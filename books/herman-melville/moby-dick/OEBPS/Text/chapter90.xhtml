“𐑦𐑟 𐑞 ·𐑛𐑿𐑒 𐑕𐑴 𐑝𐑧𐑮𐑦 𐑐𐑫𐑼 𐑨𐑟 𐑑 𐑚𐑰 𐑓𐑹𐑕𐑑 𐑑 𐑞𐑦𐑕 𐑛𐑧𐑕𐑐𐑼𐑩𐑑 𐑥𐑴𐑛 𐑝 𐑜𐑧𐑑𐑦𐑙 𐑩 𐑤𐑲𐑝𐑤𐑦𐑣𐑫𐑛?”

“𐑦𐑑 𐑦𐑟 𐑣𐑦𐑟.”

“𐑲 𐑔𐑷𐑑 𐑑 𐑮𐑦𐑤𐑰𐑝 𐑥𐑲 𐑴𐑤𐑛 𐑚𐑧𐑛-𐑮𐑦𐑛𐑩𐑯 𐑥𐑳𐑞𐑼 𐑚𐑲 𐑐𐑸𐑑 𐑝 𐑥𐑲 𐑖𐑺 𐑝 𐑞𐑦𐑕 𐑢𐑱𐑤.”

“𐑦𐑑 𐑦𐑟 𐑣𐑦𐑟.”

“𐑢𐑴𐑯𐑑 𐑞 ·𐑛𐑿𐑒 𐑚𐑰 𐑒𐑩𐑯𐑑𐑧𐑯𐑑 𐑢𐑦𐑞 𐑩 𐑒𐑢𐑹𐑑𐑼 𐑹 𐑩 𐑣𐑭𐑓?”

“𐑦𐑑 𐑦𐑟 𐑣𐑦𐑟.”

𐑦𐑯 𐑩 𐑢𐑻𐑛, 𐑞 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑕𐑰𐑟𐑛 𐑯 𐑕𐑴𐑤𐑛, 𐑯 𐑣𐑦𐑟 ·𐑜𐑮𐑱𐑕 𐑞 ·𐑛𐑿𐑒 𐑝 ·𐑢𐑧𐑤𐑦𐑙𐑑𐑩𐑯 𐑮𐑦𐑕𐑰𐑝𐑛 𐑞 𐑥𐑳𐑯𐑦. 𐑔𐑦𐑙𐑒𐑦𐑙 𐑞𐑨𐑑 𐑝𐑿𐑛 𐑦𐑯 𐑕𐑳𐑥 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑤𐑲𐑑𐑕, 𐑞 𐑒𐑱𐑕 𐑥𐑲𐑑 𐑚𐑲 𐑩 𐑚𐑺 𐑐𐑪𐑕𐑩𐑚𐑦𐑤𐑦𐑑𐑦 𐑦𐑯 𐑕𐑳𐑥 𐑕𐑥𐑷𐑤 𐑛𐑦𐑜𐑮𐑰 𐑚𐑰 𐑛𐑰𐑥𐑛, 𐑳𐑯𐑛𐑼 𐑞 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕𐑩𐑟, 𐑩 𐑮𐑭𐑞𐑼 𐑣𐑸𐑛 𐑢𐑳𐑯, 𐑩𐑯 𐑪𐑯𐑩𐑕𐑑 𐑒𐑤𐑻𐑡𐑦𐑥𐑩𐑯 𐑝 𐑞 𐑑𐑬𐑯 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑓𐑩𐑤𐑦 𐑩𐑛𐑮𐑧𐑕𐑑 𐑩 𐑯𐑴𐑑 𐑑 𐑣𐑦𐑟 ·𐑜𐑮𐑱𐑕, 𐑚𐑧𐑜𐑦𐑙 𐑣𐑦𐑥 𐑑 𐑑𐑱𐑒 𐑞 𐑒𐑱𐑕 𐑝 𐑞𐑴𐑟 𐑳𐑯𐑓𐑹𐑗𐑩𐑯𐑩𐑑 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑦𐑯𐑑𐑵 𐑓𐑫𐑤 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑱𐑖𐑩𐑯. 𐑑 𐑢𐑦𐑗 𐑥𐑲 ·𐑤𐑹𐑛 ·𐑛𐑿𐑒 𐑦𐑯 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕 𐑮𐑦𐑐𐑤𐑲𐑛 (𐑚𐑴𐑔 𐑤𐑧𐑑𐑼𐑟 𐑢𐑻 𐑐𐑳𐑚𐑤𐑦𐑖𐑑) 𐑞𐑨𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑷𐑤𐑮𐑧𐑛𐑦 𐑛𐑳𐑯 𐑕𐑴, 𐑯 𐑮𐑦𐑕𐑰𐑝𐑛 𐑞 𐑥𐑳𐑯𐑦, 𐑯 𐑢𐑫𐑛 𐑚𐑰 𐑩𐑚𐑤𐑲𐑡𐑛 𐑑 𐑞 𐑮𐑧𐑝𐑼𐑩𐑯𐑛 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯 𐑦𐑓 𐑓 𐑞 𐑓𐑿𐑗𐑼 𐑣𐑰 (𐑞 𐑮𐑧𐑝𐑼𐑩𐑯𐑛 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯) 𐑢𐑫𐑛 𐑛𐑦𐑒𐑤𐑲𐑯 𐑥𐑧𐑛𐑩𐑤𐑦𐑙 𐑢𐑦𐑞 𐑳𐑞𐑼 𐑐𐑰𐑐𐑩𐑤'𐑟 𐑚𐑦𐑟𐑯𐑩𐑕. 𐑦𐑟 𐑞𐑦𐑕 𐑞 𐑕𐑑𐑦𐑤 𐑥𐑦𐑤𐑦𐑑𐑩𐑯𐑑 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑨𐑑 𐑞 𐑒𐑹𐑯𐑼𐑟 𐑝 𐑞 𐑔𐑮𐑰 𐑒𐑦𐑙𐑛𐑩𐑥𐑟, 𐑪𐑯 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟 𐑒𐑴𐑻𐑕𐑦𐑙 𐑭𐑥𐑟 𐑝 𐑚𐑧𐑜𐑼𐑟?

𐑦𐑑 𐑢𐑦𐑤 𐑮𐑧𐑛𐑦𐑤𐑦 𐑚𐑰 𐑕𐑰𐑯 𐑞𐑨𐑑 𐑦𐑯 𐑞𐑦𐑕 𐑒𐑱𐑕 𐑞 𐑩𐑤𐑧𐑡𐑛 𐑮𐑲𐑑 𐑝 𐑞 ·𐑛𐑿𐑒 𐑑 𐑞 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑩 𐑛𐑧𐑤𐑦𐑜𐑱𐑑𐑩𐑛 𐑢𐑳𐑯 𐑓𐑮𐑪𐑥 𐑞 ·𐑕𐑪𐑝𐑮𐑩𐑯. 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑯𐑰𐑛𐑟 𐑦𐑯𐑒𐑢𐑲𐑼 𐑞𐑧𐑯 𐑪𐑯 𐑢𐑪𐑑 𐑐𐑮𐑦𐑯𐑕𐑩𐑐𐑩𐑤 𐑞 ·𐑕𐑪𐑝𐑮𐑩𐑯 𐑦𐑟 𐑼𐑦𐑡𐑦𐑯𐑩𐑤𐑦 𐑦𐑯𐑝𐑧𐑕𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑮𐑲𐑑. 𐑞 𐑤𐑷 𐑦𐑑𐑕𐑧𐑤𐑓 𐑣𐑨𐑟 𐑷𐑤𐑮𐑧𐑛𐑦 𐑚𐑰𐑯 𐑕𐑧𐑑 𐑓𐑹𐑔. 𐑚𐑳𐑑 ·𐑐𐑤𐑬𐑛𐑩𐑯 𐑜𐑦𐑝𐑟 𐑳𐑕 𐑞 𐑮𐑰𐑟𐑩𐑯 𐑓 𐑦𐑑. 𐑕𐑧𐑟 ·𐑐𐑤𐑬𐑛𐑩𐑯, 𐑞 𐑢𐑱𐑤 𐑕𐑴 𐑒𐑷𐑑 𐑚𐑦𐑤𐑪𐑙𐑟 𐑑 𐑞 ·𐑒𐑦𐑙 𐑯 ·𐑒𐑢𐑰𐑯, “𐑚𐑦𐑒𐑪𐑟 𐑝 𐑦𐑑𐑕 𐑕𐑵𐑐𐑽𐑽 𐑧𐑒𐑕𐑩𐑤𐑩𐑯𐑕.” 𐑯 𐑚𐑲 𐑞 𐑕𐑬𐑯𐑛𐑩𐑕𐑑 𐑒𐑪𐑥𐑩𐑯𐑑𐑱𐑑𐑼𐑟 𐑞𐑦𐑕 𐑣𐑨𐑟 𐑧𐑝𐑼 𐑚𐑰𐑯 𐑣𐑧𐑤𐑛 𐑩 𐑒𐑴𐑡𐑩𐑯𐑑 𐑸𐑜𐑿𐑥𐑩𐑯𐑑 𐑦𐑯 𐑕𐑳𐑗 𐑥𐑨𐑑𐑼𐑟.

𐑚𐑳𐑑 𐑢𐑲 𐑖𐑵𐑛 𐑞 ·𐑒𐑦𐑙 𐑣𐑨𐑝 𐑞 𐑣𐑧𐑛, 𐑯 𐑞 ·𐑒𐑢𐑰𐑯 𐑞 𐑑𐑱𐑤? 𐑩 𐑮𐑰𐑟𐑩𐑯 𐑓 𐑞𐑨𐑑, 𐑘𐑰 𐑤𐑷𐑘𐑼𐑟!

𐑦𐑯 𐑣𐑦𐑟 𐑑𐑮𐑰𐑑𐑦𐑟 𐑪𐑯 “𐑒𐑢𐑰𐑯-·𐑜𐑴𐑤𐑛,” 𐑹 ·𐑒𐑢𐑰𐑯-𐑐𐑦𐑯𐑥𐑳𐑯𐑦, 𐑩𐑯 𐑴𐑤𐑛 ·𐑒𐑦𐑙'𐑟 ·𐑚𐑧𐑯𐑗 𐑷𐑔𐑼, 𐑢𐑳𐑯 ·𐑢𐑦𐑤𐑾𐑥 ·𐑐𐑮𐑦𐑯, 𐑞𐑳𐑕 𐑛𐑦𐑕𐑒𐑹𐑕𐑩𐑔: “𐑘𐑰 𐑑𐑱𐑤 𐑦𐑟 𐑘𐑰 ·𐑒𐑢𐑰𐑯'𐑟, 𐑞𐑨𐑑 𐑘𐑰 ·𐑒𐑢𐑰𐑯'𐑟 𐑢𐑹𐑛𐑮𐑴𐑚 𐑥𐑱 𐑚𐑰 𐑕𐑩𐑐𐑤𐑲𐑛 𐑢𐑦𐑞 𐑘𐑰 𐑢𐑱𐑤𐑚𐑴𐑯.” 𐑯𐑬 𐑞𐑦𐑕 𐑢𐑪𐑟 𐑮𐑦𐑑𐑩𐑯 𐑨𐑑 𐑩 𐑑𐑲𐑥 𐑢𐑧𐑯 𐑞 𐑚𐑤𐑨𐑒 𐑤𐑦𐑥𐑚𐑼 𐑚𐑴𐑯 𐑝 𐑞 ·𐑜𐑮𐑰𐑯𐑤𐑩𐑯𐑛 𐑹 ·𐑮𐑲𐑑 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑤𐑸𐑡𐑤𐑦 𐑿𐑟𐑛 𐑦𐑯 𐑤𐑱𐑛𐑦𐑟' 𐑚𐑪𐑛𐑦𐑕𐑩𐑟. 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑕𐑱𐑥 𐑚𐑴𐑯 𐑦𐑟 𐑯𐑪𐑑 𐑦𐑯 𐑞 𐑑𐑱𐑤; 𐑦𐑑 𐑦𐑟 𐑦𐑯 𐑞 𐑣𐑧𐑛, 𐑢𐑦𐑗 𐑦𐑟 𐑩 𐑕𐑨𐑛 𐑥𐑦𐑕𐑑𐑱𐑒 𐑓 𐑩 𐑕𐑩𐑜𐑱𐑖𐑩𐑕 𐑤𐑷𐑘𐑼 𐑤𐑲𐑒 ·𐑐𐑮𐑦𐑯. 𐑚𐑳𐑑 𐑦𐑟 𐑞 ·𐑒𐑢𐑰𐑯 𐑩 𐑥𐑻𐑥𐑱𐑛, 𐑑 𐑚𐑰 𐑐𐑮𐑦𐑟𐑧𐑯𐑑𐑩𐑛 𐑢𐑦𐑞 𐑩 𐑑𐑱𐑤? 𐑩𐑯 𐑨𐑤𐑦𐑜𐑹𐑦𐑒𐑩𐑤 𐑥𐑰𐑯𐑦𐑙 𐑥𐑱 𐑤𐑻𐑒 𐑣𐑽.

𐑞𐑺 𐑸 𐑑𐑵 𐑮𐑶𐑩𐑤 𐑓𐑦𐑖 𐑕𐑴 𐑕𐑑𐑲𐑤𐑛 𐑚𐑲 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖 𐑤𐑷 𐑮𐑲𐑑𐑼𐑟—𐑞 𐑢𐑱𐑤 𐑯 𐑞 𐑕𐑑𐑻𐑡𐑩𐑯; 𐑚𐑴𐑔 𐑮𐑶𐑩𐑤 𐑐𐑮𐑪𐑐𐑼𐑑𐑦 𐑳𐑯𐑛𐑼 𐑕𐑻𐑑𐑩𐑯 𐑤𐑦𐑥𐑦𐑑𐑱𐑖𐑩𐑯𐑟, 𐑯 𐑯𐑪𐑥𐑦𐑯𐑩𐑤𐑦 𐑕𐑩𐑐𐑤𐑲𐑦𐑙 𐑞 𐑑𐑧𐑯𐑔 𐑚𐑮𐑭𐑯𐑗 𐑝 𐑞 𐑒𐑮𐑬𐑯'𐑟 𐑹𐑛𐑦𐑯𐑼𐑦 𐑮𐑧𐑝𐑩𐑯𐑿. 𐑲 𐑯𐑴 𐑯𐑪𐑑 𐑞𐑨𐑑 𐑧𐑯𐑦 𐑳𐑞𐑼 𐑷𐑔𐑼 𐑣𐑨𐑟 𐑣𐑦𐑯𐑑𐑩𐑛 𐑝 𐑞 𐑥𐑨𐑑𐑼; 𐑚𐑳𐑑 𐑚𐑲 𐑦𐑯𐑓𐑼𐑩𐑯𐑕 𐑦𐑑 𐑕𐑰𐑥𐑟 𐑑 𐑥𐑰 𐑞𐑨𐑑 𐑞 𐑕𐑑𐑻𐑡𐑩𐑯 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑛𐑦𐑝𐑲𐑛𐑩𐑛 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑢𐑱 𐑨𐑟 𐑞 𐑢𐑱𐑤, 𐑞 ·𐑒𐑦𐑙 𐑮𐑦𐑕𐑰𐑝𐑦𐑙 𐑞 𐑣𐑲𐑤𐑦 𐑛𐑧𐑯𐑕 𐑯 𐑦𐑤𐑨𐑕𐑑𐑦𐑒 𐑣𐑧𐑛 𐑐𐑦𐑒𐑿𐑤𐑽 𐑑 𐑞𐑨𐑑 𐑓𐑦𐑖, 𐑢𐑦𐑗, 𐑕𐑦𐑥𐑚𐑪𐑤𐑦𐑒𐑤𐑦 𐑮𐑦𐑜𐑸𐑛𐑩𐑛, 𐑥𐑱 𐑐𐑪𐑕𐑩𐑚𐑤𐑦 𐑚𐑰 𐑣𐑿𐑥𐑼𐑩𐑕𐑤𐑦 𐑜𐑮𐑬𐑯𐑛𐑩𐑛 𐑩𐑐𐑪𐑯 𐑕𐑳𐑥 𐑐𐑮𐑦𐑟𐑿𐑥𐑛 𐑒𐑩𐑯𐑡𐑰𐑯𐑦𐑨𐑤𐑦𐑑𐑦. 𐑯 𐑞𐑳𐑕 𐑞𐑺 𐑕𐑰𐑥𐑟 𐑩 𐑮𐑰𐑟𐑩𐑯 𐑦𐑯 𐑷𐑤 𐑔𐑦𐑙𐑟, 𐑰𐑝𐑩𐑯 𐑦𐑯 𐑤𐑷.
