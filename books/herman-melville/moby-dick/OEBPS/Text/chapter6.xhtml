·𐑗𐑨𐑐𐑑𐑼 6. 𐑞 ·𐑕𐑑𐑮𐑰𐑑.

𐑦𐑓 𐑲 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑩𐑕𐑑𐑪𐑯𐑦𐑖𐑑 𐑨𐑑 𐑓𐑻𐑕𐑑 𐑒𐑨𐑗𐑦𐑙 𐑩 𐑜𐑤𐑦𐑥𐑐𐑕 𐑝 𐑕𐑴 𐑬𐑑𐑤𐑨𐑯𐑛𐑦𐑖 𐑩𐑯 𐑦𐑯𐑛𐑦𐑝𐑦𐑡𐑵𐑩𐑤 𐑨𐑟 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑕𐑻𐑒𐑿𐑤𐑱𐑑𐑦𐑙 𐑩𐑥𐑳𐑙 𐑞 𐑐𐑩𐑤𐑲𐑑 𐑕𐑩𐑕𐑲𐑩𐑑𐑦 𐑝 𐑩 𐑕𐑦𐑝𐑦𐑤𐑲𐑟𐑛 𐑑𐑬𐑯, 𐑞𐑨𐑑 𐑩𐑕𐑑𐑪𐑯𐑦𐑖𐑥𐑩𐑯𐑑 𐑕𐑵𐑯 𐑛𐑦𐑐𐑸𐑑𐑩𐑛 𐑩𐑐𐑪𐑯 𐑑𐑱𐑒𐑦𐑙 𐑥𐑲 𐑓𐑻𐑕𐑑 𐑛𐑱𐑤𐑲𐑑 𐑕𐑑𐑮𐑴𐑤 𐑔𐑮𐑵 𐑞 𐑕𐑑𐑮𐑰𐑑𐑕 𐑝 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛.

𐑦𐑯 𐑔𐑳𐑮𐑴𐑓𐑺𐑟 𐑯𐑲 𐑞 𐑛𐑪𐑒𐑕, 𐑧𐑯𐑦 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑕𐑰𐑐𐑹𐑑 𐑢𐑦𐑤 𐑓𐑮𐑰𐑒𐑢𐑩𐑯𐑑𐑤𐑦 𐑪𐑓𐑼 𐑑 𐑝𐑿 𐑞 𐑒𐑢𐑽𐑩𐑕𐑑 𐑤𐑫𐑒𐑦𐑙 𐑯𐑪𐑯𐑛𐑦𐑕𐑒𐑮𐑦𐑐𐑑𐑕 𐑓𐑮𐑪𐑥 𐑓𐑪𐑮𐑦𐑯 𐑐𐑸𐑑𐑕. 𐑰𐑝𐑩𐑯 𐑦𐑯 ·𐑚𐑮𐑷𐑛𐑢𐑱 𐑯 ·𐑗𐑧𐑕𐑑𐑯𐑳𐑑 𐑕𐑑𐑮𐑰𐑑𐑕, ·𐑥𐑧𐑛𐑦𐑑𐑼𐑱𐑯𐑾𐑯 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑡𐑪𐑕𐑩𐑤 𐑞 𐑩𐑓𐑮𐑲𐑑𐑩𐑛 𐑤𐑱𐑛𐑦𐑟. 𐑮𐑰𐑡𐑩𐑯𐑑 ·𐑕𐑑𐑮𐑰𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑳𐑯𐑯𐑴𐑯 𐑑 ·𐑤𐑨𐑕𐑒𐑼𐑟 𐑯 ·𐑥𐑩𐑤𐑱𐑟; 𐑯 𐑨𐑑 ·𐑚𐑪𐑥𐑚𐑱, 𐑦𐑯 𐑞 ·𐑩𐑐𐑪𐑤𐑴 ·𐑜𐑮𐑰𐑯, 𐑤𐑦𐑝 ·𐑘𐑨𐑙𐑒𐑦𐑟 𐑣𐑨𐑝 𐑪𐑓𐑩𐑯 𐑕𐑒𐑺𐑛 𐑞 𐑯𐑱𐑑𐑦𐑝𐑟. 𐑚𐑳𐑑 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛 𐑚𐑰𐑑𐑕 𐑷𐑤 ·𐑢𐑷𐑑𐑼 ·𐑕𐑑𐑮𐑰𐑑 𐑯 ·𐑢𐑪𐑐𐑦𐑙. 𐑦𐑯 𐑞𐑰𐑟 𐑤𐑭𐑕𐑑-𐑥𐑧𐑯𐑖𐑩𐑯𐑛 𐑣𐑷𐑯𐑑𐑕 𐑿 𐑕𐑰 𐑴𐑯𐑤𐑦 𐑕𐑱𐑤𐑼𐑟; 𐑚𐑳𐑑 𐑦𐑯 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛, 𐑨𐑒𐑗𐑵𐑩𐑤 𐑒𐑨𐑯𐑦𐑚𐑩𐑤𐑟 𐑕𐑑𐑨𐑯𐑛 𐑗𐑨𐑑𐑦𐑙 𐑨𐑑 𐑕𐑑𐑮𐑰𐑑 𐑒𐑹𐑯𐑼𐑟; 𐑕𐑨𐑝𐑦𐑡𐑩𐑟 𐑬𐑑𐑮𐑲𐑑; 𐑥𐑧𐑯𐑦 𐑝 𐑣𐑵𐑥 𐑘𐑧𐑑 𐑒𐑨𐑮𐑦 𐑪𐑯 𐑞𐑺 𐑚𐑴𐑯𐑟 𐑳𐑯𐑣𐑴𐑤𐑦 𐑓𐑤𐑧𐑖. 𐑦𐑑 𐑥𐑱𐑒𐑕 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑕𐑑𐑺.

𐑚𐑳𐑑, 𐑚𐑦𐑕𐑲𐑛𐑟 𐑞 ·𐑝𐑰𐑡𐑰𐑩𐑯𐑕, ·𐑑𐑪𐑙𐑱𐑑𐑴𐑚𐑵𐑼𐑟, ·𐑻𐑩𐑥𐑨𐑙𐑴𐑩𐑯𐑟, ·𐑐𐑨𐑯𐑨𐑙𐑾𐑯𐑟, 𐑯 ·𐑚𐑮𐑦𐑜𐑦𐑜𐑾𐑯𐑕, 𐑯, 𐑚𐑦𐑕𐑲𐑛𐑟 𐑞 𐑢𐑲𐑤𐑛 𐑕𐑐𐑧𐑕𐑩𐑥𐑩𐑯𐑟 𐑝 𐑞 𐑢𐑱𐑤𐑦𐑙-𐑒𐑮𐑭𐑓𐑑 𐑢𐑦𐑗 𐑳𐑯𐑣𐑰𐑛𐑩𐑛 𐑮𐑰𐑤 𐑩𐑚𐑬𐑑 𐑞 𐑕𐑑𐑮𐑰𐑑𐑕, 𐑿 𐑢𐑦𐑤 𐑕𐑰 𐑳𐑞𐑼 𐑕𐑲𐑑𐑕 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑒𐑘𐑫𐑼𐑾𐑕, 𐑕𐑻𐑑𐑩𐑯𐑤𐑦 𐑥𐑹 𐑒𐑪𐑥𐑦𐑒𐑩𐑤. 𐑞𐑺 𐑢𐑰𐑒𐑤𐑦 𐑼𐑲𐑝 𐑦𐑯 𐑞𐑦𐑕 𐑑𐑬𐑯 𐑕𐑒𐑹𐑟 𐑝 𐑜𐑮𐑰𐑯 ·𐑝𐑼𐑥𐑪𐑯𐑑𐑼𐑟 𐑯 ·𐑯𐑿 ·𐑣𐑨𐑥𐑐𐑖𐑼 𐑥𐑧𐑯, 𐑷𐑤 𐑩𐑔𐑻𐑕𐑑 𐑓 𐑜𐑱𐑯 𐑯 𐑜𐑤𐑹𐑦 𐑦𐑯 𐑞 𐑓𐑦𐑖𐑼𐑦. 𐑞𐑱 𐑸 𐑥𐑴𐑕𐑑𐑤𐑦 𐑘𐑳𐑙, 𐑝 𐑕𐑑𐑷𐑤𐑢𐑼𐑑 𐑓𐑮𐑱𐑥𐑟; 𐑓𐑧𐑤𐑴𐑟 𐑣𐑵 𐑣𐑨𐑝 𐑓𐑧𐑤𐑛 𐑓𐑪𐑮𐑦𐑕𐑑𐑕, 𐑯 𐑯𐑬 𐑕𐑰𐑒 𐑑 𐑛𐑮𐑪𐑐 𐑞 𐑨𐑒𐑕 𐑯 𐑕𐑯𐑨𐑗 𐑞 𐑢𐑱𐑤-𐑤𐑭𐑯𐑕. 𐑥𐑧𐑯𐑦 𐑸 𐑨𐑟 𐑜𐑮𐑰𐑯 𐑨𐑟 𐑞 ·𐑜𐑮𐑰𐑯 ·𐑥𐑬𐑯𐑑𐑩𐑯𐑟 𐑢𐑧𐑯𐑕 𐑞𐑱 𐑒𐑱𐑥. 𐑦𐑯 𐑕𐑳𐑥 𐑔𐑦𐑙𐑟 𐑿 𐑢𐑫𐑛 𐑔𐑦𐑙𐑒 𐑞𐑧𐑥 𐑚𐑳𐑑 𐑩 𐑓𐑿 𐑬𐑼𐑟 𐑴𐑤𐑛. 𐑤𐑫𐑒 𐑞𐑺! 𐑞𐑨𐑑 𐑗𐑨𐑐 𐑕𐑑𐑮𐑳𐑑𐑦𐑙 𐑮𐑬𐑯𐑛 𐑞 𐑒𐑹𐑯𐑼. 𐑣𐑰 𐑢𐑺𐑟 𐑩 𐑚𐑰𐑝𐑼 𐑣𐑨𐑑 𐑯 𐑕𐑢𐑪𐑤𐑴-𐑑𐑱𐑤𐑛 𐑒𐑴𐑑, 𐑜𐑻𐑛𐑩𐑤𐑛 𐑢𐑦𐑞 𐑩 𐑕𐑱𐑤𐑼-𐑚𐑧𐑤𐑑 𐑯 𐑖𐑰𐑔-𐑯𐑲𐑓. 𐑣𐑽 𐑒𐑳𐑥𐑟 𐑩𐑯𐑳𐑞𐑼 𐑢𐑦𐑞 𐑩 𐑕𐑵'-𐑢𐑧𐑕𐑑𐑼 𐑯 𐑩 𐑚𐑪𐑥𐑚𐑩𐑟𐑰𐑯 𐑒𐑤𐑴𐑒.

𐑯𐑴 𐑑𐑬𐑯-𐑚𐑮𐑧𐑛 𐑛𐑨𐑯𐑛𐑦 𐑢𐑦𐑤 𐑒𐑩𐑥𐑐𐑺 𐑢𐑦𐑞 𐑩 𐑒𐑳𐑯𐑑𐑮𐑦-𐑚𐑮𐑧𐑛 𐑢𐑳𐑯—𐑲 𐑥𐑰𐑯 𐑩 𐑛𐑬𐑯𐑮𐑲𐑑 𐑚𐑳𐑥𐑐𐑒𐑦𐑯 𐑛𐑨𐑯𐑛𐑦—𐑩 𐑓𐑧𐑤𐑴 𐑞𐑨𐑑, 𐑦𐑯 𐑞 𐑛𐑪𐑜-𐑛𐑱𐑟, 𐑢𐑦𐑤 𐑥𐑴 𐑣𐑦𐑟 𐑑𐑵 𐑱𐑒𐑼𐑟 𐑦𐑯 𐑚𐑳𐑒𐑕𐑒𐑦𐑯 𐑜𐑤𐑳𐑝𐑟 𐑓 𐑓𐑽 𐑝 𐑑𐑨𐑯𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛𐑟. 𐑯𐑬 𐑢𐑧𐑯 𐑩 𐑒𐑳𐑯𐑑𐑮𐑦 𐑛𐑨𐑯𐑛𐑦 𐑤𐑲𐑒 𐑞𐑦𐑕 𐑑𐑱𐑒𐑕 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑑 𐑥𐑱𐑒 𐑩 𐑛𐑦𐑕𐑑𐑦𐑙𐑜𐑢𐑦𐑖𐑑 𐑮𐑧𐑐𐑿𐑑𐑱𐑖𐑩𐑯, 𐑯 𐑡𐑶𐑯𐑟 𐑞 𐑜𐑮𐑱𐑑 𐑢𐑱𐑤-𐑓𐑦𐑖𐑼𐑦, 𐑿 𐑖𐑵𐑛 𐑕𐑰 𐑞 𐑒𐑪𐑥𐑦𐑒𐑩𐑤 𐑔𐑦𐑙𐑟 𐑣𐑰 𐑛𐑳𐑟 𐑩𐑐𐑪𐑯 𐑮𐑰𐑗𐑦𐑙 𐑞 𐑕𐑰𐑐𐑹𐑑. 𐑦𐑯 𐑚𐑦𐑕𐑐𐑰𐑒𐑦𐑙 𐑣𐑦𐑟 𐑕𐑰-𐑬𐑑𐑓𐑦𐑑, 𐑣𐑰 𐑹𐑛𐑼𐑟 𐑚𐑧𐑤-𐑚𐑳𐑑𐑩𐑯𐑟 𐑑 𐑣𐑦𐑟 𐑢𐑱𐑕𐑑𐑒𐑴𐑑𐑕; 𐑕𐑑𐑮𐑨𐑐𐑕 𐑑 𐑣𐑦𐑟 𐑒𐑨𐑯𐑝𐑩𐑕 𐑑𐑮𐑴𐑟𐑼𐑟. 𐑭, 𐑐𐑫𐑼 ·𐑣𐑱-·𐑕𐑰𐑛! 𐑣𐑬 𐑚𐑦𐑑𐑼𐑤𐑦 𐑢𐑦𐑤 𐑚𐑻𐑕𐑑 𐑞𐑴𐑟 𐑕𐑑𐑮𐑨𐑐𐑕 𐑦𐑯 𐑞 𐑓𐑻𐑕𐑑 𐑣𐑬𐑤𐑦𐑙 𐑜𐑱𐑤, 𐑢𐑧𐑯 𐑞𐑬 𐑸𐑑 𐑛𐑮𐑦𐑝𐑩𐑯, 𐑕𐑑𐑮𐑨𐑐𐑕, 𐑚𐑳𐑑𐑩𐑯𐑟, 𐑯 𐑷𐑤, 𐑛𐑬𐑯 𐑞 𐑔𐑮𐑴𐑑 𐑝 𐑞 𐑑𐑧𐑥𐑐𐑦𐑕𐑑.

𐑚𐑳𐑑 𐑔𐑦𐑙𐑒 𐑯𐑪𐑑 𐑞𐑨𐑑 𐑞𐑦𐑕 𐑓𐑱𐑥𐑩𐑕 𐑑𐑬𐑯 𐑣𐑨𐑟 𐑴𐑯𐑤𐑦 𐑣𐑸𐑐𐑵𐑯𐑽𐑟, 𐑒𐑨𐑯𐑦𐑚𐑩𐑤𐑟, 𐑯 𐑚𐑳𐑥𐑐𐑒𐑦𐑯𐑟 𐑑 𐑖𐑴 𐑣𐑻 𐑝𐑦𐑟𐑦𐑑𐑼𐑟. 𐑯𐑪𐑑 𐑨𐑑 𐑷𐑤. 𐑕𐑑𐑦𐑤 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛 𐑦𐑟 𐑩 𐑒𐑢𐑽 𐑐𐑤𐑱𐑕. 𐑣𐑨𐑛 𐑦𐑑 𐑯𐑪𐑑 𐑚𐑰𐑯 𐑓 𐑳𐑕 𐑢𐑱𐑤𐑥𐑧𐑯, 𐑞𐑨𐑑 𐑑𐑮𐑨𐑒𐑑 𐑝 𐑤𐑨𐑯𐑛 𐑢𐑫𐑛 𐑞𐑦𐑕 𐑛𐑱 𐑐𐑼𐑣𐑨𐑐𐑕 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑦𐑯 𐑨𐑟 𐑣𐑬𐑤𐑦𐑙 𐑒𐑩𐑯𐑛𐑦𐑖𐑩𐑯 𐑨𐑟 𐑞 𐑒𐑴𐑕𐑑 𐑝 ·𐑤𐑨𐑚𐑮𐑩𐑛𐑹. 𐑨𐑟 𐑦𐑑 𐑦𐑟, 𐑐𐑸𐑑𐑕 𐑝 𐑣𐑻 𐑚𐑨𐑒 𐑒𐑳𐑯𐑑𐑮𐑦 𐑸 𐑦𐑯𐑳𐑓 𐑑 𐑓𐑮𐑲𐑑𐑩𐑯 𐑢𐑳𐑯, 𐑞𐑱 𐑤𐑫𐑒 𐑕𐑴 𐑚𐑴𐑯𐑦. 𐑞 𐑑𐑬𐑯 𐑦𐑑𐑕𐑧𐑤𐑓 𐑦𐑟 𐑐𐑼𐑣𐑨𐑐𐑕 𐑞 𐑛𐑽𐑩𐑕𐑑 𐑐𐑤𐑱𐑕 𐑑 𐑤𐑦𐑝 𐑦𐑯, 𐑦𐑯 𐑷𐑤 ·𐑯𐑿 ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛. 𐑦𐑑 𐑦𐑟 𐑩 𐑤𐑨𐑯𐑛 𐑝 𐑶𐑤, 𐑑𐑮𐑵 𐑦𐑯𐑳𐑓: 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑤𐑲𐑒 ·𐑒𐑱𐑯𐑩𐑯; 𐑩 𐑤𐑨𐑯𐑛, 𐑷𐑤𐑕𐑴, 𐑝 𐑒𐑹𐑯 𐑯 𐑢𐑲𐑯. 𐑞 𐑕𐑑𐑮𐑰𐑑𐑕 𐑛𐑵 𐑯𐑪𐑑 𐑮𐑳𐑯 𐑢𐑦𐑞 𐑥𐑦𐑤𐑒; 𐑯𐑹 𐑦𐑯 𐑞 𐑕𐑐𐑮𐑦𐑙-𐑑𐑲𐑥 𐑛𐑵 𐑞𐑱 𐑐𐑱𐑝 𐑞𐑧𐑥 𐑢𐑦𐑞 𐑓𐑮𐑧𐑖 𐑧𐑜𐑟. 𐑘𐑧𐑑, 𐑦𐑯 𐑕𐑐𐑲𐑑 𐑝 𐑞𐑦𐑕, 𐑯𐑴𐑢𐑺 𐑦𐑯 𐑷𐑤 ·𐑩𐑥𐑧𐑮𐑦𐑒𐑩 𐑢𐑦𐑤 𐑿 𐑓𐑲𐑯𐑛 𐑥𐑹 𐑐𐑩𐑑𐑮𐑦𐑖𐑩𐑯-𐑤𐑲𐑒 𐑣𐑬𐑟𐑩𐑟; 𐑐𐑸𐑒𐑕 𐑯 𐑜𐑸𐑛𐑩𐑯𐑟 𐑥𐑹 𐑪𐑐𐑿𐑤𐑩𐑯𐑑, 𐑞𐑨𐑯 𐑦𐑯 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛. 𐑢𐑧𐑯𐑕 𐑒𐑱𐑥 𐑞𐑱? 𐑣𐑬 𐑐𐑤𐑭𐑯𐑑𐑩𐑛 𐑩𐑐𐑪𐑯 𐑞𐑦𐑕 𐑢𐑳𐑯𐑕 𐑕𐑒𐑮𐑨𐑜𐑦 𐑕𐑒𐑹𐑾 𐑝 𐑩 𐑒𐑳𐑯𐑑𐑮𐑦?

𐑜𐑴 𐑯 𐑜𐑱𐑟 𐑩𐑐𐑪𐑯 𐑞 𐑲𐑼𐑯 𐑧𐑥𐑚𐑤𐑩𐑥𐑨𐑑𐑦𐑒𐑩𐑤 𐑣𐑸𐑐𐑵𐑯𐑟 𐑮𐑬𐑯𐑛 𐑘𐑪𐑯𐑛𐑼 𐑤𐑪𐑓𐑑𐑦 𐑥𐑨𐑯𐑖𐑩𐑯, 𐑯 𐑘𐑹 𐑒𐑢𐑧𐑕𐑗𐑩𐑯 𐑢𐑦𐑤 𐑚𐑰 𐑭𐑯𐑕𐑼𐑛. 𐑘𐑧𐑕; 𐑷𐑤 𐑞𐑰𐑟 𐑚𐑮𐑱𐑝 𐑣𐑬𐑟𐑩𐑟 𐑯 𐑓𐑤𐑬𐑼𐑦 𐑜𐑸𐑛𐑩𐑯𐑟 𐑒𐑱𐑥 𐑓𐑮𐑪𐑥 𐑞 ·𐑩𐑑𐑤𐑨𐑯𐑑𐑦𐑒, ·𐑐𐑩𐑕𐑦𐑓𐑦𐑒, 𐑯 ·𐑦𐑯𐑛𐑾𐑯 𐑴𐑖𐑩𐑯𐑟. 𐑢𐑳𐑯 𐑯 𐑷𐑤, 𐑞𐑱 𐑢𐑻 𐑣𐑸𐑐𐑵𐑯𐑛 𐑯 𐑛𐑮𐑨𐑜𐑛 𐑳𐑐 𐑣𐑦𐑞𐑼 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 𐑞 𐑕𐑰. 𐑒𐑨𐑯 ·𐑣𐑧𐑮 ·𐑨𐑤𐑦𐑜𐑟𐑭𐑯𐑛𐑼 𐑐𐑼𐑓𐑹𐑥 𐑩 𐑓𐑰𐑑 𐑤𐑲𐑒 𐑞𐑨𐑑?

𐑦𐑯 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛, 𐑓𐑭𐑞𐑼𐑟, 𐑞𐑱 𐑕𐑱, 𐑜𐑦𐑝 𐑢𐑱𐑤𐑟 𐑓 𐑛𐑬𐑼𐑟 𐑑 𐑞𐑺 𐑛𐑷𐑑𐑼𐑟, 𐑯 𐑐𐑹𐑖𐑩𐑯 𐑪𐑓 𐑞𐑺 𐑯𐑰𐑕𐑩𐑟 𐑢𐑦𐑞 𐑩 𐑓𐑿 𐑐𐑹𐑐𐑩𐑕𐑩𐑟 𐑩-𐑐𐑰𐑕. 𐑿 𐑥𐑳𐑕𐑑 𐑜𐑴 𐑑 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛 𐑑 𐑕𐑰 𐑩 𐑚𐑮𐑦𐑤𐑾𐑯𐑑 𐑢𐑧𐑛𐑦𐑙; 𐑓, 𐑞𐑱 𐑕𐑱, 𐑞𐑱 𐑣𐑨𐑝 𐑮𐑧𐑟𐑼𐑝𐑢𐑸𐑟 𐑝 𐑶𐑤 𐑦𐑯 𐑧𐑝𐑮𐑦 𐑣𐑬𐑕, 𐑯 𐑧𐑝𐑮𐑦 𐑯𐑲𐑑 𐑮𐑧𐑒𐑤𐑩𐑕𐑤𐑦 𐑚𐑻𐑯 𐑞𐑺 𐑤𐑧𐑙𐑔𐑕 𐑦𐑯 𐑕𐑐𐑻𐑥𐑩𐑕𐑧𐑑𐑦 𐑒𐑨𐑯𐑛𐑩𐑤𐑟.

𐑦𐑯 𐑕𐑳𐑥𐑼 𐑑𐑲𐑥, 𐑞 𐑑𐑬𐑯 𐑦𐑟 𐑕𐑢𐑰𐑑 𐑑 𐑕𐑰; 𐑓𐑫𐑤 𐑝 𐑓𐑲𐑯 𐑥𐑱𐑐𐑩𐑤𐑟—𐑤𐑪𐑙 𐑨𐑝𐑩𐑯𐑿𐑟 𐑝 𐑜𐑮𐑰𐑯 𐑯 𐑜𐑴𐑤𐑛. 𐑯 𐑦𐑯 𐑷𐑜𐑩𐑕𐑑, 𐑣𐑲 𐑦𐑯 𐑺, 𐑞 𐑚𐑿𐑑𐑦𐑓𐑩𐑤 𐑯 𐑚𐑬𐑯𐑑𐑦𐑓𐑩𐑤 𐑣𐑹𐑕-𐑗𐑧𐑕𐑑𐑯𐑳𐑑𐑕, 𐑒𐑨𐑯𐑛𐑩𐑤𐑭𐑚𐑮𐑩-𐑢𐑲𐑟, 𐑐𐑮𐑪𐑓𐑼 𐑞 𐑐𐑭𐑕𐑼-𐑚𐑲 𐑞𐑺 𐑑𐑱𐑐𐑼𐑦𐑙 𐑳𐑐𐑮𐑲𐑑 𐑒𐑴𐑯𐑟 𐑝 𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑑𐑩𐑛 𐑚𐑤𐑪𐑕𐑩𐑥𐑟. 𐑕𐑴 𐑪𐑥𐑯𐑦𐑐𐑩𐑑𐑩𐑯𐑑 𐑦𐑟 𐑸𐑑; 𐑢𐑦𐑗 𐑦𐑯 𐑥𐑧𐑯𐑦 𐑩 𐑛𐑦𐑕𐑑𐑮𐑦𐑒𐑑 𐑝 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛 𐑣𐑨𐑟 𐑕𐑵𐑐𐑼𐑦𐑯𐑛𐑿𐑕𐑑 𐑚𐑮𐑲𐑑 𐑑𐑧𐑮𐑩𐑕𐑩𐑟 𐑝 𐑓𐑤𐑬𐑼𐑟 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑨𐑮𐑩𐑯 𐑮𐑧𐑓𐑿𐑕 𐑮𐑪𐑒𐑕 𐑔𐑮𐑴𐑯 𐑩𐑕𐑲𐑛 𐑨𐑑 𐑒𐑮𐑦𐑱𐑖𐑩𐑯'𐑟 𐑓𐑲𐑯𐑩𐑤 𐑛𐑱.

𐑯 𐑞 𐑢𐑦𐑥𐑦𐑯 𐑝 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛, 𐑞𐑱 𐑚𐑤𐑵𐑥 𐑤𐑲𐑒 𐑞𐑺 𐑴𐑯 𐑮𐑧𐑛 𐑮𐑴𐑟𐑩𐑟. 𐑚𐑳𐑑 𐑮𐑴𐑟𐑩𐑟 𐑴𐑯𐑤𐑦 𐑚𐑤𐑵𐑥 𐑦𐑯 𐑕𐑳𐑥𐑼; 𐑢𐑺𐑨𐑟 𐑞 𐑓𐑲𐑯 𐑒𐑸𐑯𐑱𐑖𐑩𐑯 𐑝 𐑞𐑺 𐑗𐑰𐑒𐑕 𐑦𐑟 𐑐𐑼𐑧𐑯𐑾𐑤 𐑨𐑟 𐑕𐑳𐑯𐑤𐑲𐑑 𐑦𐑯 𐑞 𐑕𐑧𐑝𐑩𐑯𐑔 𐑣𐑧𐑝𐑩𐑯𐑟. 𐑧𐑤𐑕𐑢𐑺 𐑥𐑨𐑗 𐑞𐑨𐑑 𐑚𐑤𐑵𐑥 𐑝 𐑞𐑺𐑟, 𐑘𐑰 𐑒𐑨𐑯𐑪𐑑, 𐑕𐑱𐑝 𐑦𐑯 ·𐑕𐑱𐑤𐑩𐑥, 𐑢𐑺 𐑞𐑱 𐑑𐑧𐑤 𐑥𐑰 𐑞 𐑘𐑳𐑙 𐑜𐑻𐑤𐑟 𐑚𐑮𐑰𐑞 𐑕𐑳𐑗 𐑥𐑳𐑕𐑒, 𐑞𐑺 𐑕𐑱𐑤𐑼 𐑕𐑢𐑰𐑑𐑣𐑸𐑑𐑕 𐑕𐑥𐑧𐑤 𐑞𐑧𐑥 𐑥𐑲𐑤𐑟 𐑪𐑓 𐑖𐑹, 𐑨𐑟 𐑞𐑴 𐑞𐑱 𐑢𐑻 𐑛𐑮𐑷𐑦𐑙 𐑯𐑲 𐑞 𐑴𐑛𐑼𐑩𐑕 ·𐑥𐑩𐑤𐑳𐑒𐑩𐑟 𐑦𐑯𐑕𐑑𐑧𐑛 𐑝 𐑞 ·𐑐𐑘𐑫𐑼𐑦𐑑𐑨𐑯𐑦𐑒 𐑕𐑨𐑯𐑛𐑟.
