·𐑗𐑨𐑐𐑑𐑼 12. 𐑚𐑲𐑩𐑜𐑮𐑨𐑓𐑦𐑒𐑩𐑤.

·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑢𐑪𐑟 𐑩 𐑯𐑱𐑑𐑦𐑝 𐑝 ·𐑮𐑴𐑒𐑴𐑝𐑩𐑒𐑴, 𐑩𐑯 𐑲𐑤𐑩𐑯𐑛 𐑓𐑸 𐑩𐑢𐑱 𐑑 𐑞 ·𐑢𐑧𐑕𐑑 𐑯 ·𐑕𐑬𐑔. 𐑦𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑛𐑬𐑯 𐑦𐑯 𐑧𐑯𐑦 𐑥𐑨𐑐; 𐑑𐑮𐑵 𐑐𐑤𐑱𐑕𐑩𐑟 𐑯𐑧𐑝𐑼 𐑸.

𐑢𐑧𐑯 𐑩 𐑯𐑿-𐑣𐑨𐑗𐑑 𐑕𐑨𐑝𐑦𐑡 𐑮𐑳𐑯𐑦𐑙 𐑢𐑲𐑤𐑛 𐑩𐑚𐑬𐑑 𐑣𐑦𐑟 𐑯𐑱𐑑𐑦𐑝 𐑢𐑫𐑛𐑤𐑩𐑯𐑛𐑟 𐑦𐑯 𐑩 𐑜𐑮𐑭𐑕 𐑒𐑤𐑬𐑑, 𐑓𐑪𐑤𐑴𐑛 𐑚𐑲 𐑞 𐑯𐑦𐑚𐑩𐑤𐑦𐑙 𐑜𐑴𐑑𐑕, 𐑨𐑟 𐑦𐑓 𐑣𐑰 𐑢𐑻 𐑩 𐑜𐑮𐑰𐑯 𐑕𐑨𐑐𐑤𐑦𐑙; 𐑰𐑝𐑩𐑯 𐑞𐑧𐑯, 𐑦𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑨𐑥𐑚𐑦𐑖𐑩𐑕 𐑕𐑴𐑤, 𐑤𐑻𐑒𐑑 𐑩 𐑕𐑑𐑮𐑪𐑙 𐑛𐑦𐑟𐑲𐑼 𐑑 𐑕𐑰 𐑕𐑳𐑥𐑔𐑦𐑙 𐑥𐑹 𐑝 ·𐑒𐑮𐑦𐑕𐑩𐑯𐑛𐑩𐑥 𐑞𐑨𐑯 𐑩 𐑕𐑐𐑧𐑕𐑩𐑥𐑩𐑯 𐑢𐑱𐑤𐑼 𐑹 𐑑𐑵. 𐑣𐑦𐑟 𐑓𐑭𐑞𐑼 𐑢𐑪𐑟 𐑩 ·𐑣𐑲 ·𐑗𐑰𐑓, 𐑩 ·𐑒𐑦𐑙; 𐑣𐑦𐑟 𐑳𐑙𐑒𐑩𐑤 𐑩 ·𐑣𐑲 ·𐑐𐑮𐑰𐑕𐑑; 𐑯 𐑪𐑯 𐑞 𐑥𐑩𐑑𐑻𐑯𐑩𐑤 𐑕𐑲𐑛 𐑣𐑰 𐑚𐑴𐑕𐑑𐑩𐑛 𐑭𐑯𐑑𐑕 𐑣𐑵 𐑢𐑻 𐑞 𐑢𐑲𐑝𐑟 𐑝 𐑳𐑯𐑒𐑪𐑙𐑒𐑼𐑩𐑚𐑩𐑤 𐑢𐑹𐑽𐑟. 𐑞𐑺 𐑢𐑪𐑟 𐑧𐑒𐑕𐑩𐑤𐑩𐑯𐑑 𐑚𐑤𐑳𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑝𐑱𐑯𐑟—𐑮𐑶𐑩𐑤 𐑕𐑑𐑳𐑓; 𐑞𐑴 𐑕𐑨𐑛𐑤𐑦 𐑝𐑦𐑖𐑦𐑱𐑑𐑩𐑛, 𐑲 𐑓𐑽, 𐑚𐑲 𐑞 𐑒𐑨𐑯𐑦𐑚𐑩𐑤 𐑐𐑮𐑩𐑐𐑧𐑯𐑕𐑦𐑑𐑦 𐑣𐑰 𐑯𐑳𐑮𐑦𐑖𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑳𐑯𐑑𐑿𐑑𐑼𐑛 𐑿𐑔.

𐑩 ·𐑕𐑨𐑜 ·𐑣𐑸𐑚𐑼 𐑖𐑦𐑐 𐑝𐑦𐑟𐑦𐑑𐑩𐑛 𐑣𐑦𐑟 𐑓𐑭𐑞𐑼'𐑟 𐑚𐑱, 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑕𐑷𐑑 𐑩 𐑐𐑨𐑕𐑦𐑡 𐑑 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯 𐑤𐑨𐑯𐑛𐑟. 𐑚𐑳𐑑 𐑞 𐑖𐑦𐑐, 𐑣𐑨𐑝𐑦𐑙 𐑣𐑻 𐑓𐑫𐑤 𐑒𐑪𐑥𐑐𐑤𐑦𐑥𐑩𐑯𐑑 𐑝 𐑕𐑰𐑥𐑧𐑯, 𐑕𐑐𐑻𐑯𐑛 𐑣𐑦𐑟 𐑕𐑵𐑑; 𐑯 𐑯𐑪𐑑 𐑷𐑤 𐑞 ·𐑒𐑦𐑙 𐑣𐑦𐑟 𐑓𐑭𐑞𐑼'𐑟 𐑦𐑯𐑓𐑤𐑵𐑩𐑯𐑕 𐑒𐑫𐑛 𐑐𐑮𐑦𐑝𐑱𐑤. 𐑚𐑳𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑝𐑬𐑛 𐑩 𐑝𐑬. 𐑩𐑤𐑴𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑒𐑩𐑯𐑵, 𐑣𐑰 𐑐𐑨𐑛𐑩𐑤𐑛 𐑪𐑓 𐑑 𐑩 𐑛𐑦𐑕𐑑𐑩𐑯𐑑 𐑕𐑑𐑮𐑱𐑑, 𐑢𐑦𐑗 𐑣𐑰 𐑯𐑿 𐑞 𐑖𐑦𐑐 𐑥𐑳𐑕𐑑 𐑐𐑭𐑕 𐑔𐑮𐑵 𐑢𐑧𐑯 𐑖𐑰 𐑒𐑢𐑦𐑑𐑩𐑛 𐑞 𐑲𐑤𐑩𐑯𐑛. 𐑪𐑯 𐑢𐑳𐑯 𐑕𐑲𐑛 𐑢𐑪𐑟 𐑩 𐑒𐑪𐑮𐑩𐑤 𐑮𐑰𐑓; 𐑪𐑯 𐑞 𐑳𐑞𐑼 𐑩 𐑤𐑴 𐑑𐑳𐑙 𐑝 𐑤𐑨𐑯𐑛, 𐑒𐑳𐑝𐑼𐑛 𐑢𐑦𐑞 𐑥𐑨𐑙𐑜𐑮𐑴𐑝 𐑔𐑦𐑒𐑩𐑑𐑕 𐑞𐑨𐑑 𐑜𐑮𐑵 𐑬𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑷𐑑𐑼. 𐑣𐑲𐑛𐑦𐑙 𐑣𐑦𐑟 𐑒𐑩𐑯𐑵, 𐑕𐑑𐑦𐑤 𐑩𐑓𐑤𐑴𐑑, 𐑩𐑥𐑳𐑙 𐑞𐑰𐑟 𐑔𐑦𐑒𐑩𐑑𐑕, 𐑢𐑦𐑞 𐑦𐑑𐑕 𐑐𐑮𐑬 𐑕𐑰𐑢𐑼𐑛, 𐑣𐑰 𐑕𐑨𐑑 𐑛𐑬𐑯 𐑦𐑯 𐑞 𐑕𐑑𐑻𐑯, 𐑐𐑨𐑛𐑩𐑤 𐑤𐑴 𐑦𐑯 𐑣𐑨𐑯𐑛; 𐑯 𐑢𐑧𐑯 𐑞 𐑖𐑦𐑐 𐑢𐑪𐑟 𐑜𐑤𐑲𐑛𐑦𐑙 𐑚𐑲, 𐑤𐑲𐑒 𐑩 𐑓𐑤𐑨𐑖 𐑣𐑰 𐑛𐑸𐑑𐑩𐑛 𐑬𐑑; 𐑜𐑱𐑯𐑛 𐑣𐑻 𐑕𐑲𐑛; 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑚𐑨𐑒𐑢𐑼𐑛 𐑛𐑨𐑖 𐑝 𐑣𐑦𐑟 𐑓𐑫𐑑 𐑒𐑨𐑐𐑕𐑲𐑟𐑛 𐑯 𐑕𐑨𐑙𐑒 𐑣𐑦𐑟 𐑒𐑩𐑯𐑵; 𐑒𐑤𐑲𐑥𐑛 𐑳𐑐 𐑞 𐑗𐑱𐑯𐑟; 𐑯 𐑔𐑮𐑴𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑨𐑑 𐑓𐑫𐑤 𐑤𐑧𐑙𐑔 𐑩𐑐𐑪𐑯 𐑞 𐑛𐑧𐑒, 𐑜𐑮𐑨𐑐𐑩𐑤𐑛 𐑩 𐑮𐑦𐑙-𐑚𐑴𐑤𐑑 𐑞𐑺, 𐑯 𐑕𐑢𐑹 𐑯𐑪𐑑 𐑑 𐑤𐑧𐑑 𐑦𐑑 𐑜𐑴, 𐑞𐑴 𐑣𐑨𐑒𐑑 𐑦𐑯 𐑐𐑰𐑕𐑩𐑟.

𐑦𐑯 𐑝𐑱𐑯 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑔𐑮𐑧𐑑𐑩𐑯𐑛 𐑑 𐑔𐑮𐑴 𐑣𐑦𐑥 𐑴𐑝𐑼𐑚𐑹𐑛; 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑩 𐑒𐑳𐑑𐑤𐑩𐑕 𐑴𐑝𐑼 𐑣𐑦𐑟 𐑯𐑱𐑒𐑩𐑛 𐑮𐑦𐑕𐑑𐑕; ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑢𐑪𐑟 𐑞 𐑕𐑳𐑯 𐑝 𐑩 ·𐑒𐑦𐑙, 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑚𐑳𐑡𐑛 𐑯𐑪𐑑. 𐑕𐑑𐑮𐑳𐑒 𐑚𐑲 𐑣𐑦𐑟 𐑛𐑧𐑕𐑐𐑼𐑩𐑑 𐑛𐑷𐑯𐑑𐑤𐑩𐑕𐑯𐑩𐑕, 𐑯 𐑣𐑦𐑟 𐑢𐑲𐑤𐑛 𐑛𐑦𐑟𐑲𐑼 𐑑 𐑝𐑦𐑟𐑦𐑑 ·𐑒𐑮𐑦𐑕𐑩𐑯𐑛𐑩𐑥, 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑮𐑦𐑤𐑧𐑯𐑑𐑩𐑛, 𐑯 𐑑𐑴𐑤𐑛 𐑣𐑦𐑥 𐑣𐑰 𐑥𐑲𐑑 𐑥𐑱𐑒 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑨𐑑 𐑣𐑴𐑥. 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑓𐑲𐑯 𐑘𐑳𐑙 𐑕𐑨𐑝𐑦𐑡—𐑞𐑦𐑕 𐑕𐑰 ·𐑐𐑮𐑦𐑯𐑕 𐑝 ·𐑢𐑱𐑤𐑟, 𐑯𐑧𐑝𐑼 𐑕𐑷 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑒𐑨𐑚𐑦𐑯. 𐑞𐑱 𐑐𐑫𐑑 𐑣𐑦𐑥 𐑛𐑬𐑯 𐑩𐑥𐑳𐑙 𐑞 𐑕𐑱𐑤𐑼𐑟, 𐑯 𐑥𐑱𐑛 𐑩 𐑢𐑱𐑤𐑥𐑩𐑯 𐑝 𐑣𐑦𐑥. 𐑚𐑳𐑑 𐑤𐑲𐑒 ·𐑑𐑕𐑸 ·𐑐𐑰𐑑𐑼 𐑒𐑪𐑯𐑑𐑧𐑯𐑑 𐑑 𐑑𐑶𐑤 𐑦𐑯 𐑞 𐑖𐑦𐑐𐑘𐑸𐑛𐑟 𐑝 𐑓𐑪𐑮𐑦𐑯 𐑕𐑦𐑑𐑦𐑟, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑛𐑦𐑕𐑛𐑱𐑯𐑛 𐑯𐑴 𐑕𐑰𐑥𐑦𐑙 𐑦𐑜𐑯𐑩𐑥𐑦𐑯𐑦, 𐑦𐑓 𐑞𐑺𐑚𐑲 𐑣𐑰 𐑥𐑲𐑑 𐑣𐑨𐑐𐑦𐑤𐑦 𐑜𐑱𐑯 𐑞 𐑐𐑬𐑼 𐑝 𐑦𐑯𐑤𐑲𐑑𐑩𐑯𐑦𐑙 𐑣𐑦𐑟 𐑳𐑯𐑑𐑿𐑑𐑼𐑛 𐑒𐑳𐑯𐑑𐑮𐑦𐑥𐑧𐑯. 𐑓 𐑨𐑑 𐑚𐑪𐑑𐑩𐑥—𐑕𐑴 𐑣𐑰 𐑑𐑴𐑤𐑛 𐑥𐑰—𐑣𐑰 𐑢𐑪𐑟 𐑨𐑒𐑗𐑵𐑱𐑑𐑩𐑛 𐑚𐑲 𐑩 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑛𐑦𐑟𐑲𐑼 𐑑 𐑤𐑻𐑯 𐑩𐑥𐑳𐑙 𐑞 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯𐑟, 𐑞 𐑸𐑑𐑕 𐑢𐑺𐑚𐑲 𐑑 𐑥𐑱𐑒 𐑣𐑦𐑟 𐑐𐑰𐑐𐑩𐑤 𐑕𐑑𐑦𐑤 𐑣𐑨𐑐𐑽 𐑞𐑨𐑯 𐑞𐑱 𐑢𐑻; 𐑯 𐑥𐑹 𐑞𐑨𐑯 𐑞𐑨𐑑, 𐑕𐑑𐑦𐑤 𐑚𐑧𐑑𐑼 𐑞𐑨𐑯 𐑞𐑱 𐑢𐑻. 𐑚𐑳𐑑, 𐑩𐑤𐑨𐑕! 𐑞 𐑐𐑮𐑨𐑒𐑑𐑦𐑕𐑩𐑟 𐑝 𐑢𐑱𐑤𐑥𐑧𐑯 𐑕𐑵𐑯 𐑒𐑩𐑯𐑝𐑦𐑯𐑕𐑑 𐑣𐑦𐑥 𐑞𐑨𐑑 𐑰𐑝𐑩𐑯 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯𐑟 𐑒𐑫𐑛 𐑚𐑰 𐑚𐑴𐑔 𐑥𐑦𐑟𐑼𐑩𐑚𐑩𐑤 𐑯 𐑢𐑦𐑒𐑑; 𐑦𐑯𐑓𐑦𐑯𐑦𐑑𐑤𐑦 𐑥𐑹 𐑕𐑴, 𐑞𐑨𐑯 𐑷𐑤 𐑣𐑦𐑟 𐑓𐑭𐑞𐑼'𐑟 𐑣𐑰𐑞𐑩𐑯𐑟. 𐑼𐑲𐑝𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑦𐑯 𐑴𐑤𐑛 ·𐑕𐑨𐑜 ·𐑣𐑸𐑚𐑼; 𐑯 𐑕𐑰𐑦𐑙 𐑢𐑪𐑑 𐑞 𐑕𐑱𐑤𐑼𐑟 𐑛𐑦𐑛 𐑞𐑺; 𐑯 𐑞𐑧𐑯 𐑜𐑴𐑦𐑙 𐑪𐑯 𐑑 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑, 𐑯 𐑕𐑰𐑦𐑙 𐑣𐑬 𐑞𐑱 𐑕𐑐𐑧𐑯𐑑 𐑞𐑺 𐑢𐑱𐑡𐑩𐑟 𐑦𐑯 𐑞𐑨𐑑 𐑐𐑤𐑱𐑕 𐑷𐑤𐑕𐑴, 𐑐𐑫𐑼 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑜𐑱𐑝 𐑦𐑑 𐑳𐑐 𐑓 𐑤𐑪𐑕𐑑. 𐑔𐑷𐑑 𐑣𐑰, 𐑦𐑑'𐑕 𐑩 𐑢𐑦𐑒𐑩𐑛 𐑢𐑻𐑤𐑛 𐑦𐑯 𐑷𐑤 𐑥𐑼𐑦𐑛𐑾𐑯𐑟; 𐑲'𐑤 𐑛𐑲 𐑩 𐑐𐑱𐑜𐑩𐑯.

𐑯 𐑞𐑳𐑕 𐑩𐑯 𐑴𐑤𐑛 𐑲𐑛𐑪𐑤𐑩𐑑𐑼 𐑨𐑑 𐑣𐑸𐑑, 𐑣𐑰 𐑘𐑧𐑑 𐑤𐑦𐑝𐑛 𐑩𐑥𐑳𐑙 𐑞𐑰𐑟 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯𐑟, 𐑢𐑹 𐑞𐑺 𐑒𐑤𐑴𐑞𐑟, 𐑯 𐑑𐑮𐑲𐑛 𐑑 𐑑𐑷𐑒 𐑞𐑺 𐑡𐑦𐑚𐑼𐑦𐑖. 𐑣𐑧𐑯𐑕 𐑞 𐑒𐑢𐑽 𐑢𐑱𐑟 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥, 𐑞𐑴 𐑯𐑬 𐑕𐑳𐑥 𐑑𐑲𐑥 𐑓𐑮𐑪𐑥 𐑣𐑴𐑥.

𐑚𐑲 𐑣𐑦𐑯𐑑𐑕, 𐑲 𐑭𐑕𐑒𐑑 𐑣𐑦𐑥 𐑢𐑧𐑞𐑼 𐑣𐑰 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑐𐑮𐑩𐑐𐑴𐑟 𐑜𐑴𐑦𐑙 𐑚𐑨𐑒, 𐑯 𐑣𐑨𐑝𐑦𐑙 𐑩 𐑒𐑪𐑮𐑩𐑯𐑱𐑖𐑩𐑯; 𐑕𐑦𐑯𐑕 𐑣𐑰 𐑥𐑲𐑑 𐑯𐑬 𐑒𐑩𐑯𐑕𐑦𐑛𐑼 𐑣𐑦𐑟 𐑓𐑭𐑞𐑼 𐑛𐑧𐑛 𐑯 𐑜𐑪𐑯, 𐑣𐑰 𐑚𐑰𐑦𐑙 𐑝𐑧𐑮𐑦 𐑴𐑤𐑛 𐑯 𐑓𐑰𐑚𐑩𐑤 𐑨𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑩𐑒𐑬𐑯𐑑𐑕. 𐑣𐑰 𐑭𐑯𐑕𐑼𐑛 𐑯𐑴, 𐑯𐑪𐑑 𐑘𐑧𐑑; 𐑯 𐑨𐑛𐑩𐑛 𐑞𐑨𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑓𐑽𐑓𐑩𐑤 ·𐑒𐑮𐑦𐑕𐑑𐑦𐑨𐑯𐑦𐑑𐑦, 𐑹 𐑮𐑭𐑞𐑼 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯𐑟, 𐑣𐑨𐑛 𐑳𐑯𐑓𐑦𐑑𐑩𐑛 𐑣𐑦𐑥 𐑓 𐑩𐑕𐑧𐑯𐑛𐑦𐑙 𐑞 𐑐𐑘𐑫𐑼 𐑯 𐑳𐑯𐑛𐑦𐑓𐑲𐑤𐑛 𐑔𐑮𐑴𐑯 𐑝 𐑔𐑻𐑑𐑦 𐑐𐑱𐑜𐑩𐑯 ·𐑒𐑦𐑙𐑟 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥. 𐑚𐑳𐑑 𐑚𐑲 𐑯 𐑚𐑲, 𐑣𐑰 𐑕𐑧𐑛, 𐑣𐑰 𐑢𐑫𐑛 𐑮𐑦𐑑𐑻𐑯,—𐑨𐑟 𐑕𐑵𐑯 𐑨𐑟 𐑣𐑰 𐑓𐑧𐑤𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑚𐑨𐑐𐑑𐑲𐑟𐑛 𐑩𐑜𐑱𐑯. 𐑓 𐑞 𐑯𐑪𐑯𐑕, 𐑣𐑬𐑧𐑝𐑼, 𐑣𐑰 𐑐𐑮𐑩𐑐𐑴𐑟𐑛 𐑑 𐑕𐑱𐑤 𐑩𐑚𐑬𐑑, 𐑯 𐑕𐑴 𐑣𐑦𐑟 𐑢𐑲𐑤𐑛 𐑴𐑑𐑕 𐑦𐑯 𐑷𐑤 𐑓𐑹 𐑴𐑖𐑩𐑯𐑟. 𐑞𐑱 𐑣𐑨𐑛 𐑥𐑱𐑛 𐑩 𐑣𐑸𐑐𐑵𐑯𐑽 𐑝 𐑣𐑦𐑥, 𐑯 𐑞𐑨𐑑 𐑚𐑸𐑚𐑛 𐑲𐑼𐑯 𐑢𐑪𐑟 𐑦𐑯 𐑤𐑵 𐑝 𐑩 𐑕𐑧𐑐𐑑𐑼 𐑯𐑬.

𐑲 𐑭𐑕𐑒𐑑 𐑣𐑦𐑥 𐑢𐑪𐑑 𐑥𐑲𐑑 𐑚𐑰 𐑣𐑦𐑟 𐑦𐑥𐑰𐑛𐑾𐑑 𐑐𐑻𐑐𐑩𐑕, 𐑑𐑳𐑗𐑦𐑙 𐑣𐑦𐑟 𐑓𐑿𐑗𐑼 𐑥𐑵𐑝𐑥𐑩𐑯𐑑𐑕. 𐑣𐑰 𐑭𐑯𐑕𐑼𐑛, 𐑑 𐑜𐑴 𐑑 𐑕𐑰 𐑩𐑜𐑱𐑯, 𐑦𐑯 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑝𐑴𐑒𐑱𐑖𐑩𐑯. 𐑩𐑐𐑪𐑯 𐑞𐑦𐑕, 𐑲 𐑑𐑴𐑤𐑛 𐑣𐑦𐑥 𐑞𐑨𐑑 𐑢𐑱𐑤𐑦𐑙 𐑢𐑪𐑟 𐑥𐑲 𐑴𐑯 𐑛𐑦𐑟𐑲𐑯, 𐑯 𐑦𐑯𐑓𐑹𐑥𐑛 𐑣𐑦𐑥 𐑝 𐑥𐑲 𐑦𐑯𐑑𐑧𐑯𐑖𐑩𐑯 𐑑 𐑕𐑱𐑤 𐑬𐑑 𐑝 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑, 𐑨𐑟 𐑚𐑰𐑦𐑙 𐑞 𐑥𐑴𐑕𐑑 𐑐𐑮𐑪𐑥𐑦𐑕𐑦𐑙 𐑐𐑹𐑑 𐑓 𐑩𐑯 𐑩𐑛𐑝𐑧𐑯𐑗𐑼𐑩𐑕 𐑢𐑱𐑤𐑥𐑩𐑯 𐑑 𐑦𐑥𐑚𐑸𐑒 𐑓𐑮𐑪𐑥. 𐑣𐑰 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑮𐑦𐑟𐑪𐑤𐑝𐑛 𐑑 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑥𐑰 𐑑 𐑞𐑨𐑑 𐑲𐑤𐑩𐑯𐑛, 𐑖𐑦𐑐 𐑩𐑚𐑹𐑛 𐑞 𐑕𐑱𐑥 𐑝𐑧𐑕𐑩𐑤, 𐑜𐑧𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑱𐑥 𐑢𐑪𐑗, 𐑞 𐑕𐑱𐑥 𐑚𐑴𐑑, 𐑞 𐑕𐑱𐑥 𐑥𐑧𐑕 𐑢𐑦𐑞 𐑥𐑰, 𐑦𐑯 𐑖𐑹𐑑 𐑑 𐑖𐑺 𐑥𐑲 𐑧𐑝𐑮𐑦 𐑣𐑨𐑐; 𐑢𐑦𐑞 𐑚𐑴𐑔 𐑥𐑲 𐑣𐑨𐑯𐑛𐑟 𐑦𐑯 𐑣𐑦𐑟, 𐑚𐑴𐑤𐑛𐑤𐑦 𐑛𐑦𐑐 𐑦𐑯𐑑𐑵 𐑞 ·𐑐𐑪𐑑𐑤𐑳𐑒 𐑝 𐑚𐑴𐑔 𐑢𐑻𐑤𐑛𐑟. 𐑑 𐑷𐑤 𐑞𐑦𐑕 𐑲 𐑡𐑶𐑩𐑕𐑤𐑦 𐑩𐑕𐑧𐑯𐑑𐑩𐑛; 𐑓 𐑚𐑦𐑕𐑲𐑛𐑟 𐑞 𐑩𐑓𐑧𐑒𐑖𐑩𐑯 𐑲 𐑯𐑬 𐑓𐑧𐑤𐑑 𐑓 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑣𐑰 𐑢𐑪𐑟 𐑩𐑯 𐑦𐑒𐑕𐑐𐑽𐑾𐑯𐑕𐑑 𐑣𐑸𐑐𐑵𐑯𐑽, 𐑯 𐑨𐑟 𐑕𐑳𐑗, 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑓𐑱𐑤 𐑑 𐑚𐑰 𐑝 𐑜𐑮𐑱𐑑 𐑿𐑕𐑓𐑩𐑤𐑯𐑩𐑕 𐑑 𐑢𐑳𐑯, 𐑣𐑵, 𐑤𐑲𐑒 𐑥𐑰, 𐑢𐑪𐑟 𐑣𐑴𐑤𐑦 𐑦𐑜𐑯𐑼𐑩𐑯𐑑 𐑝 𐑞 𐑥𐑦𐑕𐑑𐑼𐑦𐑟 𐑝 𐑢𐑱𐑤𐑦𐑙, 𐑞𐑴 𐑢𐑧𐑤 𐑩𐑒𐑢𐑱𐑯𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞 𐑕𐑰, 𐑨𐑟 𐑯𐑴𐑯 𐑑 𐑥𐑻𐑗𐑩𐑯𐑑 𐑕𐑰𐑥𐑧𐑯.

𐑣𐑦𐑟 𐑕𐑑𐑹𐑦 𐑚𐑰𐑦𐑙 𐑧𐑯𐑛𐑩𐑛 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑐𐑲𐑐'𐑕 𐑤𐑭𐑕𐑑 𐑛𐑲𐑦𐑙 𐑐𐑳𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑦𐑥𐑚𐑮𐑱𐑕𐑑 𐑥𐑰, 𐑐𐑮𐑧𐑕𐑑 𐑣𐑦𐑟 𐑓𐑪𐑮𐑦𐑛 𐑩𐑜𐑱𐑯𐑕𐑑 𐑥𐑲𐑯, 𐑯 𐑚𐑤𐑴𐑦𐑙 𐑬𐑑 𐑞 𐑤𐑲𐑑, 𐑢𐑰 𐑮𐑴𐑤𐑛 𐑴𐑝𐑼 𐑓𐑮𐑪𐑥 𐑰𐑗 𐑳𐑞𐑼, 𐑞𐑦𐑕 𐑢𐑱 𐑯 𐑞𐑨𐑑, 𐑯 𐑝𐑧𐑮𐑦 𐑕𐑵𐑯 𐑢𐑻 𐑕𐑤𐑰𐑐𐑦𐑙.
