·𐑗𐑨𐑐𐑑𐑼 116. 𐑞 ·𐑛𐑲𐑦𐑙 ·𐑢𐑱𐑤.

𐑯𐑪𐑑 𐑕𐑧𐑤𐑛𐑩𐑥 𐑦𐑯 𐑞𐑦𐑕 𐑤𐑲𐑓, 𐑢𐑧𐑯, 𐑪𐑯 𐑞 𐑮𐑲𐑑 𐑕𐑲𐑛, 𐑓𐑹𐑗𐑵𐑯'𐑟 𐑓𐑱𐑝𐑼𐑦𐑑𐑕 𐑕𐑱𐑤 𐑒𐑤𐑴𐑕 𐑚𐑲 𐑳𐑕, 𐑢𐑰, 𐑞𐑴 𐑷𐑤 𐑱𐑛𐑮𐑵𐑐 𐑚𐑦𐑓𐑹, 𐑒𐑨𐑗 𐑕𐑳𐑥𐑢𐑪𐑑 𐑝 𐑞 𐑮𐑳𐑖𐑦𐑙 𐑚𐑮𐑰𐑟, 𐑯 𐑡𐑶𐑓𐑩𐑤𐑦 𐑓𐑰𐑤 𐑬𐑼 𐑚𐑨𐑜𐑦𐑙 𐑕𐑱𐑤𐑟 𐑓𐑦𐑤 𐑬𐑑. 𐑕𐑴 𐑕𐑰𐑥𐑛 𐑦𐑑 𐑢𐑦𐑞 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛. 𐑓 𐑯𐑧𐑒𐑕𐑑 𐑛𐑱 𐑭𐑓𐑑𐑼 𐑦𐑯𐑒𐑬𐑯𐑑𐑼𐑦𐑙 𐑞 𐑜𐑱 ·𐑚𐑨𐑗𐑩𐑤𐑼, 𐑢𐑱𐑤𐑟 𐑢𐑻 𐑕𐑰𐑯 𐑯 𐑓𐑹 𐑢𐑻 𐑕𐑤𐑱𐑯; 𐑯 𐑢𐑳𐑯 𐑝 𐑞𐑧𐑥 𐑚𐑲 ·𐑱𐑣𐑨𐑚.

𐑦𐑑 𐑢𐑪𐑟 𐑓𐑸 𐑛𐑬𐑯 𐑞 𐑭𐑓𐑑𐑼𐑯𐑵𐑯; 𐑯 𐑢𐑧𐑯 𐑷𐑤 𐑞 𐑕𐑐𐑽𐑦𐑙𐑟 𐑝 𐑞 𐑒𐑮𐑦𐑥𐑟𐑩𐑯 𐑓𐑲𐑑 𐑢𐑻 𐑛𐑳𐑯: 𐑯 𐑓𐑤𐑴𐑑𐑦𐑙 𐑦𐑯 𐑞 𐑤𐑳𐑝𐑤𐑦 𐑕𐑳𐑯𐑕𐑧𐑑 𐑕𐑰 𐑯 𐑕𐑒𐑲, 𐑕𐑳𐑯 𐑯 𐑢𐑱𐑤 𐑚𐑴𐑔 𐑕𐑑𐑦𐑤𐑦 𐑛𐑲𐑛 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑞𐑧𐑯, 𐑕𐑳𐑗 𐑩 𐑕𐑢𐑰𐑑𐑯𐑩𐑕 𐑯 𐑕𐑳𐑗 𐑐𐑤𐑱𐑯𐑑𐑦𐑝𐑯𐑩𐑕, 𐑕𐑳𐑗 𐑦𐑯𐑮𐑰𐑞𐑦𐑙 𐑪𐑮𐑦𐑟𐑩𐑯𐑟 𐑒𐑻𐑤𐑛 𐑳𐑐 𐑦𐑯 𐑞𐑨𐑑 𐑮𐑴𐑟𐑦 𐑺, 𐑞𐑨𐑑 𐑦𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰𐑥𐑛 𐑨𐑟 𐑦𐑓 𐑓𐑸 𐑴𐑝𐑼 𐑓𐑮𐑪𐑥 𐑞 𐑛𐑰𐑐 𐑜𐑮𐑰𐑯 𐑒𐑪𐑯𐑝𐑧𐑯𐑑 𐑝𐑨𐑤𐑦𐑟 𐑝 𐑞 ·𐑥𐑩𐑯𐑦𐑤𐑩 𐑲𐑤𐑟, 𐑞 ·𐑕𐑐𐑨𐑯𐑦𐑖 𐑤𐑨𐑯𐑛-𐑚𐑮𐑰𐑟, 𐑢𐑪𐑯𐑑𐑩𐑯𐑤𐑦 𐑑𐑻𐑯𐑛 𐑕𐑱𐑤𐑼, 𐑣𐑨𐑛 𐑜𐑪𐑯 𐑑 𐑕𐑰, 𐑓𐑮𐑱𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞𐑰𐑟 𐑝𐑧𐑕𐑐𐑼 𐑣𐑦𐑥𐑟.

𐑕𐑵𐑞𐑛 𐑩𐑜𐑱𐑯, 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑕𐑵𐑞𐑛 𐑑 𐑛𐑰𐑐𐑼 𐑜𐑤𐑵𐑥, ·𐑱𐑣𐑨𐑚, 𐑣𐑵 𐑣𐑨𐑛 𐑕𐑑𐑻𐑯𐑛 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑱𐑤, 𐑕𐑨𐑑 𐑦𐑯𐑑𐑧𐑯𐑑𐑤𐑦 𐑢𐑪𐑗𐑦𐑙 𐑣𐑦𐑟 𐑓𐑲𐑯𐑩𐑤 𐑢𐑱𐑯𐑦𐑙𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑯𐑬 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤 𐑚𐑴𐑑. 𐑓 𐑞𐑨𐑑 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑐𐑧𐑒𐑑𐑩𐑒𐑩𐑤 𐑩𐑚𐑟𐑻𐑝𐑩𐑚𐑩𐑤 𐑦𐑯 𐑷𐑤 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤𐑟 𐑛𐑲𐑦𐑙—𐑞 𐑑𐑻𐑯𐑦𐑙 𐑕𐑳𐑯𐑢𐑼𐑛𐑟 𐑝 𐑞 𐑣𐑧𐑛, 𐑯 𐑕𐑴 𐑦𐑒𐑕𐑐𐑲𐑼𐑦𐑙—𐑞𐑨𐑑 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑐𐑧𐑒𐑑𐑩𐑒𐑩𐑤, 𐑚𐑦𐑣𐑧𐑤𐑛 𐑝 𐑕𐑳𐑗 𐑩 𐑐𐑤𐑨𐑕𐑦𐑛 𐑰𐑝𐑯𐑦𐑙, 𐑕𐑳𐑥𐑣𐑬 𐑑 ·𐑱𐑣𐑨𐑚 𐑒𐑩𐑯𐑝𐑱𐑛 𐑩 𐑢𐑳𐑯𐑛𐑮𐑩𐑕𐑯𐑩𐑕 𐑳𐑯𐑯𐑴𐑯 𐑚𐑦𐑓𐑹.

“𐑣𐑰 𐑑𐑻𐑯𐑟 𐑯 𐑑𐑻𐑯𐑟 𐑣𐑦𐑥 𐑑 𐑦𐑑,—𐑣𐑬 𐑕𐑤𐑴𐑤𐑦, 𐑚𐑳𐑑 𐑣𐑬 𐑕𐑑𐑧𐑛𐑓𐑭𐑕𐑑𐑤𐑦, 𐑣𐑦𐑟 𐑣𐑪𐑥𐑦𐑡-𐑮𐑧𐑯𐑛𐑼𐑦𐑙 𐑯 𐑦𐑯𐑝𐑴𐑒𐑦𐑙 𐑚𐑮𐑬, 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑤𐑭𐑕𐑑 𐑛𐑲𐑦𐑙 𐑥𐑴𐑖𐑩𐑯𐑟. 𐑣𐑰 𐑑𐑵 𐑢𐑻𐑖𐑦𐑐𐑕 𐑓𐑲𐑼; 𐑥𐑴𐑕𐑑 𐑓𐑱𐑔𐑓𐑩𐑤, 𐑚𐑮𐑷𐑛, 𐑚𐑼𐑴𐑯𐑾𐑤 𐑝𐑨𐑕𐑩𐑤 𐑝 𐑞 𐑕𐑳𐑯!—𐑴 𐑞𐑨𐑑 𐑞𐑰𐑟 𐑑𐑵-𐑓𐑱𐑝𐑼𐑦𐑙 𐑲𐑟 𐑖𐑵𐑛 𐑕𐑰 𐑞𐑰𐑟 𐑑𐑵-𐑓𐑱𐑝𐑼𐑦𐑙 𐑕𐑲𐑑𐑕. 𐑤𐑫𐑒! 𐑣𐑽, 𐑓𐑸 𐑢𐑷𐑑𐑼-𐑤𐑪𐑒𐑑; 𐑚𐑦𐑘𐑪𐑯𐑛 𐑷𐑤 𐑣𐑳𐑥 𐑝 𐑣𐑿𐑥𐑩𐑯 𐑢𐑰𐑤 𐑹 𐑢𐑴; 𐑦𐑯 𐑞𐑰𐑟 𐑥𐑴𐑕𐑑 𐑒𐑨𐑯𐑛𐑦𐑛 𐑯 𐑦𐑥𐑐𐑸𐑖𐑩𐑤 𐑕𐑰𐑟; 𐑢𐑺 𐑑 𐑑𐑮𐑩𐑛𐑦𐑖𐑩𐑯𐑟 𐑯𐑴 𐑮𐑪𐑒𐑕 𐑓𐑻𐑯𐑦𐑖 𐑑𐑨𐑚𐑤𐑩𐑑𐑕; 𐑢𐑺 𐑓 𐑤𐑪𐑙 ·𐑗𐑲𐑯𐑰𐑟 𐑱𐑡𐑩𐑟, 𐑞 𐑚𐑦𐑤𐑴𐑟 𐑣𐑨𐑝 𐑕𐑑𐑦𐑤 𐑮𐑴𐑤𐑛 𐑪𐑯 𐑕𐑐𐑰𐑗𐑤𐑩𐑕 𐑯 𐑳𐑯𐑕𐑐𐑴𐑒𐑩𐑯 𐑑, 𐑨𐑟 𐑕𐑑𐑸𐑟 𐑞𐑨𐑑 𐑖𐑲𐑯 𐑩𐑐𐑪𐑯 𐑞 ·𐑯𐑰𐑠𐑺'𐑟 𐑳𐑯𐑯𐑴𐑯 𐑕𐑹𐑕; 𐑣𐑽, 𐑑𐑵, 𐑤𐑲𐑓 𐑛𐑲𐑟 𐑕𐑳𐑯𐑢𐑼𐑛𐑟 𐑓𐑫𐑤 𐑝 𐑓𐑱𐑔; 𐑚𐑳𐑑 𐑕𐑰! 𐑯𐑴 𐑕𐑵𐑯𐑼 𐑛𐑧𐑛, 𐑞𐑨𐑯 𐑛𐑧𐑔 𐑢𐑻𐑤𐑟 𐑮𐑬𐑯𐑛 𐑞 𐑒𐑹𐑐𐑕, 𐑯 𐑦𐑑 𐑣𐑧𐑛𐑟 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑢𐑱.

“𐑴, 𐑞𐑬 𐑛𐑸𐑒 ·𐑣𐑦𐑯𐑛𐑵 𐑣𐑭𐑓 𐑝 𐑯𐑱𐑗𐑼, 𐑣𐑵 𐑝 𐑛𐑮𐑬𐑯𐑛 𐑚𐑴𐑯𐑟 𐑣𐑨𐑕𐑑 𐑚𐑦𐑤𐑛𐑩𐑛 𐑞𐑲 𐑕𐑧𐑐𐑼𐑩𐑑 𐑔𐑮𐑴𐑯 𐑕𐑳𐑥𐑢𐑺 𐑦𐑯 𐑞 𐑣𐑸𐑑 𐑝 𐑞𐑰𐑟 𐑳𐑯𐑝𐑻𐑡𐑼𐑛 𐑕𐑰𐑟; 𐑞𐑬 𐑸𐑑 𐑩𐑯 𐑦𐑯𐑓𐑦𐑛𐑩𐑤, 𐑞𐑬 𐑒𐑢𐑰𐑯, 𐑯 𐑑𐑵 𐑑𐑮𐑵𐑤𐑦 𐑕𐑐𐑰𐑒𐑩𐑕𐑑 𐑑 𐑥𐑰 𐑦𐑯 𐑞 𐑢𐑲𐑛-𐑕𐑤𐑷𐑑𐑼𐑦𐑙 ·𐑑𐑲𐑓𐑵𐑯, 𐑯 𐑞 𐑣𐑳𐑖𐑑 𐑚𐑧𐑮𐑾𐑤 𐑝 𐑦𐑑𐑕 𐑭𐑓𐑑𐑼 𐑒𐑭𐑥. 𐑯𐑹 𐑣𐑨𐑟 𐑞𐑦𐑕 𐑞𐑲 𐑢𐑱𐑤 𐑕𐑳𐑯𐑢𐑼𐑛𐑟 𐑑𐑻𐑯𐑛 𐑣𐑦𐑟 𐑛𐑲𐑦𐑙 𐑣𐑧𐑛, 𐑯 𐑞𐑧𐑯 𐑜𐑪𐑯 𐑮𐑬𐑯𐑛 𐑩𐑜𐑱𐑯, 𐑢𐑦𐑞𐑬𐑑 𐑩 𐑤𐑧𐑕𐑩𐑯 𐑑 𐑥𐑰.

“𐑴, 𐑑𐑮𐑧𐑚𐑤𐑦 𐑣𐑵𐑐𐑑 𐑯 𐑢𐑧𐑤𐑛𐑩𐑛 𐑣𐑦𐑐 𐑝 𐑐𐑬𐑼! 𐑴, 𐑣𐑲 𐑩𐑕𐑐𐑲𐑼𐑦𐑙, 𐑮𐑱𐑯𐑚𐑴𐑛 𐑡𐑧𐑑!—𐑞𐑨𐑑 𐑢𐑳𐑯 𐑕𐑑𐑮𐑲𐑝𐑩𐑕𐑑, 𐑞𐑦𐑕 𐑢𐑳𐑯 𐑡𐑧𐑑𐑩𐑕𐑑 𐑷𐑤 𐑦𐑯 𐑝𐑱𐑯! 𐑦𐑯 𐑝𐑱𐑯, 𐑴 𐑢𐑱𐑤, 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑕𐑰𐑒 𐑦𐑯𐑑𐑼𐑕𐑰𐑛𐑦𐑙𐑟 𐑢𐑦𐑞 𐑘𐑪𐑯 𐑷𐑤-𐑒𐑢𐑦𐑒𐑩𐑯𐑦𐑙 𐑕𐑳𐑯, 𐑞𐑨𐑑 𐑴𐑯𐑤𐑦 𐑒𐑷𐑤𐑟 𐑓𐑹𐑔 𐑤𐑲𐑓, 𐑚𐑳𐑑 𐑜𐑦𐑝𐑟 𐑦𐑑 𐑯𐑪𐑑 𐑩𐑜𐑱𐑯. 𐑘𐑧𐑑 𐑛𐑳𐑕𐑑 𐑞𐑬, 𐑛𐑸𐑒𐑼 𐑣𐑭𐑓, 𐑮𐑪𐑒 𐑥𐑰 𐑢𐑦𐑞 𐑩 𐑐𐑮𐑬𐑛𐑼, 𐑦𐑓 𐑩 𐑛𐑸𐑒𐑼 𐑓𐑱𐑔. 𐑷𐑤 𐑞𐑲 𐑳𐑯𐑯𐑱𐑥𐑩𐑚𐑩𐑤 𐑦𐑥𐑦𐑙𐑜𐑩𐑤𐑦𐑙𐑟 𐑓𐑤𐑴𐑑 𐑚𐑦𐑯𐑰𐑔 𐑥𐑰 𐑣𐑽; 𐑲 𐑨𐑥 𐑚𐑶𐑛 𐑚𐑲 𐑚𐑮𐑧𐑔𐑕 𐑝 𐑢𐑳𐑯𐑕 𐑤𐑦𐑝𐑦𐑙 𐑔𐑦𐑙𐑟, 𐑧𐑒𐑕𐑣𐑱𐑤𐑛 𐑨𐑟 𐑺, 𐑚𐑳𐑑 𐑢𐑷𐑑𐑼 𐑯𐑬.

“𐑞𐑧𐑯 𐑣𐑱𐑤, 𐑓 𐑧𐑝𐑼 𐑣𐑱𐑤, 𐑴 𐑕𐑰, 𐑦𐑯 𐑣𐑵𐑟 𐑦𐑑𐑻𐑯𐑩𐑤 𐑑𐑪𐑕𐑦𐑙𐑟 𐑞 𐑢𐑲𐑤𐑛 𐑓𐑬𐑤 𐑓𐑲𐑯𐑛𐑟 𐑣𐑦𐑟 𐑴𐑯𐑤𐑦 𐑮𐑧𐑕𐑑. 𐑚𐑹𐑯 𐑝 𐑻𐑔, 𐑘𐑧𐑑 𐑕𐑳𐑒𐑩𐑤𐑛 𐑚𐑲 𐑞 𐑕𐑰; 𐑞𐑴 𐑣𐑦𐑤 𐑯 𐑝𐑨𐑤𐑦 𐑥𐑳𐑞𐑼𐑛 𐑥𐑰, 𐑘𐑰 𐑚𐑦𐑤𐑴𐑟 𐑸 𐑥𐑲 𐑓𐑪𐑕𐑑𐑼-𐑚𐑮𐑳𐑞𐑼𐑟!”
