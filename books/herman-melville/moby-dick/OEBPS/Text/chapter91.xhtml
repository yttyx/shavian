·𐑗𐑨𐑐𐑑𐑼 91. 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 ·𐑥𐑰𐑑𐑕 𐑞 ·𐑮𐑴𐑟-·𐑚𐑳𐑛.

“𐑦𐑯 𐑝𐑱𐑯 𐑦𐑑 𐑢𐑪𐑟 𐑑 𐑮𐑱𐑒 𐑓 ·𐑨𐑥𐑚𐑼𐑜𐑮𐑦𐑰𐑟 𐑦𐑯 𐑞 𐑐𐑷𐑯𐑗 𐑝 𐑞𐑦𐑕 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯, 𐑦𐑯𐑕𐑳𐑓𐑼𐑩𐑚𐑩𐑤 𐑓𐑰𐑑𐑼 𐑛𐑦𐑯𐑲𐑦𐑙 𐑯𐑪𐑑 𐑦𐑯𐑒𐑢𐑲𐑼𐑦.” 𐑕𐑻 T. 𐑚𐑮𐑬𐑯, V.E.

·𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑢𐑰𐑒 𐑹 𐑑𐑵 𐑭𐑓𐑑𐑼 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑱𐑤𐑦𐑙 𐑕𐑰𐑯 𐑮𐑦𐑒𐑬𐑯𐑑𐑩𐑛, 𐑯 𐑢𐑧𐑯 𐑢𐑰 𐑢𐑻 𐑕𐑤𐑴𐑤𐑦 𐑕𐑱𐑤𐑦𐑙 𐑴𐑝𐑼 𐑩 𐑕𐑤𐑰𐑐𐑦, 𐑝𐑱𐑐𐑼𐑦, 𐑥𐑦𐑛-𐑛𐑱 𐑕𐑰, 𐑞𐑨𐑑 𐑞 𐑥𐑧𐑯𐑦 𐑯𐑴𐑟𐑩𐑟 𐑪𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑛𐑧𐑒 𐑐𐑮𐑵𐑝𐑛 𐑥𐑹 𐑝𐑦𐑡𐑦𐑤𐑩𐑯𐑑 𐑛𐑦𐑕𐑒𐑳𐑝𐑼𐑼𐑟 𐑞𐑨𐑯 𐑞 𐑔𐑮𐑰 𐑐𐑺𐑟 𐑝 𐑲𐑟 𐑩𐑤𐑪𐑓𐑑. 𐑩 𐑐𐑦𐑒𐑿𐑤𐑽 𐑯 𐑯𐑪𐑑 𐑝𐑧𐑮𐑦 𐑐𐑤𐑧𐑟𐑩𐑯𐑑 𐑕𐑥𐑧𐑤 𐑢𐑪𐑟 𐑕𐑥𐑧𐑤𐑑 𐑦𐑯 𐑞 𐑕𐑰.

“𐑲 𐑢𐑦𐑤 𐑚𐑧𐑑 𐑕𐑳𐑥𐑔𐑦𐑙 𐑯𐑬,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, “𐑞𐑨𐑑 𐑕𐑳𐑥𐑢𐑺 𐑣𐑽𐑩𐑚𐑬𐑑𐑕 𐑸 𐑕𐑳𐑥 𐑝 𐑞𐑴𐑟 𐑛𐑮𐑳𐑜𐑛 𐑢𐑱𐑤𐑟 𐑢𐑰 𐑑𐑦𐑒𐑩𐑤𐑛 𐑞 𐑳𐑞𐑼 𐑛𐑱. 𐑲 𐑔𐑷𐑑 𐑞𐑱 𐑢𐑫𐑛 𐑒𐑰𐑤 𐑳𐑐 𐑚𐑦𐑓𐑹 𐑤𐑪𐑙.”

𐑐𐑮𐑧𐑟𐑩𐑯𐑑𐑤𐑦, 𐑞 𐑝𐑱𐑐𐑼𐑟 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕 𐑕𐑤𐑦𐑛 𐑩𐑕𐑲𐑛; 𐑯 𐑞𐑺 𐑦𐑯 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑤𐑱 𐑩 𐑖𐑦𐑐, 𐑣𐑵𐑟 𐑓𐑻𐑤𐑛 𐑕𐑱𐑤𐑟 𐑚𐑦𐑑𐑴𐑒𐑩𐑯𐑛 𐑞𐑨𐑑 𐑕𐑳𐑥 𐑕𐑹𐑑 𐑝 𐑢𐑱𐑤 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑩𐑤𐑪𐑙𐑕𐑲𐑛. 𐑨𐑟 𐑢𐑰 𐑜𐑤𐑲𐑛𐑩𐑛 𐑯𐑽𐑼, 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑖𐑴𐑛 ·𐑓𐑮𐑧𐑯𐑗 𐑒𐑳𐑤𐑼𐑟 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑐𐑰𐑒; 𐑯 𐑚𐑲 𐑞 𐑧𐑛𐑦𐑦𐑙 𐑒𐑤𐑬𐑛 𐑝 𐑝𐑳𐑤𐑗𐑼 𐑕𐑰-𐑓𐑬𐑤 𐑞𐑨𐑑 𐑕𐑻𐑒𐑩𐑤𐑛, 𐑯 𐑣𐑪𐑝𐑼𐑛, 𐑯 𐑕𐑢𐑵𐑐𐑑 𐑼𐑬𐑯𐑛 𐑣𐑦𐑥, 𐑦𐑑 𐑢𐑪𐑟 𐑐𐑤𐑱𐑯 𐑞𐑨𐑑 𐑞 𐑢𐑱𐑤 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑢𐑪𐑑 𐑞 𐑓𐑦𐑖𐑼𐑥𐑧𐑯 𐑒𐑷𐑤 𐑩 𐑚𐑤𐑭𐑕𐑑𐑩𐑛 𐑢𐑱𐑤, 𐑞𐑨𐑑 𐑦𐑟, 𐑩 𐑢𐑱𐑤 𐑞𐑨𐑑 𐑣𐑨𐑟 𐑛𐑲𐑛 𐑳𐑯𐑥𐑩𐑤𐑧𐑕𐑑𐑩𐑛 𐑪𐑯 𐑞 𐑕𐑰, 𐑯 𐑕𐑴 𐑓𐑤𐑴𐑑𐑩𐑛 𐑩𐑯 𐑳𐑯𐑩𐑐𐑮𐑴𐑐𐑮𐑦𐑱𐑑𐑩𐑛 𐑒𐑹𐑐𐑕. 𐑦𐑑 𐑥𐑱 𐑢𐑧𐑤 𐑚𐑰 𐑒𐑩𐑯𐑕𐑰𐑝𐑛, 𐑢𐑪𐑑 𐑩𐑯 𐑳𐑯𐑕𐑱𐑝𐑼𐑦 𐑴𐑛𐑼 𐑕𐑳𐑗 𐑩 𐑥𐑨𐑕 𐑥𐑳𐑕𐑑 𐑧𐑒𐑕𐑣𐑱𐑤; 𐑢𐑻𐑕 𐑞𐑨𐑯 𐑩𐑯 ·𐑩𐑕𐑽𐑾𐑯 𐑕𐑦𐑑𐑦 𐑦𐑯 𐑞 𐑐𐑤𐑱𐑜, 𐑢𐑧𐑯 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑸 𐑦𐑯𐑒𐑪𐑥𐑐𐑦𐑑𐑩𐑯𐑑 𐑑 𐑚𐑧𐑮𐑦 𐑞 𐑛𐑦𐑐𐑸𐑑𐑩𐑛. 𐑕𐑴 𐑦𐑯𐑑𐑪𐑤𐑼𐑩𐑚𐑩𐑤 𐑦𐑯𐑛𐑰𐑛 𐑦𐑟 𐑦𐑑 𐑮𐑦𐑜𐑸𐑛𐑩𐑛 𐑚𐑲 𐑕𐑳𐑥, 𐑞𐑨𐑑 𐑯𐑴 𐑒𐑿𐑐𐑦𐑛𐑦𐑑𐑦 𐑒𐑫𐑛 𐑐𐑼𐑕𐑢𐑱𐑛 𐑞𐑧𐑥 𐑑 𐑥𐑫𐑼 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑝 𐑦𐑑. 𐑘𐑧𐑑 𐑸 𐑞𐑺 𐑞𐑴𐑟 𐑣𐑵 𐑢𐑦𐑤 𐑕𐑑𐑦𐑤 𐑛𐑵 𐑦𐑑; 𐑯𐑪𐑑𐑢𐑦𐑞𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑞 𐑓𐑨𐑒𐑑 𐑞𐑨𐑑 𐑞 𐑶𐑤 𐑩𐑚𐑑𐑱𐑯𐑛 𐑓𐑮𐑪𐑥 𐑕𐑳𐑗 𐑕𐑳𐑚𐑡𐑧𐑒𐑑𐑕 𐑦𐑟 𐑝 𐑩 𐑝𐑧𐑮𐑦 𐑦𐑯𐑓𐑽𐑽 𐑒𐑢𐑪𐑤𐑦𐑑𐑦, 𐑯 𐑚𐑲 𐑯𐑴 𐑥𐑰𐑯𐑟 𐑝 𐑞 𐑯𐑱𐑗𐑼 𐑝 𐑨𐑑𐑼-𐑝-𐑮𐑴𐑟.

𐑒𐑳𐑥𐑦𐑙 𐑕𐑑𐑦𐑤 𐑯𐑽𐑼 𐑢𐑦𐑞 𐑞 𐑦𐑒𐑕𐑐𐑲𐑼𐑦𐑙 𐑚𐑮𐑰𐑟, 𐑢𐑰 𐑕𐑷 𐑞𐑨𐑑 𐑞 ·𐑓𐑮𐑧𐑯𐑗𐑥𐑩𐑯 𐑣𐑨𐑛 𐑩 𐑕𐑧𐑒𐑩𐑯𐑛 𐑢𐑱𐑤 𐑩𐑤𐑪𐑙𐑕𐑲𐑛; 𐑯 𐑞𐑦𐑕 𐑕𐑧𐑒𐑩𐑯𐑛 𐑢𐑱𐑤 𐑕𐑰𐑥𐑛 𐑰𐑝𐑩𐑯 𐑥𐑹 𐑝 𐑩 𐑯𐑴𐑟𐑜𐑱 𐑞𐑨𐑯 𐑞 𐑓𐑻𐑕𐑑. 𐑦𐑯 𐑑𐑮𐑵𐑔, 𐑦𐑑 𐑑𐑻𐑯𐑛 𐑬𐑑 𐑑 𐑚𐑰 𐑢𐑳𐑯 𐑝 𐑞𐑴𐑟 𐑐𐑮𐑪𐑚𐑤𐑩𐑥𐑨𐑑𐑦𐑒𐑩𐑤 𐑢𐑱𐑤𐑟 𐑞𐑨𐑑 𐑕𐑰𐑥 𐑑 𐑛𐑮𐑲 𐑳𐑐 𐑯 𐑛𐑲 𐑢𐑦𐑞 𐑩 𐑕𐑹𐑑 𐑝 𐑐𐑮𐑩𐑛𐑦𐑡𐑩𐑕 𐑛𐑦𐑕𐑐𐑧𐑐𐑕𐑾, 𐑹 𐑦𐑯𐑛𐑦𐑡𐑧𐑕𐑗𐑩𐑯; 𐑤𐑰𐑝𐑦𐑙 𐑞𐑺 𐑛𐑦𐑓𐑳𐑙𐑒𐑑 𐑚𐑪𐑛𐑦𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑯𐑑𐑲𐑼𐑤𐑦 𐑚𐑨𐑙𐑒𐑮𐑳𐑐𐑑 𐑝 𐑧𐑯𐑦𐑔𐑦𐑙 𐑤𐑲𐑒 𐑶𐑤. 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕, 𐑦𐑯 𐑞 𐑐𐑮𐑪𐑐𐑼 𐑐𐑤𐑱𐑕 𐑢𐑰 𐑖𐑨𐑤 𐑕𐑰 𐑞𐑨𐑑 𐑯𐑴 𐑯𐑴𐑦𐑙 𐑓𐑦𐑖𐑼𐑥𐑩𐑯 𐑢𐑦𐑤 𐑧𐑝𐑼 𐑑𐑻𐑯 𐑳𐑐 𐑣𐑦𐑟 𐑯𐑴𐑟 𐑨𐑑 𐑕𐑳𐑗 𐑩 𐑢𐑱𐑤 𐑨𐑟 𐑞𐑦𐑕, 𐑣𐑬𐑧𐑝𐑼 𐑥𐑳𐑗 𐑣𐑰 𐑥𐑱 𐑖𐑳𐑯 𐑚𐑤𐑭𐑕𐑑𐑩𐑛 𐑢𐑱𐑤𐑟 𐑦𐑯 𐑡𐑧𐑯𐑼𐑩𐑤.

𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑣𐑨𐑛 𐑯𐑬 𐑕𐑢𐑧𐑐𐑑 𐑕𐑴 𐑯𐑲 𐑑 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, 𐑞𐑨𐑑 ·𐑕𐑑𐑳𐑚 𐑝𐑬𐑛 𐑣𐑰 𐑮𐑧𐑒𐑩𐑜𐑯𐑲𐑟𐑛 𐑣𐑦𐑟 𐑒𐑳𐑑𐑦𐑙 𐑕𐑐𐑱𐑛-𐑐𐑴𐑤 𐑦𐑯𐑑𐑨𐑙𐑜𐑩𐑤𐑛 𐑦𐑯 𐑞 𐑤𐑲𐑯𐑟 𐑞𐑨𐑑 𐑢𐑻 𐑯𐑪𐑑𐑩𐑛 𐑮𐑬𐑯𐑛 𐑞 𐑑𐑱𐑤 𐑝 𐑢𐑳𐑯 𐑝 𐑞𐑰𐑟 𐑢𐑱𐑤𐑟.

“𐑞𐑺'𐑟 𐑩 𐑐𐑮𐑦𐑑𐑦 𐑓𐑧𐑤𐑴, 𐑯𐑬,” 𐑣𐑰 𐑚𐑨𐑯𐑑𐑼𐑦𐑙𐑤𐑦 𐑤𐑭𐑓𐑑, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑞 𐑖𐑦𐑐'𐑕 𐑚𐑬𐑟, “𐑞𐑺'𐑟 𐑩 𐑡𐑨𐑒𐑩𐑤 𐑓 𐑘𐑰! 𐑲 𐑢𐑧𐑤 𐑯𐑴 𐑞𐑨𐑑 𐑞𐑰𐑟 ·𐑒𐑮𐑨𐑐𐑴𐑟 𐑝 ·𐑓𐑮𐑧𐑯𐑗𐑥𐑧𐑯 𐑸 𐑚𐑳𐑑 𐑐𐑫𐑼 𐑛𐑧𐑝𐑩𐑤𐑟 𐑦𐑯 𐑞 𐑓𐑦𐑖𐑼𐑦; 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑤𐑴𐑼𐑦𐑙 𐑞𐑺 𐑚𐑴𐑑𐑕 𐑓 𐑚𐑮𐑱𐑒𐑼𐑟, 𐑥𐑦𐑕𐑑𐑱𐑒𐑦𐑙 𐑞𐑧𐑥 𐑓 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑕𐑐𐑬𐑑𐑕; 𐑘𐑧𐑕, 𐑯 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑕𐑱𐑤𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞𐑺 𐑐𐑹𐑑 𐑢𐑦𐑞 𐑞𐑺 𐑣𐑴𐑤𐑛 𐑓𐑫𐑤 𐑝 𐑚𐑪𐑒𐑕𐑩𐑟 𐑝 𐑑𐑨𐑤𐑴 𐑒𐑨𐑯𐑛𐑩𐑤𐑟, 𐑯 𐑒𐑱𐑕𐑩𐑟 𐑝 𐑕𐑯𐑳𐑓𐑼𐑟, 𐑓𐑹𐑕𐑰𐑦𐑙 𐑞𐑨𐑑 𐑷𐑤 𐑞 𐑶𐑤 𐑞𐑱 𐑢𐑦𐑤 𐑜𐑧𐑑 𐑢𐑴𐑯𐑑 𐑚𐑰 𐑦𐑯𐑳𐑓 𐑑 𐑛𐑦𐑐 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑢𐑦𐑒 𐑦𐑯𐑑𐑵; 𐑲, 𐑢𐑰 𐑷𐑤 𐑯𐑴 𐑞𐑰𐑟 𐑔𐑦𐑙𐑟; 𐑚𐑳𐑑 𐑤𐑫𐑒 𐑘𐑰, 𐑣𐑽'𐑟 𐑩 ·𐑒𐑮𐑨𐑐𐑴 𐑞𐑨𐑑 𐑦𐑟 𐑒𐑩𐑯𐑑𐑧𐑯𐑑 𐑢𐑦𐑞 𐑬𐑼 𐑤𐑰𐑝𐑦𐑙𐑟, 𐑞 𐑛𐑮𐑳𐑜𐑛 𐑢𐑱𐑤 𐑞𐑺, 𐑲 𐑥𐑰𐑯; 𐑲, 𐑯 𐑦𐑟 𐑒𐑩𐑯𐑑𐑧𐑯𐑑 𐑑𐑵 𐑢𐑦𐑞 𐑕𐑒𐑮𐑱𐑐𐑦𐑙 𐑞 𐑛𐑮𐑲 𐑚𐑴𐑯𐑟 𐑝 𐑞𐑨𐑑 𐑳𐑞𐑼 𐑐𐑮𐑧𐑖𐑩𐑕 𐑓𐑦𐑖 𐑣𐑰 𐑣𐑨𐑟 𐑞𐑺. 𐑐𐑫𐑼 𐑛𐑧𐑝𐑩𐑤! 𐑲 𐑕𐑱, 𐑐𐑭𐑕 𐑮𐑬𐑯𐑛 𐑩 𐑣𐑨𐑑, 𐑕𐑳𐑥 𐑢𐑳𐑯, 𐑯 𐑤𐑧𐑑'𐑕 𐑥𐑱𐑒 𐑣𐑦𐑥 𐑩 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑝 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑶𐑤 𐑓 𐑛𐑽 𐑗𐑨𐑮𐑦𐑑𐑦'𐑟 𐑕𐑱𐑒. 𐑓 𐑢𐑪𐑑 𐑶𐑤 𐑣𐑰'𐑤 𐑜𐑧𐑑 𐑓𐑮𐑪𐑥 𐑞𐑨𐑑 𐑛𐑮𐑳𐑜𐑛 𐑢𐑱𐑤 𐑞𐑺, 𐑢𐑫𐑛𐑩𐑯𐑑 𐑚𐑰 𐑓𐑦𐑑 𐑑 𐑚𐑻𐑯 𐑦𐑯 𐑩 𐑡𐑱𐑤; 𐑯𐑴, 𐑯𐑪𐑑 𐑦𐑯 𐑩 𐑒𐑩𐑯𐑛𐑧𐑥𐑛 𐑕𐑧𐑤. 𐑯 𐑨𐑟 𐑓 𐑞 𐑳𐑞𐑼 𐑢𐑱𐑤, 𐑢𐑲, 𐑲'𐑤 𐑩𐑜𐑮𐑰 𐑑 𐑜𐑧𐑑 𐑥𐑹 𐑶𐑤 𐑚𐑲 𐑗𐑪𐑐𐑦𐑙 𐑳𐑐 𐑯 𐑑𐑮𐑲𐑦𐑙 𐑬𐑑 𐑞𐑰𐑟 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑𐑕 𐑝 𐑬𐑼𐑟, 𐑞𐑨𐑯 𐑣𐑰'𐑤 𐑜𐑧𐑑 𐑓𐑮𐑪𐑥 𐑞𐑨𐑑 𐑚𐑳𐑯𐑛𐑩𐑤 𐑝 𐑚𐑴𐑯𐑟; 𐑞𐑴, 𐑯𐑬 𐑞𐑨𐑑 𐑲 𐑔𐑦𐑙𐑒 𐑝 𐑦𐑑, 𐑦𐑑 𐑥𐑱 𐑒𐑩𐑯𐑑𐑱𐑯 𐑕𐑳𐑥𐑔𐑦𐑙 𐑢𐑻𐑔 𐑩 𐑜𐑫𐑛 𐑛𐑰𐑤 𐑥𐑹 𐑞𐑨𐑯 𐑶𐑤; 𐑘𐑧𐑕, 𐑨𐑥𐑚𐑼𐑜𐑮𐑰𐑕. 𐑲 𐑢𐑳𐑯𐑛𐑼 𐑯𐑬 𐑦𐑓 𐑬𐑼 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑣𐑨𐑟 𐑔𐑷𐑑 𐑝 𐑞𐑨𐑑. 𐑦𐑑'𐑕 𐑢𐑻𐑔 𐑑𐑮𐑲𐑦𐑙. 𐑘𐑧𐑕, 𐑲'𐑥 𐑓 𐑦𐑑;” 𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙 𐑣𐑰 𐑕𐑑𐑸𐑑𐑩𐑛 𐑓 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒.

𐑚𐑲 𐑞𐑦𐑕 𐑑𐑲𐑥 𐑞 𐑓𐑱𐑯𐑑 𐑺 𐑣𐑨𐑛 𐑚𐑦𐑒𐑳𐑥 𐑩 𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑒𐑭𐑥; 𐑕𐑴 𐑞𐑨𐑑 𐑢𐑧𐑞𐑼 𐑹 𐑯𐑴, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑯𐑬 𐑓𐑺𐑤𐑦 𐑦𐑯𐑑𐑮𐑨𐑐𐑑 𐑦𐑯 𐑞 𐑕𐑥𐑧𐑤, 𐑢𐑦𐑞 𐑯𐑴 𐑣𐑴𐑐 𐑝 𐑦𐑕𐑒𐑱𐑐𐑦𐑙 𐑦𐑒𐑕𐑧𐑐𐑑 𐑚𐑲 𐑦𐑑𐑕 𐑚𐑮𐑰𐑟𐑦𐑙 𐑳𐑐 𐑩𐑜𐑱𐑯. 𐑦𐑖𐑿𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑨𐑚𐑦𐑯, ·𐑕𐑑𐑳𐑚 𐑯𐑬 𐑒𐑷𐑤𐑛 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵, 𐑯 𐑐𐑫𐑤𐑛 𐑪𐑓 𐑓 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼. 𐑛𐑮𐑷𐑦𐑙 𐑩𐑒𐑮𐑪𐑕 𐑣𐑻 𐑚𐑬, 𐑣𐑰 𐑐𐑼𐑕𐑰𐑝𐑛 𐑞𐑨𐑑 𐑦𐑯 𐑩𐑒𐑹𐑛𐑩𐑯𐑕 𐑢𐑦𐑞 𐑞 𐑓𐑨𐑯𐑕𐑦𐑓𐑩𐑤 ·𐑓𐑮𐑧𐑯𐑗 𐑑𐑱𐑕𐑑, 𐑞 𐑳𐑐𐑼 𐑐𐑸𐑑 𐑝 𐑣𐑻 𐑕𐑑𐑧𐑥-𐑐𐑰𐑕 𐑢𐑪𐑟 𐑒𐑸𐑝𐑛 𐑦𐑯 𐑞 𐑤𐑲𐑒𐑯𐑩𐑕 𐑝 𐑩 𐑣𐑿𐑡 𐑛𐑮𐑵𐑐𐑦𐑙 𐑕𐑑𐑷𐑒, 𐑢𐑪𐑟 𐑐𐑱𐑯𐑑𐑩𐑛 𐑜𐑮𐑰𐑯, 𐑯 𐑓 𐑔𐑹𐑯𐑟 𐑣𐑨𐑛 𐑒𐑪𐑐𐑼 𐑕𐑐𐑲𐑒𐑕 𐑐𐑮𐑩𐑡𐑧𐑒𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑦𐑑 𐑣𐑽 𐑯 𐑞𐑺; 𐑞 𐑣𐑴𐑤 𐑑𐑻𐑥𐑦𐑯𐑱𐑑𐑦𐑙 𐑦𐑯 𐑩 𐑕𐑦𐑥𐑧𐑑𐑮𐑦𐑒𐑩𐑤 𐑓𐑴𐑤𐑛𐑩𐑛 𐑚𐑳𐑤𐑚 𐑝 𐑩 𐑚𐑮𐑲𐑑 𐑮𐑧𐑛 𐑒𐑳𐑤𐑼. 𐑩𐑐𐑪𐑯 𐑣𐑻 𐑣𐑧𐑛 𐑚𐑹𐑛𐑟, 𐑦𐑯 𐑤𐑸𐑡 𐑜𐑦𐑤𐑑 𐑤𐑧𐑑𐑼𐑟, 𐑣𐑰 𐑮𐑧𐑛 “𐑚𐑬𐑑𐑩𐑯 𐑛𐑦 ·𐑮𐑴𐑟,”—𐑮𐑴𐑟-𐑚𐑳𐑑𐑩𐑯, 𐑹 ·𐑮𐑴𐑟-𐑚𐑳𐑛; 𐑯 𐑞𐑦𐑕 𐑢𐑪𐑟 𐑞 𐑮𐑴𐑥𐑨𐑯𐑑𐑦𐑒 𐑯𐑱𐑥 𐑝 𐑞𐑦𐑕 𐑨𐑮𐑩𐑥𐑨𐑑𐑦𐑒 𐑖𐑦𐑐.

𐑞𐑴 ·𐑕𐑑𐑳𐑚 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑞 ·𐑚𐑬𐑑𐑩𐑯 𐑐𐑸𐑑 𐑝 𐑞 𐑦𐑯𐑕𐑒𐑮𐑦𐑐𐑖𐑩𐑯, 𐑘𐑧𐑑 𐑞 𐑢𐑻𐑛 𐑮𐑴𐑟, 𐑯 𐑞 𐑚𐑳𐑤𐑚𐑩𐑕 𐑓𐑦𐑜𐑼-𐑣𐑧𐑛 𐑐𐑫𐑑 𐑑𐑩𐑜𐑧𐑞𐑼, 𐑕𐑩𐑓𐑦𐑖𐑩𐑯𐑑𐑤𐑦 𐑦𐑒𐑕𐑐𐑤𐑱𐑯𐑛 𐑞 𐑣𐑴𐑤 𐑑 𐑣𐑦𐑥.

“𐑩 𐑢𐑫𐑛𐑩𐑯 𐑮𐑴𐑟-𐑚𐑳𐑛, 𐑱?” 𐑣𐑰 𐑒𐑮𐑲𐑛 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑑 𐑣𐑦𐑟 𐑯𐑴𐑟, “𐑞𐑨𐑑 𐑢𐑦𐑤 𐑛𐑵 𐑝𐑧𐑮𐑦 𐑢𐑧𐑤; 𐑚𐑳𐑑 𐑣𐑬 𐑤𐑲𐑒 𐑷𐑤 𐑒𐑮𐑦𐑱𐑖𐑩𐑯 𐑦𐑑 𐑕𐑥𐑧𐑤𐑟!”

𐑯𐑬 𐑦𐑯 𐑹𐑛𐑼 𐑑 𐑣𐑴𐑤𐑛 𐑛𐑦𐑮𐑧𐑒𐑑 𐑒𐑩𐑥𐑿𐑯𐑦𐑒𐑱𐑖𐑩𐑯 𐑢𐑦𐑞 𐑞 𐑐𐑰𐑐𐑩𐑤 𐑪𐑯 𐑛𐑧𐑒, 𐑣𐑰 𐑣𐑨𐑛 𐑑 𐑐𐑫𐑤 𐑮𐑬𐑯𐑛 𐑞 𐑚𐑬𐑟 𐑑 𐑞 𐑕𐑑𐑸𐑚𐑼𐑛 𐑕𐑲𐑛, 𐑯 𐑞𐑳𐑕 𐑒𐑳𐑥 𐑒𐑤𐑴𐑕 𐑑 𐑞 𐑚𐑤𐑭𐑕𐑑𐑩𐑛 𐑢𐑱𐑤; 𐑯 𐑕𐑴 𐑑𐑷𐑒 𐑴𐑝𐑼 𐑦𐑑.

𐑼𐑲𐑝𐑛 𐑞𐑧𐑯 𐑨𐑑 𐑞𐑦𐑕 𐑕𐑐𐑪𐑑, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑕𐑑𐑦𐑤 𐑑 𐑣𐑦𐑟 𐑯𐑴𐑟, 𐑣𐑰 𐑚𐑷𐑤𐑛—“𐑚𐑬𐑑𐑩𐑯-𐑛𐑦-·𐑮𐑴𐑟, 𐑩𐑣𐑶! 𐑸 𐑞𐑺 𐑧𐑯𐑦 𐑝 𐑿 ·𐑚𐑬𐑑𐑩𐑯-𐑛𐑦-·𐑮𐑴𐑟𐑩𐑟 𐑞𐑨𐑑 𐑕𐑐𐑰𐑒 ·𐑦𐑙𐑜𐑤𐑦𐑖?”

“𐑘𐑧𐑕,” 𐑮𐑰𐑡𐑶𐑯𐑛 𐑩 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑣𐑵 𐑑𐑻𐑯𐑛 𐑬𐑑 𐑑 𐑚𐑰 𐑞 𐑗𐑰𐑓-𐑥𐑱𐑑.

“𐑢𐑧𐑤, 𐑞𐑧𐑯, 𐑥𐑲 ·𐑚𐑬𐑑𐑩𐑯-𐑛𐑦-·𐑮𐑴𐑟-𐑚𐑳𐑛, 𐑣𐑨𐑝 𐑿 𐑕𐑰𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?”

“𐑢𐑪𐑑 𐑢𐑱𐑤?”

“𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤—𐑩 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤—·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒, 𐑣𐑨𐑝 𐑘𐑰 𐑕𐑰𐑯 𐑣𐑦𐑥?

“𐑯𐑧𐑝𐑼 𐑣𐑻𐑛 𐑝 𐑕𐑳𐑗 𐑩 𐑢𐑱𐑤. 𐑒𐑨𐑖𐑩𐑤𐑪𐑑 ·𐑚𐑤𐑭𐑯𐑖! 𐑢𐑲𐑑 ·𐑢𐑱𐑤—𐑯𐑴.”

“𐑝𐑧𐑮𐑦 𐑜𐑫𐑛, 𐑞𐑧𐑯; 𐑜𐑫𐑛 𐑚𐑲 𐑯𐑬, 𐑯 𐑲'𐑤 𐑒𐑷𐑤 𐑩𐑜𐑱𐑯 𐑦𐑯 𐑩 𐑥𐑦𐑯𐑦𐑑.”

𐑞𐑧𐑯 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑐𐑫𐑤𐑦𐑙 𐑚𐑨𐑒 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑯 𐑕𐑰𐑦𐑙 ·𐑱𐑣𐑨𐑚 𐑤𐑰𐑯𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒 𐑮𐑱𐑤 𐑩𐑢𐑱𐑑𐑦𐑙 𐑣𐑦𐑟 𐑮𐑦𐑐𐑹𐑑, 𐑣𐑰 𐑥𐑴𐑤𐑛𐑩𐑛 𐑣𐑦𐑟 𐑑𐑵 𐑣𐑨𐑯𐑛𐑟 𐑦𐑯𐑑𐑵 𐑩 𐑑𐑮𐑳𐑥𐑐𐑩𐑑 𐑯 𐑖𐑬𐑑𐑩𐑛—“𐑯𐑴, ·𐑕𐑻! 𐑯𐑴!” 𐑩𐑐𐑪𐑯 𐑢𐑦𐑗 ·𐑱𐑣𐑨𐑚 𐑮𐑦𐑑𐑲𐑼𐑛, 𐑯 ·𐑕𐑑𐑳𐑚 𐑮𐑩𐑑𐑻𐑯𐑛 𐑑 𐑞 ·𐑓𐑮𐑧𐑯𐑗𐑥𐑩𐑯.

𐑣𐑰 𐑯𐑬 𐑐𐑼𐑕𐑰𐑝𐑛 𐑞𐑨𐑑 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯, 𐑣𐑵 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑜𐑪𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑗𐑱𐑯𐑟, 𐑯 𐑢𐑪𐑟 𐑿𐑟𐑦𐑙 𐑩 𐑒𐑳𐑑𐑦𐑙-𐑕𐑐𐑱𐑛, 𐑣𐑨𐑛 𐑕𐑤𐑳𐑙 𐑣𐑦𐑟 𐑯𐑴𐑟 𐑦𐑯 𐑩 𐑕𐑹𐑑 𐑝 𐑚𐑨𐑜.

“𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑘𐑹 𐑯𐑴𐑟, 𐑞𐑺?” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚. “𐑚𐑮𐑴𐑒 𐑦𐑑?”

“𐑲 𐑢𐑦𐑖 𐑦𐑑 𐑢𐑪𐑟 𐑚𐑮𐑴𐑒𐑩𐑯, 𐑹 𐑞𐑨𐑑 𐑲 𐑛𐑦𐑛𐑩𐑯𐑑 𐑣𐑨𐑝 𐑧𐑯𐑦 𐑯𐑴𐑟 𐑨𐑑 𐑷𐑤!” 𐑭𐑯𐑕𐑼𐑛 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯, 𐑣𐑵 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑕𐑰𐑥 𐑑 𐑮𐑧𐑤𐑦𐑖 𐑞 𐑡𐑪𐑚 𐑣𐑰 𐑢𐑪𐑟 𐑨𐑑 𐑝𐑧𐑮𐑦 𐑥𐑳𐑗. “𐑚𐑳𐑑 𐑢𐑪𐑑 𐑸 𐑿 𐑣𐑴𐑤𐑛𐑦𐑙 𐑘𐑹𐑟 𐑓?”

“𐑴, 𐑯𐑳𐑔𐑦𐑙! 𐑦𐑑'𐑕 𐑩 𐑢𐑨𐑒𐑕 𐑯𐑴𐑟; 𐑲 𐑣𐑨𐑓 𐑑 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑪𐑯. 𐑓𐑲𐑯 𐑛𐑱, 𐑱𐑯𐑑 𐑦𐑑? 𐑺 𐑮𐑭𐑞𐑼 𐑜𐑸𐑛𐑩𐑯𐑦, 𐑲 𐑖𐑵𐑛 𐑕𐑱; 𐑔𐑮𐑴 𐑳𐑕 𐑩 𐑚𐑳𐑯𐑗 𐑝 𐑐𐑴𐑟𐑦𐑟, 𐑢𐑦𐑤 𐑘𐑰, ·𐑚𐑬𐑑𐑩𐑯-𐑛𐑦-·𐑮𐑴𐑟?”

“𐑢𐑪𐑑 𐑦𐑯 𐑞 𐑛𐑧𐑝𐑩𐑤'𐑟 𐑯𐑱𐑥 𐑛𐑵 𐑿 𐑢𐑪𐑯𐑑 𐑣𐑽?” 𐑮𐑹𐑛 𐑞 ·𐑜𐑻𐑯𐑟𐑦𐑥𐑩𐑯, 𐑓𐑤𐑲𐑦𐑙 𐑦𐑯𐑑𐑵 𐑩 𐑕𐑳𐑛𐑩𐑯 𐑐𐑭𐑖𐑩𐑯.

“𐑴! 𐑒𐑰𐑐 𐑒𐑵𐑤—𐑒𐑵𐑤? 𐑘𐑧𐑕, 𐑞𐑨𐑑'𐑕 𐑞 𐑢𐑻𐑛! 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑿 𐑐𐑨𐑒 𐑞𐑴𐑟 𐑢𐑱𐑤𐑟 𐑦𐑯 𐑲𐑕 𐑢𐑲𐑤 𐑿'𐑼 𐑢𐑻𐑒𐑦𐑙 𐑨𐑑 '𐑩𐑥? 𐑚𐑳𐑑 𐑡𐑴𐑒𐑦𐑙 𐑩𐑕𐑲𐑛, 𐑞𐑴; 𐑛𐑵 𐑿 𐑯𐑴, ·𐑮𐑴𐑟-𐑚𐑳𐑛, 𐑞𐑨𐑑 𐑦𐑑'𐑕 𐑷𐑤 𐑯𐑪𐑯𐑕𐑧𐑯𐑕 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑜𐑧𐑑 𐑧𐑯𐑦 𐑶𐑤 𐑬𐑑 𐑝 𐑕𐑳𐑗 𐑢𐑱𐑤𐑟? 𐑨𐑟 𐑓 𐑞𐑨𐑑 𐑛𐑮𐑲𐑛 𐑳𐑐 𐑢𐑳𐑯, 𐑞𐑺, 𐑣𐑰 𐑣𐑨𐑟𐑩𐑯𐑑 𐑩 𐑜𐑦𐑤 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑴𐑤 𐑒𐑸𐑒𐑩𐑕.”

“𐑲 𐑯𐑴 𐑞𐑨𐑑 𐑢𐑧𐑤 𐑦𐑯𐑳𐑓; 𐑚𐑳𐑑, 𐑛'𐑘𐑰 𐑕𐑰, 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑣𐑽 𐑢𐑴𐑯𐑑 𐑚𐑦𐑤𐑰𐑝 𐑦𐑑; 𐑞𐑦𐑕 𐑦𐑟 𐑣𐑦𐑟 𐑓𐑻𐑕𐑑 𐑝𐑶𐑦𐑡; 𐑣𐑰 𐑢𐑪𐑟 𐑩 ·𐑒𐑩𐑤𐑴𐑯 𐑥𐑨𐑯𐑿𐑓𐑨𐑒𐑗𐑼𐑼 𐑚𐑦𐑓𐑹. 𐑚𐑳𐑑 𐑒𐑳𐑥 𐑩𐑚𐑹𐑛, 𐑯 𐑥𐑱𐑣𐑨𐑐 𐑣𐑰'𐑤 𐑚𐑦𐑤𐑰𐑝 𐑿, 𐑦𐑓 𐑣𐑰 𐑢𐑴𐑯𐑑 𐑥𐑰; 𐑯 𐑕𐑴 𐑲'𐑤 𐑜𐑧𐑑 𐑬𐑑 𐑝 𐑞𐑦𐑕 𐑛𐑻𐑑𐑦 𐑕𐑒𐑮𐑱𐑐.”

𐑚𐑲 𐑞𐑦𐑕 𐑑𐑲𐑥 𐑞𐑺 𐑛𐑧𐑕𐑑𐑦𐑯𐑛 𐑝𐑦𐑒𐑑𐑦𐑥 𐑩𐑐𐑽𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯. 𐑣𐑰 𐑢𐑪𐑟 𐑩 𐑕𐑥𐑷𐑤 𐑯 𐑛𐑸𐑒, 𐑚𐑳𐑑 𐑮𐑭𐑞𐑼 𐑛𐑧𐑤𐑦𐑒𐑩𐑑 𐑤𐑫𐑒𐑦𐑙 𐑥𐑨𐑯 𐑓 𐑩 𐑕𐑰-𐑒𐑨𐑐𐑑𐑦𐑯, 𐑢𐑦𐑞 𐑤𐑸𐑡 𐑢𐑦𐑕𐑒𐑼𐑟 𐑯 𐑥𐑳𐑕𐑑𐑭𐑖, 𐑣𐑬𐑧𐑝𐑼; 𐑯 𐑢𐑹 𐑩 𐑮𐑧𐑛 𐑒𐑪𐑑𐑩𐑯 𐑝𐑧𐑤𐑝𐑩𐑑 𐑝𐑧𐑕𐑑 𐑢𐑦𐑞 𐑢𐑪𐑗-𐑕𐑰𐑤𐑟 𐑨𐑑 𐑣𐑦𐑟 𐑕𐑲𐑛. 𐑑 𐑞𐑦𐑕 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯, ·𐑕𐑑𐑳𐑚 𐑢𐑪𐑟 𐑯𐑬 𐑐𐑩𐑤𐑲𐑑𐑤𐑦 𐑦𐑯𐑑𐑮𐑩𐑛𐑿𐑕𐑑 𐑚𐑲 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯, 𐑣𐑵 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑪𐑕𐑑𐑧𐑯𐑑𐑱𐑖𐑩𐑕𐑤𐑦 𐑐𐑫𐑑 𐑪𐑯 𐑞 𐑨𐑕𐑐𐑧𐑒𐑑 𐑝 𐑦𐑯𐑑𐑻𐑐𐑮𐑩𐑑𐑦𐑙 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞𐑧𐑥.

“𐑢𐑪𐑑 𐑖𐑨𐑤 𐑲 𐑕𐑱 𐑑 𐑣𐑦𐑥 𐑓𐑻𐑕𐑑?” 𐑕𐑧𐑛 𐑣𐑰.

“𐑢𐑲,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑲𐑦𐑙 𐑞 𐑝𐑧𐑤𐑝𐑩𐑑 𐑝𐑧𐑕𐑑 𐑯 𐑞 𐑢𐑪𐑗 𐑯 𐑕𐑰𐑤𐑟, “𐑿 𐑥𐑱 𐑨𐑟 𐑢𐑧𐑤 𐑚𐑦𐑜𐑦𐑯 𐑚𐑲 𐑑𐑧𐑤𐑦𐑙 𐑣𐑦𐑥 𐑞𐑨𐑑 𐑣𐑰 𐑤𐑫𐑒𐑕 𐑩 𐑕𐑹𐑑 𐑝 𐑚𐑱𐑚𐑦𐑦𐑖 𐑑 𐑥𐑰, 𐑞𐑴 𐑲 𐑛𐑴𐑯𐑑 𐑐𐑮𐑦𐑑𐑧𐑯𐑛 𐑑 𐑚𐑰 𐑩 𐑡𐑳𐑡.”

“𐑣𐑰 𐑕𐑧𐑟, ·𐑥𐑩𐑕𐑘𐑻,” 𐑕𐑧𐑛 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯, 𐑦𐑯 ·𐑓𐑮𐑧𐑯𐑗, 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑣𐑦𐑟 𐑒𐑨𐑐𐑑𐑦𐑯, “𐑞𐑨𐑑 𐑴𐑯𐑤𐑦 𐑘𐑧𐑕𐑑𐑼𐑛𐑱 𐑣𐑦𐑟 𐑖𐑦𐑐 𐑕𐑐𐑴𐑒 𐑩 𐑝𐑧𐑕𐑩𐑤, 𐑣𐑵𐑟 𐑒𐑨𐑐𐑑𐑦𐑯 𐑯 𐑗𐑰𐑓-𐑥𐑱𐑑, 𐑢𐑦𐑞 𐑕𐑦𐑒𐑕 𐑕𐑱𐑤𐑼𐑟, 𐑣𐑨𐑛 𐑷𐑤 𐑛𐑲𐑛 𐑝 𐑩 𐑓𐑰𐑝𐑼 𐑒𐑷𐑑 𐑓𐑮𐑪𐑥 𐑩 𐑚𐑤𐑭𐑕𐑑𐑩𐑛 𐑢𐑱𐑤 𐑞𐑱 𐑣𐑨𐑛 𐑚𐑮𐑷𐑑 𐑩𐑤𐑪𐑙𐑕𐑲𐑛.”

𐑩𐑐𐑪𐑯 𐑞𐑦𐑕 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑕𐑑𐑸𐑑𐑩𐑛, 𐑯 𐑰𐑜𐑼𐑤𐑦 𐑛𐑦𐑟𐑲𐑼𐑛 𐑑 𐑯𐑴 𐑥𐑹.

“𐑢𐑪𐑑 𐑯𐑬?” 𐑕𐑧𐑛 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯 𐑑 ·𐑕𐑑𐑳𐑚.

“𐑢𐑲, 𐑕𐑦𐑯𐑕 𐑣𐑰 𐑑𐑱𐑒𐑕 𐑦𐑑 𐑕𐑴 𐑰𐑟𐑦, 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑞𐑨𐑑 𐑯𐑬 𐑲 𐑣𐑨𐑝 𐑲𐑛 𐑣𐑦𐑥 𐑒𐑺𐑓𐑩𐑤𐑦, 𐑲'𐑥 𐑒𐑢𐑲𐑑 𐑕𐑻𐑑𐑩𐑯 𐑞𐑨𐑑 𐑣𐑰'𐑟 𐑯𐑴 𐑥𐑹 𐑓𐑦𐑑 𐑑 𐑒𐑩𐑥𐑭𐑯𐑛 𐑩 𐑢𐑱𐑤-𐑖𐑦𐑐 𐑞𐑨𐑯 𐑩 ·𐑕𐑑 ·𐑡𐑨𐑜𐑴 𐑥𐑳𐑙𐑒𐑦. 𐑦𐑯 𐑓𐑨𐑒𐑑, 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑓𐑮𐑪𐑥 𐑥𐑰 𐑣𐑰'𐑟 𐑩 𐑚𐑩𐑚𐑵𐑯.”

“𐑣𐑰 𐑝𐑬𐑟 𐑯 𐑛𐑦𐑒𐑤𐑺𐑟, ·𐑥𐑩𐑕𐑘𐑻, 𐑞𐑨𐑑 𐑞 𐑳𐑞𐑼 𐑢𐑱𐑤, 𐑞 𐑛𐑮𐑲𐑛 𐑢𐑳𐑯, 𐑦𐑟 𐑓𐑸 𐑥𐑹 𐑛𐑧𐑛𐑤𐑦 𐑞𐑨𐑯 𐑞 𐑚𐑤𐑭𐑕𐑑𐑩𐑛 𐑢𐑳𐑯; 𐑦𐑯 𐑓𐑲𐑯, ·𐑥𐑩𐑕𐑘𐑻, 𐑣𐑰 𐑒𐑳𐑯𐑡𐑼𐑟 𐑳𐑕, 𐑨𐑟 𐑢𐑰 𐑝𐑨𐑤𐑿 𐑬𐑼 𐑤𐑲𐑝𐑟, 𐑑 𐑒𐑳𐑑 𐑤𐑵𐑕 𐑓𐑮𐑪𐑥 𐑞𐑰𐑟 𐑓𐑦𐑖.”

𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑮𐑨𐑯 𐑓𐑹𐑢𐑼𐑛, 𐑯 𐑦𐑯 𐑩 𐑤𐑬𐑛 𐑝𐑶𐑕 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛 𐑣𐑦𐑟 𐑒𐑮𐑵 𐑑 𐑛𐑦𐑟𐑦𐑕𐑑 𐑓𐑮𐑪𐑥 𐑣𐑶𐑕𐑑𐑦𐑙 𐑞 𐑒𐑳𐑑𐑦𐑙-𐑑𐑨𐑒𐑩𐑤𐑟, 𐑯 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑒𐑭𐑕𐑑 𐑤𐑵𐑕 𐑞 𐑒𐑱𐑚𐑩𐑤𐑟 𐑯 𐑗𐑱𐑯𐑟 𐑒𐑩𐑯𐑓𐑲𐑯𐑦𐑙 𐑞 𐑢𐑱𐑤𐑟 𐑑 𐑞 𐑖𐑦𐑐.

“𐑢𐑪𐑑 𐑯𐑬?” 𐑕𐑧𐑛 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯, 𐑢𐑧𐑯 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑣𐑨𐑛 𐑮𐑩𐑑𐑻𐑯𐑛 𐑑 𐑞𐑧𐑥.

“𐑢𐑲, 𐑤𐑧𐑑 𐑥𐑰 𐑕𐑰; 𐑘𐑧𐑕, 𐑿 𐑥𐑱 𐑨𐑟 𐑢𐑧𐑤 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑯𐑬 𐑞𐑨𐑑—𐑞𐑨𐑑—𐑦𐑯 𐑓𐑨𐑒𐑑, 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑲'𐑝 𐑛𐑦𐑛𐑩𐑤𐑛 𐑣𐑦𐑥, 𐑯 (𐑩𐑕𐑲𐑛 𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓) 𐑐𐑼𐑣𐑨𐑐𐑕 𐑕𐑳𐑥𐑚𐑪𐑛𐑦 𐑧𐑤𐑕.”

“𐑣𐑰 𐑕𐑧𐑟, ·𐑥𐑩𐑕𐑘𐑻, 𐑞𐑨𐑑 𐑣𐑰'𐑟 𐑝𐑧𐑮𐑦 𐑣𐑨𐑐𐑦 𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑝 𐑧𐑯𐑦 𐑕𐑻𐑝𐑦𐑕 𐑑 𐑳𐑕.”

𐑣𐑽𐑦𐑙 𐑞𐑦𐑕, 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑝𐑬𐑛 𐑞𐑨𐑑 𐑞𐑱 𐑢𐑻 𐑞 𐑜𐑮𐑱𐑑𐑓𐑩𐑤 𐑐𐑸𐑑𐑦𐑟 (𐑥𐑰𐑯𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑯 𐑥𐑱𐑑) 𐑯 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑚𐑲 𐑦𐑯𐑝𐑲𐑑𐑦𐑙 ·𐑕𐑑𐑳𐑚 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯 𐑑 𐑛𐑮𐑦𐑙𐑒 𐑩 𐑚𐑪𐑑𐑩𐑤 𐑝 ·𐑚𐑹𐑛𐑴.

“𐑣𐑰 𐑢𐑪𐑯𐑑𐑕 𐑿 𐑑 𐑑𐑱𐑒 𐑩 𐑜𐑤𐑭𐑕 𐑝 𐑢𐑲𐑯 𐑢𐑦𐑞 𐑣𐑦𐑥,” 𐑕𐑧𐑛 𐑞 𐑦𐑯𐑑𐑻𐑐𐑮𐑩𐑑𐑼.

“𐑔𐑨𐑙𐑒 𐑣𐑦𐑥 𐑣𐑸𐑑𐑦𐑤𐑦; 𐑚𐑳𐑑 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑦𐑑'𐑕 𐑩𐑜𐑱𐑯𐑕𐑑 𐑥𐑲 𐑐𐑮𐑦𐑯𐑕𐑩𐑐𐑩𐑤𐑟 𐑑 𐑛𐑮𐑦𐑙𐑒 𐑢𐑦𐑞 𐑞 𐑥𐑨𐑯 𐑲'𐑝 𐑛𐑦𐑛𐑩𐑤𐑛. 𐑦𐑯 𐑓𐑨𐑒𐑑, 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑲 𐑥𐑳𐑕𐑑 𐑜𐑴.”

“𐑣𐑰 𐑕𐑧𐑟, ·𐑥𐑩𐑕𐑘𐑻, 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑐𐑮𐑦𐑯𐑕𐑩𐑐𐑩𐑤𐑟 𐑢𐑴𐑯𐑑 𐑩𐑛𐑥𐑦𐑑 𐑝 𐑣𐑦𐑟 𐑛𐑮𐑦𐑙𐑒𐑦𐑙; 𐑚𐑳𐑑 𐑞𐑨𐑑 𐑦𐑓 ·𐑥𐑩𐑕𐑘𐑻 𐑢𐑪𐑯𐑑𐑕 𐑑 𐑤𐑦𐑝 𐑩𐑯𐑳𐑞𐑼 𐑛𐑱 𐑑 𐑛𐑮𐑦𐑙𐑒, 𐑞𐑧𐑯 ·𐑥𐑩𐑕𐑘𐑻 𐑣𐑨𐑛 𐑚𐑧𐑕𐑑 𐑛𐑮𐑪𐑐 𐑷𐑤 𐑓𐑹 𐑚𐑴𐑑𐑕, 𐑯 𐑐𐑫𐑤 𐑞 𐑖𐑦𐑐 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞𐑰𐑟 𐑢𐑱𐑤𐑟, 𐑓 𐑦𐑑'𐑕 𐑕𐑴 𐑒𐑭𐑥 𐑞𐑱 𐑢𐑴𐑯𐑑 𐑛𐑮𐑦𐑓𐑑.”

𐑚𐑲 𐑞𐑦𐑕 𐑑𐑲𐑥 ·𐑕𐑑𐑳𐑚 𐑢𐑪𐑟 𐑴𐑝𐑼 𐑞 𐑕𐑲𐑛, 𐑯 𐑜𐑧𐑑𐑦𐑙 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑚𐑴𐑑, 𐑣𐑱𐑤𐑛 𐑞 ·𐑜𐑻𐑯𐑟𐑦-𐑥𐑨𐑯 𐑑 𐑞𐑦𐑕 𐑦𐑓𐑧𐑒𐑑,—𐑞𐑨𐑑 𐑣𐑨𐑝𐑦𐑙 𐑩 𐑤𐑪𐑙 𐑑𐑴-𐑤𐑲𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑴𐑑, 𐑣𐑰 𐑢𐑫𐑛 𐑛𐑵 𐑢𐑪𐑑 𐑣𐑰 𐑒𐑫𐑛 𐑑 𐑣𐑧𐑤𐑐 𐑞𐑧𐑥, 𐑚𐑲 𐑐𐑫𐑤𐑦𐑙 𐑬𐑑 𐑞 𐑤𐑲𐑑𐑼 𐑢𐑱𐑤 𐑝 𐑞 𐑑𐑵 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐'𐑕 𐑕𐑲𐑛. 𐑢𐑲𐑤 𐑞 ·𐑓𐑮𐑧𐑯𐑗𐑥𐑩𐑯'𐑟 𐑚𐑴𐑑𐑕, 𐑞𐑧𐑯, 𐑢𐑻 𐑦𐑯𐑜𐑱𐑡𐑛 𐑦𐑯 𐑑𐑴𐑦𐑙 𐑞 𐑖𐑦𐑐 𐑢𐑳𐑯 𐑢𐑱, ·𐑕𐑑𐑳𐑚 𐑚𐑦𐑯𐑧𐑝𐑩𐑤𐑩𐑯𐑑𐑤𐑦 𐑑𐑴𐑛 𐑩𐑢𐑱 𐑨𐑑 𐑣𐑦𐑟 𐑢𐑱𐑤 𐑞 𐑳𐑞𐑼 𐑢𐑱, 𐑪𐑕𐑑𐑧𐑯𐑑𐑱𐑖𐑩𐑕𐑤𐑦 𐑕𐑤𐑨𐑒𐑦𐑙 𐑬𐑑 𐑩 𐑥𐑴𐑕𐑑 𐑳𐑯𐑿𐑠𐑵𐑩𐑤𐑦 𐑤𐑪𐑙 𐑑𐑴-𐑤𐑲𐑯.

𐑐𐑮𐑧𐑟𐑩𐑯𐑑𐑤𐑦 𐑩 𐑚𐑮𐑰𐑟 𐑕𐑐𐑮𐑨𐑙 𐑳𐑐; ·𐑕𐑑𐑳𐑚 𐑓𐑱𐑯𐑛 𐑑 𐑒𐑭𐑕𐑑 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑱𐑤; 𐑣𐑶𐑕𐑑𐑦𐑙 𐑣𐑦𐑟 𐑚𐑴𐑑𐑕, 𐑞 ·𐑓𐑮𐑧𐑯𐑗𐑥𐑩𐑯 𐑕𐑵𐑯 𐑦𐑯𐑒𐑮𐑰𐑕𐑑 𐑣𐑦𐑟 𐑛𐑦𐑕𐑑𐑩𐑯𐑕, 𐑢𐑲𐑤 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑕𐑤𐑦𐑛 𐑦𐑯 𐑚𐑦𐑑𐑢𐑰𐑯 𐑣𐑦𐑥 𐑯 ·𐑕𐑑𐑳𐑚'𐑟 𐑢𐑱𐑤. 𐑢𐑺𐑩𐑐𐑪𐑯 ·𐑕𐑑𐑳𐑚 𐑒𐑢𐑦𐑒𐑤𐑦 𐑐𐑫𐑤𐑛 𐑑 𐑞 𐑓𐑤𐑴𐑑𐑦𐑙 𐑚𐑪𐑛𐑦, 𐑯 𐑣𐑱𐑤𐑦𐑙 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑑 𐑜𐑦𐑝 𐑯𐑴𐑑𐑦𐑕 𐑝 𐑣𐑦𐑟 𐑦𐑯𐑑𐑧𐑯𐑖𐑩𐑯𐑟, 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑐𐑮𐑴𐑕𐑰𐑛𐑩𐑛 𐑑 𐑮𐑰𐑐 𐑞 𐑓𐑮𐑵𐑑 𐑝 𐑣𐑦𐑟 𐑳𐑯𐑮𐑲𐑗𐑩𐑕 𐑒𐑳𐑯𐑦𐑙. 𐑕𐑰𐑟𐑦𐑙 𐑣𐑦𐑟 𐑖𐑸𐑐 𐑚𐑴𐑑-𐑕𐑐𐑱𐑛, 𐑣𐑰 𐑒𐑩𐑥𐑧𐑯𐑕𐑑 𐑩𐑯 𐑧𐑒𐑕𐑒𐑩𐑝𐑱𐑖𐑩𐑯 𐑦𐑯 𐑞 𐑚𐑪𐑛𐑦, 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑚𐑦𐑣𐑲𐑯𐑛 𐑞 𐑕𐑲𐑛 𐑓𐑦𐑯. 𐑿 𐑢𐑫𐑛 𐑷𐑤𐑥𐑴𐑕𐑑 𐑣𐑨𐑝 𐑔𐑷𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑛𐑦𐑜𐑦𐑙 𐑩 𐑕𐑧𐑤𐑼 𐑞𐑺 𐑦𐑯 𐑞 𐑕𐑰; 𐑯 𐑢𐑧𐑯 𐑨𐑑 𐑤𐑧𐑙𐑔 𐑣𐑦𐑟 𐑕𐑐𐑱𐑛 𐑕𐑑𐑮𐑳𐑒 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑜𐑷𐑯𐑑 𐑮𐑦𐑚𐑟, 𐑦𐑑 𐑢𐑪𐑟 𐑤𐑲𐑒 𐑑𐑻𐑯𐑦𐑙 𐑳𐑐 𐑴𐑤𐑛 ·𐑮𐑴𐑥𐑩𐑯 𐑑𐑲𐑤𐑟 𐑯 𐑐𐑪𐑑𐑼𐑦 𐑚𐑧𐑮𐑦𐑛 𐑦𐑯 𐑓𐑨𐑑 ·𐑦𐑙𐑜𐑤𐑦𐑖 𐑤𐑴𐑥. 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵 𐑢𐑻 𐑷𐑤 𐑦𐑯 𐑣𐑲 𐑦𐑒𐑕𐑲𐑑𐑥𐑩𐑯𐑑, 𐑰𐑜𐑼𐑤𐑦 𐑣𐑧𐑤𐑐𐑦𐑙 𐑞𐑺 𐑗𐑰𐑓, 𐑯 𐑤𐑫𐑒𐑦𐑙 𐑨𐑟 𐑨𐑙𐑒𐑖𐑩𐑕 𐑨𐑟 𐑜𐑴𐑤𐑛-𐑣𐑳𐑯𐑑𐑼𐑟.

𐑯 𐑷𐑤 𐑞 𐑑𐑲𐑥 𐑯𐑳𐑥𐑚𐑼𐑤𐑩𐑕 𐑓𐑬𐑤𐑟 𐑢𐑻 𐑛𐑲𐑝𐑦𐑙, 𐑯 𐑛𐑳𐑒𐑦𐑙, 𐑯 𐑕𐑒𐑮𐑰𐑥𐑦𐑙, 𐑯 𐑘𐑧𐑤𐑦𐑙, 𐑯 𐑓𐑲𐑑𐑦𐑙 𐑼𐑬𐑯𐑛 𐑞𐑧𐑥. 𐑕𐑑𐑳𐑚 𐑢𐑪𐑟 𐑚𐑦𐑜𐑦𐑯𐑦𐑙 𐑑 𐑤𐑫𐑒 𐑛𐑦𐑕𐑩𐑐𐑶𐑯𐑑𐑩𐑛, 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑨𐑟 𐑞 𐑣𐑪𐑮𐑦𐑚𐑩𐑤 𐑯𐑴𐑟𐑜𐑱 𐑦𐑯𐑒𐑮𐑰𐑕𐑑, 𐑢𐑧𐑯 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑞 𐑝𐑧𐑮𐑦 𐑣𐑸𐑑 𐑝 𐑞𐑦𐑕 𐑐𐑤𐑱𐑜, 𐑞𐑺 𐑕𐑑𐑴𐑤 𐑩 𐑓𐑱𐑯𐑑 𐑕𐑑𐑮𐑰𐑥 𐑝 𐑐𐑻𐑓𐑿𐑥, 𐑢𐑦𐑗 𐑓𐑤𐑴𐑛 𐑔𐑮𐑵 𐑞 𐑑𐑲𐑛 𐑝 𐑚𐑨𐑛 𐑕𐑥𐑧𐑤𐑟 𐑢𐑦𐑞𐑬𐑑 𐑚𐑰𐑦𐑙 𐑩𐑚𐑕𐑹𐑚𐑛 𐑚𐑲 𐑦𐑑, 𐑨𐑟 𐑢𐑳𐑯 𐑮𐑦𐑝𐑼 𐑢𐑦𐑤 𐑓𐑤𐑴 𐑦𐑯𐑑𐑵 𐑯 𐑞𐑧𐑯 𐑩𐑤𐑪𐑙 𐑢𐑦𐑞 𐑩𐑯𐑳𐑞𐑼, 𐑢𐑦𐑞𐑬𐑑 𐑨𐑑 𐑷𐑤 𐑚𐑤𐑧𐑯𐑛𐑦𐑙 𐑢𐑦𐑞 𐑦𐑑 𐑓 𐑩 𐑑𐑲𐑥.

“𐑲 𐑣𐑨𐑝 𐑦𐑑, 𐑲 𐑣𐑨𐑝 𐑦𐑑,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, 𐑢𐑦𐑞 𐑛𐑦𐑤𐑲𐑑, 𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑕𐑳𐑥𐑔𐑦𐑙 𐑦𐑯 𐑞 𐑕𐑳𐑚𐑑𐑼𐑱𐑯𐑾𐑯 𐑮𐑰𐑡𐑩𐑯𐑟, “𐑩 𐑐𐑻𐑕! 𐑩 𐑐𐑻𐑕!”
