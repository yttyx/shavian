𐑨𐑟 𐑢𐑧𐑯 𐑞 𐑕𐑑𐑮𐑦𐑒𐑩𐑯 𐑢𐑱𐑤, 𐑞𐑨𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑳𐑚 𐑣𐑨𐑟 𐑮𐑰𐑤𐑛 𐑬𐑑 𐑣𐑳𐑯𐑛𐑮𐑩𐑛𐑟 𐑝 𐑓𐑨𐑞𐑩𐑥𐑟 𐑝 𐑮𐑴𐑐; 𐑨𐑟, 𐑭𐑓𐑑𐑼 𐑛𐑰𐑐 𐑕𐑬𐑯𐑛𐑦𐑙, 𐑣𐑰 𐑓𐑤𐑴𐑑𐑕 𐑳𐑐 𐑩𐑜𐑱𐑯, 𐑯 𐑖𐑴𐑟 𐑞 𐑕𐑤𐑨𐑒𐑩𐑯𐑛 𐑒𐑻𐑤𐑦𐑙 𐑤𐑲𐑯 𐑚𐑶𐑩𐑯𐑑𐑤𐑦 𐑮𐑲𐑟𐑦𐑙 𐑯 𐑕𐑐𐑲𐑮𐑩𐑤𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑺; 𐑕𐑴 𐑯𐑬, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑕𐑷 𐑤𐑪𐑙 𐑒𐑶𐑤𐑟 𐑝 𐑞 𐑳𐑥𐑚𐑦𐑤𐑦𐑒𐑩𐑤 𐑒𐑹𐑛 𐑝 ·𐑥𐑩𐑛𐑭𐑥 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯, 𐑚𐑲 𐑢𐑦𐑗 𐑞 𐑘𐑳𐑙 𐑒𐑳𐑚 𐑕𐑰𐑥𐑛 𐑕𐑑𐑦𐑤 𐑑𐑧𐑞𐑼𐑛 𐑑 𐑦𐑑𐑕 𐑛𐑨𐑥. 𐑯𐑪𐑑 𐑕𐑧𐑤𐑛𐑩𐑥 𐑦𐑯 𐑞 𐑮𐑨𐑐𐑦𐑛 𐑝𐑦𐑕𐑦𐑕𐑦𐑑𐑿𐑛𐑟 𐑝 𐑞 𐑗𐑱𐑕, 𐑞𐑦𐑕 𐑯𐑨𐑗𐑼𐑩𐑤 𐑤𐑲𐑯, 𐑢𐑦𐑞 𐑞 𐑥𐑩𐑑𐑻𐑯𐑩𐑤 𐑧𐑯𐑛 𐑤𐑵𐑕, 𐑚𐑦𐑒𐑳𐑥𐑟 𐑦𐑯𐑑𐑨𐑙𐑜𐑩𐑤𐑛 𐑢𐑦𐑞 𐑞 𐑣𐑧𐑥𐑐𐑩𐑯 𐑢𐑳𐑯, 𐑕𐑴 𐑞𐑨𐑑 𐑞 𐑒𐑳𐑚 𐑦𐑟 𐑞𐑺𐑚𐑲 𐑑𐑮𐑨𐑐𐑑. 𐑕𐑳𐑥 𐑝 𐑞 𐑕𐑳𐑑𐑩𐑤𐑩𐑕𐑑 𐑕𐑰𐑒𐑮𐑩𐑑𐑕 𐑝 𐑞 𐑕𐑰𐑟 𐑕𐑰𐑥𐑛 𐑛𐑲𐑝𐑳𐑤𐑡𐑛 𐑑 𐑳𐑕 𐑦𐑯 𐑞𐑦𐑕 𐑦𐑯𐑗𐑭𐑯𐑑𐑩𐑛 𐑐𐑪𐑯𐑛. 𐑢𐑰 𐑕𐑷 𐑘𐑳𐑙 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑩𐑥𐑫𐑼𐑟 𐑦𐑯 𐑞 𐑛𐑰𐑐.*

𐑞 𐑚𐑴𐑑 𐑢𐑪𐑟 𐑯𐑬 𐑷𐑤 𐑚𐑳𐑑 𐑡𐑨𐑥𐑛 𐑚𐑦𐑑𐑢𐑰𐑯 𐑑𐑵 𐑝𐑭𐑕𐑑 𐑚𐑤𐑨𐑒 𐑚𐑳𐑤𐑒𐑕, 𐑤𐑰𐑝𐑦𐑙 𐑩 𐑯𐑼𐑴 ·𐑛𐑸𐑛𐑩𐑯𐑧𐑤𐑟 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞𐑺 𐑤𐑪𐑙 𐑤𐑧𐑙𐑔𐑕. 𐑚𐑳𐑑 𐑚𐑲 𐑛𐑧𐑕𐑐𐑼𐑩𐑑 𐑦𐑯𐑛𐑧𐑝𐑼 𐑢𐑰 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑖𐑪𐑑 𐑦𐑯𐑑𐑵 𐑩 𐑑𐑧𐑥𐑐𐑼𐑼𐑦 𐑴𐑐𐑩𐑯𐑦𐑙; 𐑞𐑧𐑯 𐑜𐑦𐑝𐑦𐑙 𐑢𐑱 𐑮𐑨𐑐𐑦𐑛𐑤𐑦, 𐑯 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥 𐑻𐑯𐑩𐑕𐑑𐑤𐑦 𐑢𐑪𐑗𐑦𐑙 𐑓 𐑩𐑯𐑳𐑞𐑼 𐑬𐑑𐑤𐑧𐑑. 𐑭𐑓𐑑𐑼 𐑥𐑧𐑯𐑦 𐑕𐑦𐑥𐑦𐑤𐑼 𐑣𐑺-𐑚𐑮𐑧𐑛𐑔 𐑦𐑕𐑒𐑱𐑐𐑕, 𐑢𐑰 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑜𐑤𐑲𐑛𐑩𐑛 𐑦𐑯𐑑𐑵 𐑢𐑪𐑑 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑚𐑰𐑯 𐑢𐑳𐑯 𐑝 𐑞 𐑬𐑑𐑼 𐑕𐑻𐑒𐑩𐑤𐑟, 𐑚𐑳𐑑 𐑯𐑬 𐑒𐑮𐑪𐑕𐑑 𐑚𐑲 𐑮𐑨𐑯𐑛𐑩𐑥 𐑢𐑱𐑤𐑟, 𐑷𐑤 𐑝𐑲𐑩𐑤𐑩𐑯𐑑𐑤𐑦 𐑥𐑱𐑒𐑦𐑙 𐑓 𐑢𐑳𐑯 𐑕𐑧𐑯𐑑𐑼. 𐑞𐑦𐑕 𐑤𐑳𐑒𐑦 𐑕𐑨𐑤𐑝𐑱𐑖𐑩𐑯 𐑢𐑪𐑟 𐑗𐑰𐑐𐑤𐑦 𐑐𐑻𐑗𐑩𐑕𐑑 𐑚𐑲 𐑞 𐑤𐑪𐑕 𐑝 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑣𐑨𐑑, 𐑣𐑵, 𐑢𐑲𐑤 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑞 𐑚𐑬𐑟 𐑑 𐑐𐑮𐑦𐑒 𐑞 𐑓𐑿𐑡𐑩𐑑𐑦𐑝 𐑢𐑱𐑤𐑟, 𐑣𐑨𐑛 𐑣𐑦𐑟 𐑣𐑨𐑑 𐑑𐑱𐑒𐑩𐑯 𐑒𐑤𐑰𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑚𐑲 𐑞 𐑺-𐑧𐑛𐑦 𐑥𐑱𐑛 𐑚𐑲 𐑞 𐑕𐑳𐑛𐑩𐑯 𐑑𐑪𐑕𐑦𐑙 𐑝 𐑩 𐑐𐑺 𐑝 𐑚𐑮𐑷𐑛 𐑓𐑤𐑵𐑒𐑕 𐑒𐑤𐑴𐑕 𐑚𐑲.

𐑮𐑲𐑩𐑑𐑩𐑕 𐑯 𐑛𐑦𐑕𐑹𐑛𐑼𐑛 𐑨𐑟 𐑞 𐑿𐑯𐑦𐑝𐑻𐑕𐑩𐑤 𐑒𐑩𐑥𐑴𐑖𐑩𐑯 𐑯𐑬 𐑢𐑪𐑟, 𐑦𐑑 𐑕𐑵𐑯 𐑮𐑦𐑟𐑪𐑤𐑝𐑛 𐑦𐑑𐑕𐑧𐑤𐑓 𐑦𐑯𐑑𐑵 𐑢𐑪𐑑 𐑕𐑰𐑥𐑛 𐑩 𐑕𐑦𐑕𐑑𐑩𐑥𐑨𐑑𐑦𐑒 𐑥𐑵𐑝𐑥𐑩𐑯𐑑; 𐑓 𐑣𐑨𐑝𐑦𐑙 𐑒𐑤𐑳𐑥𐑐𐑑 𐑑𐑩𐑜𐑧𐑞𐑼 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑦𐑯 𐑢𐑳𐑯 𐑛𐑧𐑯𐑕 𐑚𐑪𐑛𐑦, 𐑞𐑱 𐑞𐑧𐑯 𐑮𐑦𐑯𐑿𐑛 𐑞𐑺 𐑪𐑯𐑢𐑼𐑛 𐑓𐑤𐑲𐑑 𐑢𐑦𐑞 𐑷𐑜𐑥𐑧𐑯𐑑𐑩𐑛 𐑓𐑤𐑰𐑑𐑯𐑩𐑕. 𐑓𐑻𐑞𐑼 𐑐𐑼𐑕𐑿𐑑 𐑢𐑪𐑟 𐑿𐑕𐑤𐑩𐑕; 𐑚𐑳𐑑 𐑞 𐑚𐑴𐑑𐑕 𐑕𐑑𐑦𐑤 𐑤𐑦𐑙𐑜𐑼𐑛 𐑦𐑯 𐑞𐑺 𐑢𐑱𐑒 𐑑 𐑐𐑦𐑒 𐑳𐑐 𐑢𐑪𐑑 𐑛𐑮𐑳𐑜𐑛 𐑢𐑱𐑤𐑟 𐑥𐑲𐑑 𐑚𐑰 𐑛𐑮𐑪𐑐𐑑 𐑩𐑕𐑑𐑻𐑯, 𐑯 𐑤𐑲𐑒𐑢𐑲𐑟 𐑑 𐑕𐑦𐑒𐑘𐑫𐑼 𐑢𐑳𐑯 𐑢𐑦𐑗 ·𐑓𐑤𐑭𐑕𐑒 𐑣𐑨𐑛 𐑒𐑦𐑤𐑛 𐑯 𐑢𐑱𐑓𐑑. 𐑞 𐑢𐑱𐑓 𐑦𐑟 𐑩 𐑐𐑧𐑯𐑩𐑯𐑛 𐑐𐑴𐑤, 𐑑𐑵 𐑹 𐑔𐑮𐑰 𐑝 𐑢𐑦𐑗 𐑸 𐑒𐑨𐑮𐑦𐑛 𐑚𐑲 𐑧𐑝𐑮𐑦 𐑚𐑴𐑑; 𐑯 𐑢𐑦𐑗, 𐑢𐑧𐑯 𐑩𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑜𐑱𐑥 𐑦𐑟 𐑨𐑑 𐑣𐑨𐑯𐑛, 𐑸 𐑦𐑯𐑕𐑻𐑑𐑩𐑛 𐑳𐑐𐑮𐑲𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑓𐑤𐑴𐑑𐑦𐑙 𐑚𐑪𐑛𐑦 𐑝 𐑩 𐑛𐑧𐑛 𐑢𐑱𐑤, 𐑚𐑴𐑔 𐑑 𐑥𐑸𐑒 𐑦𐑑𐑕 𐑐𐑤𐑱𐑕 𐑪𐑯 𐑞 𐑕𐑰, 𐑯 𐑷𐑤𐑕𐑴 𐑨𐑟 𐑑𐑴𐑒𐑩𐑯 𐑝 𐑐𐑮𐑲𐑼 𐑐𐑩𐑟𐑧𐑖𐑩𐑯, 𐑖𐑵𐑛 𐑞 𐑚𐑴𐑑𐑕 𐑝 𐑧𐑯𐑦 𐑳𐑞𐑼 𐑖𐑦𐑐 𐑛𐑮𐑷 𐑯𐑽.

𐑞 𐑮𐑦𐑟𐑳𐑤𐑑 𐑝 𐑞𐑦𐑕 𐑤𐑴𐑼𐑦𐑙 𐑢𐑪𐑟 𐑕𐑳𐑥𐑢𐑪𐑑 𐑦𐑤𐑩𐑕𐑑𐑮𐑩𐑑𐑦𐑝 𐑝 𐑞𐑨𐑑 𐑕𐑩𐑜𐑱𐑖𐑩𐑕 𐑕𐑱𐑦𐑙 𐑦𐑯 𐑞 ·𐑓𐑦𐑖𐑼𐑦,—𐑞 𐑥𐑹 𐑢𐑱𐑤𐑟 𐑞 𐑤𐑧𐑕 𐑓𐑦𐑖. 𐑝 𐑷𐑤 𐑞 𐑛𐑮𐑳𐑜𐑛 𐑢𐑱𐑤𐑟 𐑴𐑯𐑤𐑦 𐑢𐑳𐑯 𐑢𐑪𐑟 𐑒𐑨𐑐𐑗𐑼𐑛. 𐑞 𐑮𐑧𐑕𐑑 𐑒𐑩𐑯𐑑𐑮𐑲𐑝𐑛 𐑑 𐑦𐑕𐑒𐑱𐑐 𐑓 𐑞 𐑑𐑲𐑥, 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑑 𐑚𐑰 𐑑𐑱𐑒𐑩𐑯, 𐑨𐑟 𐑢𐑦𐑤 𐑣𐑽𐑭𐑓𐑑𐑼 𐑚𐑰 𐑕𐑰𐑯, 𐑚𐑲 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑒𐑮𐑭𐑓𐑑 𐑞𐑨𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛.
