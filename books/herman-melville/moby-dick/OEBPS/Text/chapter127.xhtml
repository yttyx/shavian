·𐑗𐑨𐑐𐑑𐑼 127. 𐑞 ·𐑛𐑧𐑒.

𐑞 𐑒𐑪𐑓𐑦𐑯 𐑤𐑱𐑛 𐑩𐑐𐑪𐑯 𐑑𐑵 𐑤𐑲𐑯-𐑑𐑳𐑚𐑟, 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑝𐑲𐑕-𐑚𐑧𐑯𐑗 𐑯 𐑞 𐑴𐑐𐑩𐑯 𐑣𐑨𐑗𐑢𐑱; 𐑞 ·𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑒𐑷𐑒𐑦𐑙 𐑦𐑑𐑕 𐑕𐑰𐑥𐑟; 𐑞 𐑕𐑑𐑮𐑦𐑙 𐑝 𐑑𐑢𐑦𐑕𐑑𐑩𐑛 𐑴𐑒𐑩𐑥 𐑕𐑤𐑴𐑤𐑦 𐑳𐑯𐑢𐑲𐑯𐑛𐑦𐑙 𐑓𐑮𐑪𐑥 𐑩 𐑤𐑸𐑡 𐑮𐑴𐑤 𐑝 𐑦𐑑 𐑐𐑤𐑱𐑕𐑑 𐑦𐑯 𐑞 𐑚𐑫𐑟𐑩𐑥 𐑝 𐑣𐑦𐑟 𐑓𐑮𐑪𐑒.—𐑱𐑣𐑨𐑚 𐑒𐑳𐑥𐑟 𐑕𐑤𐑴𐑤𐑦 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑨𐑚𐑦𐑯-𐑜𐑨𐑙𐑢𐑱, 𐑯 𐑣𐑽𐑟 ·𐑐𐑦𐑐 𐑓𐑪𐑤𐑴𐑦𐑙 𐑣𐑦𐑥.

“𐑚𐑨𐑒, 𐑤𐑨𐑛; 𐑲 𐑢𐑦𐑤 𐑚𐑰 𐑢𐑦𐑞 𐑘𐑰 𐑩𐑜𐑱𐑯 𐑐𐑮𐑧𐑟𐑩𐑯𐑑𐑤𐑦. 𐑣𐑰 𐑜𐑴𐑟! 𐑯𐑪𐑑 𐑞𐑦𐑕 𐑣𐑨𐑯𐑛 𐑒𐑩𐑥𐑐𐑤𐑲𐑟 𐑢𐑦𐑞 𐑥𐑲 𐑣𐑿𐑥𐑼 𐑥𐑹 𐑡𐑰𐑯𐑾𐑤𐑦 𐑞𐑨𐑯 𐑞𐑨𐑑 𐑚𐑶.—𐑥𐑦𐑛𐑩𐑤 𐑲𐑤 𐑝 𐑩 𐑗𐑻𐑗! 𐑢𐑪𐑑'𐑕 𐑣𐑽?”

“𐑤𐑲𐑓-𐑚𐑶, 𐑕𐑻. ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑹𐑛𐑼𐑟. 𐑴, 𐑤𐑫𐑒, 𐑕𐑻! 𐑚𐑦𐑢𐑺 𐑞 𐑣𐑨𐑗𐑢𐑱!”

“𐑔𐑨𐑙𐑒 𐑘𐑰, 𐑥𐑨𐑯. 𐑞𐑲 𐑒𐑪𐑓𐑦𐑯 𐑤𐑲𐑟 𐑣𐑨𐑯𐑛𐑦 𐑑 𐑞 𐑝𐑷𐑤𐑑.”

“𐑕𐑻? 𐑞 𐑣𐑨𐑗𐑢𐑱? 𐑴! 𐑕𐑴 𐑦𐑑 𐑛𐑳𐑟, 𐑕𐑻, 𐑕𐑴 𐑦𐑑 𐑛𐑳𐑟.”

“𐑸𐑑 𐑯𐑪𐑑 𐑞𐑬 𐑞 𐑤𐑧𐑜-𐑥𐑱𐑒𐑼? 𐑤𐑫𐑒, 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑞𐑦𐑕 𐑕𐑑𐑳𐑥𐑐 𐑒𐑳𐑥 𐑓𐑮𐑪𐑥 𐑞𐑲 𐑖𐑪𐑐?”

“𐑲 𐑚𐑦𐑤𐑰𐑝 𐑦𐑑 𐑛𐑦𐑛, 𐑕𐑻; 𐑛𐑳𐑟 𐑞 𐑓𐑧𐑮𐑵𐑤 𐑕𐑑𐑨𐑯𐑛, 𐑕𐑻?”

“𐑢𐑧𐑤 𐑦𐑯𐑳𐑓. 𐑚𐑳𐑑 𐑸𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑷𐑤𐑕𐑴 𐑞 𐑳𐑯𐑛𐑼𐑑𐑱𐑒𐑼?”

“𐑲, 𐑕𐑻; 𐑲 𐑐𐑨𐑗𐑑 𐑳𐑐 𐑞𐑦𐑕 𐑔𐑦𐑙 𐑣𐑽 𐑨𐑟 𐑩 𐑒𐑪𐑓𐑦𐑯 𐑓 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜; 𐑚𐑳𐑑 𐑞𐑱'𐑝 𐑕𐑧𐑑 𐑥𐑰 𐑯𐑬 𐑑 𐑑𐑻𐑯𐑦𐑙 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑕𐑳𐑥𐑔𐑦𐑙 𐑧𐑤𐑕.”

“𐑞𐑧𐑯 𐑑𐑧𐑤 𐑥𐑰; 𐑸𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑩𐑯 𐑨𐑮𐑩𐑯𐑑, 𐑷𐑤-𐑜𐑮𐑭𐑕𐑐𐑦𐑙, 𐑦𐑯𐑑𐑼𐑥𐑧𐑛𐑩𐑤𐑦𐑙, 𐑥𐑩𐑯𐑪𐑐𐑩𐑤𐑲𐑟𐑦𐑙, 𐑣𐑰𐑞𐑩𐑯𐑦𐑖 𐑴𐑤𐑛 𐑕𐑒𐑨𐑥𐑐, 𐑑 𐑚𐑰 𐑢𐑳𐑯 𐑛𐑱 𐑥𐑱𐑒𐑦𐑙 𐑤𐑧𐑜𐑟, 𐑯 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑛𐑱 𐑒𐑪𐑓𐑦𐑯𐑟 𐑑 𐑒𐑤𐑨𐑐 𐑞𐑧𐑥 𐑦𐑯, 𐑯 𐑘𐑧𐑑 𐑩𐑜𐑱𐑯 𐑤𐑲𐑓-𐑚𐑶𐑟 𐑬𐑑 𐑝 𐑞𐑴𐑟 𐑕𐑱𐑥 𐑒𐑪𐑓𐑦𐑯𐑟? 𐑞𐑬 𐑸𐑑 𐑨𐑟 𐑳𐑯𐑐𐑮𐑦𐑯𐑕𐑩𐑐𐑩𐑤𐑛 𐑨𐑟 𐑞 𐑜𐑪𐑛𐑟, 𐑯 𐑨𐑟 𐑥𐑳𐑗 𐑝 𐑩 𐑡𐑨𐑒-𐑝-𐑷𐑤-𐑑𐑮𐑱𐑛𐑟.”

“𐑚𐑳𐑑 𐑲 𐑛𐑵 𐑯𐑪𐑑 𐑥𐑰𐑯 𐑧𐑯𐑦𐑔𐑦𐑙, 𐑕𐑻. 𐑲 𐑛𐑵 𐑨𐑟 𐑲 𐑛𐑵.”

“𐑞 𐑜𐑪𐑛𐑟 𐑩𐑜𐑱𐑯. 𐑣𐑸𐑒 𐑘𐑰, 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑧𐑝𐑼 𐑕𐑦𐑙 𐑢𐑻𐑒𐑦𐑙 𐑩𐑚𐑬𐑑 𐑩 𐑒𐑪𐑓𐑦𐑯? 𐑞 ·𐑑𐑲𐑑𐑩𐑯𐑟, 𐑞𐑱 𐑕𐑱, 𐑣𐑳𐑥𐑛 𐑕𐑯𐑨𐑗𐑩𐑟 𐑢𐑧𐑯 𐑗𐑦𐑐𐑦𐑙 𐑬𐑑 𐑞 𐑒𐑮𐑱𐑑𐑼𐑟 𐑓 𐑝𐑪𐑤𐑒𐑱𐑯𐑴𐑟; 𐑯 𐑞 𐑜𐑮𐑱𐑝-𐑛𐑦𐑜𐑼 𐑦𐑯 𐑞 𐑐𐑤𐑱 𐑕𐑦𐑙𐑟, 𐑕𐑐𐑱𐑛 𐑦𐑯 𐑣𐑨𐑯𐑛. 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑯𐑧𐑝𐑼?”

“𐑕𐑦𐑙, 𐑕𐑻? 𐑛𐑵 𐑲 𐑕𐑦𐑙? 𐑴, 𐑲'𐑥 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑑 𐑦𐑯𐑳𐑓, 𐑕𐑻, 𐑓 𐑞𐑨𐑑; 𐑚𐑳𐑑 𐑞 𐑮𐑰𐑟𐑩𐑯 𐑢𐑲 𐑞 𐑜𐑮𐑱𐑝-𐑛𐑦𐑜𐑼 𐑥𐑱𐑛 𐑥𐑿𐑟𐑦𐑒 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑚𐑦𐑒𐑪𐑟 𐑞𐑺 𐑢𐑪𐑟 𐑯𐑳𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑕𐑐𐑱𐑛, 𐑕𐑻. 𐑚𐑳𐑑 𐑞 𐑒𐑷𐑒𐑦𐑙 𐑥𐑨𐑤𐑩𐑑 𐑦𐑟 𐑓𐑫𐑤 𐑝 𐑦𐑑. 𐑣𐑸𐑒 𐑑 𐑦𐑑.”

“𐑲, 𐑯 𐑞𐑨𐑑'𐑕 𐑚𐑦𐑒𐑪𐑟 𐑞 𐑤𐑦𐑛 𐑞𐑺'𐑟 𐑩 𐑕𐑬𐑯𐑛𐑦𐑙-𐑚𐑹𐑛; 𐑯 𐑢𐑪𐑑 𐑦𐑯 𐑷𐑤 𐑔𐑦𐑙𐑟 𐑥𐑱𐑒𐑕 𐑞 𐑕𐑬𐑯𐑛𐑦𐑙-𐑚𐑹𐑛 𐑦𐑟 𐑞𐑦𐑕—𐑞𐑺'𐑟 𐑯𐑷𐑑 𐑚𐑦𐑯𐑰𐑔. 𐑯 𐑘𐑧𐑑, 𐑩 𐑒𐑪𐑓𐑦𐑯 𐑢𐑦𐑞 𐑩 𐑚𐑪𐑛𐑦 𐑦𐑯 𐑦𐑑 𐑮𐑦𐑙𐑟 𐑐𐑮𐑦𐑑𐑦 𐑥𐑳𐑗 𐑞 𐑕𐑱𐑥, ·𐑒𐑸𐑐𐑩𐑯𐑑𐑼. 𐑣𐑨𐑕𐑑 𐑞𐑬 𐑧𐑝𐑼 𐑣𐑧𐑤𐑐𐑑 𐑒𐑨𐑮𐑦 𐑩 𐑚𐑽, 𐑯 𐑣𐑻𐑛 𐑞 𐑒𐑪𐑓𐑦𐑯 𐑯𐑪𐑒 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑗𐑻𐑗𐑘𐑸𐑛 𐑜𐑱𐑑, 𐑜𐑴𐑦𐑙 𐑦𐑯?

“𐑓𐑱𐑔, 𐑕𐑻, 𐑲'𐑝——”

“𐑓𐑱𐑔? 𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑?”

“𐑢𐑲, 𐑓𐑱𐑔, 𐑕𐑻, 𐑦𐑑'𐑕 𐑴𐑯𐑤𐑦 𐑩 𐑕𐑹𐑑 𐑝 𐑧𐑒𐑕𐑒𐑤𐑩𐑥𐑱𐑖𐑩𐑯-𐑤𐑲𐑒—𐑞𐑨𐑑'𐑕 𐑷𐑤, 𐑕𐑻.”

“𐑳𐑥, 𐑳𐑥; 𐑜𐑴 𐑪𐑯.”

“𐑲 𐑢𐑪𐑟 𐑩𐑚𐑬𐑑 𐑑 𐑕𐑱, 𐑕𐑻, 𐑞𐑨𐑑——”

“𐑸𐑑 𐑞𐑬 𐑩 𐑕𐑦𐑤𐑒-𐑢𐑻𐑥? 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑕𐑐𐑦𐑯 𐑞𐑲 𐑴𐑯 𐑖𐑮𐑬𐑛 𐑬𐑑 𐑝 𐑞𐑲𐑕𐑧𐑤𐑓? 𐑤𐑫𐑒 𐑨𐑑 𐑞𐑲 𐑚𐑫𐑟𐑩𐑥! 𐑛𐑦𐑕𐑐𐑨𐑗! 𐑯 𐑜𐑧𐑑 𐑞𐑰𐑟 𐑑𐑮𐑨𐑐𐑕 𐑬𐑑 𐑝 𐑕𐑲𐑑.”

“𐑣𐑰 𐑜𐑴𐑟 𐑭𐑓𐑑. 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑕𐑳𐑛𐑩𐑯, 𐑯𐑬; 𐑚𐑳𐑑 𐑕𐑒𐑢𐑷𐑤𐑟 𐑒𐑳𐑥 𐑕𐑳𐑛𐑩𐑯 𐑦𐑯 𐑣𐑪𐑑 𐑤𐑨𐑑𐑦𐑑𐑿𐑛𐑟. 𐑲'𐑝 𐑣𐑻𐑛 𐑞𐑨𐑑 𐑞 ·𐑲𐑤 𐑝 ·𐑨𐑤𐑚𐑩𐑥𐑸𐑤, 𐑢𐑳𐑯 𐑝 𐑞 ·𐑜𐑨𐑤𐑲𐑐𐑩𐑜𐑴𐑟, 𐑦𐑟 𐑒𐑳𐑑 𐑚𐑲 𐑞 ·𐑦𐑒𐑢𐑱𐑑𐑼 𐑮𐑲𐑑 𐑦𐑯 𐑞 𐑥𐑦𐑛𐑩𐑤. 𐑕𐑰𐑥𐑟 𐑑 𐑥𐑰 𐑕𐑳𐑥 𐑕𐑹𐑑 𐑝 ·𐑦𐑒𐑢𐑱𐑑𐑼 𐑒𐑳𐑑𐑕 𐑘𐑪𐑯 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑑𐑵, 𐑮𐑲𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑥𐑦𐑛𐑩𐑤. 𐑣𐑰'𐑟 𐑷𐑤𐑢𐑱𐑟 𐑳𐑯𐑛𐑼 𐑞 ·𐑤𐑲𐑯—𐑓𐑲𐑼𐑦 𐑣𐑪𐑑, 𐑲 𐑑𐑧𐑤 𐑘𐑰! 𐑣𐑰'𐑟 𐑤𐑫𐑒𐑦𐑙 𐑞𐑦𐑕 𐑢𐑱—𐑒𐑳𐑥, 𐑴𐑒𐑩𐑥; 𐑒𐑢𐑦𐑒. 𐑣𐑽 𐑢𐑰 𐑜𐑴 𐑩𐑜𐑱𐑯. 𐑞𐑦𐑕 𐑢𐑫𐑛𐑩𐑯 𐑥𐑨𐑤𐑩𐑑 𐑦𐑟 𐑞 𐑒𐑹𐑒, 𐑯 𐑲'𐑥 𐑞 𐑐𐑮𐑩𐑓𐑧𐑕𐑼 𐑝 𐑥𐑿𐑟𐑦𐑒𐑩𐑤 𐑜𐑤𐑭𐑕𐑩𐑟—𐑑𐑨𐑐, 𐑑𐑨𐑐!”

(𐑱𐑣𐑨𐑚 𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓.)

“𐑞𐑺'𐑟 𐑩 𐑕𐑲𐑑! 𐑞𐑺'𐑟 𐑩 𐑕𐑬𐑯𐑛! 𐑞 𐑜𐑮𐑱𐑣𐑧𐑛𐑩𐑛 𐑢𐑫𐑛𐑐𐑧𐑒𐑼 𐑑𐑨𐑐𐑦𐑙 𐑞 𐑣𐑪𐑤𐑴 𐑑𐑮𐑰! 𐑚𐑤𐑲𐑯𐑛 𐑯 𐑛𐑳𐑥 𐑥𐑲𐑑 𐑢𐑧𐑤 𐑚𐑰 𐑧𐑯𐑝𐑦𐑛 𐑯𐑬. 𐑕𐑰! 𐑞𐑨𐑑 𐑔𐑦𐑙 𐑮𐑧𐑕𐑑𐑕 𐑪𐑯 𐑑𐑵 𐑤𐑲𐑯-𐑑𐑳𐑚𐑟, 𐑓𐑫𐑤 𐑝 𐑑𐑴-𐑤𐑲𐑯𐑟. 𐑩 𐑥𐑴𐑕𐑑 𐑥𐑩𐑤𐑦𐑖𐑩𐑕 𐑢𐑨𐑜, 𐑞𐑨𐑑 𐑓𐑧𐑤𐑴. 𐑮𐑨𐑑-𐑑𐑨𐑑! 𐑕𐑴 𐑥𐑨𐑯'𐑟 𐑕𐑧𐑒𐑩𐑯𐑛𐑟 𐑑𐑦𐑒! 𐑴! 𐑣𐑬 𐑦𐑥𐑩𐑑𐑽𐑾𐑤 𐑸 𐑷𐑤 𐑥𐑩𐑑𐑽𐑾𐑤𐑟! 𐑢𐑪𐑑 𐑔𐑦𐑙𐑟 𐑮𐑾𐑤 𐑸 𐑞𐑺, 𐑚𐑳𐑑 𐑦𐑥𐑐𐑪𐑯𐑛𐑼𐑩𐑚𐑩𐑤 𐑔𐑷𐑑𐑕? 𐑣𐑽 𐑯𐑬'𐑟 𐑞 𐑝𐑧𐑮𐑦 𐑛𐑮𐑧𐑛𐑩𐑛 𐑕𐑦𐑥𐑚𐑩𐑤 𐑝 𐑜𐑮𐑦𐑥 𐑛𐑧𐑔, 𐑚𐑲 𐑩 𐑥𐑽 𐑣𐑨𐑐, 𐑥𐑱𐑛 𐑞 𐑦𐑒𐑕𐑐𐑮𐑧𐑕𐑦𐑝 𐑕𐑲𐑯 𐑝 𐑞 𐑣𐑧𐑤𐑐 𐑯 𐑣𐑴𐑐 𐑝 𐑥𐑴𐑕𐑑 𐑦𐑯𐑛𐑱𐑯𐑡𐑼𐑛 𐑤𐑲𐑓. 𐑩 𐑤𐑲𐑓-𐑚𐑶 𐑝 𐑩 𐑒𐑪𐑓𐑦𐑯! 𐑛𐑳𐑟 𐑦𐑑 𐑜𐑴 𐑓𐑻𐑞𐑼? 𐑒𐑨𐑯 𐑦𐑑 𐑚𐑰 𐑞𐑨𐑑 𐑦𐑯 𐑕𐑳𐑥 𐑕𐑐𐑦𐑮𐑦𐑗𐑵𐑩𐑤 𐑕𐑧𐑯𐑕 𐑞 𐑒𐑪𐑓𐑦𐑯 𐑦𐑟, 𐑭𐑓𐑑𐑼 𐑷𐑤, 𐑚𐑳𐑑 𐑩𐑯 𐑦𐑥𐑹𐑑𐑨𐑤𐑦𐑑𐑦-𐑐𐑮𐑦𐑟𐑻𐑝𐑼! 𐑲'𐑤 𐑔𐑦𐑙𐑒 𐑝 𐑞𐑨𐑑. 𐑚𐑳𐑑 𐑯𐑴. 𐑕𐑴 𐑓𐑸 𐑜𐑪𐑯 𐑨𐑥 𐑲 𐑦𐑯 𐑞 𐑛𐑸𐑒 𐑕𐑲𐑛 𐑝 𐑻𐑔, 𐑞𐑨𐑑 𐑦𐑑𐑕 𐑳𐑞𐑼 𐑕𐑲𐑛, 𐑞 𐑔𐑽𐑧𐑑𐑦𐑒 𐑚𐑮𐑲𐑑 𐑢𐑳𐑯, 𐑕𐑰𐑥𐑟 𐑚𐑳𐑑 𐑳𐑯𐑕𐑻𐑑𐑩𐑯 𐑑𐑢𐑲𐑤𐑲𐑑 𐑑 𐑥𐑰. 𐑢𐑦𐑤 𐑘𐑰 𐑯𐑧𐑝𐑼 𐑣𐑨𐑝 𐑛𐑳𐑯, ·𐑒𐑸𐑐𐑩𐑯𐑑𐑼, 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑩𐑒𐑻𐑕𐑩𐑛 𐑕𐑬𐑯𐑛? 𐑲 𐑜𐑴 𐑚𐑦𐑤𐑴; 𐑤𐑧𐑑 𐑥𐑰 𐑯𐑪𐑑 𐑕𐑰 𐑞𐑨𐑑 𐑔𐑦𐑙 𐑣𐑽 𐑢𐑧𐑯 𐑲 𐑮𐑦𐑑𐑻𐑯 𐑩𐑜𐑱𐑯. 𐑯𐑬, 𐑞𐑧𐑯, ·𐑐𐑦𐑐, 𐑢𐑰'𐑤 𐑑𐑷𐑒 𐑞𐑦𐑕 𐑴𐑝𐑼; 𐑲 𐑛𐑵 𐑕𐑳𐑒 𐑥𐑴𐑕𐑑 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑓𐑦𐑤𐑪𐑕𐑩𐑓𐑦𐑟 𐑓𐑮𐑪𐑥 𐑞𐑰! 𐑕𐑳𐑥 𐑳𐑯𐑯𐑴𐑯 𐑒𐑪𐑯𐑛𐑿𐑦𐑑𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑳𐑯𐑯𐑴𐑯 𐑢𐑻𐑤𐑛𐑟 𐑥𐑳𐑕𐑑 𐑧𐑥𐑐𐑑𐑦 𐑦𐑯𐑑𐑵 𐑞𐑰!”
