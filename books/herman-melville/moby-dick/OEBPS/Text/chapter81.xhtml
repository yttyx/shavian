·𐑗𐑨𐑐𐑑𐑼 81. 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 ·𐑥𐑰𐑑𐑕 𐑞 ·𐑝𐑻𐑡𐑦𐑯.

𐑞 𐑐𐑮𐑰𐑛𐑧𐑕𐑑𐑦𐑯𐑱𐑑𐑩𐑛 𐑛𐑱 𐑼𐑲𐑝𐑛, 𐑯 𐑢𐑰 𐑛𐑵𐑤𐑦 𐑥𐑧𐑑 𐑞 𐑖𐑦𐑐 ·𐑿𐑙𐑓𐑮𐑬, ·𐑛𐑧𐑮𐑦𐑒 ·𐑛𐑦 ·𐑛𐑽, 𐑥𐑭𐑕𐑑𐑼, 𐑝 ·𐑚𐑮𐑱𐑥𐑩𐑯.

𐑨𐑑 𐑢𐑳𐑯 𐑑𐑲𐑥 𐑞 𐑜𐑮𐑱𐑑𐑩𐑕𐑑 𐑢𐑱𐑤𐑦𐑙 𐑐𐑰𐑐𐑩𐑤 𐑦𐑯 𐑞 𐑢𐑻𐑤𐑛, 𐑞 ·𐑛𐑳𐑗 𐑯 ·𐑡𐑻𐑥𐑩𐑯𐑟 𐑸 𐑯𐑬 𐑩𐑥𐑳𐑙 𐑞 𐑤𐑰𐑕𐑑; 𐑚𐑳𐑑 𐑣𐑽 𐑯 𐑞𐑺 𐑨𐑑 𐑝𐑧𐑮𐑦 𐑢𐑲𐑛 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑝 𐑤𐑨𐑑𐑦𐑑𐑿𐑛 𐑯 𐑤𐑪𐑯𐑜𐑦𐑑𐑿𐑛, 𐑿 𐑕𐑑𐑦𐑤 𐑩𐑒𐑱𐑠𐑩𐑯𐑩𐑤𐑦 𐑥𐑰𐑑 𐑢𐑦𐑞 𐑞𐑺 𐑓𐑤𐑨𐑜 𐑦𐑯 𐑞 ·𐑐𐑩𐑕𐑦𐑓𐑦𐑒.

𐑓 𐑕𐑳𐑥 𐑮𐑰𐑟𐑩𐑯, 𐑞 ·𐑿𐑙𐑓𐑮𐑬 𐑕𐑰𐑥𐑛 𐑒𐑢𐑲𐑑 𐑰𐑜𐑼 𐑑 𐑐𐑱 𐑣𐑻 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑕. 𐑢𐑲𐑤 𐑘𐑧𐑑 𐑕𐑳𐑥 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑓𐑮𐑪𐑥 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑖𐑰 𐑮𐑬𐑯𐑛𐑩𐑛 𐑑, 𐑯 𐑛𐑮𐑪𐑐𐑦𐑙 𐑩 𐑚𐑴𐑑, 𐑣𐑻 𐑒𐑨𐑐𐑑𐑦𐑯 𐑢𐑪𐑟 𐑦𐑥𐑐𐑧𐑤𐑛 𐑑𐑩𐑢𐑹𐑛𐑟 𐑳𐑕, 𐑦𐑥𐑐𐑱𐑖𐑩𐑯𐑑𐑤𐑦 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑞 𐑚𐑬𐑟 𐑦𐑯𐑕𐑑𐑧𐑛 𐑝 𐑞 𐑕𐑑𐑻𐑯.

“𐑢𐑪𐑑 𐑣𐑨𐑟 𐑣𐑰 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑞𐑺?” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑐𐑶𐑯𐑑𐑦𐑙 𐑑 𐑕𐑳𐑥𐑔𐑦𐑙 𐑢𐑱𐑝𐑦𐑙𐑤𐑦 𐑣𐑧𐑤𐑛 𐑚𐑲 𐑞 ·𐑡𐑻𐑥𐑩𐑯. “𐑦𐑥𐑐𐑪𐑕𐑩𐑚𐑩𐑤!—𐑩 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼!”

“𐑯𐑪𐑑 𐑞𐑨𐑑,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, “𐑯𐑴, 𐑯𐑴, 𐑦𐑑'𐑕 𐑩 𐑒𐑪𐑓𐑦-𐑐𐑪𐑑, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑣𐑰'𐑟 𐑒𐑳𐑥𐑦𐑙 𐑪𐑓 𐑑 𐑥𐑱𐑒 𐑳𐑕 𐑬𐑼 𐑒𐑪𐑓𐑦, 𐑦𐑟 𐑞 ·𐑘𐑸𐑥𐑩𐑯; 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑰 𐑞𐑨𐑑 𐑚𐑦𐑜 𐑑𐑦𐑯 𐑒𐑨𐑯 𐑞𐑺 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑝 𐑣𐑦𐑥?—𐑞𐑨𐑑'𐑕 𐑣𐑦𐑟 𐑚𐑶𐑤𐑦𐑙 𐑢𐑷𐑑𐑼. 𐑴! 𐑣𐑰'𐑟 𐑷𐑤 𐑮𐑲𐑑, 𐑦𐑟 𐑞 ·𐑘𐑸𐑥𐑩𐑯.”

“𐑜𐑴 𐑩𐑤𐑪𐑙 𐑢𐑦𐑞 𐑿,” 𐑒𐑮𐑲𐑛 ·𐑓𐑤𐑭𐑕𐑒, “𐑦𐑑'𐑕 𐑩 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼 𐑯 𐑩𐑯 𐑶𐑤-𐑒𐑨𐑯. 𐑣𐑰'𐑟 𐑬𐑑 𐑝 𐑶𐑤, 𐑯 𐑣𐑨𐑟 𐑒𐑳𐑥 𐑩-𐑚𐑧𐑜𐑦𐑙.”

𐑣𐑬𐑧𐑝𐑼 𐑒𐑘𐑫𐑼𐑾𐑕 𐑦𐑑 𐑥𐑱 𐑕𐑰𐑥 𐑓 𐑩𐑯 𐑶𐑤-𐑖𐑦𐑐 𐑑 𐑚𐑰 𐑚𐑪𐑮𐑴𐑦𐑙 𐑶𐑤 𐑪𐑯 𐑞 𐑢𐑱𐑤-𐑜𐑮𐑬𐑯𐑛, 𐑯 𐑣𐑬𐑧𐑝𐑼 𐑥𐑳𐑗 𐑦𐑑 𐑥𐑱 𐑦𐑯𐑝𐑻𐑑𐑩𐑛𐑤𐑦 𐑒𐑪𐑯𐑑𐑮𐑩𐑛𐑦𐑒𐑑 𐑞 𐑴𐑤𐑛 𐑐𐑮𐑪𐑝𐑻𐑚 𐑩𐑚𐑬𐑑 𐑒𐑨𐑮𐑦𐑦𐑙 𐑒𐑴𐑤𐑟 𐑑 ·𐑯𐑿𐑒𐑭𐑕𐑩𐑤, 𐑘𐑧𐑑 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑕𐑳𐑗 𐑩 𐑔𐑦𐑙 𐑮𐑾𐑤𐑦 𐑣𐑨𐑐𐑩𐑯𐑟; 𐑯 𐑦𐑯 𐑞 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑒𐑱𐑕 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑛𐑧𐑮𐑦𐑒 ·𐑛𐑦 ·𐑛𐑽 𐑛𐑦𐑛 𐑦𐑯𐑛𐑿𐑚𐑦𐑑𐑩𐑚𐑤𐑦 𐑒𐑩𐑯𐑛𐑳𐑒𐑑 𐑩 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼 𐑨𐑟 ·𐑓𐑤𐑭𐑕𐑒 𐑛𐑦𐑛 𐑛𐑦𐑒𐑤𐑺.

𐑨𐑟 𐑣𐑰 𐑥𐑬𐑯𐑑𐑩𐑛 𐑞 𐑛𐑧𐑒, ·𐑱𐑣𐑨𐑚 𐑩𐑚𐑮𐑳𐑐𐑑𐑤𐑦 𐑩𐑒𐑪𐑕𐑑𐑩𐑛 𐑣𐑦𐑥, 𐑢𐑦𐑞𐑬𐑑 𐑨𐑑 𐑷𐑤 𐑣𐑰𐑛𐑦𐑙 𐑢𐑪𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛; 𐑚𐑳𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑮𐑴𐑒𐑩𐑯 𐑤𐑦𐑙𐑜𐑴, 𐑞 ·𐑡𐑻𐑥𐑩𐑯 𐑕𐑵𐑯 𐑦𐑝𐑦𐑯𐑕𐑑 𐑣𐑦𐑟 𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑦𐑜𐑯𐑼𐑩𐑯𐑕 𐑝 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤; 𐑦𐑥𐑰𐑛𐑾𐑑𐑤𐑦 𐑑𐑻𐑯𐑦𐑙 𐑞 𐑒𐑪𐑯𐑝𐑼𐑕𐑱𐑖𐑩𐑯 𐑑 𐑣𐑦𐑟 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼 𐑯 𐑶𐑤 𐑒𐑨𐑯, 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑮𐑦𐑥𐑸𐑒𐑕 𐑑𐑳𐑗𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑝𐑦𐑙 𐑑 𐑑𐑻𐑯 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑣𐑨𐑥𐑩𐑒 𐑨𐑑 𐑯𐑲𐑑 𐑦𐑯 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑛𐑸𐑒𐑯𐑩𐑕—𐑣𐑦𐑟 𐑤𐑭𐑕𐑑 𐑛𐑮𐑪𐑐 𐑝 ·𐑚𐑮𐑱𐑥𐑩𐑯 𐑶𐑤 𐑚𐑰𐑦𐑙 𐑜𐑪𐑯, 𐑯 𐑯𐑪𐑑 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑓𐑤𐑲𐑦𐑙-𐑓𐑦𐑖 𐑘𐑧𐑑 𐑒𐑨𐑐𐑗𐑼𐑛 𐑑 𐑕𐑩𐑐𐑤𐑲 𐑞 𐑛𐑦𐑓𐑦𐑖𐑩𐑯𐑕𐑦; 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑦𐑙 𐑚𐑲 𐑣𐑦𐑯𐑑𐑦𐑙 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑖𐑦𐑐 𐑢𐑪𐑟 𐑦𐑯𐑛𐑰𐑛 𐑢𐑪𐑑 𐑦𐑯 𐑞 ·𐑓𐑦𐑖𐑼𐑦 𐑦𐑟 𐑑𐑧𐑒𐑯𐑦𐑒𐑤𐑦 𐑒𐑷𐑤𐑛 𐑩 𐑒𐑤𐑰𐑯 𐑢𐑳𐑯 (𐑞𐑨𐑑 𐑦𐑟, 𐑩𐑯 𐑧𐑥𐑐𐑑𐑦 𐑢𐑳𐑯), 𐑢𐑧𐑤 𐑛𐑦𐑟𐑻𐑝𐑦𐑙 𐑞 𐑯𐑱𐑥 𐑝 ·𐑿𐑙𐑓𐑮𐑬 𐑹 𐑞 ·𐑝𐑻𐑡𐑦𐑯.

𐑣𐑦𐑟 𐑯𐑦𐑕𐑧𐑕𐑦𐑑𐑦𐑟 𐑕𐑩𐑐𐑤𐑲𐑛, ·𐑛𐑧𐑮𐑦𐑒 𐑛𐑦𐑐𐑸𐑑𐑩𐑛; 𐑚𐑳𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑜𐑱𐑯𐑛 𐑣𐑦𐑟 𐑖𐑦𐑐'𐑕 𐑕𐑲𐑛, 𐑢𐑧𐑯 𐑢𐑱𐑤𐑟 𐑢𐑻 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑦𐑥𐑩𐑤𐑑𐑱𐑯𐑾𐑕𐑤𐑦 𐑮𐑱𐑟𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑝 𐑚𐑴𐑔 𐑝𐑧𐑕𐑩𐑤𐑟; 𐑯 𐑕𐑴 𐑰𐑜𐑼 𐑓 𐑞 𐑗𐑱𐑕 𐑢𐑪𐑟 ·𐑛𐑧𐑮𐑦𐑒, 𐑞𐑨𐑑 𐑢𐑦𐑞𐑬𐑑 𐑐𐑷𐑟𐑦𐑙 𐑑 𐑐𐑫𐑑 𐑣𐑦𐑟 𐑶𐑤-𐑒𐑨𐑯 𐑯 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼 𐑩𐑚𐑹𐑛, 𐑣𐑰 𐑕𐑤𐑵𐑛 𐑮𐑬𐑯𐑛 𐑣𐑦𐑟 𐑚𐑴𐑑 𐑯 𐑥𐑱𐑛 𐑭𐑓𐑑𐑼 𐑞 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼𐑟.

𐑯𐑬, 𐑞 𐑜𐑱𐑥 𐑣𐑨𐑝𐑦𐑙 𐑮𐑦𐑟𐑩𐑯 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑣𐑰 𐑯 𐑞 𐑳𐑞𐑼 𐑔𐑮𐑰 ·𐑡𐑻𐑥𐑩𐑯 𐑚𐑴𐑑𐑕 𐑞𐑨𐑑 𐑕𐑵𐑯 𐑓𐑪𐑤𐑴𐑛 𐑣𐑦𐑥, 𐑣𐑨𐑛 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑤𐑦 𐑞 𐑕𐑑𐑸𐑑 𐑝 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑒𐑰𐑤𐑟. 𐑞𐑺 𐑢𐑻 𐑱𐑑 𐑢𐑱𐑤𐑟, 𐑩𐑯 𐑨𐑝𐑼𐑦𐑡 𐑐𐑪𐑛. 𐑩𐑢𐑺 𐑝 𐑞𐑺 𐑛𐑱𐑯𐑡𐑼, 𐑞𐑱 𐑢𐑻 𐑜𐑴𐑦𐑙 𐑷𐑤 𐑩𐑚𐑮𐑧𐑕𐑑 𐑢𐑦𐑞 𐑜𐑮𐑱𐑑 𐑕𐑐𐑰𐑛 𐑕𐑑𐑮𐑱𐑑 𐑚𐑦𐑓𐑹 𐑞 𐑢𐑦𐑯𐑛, 𐑮𐑳𐑚𐑦𐑙 𐑞𐑺 𐑓𐑤𐑨𐑙𐑒𐑕 𐑨𐑟 𐑒𐑤𐑴𐑕𐑤𐑦 𐑨𐑟 𐑕𐑴 𐑥𐑧𐑯𐑦 𐑕𐑐𐑨𐑯𐑟 𐑝 𐑣𐑹𐑕𐑩𐑟 𐑦𐑯 𐑣𐑸𐑯𐑩𐑕. 𐑞𐑱 𐑤𐑧𐑓𐑑 𐑩 𐑜𐑮𐑱𐑑, 𐑢𐑲𐑛 𐑢𐑱𐑒, 𐑨𐑟 𐑞𐑴 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤𐑦 𐑳𐑯𐑮𐑴𐑤𐑦𐑙 𐑩 𐑜𐑮𐑱𐑑 𐑢𐑲𐑛 𐑐𐑸𐑗𐑥𐑩𐑯𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑰.

𐑓𐑫𐑤 𐑦𐑯 𐑞𐑦𐑕 𐑮𐑨𐑐𐑦𐑛 𐑢𐑱𐑒, 𐑯 𐑥𐑧𐑯𐑦 𐑓𐑨𐑞𐑩𐑥𐑟 𐑦𐑯 𐑞 𐑮𐑽, 𐑕𐑢𐑨𐑥 𐑩 𐑣𐑿𐑡, 𐑣𐑳𐑥𐑐𐑑 𐑴𐑤𐑛 𐑚𐑫𐑤, 𐑢𐑦𐑗 𐑚𐑲 𐑣𐑦𐑟 𐑒𐑩𐑥𐑐𐑨𐑮𐑩𐑑𐑦𐑝𐑤𐑦 𐑕𐑤𐑴 𐑐𐑮𐑴𐑜𐑮𐑧𐑕, 𐑨𐑟 𐑢𐑧𐑤 𐑨𐑟 𐑚𐑲 𐑞 𐑳𐑯𐑿𐑠𐑵𐑩𐑤 𐑘𐑧𐑤𐑴𐑦𐑖 𐑦𐑯𐑒𐑮𐑳𐑕𐑑𐑱𐑖𐑩𐑯𐑟 𐑴𐑝𐑼𐑜𐑮𐑴𐑦𐑙 𐑣𐑦𐑥, 𐑕𐑰𐑥𐑛 𐑩𐑓𐑤𐑦𐑒𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞 𐑡𐑷𐑯𐑛𐑦𐑕, 𐑹 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑦𐑯𐑓𐑻𐑥𐑦𐑑𐑦. 𐑢𐑧𐑞𐑼 𐑞𐑦𐑕 𐑢𐑱𐑤 𐑚𐑦𐑤𐑪𐑙𐑛 𐑑 𐑞 𐑐𐑪𐑛 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕, 𐑕𐑰𐑥𐑛 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑩𐑚𐑩𐑤; 𐑓 𐑦𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑒𐑳𐑕𐑑𐑩𐑥𐑼𐑦 𐑓 𐑕𐑳𐑗 𐑝𐑧𐑯𐑼𐑩𐑚𐑩𐑤 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯𐑟 𐑑 𐑚𐑰 𐑨𐑑 𐑷𐑤 𐑕𐑴𐑖𐑩𐑤. 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕, 𐑣𐑰 𐑕𐑑𐑳𐑒 𐑑 𐑞𐑺 𐑢𐑱𐑒, 𐑞𐑴 𐑦𐑯𐑛𐑰𐑛 𐑞𐑺 𐑚𐑨𐑒 𐑢𐑷𐑑𐑼 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑮𐑦𐑑𐑸𐑛𐑩𐑛 𐑣𐑦𐑥, 𐑚𐑦𐑒𐑪𐑟 𐑞 𐑢𐑲𐑑-𐑚𐑴𐑯 𐑹 𐑕𐑢𐑧𐑤 𐑨𐑑 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛 𐑥𐑳𐑟𐑩𐑤 𐑢𐑪𐑟 𐑩 𐑛𐑨𐑖𐑑 𐑢𐑳𐑯, 𐑤𐑲𐑒 𐑞 𐑕𐑢𐑧𐑤 𐑓𐑹𐑥𐑛 𐑢𐑧𐑯 𐑑𐑵 𐑣𐑪𐑕𐑑𐑲𐑤 𐑒𐑳𐑮𐑩𐑯𐑑𐑕 𐑥𐑰𐑑. 𐑣𐑦𐑟 𐑕𐑐𐑬𐑑 𐑢𐑪𐑟 𐑖𐑹𐑑, 𐑕𐑤𐑴, 𐑯 𐑤𐑩𐑚𐑹𐑾𐑕; 𐑒𐑳𐑥𐑦𐑙 𐑓𐑹𐑔 𐑢𐑦𐑞 𐑩 𐑗𐑴𐑒𐑦𐑙 𐑕𐑹𐑑 𐑝 𐑜𐑳𐑖, 𐑯 𐑕𐑐𐑧𐑯𐑛𐑦𐑙 𐑦𐑑𐑕𐑧𐑤𐑓 𐑦𐑯 𐑑𐑹𐑯 𐑖𐑮𐑧𐑛𐑟, 𐑓𐑪𐑤𐑴𐑛 𐑚𐑲 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑳𐑚𐑑𐑼𐑱𐑯𐑾𐑯 𐑒𐑩𐑥𐑴𐑖𐑩𐑯𐑟 𐑦𐑯 𐑣𐑦𐑥, 𐑢𐑦𐑗 𐑕𐑰𐑥𐑛 𐑑 𐑣𐑨𐑝 𐑰𐑜𐑮𐑧𐑕 𐑨𐑑 𐑣𐑦𐑟 𐑳𐑞𐑼 𐑚𐑧𐑮𐑦𐑛 𐑦𐑒𐑕𐑑𐑮𐑧𐑥𐑦𐑑𐑦, 𐑒𐑷𐑟𐑦𐑙 𐑞 𐑢𐑷𐑑𐑼𐑟 𐑚𐑦𐑣𐑲𐑯𐑛 𐑣𐑦𐑥 𐑑 𐑳𐑐𐑚𐑳𐑚𐑩𐑤.

“𐑣𐑵'𐑟 𐑜𐑪𐑑 𐑕𐑳𐑥 𐑐𐑨𐑮𐑩𐑜𐑪𐑮𐑦𐑒?” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, “𐑣𐑰 𐑣𐑨𐑟 𐑞 𐑕𐑑𐑳𐑥𐑩𐑒-𐑱𐑒, 𐑲'𐑥 𐑩𐑓𐑮𐑱𐑛. 𐑤𐑹𐑛, 𐑔𐑦𐑙𐑒 𐑝 𐑣𐑨𐑝𐑦𐑙 𐑣𐑭𐑓 𐑩𐑯 𐑱𐑒𐑼 𐑝 𐑕𐑑𐑳𐑥𐑩𐑒-𐑱𐑒! 𐑨𐑛𐑝𐑻𐑕 𐑢𐑦𐑯𐑛𐑟 𐑸 𐑣𐑴𐑤𐑛𐑦𐑙 𐑥𐑨𐑛 ·𐑒𐑮𐑦𐑕𐑥𐑩𐑕 𐑦𐑯 𐑣𐑦𐑥, 𐑚𐑶𐑟. 𐑦𐑑'𐑕 𐑞 𐑓𐑻𐑕𐑑 𐑓𐑬𐑤 𐑢𐑦𐑯𐑛 𐑲 𐑧𐑝𐑼 𐑯𐑿 𐑑 𐑚𐑤𐑴 𐑓𐑮𐑪𐑥 𐑩𐑕𐑑𐑻𐑯; 𐑚𐑳𐑑 𐑤𐑫𐑒, 𐑛𐑦𐑛 𐑧𐑝𐑼 𐑢𐑱𐑤 𐑘𐑷 𐑕𐑴 𐑚𐑦𐑓𐑹? 𐑦𐑑 𐑥𐑳𐑕𐑑 𐑚𐑰, 𐑣𐑰'𐑟 𐑤𐑪𐑕𐑑 𐑣𐑦𐑟 𐑑𐑦𐑤𐑼.”

𐑨𐑟 𐑩𐑯 𐑴𐑝𐑼𐑤𐑱𐑛𐑩𐑯 ·𐑦𐑯𐑛𐑾𐑥𐑩𐑯 𐑚𐑺𐑦𐑙 𐑛𐑬𐑯 𐑞 ·𐑣𐑲𐑯𐑛𐑴𐑕𐑑𐑭𐑯 𐑒𐑴𐑕𐑑 𐑢𐑦𐑞 𐑩 𐑛𐑧𐑒 𐑤𐑴𐑛 𐑝 𐑓𐑮𐑲𐑑𐑩𐑯𐑛 𐑣𐑹𐑕𐑩𐑟, 𐑒𐑼𐑰𐑯𐑟, 𐑚𐑧𐑮𐑦𐑟, 𐑮𐑴𐑤𐑟, 𐑯 𐑢𐑪𐑤𐑴𐑟 𐑪𐑯 𐑣𐑻 𐑢𐑱; 𐑕𐑴 𐑛𐑦𐑛 𐑞𐑦𐑕 𐑴𐑤𐑛 𐑢𐑱𐑤 𐑣𐑰𐑝 𐑣𐑦𐑟 𐑱𐑡𐑛 𐑚𐑳𐑤𐑒, 𐑯 𐑯𐑬 𐑯 𐑞𐑧𐑯 𐑐𐑸𐑑𐑤𐑦 𐑑𐑻𐑯𐑦𐑙 𐑴𐑝𐑼 𐑪𐑯 𐑣𐑦𐑟 𐑒𐑳𐑥𐑚𐑮𐑩𐑕 𐑮𐑦𐑚-𐑧𐑯𐑛𐑟, 𐑦𐑒𐑕𐑐𐑴𐑟 𐑞 𐑒𐑷𐑟 𐑝 𐑣𐑦𐑟 𐑛𐑰𐑝𐑾𐑕 𐑢𐑱𐑒 𐑦𐑯 𐑞 𐑳𐑯𐑯𐑨𐑗𐑼𐑩𐑤 𐑕𐑑𐑳𐑥𐑐 𐑝 𐑣𐑦𐑟 𐑕𐑑𐑸𐑚𐑼𐑛 𐑓𐑦𐑯. 𐑢𐑧𐑞𐑼 𐑣𐑰 𐑣𐑨𐑛 𐑤𐑪𐑕𐑑 𐑞𐑨𐑑 𐑓𐑦𐑯 𐑦𐑯 𐑚𐑨𐑑𐑩𐑤, 𐑹 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑚𐑹𐑯 𐑢𐑦𐑞𐑬𐑑 𐑦𐑑, 𐑦𐑑 𐑢𐑻 𐑣𐑸𐑛 𐑑 𐑕𐑱.

“𐑴𐑯𐑤𐑦 𐑢𐑱𐑑 𐑩 𐑚𐑦𐑑, 𐑴𐑤𐑛 𐑗𐑨𐑐, 𐑯 𐑲'𐑤 𐑜𐑦𐑝 𐑘𐑰 𐑩 𐑕𐑤𐑦𐑙 𐑓 𐑞𐑨𐑑 𐑢𐑵𐑯𐑛𐑩𐑛 𐑸𐑥,” 𐑒𐑮𐑲𐑛 𐑒𐑮𐑵𐑩𐑤 ·𐑓𐑤𐑭𐑕𐑒, 𐑐𐑶𐑯𐑑𐑦𐑙 𐑑 𐑞 𐑢𐑱𐑤-𐑤𐑲𐑯 𐑯𐑽 𐑣𐑦𐑥.

“𐑥𐑲𐑯𐑛 𐑣𐑰 𐑛𐑴𐑯𐑑 𐑕𐑤𐑦𐑙 𐑞𐑰 𐑢𐑦𐑞 𐑦𐑑,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒. “𐑜𐑦𐑝 𐑢𐑱, 𐑹 𐑞 ·𐑡𐑻𐑥𐑩𐑯 𐑢𐑦𐑤 𐑣𐑨𐑝 𐑣𐑦𐑥.”

𐑢𐑦𐑞 𐑢𐑳𐑯 𐑦𐑯𐑑𐑧𐑯𐑑 𐑷𐑤 𐑞 𐑒𐑩𐑥𐑚𐑲𐑯𐑛 𐑮𐑲𐑝𐑩𐑤 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑐𐑶𐑯𐑑𐑩𐑛 𐑓 𐑞𐑦𐑕 𐑢𐑳𐑯 𐑓𐑦𐑖, 𐑚𐑦𐑒𐑪𐑟 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑢𐑪𐑟 𐑣𐑰 𐑞 𐑤𐑸𐑡𐑩𐑕𐑑, 𐑯 𐑞𐑺𐑓𐑹 𐑞 𐑥𐑴𐑕𐑑 𐑝𐑨𐑤𐑿𐑩𐑚𐑩𐑤 𐑢𐑱𐑤, 𐑚𐑳𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑯𐑽𐑩𐑕𐑑 𐑑 𐑞𐑧𐑥, 𐑯 𐑞 𐑳𐑞𐑼 𐑢𐑱𐑤𐑟 𐑢𐑻 𐑜𐑴𐑦𐑙 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑜𐑮𐑱𐑑 𐑝𐑦𐑤𐑪𐑕𐑦𐑑𐑦, 𐑥𐑹𐑴𐑝𐑼, 𐑨𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑑 𐑛𐑦𐑓𐑲 𐑐𐑼𐑕𐑿𐑑 𐑓 𐑞 𐑑𐑲𐑥. 𐑨𐑑 𐑞𐑦𐑕 𐑡𐑳𐑙𐑒𐑗𐑼 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑒𐑰𐑤𐑟 𐑣𐑨𐑛 𐑖𐑪𐑑 𐑚𐑲 𐑞 𐑔𐑮𐑰 ·𐑡𐑻𐑥𐑩𐑯 𐑚𐑴𐑑𐑕 𐑤𐑭𐑕𐑑 𐑤𐑴𐑼𐑛; 𐑚𐑳𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑜𐑮𐑱𐑑 𐑕𐑑𐑸𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑣𐑨𐑛, ·𐑛𐑧𐑮𐑦𐑒'𐑕 𐑚𐑴𐑑 𐑕𐑑𐑦𐑤 𐑤𐑧𐑛 𐑞 𐑗𐑱𐑕, 𐑞𐑴 𐑧𐑝𐑮𐑦 𐑥𐑴𐑥𐑧𐑯𐑑 𐑯𐑽𐑛 𐑚𐑲 𐑣𐑦𐑟 𐑓𐑪𐑮𐑦𐑯 𐑮𐑲𐑝𐑩𐑤𐑟. 𐑞 𐑴𐑯𐑤𐑦 𐑔𐑦𐑙 𐑞𐑱 𐑓𐑽𐑛, 𐑢𐑪𐑟, 𐑞𐑨𐑑 𐑓𐑮𐑪𐑥 𐑚𐑰𐑦𐑙 𐑷𐑤𐑮𐑧𐑛𐑦 𐑕𐑴 𐑯𐑲 𐑑 𐑣𐑦𐑟 𐑥𐑸𐑒, 𐑣𐑰 𐑢𐑫𐑛 𐑚𐑰 𐑦𐑯𐑱𐑚𐑩𐑤𐑛 𐑑 𐑛𐑸𐑑 𐑣𐑦𐑟 𐑲𐑼𐑯 𐑚𐑦𐑓𐑹 𐑞𐑱 𐑒𐑫𐑛 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑴𐑝𐑼𐑑𐑱𐑒 𐑯 𐑐𐑭𐑕 𐑣𐑦𐑥. 𐑨𐑟 𐑓 ·𐑛𐑧𐑮𐑦𐑒, 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑒𐑢𐑲𐑑 𐑒𐑪𐑯𐑓𐑦𐑛𐑩𐑯𐑑 𐑞𐑨𐑑 𐑞𐑦𐑕 𐑢𐑫𐑛 𐑚𐑰 𐑞 𐑒𐑱𐑕, 𐑯 𐑩𐑒𐑱𐑠𐑩𐑯𐑩𐑤𐑦 𐑢𐑦𐑞 𐑩 𐑛𐑦𐑮𐑲𐑛𐑦𐑙 𐑡𐑧𐑕𐑗𐑼 𐑖𐑫𐑒 𐑣𐑦𐑟 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼 𐑨𐑑 𐑞 𐑳𐑞𐑼 𐑚𐑴𐑑𐑕.

“𐑞 𐑳𐑯𐑜𐑮𐑱𐑖𐑩𐑕 𐑯 𐑳𐑯𐑜𐑮𐑱𐑑𐑓𐑩𐑤 𐑛𐑪𐑜!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒; “𐑣𐑰 𐑥𐑪𐑒𐑕 𐑯 𐑛𐑺𐑟 𐑥𐑰 𐑢𐑦𐑞 𐑞 𐑝𐑧𐑮𐑦 𐑐𐑫𐑼-𐑚𐑪𐑒𐑕 𐑲 𐑓𐑦𐑤𐑛 𐑓 𐑣𐑦𐑥 𐑯𐑪𐑑 𐑓𐑲𐑝 𐑥𐑦𐑯𐑦𐑑𐑕 𐑩𐑜𐑴!”—𐑞𐑧𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑦𐑯𐑑𐑧𐑯𐑕 𐑢𐑦𐑕𐑐𐑼—“𐑜𐑦𐑝 𐑢𐑱, 𐑜𐑮𐑱𐑣𐑬𐑯𐑛𐑟! 𐑛𐑪𐑜 𐑑 𐑦𐑑!”

“𐑲 𐑑𐑧𐑤 𐑘𐑰 𐑢𐑪𐑑 𐑦𐑑 𐑦𐑟, 𐑥𐑧𐑯”—𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚 𐑑 𐑣𐑦𐑟 𐑒𐑮𐑵—“𐑦𐑑'𐑕 𐑩𐑜𐑱𐑯𐑕𐑑 𐑥𐑲 𐑮𐑦𐑤𐑦𐑡𐑩𐑯 𐑑 𐑜𐑧𐑑 𐑥𐑨𐑛; 𐑚𐑳𐑑 𐑲'𐑛 𐑤𐑲𐑒 𐑑 𐑰𐑑 𐑞𐑨𐑑 𐑝𐑦𐑤𐑩𐑯𐑩𐑕 ·𐑘𐑸𐑥𐑩𐑯—·𐑐𐑫𐑤—𐑢𐑴𐑯𐑑 𐑘𐑰? 𐑸 𐑘𐑰 𐑜𐑴𐑦𐑙 𐑑 𐑤𐑧𐑑 𐑞𐑨𐑑 𐑮𐑭𐑕𐑒𐑩𐑤 𐑚𐑰𐑑 𐑘𐑰? 𐑛𐑵 𐑘𐑰 𐑤𐑳𐑝 𐑚𐑮𐑨𐑯𐑛𐑦? 𐑩 𐑣𐑪𐑜𐑟𐑣𐑧𐑛 𐑝 𐑚𐑮𐑨𐑯𐑛𐑦, 𐑞𐑧𐑯, 𐑑 𐑞 𐑚𐑧𐑕𐑑 𐑥𐑨𐑯. 𐑒𐑳𐑥, 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑕𐑳𐑥 𐑝 𐑘𐑰 𐑚𐑻𐑕𐑑 𐑩 𐑚𐑤𐑳𐑛-𐑝𐑧𐑕𐑩𐑤? 𐑣𐑵'𐑟 𐑞𐑨𐑑 𐑚𐑰𐑯 𐑛𐑮𐑪𐑐𐑦𐑙 𐑩𐑯 𐑨𐑙𐑒𐑼 𐑴𐑝𐑼𐑚𐑹𐑛—𐑢𐑰 𐑛𐑴𐑯𐑑 𐑚𐑳𐑡 𐑩𐑯 𐑦𐑯𐑗—𐑢𐑰'𐑼 𐑚𐑦𐑒𐑭𐑥𐑛. 𐑣𐑩𐑤𐑵, 𐑣𐑽'𐑟 𐑜𐑮𐑭𐑕 𐑜𐑮𐑴𐑦𐑙 𐑦𐑯 𐑞 𐑚𐑴𐑑'𐑕 𐑚𐑪𐑑𐑩𐑥—𐑯 𐑚𐑲 𐑞 ·𐑤𐑹𐑛, 𐑞 𐑥𐑭𐑕𐑑 𐑞𐑺'𐑟 𐑚𐑳𐑛𐑦𐑙. 𐑞𐑦𐑕 𐑢𐑴𐑯𐑑 𐑛𐑵, 𐑚𐑶𐑟. 𐑤𐑫𐑒 𐑨𐑑 𐑞𐑨𐑑 ·𐑘𐑸𐑥𐑩𐑯! 𐑞 𐑖𐑹𐑑 𐑯 𐑤𐑪𐑙 𐑝 𐑦𐑑 𐑦𐑟, 𐑥𐑧𐑯, 𐑢𐑦𐑤 𐑘𐑰 𐑕𐑐𐑦𐑑 𐑓𐑲𐑼 𐑹 𐑯𐑪𐑑?”

“𐑴! 𐑕𐑰 𐑞 𐑕𐑳𐑛𐑟 𐑣𐑰 𐑥𐑱𐑒𐑕!” 𐑒𐑮𐑲𐑛 ·𐑓𐑤𐑭𐑕𐑒, 𐑛𐑭𐑯𐑕𐑦𐑙 𐑳𐑐 𐑯 𐑛𐑬𐑯—“𐑢𐑪𐑑 𐑩 𐑣𐑳𐑥𐑐—𐑴, 𐑛𐑵 𐑐𐑲𐑤 𐑪𐑯 𐑞 𐑚𐑰𐑓—𐑤𐑱𐑟 𐑤𐑲𐑒 𐑩 𐑤𐑪𐑜! 𐑴! 𐑥𐑲 𐑤𐑨𐑛𐑟, 𐑛𐑵 𐑕𐑐𐑮𐑦𐑙—𐑕𐑤𐑨𐑐-𐑡𐑨𐑒𐑕 𐑯 𐑒𐑢𐑭𐑣𐑪𐑜𐑟 𐑓 𐑕𐑳𐑐𐑼, 𐑿 𐑯𐑴, 𐑥𐑲 𐑤𐑨𐑛𐑟—𐑚𐑱𐑒𐑑 𐑒𐑤𐑨𐑥𐑟 𐑯 𐑥𐑳𐑓𐑦𐑯𐑟—𐑴, 𐑛𐑵, 𐑛𐑵, 𐑕𐑐𐑮𐑦𐑙,—𐑣𐑰'𐑟 𐑩 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑚𐑨𐑮𐑩𐑤𐑼—𐑛𐑴𐑯𐑑 𐑤𐑵𐑟 𐑣𐑦𐑥 𐑯𐑬—𐑛𐑴𐑯𐑑 𐑴, 𐑛𐑴𐑯𐑑!—𐑕𐑰 𐑞𐑨𐑑 ·𐑘𐑸𐑥𐑩𐑯—𐑴, 𐑢𐑴𐑯𐑑 𐑘𐑰 𐑐𐑫𐑤 𐑓 𐑘𐑹 𐑛𐑳𐑓, 𐑥𐑲 𐑤𐑨𐑛𐑟—𐑕𐑳𐑗 𐑩 𐑕𐑪𐑜! 𐑕𐑳𐑗 𐑩 𐑕𐑪𐑜𐑼! 𐑛𐑴𐑯𐑑 𐑘𐑰 𐑤𐑳𐑝 𐑕𐑐𐑻𐑥? 𐑞𐑺 𐑜𐑴𐑟 𐑔𐑮𐑰 𐑔𐑬𐑟𐑩𐑯𐑛 𐑛𐑪𐑤𐑼𐑟, 𐑥𐑧𐑯!—𐑩 𐑚𐑨𐑙𐑒!—𐑩 𐑣𐑴𐑤 𐑚𐑨𐑙𐑒! 𐑞 𐑚𐑨𐑙𐑒 𐑝 ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛!—𐑴, 𐑛𐑵, 𐑛𐑵, 𐑛𐑵!—𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 ·𐑘𐑸𐑥𐑩𐑯 𐑩𐑚𐑬𐑑 𐑯𐑬?”

𐑨𐑑 𐑞𐑦𐑕 𐑥𐑴𐑥𐑧𐑯𐑑 ·𐑛𐑧𐑮𐑦𐑒 𐑢𐑪𐑟 𐑦𐑯 𐑞 𐑨𐑒𐑑 𐑝 𐑐𐑦𐑗𐑦𐑙 𐑣𐑦𐑟 𐑤𐑨𐑥𐑐-𐑓𐑰𐑛𐑼 𐑨𐑑 𐑞 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑚𐑴𐑑𐑕, 𐑯 𐑷𐑤𐑕𐑴 𐑣𐑦𐑟 𐑶𐑤-𐑒𐑨𐑯; 𐑐𐑼𐑣𐑨𐑐𐑕 𐑢𐑦𐑞 𐑞 𐑛𐑳𐑚𐑩𐑤 𐑝𐑿 𐑝 𐑮𐑦𐑑𐑸𐑛𐑦𐑙 𐑣𐑦𐑟 𐑮𐑲𐑝𐑩𐑤𐑟' 𐑢𐑱, 𐑯 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥 𐑰𐑒𐑩𐑯𐑪𐑥𐑦𐑒𐑤𐑦 𐑩𐑒𐑕𐑧𐑤𐑼𐑱𐑑𐑦𐑙 𐑣𐑦𐑟 𐑴𐑯 𐑚𐑲 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑𐑼𐑦 𐑦𐑥𐑐𐑦𐑑𐑩𐑕 𐑝 𐑞 𐑚𐑨𐑒𐑢𐑼𐑛 𐑑𐑪𐑕.

“𐑞 𐑳𐑯𐑥𐑨𐑯𐑼𐑤𐑦 ·𐑛𐑳𐑗 𐑛𐑪𐑜𐑼!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚. “𐑐𐑫𐑤 𐑯𐑬, 𐑥𐑧𐑯, 𐑤𐑲𐑒 𐑓𐑦𐑓𐑑𐑦 𐑔𐑬𐑟𐑩𐑯𐑛 𐑤𐑲𐑯-𐑝-𐑚𐑨𐑑𐑩𐑤-𐑖𐑦𐑐 𐑤𐑴𐑛𐑟 𐑝 𐑮𐑧𐑛-𐑣𐑺𐑛 𐑛𐑧𐑝𐑩𐑤𐑟. 𐑢𐑪𐑑 𐑛'𐑘𐑰 𐑕𐑱, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴; 𐑸 𐑿 𐑞 𐑥𐑨𐑯 𐑑 𐑕𐑯𐑨𐑐 𐑘𐑹 𐑕𐑐𐑲𐑯 𐑦𐑯 𐑑𐑵-𐑯-𐑑𐑢𐑧𐑯𐑑𐑦 𐑐𐑰𐑕𐑩𐑟 𐑓 𐑞 𐑪𐑯𐑼 𐑝 𐑴𐑤𐑛 ·𐑜𐑱𐑣𐑧𐑛? 𐑢𐑪𐑑 𐑛'𐑘𐑰 𐑕𐑱?”

“𐑲 𐑕𐑱, 𐑐𐑫𐑤 𐑤𐑲𐑒 𐑜𐑪𐑛-𐑛𐑨𐑥,”—𐑒𐑮𐑲𐑛 𐑞 ·𐑦𐑯𐑛𐑾𐑯.

𐑓𐑽𐑕𐑤𐑦, 𐑚𐑳𐑑 𐑰𐑝𐑩𐑯𐑤𐑦 𐑦𐑯𐑕𐑲𐑑𐑩𐑛 𐑚𐑲 𐑞 𐑑𐑷𐑯𐑑𐑕 𐑝 𐑞 ·𐑡𐑻𐑥𐑩𐑯, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕 𐑯𐑬 𐑚𐑦𐑜𐑨𐑯 𐑮𐑱𐑯𐑡𐑦𐑙 𐑷𐑤𐑥𐑴𐑕𐑑 𐑩𐑚𐑮𐑧𐑕𐑑; 𐑯, 𐑕𐑴 𐑛𐑦𐑕𐑐𐑴𐑟𐑛, 𐑥𐑴𐑥𐑧𐑯𐑑𐑺𐑦𐑤𐑦 𐑯𐑽𐑛 𐑣𐑦𐑥. 𐑦𐑯 𐑞𐑨𐑑 𐑓𐑲𐑯, 𐑤𐑵𐑕, 𐑖𐑦𐑝𐑩𐑤𐑮𐑩𐑕 𐑨𐑑𐑦𐑑𐑿𐑛 𐑝 𐑞 𐑣𐑧𐑛𐑟𐑥𐑩𐑯 𐑢𐑧𐑯 𐑛𐑮𐑷𐑦𐑙 𐑯𐑽 𐑑 𐑣𐑦𐑟 𐑐𐑮𐑱, 𐑞 𐑔𐑮𐑰 𐑥𐑱𐑑𐑕 𐑕𐑑𐑫𐑛 𐑳𐑐 𐑐𐑮𐑬𐑛𐑤𐑦, 𐑩𐑒𐑱𐑠𐑩𐑯𐑩𐑤𐑦 𐑚𐑨𐑒𐑦𐑙 𐑞 𐑭𐑓𐑑𐑼 𐑹𐑟𐑥𐑩𐑯 𐑢𐑦𐑞 𐑩𐑯 𐑦𐑜𐑟𐑦𐑤𐑼𐑱𐑑𐑦𐑙 𐑒𐑮𐑲 𐑝, “𐑞𐑺 𐑖𐑰 𐑕𐑤𐑲𐑛𐑟, 𐑯𐑬! 𐑣𐑼𐑭 𐑓 𐑞 𐑢𐑲𐑑-𐑨𐑖 𐑚𐑮𐑰𐑟! 𐑛𐑬𐑯 𐑢𐑦𐑞 𐑞 ·𐑘𐑸𐑥𐑩𐑯! 𐑕𐑱𐑤 𐑴𐑝𐑼 𐑣𐑦𐑥!”

𐑚𐑳𐑑 𐑕𐑴 𐑛𐑦𐑕𐑲𐑛𐑩𐑛 𐑩𐑯 𐑼𐑦𐑡𐑦𐑯𐑩𐑤 𐑕𐑑𐑸𐑑 𐑣𐑨𐑛 ·𐑛𐑧𐑮𐑦𐑒 𐑣𐑨𐑛, 𐑞𐑨𐑑 𐑕𐑐𐑲𐑑 𐑝 𐑷𐑤 𐑞𐑺 𐑜𐑨𐑤𐑩𐑯𐑑𐑮𐑦, 𐑣𐑰 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑐𐑮𐑵𐑝𐑛 𐑞 𐑝𐑦𐑒𐑑𐑼 𐑦𐑯 𐑞𐑦𐑕 𐑮𐑱𐑕, 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑩 𐑮𐑲𐑗𐑩𐑕 𐑡𐑳𐑡𐑥𐑩𐑯𐑑 𐑛𐑦𐑕𐑧𐑯𐑛𐑩𐑛 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥 𐑦𐑯 𐑩 𐑒𐑮𐑨𐑚 𐑢𐑦𐑗 𐑒𐑷𐑑 𐑞 𐑚𐑤𐑱𐑛 𐑝 𐑣𐑦𐑟 𐑥𐑦𐑛𐑖𐑦𐑐 𐑹𐑟𐑥𐑩𐑯. 𐑢𐑲𐑤 𐑞𐑦𐑕 𐑒𐑤𐑳𐑥𐑟𐑦 𐑤𐑳𐑚𐑼 𐑢𐑪𐑟 𐑕𐑑𐑮𐑲𐑝𐑦𐑙 𐑑 𐑓𐑮𐑰 𐑣𐑦𐑟 𐑢𐑲𐑑-𐑨𐑖, 𐑯 𐑢𐑲𐑤, 𐑦𐑯 𐑒𐑪𐑯𐑕𐑦𐑒𐑢𐑩𐑯𐑕, ·𐑛𐑧𐑮𐑦𐑒'𐑕 𐑚𐑴𐑑 𐑢𐑪𐑟 𐑯𐑲 𐑑 𐑒𐑨𐑐𐑕𐑲𐑟𐑦𐑙, 𐑯 𐑣𐑰 𐑔𐑳𐑯𐑛𐑼𐑦𐑙 𐑩𐑢𐑱 𐑨𐑑 𐑣𐑦𐑟 𐑥𐑧𐑯 𐑦𐑯 𐑩 𐑥𐑲𐑑𐑦 𐑮𐑱𐑡;—𐑞𐑨𐑑 𐑢𐑪𐑟 𐑩 𐑜𐑫𐑛 𐑑𐑲𐑥 𐑓 ·𐑕𐑑𐑸𐑚𐑳𐑒, ·𐑕𐑑𐑳𐑚, 𐑯 ·𐑓𐑤𐑭𐑕𐑒. 𐑢𐑦𐑞 𐑩 𐑖𐑬𐑑, 𐑞𐑱 𐑑𐑫𐑒 𐑩 𐑥𐑹𐑑𐑩𐑤 𐑕𐑑𐑸𐑑 𐑓𐑹𐑢𐑼𐑛𐑟, 𐑯 𐑕𐑤𐑭𐑯𐑑𐑦𐑙𐑤𐑦 𐑮𐑱𐑯𐑡𐑛 𐑳𐑐 𐑪𐑯 𐑞 ·𐑡𐑻𐑥𐑩𐑯'𐑟 𐑒𐑢𐑹𐑑𐑼. 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑥𐑹, 𐑯 𐑷𐑤 𐑓𐑹 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑛𐑲𐑨𐑜𐑩𐑯𐑦𐑒𐑤𐑦 𐑦𐑯 𐑞 𐑢𐑱𐑤'𐑟 𐑦𐑥𐑰𐑛𐑾𐑑 𐑢𐑱𐑒, 𐑢𐑲𐑤 𐑕𐑑𐑮𐑧𐑗𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞𐑧𐑥, 𐑪𐑯 𐑚𐑴𐑔 𐑕𐑲𐑛𐑟, 𐑢𐑪𐑟 𐑞 𐑓𐑴𐑥𐑦𐑙 𐑕𐑢𐑧𐑤 𐑞𐑨𐑑 𐑣𐑰 𐑥𐑱𐑛.

𐑦𐑯 𐑞𐑨𐑑 𐑕𐑤𐑴𐑐𐑦𐑙 𐑭𐑓𐑑𐑼𐑯𐑵𐑯 𐑕𐑳𐑯𐑤𐑲𐑑, 𐑞 𐑖𐑨𐑛𐑴𐑟 𐑞𐑨𐑑 𐑞 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕 𐑕𐑧𐑯𐑑 𐑛𐑬𐑯 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑕𐑻𐑓𐑦𐑕, 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑤𐑪𐑙 𐑦𐑯𐑳𐑓 𐑯 𐑚𐑮𐑷𐑛 𐑦𐑯𐑳𐑓 𐑑 𐑖𐑱𐑛 𐑣𐑭𐑓 ·𐑟𐑻𐑒𐑕𐑰𐑟' 𐑸𐑥𐑦. 𐑣𐑵 𐑒𐑨𐑯 𐑑𐑧𐑤 𐑣𐑬 𐑩𐑐𐑷𐑤𐑦𐑙 𐑑 𐑞 𐑢𐑵𐑯𐑛𐑩𐑛 𐑢𐑱𐑤 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑕𐑳𐑗 𐑣𐑿𐑡 𐑓𐑨𐑯𐑑𐑩𐑥𐑟 𐑓𐑤𐑦𐑑𐑦𐑙 𐑴𐑝𐑼 𐑣𐑦𐑟 𐑣𐑧𐑛!

“𐑕𐑑𐑨𐑯𐑛 𐑚𐑲, 𐑥𐑧𐑯; 𐑣𐑰 𐑕𐑑𐑻𐑟,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑨𐑟 𐑞 𐑔𐑮𐑰 𐑤𐑲𐑯𐑟 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑝𐑲𐑚𐑮𐑱𐑑𐑩𐑛 𐑦𐑯 𐑞 𐑢𐑷𐑑𐑼, 𐑛𐑦𐑕𐑑𐑦𐑙𐑒𐑑𐑤𐑦 𐑒𐑪𐑯𐑛𐑳𐑒𐑑𐑦𐑙 𐑳𐑐𐑢𐑼𐑛𐑟 𐑑 𐑞𐑧𐑥, 𐑨𐑟 𐑚𐑲 𐑥𐑨𐑜𐑯𐑧𐑑𐑦𐑒 𐑢𐑲𐑼𐑟, 𐑞 𐑤𐑲𐑓 𐑯 𐑛𐑧𐑔 𐑔𐑮𐑪𐑚𐑟 𐑝 𐑞 𐑢𐑱𐑤, 𐑕𐑴 𐑞𐑨𐑑 𐑧𐑝𐑮𐑦 𐑹𐑟𐑥𐑩𐑯 𐑓𐑧𐑤𐑑 𐑞𐑧𐑥 𐑦𐑯 𐑣𐑦𐑟 𐑕𐑰𐑑. 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑮𐑦𐑤𐑰𐑝𐑛 𐑦𐑯 𐑜𐑮𐑱𐑑 𐑐𐑸𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑛𐑬𐑯𐑢𐑼𐑛 𐑕𐑑𐑮𐑱𐑯 𐑨𐑑 𐑞 𐑚𐑬𐑟, 𐑞 𐑚𐑴𐑑𐑕 𐑜𐑱𐑝 𐑩 𐑕𐑳𐑛𐑩𐑯 𐑚𐑬𐑯𐑕 𐑳𐑐𐑢𐑼𐑛𐑟, 𐑨𐑟 𐑩 𐑕𐑥𐑷𐑤 𐑲𐑕𐑓𐑰𐑤𐑛 𐑢𐑦𐑤, 𐑢𐑧𐑯 𐑩 𐑛𐑧𐑯𐑕 𐑣𐑻𐑛 𐑝 𐑢𐑲𐑑 𐑚𐑺𐑟 𐑸 𐑕𐑒𐑺𐑛 𐑓𐑮𐑪𐑥 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰.

“𐑣𐑷𐑤 𐑦𐑯! 𐑣𐑷𐑤 𐑦𐑯!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑩𐑜𐑱𐑯; “𐑣𐑰'𐑟 𐑮𐑲𐑟𐑦𐑙.”

𐑨𐑟 𐑞 𐑚𐑴𐑑𐑕 𐑯𐑬 𐑥𐑹 𐑒𐑤𐑴𐑕𐑤𐑦 𐑕𐑼𐑬𐑯𐑛𐑩𐑛 𐑣𐑦𐑥, 𐑞 𐑣𐑴𐑤 𐑳𐑐𐑼 𐑐𐑸𐑑 𐑝 𐑣𐑦𐑟 𐑓𐑹𐑥, 𐑢𐑦𐑞 𐑥𐑳𐑗 𐑝 𐑦𐑑 𐑞𐑨𐑑 𐑦𐑟 𐑹𐑛𐑦𐑯𐑺𐑦𐑤𐑦 𐑕𐑩𐑚𐑥𐑻𐑡𐑛, 𐑢𐑪𐑟 𐑐𐑤𐑱𐑯𐑤𐑦 𐑮𐑦𐑝𐑰𐑤𐑛. 𐑣𐑦𐑟 𐑲𐑟, 𐑹 𐑮𐑭𐑞𐑼 𐑞 𐑐𐑤𐑱𐑕𐑩𐑟 𐑢𐑺 𐑣𐑦𐑟 𐑲𐑟 𐑣𐑨𐑛 𐑚𐑰𐑯, 𐑢𐑻 𐑚𐑦𐑣𐑧𐑤𐑛. 𐑨𐑟 𐑕𐑑𐑮𐑱𐑯𐑡 𐑥𐑦𐑕𐑜𐑮𐑴𐑯 𐑥𐑨𐑕𐑩𐑟 𐑜𐑨𐑞𐑼 𐑦𐑯 𐑞 𐑯𐑪𐑑-𐑣𐑴𐑤𐑟 𐑝 𐑞 𐑯𐑴𐑚𐑩𐑤𐑩𐑕𐑑 𐑴𐑒𐑕 𐑢𐑧𐑯 𐑐𐑮𐑪𐑕𐑑𐑮𐑱𐑑, 𐑕𐑴 𐑓𐑮𐑪𐑥 𐑞 𐑐𐑶𐑯𐑑𐑕 𐑢𐑦𐑗 𐑞 𐑢𐑱𐑤'𐑟 𐑲𐑟 𐑣𐑨𐑛 𐑢𐑳𐑯𐑕 𐑪𐑒𐑿𐑐𐑲𐑛, 𐑯𐑬 𐑐𐑮𐑴𐑑𐑮𐑵𐑛𐑩𐑛 𐑚𐑤𐑲𐑯𐑛 𐑚𐑳𐑤𐑚𐑟, 𐑣𐑪𐑮𐑦𐑚𐑤𐑦 𐑐𐑦𐑑𐑾𐑚𐑩𐑤 𐑑 𐑕𐑰. 𐑚𐑳𐑑 𐑐𐑦𐑑𐑦 𐑞𐑺 𐑢𐑪𐑟 𐑯𐑳𐑯. 𐑓 𐑷𐑤 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑱𐑡, 𐑯 𐑣𐑦𐑟 𐑢𐑳𐑯 𐑸𐑥, 𐑯 𐑣𐑦𐑟 𐑚𐑤𐑲𐑯𐑛 𐑲𐑟, 𐑣𐑰 𐑥𐑳𐑕𐑑 𐑛𐑲 𐑞 𐑛𐑧𐑔 𐑯 𐑚𐑰 𐑥𐑻𐑛𐑼𐑛, 𐑦𐑯 𐑹𐑛𐑼 𐑑 𐑤𐑲𐑑 𐑞 𐑜𐑱 𐑚𐑮𐑲𐑛𐑩𐑤𐑟 𐑯 𐑳𐑞𐑼 𐑥𐑧𐑮𐑦-𐑥𐑱𐑒𐑦𐑙𐑟 𐑝 𐑥𐑧𐑯, 𐑯 𐑷𐑤𐑕𐑴 𐑑 𐑦𐑤𐑵𐑥𐑦𐑯𐑱𐑑 𐑞 𐑕𐑪𐑤𐑩𐑥 𐑗𐑻𐑗𐑩𐑟 𐑞𐑨𐑑 𐑐𐑮𐑰𐑗 𐑳𐑯𐑒𐑩𐑯𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑦𐑯𐑩𐑓𐑧𐑯𐑕𐑦𐑝𐑯𐑩𐑕 𐑚𐑲 𐑷𐑤 𐑑 𐑷𐑤. 𐑕𐑑𐑦𐑤 𐑮𐑴𐑤𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑤𐑳𐑛, 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑣𐑰 𐑐𐑸𐑖𐑩𐑤𐑦 𐑛𐑦𐑕𐑒𐑤𐑴𐑟𐑛 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑛𐑦𐑕𐑒𐑳𐑤𐑼𐑛 𐑚𐑳𐑯𐑗 𐑹 𐑐𐑮𐑴𐑑𐑿𐑚𐑼𐑩𐑯𐑕, 𐑞 𐑕𐑲𐑟 𐑝 𐑩 𐑚𐑫𐑖𐑩𐑤, 𐑤𐑴 𐑛𐑬𐑯 𐑪𐑯 𐑞 𐑓𐑤𐑨𐑙𐑒.
