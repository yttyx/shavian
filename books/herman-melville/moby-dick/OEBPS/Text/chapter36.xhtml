·𐑗𐑨𐑐𐑑𐑼 36. 𐑞 ·𐑒𐑢𐑹𐑑𐑼-·𐑛𐑧𐑒.

(𐑧𐑯𐑑𐑼 ·𐑱𐑣𐑨𐑚: 𐑞𐑧𐑯, 𐑷𐑤.)

𐑦𐑑 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑩 𐑜𐑮𐑱𐑑 𐑢𐑲𐑤 𐑭𐑓𐑑𐑼 𐑞 𐑩𐑓𐑺 𐑝 𐑞 𐑐𐑲𐑐, 𐑞𐑨𐑑 𐑢𐑳𐑯 𐑥𐑹𐑯𐑦𐑙 𐑖𐑹𐑑𐑤𐑦 𐑭𐑓𐑑𐑼 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑, ·𐑱𐑣𐑨𐑚, 𐑨𐑟 𐑢𐑪𐑟 𐑣𐑦𐑟 𐑢𐑴𐑯𐑑, 𐑩𐑕𐑧𐑯𐑛𐑩𐑛 𐑞 𐑒𐑨𐑚𐑦𐑯-𐑜𐑨𐑙𐑢𐑱 𐑑 𐑞 𐑛𐑧𐑒. 𐑞𐑺 𐑥𐑴𐑕𐑑 𐑕𐑰-𐑒𐑨𐑐𐑑𐑦𐑯𐑟 𐑿𐑠𐑵𐑩𐑤𐑦 𐑢𐑷𐑒 𐑨𐑑 𐑞𐑨𐑑 𐑬𐑼, 𐑨𐑟 𐑒𐑳𐑯𐑑𐑮𐑦 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑧𐑯, 𐑭𐑓𐑑𐑼 𐑞 𐑕𐑱𐑥 𐑥𐑰𐑤, 𐑑𐑱𐑒 𐑩 𐑓𐑿 𐑑𐑻𐑯𐑟 𐑦𐑯 𐑞 𐑜𐑸𐑛𐑩𐑯.

𐑕𐑵𐑯 𐑣𐑦𐑟 𐑕𐑑𐑧𐑛𐑦, 𐑲𐑝𐑼𐑦 𐑕𐑑𐑮𐑲𐑛 𐑢𐑪𐑟 𐑣𐑻𐑛, 𐑨𐑟 𐑑 𐑯 𐑓𐑮𐑴 𐑣𐑰 𐑐𐑱𐑕𐑑 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑮𐑬𐑯𐑛𐑟, 𐑩𐑐𐑪𐑯 𐑐𐑤𐑨𐑙𐑒𐑕 𐑕𐑴 𐑓𐑩𐑥𐑦𐑤𐑽 𐑑 𐑣𐑦𐑟 𐑑𐑮𐑧𐑛, 𐑞𐑨𐑑 𐑞𐑱 𐑢𐑻 𐑷𐑤 𐑴𐑝𐑼 𐑛𐑧𐑯𐑑𐑩𐑛, 𐑤𐑲𐑒 𐑡𐑾𐑤𐑪𐑡𐑦𐑒𐑩𐑤 𐑕𐑑𐑴𐑯𐑟, 𐑢𐑦𐑞 𐑞 𐑐𐑦𐑒𐑿𐑤𐑽 𐑥𐑸𐑒 𐑝 𐑣𐑦𐑟 𐑢𐑷𐑒. 𐑛𐑦𐑛 𐑿 𐑓𐑦𐑒𐑕𐑩𐑛𐑤𐑦 𐑜𐑱𐑟, 𐑑𐑵, 𐑩𐑐𐑪𐑯 𐑞𐑨𐑑 𐑮𐑦𐑚𐑛 𐑯 𐑛𐑧𐑯𐑑𐑩𐑛 𐑚𐑮𐑬; 𐑞𐑺 𐑷𐑤𐑕𐑴, 𐑿 𐑢𐑫𐑛 𐑕𐑰 𐑕𐑑𐑦𐑤 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑓𐑫𐑑-𐑐𐑮𐑦𐑯𐑑𐑕—𐑞 𐑓𐑫𐑑-𐑐𐑮𐑦𐑯𐑑𐑕 𐑝 𐑣𐑦𐑟 𐑢𐑳𐑯 𐑳𐑯𐑕𐑤𐑰𐑐𐑦𐑙, 𐑧𐑝𐑼-𐑐𐑱𐑕𐑦𐑙 𐑔𐑷𐑑.

𐑚𐑳𐑑 𐑪𐑯 𐑞 𐑩𐑒𐑱𐑠𐑩𐑯 𐑦𐑯 𐑒𐑢𐑧𐑕𐑗𐑩𐑯, 𐑞𐑴𐑟 𐑛𐑧𐑯𐑑𐑕 𐑤𐑫𐑒𐑑 𐑛𐑰𐑐𐑼, 𐑰𐑝𐑩𐑯 𐑨𐑟 𐑣𐑦𐑟 𐑯𐑻𐑝𐑩𐑕 𐑕𐑑𐑧𐑐 𐑞𐑨𐑑 𐑥𐑹𐑯𐑦𐑙 𐑤𐑧𐑓𐑑 𐑩 𐑛𐑰𐑐𐑼 𐑥𐑸𐑒. 𐑯, 𐑕𐑴 𐑓𐑫𐑤 𐑝 𐑣𐑦𐑟 𐑔𐑷𐑑 𐑢𐑪𐑟 ·𐑱𐑣𐑨𐑚, 𐑞𐑨𐑑 𐑨𐑑 𐑧𐑝𐑮𐑦 𐑿𐑯𐑦𐑓𐑹𐑥 𐑑𐑻𐑯 𐑞𐑨𐑑 𐑣𐑰 𐑥𐑱𐑛, 𐑯𐑬 𐑨𐑑 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑 𐑯 𐑯𐑬 𐑨𐑑 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤, 𐑿 𐑒𐑫𐑛 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰 𐑞𐑨𐑑 𐑔𐑷𐑑 𐑑𐑻𐑯 𐑦𐑯 𐑣𐑦𐑥 𐑨𐑟 𐑣𐑰 𐑑𐑻𐑯𐑛, 𐑯 𐑐𐑱𐑕 𐑦𐑯 𐑣𐑦𐑥 𐑨𐑟 𐑣𐑰 𐑐𐑱𐑕𐑑; 𐑕𐑴 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑐𐑩𐑟𐑧𐑕𐑦𐑙 𐑣𐑦𐑥, 𐑦𐑯𐑛𐑰𐑛, 𐑞𐑨𐑑 𐑦𐑑 𐑷𐑤 𐑚𐑳𐑑 𐑕𐑰𐑥𐑛 𐑞 𐑦𐑯𐑢𐑼𐑛 𐑥𐑴𐑤𐑛 𐑝 𐑧𐑝𐑮𐑦 𐑬𐑑𐑼 𐑥𐑵𐑝𐑥𐑩𐑯𐑑.

“·𐑛'𐑘𐑰 𐑥𐑸𐑒 𐑣𐑦𐑥, ·𐑓𐑤𐑭𐑕𐑒?” 𐑢𐑦𐑕𐑐𐑼𐑛 ·𐑕𐑑𐑳𐑚; “𐑞 𐑗𐑦𐑒 𐑞𐑨𐑑'𐑕 𐑦𐑯 𐑣𐑦𐑥 𐑐𐑧𐑒𐑕 𐑞 𐑖𐑧𐑤. '𐑑𐑢𐑦𐑤 𐑕𐑵𐑯 𐑚𐑰 𐑬𐑑.”

𐑞 𐑬𐑼𐑟 𐑢𐑹 𐑪𐑯;—·𐑱𐑣𐑨𐑚 𐑯𐑬 𐑖𐑳𐑑 𐑳𐑐 𐑢𐑦𐑞𐑦𐑯 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯; 𐑩𐑯𐑪𐑯, 𐑐𐑱𐑕𐑦𐑙 𐑞 𐑛𐑧𐑒, 𐑢𐑦𐑞 𐑞 𐑕𐑱𐑥 𐑦𐑯𐑑𐑧𐑯𐑕 𐑚𐑦𐑜𐑩𐑑𐑮𐑦 𐑝 𐑐𐑻𐑐𐑩𐑕 𐑦𐑯 𐑣𐑦𐑟 𐑨𐑕𐑐𐑧𐑒𐑑.

𐑦𐑑 𐑛𐑮𐑵 𐑯𐑽 𐑞 𐑒𐑤𐑴𐑟 𐑝 𐑛𐑱. 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑣𐑰 𐑒𐑱𐑥 𐑑 𐑩 𐑣𐑷𐑤𐑑 𐑚𐑲 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑯 𐑦𐑯𐑕𐑻𐑑𐑦𐑙 𐑣𐑦𐑟 𐑚𐑴𐑯 𐑤𐑧𐑜 𐑦𐑯𐑑𐑵 𐑞 𐑷𐑜𐑼-𐑣𐑴𐑤 𐑞𐑺, 𐑯 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑜𐑮𐑭𐑕𐑐𐑦𐑙 𐑩 𐑖𐑮𐑬𐑛, 𐑣𐑰 𐑹𐑛𐑼𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑑 𐑕𐑧𐑯𐑛 𐑧𐑝𐑮𐑦𐑚𐑪𐑛𐑦 𐑭𐑓𐑑.

“𐑕𐑻!” 𐑕𐑧𐑛 𐑞 𐑥𐑱𐑑, 𐑩𐑕𐑑𐑪𐑯𐑦𐑖𐑑 𐑨𐑑 𐑩𐑯 𐑹𐑛𐑼 𐑕𐑧𐑤𐑛𐑩𐑥 𐑹 𐑯𐑧𐑝𐑼 𐑜𐑦𐑝𐑩𐑯 𐑪𐑯 𐑖𐑦𐑐-𐑚𐑹𐑛 𐑦𐑒𐑕𐑧𐑐𐑑 𐑦𐑯 𐑕𐑳𐑥 𐑧𐑒𐑕𐑑𐑮𐑹𐑛𐑦𐑯𐑼𐑦 𐑒𐑱𐑕.

“𐑕𐑧𐑯𐑛 𐑧𐑝𐑮𐑦𐑚𐑪𐑛𐑦 𐑭𐑓𐑑,” 𐑮𐑦𐑐𐑰𐑑𐑩𐑛 ·𐑱𐑣𐑨𐑚. “𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟, 𐑞𐑺! 𐑒𐑳𐑥 𐑛𐑬𐑯!”

𐑢𐑧𐑯 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑢𐑻 𐑩𐑕𐑧𐑥𐑚𐑩𐑤𐑛, 𐑯 𐑢𐑦𐑞 𐑒𐑘𐑫𐑼𐑾𐑕 𐑯 𐑯𐑪𐑑 𐑣𐑴𐑤𐑦 𐑳𐑯𐑨𐑐𐑮𐑦𐑣𐑧𐑯𐑕𐑦𐑝 𐑓𐑱𐑕𐑩𐑟, 𐑢𐑻 𐑲𐑦𐑙 𐑣𐑦𐑥, 𐑓 𐑣𐑰 𐑤𐑫𐑒𐑑 𐑯𐑪𐑑 𐑳𐑯𐑤𐑲𐑒 𐑞 𐑢𐑧𐑞𐑼 𐑣𐑼𐑲𐑟𐑩𐑯 𐑢𐑧𐑯 𐑩 𐑕𐑑𐑹𐑥 𐑦𐑟 𐑒𐑳𐑥𐑦𐑙 𐑳𐑐, ·𐑱𐑣𐑨𐑚, 𐑭𐑓𐑑𐑼 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑯 𐑞𐑧𐑯 𐑛𐑸𐑑𐑦𐑙 𐑣𐑦𐑟 𐑲𐑟 𐑩𐑥𐑳𐑙 𐑞 𐑒𐑮𐑵, 𐑕𐑑𐑸𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑕𐑑𐑨𐑯𐑛𐑐𐑶𐑯𐑑; 𐑯 𐑨𐑟 𐑞𐑴 𐑯𐑪𐑑 𐑩 𐑕𐑴𐑤 𐑢𐑻 𐑯𐑲 𐑣𐑦𐑥 𐑮𐑦𐑟𐑿𐑥𐑛 𐑣𐑦𐑟 𐑣𐑧𐑝𐑦 𐑑𐑻𐑯𐑟 𐑩𐑐𐑪𐑯 𐑞 𐑛𐑧𐑒. 𐑢𐑦𐑞 𐑚𐑧𐑯𐑑 𐑣𐑧𐑛 𐑯 𐑣𐑭𐑓-𐑕𐑤𐑬𐑗𐑑 𐑣𐑨𐑑 𐑣𐑰 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑛 𐑑 𐑐𐑱𐑕, 𐑳𐑯𐑥𐑲𐑯𐑛𐑓𐑩𐑤 𐑝 𐑞 𐑢𐑳𐑯𐑛𐑼𐑦𐑙 𐑢𐑦𐑕𐑐𐑼𐑦𐑙 𐑩𐑥𐑳𐑙 𐑞 𐑥𐑧𐑯; 𐑑𐑦𐑤 ·𐑕𐑑𐑳𐑚 𐑒𐑷𐑖𐑩𐑕𐑤𐑦 𐑢𐑦𐑕𐑐𐑼𐑛 𐑑 ·𐑓𐑤𐑭𐑕𐑒, 𐑞𐑨𐑑 ·𐑱𐑣𐑨𐑚 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑕𐑳𐑥𐑩𐑯𐑛 𐑞𐑧𐑥 𐑞𐑺 𐑓 𐑞 𐑐𐑻𐑐𐑩𐑕 𐑝 𐑢𐑦𐑑𐑯𐑩𐑕𐑦𐑙 𐑩 𐑐𐑦𐑛𐑧𐑕𐑑𐑮𐑾𐑯 𐑓𐑰𐑑. 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑤𐑭𐑕𐑑 𐑤𐑪𐑙. 𐑝𐑾𐑥𐑩𐑯𐑑𐑤𐑦 𐑐𐑷𐑟𐑦𐑙, 𐑣𐑰 𐑒𐑮𐑲𐑛:—

“𐑢𐑪𐑑 𐑛𐑵 𐑘𐑰 𐑛𐑵 𐑢𐑧𐑯 𐑘𐑰 𐑕𐑰 𐑩 𐑢𐑱𐑤, 𐑥𐑧𐑯?”

“𐑕𐑦𐑙 𐑬𐑑 𐑓 𐑣𐑦𐑥!” 𐑢𐑪𐑟 𐑞 𐑦𐑥𐑐𐑳𐑤𐑕𐑦𐑝 𐑮𐑦𐑡𐑶𐑯𐑛𐑼 𐑓𐑮𐑪𐑥 𐑩 𐑕𐑒𐑹 𐑝 𐑒𐑤𐑳𐑚𐑛 𐑝𐑶𐑕𐑩𐑟.

“𐑜𐑫𐑛!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑢𐑦𐑞 𐑩 𐑢𐑲𐑤𐑛 𐑩𐑐𐑮𐑵𐑝𐑩𐑤 𐑦𐑯 𐑣𐑦𐑟 𐑑𐑴𐑯𐑟; 𐑩𐑚𐑟𐑻𐑝𐑦𐑙 𐑞 𐑣𐑸𐑑𐑦 𐑨𐑯𐑦𐑥𐑱𐑖𐑩𐑯 𐑦𐑯𐑑𐑵 𐑢𐑦𐑗 𐑣𐑦𐑟 𐑳𐑯𐑦𐑒𐑕𐑐𐑧𐑒𐑑𐑩𐑛 𐑒𐑢𐑧𐑕𐑗𐑩𐑯 𐑣𐑨𐑛 𐑕𐑴 𐑥𐑨𐑜𐑯𐑧𐑑𐑦𐑒𐑤𐑦 𐑔𐑮𐑴𐑯 𐑞𐑧𐑥.

“𐑯 𐑢𐑪𐑑 𐑛𐑵 𐑘𐑰 𐑯𐑧𐑒𐑕𐑑, 𐑥𐑧𐑯?”

“𐑤𐑴𐑼 𐑩𐑢𐑱, 𐑯 𐑭𐑓𐑑𐑼 𐑣𐑦𐑥!”

“𐑯 𐑢𐑪𐑑 𐑑𐑿𐑯 𐑦𐑟 𐑦𐑑 𐑘𐑰 𐑐𐑫𐑤 𐑑, 𐑥𐑧𐑯?”

“𐑩 𐑛𐑧𐑛 𐑢𐑱𐑤 𐑹 𐑩 𐑕𐑑𐑴𐑝 𐑚𐑴𐑑!”

𐑥𐑹 𐑯 𐑥𐑹 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑯 𐑓𐑽𐑕𐑤𐑦 𐑜𐑤𐑨𐑛 𐑯 𐑩𐑐𐑮𐑵𐑝𐑦𐑙, 𐑜𐑮𐑵 𐑞 𐑒𐑬𐑯𐑑𐑩𐑯𐑩𐑯𐑕 𐑝 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑨𐑑 𐑧𐑝𐑮𐑦 𐑖𐑬𐑑; 𐑢𐑲𐑤 𐑞 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑜𐑱𐑟 𐑒𐑘𐑫𐑼𐑾𐑕𐑤𐑦 𐑨𐑑 𐑰𐑗 𐑳𐑞𐑼, 𐑨𐑟 𐑦𐑓 𐑥𐑸𐑝𐑩𐑤𐑦𐑙 𐑣𐑬 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑞𐑱 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑚𐑦𐑒𐑱𐑥 𐑕𐑴 𐑦𐑒𐑕𐑲𐑑𐑩𐑛 𐑨𐑑 𐑕𐑳𐑗 𐑕𐑰𐑥𐑦𐑙𐑤𐑦 𐑐𐑻𐑐𐑩𐑕𐑤𐑩𐑕 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑟.

𐑚𐑳𐑑, 𐑞𐑱 𐑢𐑻 𐑷𐑤 𐑰𐑜𐑼𐑯𐑩𐑕 𐑩𐑜𐑱𐑯, 𐑨𐑟 ·𐑱𐑣𐑨𐑚, 𐑯𐑬 𐑣𐑭𐑓-𐑮𐑦𐑝𐑪𐑤𐑝𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑐𐑦𐑝𐑩𐑑-𐑣𐑴𐑤, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑮𐑰𐑗𐑦𐑙 𐑣𐑲 𐑳𐑐 𐑩 𐑖𐑮𐑬𐑛, 𐑯 𐑑𐑲𐑑𐑤𐑦, 𐑷𐑤𐑥𐑴𐑕𐑑 𐑒𐑩𐑯𐑝𐑳𐑤𐑕𐑦𐑝𐑤𐑦 𐑜𐑮𐑭𐑕𐑐𐑦𐑙 𐑦𐑑, 𐑩𐑛𐑮𐑧𐑕𐑑 𐑞𐑧𐑥 𐑞𐑳𐑕:—

“𐑷𐑤 𐑘𐑰 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑼𐑟 𐑣𐑨𐑝 𐑚𐑦𐑓𐑹 𐑯𐑬 𐑣𐑻𐑛 𐑥𐑰 𐑜𐑦𐑝 𐑹𐑛𐑼𐑟 𐑩𐑚𐑬𐑑 𐑩 𐑢𐑲𐑑 𐑢𐑱𐑤. 𐑤𐑫𐑒 𐑘𐑰! 𐑛'𐑘𐑰 𐑕𐑰 𐑞𐑦𐑕 ·𐑕𐑐𐑨𐑯𐑦𐑖 𐑬𐑯𐑕 𐑝 𐑜𐑴𐑤𐑛?”—𐑣𐑴𐑤𐑛𐑦𐑙 𐑳𐑐 𐑩 𐑚𐑮𐑷𐑛 𐑚𐑮𐑲𐑑 𐑒𐑶𐑯 𐑑 𐑞 𐑕𐑳𐑯—“𐑦𐑑 𐑦𐑟 𐑩 𐑕𐑦𐑒𐑕𐑑𐑰𐑯 𐑛𐑪𐑤𐑼 𐑐𐑰𐑕, 𐑥𐑧𐑯. ·𐑛'𐑘𐑰 𐑕𐑰 𐑦𐑑? ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑣𐑨𐑯𐑛 𐑥𐑰 𐑘𐑪𐑯 𐑑𐑪𐑐-𐑥𐑷𐑤.”

𐑢𐑲𐑤 𐑞 𐑥𐑱𐑑 𐑢𐑪𐑟 𐑜𐑧𐑑𐑦𐑙 𐑞 𐑣𐑨𐑥𐑼, ·𐑱𐑣𐑨𐑚, 𐑢𐑦𐑞𐑬𐑑 𐑕𐑐𐑰𐑒𐑦𐑙, 𐑢𐑪𐑟 𐑕𐑤𐑴𐑤𐑦 𐑮𐑳𐑚𐑦𐑙 𐑞 𐑜𐑴𐑤𐑛 𐑐𐑰𐑕 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑕𐑒𐑻𐑑𐑕 𐑝 𐑣𐑦𐑟 𐑡𐑨𐑒𐑩𐑑, 𐑨𐑟 𐑦𐑓 𐑑 𐑣𐑲𐑑𐑩𐑯 𐑦𐑑𐑕 𐑤𐑳𐑕𐑑𐑼, 𐑯 𐑢𐑦𐑞𐑬𐑑 𐑿𐑟𐑦𐑙 𐑧𐑯𐑦 𐑢𐑻𐑛𐑟 𐑢𐑪𐑟 𐑥𐑰𐑯𐑢𐑲𐑤 𐑤𐑴𐑤𐑦 𐑣𐑳𐑥𐑦𐑙 𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑐𐑮𐑩𐑛𐑿𐑕𐑦𐑙 𐑩 𐑕𐑬𐑯𐑛 𐑕𐑴 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑥𐑳𐑓𐑩𐑤𐑛 𐑯 𐑦𐑯𐑸𐑑𐑦𐑒𐑿𐑤𐑩𐑑 𐑞𐑨𐑑 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑞 𐑥𐑦𐑒𐑨𐑯𐑦𐑒𐑩𐑤 𐑣𐑳𐑥𐑦𐑙 𐑝 𐑞 𐑢𐑰𐑤𐑟 𐑝 𐑣𐑦𐑟 𐑝𐑲𐑑𐑨𐑤𐑦𐑑𐑦 𐑦𐑯 𐑣𐑦𐑥.

𐑮𐑦𐑕𐑰𐑝𐑦𐑙 𐑞 𐑑𐑪𐑐-𐑥𐑷𐑤 𐑓𐑮𐑪𐑥 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑣𐑰 𐑩𐑛𐑝𐑭𐑯𐑕𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑 𐑢𐑦𐑞 𐑞 𐑣𐑨𐑥𐑼 𐑳𐑐𐑤𐑦𐑓𐑑𐑩𐑛 𐑦𐑯 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛, 𐑦𐑜𐑟𐑦𐑚𐑦𐑑𐑦𐑙 𐑞 𐑜𐑴𐑤𐑛 𐑢𐑦𐑞 𐑞 𐑳𐑞𐑼, 𐑯 𐑢𐑦𐑞 𐑩 𐑣𐑲 𐑮𐑱𐑟𐑛 𐑝𐑶𐑕 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑦𐑙: “𐑣𐑵𐑕𐑴𐑧𐑝𐑼 𐑝 𐑘𐑰 𐑮𐑱𐑟𐑩𐑟 𐑥𐑰 𐑩 𐑢𐑲𐑑-𐑣𐑧𐑛𐑩𐑛 𐑢𐑱𐑤 𐑢𐑦𐑞 𐑩 𐑮𐑦𐑙𐑒𐑩𐑤𐑛 𐑚𐑮𐑬 𐑯 𐑩 𐑒𐑮𐑫𐑒𐑩𐑛 𐑡𐑷; 𐑣𐑵𐑕𐑴𐑧𐑝𐑼 𐑝 𐑘𐑰 𐑮𐑱𐑟𐑩𐑟 𐑥𐑰 𐑞𐑨𐑑 𐑢𐑲𐑑-𐑣𐑧𐑛𐑩𐑛 𐑢𐑱𐑤, 𐑢𐑦𐑞 𐑔𐑮𐑰 𐑣𐑴𐑤𐑟 𐑐𐑳𐑙𐑒𐑗𐑼𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑕𐑑𐑸𐑚𐑼𐑛 𐑓𐑤𐑵𐑒—𐑤𐑫𐑒 𐑘𐑰, 𐑣𐑵𐑕𐑴𐑧𐑝𐑼 𐑝 𐑘𐑰 𐑮𐑱𐑟𐑩𐑟 𐑥𐑰 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑢𐑲𐑑 𐑢𐑱𐑤, 𐑣𐑰 𐑖𐑨𐑤 𐑣𐑨𐑝 𐑞𐑦𐑕 𐑜𐑴𐑤𐑛 𐑬𐑯𐑕, 𐑥𐑲 𐑚𐑶𐑟!”

“𐑣𐑩𐑟𐑭! 𐑣𐑩𐑟𐑭!” 𐑒𐑮𐑲𐑛 𐑞 𐑕𐑰𐑥𐑧𐑯, 𐑨𐑟 𐑢𐑦𐑞 𐑕𐑢𐑦𐑙𐑦𐑙 𐑑𐑸𐑐𐑷𐑤𐑦𐑯𐑟 𐑞𐑱 𐑣𐑱𐑤𐑛 𐑞 𐑨𐑒𐑑 𐑝 𐑯𐑱𐑤𐑦𐑙 𐑞 𐑜𐑴𐑤𐑛 𐑑 𐑞 𐑥𐑭𐑕𐑑.

“𐑦𐑑'𐑕 𐑩 𐑢𐑲𐑑 𐑢𐑱𐑤, 𐑲 𐑕𐑱,” 𐑮𐑦𐑟𐑿𐑥𐑛 ·𐑱𐑣𐑨𐑚, 𐑨𐑟 𐑣𐑰 𐑔𐑮𐑵 𐑛𐑬𐑯 𐑞 𐑑𐑪𐑐𐑥𐑷𐑤: “𐑩 𐑢𐑲𐑑 𐑢𐑱𐑤. 𐑕𐑒𐑦𐑯 𐑘𐑹 𐑲𐑟 𐑓 𐑣𐑦𐑥, 𐑥𐑧𐑯; 𐑤𐑫𐑒 𐑖𐑸𐑐 𐑓 𐑢𐑲𐑑 𐑢𐑷𐑑𐑼; 𐑦𐑓 𐑘𐑰 𐑕𐑰 𐑚𐑳𐑑 𐑩 𐑚𐑳𐑚𐑩𐑤, 𐑕𐑦𐑙 𐑬𐑑.”

𐑷𐑤 𐑞𐑦𐑕 𐑢𐑲𐑤 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, ·𐑛𐑨𐑜𐑵, 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑣𐑨𐑛 𐑤𐑫𐑒𐑑 𐑪𐑯 𐑢𐑦𐑞 𐑰𐑝𐑩𐑯 𐑥𐑹 𐑦𐑯𐑑𐑧𐑯𐑕 𐑦𐑯𐑑𐑼𐑩𐑕𐑑 𐑯 𐑕𐑼𐑐𐑮𐑲𐑟 𐑞𐑨𐑯 𐑞 𐑮𐑧𐑕𐑑, 𐑯 𐑨𐑑 𐑞 𐑥𐑧𐑯𐑖𐑩𐑯 𐑝 𐑞 𐑮𐑦𐑙𐑒𐑩𐑤𐑛 𐑚𐑮𐑬 𐑯 𐑒𐑮𐑫𐑒𐑑 𐑡𐑷 𐑞𐑱 𐑣𐑨𐑛 𐑕𐑑𐑸𐑑𐑩𐑛 𐑨𐑟 𐑦𐑓 𐑰𐑗 𐑢𐑪𐑟 𐑕𐑧𐑐𐑼𐑩𐑑𐑤𐑦 𐑑𐑳𐑗𐑑 𐑚𐑲 𐑕𐑳𐑥 𐑕𐑐𐑩𐑕𐑦𐑓𐑦𐑒 𐑮𐑧𐑒𐑩𐑤𐑧𐑒𐑖𐑩𐑯.

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚,” 𐑕𐑧𐑛 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, “𐑞𐑨𐑑 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑞 𐑕𐑱𐑥 𐑞𐑨𐑑 𐑕𐑳𐑥 𐑒𐑷𐑤 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒.”

“𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒?” 𐑖𐑬𐑑𐑩𐑛 ·𐑱𐑣𐑨𐑚. “𐑛𐑵 𐑘𐑰 𐑯𐑴 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑞𐑧𐑯, ·𐑑𐑨𐑖?”

“𐑛𐑳𐑟 𐑣𐑰 𐑓𐑨𐑯-𐑑𐑱𐑤 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑒𐑘𐑫𐑼𐑾𐑕, 𐑕𐑻, 𐑚𐑦𐑓𐑹 𐑣𐑰 𐑜𐑴𐑟 𐑛𐑬𐑯?” 𐑕𐑧𐑛 𐑞 ·𐑜𐑱-·𐑣𐑧𐑛𐑼 𐑛𐑦𐑤𐑦𐑚𐑼𐑩𐑑𐑤𐑦.

“𐑯 𐑣𐑨𐑟 𐑣𐑰 𐑩 𐑒𐑘𐑫𐑼𐑾𐑕 𐑕𐑐𐑬𐑑, 𐑑𐑵,” 𐑕𐑧𐑛 ·𐑛𐑨𐑜𐑵, “𐑝𐑧𐑮𐑦 𐑚𐑫𐑖𐑦, 𐑰𐑝𐑩𐑯 𐑓 𐑩 ·𐑐𐑸𐑥𐑩𐑕𐑧𐑑𐑦, 𐑯 𐑥𐑲𐑑𐑦 𐑒𐑢𐑦𐑒, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚?”

“𐑯 𐑣𐑰 𐑣𐑨𐑝 𐑢𐑳𐑯, 𐑑𐑵, 𐑔𐑮𐑰—𐑴! 𐑜𐑫𐑛 𐑥𐑧𐑯𐑦 𐑲𐑼𐑯 𐑦𐑯 𐑣𐑦𐑥 𐑣𐑲𐑛, 𐑑𐑵, ·𐑒𐑨𐑐𐑑𐑦𐑯,” 𐑒𐑮𐑲𐑛 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑛𐑦𐑕𐑡𐑶𐑯𐑑𐑩𐑛𐑤𐑦, “𐑷𐑤 𐑑𐑢𐑦𐑕𐑒-𐑑𐑰 𐑚𐑰-𐑑𐑢𐑲𐑕𐑒, 𐑤𐑲𐑒 𐑣𐑦𐑥—𐑣𐑦𐑥—” 𐑓𐑷𐑤𐑑𐑼𐑦𐑙 𐑣𐑸𐑛 𐑓 𐑩 𐑢𐑻𐑛, 𐑯 𐑕𐑒𐑮𐑵𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛 𐑨𐑟 𐑞𐑴 𐑳𐑯𐑒𐑹𐑒𐑦𐑙 𐑩 𐑚𐑪𐑑𐑩𐑤—“𐑤𐑲𐑒 𐑣𐑦𐑥—𐑣𐑦𐑥—”

“𐑒𐑹𐑒𐑕𐑒𐑮𐑵!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, “𐑲, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑞 𐑣𐑸𐑐𐑵𐑯𐑟 𐑤𐑲 𐑷𐑤 𐑑𐑢𐑦𐑕𐑑𐑩𐑛 𐑯 𐑮𐑧𐑯𐑗𐑑 𐑦𐑯 𐑣𐑦𐑥; 𐑲, ·𐑛𐑨𐑜𐑵, 𐑣𐑦𐑟 𐑕𐑐𐑬𐑑 𐑦𐑟 𐑩 𐑚𐑦𐑜 𐑢𐑳𐑯, 𐑤𐑲𐑒 𐑩 𐑣𐑴𐑤 𐑖𐑪𐑒 𐑝 𐑢𐑰𐑑, 𐑯 𐑢𐑲𐑑 𐑨𐑟 𐑩 𐑐𐑲𐑤 𐑝 𐑬𐑼 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑢𐑫𐑤 𐑭𐑓𐑑𐑼 𐑞 𐑜𐑮𐑱𐑑 𐑨𐑯𐑿𐑩𐑤 𐑖𐑰𐑐-𐑖𐑽𐑦𐑙; 𐑲, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, 𐑯 𐑣𐑰 𐑓𐑨𐑯-𐑑𐑱𐑤𐑟 𐑤𐑲𐑒 𐑩 𐑕𐑐𐑤𐑦𐑑 𐑡𐑦𐑚 𐑦𐑯 𐑩 𐑕𐑒𐑢𐑷𐑤. 𐑛𐑧𐑔 𐑯 𐑛𐑧𐑝𐑩𐑤𐑟! 𐑥𐑧𐑯, 𐑦𐑑 𐑦𐑟 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑘𐑰 𐑣𐑨𐑝 𐑕𐑰𐑯—·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒—·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒!”

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚,” 𐑕𐑧𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑣𐑵, 𐑢𐑦𐑞 ·𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒, 𐑣𐑨𐑛 𐑞𐑳𐑕 𐑓𐑸 𐑚𐑰𐑯 𐑲𐑦𐑙 𐑣𐑦𐑟 𐑕𐑵𐑐𐑽𐑽 𐑢𐑦𐑞 𐑦𐑯𐑒𐑮𐑰𐑕𐑦𐑙 𐑕𐑼𐑐𐑮𐑲𐑟, 𐑚𐑳𐑑 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑕𐑰𐑥𐑛 𐑕𐑑𐑮𐑳𐑒 𐑢𐑦𐑞 𐑩 𐑔𐑷𐑑 𐑢𐑦𐑗 𐑕𐑳𐑥𐑢𐑪𐑑 𐑦𐑒𐑕𐑐𐑤𐑱𐑯𐑛 𐑷𐑤 𐑞 𐑢𐑳𐑯𐑛𐑼. “𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑲 𐑣𐑨𐑝 𐑣𐑻𐑛 𐑝 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒—𐑚𐑳𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑯𐑪𐑑 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑞𐑨𐑑 𐑑𐑫𐑒 𐑪𐑓 𐑞𐑲 𐑤𐑧𐑜?”

“𐑣𐑵 𐑑𐑴𐑤𐑛 𐑞𐑰 𐑞𐑨𐑑?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚; 𐑞𐑧𐑯 𐑐𐑷𐑟𐑦𐑙, “𐑲, ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑲, 𐑥𐑲 𐑣𐑸𐑑𐑦𐑟 𐑷𐑤 𐑮𐑬𐑯𐑛; 𐑦𐑑 𐑢𐑪𐑟 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑞𐑨𐑑 𐑛𐑦𐑕𐑥𐑭𐑕𐑑𐑩𐑛 𐑥𐑰; ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑞𐑨𐑑 𐑚𐑮𐑷𐑑 𐑥𐑰 𐑑 𐑞𐑦𐑕 𐑛𐑧𐑛 𐑕𐑑𐑳𐑥𐑐 𐑲 𐑕𐑑𐑨𐑯𐑛 𐑪𐑯 𐑯𐑬. 𐑲, 𐑲,” 𐑣𐑰 𐑖𐑬𐑑𐑩𐑛 𐑢𐑦𐑞 𐑩 𐑑𐑼𐑦𐑓𐑦𐑒, 𐑤𐑬𐑛, 𐑨𐑯𐑦𐑥𐑩𐑤 𐑕𐑪𐑚, 𐑤𐑲𐑒 𐑞𐑨𐑑 𐑝 𐑩 𐑣𐑸𐑑-𐑕𐑑𐑮𐑦𐑒𐑩𐑯 𐑥𐑵𐑕; “𐑲, 𐑲! 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑩𐑒𐑻𐑕𐑩𐑛 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑞𐑨𐑑 𐑮𐑱𐑟𐑛 𐑥𐑰; 𐑥𐑱𐑛 𐑩 𐑐𐑫𐑼 𐑐𐑧𐑜𐑦𐑙 𐑤𐑳𐑚𐑼 𐑝 𐑥𐑰 𐑓 𐑧𐑝𐑼 𐑯 𐑩 𐑛𐑱!” 𐑞𐑧𐑯 𐑑𐑪𐑕𐑦𐑙 𐑚𐑴𐑔 𐑸𐑥𐑟, 𐑢𐑦𐑞 𐑥𐑧𐑠𐑼𐑤𐑩𐑕 𐑦𐑥𐑐𐑮𐑦𐑒𐑱𐑖𐑩𐑯𐑟 𐑣𐑰 𐑖𐑬𐑑𐑩𐑛 𐑬𐑑: “𐑲, 𐑲! 𐑯 𐑲'𐑤 𐑗𐑱𐑕 𐑣𐑦𐑥 𐑮𐑬𐑯𐑛 𐑜𐑫𐑛 ·𐑣𐑴𐑐, 𐑯 𐑮𐑬𐑯𐑛 𐑞 ·𐑣𐑹𐑯, 𐑯 𐑮𐑬𐑯𐑛 𐑞 ·𐑯𐑹𐑢𐑱 ·𐑥𐑱𐑤𐑕𐑑𐑮𐑩𐑥, 𐑯 𐑮𐑬𐑯𐑛 𐑐𐑼𐑛𐑦𐑖𐑩𐑯'𐑟 𐑓𐑤𐑱𐑥𐑟 𐑚𐑦𐑓𐑹 𐑲 𐑜𐑦𐑝 𐑣𐑦𐑥 𐑳𐑐. 𐑯 𐑞𐑦𐑕 𐑦𐑟 𐑢𐑪𐑑 𐑘𐑰 𐑣𐑨𐑝 𐑖𐑦𐑐𐑑 𐑓, 𐑥𐑧𐑯! 𐑑 𐑗𐑱𐑕 𐑞𐑨𐑑 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑪𐑯 𐑚𐑴𐑔 𐑕𐑲𐑛𐑟 𐑝 𐑤𐑨𐑯𐑛, 𐑯 𐑴𐑝𐑼 𐑷𐑤 𐑕𐑲𐑛𐑟 𐑝 𐑻𐑔, 𐑑𐑦𐑤 𐑣𐑰 𐑕𐑐𐑬𐑑𐑕 𐑚𐑤𐑨𐑒 𐑚𐑤𐑳𐑛 𐑯 𐑮𐑴𐑤𐑟 𐑓𐑦𐑯 𐑬𐑑. 𐑢𐑪𐑑 𐑕𐑱 𐑘𐑰, 𐑥𐑧𐑯, 𐑢𐑦𐑤 𐑘𐑰 𐑕𐑐𐑤𐑲𐑕 𐑣𐑨𐑯𐑛𐑟 𐑪𐑯 𐑦𐑑, 𐑯𐑬? 𐑲 𐑔𐑦𐑙𐑒 𐑘𐑰 𐑛𐑵 𐑤𐑫𐑒 𐑚𐑮𐑱𐑝.”

“𐑲, 𐑲!” 𐑖𐑬𐑑𐑩𐑛 𐑞 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑯 𐑕𐑰𐑥𐑧𐑯, 𐑮𐑳𐑯𐑦𐑙 𐑒𐑤𐑴𐑕𐑼 𐑑 𐑞 𐑦𐑒𐑕𐑲𐑑𐑩𐑛 𐑴𐑤𐑛 𐑥𐑨𐑯: “𐑩 𐑖𐑸𐑐 𐑲 𐑓 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤; 𐑩 𐑖𐑸𐑐 𐑤𐑭𐑯𐑕 𐑓 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒!”

“𐑜𐑪𐑛 𐑚𐑤𐑧𐑕 𐑘𐑰,” 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑑 𐑣𐑭𐑓 𐑕𐑪𐑚 𐑯 𐑣𐑭𐑓 𐑖𐑬𐑑. “𐑜𐑪𐑛 𐑚𐑤𐑧𐑕 𐑘𐑰, 𐑥𐑧𐑯. 𐑕𐑑𐑿𐑼𐑛! 𐑜𐑴 𐑛𐑮𐑷 𐑞 𐑜𐑮𐑱𐑑 𐑥𐑧𐑠𐑼 𐑝 𐑜𐑮𐑪𐑜. 𐑚𐑳𐑑 𐑢𐑪𐑑'𐑕 𐑞𐑦𐑕 𐑤𐑪𐑙 𐑓𐑱𐑕 𐑩𐑚𐑬𐑑, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑢𐑦𐑤𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑗𐑱𐑕 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤? 𐑸𐑑 𐑯𐑪𐑑 𐑜𐑱𐑥 𐑓 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒?”

“𐑲 𐑨𐑥 𐑜𐑱𐑥 𐑓 𐑣𐑦𐑟 𐑒𐑮𐑫𐑒𐑩𐑛 𐑡𐑷, 𐑯 𐑓 𐑞 𐑡𐑷𐑟 𐑝 ·𐑛𐑧𐑔 𐑑𐑵, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑦𐑓 𐑦𐑑 𐑓𐑺𐑤𐑦 𐑒𐑳𐑥𐑟 𐑦𐑯 𐑞 𐑢𐑱 𐑝 𐑞 𐑚𐑦𐑟𐑯𐑩𐑕 𐑢𐑰 𐑓𐑪𐑤𐑴; 𐑚𐑳𐑑 𐑲 𐑒𐑱𐑥 𐑣𐑽 𐑑 𐑣𐑳𐑯𐑑 𐑢𐑱𐑤𐑟, 𐑯𐑪𐑑 𐑥𐑲 𐑒𐑩𐑥𐑭𐑯𐑛𐑼'𐑟 𐑝𐑧𐑯𐑡𐑩𐑯𐑕. 𐑣𐑬 𐑥𐑧𐑯𐑦 𐑚𐑨𐑮𐑩𐑤𐑟 𐑢𐑦𐑤 𐑞𐑲 𐑝𐑧𐑯𐑡𐑩𐑯𐑕 𐑘𐑰𐑤𐑛 𐑞𐑰 𐑰𐑝𐑩𐑯 𐑦𐑓 𐑞𐑬 𐑜𐑧𐑑𐑩𐑕𐑑 𐑦𐑑, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚? 𐑦𐑑 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑓𐑧𐑗 𐑞𐑰 𐑥𐑳𐑗 𐑦𐑯 𐑬𐑼 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑥𐑸𐑒𐑩𐑑.”

“·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑥𐑸𐑒𐑩𐑑! 𐑣𐑵𐑑! 𐑚𐑳𐑑 𐑒𐑳𐑥 𐑒𐑤𐑴𐑕𐑼, ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑞𐑬 𐑮𐑦𐑒𐑢𐑲𐑼𐑩𐑕𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑤𐑴𐑼 𐑤𐑱𐑼. 𐑦𐑓 𐑥𐑳𐑯𐑦'𐑟 𐑑 𐑚𐑰 𐑞 𐑥𐑧𐑠𐑼𐑼, 𐑥𐑨𐑯, 𐑯 𐑞 𐑩𐑒𐑬𐑯𐑑𐑩𐑯𐑑𐑕 𐑣𐑨𐑝 𐑒𐑩𐑥𐑐𐑿𐑑𐑩𐑛 𐑞𐑺 𐑜𐑮𐑱𐑑 𐑒𐑬𐑯𐑑𐑦𐑙-𐑣𐑬𐑕 𐑞 𐑜𐑤𐑴𐑚, 𐑚𐑲 𐑜𐑻𐑛𐑩𐑤𐑦𐑙 𐑦𐑑 𐑢𐑦𐑞 𐑜𐑦𐑯𐑦𐑟, 𐑢𐑳𐑯 𐑑 𐑧𐑝𐑮𐑦 𐑔𐑮𐑰 𐑐𐑸𐑑𐑕 𐑝 𐑩𐑯 𐑦𐑯𐑗; 𐑞𐑧𐑯, 𐑤𐑧𐑑 𐑥𐑰 𐑑𐑧𐑤 𐑞𐑰, 𐑞𐑨𐑑 𐑥𐑲 𐑝𐑧𐑯𐑡𐑩𐑯𐑕 𐑢𐑦𐑤 𐑓𐑧𐑗 𐑩 𐑜𐑮𐑱𐑑 𐑐𐑮𐑰𐑥𐑾𐑥 𐑣𐑽!”

𐑚𐑳𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑡𐑶 𐑨𐑑 𐑞 𐑦𐑯𐑗𐑭𐑯𐑑𐑩𐑛, 𐑑𐑨𐑕𐑦𐑑 𐑨𐑒𐑢𐑦𐑧𐑕𐑩𐑯𐑕 𐑝 𐑞 𐑥𐑱𐑑, ·𐑱𐑣𐑨𐑚 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑣𐑽 𐑣𐑦𐑟 𐑓𐑹𐑚𐑴𐑛𐑦𐑙 𐑦𐑯𐑝𐑩𐑒𐑱𐑖𐑩𐑯; 𐑯𐑹 𐑘𐑧𐑑 𐑞 𐑤𐑴 𐑤𐑭𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑣𐑴𐑤𐑛; 𐑯𐑹 𐑘𐑧𐑑 𐑞 𐑐𐑮𐑦𐑕𐑱𐑡𐑦𐑙 𐑝𐑲𐑚𐑮𐑱𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑢𐑦𐑯𐑛𐑟 𐑦𐑯 𐑞 𐑒𐑹𐑛𐑦𐑡; 𐑯𐑹 𐑘𐑧𐑑 𐑞 𐑣𐑪𐑤𐑴 𐑓𐑤𐑨𐑐 𐑝 𐑞 𐑕𐑱𐑤𐑟 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑥𐑭𐑕𐑑𐑕, 𐑨𐑟 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑞𐑺 𐑣𐑸𐑑𐑕 𐑕𐑨𐑙𐑒 𐑦𐑯. 𐑓 𐑩𐑜𐑱𐑯 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑛𐑬𐑯𐑒𐑭𐑕𐑑 𐑲𐑟 𐑤𐑲𐑑𐑩𐑛 𐑳𐑐 𐑢𐑦𐑞 𐑞 𐑕𐑑𐑳𐑚𐑼𐑯𐑯𐑩𐑕 𐑝 𐑤𐑲𐑓; 𐑞 𐑕𐑳𐑚𐑑𐑼𐑱𐑯𐑾𐑯 𐑤𐑭𐑓 𐑛𐑲𐑛 𐑩𐑢𐑱; 𐑞 𐑢𐑦𐑯𐑛𐑟 𐑚𐑤𐑵 𐑪𐑯; 𐑞 𐑕𐑱𐑤𐑟 𐑓𐑦𐑤𐑛 𐑬𐑑; 𐑞 𐑖𐑦𐑐 𐑣𐑰𐑝𐑛 𐑯 𐑮𐑴𐑤𐑛 𐑨𐑟 𐑚𐑦𐑓𐑹. 𐑭, 𐑘𐑰 𐑨𐑛𐑥𐑪𐑯𐑦𐑖𐑩𐑯𐑟 𐑯 𐑢𐑹𐑯𐑦𐑙𐑟! 𐑢𐑲 𐑕𐑑𐑱 𐑘𐑰 𐑯𐑪𐑑 𐑢𐑧𐑯 𐑘𐑰 𐑒𐑳𐑥? 𐑚𐑳𐑑 𐑮𐑭𐑞𐑼 𐑸 𐑘𐑰 𐑐𐑮𐑦𐑛𐑦𐑒𐑖𐑩𐑯𐑟 𐑞𐑨𐑯 𐑢𐑹𐑯𐑦𐑙𐑟, 𐑘𐑰 𐑖𐑨𐑛𐑴𐑟! 𐑘𐑧𐑑 𐑯𐑪𐑑 𐑕𐑴 𐑥𐑳𐑗 𐑐𐑮𐑦𐑛𐑦𐑒𐑖𐑩𐑯𐑟 𐑓𐑮𐑪𐑥 𐑢𐑦𐑞𐑬𐑑, 𐑨𐑟 𐑝𐑧𐑮𐑦𐑓𐑦𐑒𐑱𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑓𐑹𐑜𐑴𐑦𐑙 𐑔𐑦𐑙𐑟 𐑢𐑦𐑞𐑦𐑯. 𐑓 𐑢𐑦𐑞 𐑤𐑦𐑑𐑩𐑤 𐑧𐑒𐑕𐑑𐑼𐑯𐑩𐑤 𐑑 𐑒𐑩𐑯𐑕𐑑𐑮𐑱𐑯 𐑳𐑕, 𐑞 𐑦𐑯𐑼𐑥𐑴𐑕𐑑 𐑯𐑦𐑕𐑧𐑕𐑦𐑑𐑦𐑟 𐑦𐑯 𐑬𐑼 𐑚𐑰𐑦𐑙, 𐑞𐑰𐑟 𐑕𐑑𐑦𐑤 𐑛𐑮𐑲𐑝 𐑳𐑕 𐑪𐑯.

“𐑞 𐑥𐑧𐑠𐑼! 𐑞 𐑥𐑧𐑠𐑼!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚.

𐑮𐑦𐑕𐑰𐑝𐑦𐑙 𐑞 𐑚𐑮𐑦𐑥𐑦𐑙 𐑐𐑿𐑑𐑼, 𐑯 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑞 𐑣𐑸𐑐𐑵𐑯𐑽𐑟, 𐑣𐑰 𐑹𐑛𐑼𐑛 𐑞𐑧𐑥 𐑑 𐑐𐑮𐑩𐑛𐑿𐑕 𐑞𐑺 𐑢𐑧𐑐𐑩𐑯𐑟. 𐑞𐑧𐑯 𐑮𐑱𐑯𐑡𐑦𐑙 𐑞𐑧𐑥 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥 𐑯𐑽 𐑞 𐑒𐑨𐑐𐑕𐑑𐑩𐑯, 𐑢𐑦𐑞 𐑞𐑺 𐑣𐑸𐑐𐑵𐑯𐑟 𐑦𐑯 𐑞𐑺 𐑣𐑨𐑯𐑛𐑟, 𐑢𐑲𐑤 𐑣𐑦𐑟 𐑔𐑮𐑰 𐑥𐑱𐑑𐑕 𐑕𐑑𐑫𐑛 𐑨𐑑 𐑣𐑦𐑟 𐑕𐑲𐑛 𐑢𐑦𐑞 𐑞𐑺 𐑤𐑭𐑯𐑕𐑩𐑟, 𐑯 𐑞 𐑮𐑧𐑕𐑑 𐑝 𐑞 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑓𐑹𐑥𐑛 𐑩 𐑕𐑻𐑒𐑩𐑤 𐑮𐑬𐑯𐑛 𐑞 𐑜𐑮𐑵𐑐; 𐑣𐑰 𐑕𐑑𐑫𐑛 𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑕𐑻𐑗𐑦𐑙𐑤𐑦 𐑲𐑦𐑙 𐑧𐑝𐑮𐑦 𐑥𐑨𐑯 𐑝 𐑣𐑦𐑟 𐑒𐑮𐑵. 𐑚𐑳𐑑 𐑞𐑴𐑟 𐑢𐑲𐑤𐑛 𐑲𐑟 𐑥𐑧𐑑 𐑣𐑦𐑟, 𐑨𐑟 𐑞 𐑚𐑤𐑳𐑛𐑖𐑪𐑑 𐑲𐑟 𐑝 𐑞 𐑐𐑮𐑺𐑦 𐑢𐑫𐑤𐑝𐑟 𐑥𐑰𐑑 𐑞 𐑲 𐑝 𐑞𐑺 𐑤𐑰𐑛𐑼, 𐑺 𐑣𐑰 𐑮𐑳𐑖𐑩𐑟 𐑪𐑯 𐑨𐑑 𐑞𐑺 𐑣𐑧𐑛 𐑦𐑯 𐑞 𐑑𐑮𐑱𐑤 𐑝 𐑞 𐑚𐑲𐑕𐑩𐑯; 𐑚𐑳𐑑, 𐑩𐑤𐑨𐑕! 𐑴𐑯𐑤𐑦 𐑑 𐑓𐑷𐑤 𐑦𐑯𐑑𐑵 𐑞 𐑣𐑦𐑛𐑩𐑯 𐑕𐑯𐑺 𐑝 𐑞 ·𐑦𐑯𐑛𐑾𐑯.

“𐑛𐑮𐑦𐑙𐑒 𐑯 𐑐𐑭𐑕!” 𐑣𐑰 𐑒𐑮𐑲𐑛, 𐑣𐑨𐑯𐑛𐑦𐑙 𐑞 𐑣𐑧𐑝𐑦 𐑗𐑸𐑡𐑛 𐑓𐑤𐑨𐑜𐑩𐑯 𐑑 𐑞 𐑯𐑽𐑩𐑕𐑑 𐑕𐑰𐑥𐑩𐑯. “𐑞 𐑒𐑮𐑵 𐑩𐑤𐑴𐑯 𐑯𐑬 𐑛𐑮𐑦𐑙𐑒. 𐑮𐑬𐑯𐑛 𐑢𐑦𐑞 𐑦𐑑, 𐑮𐑬𐑯𐑛! 𐑖𐑹𐑑 𐑛𐑮𐑭𐑓𐑑𐑕—𐑤𐑪𐑙 𐑕𐑢𐑪𐑤𐑴𐑟, 𐑥𐑧𐑯; '𐑑𐑦𐑟 𐑣𐑪𐑑 𐑨𐑟 ·𐑕𐑱𐑑𐑩𐑯'𐑟 𐑣𐑵𐑓. 𐑕𐑴, 𐑕𐑴; 𐑦𐑑 𐑜𐑴𐑟 𐑮𐑬𐑯𐑛 𐑧𐑒𐑕𐑩𐑤𐑩𐑯𐑑𐑤𐑦. 𐑦𐑑 𐑕𐑐𐑲𐑮𐑩𐑤𐑲𐑟𐑩𐑟 𐑦𐑯 𐑘𐑰; 𐑓𐑹𐑒𐑕 𐑬𐑑 𐑨𐑑 𐑞 𐑕𐑻𐑐𐑩𐑯𐑑-𐑕𐑯𐑨𐑐𐑦𐑙 𐑲. 𐑢𐑧𐑤 𐑛𐑳𐑯; 𐑷𐑤𐑥𐑴𐑕𐑑 𐑛𐑮𐑱𐑯𐑛. 𐑞𐑨𐑑 𐑢𐑱 𐑦𐑑 𐑢𐑧𐑯𐑑, 𐑞𐑦𐑕 𐑢𐑱 𐑦𐑑 𐑒𐑳𐑥𐑟. 𐑣𐑨𐑯𐑛 𐑦𐑑 𐑥𐑰—𐑣𐑽'𐑟 𐑩 𐑣𐑪𐑤𐑴! 𐑥𐑧𐑯, 𐑘𐑰 𐑕𐑰𐑥 𐑞 𐑘𐑽𐑟; 𐑕𐑴 𐑚𐑮𐑦𐑥𐑦𐑙 𐑤𐑲𐑓 𐑦𐑟 𐑜𐑳𐑤𐑐𐑑 𐑯 𐑜𐑪𐑯. 𐑕𐑑𐑿𐑼𐑛, 𐑮𐑰𐑓𐑦𐑤!

“𐑩𐑑𐑧𐑯𐑛 𐑯𐑬, 𐑥𐑲 𐑚𐑮𐑱𐑝𐑟. 𐑲 𐑣𐑨𐑝 𐑥𐑳𐑕𐑑𐑼𐑛 𐑘𐑰 𐑷𐑤 𐑮𐑬𐑯𐑛 𐑞𐑦𐑕 𐑒𐑨𐑐𐑕𐑑𐑩𐑯; 𐑯 𐑘𐑰 𐑥𐑱𐑑𐑕, 𐑓𐑤𐑨𐑙𐑒 𐑥𐑰 𐑢𐑦𐑞 𐑘𐑹 𐑤𐑭𐑯𐑕𐑩𐑟; 𐑯 𐑘𐑰 𐑣𐑸𐑐𐑵𐑯𐑽𐑟, 𐑕𐑑𐑨𐑯𐑛 𐑞𐑺 𐑢𐑦𐑞 𐑘𐑹 𐑲𐑼𐑯𐑟; 𐑯 𐑘𐑰, 𐑕𐑑𐑬𐑑 𐑥𐑨𐑮𐑦𐑯𐑼𐑟, 𐑮𐑦𐑙 𐑥𐑰 𐑦𐑯, 𐑞𐑨𐑑 𐑲 𐑥𐑱 𐑦𐑯 𐑕𐑳𐑥 𐑕𐑹𐑑 𐑮𐑦𐑝𐑲𐑝 𐑩 𐑯𐑴𐑚𐑩𐑤 𐑒𐑳𐑕𐑑𐑩𐑥 𐑝 𐑥𐑲 𐑓𐑦𐑖𐑼𐑥𐑩𐑯 𐑓𐑭𐑞𐑼𐑟 𐑚𐑦𐑓𐑹 𐑥𐑰. 𐑴 𐑥𐑧𐑯, 𐑿 𐑢𐑦𐑤 𐑘𐑧𐑑 𐑕𐑰 𐑞𐑨𐑑—·𐑣𐑭! 𐑚𐑶, 𐑒𐑳𐑥 𐑚𐑨𐑒? 𐑚𐑨𐑛 𐑐𐑧𐑯𐑦𐑟 𐑒𐑳𐑥 𐑯𐑪𐑑 𐑕𐑵𐑯𐑼. 𐑣𐑨𐑯𐑛 𐑦𐑑 𐑥𐑰. 𐑢𐑲, 𐑯𐑬, 𐑞𐑦𐑕 𐑐𐑿𐑑𐑼 𐑣𐑨𐑛 𐑮𐑳𐑯 𐑚𐑮𐑦𐑥𐑦𐑙 𐑩𐑜𐑱𐑯, 𐑢𐑻'𐑑 𐑯𐑪𐑑 𐑞𐑬 ·𐑕𐑑 ·𐑝𐑰𐑑𐑵𐑕' 𐑦𐑥𐑐—𐑩𐑢𐑱, 𐑞𐑬 𐑱𐑜𐑿!

“𐑩𐑛𐑝𐑭𐑯𐑕, 𐑘𐑰 𐑥𐑱𐑑𐑕! 𐑒𐑮𐑪𐑕 𐑘𐑹 𐑤𐑭𐑯𐑕𐑩𐑟 𐑓𐑫𐑤 𐑚𐑦𐑓𐑹 𐑥𐑰. 𐑢𐑧𐑤 𐑛𐑳𐑯! 𐑤𐑧𐑑 𐑥𐑰 𐑑𐑳𐑗 𐑞 𐑨𐑒𐑕𐑦𐑕.” 𐑕𐑴 𐑕𐑱𐑦𐑙, 𐑢𐑦𐑞 𐑧𐑒𐑕𐑑𐑧𐑯𐑛𐑩𐑛 𐑸𐑥, 𐑣𐑰 𐑜𐑮𐑭𐑕𐑐𐑑 𐑞 𐑔𐑮𐑰 𐑤𐑧𐑝𐑩𐑤, 𐑮𐑱𐑛𐑦𐑱𐑑𐑦𐑙 𐑤𐑭𐑯𐑕𐑩𐑟 𐑨𐑑 𐑞𐑺 𐑒𐑮𐑪𐑕𐑑 𐑕𐑧𐑯𐑑𐑼; 𐑢𐑲𐑤 𐑕𐑴 𐑛𐑵𐑦𐑙, 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑯 𐑯𐑻𐑝𐑩𐑕𐑤𐑦 𐑑𐑢𐑦𐑗𐑑 𐑞𐑧𐑥; 𐑥𐑰𐑯𐑢𐑲𐑤, 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑦𐑯𐑑𐑧𐑯𐑑𐑤𐑦 𐑓𐑮𐑪𐑥 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑑 ·𐑕𐑑𐑳𐑚; 𐑓𐑮𐑪𐑥 ·𐑕𐑑𐑳𐑚 𐑑 ·𐑓𐑤𐑭𐑕𐑒. 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑨𐑟 𐑞𐑴, 𐑚𐑲 𐑕𐑳𐑥 𐑯𐑱𐑥𐑤𐑩𐑕, 𐑦𐑯𐑑𐑽𐑽 𐑝𐑴𐑤𐑦𐑖𐑩𐑯, 𐑣𐑰 𐑢𐑫𐑛 𐑓𐑱𐑯 𐑣𐑨𐑝 𐑖𐑪𐑒𐑑 𐑦𐑯𐑑𐑵 𐑞𐑧𐑥 𐑞 𐑕𐑱𐑥 𐑓𐑲𐑼𐑦 𐑦𐑥𐑴𐑖𐑩𐑯 𐑩𐑒𐑿𐑥𐑿𐑤𐑱𐑑𐑩𐑛 𐑢𐑦𐑞𐑦𐑯 𐑞 ·𐑤𐑱𐑛𐑩𐑯 𐑡𐑸 𐑝 𐑣𐑦𐑟 𐑴𐑯 𐑥𐑨𐑜𐑯𐑧𐑑𐑦𐑒 𐑤𐑲𐑓. 𐑞 𐑔𐑮𐑰 𐑥𐑱𐑑𐑕 𐑒𐑢𐑱𐑤𐑛 𐑚𐑦𐑓𐑹 𐑣𐑦𐑟 𐑕𐑑𐑮𐑪𐑙, 𐑕𐑩𐑕𐑑𐑱𐑯𐑛, 𐑯 𐑥𐑦𐑕𐑑𐑦𐑒 𐑨𐑕𐑐𐑧𐑒𐑑. 𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒 𐑤𐑫𐑒𐑑 𐑕𐑲𐑛𐑢𐑱𐑟 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥; 𐑞 𐑪𐑯𐑩𐑕𐑑 𐑲 𐑝 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑓𐑧𐑤 𐑛𐑬𐑯𐑮𐑲𐑑.

“𐑦𐑯 𐑝𐑱𐑯!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚; “𐑚𐑳𐑑, 𐑥𐑱𐑚𐑰, '𐑑𐑦𐑟 𐑢𐑧𐑤. 𐑓 𐑛𐑦𐑛 𐑘𐑰 𐑔𐑮𐑰 𐑚𐑳𐑑 𐑢𐑳𐑯𐑕 𐑑𐑱𐑒 𐑞 𐑓𐑫𐑤-𐑓𐑹𐑕𐑑 𐑖𐑪𐑒, 𐑞𐑧𐑯 𐑥𐑲𐑯 𐑴𐑯 𐑦𐑤𐑧𐑒𐑑𐑮𐑦𐑒 𐑔𐑦𐑙, 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑐𐑼𐑣𐑨𐑐𐑕 𐑦𐑒𐑕𐑐𐑲𐑼𐑛 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑥𐑰. 𐑐𐑼𐑗𐑨𐑯𐑕, 𐑑𐑵, 𐑦𐑑 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑛𐑮𐑪𐑐𐑑 𐑘𐑰 𐑛𐑧𐑛. 𐑐𐑼𐑗𐑨𐑯𐑕 𐑘𐑰 𐑯𐑰𐑛 𐑦𐑑 𐑯𐑪𐑑. 𐑛𐑬𐑯 𐑤𐑭𐑯𐑕𐑩𐑟! 𐑯 𐑯𐑬, 𐑘𐑰 𐑥𐑱𐑑𐑕, 𐑲 𐑛𐑵 𐑩𐑐𐑶𐑯𐑑 𐑘𐑰 𐑔𐑮𐑰 𐑒𐑳𐑐𐑚𐑺𐑼𐑟 𐑑 𐑥𐑲 𐑔𐑮𐑰 𐑐𐑱𐑜𐑩𐑯 𐑒𐑦𐑯𐑟𐑥𐑧𐑯 𐑞𐑺—𐑘𐑪𐑯 𐑔𐑮𐑰 𐑥𐑴𐑕𐑑 𐑪𐑯𐑼𐑩𐑚𐑩𐑤 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑧𐑯 𐑯 𐑯𐑴𐑚𐑩𐑤𐑥𐑧𐑯, 𐑥𐑲 𐑝𐑨𐑤𐑾𐑯𐑑 𐑣𐑸𐑐𐑵𐑯𐑽𐑟. 𐑛𐑦𐑕𐑛𐑱𐑯 𐑞 𐑑𐑭𐑕𐑒? 𐑢𐑪𐑑, 𐑢𐑧𐑯 𐑞 𐑜𐑮𐑱𐑑 ·𐑐𐑴𐑐 𐑢𐑪𐑖𐑩𐑟 𐑞 𐑓𐑰𐑑 𐑝 𐑚𐑧𐑜𐑼𐑟, 𐑿𐑟𐑦𐑙 𐑣𐑦𐑟 𐑑𐑦𐑸𐑩 𐑓 𐑿𐑼? 𐑴, 𐑥𐑲 𐑕𐑢𐑰𐑑 𐑒𐑸𐑛𐑦𐑯𐑩𐑤𐑟! 𐑘𐑹 𐑴𐑯 𐑒𐑪𐑯𐑛𐑦𐑕𐑧𐑯𐑖𐑩𐑯, 𐑞𐑨𐑑 𐑖𐑨𐑤 𐑚𐑧𐑯𐑛 𐑘𐑰 𐑑 𐑦𐑑. 𐑲 𐑛𐑵 𐑯𐑪𐑑 𐑹𐑛𐑼 𐑘𐑰; 𐑘𐑰 𐑢𐑦𐑤 𐑦𐑑. 𐑒𐑳𐑑 𐑘𐑹 𐑕𐑰𐑟𐑦𐑙𐑟 𐑯 𐑛𐑮𐑷 𐑞 𐑐𐑴𐑤𐑟, 𐑘𐑰 𐑣𐑸𐑐𐑵𐑯𐑽𐑟!”

𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑴𐑚𐑱𐑦𐑙 𐑞 𐑹𐑛𐑼, 𐑞 𐑔𐑮𐑰 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑯𐑬 𐑕𐑑𐑫𐑛 𐑢𐑦𐑞 𐑞 𐑛𐑦𐑑𐑨𐑗𐑑 𐑲𐑼𐑯 𐑐𐑸𐑑 𐑝 𐑞𐑺 𐑣𐑸𐑐𐑵𐑯𐑟, 𐑕𐑳𐑥 𐑔𐑮𐑰 𐑓𐑰𐑑 𐑤𐑪𐑙, 𐑣𐑧𐑤𐑛, 𐑚𐑸𐑚𐑟 𐑳𐑐, 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥.

“𐑕𐑑𐑨𐑚 𐑥𐑰 𐑯𐑪𐑑 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑒𐑰𐑯 𐑕𐑑𐑰𐑤! 𐑒𐑨𐑯𐑑 𐑞𐑧𐑥; 𐑒𐑨𐑯𐑑 𐑞𐑧𐑥 𐑴𐑝𐑼! 𐑯𐑴 𐑘𐑰 𐑯𐑪𐑑 𐑞 𐑜𐑪𐑚𐑤𐑩𐑑 𐑧𐑯𐑛? 𐑑𐑻𐑯 𐑳𐑐 𐑞 𐑕𐑪𐑒𐑩𐑑! 𐑕𐑴, 𐑕𐑴; 𐑯𐑬, 𐑘𐑰 𐑒𐑳𐑐-𐑚𐑺𐑼𐑟, 𐑩𐑛𐑝𐑭𐑯𐑕. 𐑞 𐑲𐑼𐑯𐑟! 𐑑𐑱𐑒 𐑞𐑧𐑥; 𐑣𐑴𐑤𐑛 𐑞𐑧𐑥 𐑢𐑲𐑤 𐑲 𐑓𐑦𐑤!” 𐑓𐑹𐑔𐑢𐑦𐑔, 𐑕𐑤𐑴𐑤𐑦 𐑜𐑴𐑦𐑙 𐑓𐑮𐑪𐑥 𐑢𐑳𐑯 𐑪𐑓𐑦𐑕𐑼 𐑑 𐑞 𐑳𐑞𐑼, 𐑣𐑰 𐑚𐑮𐑦𐑥𐑛 𐑞 𐑣𐑸𐑐𐑵𐑯 𐑕𐑪𐑒𐑩𐑑𐑕 𐑢𐑦𐑞 𐑞 𐑓𐑲𐑼𐑦 𐑢𐑷𐑑𐑼𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑐𐑿𐑑𐑼.
