·𐑗𐑨𐑐𐑑𐑼 37. 𐑕𐑳𐑯𐑕𐑧𐑑.

𐑞 𐑒𐑨𐑚𐑦𐑯; 𐑚𐑲 𐑞 𐑕𐑑𐑻𐑯 𐑢𐑦𐑯𐑛𐑴𐑟; ·𐑱𐑣𐑨𐑚 𐑕𐑦𐑑𐑦𐑙 𐑩𐑤𐑴𐑯, 𐑯 𐑜𐑱𐑟𐑦𐑙 𐑬𐑑.

𐑲 𐑤𐑰𐑝 𐑩 𐑢𐑲𐑑 𐑯 𐑑𐑻𐑚𐑦𐑛 𐑢𐑱𐑒; 𐑐𐑱𐑤 𐑢𐑷𐑑𐑼𐑟, 𐑐𐑱𐑤𐑼 𐑗𐑰𐑒𐑕, 𐑢𐑺'𐑼 𐑲 𐑕𐑱𐑤. 𐑞 𐑧𐑯𐑝𐑾𐑕 𐑚𐑦𐑤𐑴𐑟 𐑕𐑲𐑛𐑤𐑪𐑙 𐑕𐑢𐑧𐑤 𐑑 𐑢𐑧𐑤𐑥 𐑥𐑲 𐑑𐑮𐑨𐑒; 𐑤𐑧𐑑 𐑞𐑧𐑥; 𐑚𐑳𐑑 𐑓𐑻𐑕𐑑 𐑲 𐑐𐑭𐑕.

𐑘𐑪𐑯𐑛𐑼, 𐑚𐑲 𐑧𐑝𐑼-𐑚𐑮𐑦𐑥𐑦𐑙 𐑜𐑪𐑚𐑤𐑩𐑑'𐑕 𐑮𐑦𐑥, 𐑞 𐑢𐑹𐑥 𐑢𐑱𐑝𐑟 𐑚𐑤𐑳𐑖 𐑤𐑲𐑒 𐑢𐑲𐑯. 𐑞 𐑜𐑴𐑤𐑛 𐑚𐑮𐑬 𐑐𐑤𐑳𐑥𐑟 𐑞 𐑚𐑤𐑵. 𐑞 𐑛𐑲𐑝𐑼 𐑕𐑳𐑯—𐑕𐑤𐑴 𐑛𐑲𐑝𐑛 𐑓𐑮𐑪𐑥 𐑯𐑵𐑯—𐑜𐑴𐑟 𐑛𐑬𐑯; 𐑥𐑲 𐑕𐑴𐑤 𐑥𐑬𐑯𐑑𐑕 𐑳𐑐! 𐑖𐑰 𐑢𐑽𐑦𐑟 𐑢𐑦𐑞 𐑣𐑻 𐑧𐑯𐑛𐑤𐑩𐑕 𐑣𐑦𐑤. 𐑦𐑟, 𐑞𐑧𐑯, 𐑞 𐑒𐑮𐑬𐑯 𐑑𐑵 𐑣𐑧𐑝𐑦 𐑞𐑨𐑑 𐑲 𐑢𐑺? 𐑞𐑦𐑕 ·𐑲𐑼𐑯 ·𐑒𐑮𐑬𐑯 𐑝 ·𐑤𐑴𐑥𐑚𐑸𐑛𐑦. 𐑘𐑧𐑑 𐑦𐑟 𐑦𐑑 𐑚𐑮𐑲𐑑 𐑢𐑦𐑞 𐑥𐑧𐑯𐑦 𐑩 𐑡𐑧𐑥; 𐑲 𐑞 𐑢𐑺𐑼, 𐑕𐑰 𐑯𐑪𐑑 𐑦𐑑𐑕 𐑓𐑸 𐑓𐑤𐑨𐑖𐑦𐑙𐑟; 𐑚𐑳𐑑 𐑛𐑸𐑒𐑤𐑦 𐑓𐑰𐑤 𐑞𐑨𐑑 𐑲 𐑢𐑺 𐑞𐑨𐑑, 𐑞𐑨𐑑 𐑛𐑨𐑟𐑩𐑤𐑦𐑙𐑤𐑦 𐑒𐑩𐑯𐑓𐑬𐑯𐑛𐑟. '𐑑𐑦𐑟 𐑲𐑼𐑯—𐑞𐑨𐑑 𐑲 𐑯𐑴—𐑯𐑪𐑑 𐑜𐑴𐑤𐑛. '𐑑𐑦𐑟 𐑕𐑐𐑤𐑦𐑑, 𐑑𐑵—𐑞𐑨𐑑 𐑲 𐑓𐑰𐑤; 𐑞 𐑡𐑨𐑜𐑩𐑛 𐑧𐑡 𐑜𐑷𐑤𐑟 𐑥𐑰 𐑕𐑴, 𐑥𐑲 𐑚𐑮𐑱𐑯 𐑕𐑰𐑥𐑟 𐑑 𐑚𐑰𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑕𐑪𐑤𐑦𐑛 𐑥𐑧𐑑𐑩𐑤; 𐑲, 𐑕𐑑𐑰𐑤 𐑕𐑒𐑳𐑤, 𐑥𐑲𐑯; 𐑞 𐑕𐑹𐑑 𐑞𐑨𐑑 𐑯𐑰𐑛𐑟 𐑯𐑴 𐑣𐑧𐑤𐑥𐑩𐑑 𐑦𐑯 𐑞 𐑥𐑴𐑕𐑑 𐑚𐑮𐑱𐑯-𐑚𐑨𐑑𐑼𐑦𐑙 𐑓𐑲𐑑!

𐑛𐑮𐑲 𐑣𐑰𐑑 𐑩𐑐𐑪𐑯 𐑥𐑲 𐑚𐑮𐑬? 𐑴! 𐑑𐑲𐑥 𐑢𐑪𐑟, 𐑢𐑧𐑯 𐑨𐑟 𐑞 𐑕𐑳𐑯𐑮𐑲𐑟 𐑯𐑴𐑚𐑤𐑦 𐑕𐑐𐑻𐑛 𐑥𐑰, 𐑕𐑴 𐑞 𐑕𐑳𐑯𐑕𐑧𐑑 𐑕𐑵𐑞𐑛. 𐑯𐑴 𐑥𐑹. 𐑞𐑦𐑕 𐑤𐑳𐑝𐑤𐑦 𐑤𐑲𐑑, 𐑦𐑑 𐑤𐑲𐑑𐑕 𐑯𐑪𐑑 𐑥𐑰; 𐑷𐑤 𐑤𐑳𐑝𐑤𐑦𐑯𐑩𐑕 𐑦𐑟 𐑨𐑙𐑜𐑢𐑦𐑖 𐑑 𐑥𐑰, 𐑕𐑦𐑯𐑕 𐑲 𐑒𐑨𐑯 𐑯𐑺 𐑦𐑯𐑡𐑶. 𐑜𐑦𐑓𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞 𐑣𐑲 𐑐𐑼𐑕𐑧𐑐𐑖𐑩𐑯, 𐑲 𐑤𐑨𐑒 𐑞 𐑤𐑴, 𐑦𐑯𐑡𐑶𐑦𐑙 𐑐𐑬𐑼; 𐑛𐑨𐑥𐑛, 𐑥𐑴𐑕𐑑 𐑕𐑳𐑑𐑤𐑦 𐑯 𐑥𐑴𐑕𐑑 𐑥𐑩𐑤𐑦𐑜𐑯𐑩𐑯𐑑𐑤𐑦! 𐑛𐑨𐑥𐑛 𐑦𐑯 𐑞 𐑥𐑦𐑛𐑕𐑑 𐑝 ·𐑐𐑨𐑮𐑩𐑛𐑲𐑕! 𐑜𐑫𐑛 𐑯𐑲𐑑—𐑜𐑫𐑛 𐑯𐑲𐑑! (𐑢𐑱𐑝𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛, 𐑣𐑰 𐑥𐑵𐑝𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑦𐑯𐑛𐑴.)

'𐑑𐑢𐑪𐑟 𐑯𐑪𐑑 𐑕𐑴 𐑣𐑸𐑛 𐑩 𐑑𐑭𐑕𐑒. 𐑲 𐑔𐑷𐑑 𐑑 𐑓𐑲𐑯𐑛 𐑢𐑳𐑯 𐑕𐑑𐑳𐑚𐑼𐑯, 𐑨𐑑 𐑞 𐑤𐑰𐑕𐑑; 𐑚𐑳𐑑 𐑥𐑲 𐑢𐑳𐑯 𐑒𐑪𐑜𐑛 𐑕𐑻𐑒𐑩𐑤 𐑓𐑦𐑑𐑕 𐑦𐑯𐑑𐑵 𐑷𐑤 𐑞𐑺 𐑝𐑺𐑾𐑕 𐑢𐑰𐑤𐑟, 𐑯 𐑞𐑱 𐑮𐑦𐑝𐑪𐑤𐑝. 𐑹, 𐑦𐑓 𐑿 𐑢𐑦𐑤, 𐑤𐑲𐑒 𐑕𐑴 𐑥𐑧𐑯𐑦 𐑨𐑯𐑑-𐑣𐑦𐑤𐑟 𐑝 𐑐𐑬𐑛𐑼, 𐑞𐑱 𐑷𐑤 𐑕𐑑𐑨𐑯𐑛 𐑚𐑦𐑓𐑹 𐑥𐑰; 𐑯 𐑲 𐑞𐑺 𐑥𐑨𐑗. 𐑴, 𐑣𐑸𐑛! 𐑞𐑨𐑑 𐑑 𐑓𐑲𐑼 𐑳𐑞𐑼𐑟, 𐑞 𐑥𐑨𐑗 𐑦𐑑𐑕𐑧𐑤𐑓 𐑥𐑳𐑕𐑑 𐑯𐑰𐑛𐑟 𐑚𐑰 𐑢𐑱𐑕𐑑𐑦𐑙! 𐑢𐑪𐑑 𐑲'𐑝 𐑛𐑺𐑛, 𐑲'𐑝 𐑢𐑦𐑤𐑛; 𐑯 𐑢𐑪𐑑 𐑲'𐑝 𐑢𐑦𐑤𐑛, 𐑲'𐑤 𐑛𐑵! 𐑞𐑱 𐑔𐑦𐑙𐑒 𐑥𐑰 𐑥𐑨𐑛—·𐑕𐑑𐑸𐑚𐑳𐑒 𐑛𐑳𐑟; 𐑚𐑳𐑑 𐑲'𐑥 𐑛𐑦𐑥𐑴𐑯𐑦𐑨𐑒, 𐑲 𐑨𐑥 𐑥𐑨𐑛𐑯𐑩𐑕 𐑥𐑨𐑛𐑩𐑯𐑛! 𐑞𐑨𐑑 𐑢𐑲𐑤𐑛 𐑥𐑨𐑛𐑯𐑩𐑕 𐑞𐑨𐑑'𐑕 𐑴𐑯𐑤𐑦 𐑒𐑭𐑥 𐑑 𐑒𐑪𐑥𐑐𐑮𐑦𐑣𐑧𐑯𐑛 𐑦𐑑𐑕𐑧𐑤𐑓! 𐑞 𐑐𐑮𐑪𐑓𐑩𐑕𐑦 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑲 𐑖𐑵𐑛 𐑚𐑰 𐑛𐑦𐑕𐑥𐑧𐑥𐑚𐑼𐑛; 𐑯—·𐑲! 𐑲 𐑤𐑪𐑕𐑑 𐑞𐑦𐑕 𐑤𐑧𐑜. 𐑲 𐑯𐑬 𐑐𐑮𐑪𐑓𐑩𐑕𐑲 𐑞𐑨𐑑 𐑲 𐑢𐑦𐑤 𐑛𐑦𐑕𐑥𐑧𐑥𐑚𐑼 𐑥𐑲 𐑛𐑦𐑕𐑥𐑧𐑥𐑚𐑼𐑼. 𐑯𐑬, 𐑞𐑧𐑯, 𐑚𐑰 𐑞 𐑐𐑮𐑪𐑓𐑩𐑑 𐑯 𐑞 𐑓𐑫𐑤𐑓𐑦𐑤𐑼 𐑢𐑳𐑯. 𐑞𐑨𐑑'𐑕 𐑥𐑹 𐑞𐑨𐑯 𐑘𐑰, 𐑘𐑰 𐑜𐑮𐑱𐑑 𐑜𐑪𐑛𐑟, 𐑧𐑝𐑼 𐑢𐑻. 𐑲 𐑤𐑭𐑓 𐑯 𐑣𐑵𐑑 𐑨𐑑 𐑘𐑰, 𐑘𐑰 𐑒𐑮𐑦𐑒𐑩𐑑-𐑐𐑤𐑱𐑼𐑟, 𐑘𐑰 𐑐𐑿𐑡𐑦𐑤𐑦𐑕𐑑𐑕, 𐑘𐑰 𐑛𐑧𐑓 ·𐑚𐑻𐑒𐑕 𐑯 𐑚𐑤𐑲𐑯𐑛𐑩𐑛 ·𐑚𐑧𐑯𐑛𐑦𐑜𐑴𐑟! 𐑲 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑕𐑱 𐑨𐑟 𐑕𐑒𐑵𐑤𐑚𐑶𐑟 𐑛𐑵 𐑑 𐑚𐑫𐑤𐑦𐑟—𐑑𐑱𐑒 𐑕𐑳𐑥 𐑢𐑳𐑯 𐑝 𐑘𐑹 𐑴𐑯 𐑕𐑲𐑟; 𐑛𐑴𐑯𐑑 𐑐𐑪𐑥𐑩𐑤 𐑥𐑰! 𐑯𐑴, 𐑘𐑰'𐑝 𐑯𐑪𐑒𐑑 𐑥𐑰 𐑛𐑬𐑯, 𐑯 𐑲 𐑨𐑥 𐑳𐑐 𐑩𐑜𐑱𐑯; 𐑚𐑳𐑑 𐑘𐑰 𐑣𐑨𐑝 𐑮𐑳𐑯 𐑯 𐑣𐑦𐑛𐑩𐑯. 𐑒𐑳𐑥 𐑓𐑹𐑔 𐑓𐑮𐑪𐑥 𐑚𐑦𐑣𐑲𐑯𐑛 𐑘𐑹 𐑒𐑪𐑑𐑩𐑯 𐑚𐑨𐑜𐑟! 𐑲 𐑣𐑨𐑝 𐑯𐑴 𐑤𐑪𐑙 𐑜𐑳𐑯 𐑑 𐑮𐑰𐑗 𐑘𐑰. 𐑒𐑳𐑥, ·𐑱𐑣𐑨𐑚'𐑟 𐑒𐑪𐑥𐑐𐑤𐑦𐑥𐑧𐑯𐑑𐑕 𐑑 𐑘𐑰; 𐑒𐑳𐑥 𐑯 𐑕𐑰 𐑦𐑓 𐑘𐑰 𐑒𐑨𐑯 𐑕𐑢𐑻𐑝 𐑥𐑰. 𐑕𐑢𐑻𐑝 𐑥𐑰? 𐑘𐑰 𐑒𐑨𐑯𐑪𐑑 𐑕𐑢𐑻𐑝 𐑥𐑰, 𐑧𐑤𐑕 𐑘𐑰 𐑕𐑢𐑻𐑝 𐑘𐑹𐑕𐑧𐑤𐑝𐑟! 𐑥𐑨𐑯 𐑣𐑨𐑟 𐑘𐑰 𐑞𐑺. 𐑕𐑢𐑻𐑝 𐑥𐑰? 𐑞 𐑐𐑭𐑔 𐑑 𐑥𐑲 𐑓𐑦𐑒𐑕𐑑 𐑐𐑻𐑐𐑩𐑕 𐑦𐑟 𐑤𐑱𐑛 𐑢𐑦𐑞 𐑲𐑼𐑯 𐑮𐑱𐑤𐑟, 𐑢𐑺𐑪𐑯 𐑥𐑲 𐑕𐑴𐑤 𐑦𐑟 𐑜𐑮𐑵𐑝𐑛 𐑑 𐑮𐑳𐑯. 𐑴𐑝𐑼 𐑳𐑯𐑕𐑬𐑯𐑛𐑩𐑛 𐑜𐑹𐑡𐑩𐑟, 𐑔𐑮𐑵 𐑞 𐑮𐑲𐑓𐑩𐑤𐑛 𐑣𐑸𐑑𐑕 𐑝 𐑥𐑬𐑯𐑑𐑩𐑯𐑟, 𐑳𐑯𐑛𐑼 𐑑𐑪𐑮𐑩𐑯𐑑𐑕' 𐑚𐑧𐑛𐑟, 𐑳𐑯𐑻𐑦𐑙𐑤𐑦 𐑲 𐑮𐑳𐑖! 𐑯𐑷𐑑'𐑕 𐑩𐑯 𐑪𐑚𐑕𐑑𐑩𐑒𐑩𐑤, 𐑯𐑷𐑑'𐑕 𐑩𐑯 𐑨𐑙𐑜𐑩𐑤 𐑑 𐑞 𐑲𐑼𐑯 𐑢𐑱!
