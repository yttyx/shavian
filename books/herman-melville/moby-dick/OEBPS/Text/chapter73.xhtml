“𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴, 𐑚𐑳𐑑 𐑲 𐑣𐑻𐑛 𐑞𐑨𐑑 𐑜𐑨𐑥𐑚𐑴𐑠 𐑜𐑴𐑕𐑑 𐑝 𐑩 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑕𐑱𐑦𐑙 𐑕𐑴, 𐑯 𐑣𐑰 𐑕𐑰𐑥𐑟 𐑑 𐑯𐑴 𐑷𐑤 𐑩𐑚𐑬𐑑 𐑖𐑦𐑐𐑕' 𐑗𐑸𐑥𐑟. 𐑚𐑳𐑑 𐑲 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑔𐑦𐑙𐑒 𐑣𐑰'𐑤 𐑗𐑸𐑥 𐑞 𐑖𐑦𐑐 𐑑 𐑯𐑴 𐑜𐑫𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑. 𐑲 𐑛𐑴𐑯𐑑 𐑣𐑭𐑓 𐑤𐑲𐑒 𐑞𐑨𐑑 𐑗𐑨𐑐, ·𐑕𐑑𐑳𐑚. 𐑛𐑦𐑛 𐑿 𐑧𐑝𐑼 𐑯𐑴𐑑𐑦𐑕 𐑣𐑬 𐑞𐑨𐑑 𐑑𐑳𐑕𐑒 𐑝 𐑣𐑦𐑟 𐑦𐑟 𐑩 𐑕𐑹𐑑 𐑝 𐑒𐑸𐑝𐑛 𐑦𐑯𐑑𐑵 𐑩 𐑕𐑯𐑱𐑒'𐑕 𐑣𐑧𐑛, ·𐑕𐑑𐑳𐑚?”

“𐑕𐑦𐑙𐑒 𐑣𐑦𐑥! 𐑲 𐑯𐑧𐑝𐑼 𐑤𐑫𐑒 𐑨𐑑 𐑣𐑦𐑥 𐑨𐑑 𐑷𐑤; 𐑚𐑳𐑑 𐑦𐑓 𐑧𐑝𐑼 𐑲 𐑜𐑧𐑑 𐑩 𐑗𐑭𐑯𐑕 𐑝 𐑩 𐑛𐑸𐑒 𐑯𐑲𐑑, 𐑯 𐑣𐑰 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑣𐑸𐑛 𐑚𐑲 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑯 𐑯𐑴 𐑢𐑳𐑯 𐑚𐑲; 𐑤𐑫𐑒 𐑛𐑬𐑯 𐑞𐑺, ·𐑓𐑤𐑭𐑕𐑒”—𐑐𐑶𐑯𐑑𐑦𐑙 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰 𐑢𐑦𐑞 𐑩 𐑐𐑦𐑒𐑿𐑤𐑽 𐑥𐑴𐑖𐑩𐑯 𐑝 𐑚𐑴𐑔 𐑣𐑨𐑯𐑛𐑟—“𐑲, 𐑢𐑦𐑤 𐑲! 𐑓𐑤𐑭𐑕𐑒, 𐑲 𐑑𐑱𐑒 𐑞𐑨𐑑 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑑 𐑚𐑰 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑦𐑯 𐑛𐑦𐑕𐑜𐑲𐑟. 𐑛𐑵 𐑿 𐑚𐑦𐑤𐑰𐑝 𐑞𐑨𐑑 𐑒𐑪𐑒 𐑯 𐑚𐑫𐑤 𐑕𐑑𐑹𐑦 𐑩𐑚𐑬𐑑 𐑣𐑦𐑟 𐑣𐑨𐑝𐑦𐑙 𐑚𐑰𐑯 𐑕𐑑𐑴𐑛 𐑩𐑢𐑱 𐑪𐑯 𐑚𐑹𐑛 𐑖𐑦𐑐? 𐑣𐑰'𐑟 𐑞 𐑛𐑧𐑝𐑩𐑤, 𐑲 𐑕𐑱. 𐑞 𐑮𐑰𐑟𐑩𐑯 𐑢𐑲 𐑿 𐑛𐑴𐑯𐑑 𐑕𐑰 𐑣𐑦𐑟 𐑑𐑱𐑤, 𐑦𐑟 𐑚𐑦𐑒𐑪𐑟 𐑣𐑰 𐑑𐑳𐑒𐑕 𐑦𐑑 𐑳𐑐 𐑬𐑑 𐑝 𐑕𐑲𐑑; 𐑣𐑰 𐑒𐑨𐑮𐑦𐑟 𐑦𐑑 𐑒𐑶𐑤𐑛 𐑩𐑢𐑱 𐑦𐑯 𐑣𐑦𐑟 𐑐𐑪𐑒𐑩𐑑, 𐑲 𐑜𐑧𐑕. 𐑚𐑤𐑭𐑕𐑑 𐑣𐑦𐑥! 𐑯𐑬 𐑞𐑨𐑑 𐑲 𐑔𐑦𐑙𐑒 𐑝 𐑦𐑑, 𐑣𐑰'𐑟 𐑷𐑤𐑢𐑱𐑟 𐑢𐑪𐑯𐑑𐑦𐑙 𐑴𐑒𐑩𐑥 𐑑 𐑕𐑑𐑳𐑓 𐑦𐑯𐑑𐑵 𐑞 𐑑𐑴𐑟 𐑝 𐑣𐑦𐑟 𐑚𐑵𐑑𐑕.”

“𐑣𐑰 𐑕𐑤𐑰𐑐𐑕 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑵𐑑𐑕, 𐑛𐑴𐑯𐑑 𐑣𐑰? 𐑣𐑰 𐑣𐑨𐑟𐑩𐑯𐑑 𐑜𐑪𐑑 𐑧𐑯𐑦 𐑣𐑨𐑥𐑩𐑒; 𐑚𐑳𐑑 𐑲'𐑝 𐑕𐑰𐑯 𐑣𐑦𐑥 𐑤𐑱 𐑝 𐑯𐑲𐑑𐑕 𐑦𐑯 𐑩 𐑒𐑶𐑤 𐑝 𐑮𐑦𐑜𐑦𐑙.”

“𐑯𐑴 𐑛𐑬𐑑, 𐑯 𐑦𐑑'𐑕 𐑚𐑦𐑒𐑪𐑟 𐑝 𐑣𐑦𐑟 𐑒𐑻𐑕𐑑 𐑑𐑱𐑤; 𐑣𐑰 𐑒𐑶𐑤𐑟 𐑦𐑑 𐑛𐑬𐑯, 𐑛𐑵 𐑘𐑰 𐑕𐑰, 𐑦𐑯 𐑞 𐑲 𐑝 𐑞 𐑮𐑦𐑜𐑦𐑙.”

“𐑢𐑪𐑑'𐑕 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑣𐑨𐑝 𐑕𐑴 𐑥𐑳𐑗 𐑑 𐑛𐑵 𐑢𐑦𐑞 𐑣𐑦𐑥 𐑓?”

“𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑳𐑐 𐑩 𐑕𐑢𐑪𐑐 𐑹 𐑩 𐑚𐑸𐑜𐑦𐑯, 𐑲 𐑕𐑩𐑐𐑴𐑟.”

“𐑚𐑸𐑜𐑦𐑯?—𐑩𐑚𐑬𐑑 𐑢𐑪𐑑?”

“𐑢𐑲, 𐑛𐑵 𐑘𐑰 𐑕𐑰, 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑦𐑟 𐑣𐑸𐑛 𐑚𐑧𐑯𐑑 𐑭𐑓𐑑𐑼 𐑞𐑨𐑑 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤, 𐑯 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑞𐑺 𐑦𐑟 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑒𐑳𐑥 𐑮𐑬𐑯𐑛 𐑣𐑦𐑥, 𐑯 𐑜𐑧𐑑 𐑣𐑦𐑥 𐑑 𐑕𐑢𐑪𐑐 𐑩𐑢𐑱 𐑣𐑦𐑟 𐑕𐑦𐑤𐑝𐑼 𐑢𐑪𐑗, 𐑹 𐑣𐑦𐑟 𐑕𐑴𐑤, 𐑹 𐑕𐑳𐑥𐑔𐑦𐑙 𐑝 𐑞𐑨𐑑 𐑕𐑹𐑑, 𐑯 𐑞𐑧𐑯 𐑣𐑰'𐑤 𐑕𐑼𐑧𐑯𐑛𐑼 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒.”

“𐑐𐑵! 𐑕𐑑𐑳𐑚, 𐑿 𐑸 𐑕𐑒𐑲𐑤𐑸𐑒𐑦𐑙; 𐑣𐑬 𐑒𐑨𐑯 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑛𐑵 𐑞𐑨𐑑?”

“𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴, ·𐑓𐑤𐑭𐑕𐑒, 𐑚𐑳𐑑 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑦𐑟 𐑩 𐑒𐑘𐑫𐑼𐑾𐑕 𐑗𐑨𐑐, 𐑯 𐑩 𐑢𐑦𐑒𐑩𐑛 𐑢𐑳𐑯, 𐑲 𐑑𐑧𐑤 𐑘𐑰. 𐑢𐑲, 𐑞𐑱 𐑕𐑱 𐑨𐑟 𐑣𐑬 𐑣𐑰 𐑢𐑧𐑯𐑑 𐑩 𐑕𐑷𐑯𐑑𐑼𐑦𐑙 𐑦𐑯𐑑𐑵 𐑞 𐑴𐑤𐑛 𐑓𐑤𐑨𐑜-𐑖𐑦𐑐 𐑢𐑳𐑯𐑕, 𐑕𐑢𐑦𐑗𐑦𐑙 𐑣𐑦𐑟 𐑑𐑱𐑤 𐑩𐑚𐑬𐑑 𐑛𐑧𐑝𐑩𐑤𐑦𐑖 𐑰𐑟𐑦 𐑯 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯𐑤𐑲𐑒, 𐑯 𐑦𐑯𐑒𐑢𐑲𐑼𐑦𐑙 𐑦𐑓 𐑞 𐑴𐑤𐑛 𐑜𐑳𐑝𐑼𐑯𐑼 𐑢𐑪𐑟 𐑨𐑑 𐑣𐑴𐑥. 𐑢𐑧𐑤, 𐑣𐑰 𐑢𐑪𐑟 𐑨𐑑 𐑣𐑴𐑥, 𐑯 𐑭𐑕𐑒𐑑 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑢𐑪𐑑 𐑣𐑰 𐑢𐑪𐑯𐑑𐑩𐑛. 𐑞 𐑛𐑧𐑝𐑩𐑤, 𐑕𐑢𐑦𐑗𐑦𐑙 𐑣𐑦𐑟 𐑣𐑵𐑓𐑕, 𐑳𐑐 𐑯 𐑕𐑧𐑟, '𐑲 𐑢𐑪𐑯𐑑 ·𐑡𐑪𐑯.' '𐑢𐑪𐑑 𐑓?' 𐑕𐑧𐑟 𐑞 𐑴𐑤𐑛 𐑜𐑳𐑝𐑼𐑯𐑼. '𐑢𐑪𐑑 𐑚𐑦𐑟𐑯𐑩𐑕 𐑦𐑟 𐑞𐑨𐑑 𐑝 𐑘𐑹𐑟,' 𐑕𐑧𐑟 𐑞 𐑛𐑧𐑝𐑩𐑤, 𐑜𐑧𐑑𐑦𐑙 𐑥𐑨𐑛,—'𐑲 𐑢𐑪𐑯𐑑 𐑑 𐑿𐑟 𐑣𐑦𐑥.' '𐑑𐑱𐑒 𐑣𐑦𐑥,' 𐑕𐑧𐑟 𐑞 𐑜𐑳𐑝𐑼𐑯𐑼—𐑯 𐑚𐑲 𐑞 ·𐑤𐑹𐑛, ·𐑓𐑤𐑭𐑕𐑒, 𐑦𐑓 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑛𐑦𐑛𐑩𐑯𐑑 𐑜𐑦𐑝 ·𐑡𐑪𐑯 𐑞 ·𐑱𐑠𐑦𐑨𐑑𐑦𐑒 𐑒𐑪𐑤𐑼𐑩 𐑚𐑦𐑓𐑹 𐑣𐑰 𐑜𐑪𐑑 𐑔𐑮𐑵 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑲'𐑤 𐑰𐑑 𐑞𐑦𐑕 𐑢𐑱𐑤 𐑦𐑯 𐑢𐑳𐑯 𐑥𐑬𐑔𐑓𐑫𐑤. 𐑚𐑳𐑑 𐑤𐑫𐑒 𐑖𐑸𐑐—𐑱𐑯𐑑 𐑿 𐑷𐑤 𐑮𐑧𐑛𐑦 𐑞𐑺? 𐑢𐑧𐑤, 𐑞𐑧𐑯, 𐑐𐑫𐑤 𐑩𐑣𐑧𐑛, 𐑯 𐑤𐑧𐑑'𐑕 𐑜𐑧𐑑 𐑞 𐑢𐑱𐑤 𐑩𐑤𐑪𐑙𐑕𐑲𐑛.”

“𐑲 𐑔𐑦𐑙𐑒 𐑲 𐑮𐑦𐑥𐑧𐑥𐑚𐑼 𐑕𐑳𐑥 𐑕𐑳𐑗 𐑕𐑑𐑹𐑦 𐑨𐑟 𐑿 𐑢𐑻 𐑑𐑧𐑤𐑦𐑙,” 𐑕𐑧𐑛 ·𐑓𐑤𐑭𐑕𐑒, 𐑢𐑧𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑞 𐑑𐑵 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑕𐑤𐑴𐑤𐑦 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑢𐑦𐑞 𐑞𐑺 𐑚𐑻𐑛𐑩𐑯 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑖𐑦𐑐, “𐑚𐑳𐑑 𐑲 𐑒𐑭𐑯𐑑 𐑮𐑦𐑥𐑧𐑥𐑚𐑼 𐑢𐑺.”

“𐑔𐑮𐑰 ·𐑕𐑐𐑨𐑯𐑘𐑼𐑛𐑟? 𐑩𐑛𐑝𐑧𐑯𐑗𐑼𐑟 𐑝 𐑞𐑴𐑟 𐑔𐑮𐑰 𐑚𐑤𐑳𐑛𐑦-𐑥𐑲𐑯𐑛𐑩𐑛 𐑕𐑴𐑤𐑛𐑱𐑛𐑴𐑟? 𐑛𐑦𐑛 𐑘𐑰 𐑮𐑰𐑛 𐑦𐑑 𐑞𐑺, ·𐑓𐑤𐑭𐑕𐑒? 𐑲 𐑜𐑧𐑕 𐑘𐑰 𐑛𐑦𐑛?”

“𐑯𐑴: 𐑯𐑧𐑝𐑼 𐑕𐑷 𐑕𐑳𐑗 𐑩 𐑚𐑫𐑒; 𐑣𐑻𐑛 𐑝 𐑦𐑑, 𐑞𐑴. 𐑚𐑳𐑑 𐑯𐑬, 𐑑𐑧𐑤 𐑥𐑰, ·𐑕𐑑𐑳𐑚, 𐑛𐑵 𐑿 𐑕𐑩𐑐𐑴𐑟 𐑞𐑨𐑑 𐑞𐑨𐑑 𐑛𐑧𐑝𐑩𐑤 𐑿 𐑢𐑪𐑟 𐑕𐑐𐑰𐑒𐑦𐑙 𐑝 𐑡𐑳𐑕𐑑 𐑯𐑬, 𐑢𐑪𐑟 𐑞 𐑕𐑱𐑥 𐑿 𐑕𐑱 𐑦𐑟 𐑯𐑬 𐑪𐑯 𐑚𐑹𐑛 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛?”

“𐑨𐑥 𐑲 𐑞 𐑕𐑱𐑥 𐑥𐑨𐑯 𐑞𐑨𐑑 𐑣𐑧𐑤𐑐𐑑 𐑒𐑦𐑤 𐑞𐑦𐑕 𐑢𐑱𐑤? 𐑛𐑳𐑟𐑩𐑯𐑑 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑤𐑲𐑝 𐑓 𐑧𐑝𐑼; 𐑣𐑵 𐑧𐑝𐑼 𐑣𐑻𐑛 𐑞𐑨𐑑 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑢𐑪𐑟 𐑛𐑧𐑛? 𐑛𐑦𐑛 𐑿 𐑧𐑝𐑼 𐑕𐑰 𐑧𐑯𐑦 𐑐𐑸𐑟𐑩𐑯 𐑩 𐑢𐑺𐑦𐑙 𐑥𐑹𐑯𐑦𐑙 𐑓 𐑞 𐑛𐑧𐑝𐑩𐑤? 𐑯 𐑦𐑓 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑣𐑨𐑟 𐑩 𐑤𐑨𐑗-𐑒𐑰 𐑑 𐑜𐑧𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑨𐑛𐑥𐑼𐑩𐑤'𐑟 𐑒𐑨𐑚𐑦𐑯, 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑩𐑐𐑴𐑟 𐑣𐑰 𐑒𐑨𐑯 𐑒𐑮𐑷𐑤 𐑦𐑯𐑑𐑵 𐑩 𐑐𐑹𐑑𐑣𐑴𐑤? 𐑑𐑧𐑤 𐑥𐑰 𐑞𐑨𐑑, ·𐑥𐑼 ·𐑓𐑤𐑭𐑕𐑒?”

“𐑣𐑬 𐑴𐑤𐑛 𐑛𐑵 𐑿 𐑕𐑩𐑐𐑴𐑟 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑦𐑟, ·𐑕𐑑𐑳𐑚?”

“𐑛𐑵 𐑿 𐑕𐑰 𐑞𐑨𐑑 𐑥𐑱𐑯𐑥𐑭𐑕𐑑 𐑞𐑺?” 𐑐𐑶𐑯𐑑𐑦𐑙 𐑑 𐑞 𐑖𐑦𐑐; “𐑢𐑧𐑤, 𐑞𐑨𐑑'𐑕 𐑞 𐑓𐑦𐑜𐑼 𐑢𐑳𐑯; 𐑯𐑬 𐑑𐑱𐑒 𐑷𐑤 𐑞 𐑣𐑵𐑐𐑕 𐑦𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑣𐑴𐑤𐑛, 𐑯 𐑕𐑑𐑮𐑦𐑙 𐑩𐑤𐑪𐑙 𐑦𐑯 𐑩 𐑮𐑴 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑥𐑭𐑕𐑑, 𐑓 𐑷𐑑𐑕, 𐑛𐑵 𐑿 𐑕𐑰; 𐑢𐑧𐑤, 𐑞𐑨𐑑 𐑢𐑫𐑛𐑩𐑯𐑑 𐑚𐑦𐑜𐑦𐑯 𐑑 𐑚𐑰 ·𐑓𐑧𐑛𐑩𐑤𐑩'𐑟 𐑱𐑡. 𐑯𐑹 𐑷𐑤 𐑞 𐑒𐑵𐑐𐑼𐑟 𐑦𐑯 𐑒𐑮𐑦𐑱𐑖𐑩𐑯 𐑒𐑫𐑛𐑩𐑯𐑑 𐑖𐑴 𐑣𐑵𐑐𐑕 𐑦𐑯𐑳𐑓 𐑑 𐑥𐑱𐑒 𐑷𐑑𐑕 𐑦𐑯𐑳𐑓.”

“𐑚𐑳𐑑 𐑕𐑰 𐑣𐑽, ·𐑕𐑑𐑳𐑚, 𐑲 𐑔𐑷𐑑 𐑿 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑚𐑴𐑕𐑑𐑩𐑛 𐑡𐑳𐑕𐑑 𐑯𐑬, 𐑞𐑨𐑑 𐑿 𐑥𐑧𐑯𐑑 𐑑 𐑜𐑦𐑝 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑩 𐑕𐑰-𐑑𐑪𐑕, 𐑦𐑓 𐑿 𐑜𐑪𐑑 𐑩 𐑜𐑫𐑛 𐑗𐑭𐑯𐑕. 𐑯𐑬, 𐑦𐑓 𐑣𐑰'𐑟 𐑕𐑴 𐑴𐑤𐑛 𐑨𐑟 𐑷𐑤 𐑞𐑴𐑟 𐑣𐑵𐑐𐑕 𐑝 𐑘𐑹𐑟 𐑒𐑳𐑥 𐑑, 𐑯 𐑦𐑓 𐑣𐑰 𐑦𐑟 𐑜𐑴𐑦𐑙 𐑑 𐑤𐑦𐑝 𐑓 𐑧𐑝𐑼, 𐑢𐑪𐑑 𐑜𐑫𐑛 𐑢𐑦𐑤 𐑦𐑑 𐑛𐑵 𐑑 𐑐𐑦𐑗 𐑣𐑦𐑥 𐑴𐑝𐑼𐑚𐑹𐑛—𐑑𐑧𐑤 𐑥𐑰 𐑞𐑨𐑑?

“𐑜𐑦𐑝 𐑣𐑦𐑥 𐑩 𐑜𐑫𐑛 𐑛𐑳𐑒𐑦𐑙, 𐑧𐑯𐑦𐑣𐑬.”

“𐑚𐑳𐑑 𐑣𐑰'𐑛 𐑒𐑮𐑷𐑤 𐑚𐑨𐑒.”

“𐑛𐑳𐑒 𐑣𐑦𐑥 𐑩𐑜𐑱𐑯; 𐑯 𐑒𐑰𐑐 𐑛𐑳𐑒𐑦𐑙 𐑣𐑦𐑥.”

“𐑕𐑩𐑐𐑴𐑟 𐑣𐑰 𐑖𐑵𐑛 𐑑𐑱𐑒 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑑 𐑛𐑳𐑒 𐑿, 𐑞𐑴—𐑘𐑧𐑕, 𐑯 𐑛𐑮𐑬𐑯 𐑿—𐑢𐑪𐑑 𐑞𐑧𐑯?”

“𐑲 𐑖𐑵𐑛 𐑤𐑲𐑒 𐑑 𐑕𐑰 𐑣𐑦𐑥 𐑑𐑮𐑲 𐑦𐑑; 𐑲'𐑛 𐑜𐑦𐑝 𐑣𐑦𐑥 𐑕𐑳𐑗 𐑩 𐑐𐑺 𐑝 𐑚𐑤𐑨𐑒 𐑲𐑟 𐑞𐑨𐑑 𐑣𐑰 𐑢𐑫𐑛𐑩𐑯𐑑 𐑛𐑺 𐑑 𐑖𐑴 𐑣𐑦𐑟 𐑓𐑱𐑕 𐑦𐑯 𐑞 𐑨𐑛𐑥𐑼𐑩𐑤'𐑟 𐑒𐑨𐑚𐑦𐑯 𐑩𐑜𐑱𐑯 𐑓 𐑩 𐑤𐑪𐑙 𐑢𐑲𐑤, 𐑤𐑧𐑑 𐑩𐑤𐑴𐑯 𐑛𐑬𐑯 𐑦𐑯 𐑞 𐑹𐑤𐑪𐑐 𐑞𐑺, 𐑢𐑺 𐑣𐑰 𐑤𐑦𐑝𐑟, 𐑯 𐑣𐑽𐑩𐑚𐑬𐑑𐑕 𐑪𐑯 𐑞 𐑳𐑐𐑼 𐑛𐑧𐑒𐑕 𐑢𐑺 𐑣𐑰 𐑕𐑯𐑰𐑒𐑕 𐑕𐑴 𐑥𐑳𐑗. 𐑛𐑨𐑥 𐑞 𐑛𐑧𐑝𐑩𐑤, ·𐑓𐑤𐑭𐑕𐑒; 𐑕𐑴 𐑿 𐑕𐑩𐑐𐑴𐑟 𐑲'𐑥 𐑩𐑓𐑮𐑱𐑛 𐑝 𐑞 𐑛𐑧𐑝𐑩𐑤? 𐑣𐑵'𐑟 𐑩𐑓𐑮𐑱𐑛 𐑝 𐑣𐑦𐑥, 𐑦𐑒𐑕𐑧𐑐𐑑 𐑞 𐑴𐑤𐑛 𐑜𐑳𐑝𐑼𐑯𐑼 𐑣𐑵 𐑛𐑺𐑟𐑩𐑯𐑑 𐑒𐑨𐑗 𐑣𐑦𐑥 𐑯 𐑐𐑫𐑑 𐑣𐑦𐑥 𐑦𐑯 𐑛𐑳𐑚𐑩𐑤-𐑛𐑸𐑚𐑦𐑟, 𐑨𐑟 𐑣𐑰 𐑛𐑦𐑟𐑻𐑝𐑟, 𐑚𐑳𐑑 𐑤𐑧𐑑𐑕 𐑣𐑦𐑥 𐑜𐑴 𐑩𐑚𐑬𐑑 𐑒𐑦𐑛𐑯𐑨𐑐𐑦𐑙 𐑐𐑰𐑐𐑩𐑤; 𐑲, 𐑯 𐑕𐑲𐑯𐑛 𐑩 𐑚𐑪𐑯𐑛 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑞𐑨𐑑 𐑷𐑤 𐑞 𐑐𐑰𐑐𐑩𐑤 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑒𐑦𐑛𐑯𐑨𐑐𐑑, 𐑣𐑰'𐑛 𐑮𐑴𐑕𐑑 𐑓 𐑣𐑦𐑥? 𐑞𐑺'𐑟 𐑩 𐑜𐑳𐑝𐑼𐑯𐑼!”

“𐑛𐑵 𐑿 𐑕𐑩𐑐𐑴𐑟 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑢𐑪𐑯𐑑𐑕 𐑑 𐑒𐑦𐑛𐑯𐑨𐑐 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚?”

“𐑛𐑵 𐑲 𐑕𐑩𐑐𐑴𐑟 𐑦𐑑? 𐑿'𐑤 𐑯𐑴 𐑦𐑑 𐑚𐑦𐑓𐑹 𐑤𐑪𐑙, ·𐑓𐑤𐑭𐑕𐑒. 𐑚𐑳𐑑 𐑲 𐑨𐑥 𐑜𐑴𐑦𐑙 𐑯𐑬 𐑑 𐑒𐑰𐑐 𐑩 𐑖𐑸𐑐 𐑤𐑫𐑒-𐑬𐑑 𐑪𐑯 𐑣𐑦𐑥; 𐑯 𐑦𐑓 𐑲 𐑕𐑰 𐑧𐑯𐑦𐑔𐑦𐑙 𐑝𐑧𐑮𐑦 𐑕𐑩𐑕𐑐𐑦𐑖𐑩𐑕 𐑜𐑴𐑦𐑙 𐑪𐑯, 𐑲'𐑤 𐑡𐑳𐑕𐑑 𐑑𐑱𐑒 𐑣𐑦𐑥 𐑚𐑲 𐑞 𐑯𐑱𐑐 𐑝 𐑣𐑦𐑟 𐑯𐑧𐑒, 𐑯 𐑕𐑱—𐑤𐑫𐑒 𐑣𐑽, ·𐑚𐑰𐑧𐑤𐑟𐑦𐑚𐑳𐑚, 𐑿 𐑛𐑴𐑯𐑑 𐑛𐑵 𐑦𐑑; 𐑯 𐑦𐑓 𐑣𐑰 𐑥𐑱𐑒𐑕 𐑧𐑯𐑦 𐑓𐑳𐑕, 𐑚𐑲 𐑞 ·𐑤𐑹𐑛 𐑲'𐑤 𐑥𐑱𐑒 𐑩 𐑜𐑮𐑨𐑚 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑐𐑪𐑒𐑩𐑑 𐑓 𐑣𐑦𐑟 𐑑𐑱𐑤, 𐑑𐑱𐑒 𐑦𐑑 𐑑 𐑞 𐑒𐑨𐑐𐑕𐑑𐑩𐑯, 𐑯 𐑜𐑦𐑝 𐑣𐑦𐑥 𐑕𐑳𐑗 𐑩 𐑮𐑧𐑯𐑗𐑦𐑙 𐑯 𐑣𐑰𐑝𐑦𐑙, 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑑𐑱𐑤 𐑢𐑦𐑤 𐑒𐑳𐑥 𐑖𐑹𐑑 𐑪𐑓 𐑨𐑑 𐑞 𐑕𐑑𐑳𐑥𐑐—𐑛𐑵 𐑿 𐑕𐑰; 𐑯 𐑞𐑧𐑯, 𐑲 𐑮𐑭𐑞𐑼 𐑜𐑧𐑕 𐑢𐑧𐑯 𐑣𐑰 𐑓𐑲𐑯𐑛𐑟 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑛𐑪𐑒𐑑 𐑦𐑯 𐑞𐑨𐑑 𐑒𐑢𐑽 𐑓𐑨𐑖𐑩𐑯, 𐑣𐑰'𐑤 𐑕𐑯𐑰𐑒 𐑪𐑓 𐑢𐑦𐑞𐑬𐑑 𐑞 𐑐𐑫𐑼 𐑕𐑨𐑑𐑦𐑕𐑓𐑨𐑒𐑖𐑩𐑯 𐑝 𐑓𐑰𐑤𐑦𐑙 𐑣𐑦𐑟 𐑑𐑱𐑤 𐑚𐑦𐑑𐑢𐑰𐑯 𐑣𐑦𐑟 𐑤𐑧𐑜𐑟.”

“𐑯 𐑢𐑪𐑑 𐑢𐑦𐑤 𐑿 𐑛𐑵 𐑢𐑦𐑞 𐑞 𐑑𐑱𐑤, ·𐑕𐑑𐑳𐑚?”

“𐑛𐑵 𐑢𐑦𐑞 𐑦𐑑? 𐑕𐑧𐑤 𐑦𐑑 𐑓 𐑩𐑯 𐑪𐑒𐑕 𐑢𐑦𐑐 𐑢𐑧𐑯 𐑢𐑰 𐑜𐑧𐑑 𐑣𐑴𐑥;—𐑢𐑪𐑑 𐑧𐑤𐑕?”

“𐑯𐑬, 𐑛𐑵 𐑿 𐑥𐑰𐑯 𐑢𐑪𐑑 𐑿 𐑕𐑱, 𐑯 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑕𐑱𐑦𐑙 𐑷𐑤 𐑩𐑤𐑪𐑙, ·𐑕𐑑𐑳𐑚?”

“𐑥𐑰𐑯 𐑹 𐑯𐑪𐑑 𐑥𐑰𐑯, 𐑣𐑽 𐑢𐑰 𐑸 𐑨𐑑 𐑞 𐑖𐑦𐑐.”

𐑞 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑣𐑽 𐑣𐑱𐑤𐑛, 𐑑 𐑑𐑴 𐑞 𐑢𐑱𐑤 𐑪𐑯 𐑞 𐑤𐑸𐑚𐑼𐑛 𐑕𐑲𐑛, 𐑢𐑺 𐑓𐑤𐑵𐑒 𐑗𐑱𐑯𐑟 𐑯 𐑳𐑞𐑼 𐑯𐑧𐑕𐑦𐑕𐑺𐑦𐑟 𐑢𐑻 𐑷𐑤𐑮𐑧𐑛𐑦 𐑐𐑮𐑦𐑐𐑺𐑛 𐑓 𐑕𐑦𐑒𐑘𐑫𐑼𐑦𐑙 𐑣𐑦𐑥.

“𐑛𐑦𐑛𐑩𐑯𐑑 𐑲 𐑑𐑧𐑤 𐑿 𐑕𐑴?” 𐑕𐑧𐑛 ·𐑓𐑤𐑭𐑕𐑒; “𐑘𐑧𐑕, 𐑿'𐑤 𐑕𐑵𐑯 𐑕𐑰 𐑞𐑦𐑕 𐑮𐑲𐑑 𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛 𐑣𐑶𐑕𐑑𐑩𐑛 𐑳𐑐 𐑪𐑐𐑩𐑟𐑦𐑑 𐑞𐑨𐑑 ·𐑐𐑸𐑥𐑩𐑕𐑧𐑑𐑦'𐑟.”

𐑦𐑯 𐑜𐑫𐑛 𐑑𐑲𐑥, ·𐑓𐑤𐑭𐑕𐑒'𐑕 𐑕𐑱𐑦𐑙 𐑐𐑮𐑵𐑝𐑛 𐑑𐑮𐑵. 𐑨𐑟 𐑚𐑦𐑓𐑹, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑕𐑑𐑰𐑐𐑤𐑦 𐑤𐑰𐑯𐑛 𐑴𐑝𐑼 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛, 𐑯𐑬, 𐑚𐑲 𐑞 𐑒𐑬𐑯𐑑𐑼𐑐𐑶𐑟 𐑝 𐑚𐑴𐑔 𐑣𐑧𐑛𐑟, 𐑖𐑰 𐑮𐑦𐑜𐑱𐑯𐑛 𐑣𐑻 𐑰𐑝𐑩𐑯 𐑒𐑰𐑤; 𐑞𐑴 𐑕𐑹𐑤𐑦 𐑕𐑑𐑮𐑱𐑯𐑛, 𐑿 𐑥𐑱 𐑢𐑧𐑤 𐑚𐑦𐑤𐑰𐑝. 𐑕𐑴, 𐑢𐑧𐑯 𐑪𐑯 𐑢𐑳𐑯 𐑕𐑲𐑛 𐑿 𐑣𐑶𐑕𐑑 𐑦𐑯 ·𐑤𐑪𐑒'𐑕 𐑣𐑧𐑛, 𐑿 𐑜𐑴 𐑴𐑝𐑼 𐑞𐑨𐑑 𐑢𐑱; 𐑚𐑳𐑑 𐑯𐑬, 𐑪𐑯 𐑞 𐑳𐑞𐑼 𐑕𐑲𐑛, 𐑣𐑶𐑕𐑑 𐑦𐑯 ·𐑒𐑭𐑯𐑑'𐑕 𐑯 𐑿 𐑒𐑳𐑥 𐑚𐑨𐑒 𐑩𐑜𐑱𐑯; 𐑚𐑳𐑑 𐑦𐑯 𐑝𐑧𐑮𐑦 𐑐𐑫𐑼 𐑐𐑤𐑲𐑑. 𐑞𐑳𐑕, 𐑕𐑳𐑥 𐑥𐑲𐑯𐑛𐑟 𐑓 𐑧𐑝𐑼 𐑒𐑰𐑐 𐑑𐑮𐑦𐑥𐑦𐑙 𐑚𐑴𐑑. 𐑴, 𐑘𐑰 𐑓𐑵𐑤𐑦𐑖! 𐑔𐑮𐑴 𐑷𐑤 𐑞𐑰𐑟 𐑔𐑳𐑯𐑛𐑼-𐑣𐑧𐑛𐑟 𐑴𐑝𐑼𐑚𐑹𐑛, 𐑯 𐑞𐑧𐑯 𐑿 𐑢𐑦𐑤 𐑓𐑤𐑴𐑑 𐑤𐑲𐑑 𐑯 𐑮𐑲𐑑.

𐑦𐑯 𐑛𐑦𐑕𐑐𐑴𐑟𐑦𐑙 𐑝 𐑞 𐑚𐑪𐑛𐑦 𐑝 𐑩 𐑮𐑲𐑑 𐑢𐑱𐑤, 𐑢𐑧𐑯 𐑚𐑮𐑷𐑑 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑞 𐑖𐑦𐑐, 𐑞 𐑕𐑱𐑥 𐑐𐑮𐑦𐑤𐑦𐑥𐑦𐑯𐑼𐑦 𐑐𐑮𐑴𐑕𐑰𐑛𐑦𐑙𐑟 𐑒𐑪𐑥𐑩𐑯𐑤𐑦 𐑑𐑱𐑒 𐑐𐑤𐑱𐑕 𐑨𐑟 𐑦𐑯 𐑞 𐑒𐑱𐑕 𐑝 𐑩 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤; 𐑴𐑯𐑤𐑦, 𐑦𐑯 𐑞 𐑤𐑨𐑑𐑼 𐑦𐑯𐑕𐑑𐑩𐑯𐑕, 𐑞 𐑣𐑧𐑛 𐑦𐑟 𐑒𐑳𐑑 𐑪𐑓 𐑣𐑴𐑤, 𐑚𐑳𐑑 𐑦𐑯 𐑞 𐑓𐑹𐑥𐑼 𐑞 𐑤𐑦𐑐𐑕 𐑯 𐑑𐑳𐑙 𐑸 𐑕𐑧𐑐𐑼𐑩𐑑𐑤𐑦 𐑮𐑦𐑥𐑵𐑝𐑛 𐑯 𐑣𐑶𐑕𐑑𐑩𐑛 𐑪𐑯 𐑛𐑧𐑒, 𐑢𐑦𐑞 𐑷𐑤 𐑞 𐑢𐑧𐑤 𐑯𐑴𐑯 𐑚𐑤𐑨𐑒 𐑚𐑴𐑯 𐑩𐑑𐑨𐑗𐑑 𐑑 𐑢𐑪𐑑 𐑦𐑟 𐑒𐑷𐑤𐑛 𐑞 𐑒𐑮𐑬𐑯-𐑐𐑰𐑕. 𐑚𐑳𐑑 𐑯𐑳𐑔𐑦𐑙 𐑤𐑲𐑒 𐑞𐑦𐑕, 𐑦𐑯 𐑞 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑒𐑱𐑕, 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑛𐑳𐑯. 𐑞 𐑒𐑸𐑒𐑩𐑕𐑩𐑟 𐑝 𐑚𐑴𐑔 𐑢𐑱𐑤𐑟 𐑣𐑨𐑛 𐑛𐑮𐑪𐑐𐑑 𐑩𐑕𐑑𐑻𐑯; 𐑯 𐑞 𐑣𐑧𐑛-𐑤𐑱𐑛𐑩𐑯 𐑖𐑦𐑐 𐑯𐑪𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑮𐑦𐑟𐑧𐑥𐑚𐑩𐑤𐑛 𐑩 𐑥𐑿𐑤 𐑒𐑨𐑮𐑦𐑦𐑙 𐑩 𐑐𐑺 𐑝 𐑴𐑝𐑼𐑚𐑻𐑛𐑩𐑯𐑦𐑙 𐑐𐑨𐑯𐑽𐑟.

𐑥𐑰𐑯𐑑𐑲𐑥, ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑢𐑪𐑟 𐑒𐑭𐑥𐑤𐑦 𐑲𐑦𐑙 𐑞 𐑮𐑲𐑑 𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛, 𐑯 𐑧𐑝𐑼 𐑯 𐑩𐑯𐑪𐑯 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑛𐑰𐑐 𐑮𐑦𐑙𐑒𐑩𐑤𐑟 𐑞𐑺 𐑑 𐑞 𐑤𐑲𐑯𐑟 𐑦𐑯 𐑣𐑦𐑟 𐑴𐑯 𐑣𐑨𐑯𐑛. 𐑯 ·𐑱𐑣𐑨𐑚 𐑗𐑭𐑯𐑕𐑑 𐑕𐑴 𐑑 𐑕𐑑𐑨𐑯𐑛, 𐑞𐑨𐑑 𐑞 ·𐑐𐑸𐑟𐑰 𐑪𐑒𐑿𐑐𐑲𐑛 𐑣𐑦𐑟 𐑖𐑨𐑛𐑴; 𐑢𐑲𐑤, 𐑦𐑓 𐑞 ·𐑐𐑸𐑟𐑰'𐑟 𐑖𐑨𐑛𐑴 𐑢𐑪𐑟 𐑞𐑺 𐑨𐑑 𐑷𐑤 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑴𐑯𐑤𐑦 𐑑 𐑚𐑤𐑧𐑯𐑛 𐑢𐑦𐑞, 𐑯 𐑤𐑧𐑙𐑔𐑩𐑯 ·𐑱𐑣𐑨𐑚'𐑟. 𐑨𐑟 𐑞 𐑒𐑮𐑵 𐑑𐑶𐑤𐑛 𐑪𐑯, ·𐑤𐑨𐑐𐑤𐑩𐑯𐑛𐑦𐑖 𐑕𐑐𐑧𐑒𐑿𐑤𐑱𐑖𐑩𐑯𐑟 𐑢𐑻 𐑚𐑨𐑯𐑛𐑦𐑛 𐑩𐑥𐑳𐑙 𐑞𐑧𐑥, 𐑒𐑩𐑯𐑕𐑻𐑯𐑦𐑙 𐑷𐑤 𐑞𐑰𐑟 𐑐𐑭𐑕𐑦𐑙 𐑔𐑦𐑙𐑟.
