·𐑗𐑨𐑐𐑑𐑼 109. 𐑱𐑣𐑨𐑚 𐑯 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑦𐑯 𐑞 ·𐑒𐑨𐑚𐑦𐑯.

𐑩𐑒𐑹𐑛𐑦𐑙 𐑑 𐑿𐑕𐑦𐑡 𐑞𐑱 𐑢𐑻 𐑐𐑳𐑥𐑐𐑦𐑙 𐑞 𐑖𐑦𐑐 𐑯𐑧𐑒𐑕𐑑 𐑥𐑹𐑯𐑦𐑙; 𐑯 𐑤𐑴! 𐑯𐑴 𐑦𐑯𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑶𐑤 𐑒𐑱𐑥 𐑳𐑐 𐑢𐑦𐑞 𐑞 𐑢𐑷𐑑𐑼; 𐑞 𐑒𐑭𐑕𐑒𐑕 𐑚𐑦𐑤𐑴 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑕𐑐𐑮𐑳𐑙 𐑩 𐑚𐑨𐑛 𐑤𐑰𐑒. 𐑥𐑳𐑗 𐑒𐑩𐑯𐑕𐑻𐑯 𐑢𐑪𐑟 𐑖𐑴𐑯; 𐑯 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑢𐑧𐑯𐑑 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑑 𐑮𐑦𐑐𐑹𐑑 𐑞𐑦𐑕 𐑳𐑯𐑓𐑱𐑝𐑼𐑩𐑚𐑩𐑤 𐑩𐑓𐑺.*

*𐑦𐑯 ·𐑕𐑐𐑻𐑥-𐑢𐑱𐑤𐑥𐑧𐑯 𐑢𐑦𐑞 𐑧𐑯𐑦 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑒𐑢𐑪𐑯𐑑𐑦𐑑𐑦 𐑝 𐑶𐑤 𐑪𐑯 𐑚𐑹𐑛, 𐑦𐑑 𐑦𐑟 𐑩 𐑮𐑧𐑜𐑿𐑤𐑼 𐑕𐑧𐑥𐑦-𐑢𐑰𐑒𐑤𐑦 𐑛𐑿𐑑𐑦 𐑑 𐑒𐑩𐑯𐑛𐑳𐑒𐑑 𐑩 𐑣𐑴𐑟 𐑦𐑯𐑑𐑵 𐑞 𐑣𐑴𐑤𐑛, 𐑯 𐑛𐑮𐑧𐑯𐑗 𐑞 𐑒𐑭𐑕𐑒𐑕 𐑢𐑦𐑞 𐑕𐑰-𐑢𐑷𐑑𐑼; 𐑢𐑦𐑗 𐑭𐑓𐑑𐑼𐑢𐑼𐑛𐑟, 𐑨𐑑 𐑝𐑺𐑦𐑦𐑙 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟, 𐑦𐑟 𐑮𐑦𐑥𐑵𐑝𐑛 𐑚𐑲 𐑞 𐑖𐑦𐑐'𐑕 𐑐𐑳𐑥𐑐𐑕. 𐑣𐑽𐑚𐑲 𐑞 𐑒𐑭𐑕𐑒𐑕 𐑸 𐑕𐑷𐑑 𐑑 𐑚𐑰 𐑒𐑧𐑐𐑑 𐑛𐑨𐑥𐑐𐑤𐑦 𐑑𐑲𐑑; 𐑢𐑲𐑤 𐑚𐑲 𐑞 𐑗𐑱𐑯𐑡𐑛 𐑒𐑨𐑮𐑩𐑒𐑑𐑼 𐑝 𐑞 𐑢𐑦𐑞𐑛𐑮𐑷𐑯 𐑢𐑷𐑑𐑼, 𐑞 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑮𐑧𐑛𐑦𐑤𐑦 𐑛𐑦𐑑𐑧𐑒𐑑 𐑧𐑯𐑦 𐑕𐑽𐑾𐑕 𐑤𐑰𐑒𐑦𐑡 𐑦𐑯 𐑞 𐑐𐑮𐑧𐑖𐑩𐑕 𐑒𐑸𐑜𐑴.

𐑯𐑬, 𐑓𐑮𐑪𐑥 𐑞 ·𐑕𐑬𐑔 𐑯 ·𐑢𐑧𐑕𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑛𐑮𐑷𐑦𐑙 𐑯𐑲 𐑑 ·𐑓𐑹𐑥𐑴𐑕𐑩 𐑯 𐑞 ·𐑚𐑨𐑖𐑰 ·𐑲𐑤𐑟, 𐑚𐑦𐑑𐑢𐑰𐑯 𐑢𐑦𐑗 𐑤𐑲𐑟 𐑢𐑳𐑯 𐑝 𐑞 𐑑𐑮𐑪𐑐𐑦𐑒𐑩𐑤 𐑬𐑑𐑤𐑧𐑑𐑕 𐑓𐑮𐑪𐑥 𐑞 ·𐑗𐑲𐑯𐑩 𐑢𐑷𐑑𐑼𐑟 𐑦𐑯𐑑𐑵 𐑞 ·𐑐𐑩𐑕𐑦𐑓𐑦𐑒. 𐑯 𐑕𐑴 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑓𐑬𐑯𐑛 ·𐑱𐑣𐑨𐑚 𐑢𐑦𐑞 𐑩 𐑡𐑧𐑯𐑼𐑩𐑤 𐑗𐑸𐑑 𐑝 𐑞 ·𐑹𐑦𐑧𐑯𐑑𐑩𐑤 𐑸𐑒𐑦𐑐𐑧𐑤𐑩𐑜𐑴𐑟 𐑕𐑐𐑮𐑧𐑛 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥; 𐑯 𐑩𐑯𐑳𐑞𐑼 𐑕𐑧𐑐𐑼𐑩𐑑 𐑢𐑳𐑯 𐑮𐑧𐑐𐑮𐑦𐑟𐑧𐑯𐑑𐑦𐑙 𐑞 𐑤𐑪𐑙 𐑰𐑕𐑑𐑼𐑯 𐑒𐑴𐑕𐑑𐑕 𐑝 𐑞 ·𐑡𐑨𐑐𐑩𐑯𐑰𐑟 𐑲𐑤𐑩𐑯𐑛𐑟—·𐑯𐑦𐑓𐑪𐑯, ·𐑥𐑨𐑑𐑕𐑥𐑱, 𐑯 ·𐑕𐑦𐑒𐑴𐑒. 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑕𐑯𐑴-𐑢𐑲𐑑 𐑯𐑿 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜 𐑚𐑮𐑱𐑕𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑕𐑒𐑮𐑵𐑛 𐑤𐑧𐑜 𐑝 𐑣𐑦𐑟 𐑑𐑱𐑚𐑩𐑤, 𐑯 𐑢𐑦𐑞 𐑩 𐑤𐑪𐑙 𐑐𐑮𐑵𐑯𐑦𐑙-𐑣𐑫𐑒 𐑝 𐑩 𐑡𐑨𐑒-𐑯𐑲𐑓 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛, 𐑞 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑚𐑨𐑒 𐑑 𐑞 𐑜𐑨𐑙𐑢𐑱 𐑛𐑹, 𐑢𐑪𐑟 𐑮𐑦𐑙𐑒𐑩𐑤𐑦𐑙 𐑣𐑦𐑟 𐑚𐑮𐑬, 𐑯 𐑑𐑮𐑱𐑕𐑦𐑙 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑒𐑹𐑕𐑩𐑟 𐑩𐑜𐑱𐑯.

“𐑣𐑵'𐑟 𐑞𐑺?” 𐑣𐑽𐑦𐑙 𐑞 𐑓𐑫𐑑𐑕𐑑𐑧𐑐 𐑨𐑑 𐑞 𐑛𐑹, 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑑𐑻𐑯𐑦𐑙 𐑮𐑬𐑯𐑛 𐑑 𐑦𐑑. “𐑪𐑯 𐑛𐑧𐑒! 𐑚𐑦𐑜𐑪𐑯!”

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑥𐑦𐑕𐑑𐑱𐑒𐑕; 𐑦𐑑 𐑦𐑟 𐑲. 𐑞 𐑶𐑤 𐑦𐑯 𐑞 𐑣𐑴𐑤𐑛 𐑦𐑟 𐑤𐑰𐑒𐑦𐑙, 𐑕𐑻. 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑳𐑐 ·𐑚𐑻𐑑𐑩𐑯𐑟 𐑯 𐑚𐑮𐑱𐑒 𐑬𐑑.”

“𐑳𐑐 ·𐑚𐑻𐑑𐑩𐑯𐑟 𐑯 𐑚𐑮𐑱𐑒 𐑬𐑑? 𐑯𐑬 𐑞𐑨𐑑 𐑢𐑰 𐑸 𐑯𐑽𐑦𐑙 ·𐑡𐑩𐑐𐑨𐑯; 𐑣𐑰𐑝-𐑑 𐑣𐑽 𐑓 𐑩 𐑢𐑰𐑒 𐑑 𐑑𐑦𐑙𐑒𐑼 𐑩 𐑐𐑸𐑕𐑩𐑤 𐑝 𐑴𐑤𐑛 𐑣𐑵𐑐𐑕?”

“𐑲𐑞𐑼 𐑛𐑵 𐑞𐑨𐑑, 𐑕𐑻, 𐑹 𐑢𐑱𐑕𐑑 𐑦𐑯 𐑢𐑳𐑯 𐑛𐑱 𐑥𐑹 𐑶𐑤 𐑞𐑨𐑯 𐑢𐑰 𐑥𐑱 𐑥𐑱𐑒 𐑜𐑫𐑛 𐑦𐑯 𐑩 𐑘𐑽. 𐑢𐑪𐑑 𐑢𐑰 𐑒𐑳𐑥 𐑑𐑢𐑧𐑯𐑑𐑦 𐑔𐑬𐑟𐑩𐑯𐑛 𐑥𐑲𐑤𐑟 𐑑 𐑜𐑧𐑑 𐑦𐑟 𐑢𐑻𐑔 𐑕𐑱𐑝𐑦𐑙, 𐑕𐑻.”

“𐑕𐑴 𐑦𐑑 𐑦𐑟, 𐑕𐑴 𐑦𐑑 𐑦𐑟; 𐑦𐑓 𐑢𐑰 𐑜𐑧𐑑 𐑦𐑑.”

“𐑲 𐑢𐑪𐑟 𐑕𐑐𐑰𐑒𐑦𐑙 𐑝 𐑞 𐑶𐑤 𐑦𐑯 𐑞 𐑣𐑴𐑤𐑛, 𐑕𐑻.”

“𐑯 𐑲 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑕𐑐𐑰𐑒𐑦𐑙 𐑹 𐑔𐑦𐑙𐑒𐑦𐑙 𐑝 𐑞𐑨𐑑 𐑨𐑑 𐑷𐑤. 𐑚𐑦𐑜𐑪𐑯! 𐑤𐑧𐑑 𐑦𐑑 𐑤𐑰𐑒! 𐑲'𐑥 𐑷𐑤 𐑱𐑤𐑰𐑒 𐑥𐑲𐑕𐑧𐑤𐑓. 𐑲! 𐑤𐑰𐑒𐑕 𐑦𐑯 𐑤𐑰𐑒𐑕! 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑓𐑫𐑤 𐑝 𐑤𐑰𐑒𐑦 𐑒𐑭𐑕𐑒𐑕, 𐑚𐑳𐑑 𐑞𐑴𐑟 𐑤𐑰𐑒𐑦 𐑒𐑭𐑕𐑒𐑕 𐑸 𐑦𐑯 𐑩 𐑤𐑰𐑒𐑦 𐑖𐑦𐑐; 𐑯 𐑞𐑨𐑑'𐑕 𐑩 𐑓𐑸 𐑢𐑻𐑕 𐑐𐑤𐑲𐑑 𐑞𐑨𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟, 𐑥𐑨𐑯. 𐑘𐑧𐑑 𐑲 𐑛𐑴𐑯𐑑 𐑕𐑑𐑪𐑐 𐑑 𐑐𐑤𐑳𐑜 𐑥𐑲 𐑤𐑰𐑒; 𐑓 𐑣𐑵 𐑒𐑨𐑯 𐑓𐑲𐑯𐑛 𐑦𐑑 𐑦𐑯 𐑞 𐑛𐑰𐑐-𐑤𐑴𐑛𐑩𐑛 𐑣𐑳𐑤; 𐑹 𐑣𐑬 𐑣𐑴𐑐 𐑑 𐑐𐑤𐑳𐑜 𐑦𐑑, 𐑰𐑝𐑩𐑯 𐑦𐑓 𐑓𐑬𐑯𐑛, 𐑦𐑯 𐑞𐑦𐑕 𐑤𐑲𐑓'𐑕 𐑣𐑬𐑤𐑦𐑙 𐑜𐑱𐑤? 𐑕𐑑𐑸𐑚𐑳𐑒! 𐑲'𐑤 𐑯𐑪𐑑 𐑣𐑨𐑝 𐑞 ·𐑚𐑻𐑑𐑩𐑯𐑟 𐑣𐑶𐑕𐑑𐑩𐑛.”

“𐑢𐑪𐑑 𐑢𐑦𐑤 𐑞 𐑴𐑯𐑼𐑟 𐑕𐑱, 𐑕𐑻?”

“𐑤𐑧𐑑 𐑞 𐑴𐑯𐑼𐑟 𐑕𐑑𐑨𐑯𐑛 𐑪𐑯 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑚𐑰𐑗 𐑯 𐑬𐑑𐑦𐑧𐑤 𐑞 ·𐑑𐑲𐑓𐑵𐑯𐑟. 𐑢𐑪𐑑 𐑒𐑺𐑟 ·𐑱𐑣𐑨𐑚? 𐑴𐑯𐑼𐑟, 𐑴𐑯𐑼𐑟? 𐑞𐑬 𐑸𐑑 𐑷𐑤𐑢𐑱𐑟 𐑐𐑮𐑱𐑑𐑦𐑙 𐑑 𐑥𐑰, ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑩𐑚𐑬𐑑 𐑞𐑴𐑟 𐑥𐑲𐑟𐑼𐑤𐑦 𐑴𐑯𐑼𐑟, 𐑨𐑟 𐑦𐑓 𐑞 𐑴𐑯𐑼𐑟 𐑢𐑻 𐑥𐑲 𐑒𐑪𐑯𐑖𐑩𐑯𐑕. 𐑚𐑳𐑑 𐑤𐑫𐑒 𐑘𐑰, 𐑞 𐑴𐑯𐑤𐑦 𐑮𐑾𐑤 𐑴𐑯𐑼 𐑝 𐑧𐑯𐑦𐑔𐑦𐑙 𐑦𐑟 𐑦𐑑𐑕 𐑒𐑩𐑥𐑭𐑯𐑛𐑼; 𐑯 𐑣𐑸𐑒 𐑘𐑰, 𐑥𐑲 𐑒𐑪𐑯𐑖𐑩𐑯𐑕 𐑦𐑟 𐑦𐑯 𐑞𐑦𐑕 𐑖𐑦𐑐'𐑕 𐑒𐑰𐑤.—𐑪𐑯 𐑛𐑧𐑒!”

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚,” 𐑕𐑧𐑛 𐑞 𐑮𐑧𐑛𐑩𐑯𐑦𐑙 𐑥𐑱𐑑, 𐑥𐑵𐑝𐑦𐑙 𐑓𐑻𐑞𐑼 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑢𐑦𐑞 𐑩 𐑛𐑺𐑦𐑙 𐑕𐑴 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑓𐑩𐑤 𐑯 𐑒𐑷𐑖𐑩𐑕 𐑞𐑨𐑑 𐑦𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰𐑥𐑛 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑧𐑝𐑮𐑦 𐑢𐑱 𐑕𐑰𐑒𐑦𐑙 𐑑 𐑩𐑝𐑶𐑛 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑬𐑑𐑢𐑼𐑛 𐑥𐑨𐑯𐑦𐑓𐑧𐑕𐑑𐑱𐑖𐑩𐑯 𐑝 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑚𐑳𐑑 𐑢𐑦𐑞𐑦𐑯 𐑷𐑤𐑕𐑴 𐑕𐑰𐑥𐑛 𐑥𐑹 𐑞𐑨𐑯 𐑣𐑭𐑓 𐑛𐑦𐑕𐑑𐑮𐑳𐑕𐑑𐑓𐑩𐑤 𐑝 𐑦𐑑𐑕𐑧𐑤𐑓; “𐑩 𐑚𐑧𐑑𐑼 𐑥𐑨𐑯 𐑞𐑨𐑯 𐑲 𐑥𐑲𐑑 𐑢𐑧𐑤 𐑐𐑭𐑕 𐑴𐑝𐑼 𐑦𐑯 𐑞𐑰 𐑢𐑪𐑑 𐑣𐑰 𐑢𐑫𐑛 𐑒𐑢𐑦𐑒𐑤𐑦 𐑦𐑯𐑳𐑓 𐑮𐑦𐑟𐑧𐑯𐑑 𐑦𐑯 𐑩 𐑘𐑳𐑙𐑜𐑼 𐑥𐑨𐑯; 𐑲, 𐑯 𐑦𐑯 𐑩 𐑣𐑨𐑐𐑽, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚.”

“𐑛𐑧𐑝𐑩𐑤𐑟! 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑞𐑧𐑯 𐑕𐑴 𐑥𐑳𐑗 𐑨𐑟 𐑛𐑺 𐑑 𐑒𐑮𐑦𐑑𐑦𐑒𐑤𐑦 𐑔𐑦𐑙𐑒 𐑝 𐑥𐑰?—𐑪𐑯 𐑛𐑧𐑒!”

“𐑯𐑱, 𐑕𐑻, 𐑯𐑪𐑑 𐑘𐑧𐑑; 𐑲 𐑛𐑵 𐑦𐑯𐑑𐑮𐑰𐑑. 𐑯 𐑲 𐑛𐑵 𐑛𐑺, 𐑕𐑻—𐑑 𐑚𐑰 𐑓𐑹𐑚𐑺𐑦𐑙! 𐑖𐑨𐑤 𐑢𐑰 𐑯𐑪𐑑 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑰𐑗 𐑳𐑞𐑼 𐑚𐑧𐑑𐑼 𐑞𐑨𐑯 𐑣𐑦𐑞𐑼𐑑𐑵, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚?”

𐑱𐑣𐑨𐑚 𐑕𐑰𐑟𐑛 𐑩 𐑤𐑴𐑛𐑩𐑛 𐑥𐑳𐑕𐑒𐑩𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑮𐑨𐑒 (𐑓𐑹𐑥𐑦𐑙 𐑐𐑸𐑑 𐑝 𐑥𐑴𐑕𐑑 ·𐑕𐑬𐑔-·𐑕𐑰-𐑥𐑧𐑯'𐑟 𐑒𐑨𐑚𐑦𐑯 𐑓𐑻𐑯𐑦𐑗𐑼), 𐑯 𐑐𐑶𐑯𐑑𐑦𐑙 𐑦𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛: “𐑞𐑺 𐑦𐑟 𐑢𐑳𐑯 ·𐑜𐑪𐑛 𐑞𐑨𐑑 𐑦𐑟 ·𐑤𐑹𐑛 𐑴𐑝𐑼 𐑞 𐑻𐑔, 𐑯 𐑢𐑳𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑞𐑨𐑑 𐑦𐑟 𐑤𐑹𐑛 𐑴𐑝𐑼 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛.—𐑪𐑯 𐑛𐑧𐑒!”

𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑦𐑯 𐑞 𐑓𐑤𐑨𐑖𐑦𐑙 𐑲𐑟 𐑝 𐑞 𐑥𐑱𐑑, 𐑯 𐑣𐑦𐑟 𐑓𐑲𐑼𐑦 𐑗𐑰𐑒𐑕, 𐑿 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑷𐑤𐑥𐑴𐑕𐑑 𐑔𐑷𐑑 𐑞𐑨𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑮𐑾𐑤𐑦 𐑮𐑦𐑕𐑰𐑝𐑛 𐑞 𐑚𐑤𐑱𐑟 𐑝 𐑞 𐑤𐑧𐑝𐑩𐑤𐑛 𐑑𐑿𐑚. 𐑚𐑳𐑑, 𐑥𐑭𐑕𐑑𐑼𐑦𐑙 𐑣𐑦𐑟 𐑦𐑥𐑴𐑖𐑩𐑯, 𐑣𐑰 𐑣𐑭𐑓 𐑒𐑭𐑥𐑤𐑦 𐑮𐑴𐑟, 𐑯 𐑨𐑟 𐑣𐑰 𐑒𐑢𐑦𐑑𐑩𐑛 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑐𐑷𐑟𐑛 𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑯 𐑕𐑧𐑛: “𐑞𐑬 𐑣𐑨𐑕𐑑 𐑬𐑑𐑮𐑱𐑡𐑛, 𐑯𐑪𐑑 𐑦𐑯𐑕𐑳𐑤𐑑𐑩𐑛 𐑥𐑰, 𐑕𐑻; 𐑚𐑳𐑑 𐑓 𐑞𐑨𐑑 𐑲 𐑭𐑕𐑒 𐑞𐑰 𐑯𐑪𐑑 𐑑 𐑚𐑦𐑢𐑺 𐑝 ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑞𐑬 𐑢𐑫𐑛𐑕𐑑 𐑚𐑳𐑑 𐑤𐑭𐑓; 𐑚𐑳𐑑 𐑤𐑧𐑑 ·𐑱𐑣𐑨𐑚 𐑚𐑦𐑢𐑺 𐑝 ·𐑱𐑣𐑨𐑚; 𐑚𐑦𐑢𐑺 𐑝 𐑞𐑲𐑕𐑧𐑤𐑓, 𐑴𐑤𐑛 𐑥𐑨𐑯.”

“𐑣𐑰 𐑢𐑨𐑒𐑕𐑩𐑟 𐑚𐑮𐑱𐑝, 𐑚𐑳𐑑 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕 𐑴𐑚𐑱𐑟; 𐑥𐑴𐑕𐑑 𐑒𐑺𐑓𐑩𐑤 𐑚𐑮𐑱𐑝𐑼𐑦 𐑞𐑨𐑑!” 𐑥𐑻𐑥𐑼𐑛 ·𐑱𐑣𐑨𐑚, 𐑨𐑟 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑛𐑦𐑕𐑩𐑐𐑽𐑛. “𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑣𐑰 𐑕𐑧𐑛—·𐑱𐑣𐑨𐑚 𐑚𐑦𐑢𐑺 𐑝 ·𐑱𐑣𐑨𐑚—𐑞𐑺'𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑞𐑺!” 𐑞𐑧𐑯 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕𐑤𐑦 𐑿𐑟𐑦𐑙 𐑞 𐑥𐑳𐑕𐑒𐑩𐑑 𐑓 𐑩 𐑕𐑑𐑭𐑓, 𐑢𐑦𐑞 𐑩𐑯 𐑲𐑼𐑯 𐑚𐑮𐑬 𐑣𐑰 𐑐𐑱𐑕𐑑 𐑑 𐑯 𐑓𐑮𐑴 𐑦𐑯 𐑞 𐑤𐑦𐑑𐑩𐑤 𐑒𐑨𐑚𐑦𐑯; 𐑚𐑳𐑑 𐑐𐑮𐑧𐑟𐑩𐑯𐑑𐑤𐑦 𐑞 𐑔𐑦𐑒 𐑐𐑤𐑨𐑑𐑕 𐑝 𐑣𐑦𐑟 𐑓𐑪𐑮𐑦𐑛 𐑮𐑦𐑤𐑨𐑒𐑕𐑑, 𐑯 𐑮𐑦𐑑𐑻𐑯𐑦𐑙 𐑞 𐑜𐑳𐑯 𐑑 𐑞 𐑮𐑨𐑒, 𐑣𐑰 𐑢𐑧𐑯𐑑 𐑑 𐑞 𐑛𐑧𐑒.

“𐑞𐑬 𐑸𐑑 𐑚𐑳𐑑 𐑑𐑵 𐑜𐑫𐑛 𐑩 𐑓𐑧𐑤𐑴, ·𐑕𐑑𐑸𐑚𐑳𐑒,” 𐑣𐑰 𐑕𐑧𐑛 𐑤𐑴𐑤𐑦 𐑑 𐑞 𐑥𐑱𐑑; 𐑞𐑧𐑯 𐑮𐑱𐑟𐑦𐑙 𐑣𐑦𐑟 𐑝𐑶𐑕 𐑑 𐑞 𐑒𐑮𐑵: “𐑓𐑻𐑤 𐑞 𐑑'𐑜𐑨𐑤𐑩𐑯𐑑-𐑕𐑱𐑤𐑟, 𐑯 𐑒𐑤𐑴𐑕-𐑮𐑰𐑓 𐑞 𐑑𐑪𐑐-𐑕𐑱𐑤𐑟, 𐑓𐑹 𐑯 𐑭𐑓𐑑; 𐑚𐑨𐑒 𐑞 𐑥𐑱𐑯-𐑘𐑸𐑛; 𐑳𐑐 ·𐑚𐑻𐑑𐑩𐑯, 𐑯 𐑚𐑮𐑱𐑒 𐑬𐑑 𐑦𐑯 𐑞 𐑥𐑱𐑯-𐑣𐑴𐑤𐑛.”

𐑦𐑑 𐑢𐑻 𐑐𐑼𐑣𐑨𐑐𐑕 𐑝𐑱𐑯 𐑑 𐑕𐑼𐑥𐑲𐑟 𐑦𐑜𐑟𐑨𐑒𐑑𐑤𐑦 𐑢𐑲 𐑦𐑑 𐑢𐑪𐑟, 𐑞𐑨𐑑 𐑨𐑟 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑦𐑙 ·𐑕𐑑𐑸𐑚𐑳𐑒, ·𐑱𐑣𐑨𐑚 𐑞𐑳𐑕 𐑨𐑒𐑑𐑩𐑛. 𐑦𐑑 𐑥𐑱 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑩 𐑓𐑤𐑨𐑖 𐑝 𐑪𐑯𐑩𐑕𐑑𐑦 𐑦𐑯 𐑣𐑦𐑥; 𐑹 𐑥𐑽 𐑐𐑮𐑵𐑛𐑧𐑯𐑖𐑩𐑤 𐑐𐑪𐑤𐑩𐑕𐑦 𐑢𐑦𐑗, 𐑳𐑯𐑛𐑼 𐑞 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕, 𐑦𐑥𐑐𐑽𐑾𐑕𐑤𐑦 𐑓𐑼𐑚𐑨𐑛 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑕𐑦𐑥𐑐𐑑𐑩𐑥 𐑝 𐑴𐑐𐑩𐑯 𐑛𐑦𐑕𐑩𐑓𐑧𐑒𐑖𐑩𐑯, 𐑣𐑬𐑧𐑝𐑼 𐑑𐑮𐑨𐑯𐑟𐑾𐑯𐑑, 𐑦𐑯 𐑞 𐑦𐑥𐑐𐑹𐑑𐑩𐑯𐑑 𐑗𐑰𐑓 𐑪𐑓𐑦𐑕𐑼 𐑝 𐑣𐑦𐑟 𐑖𐑦𐑐. 𐑣𐑬𐑧𐑝𐑼 𐑦𐑑 𐑢𐑪𐑟, 𐑣𐑦𐑟 𐑹𐑛𐑼𐑟 𐑢𐑻 𐑧𐑒𐑕𐑦𐑒𐑿𐑑𐑩𐑛; 𐑯 𐑞 ·𐑚𐑻𐑑𐑩𐑯𐑟 𐑢𐑻 𐑣𐑶𐑕𐑑𐑩𐑛.
