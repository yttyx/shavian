𐑲 𐑐𐑮𐑴𐑑𐑧𐑕𐑑𐑩𐑛 𐑥𐑲 𐑦𐑯𐑩𐑕𐑩𐑯𐑕 𐑝 𐑞𐑰𐑟 𐑔𐑦𐑙𐑟. 𐑲 𐑕𐑷 𐑞𐑨𐑑 𐑳𐑯𐑛𐑼 𐑞 𐑥𐑭𐑕𐑒 𐑝 𐑞𐑰𐑟 𐑣𐑭𐑓 𐑣𐑿𐑥𐑼𐑩𐑕 𐑦𐑯𐑿𐑧𐑯𐑛𐑴𐑟, 𐑞𐑦𐑕 𐑴𐑤𐑛 𐑕𐑰𐑥𐑩𐑯, 𐑨𐑟 𐑩𐑯 𐑦𐑯𐑕𐑿𐑤𐑱𐑑𐑩𐑛 ·𐑒𐑢𐑱𐑒𐑼𐑦𐑖 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼, 𐑢𐑪𐑟 𐑓𐑫𐑤 𐑝 𐑣𐑦𐑟 𐑦𐑯𐑖𐑩𐑤𐑼 𐑐𐑮𐑧𐑡𐑩𐑛𐑦𐑕𐑩𐑟, 𐑯 𐑮𐑭𐑞𐑼 𐑛𐑦𐑕𐑑𐑮𐑳𐑕𐑑𐑓𐑩𐑤 𐑝 𐑷𐑤 𐑱𐑤𐑾𐑯𐑟, 𐑩𐑯𐑤𐑧𐑕 𐑞𐑱 𐑣𐑱𐑤𐑛 𐑓𐑮𐑪𐑥 ·𐑒𐑱𐑐 ·𐑒𐑪𐑛 𐑹 𐑞 ·𐑝𐑦𐑯𐑘𐑼𐑛.

“𐑚𐑳𐑑 𐑢𐑪𐑑 𐑑𐑱𐑒𐑕 𐑞𐑰 𐑩-𐑢𐑱𐑤𐑦𐑙? 𐑲 𐑢𐑪𐑯𐑑 𐑑 𐑯𐑴 𐑞𐑨𐑑 𐑚𐑦𐑓𐑹 𐑲 𐑔𐑦𐑙𐑒 𐑝 𐑖𐑦𐑐𐑦𐑙 𐑘𐑰.”

“𐑢𐑧𐑤, 𐑕𐑻, 𐑲 𐑢𐑪𐑯𐑑 𐑑 𐑕𐑰 𐑢𐑪𐑑 𐑢𐑱𐑤𐑦𐑙 𐑦𐑟. 𐑲 𐑢𐑪𐑯𐑑 𐑑 𐑕𐑰 𐑞 𐑢𐑻𐑤𐑛.”

“𐑢𐑪𐑯𐑑 𐑑 𐑕𐑰 𐑢𐑪𐑑 𐑢𐑱𐑤𐑦𐑙 𐑦𐑟, 𐑱? 𐑣𐑨𐑝 𐑘𐑰 𐑒𐑤𐑨𐑐𐑑 𐑲 𐑪𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚?”

“𐑣𐑵 𐑦𐑟 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑕𐑻?”

“𐑲, 𐑲, 𐑲 𐑔𐑷𐑑 𐑕𐑴. 𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑝 𐑞𐑦𐑕 𐑖𐑦𐑐.”

“𐑲 𐑨𐑥 𐑥𐑦𐑕𐑑𐑱𐑒𐑩𐑯 𐑞𐑧𐑯. 𐑲 𐑔𐑷𐑑 𐑲 𐑢𐑪𐑟 𐑕𐑐𐑰𐑒𐑦𐑙 𐑑 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑣𐑦𐑥𐑕𐑧𐑤𐑓.”

“𐑞𐑬 𐑸𐑑 𐑕𐑐𐑰𐑒𐑦𐑙 𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜—𐑞𐑨𐑑'𐑕 𐑣𐑵 𐑘𐑰 𐑸 𐑕𐑐𐑰𐑒𐑦𐑙 𐑑, 𐑘𐑳𐑙 𐑥𐑨𐑯. 𐑦𐑑 𐑚𐑦𐑤𐑪𐑙𐑟 𐑑 𐑥𐑰 𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑑 𐑕𐑰 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑓𐑦𐑑𐑩𐑛 𐑬𐑑 𐑓 𐑞 𐑝𐑶𐑦𐑡, 𐑯 𐑕𐑩𐑐𐑤𐑲𐑛 𐑢𐑦𐑞 𐑷𐑤 𐑣𐑻 𐑯𐑰𐑛𐑟, 𐑦𐑯𐑒𐑤𐑵𐑛𐑦𐑙 𐑒𐑮𐑵. 𐑢𐑰 𐑸 𐑐𐑸𐑑 𐑴𐑯𐑼𐑟 𐑯 𐑱𐑡𐑩𐑯𐑑𐑕. 𐑚𐑳𐑑 𐑨𐑟 𐑲 𐑢𐑪𐑟 𐑜𐑴𐑦𐑙 𐑑 𐑕𐑱, 𐑦𐑓 𐑞𐑬 𐑢𐑪𐑯𐑑𐑩𐑕𐑑 𐑑 𐑯𐑴 𐑢𐑪𐑑 𐑢𐑱𐑤𐑦𐑙 𐑦𐑟, 𐑨𐑟 𐑞𐑬 𐑑𐑧𐑤𐑩𐑕𐑑 𐑘𐑰 𐑛𐑵, 𐑲 𐑒𐑨𐑯 𐑐𐑫𐑑 𐑘𐑰 𐑦𐑯 𐑩 𐑢𐑱 𐑝 𐑓𐑲𐑯𐑛𐑦𐑙 𐑦𐑑 𐑬𐑑 𐑚𐑦𐑓𐑹 𐑘𐑰 𐑚𐑲𐑯𐑛 𐑘𐑹𐑕𐑧𐑤𐑓 𐑑 𐑦𐑑, 𐑐𐑭𐑕𐑑 𐑚𐑨𐑒𐑦𐑙 𐑬𐑑. 𐑒𐑤𐑨𐑐 𐑲 𐑪𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑯 𐑞𐑬 𐑢𐑦𐑤𐑑 𐑓𐑲𐑯𐑛 𐑞𐑨𐑑 𐑣𐑰 𐑣𐑨𐑟 𐑴𐑯𐑤𐑦 𐑢𐑳𐑯 𐑤𐑧𐑜.”

“𐑢𐑪𐑑 𐑛𐑵 𐑿 𐑥𐑰𐑯, 𐑕𐑻? 𐑢𐑪𐑟 𐑞 𐑳𐑞𐑼 𐑢𐑳𐑯 𐑤𐑪𐑕𐑑 𐑚𐑲 𐑩 𐑢𐑱𐑤?”

“𐑤𐑪𐑕𐑑 𐑚𐑲 𐑩 𐑢𐑱𐑤! 𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑒𐑳𐑥 𐑯𐑽𐑼 𐑑 𐑥𐑰: 𐑦𐑑 𐑢𐑪𐑟 𐑛𐑦𐑝𐑬𐑼𐑛, 𐑗𐑵𐑛 𐑳𐑐, 𐑒𐑮𐑳𐑯𐑗𐑑 𐑚𐑲 𐑞 𐑥𐑪𐑯𐑕𐑑𐑮𐑩𐑕𐑩𐑕𐑑 ·𐑐𐑸𐑥𐑩𐑕𐑧𐑑𐑦 𐑞𐑨𐑑 𐑧𐑝𐑼 𐑗𐑦𐑐𐑑 𐑩 𐑚𐑴𐑑!—𐑭, 𐑭!”

𐑲 𐑢𐑪𐑟 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑩𐑤𐑸𐑥𐑛 𐑚𐑲 𐑣𐑦𐑟 𐑧𐑯𐑼𐑡𐑦, 𐑐𐑼𐑣𐑨𐑐𐑕 𐑷𐑤𐑕𐑴 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑑𐑳𐑗𐑑 𐑨𐑑 𐑞 𐑣𐑸𐑑𐑦 𐑜𐑮𐑰𐑓 𐑦𐑯 𐑣𐑦𐑟 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑦𐑙 𐑧𐑒𐑕𐑒𐑤𐑩𐑥𐑱𐑖𐑩𐑯, 𐑚𐑳𐑑 𐑕𐑧𐑛 𐑨𐑟 𐑒𐑭𐑥𐑤𐑦 𐑨𐑟 𐑲 𐑒𐑫𐑛, “𐑢𐑪𐑑 𐑿 𐑕𐑱 𐑦𐑟 𐑯𐑴 𐑛𐑬𐑑 𐑑𐑮𐑵 𐑦𐑯𐑳𐑓, 𐑕𐑻; 𐑚𐑳𐑑 𐑣𐑬 𐑒𐑫𐑛 𐑲 𐑯𐑴 𐑞𐑺 𐑢𐑪𐑟 𐑧𐑯𐑦 𐑐𐑦𐑒𐑿𐑤𐑽 𐑓𐑼𐑪𐑕𐑦𐑑𐑦 𐑦𐑯 𐑞𐑨𐑑 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑢𐑱𐑤, 𐑞𐑴 𐑦𐑯𐑛𐑰𐑛 𐑲 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑦𐑯𐑓𐑻𐑛 𐑨𐑟 𐑥𐑳𐑗 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑦𐑥𐑐𐑩𐑤 𐑓𐑨𐑒𐑑 𐑝 𐑞 𐑨𐑒𐑕𐑦𐑛𐑩𐑯𐑑.”

“𐑤𐑫𐑒 𐑘𐑰 𐑯𐑬, 𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑞𐑲 𐑤𐑳𐑙𐑟 𐑸 𐑩 𐑕𐑹𐑑 𐑝 𐑕𐑪𐑓𐑑, 𐑛'𐑘𐑰 𐑕𐑰; 𐑞𐑬 𐑛𐑳𐑕𐑑 𐑯𐑪𐑑 𐑑𐑷𐑒 𐑖𐑸𐑒 𐑩 𐑚𐑦𐑑. 𐑖𐑫𐑼, 𐑘𐑰'𐑝 𐑚𐑰𐑯 𐑑 𐑕𐑰 𐑚𐑦𐑓𐑹 𐑯𐑬; 𐑖𐑫𐑼 𐑝 𐑞𐑨𐑑?”

“𐑕𐑻,” 𐑕𐑧𐑛 𐑲, “𐑲 𐑔𐑷𐑑 𐑲 𐑑𐑴𐑤𐑛 𐑿 𐑞𐑨𐑑 𐑲 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑓𐑹 𐑝𐑶𐑦𐑡𐑩𐑟 𐑦𐑯 𐑞 𐑥𐑻𐑗𐑩𐑯𐑑—”

“𐑣𐑸𐑛 𐑛𐑬𐑯 𐑬𐑑 𐑝 𐑞𐑨𐑑! 𐑥𐑲𐑯𐑛 𐑢𐑪𐑑 𐑲 𐑕𐑧𐑛 𐑩𐑚𐑬𐑑 𐑞 𐑥𐑸𐑗𐑩𐑯𐑑 𐑕𐑻𐑝𐑦𐑕—𐑛𐑴𐑯𐑑 𐑨𐑜𐑮𐑩𐑝𐑱𐑑 𐑥𐑰—𐑲 𐑢𐑴𐑯𐑑 𐑣𐑨𐑝 𐑦𐑑. 𐑚𐑳𐑑 𐑤𐑧𐑑 𐑳𐑕 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑰𐑗 𐑳𐑞𐑼. 𐑲 𐑣𐑨𐑝 𐑜𐑦𐑝𐑩𐑯 𐑞𐑰 𐑩 𐑣𐑦𐑯𐑑 𐑩𐑚𐑬𐑑 𐑢𐑪𐑑 𐑢𐑱𐑤𐑦𐑙 𐑦𐑟; 𐑛𐑵 𐑘𐑰 𐑘𐑧𐑑 𐑓𐑰𐑤 𐑦𐑯𐑒𐑤𐑲𐑯𐑛 𐑓 𐑦𐑑?”

“𐑲 𐑛𐑵, 𐑕𐑻.”

“𐑝𐑧𐑮𐑦 𐑜𐑫𐑛. 𐑯𐑬, 𐑸𐑑 𐑞𐑬 𐑞 𐑥𐑨𐑯 𐑑 𐑐𐑦𐑗 𐑩 𐑣𐑸𐑐𐑵𐑯 𐑛𐑬𐑯 𐑩 𐑤𐑲𐑝 𐑢𐑱𐑤'𐑟 𐑔𐑮𐑴𐑑, 𐑯 𐑞𐑧𐑯 𐑡𐑳𐑥𐑐 𐑭𐑓𐑑𐑼 𐑦𐑑? 𐑭𐑯𐑕𐑼, 𐑒𐑢𐑦𐑒!”

“𐑲 𐑨𐑥, 𐑕𐑻, 𐑦𐑓 𐑦𐑑 𐑖𐑵𐑛 𐑚𐑰 𐑐𐑪𐑟𐑦𐑑𐑦𐑝𐑤𐑦 𐑦𐑯𐑛𐑦𐑕𐑐𐑧𐑯𐑕𐑩𐑚𐑩𐑤 𐑑 𐑛𐑵 𐑕𐑴; 𐑯𐑪𐑑 𐑑 𐑚𐑰 𐑜𐑪𐑑 𐑮𐑦𐑛 𐑝, 𐑞𐑨𐑑 𐑦𐑟; 𐑢𐑦𐑗 𐑲 𐑛𐑴𐑯𐑑 𐑑𐑱𐑒 𐑑 𐑚𐑰 𐑞 𐑓𐑨𐑒𐑑.”

“𐑜𐑫𐑛 𐑩𐑜𐑱𐑯. 𐑯𐑬 𐑞𐑧𐑯, 𐑞𐑬 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑢𐑪𐑯𐑑𐑩𐑕𐑑 𐑑 𐑜𐑴 𐑩-𐑢𐑱𐑤𐑦𐑙, 𐑑 𐑓𐑲𐑯𐑛 𐑬𐑑 𐑚𐑲 𐑦𐑒𐑕𐑐𐑽𐑾𐑯𐑕 𐑢𐑪𐑑 𐑢𐑱𐑤𐑦𐑙 𐑦𐑟, 𐑚𐑳𐑑 𐑘𐑰 𐑷𐑤𐑕𐑴 𐑢𐑪𐑯𐑑 𐑑 𐑜𐑴 𐑦𐑯 𐑹𐑛𐑼 𐑑 𐑕𐑰 𐑞 𐑢𐑻𐑤𐑛? 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑞𐑨𐑑 𐑢𐑪𐑑 𐑘𐑰 𐑕𐑧𐑛? 𐑲 𐑔𐑷𐑑 𐑕𐑴. 𐑢𐑧𐑤 𐑞𐑧𐑯, 𐑡𐑳𐑕𐑑 𐑕𐑑𐑧𐑐 𐑓𐑹𐑢𐑼𐑛 𐑞𐑺, 𐑯 𐑑𐑱𐑒 𐑩 𐑐𐑰𐑐 𐑴𐑝𐑼 𐑞 𐑢𐑧𐑞𐑼-𐑚𐑬, 𐑯 𐑞𐑧𐑯 𐑚𐑨𐑒 𐑑 𐑥𐑰 𐑯 𐑑𐑧𐑤 𐑥𐑰 𐑢𐑪𐑑 𐑘𐑰 𐑕𐑰 𐑞𐑺.”

𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑲 𐑕𐑑𐑫𐑛 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑐𐑳𐑟𐑩𐑤𐑛 𐑚𐑲 𐑞𐑦𐑕 𐑒𐑘𐑫𐑼𐑾𐑕 𐑮𐑦𐑒𐑢𐑧𐑕𐑑, 𐑯𐑪𐑑 𐑯𐑴𐑦𐑙 𐑦𐑜𐑟𐑨𐑒𐑑𐑤𐑦 𐑣𐑬 𐑑 𐑑𐑱𐑒 𐑦𐑑, 𐑢𐑧𐑞𐑼 𐑣𐑿𐑥𐑼𐑩𐑕𐑤𐑦 𐑹 𐑦𐑯 𐑻𐑯𐑩𐑕𐑑. 𐑚𐑳𐑑 𐑒𐑪𐑯𐑕𐑩𐑯𐑑𐑮𐑱𐑑𐑦𐑙 𐑷𐑤 𐑣𐑦𐑟 𐑒𐑮𐑴'𐑟 𐑓𐑰𐑑 𐑦𐑯𐑑𐑵 𐑢𐑳𐑯 𐑕𐑒𐑬𐑤, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑕𐑑𐑸𐑑𐑩𐑛 𐑥𐑰 𐑪𐑯 𐑞 𐑧𐑮𐑩𐑯𐑛.

𐑜𐑴𐑦𐑙 𐑓𐑹𐑢𐑼𐑛 𐑯 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑢𐑧𐑞𐑼 𐑚𐑬, 𐑲 𐑐𐑼𐑕𐑰𐑝𐑛 𐑞𐑨𐑑 𐑞 𐑖𐑦𐑐 𐑕𐑢𐑦𐑙𐑦𐑙 𐑑 𐑣𐑻 𐑨𐑙𐑒𐑼 𐑢𐑦𐑞 𐑞 𐑓𐑤𐑳𐑛-𐑑𐑲𐑛, 𐑢𐑪𐑟 𐑯𐑬 𐑴𐑚𐑤𐑰𐑒𐑤𐑦 𐑐𐑶𐑯𐑑𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑴𐑐𐑩𐑯 𐑴𐑖𐑩𐑯. 𐑞 𐑐𐑮𐑪𐑕𐑐𐑧𐑒𐑑 𐑢𐑪𐑟 𐑳𐑯𐑤𐑦𐑥𐑦𐑑𐑩𐑛, 𐑚𐑳𐑑 𐑦𐑒𐑕𐑰𐑛𐑦𐑙𐑤𐑦 𐑥𐑩𐑯𐑪𐑑𐑩𐑯𐑩𐑕 𐑯 𐑓𐑼𐑚𐑦𐑛𐑦𐑙; 𐑯𐑪𐑑 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑝𐑼𐑲𐑩𐑑𐑦 𐑞𐑨𐑑 𐑲 𐑒𐑫𐑛 𐑕𐑰.

“𐑢𐑧𐑤, 𐑢𐑪𐑑'𐑕 𐑞 𐑮𐑦𐑐𐑹𐑑?” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜 𐑢𐑧𐑯 𐑲 𐑒𐑱𐑥 𐑚𐑨𐑒; “𐑢𐑪𐑑 𐑛𐑦𐑛 𐑘𐑰 𐑕𐑰?”

“𐑯𐑪𐑑 𐑥𐑳𐑗,” 𐑲 𐑮𐑦𐑐𐑤𐑲𐑛—“𐑯𐑳𐑔𐑦𐑙 𐑚𐑳𐑑 𐑢𐑷𐑑𐑼; 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑣𐑼𐑲𐑟𐑩𐑯 𐑞𐑴, 𐑯 𐑞𐑺'𐑟 𐑩 𐑕𐑒𐑢𐑷𐑤 𐑒𐑳𐑥𐑦𐑙 𐑳𐑐, 𐑲 𐑔𐑦𐑙𐑒.”

“𐑢𐑧𐑤, 𐑢𐑪𐑑 𐑛𐑳𐑟 𐑞𐑬 𐑔𐑦𐑙𐑒 𐑞𐑧𐑯 𐑝 𐑕𐑰𐑦𐑙 𐑞 𐑢𐑻𐑤𐑛? 𐑛𐑵 𐑘𐑰 𐑢𐑦𐑖 𐑑 𐑜𐑴 𐑮𐑬𐑯𐑛 ·𐑒𐑱𐑐 ·𐑣𐑹𐑯 𐑑 𐑕𐑰 𐑧𐑯𐑦 𐑥𐑹 𐑝 𐑦𐑑, 𐑱? 𐑒𐑭𐑯𐑑 𐑘𐑰 𐑕𐑰 𐑞 𐑢𐑻𐑤𐑛 𐑢𐑺 𐑿 𐑕𐑑𐑨𐑯𐑛?”

𐑲 𐑢𐑪𐑟 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑕𐑑𐑨𐑜𐑼𐑛, 𐑚𐑳𐑑 𐑜𐑴 𐑩-𐑢𐑱𐑤𐑦𐑙 𐑲 𐑥𐑳𐑕𐑑, 𐑯 𐑲 𐑢𐑫𐑛; 𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑨𐑟 𐑜𐑫𐑛 𐑩 𐑖𐑦𐑐 𐑨𐑟 𐑧𐑯𐑦—𐑲 𐑔𐑷𐑑 𐑞 𐑚𐑧𐑕𐑑—𐑯 𐑷𐑤 𐑞𐑦𐑕 𐑲 𐑯𐑬 𐑮𐑦𐑐𐑰𐑑𐑩𐑛 𐑑 ·𐑐𐑰𐑤𐑧𐑜. 𐑕𐑰𐑦𐑙 𐑥𐑰 𐑕𐑴 𐑛𐑦𐑑𐑻𐑥𐑦𐑯𐑛, 𐑣𐑰 𐑦𐑒𐑕𐑐𐑮𐑧𐑕𐑑 𐑣𐑦𐑟 𐑢𐑦𐑤𐑦𐑙𐑯𐑩𐑕 𐑑 𐑖𐑦𐑐 𐑥𐑰.

“𐑯 𐑞𐑬 𐑥𐑱𐑩𐑕𐑑 𐑨𐑟 𐑢𐑧𐑤 𐑕𐑲𐑯 𐑞 𐑐𐑱𐑐𐑼𐑟 𐑮𐑲𐑑 𐑪𐑓,” 𐑣𐑰 𐑨𐑛𐑩𐑛—“𐑒𐑳𐑥 𐑩𐑤𐑪𐑙 𐑢𐑦𐑞 𐑘𐑰.” 𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙, 𐑣𐑰 𐑤𐑧𐑛 𐑞 𐑢𐑱 𐑚𐑦𐑤𐑴 𐑛𐑧𐑒 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑨𐑚𐑦𐑯.

𐑕𐑳𐑗, 𐑞𐑧𐑯, 𐑢𐑪𐑟 𐑞 𐑐𐑻𐑕𐑩𐑯 𐑞𐑨𐑑 𐑲 𐑕𐑷 𐑕𐑰𐑑𐑩𐑛 𐑪𐑯 𐑞 𐑑𐑮𐑨𐑯𐑕𐑩𐑥 𐑢𐑧𐑯 𐑲 𐑓𐑪𐑤𐑴𐑛 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑨𐑚𐑦𐑯. 𐑞 𐑕𐑐𐑱𐑕 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑛𐑧𐑒𐑕 𐑢𐑪𐑟 𐑕𐑥𐑷𐑤; 𐑯 𐑞𐑺, 𐑚𐑴𐑤𐑑-𐑳𐑐𐑮𐑲𐑑, 𐑕𐑨𐑑 𐑴𐑤𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑣𐑵 𐑷𐑤𐑢𐑱𐑟 𐑕𐑨𐑑 𐑕𐑴, 𐑯 𐑯𐑧𐑝𐑼 𐑤𐑰𐑯𐑛, 𐑯 𐑞𐑦𐑕 𐑑 𐑕𐑱𐑝 𐑣𐑦𐑟 𐑒𐑴𐑑 𐑑𐑱𐑤𐑟. 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛-𐑚𐑮𐑦𐑥 𐑢𐑪𐑟 𐑐𐑤𐑱𐑕𐑑 𐑚𐑦𐑕𐑲𐑛 𐑣𐑦𐑥; 𐑣𐑦𐑟 𐑤𐑧𐑜𐑟 𐑢𐑻 𐑕𐑑𐑦𐑓𐑤𐑦 𐑒𐑮𐑪𐑕𐑑; 𐑣𐑦𐑟 𐑛𐑮𐑨𐑚 𐑝𐑧𐑕𐑗𐑼 𐑢𐑪𐑟 𐑚𐑳𐑑𐑩𐑯𐑛 𐑳𐑐 𐑑 𐑣𐑦𐑟 𐑗𐑦𐑯; 𐑯 𐑕𐑐𐑧𐑒𐑑𐑩𐑒𐑩𐑤𐑟 𐑪𐑯 𐑯𐑴𐑟, 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑩𐑚𐑕𐑹𐑚𐑛 𐑦𐑯 𐑮𐑰𐑛𐑦𐑙 𐑓𐑮𐑪𐑥 𐑩 𐑐𐑪𐑯𐑛𐑼𐑩𐑕 𐑝𐑪𐑤𐑿𐑥.

“·𐑚𐑦𐑤𐑛𐑨𐑛,” 𐑒𐑮𐑲𐑛 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜, “𐑨𐑑 𐑦𐑑 𐑩𐑜𐑱𐑯, ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑱? 𐑘𐑰 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑕𐑑𐑳𐑛𐑦𐑦𐑙 𐑞𐑴𐑟 ·𐑕𐑒𐑮𐑦𐑐𐑗𐑼𐑟, 𐑯𐑬, 𐑓 𐑞 𐑤𐑭𐑕𐑑 𐑔𐑻𐑑𐑦 𐑘𐑽𐑟, 𐑑 𐑥𐑲 𐑕𐑻𐑑𐑩𐑯 𐑯𐑪𐑤𐑦𐑡. 𐑣𐑬 𐑓𐑸 𐑘𐑰 𐑜𐑪𐑑, ·𐑚𐑦𐑤𐑛𐑨𐑛?”

𐑨𐑟 𐑦𐑓 𐑤𐑪𐑙 𐑣𐑩𐑚𐑦𐑗𐑵𐑱𐑑𐑩𐑛 𐑑 𐑕𐑳𐑗 𐑐𐑮𐑴𐑓𐑱𐑯 𐑑𐑷𐑒 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑖𐑦𐑐𐑥𐑱𐑑, ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑢𐑦𐑞𐑬𐑑 𐑯𐑴𐑑𐑦𐑕𐑦𐑙 𐑣𐑦𐑟 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑦𐑮𐑧𐑝𐑼𐑩𐑯𐑕, 𐑒𐑢𐑲𐑩𐑑𐑤𐑦 𐑤𐑫𐑒𐑑 𐑳𐑐, 𐑯 𐑕𐑰𐑦𐑙 𐑥𐑰, 𐑜𐑤𐑭𐑯𐑕𐑑 𐑩𐑜𐑱𐑯 𐑦𐑯𐑒𐑢𐑲𐑼𐑦𐑙𐑤𐑦 𐑑𐑩𐑢𐑹𐑛𐑟 ·𐑐𐑰𐑤𐑧𐑜.

“𐑣𐑰 𐑕𐑧𐑟 𐑣𐑰'𐑟 𐑬𐑼 𐑥𐑨𐑯, ·𐑚𐑦𐑤𐑛𐑨𐑛,” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜, “𐑣𐑰 𐑢𐑪𐑯𐑑𐑕 𐑑 𐑖𐑦𐑐.”

“𐑛𐑳𐑕𐑑 𐑞𐑰?” 𐑕𐑧𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑦𐑯 𐑩 𐑣𐑪𐑤𐑴 𐑑𐑴𐑯, 𐑯 𐑑𐑻𐑯𐑦𐑙 𐑮𐑬𐑯𐑛 𐑑 𐑥𐑰.

“𐑲 𐑛𐑳𐑕𐑑,” 𐑕𐑧𐑛 𐑲 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕𐑤𐑦, 𐑣𐑰 𐑢𐑪𐑟 𐑕𐑴 𐑦𐑯𐑑𐑧𐑯𐑕 𐑩 ·𐑒𐑢𐑱𐑒𐑼.

“𐑞𐑬 𐑯𐑴𐑩𐑕𐑑 𐑚𐑧𐑕𐑑,” 𐑢𐑪𐑟 𐑞 𐑕𐑧𐑐𐑩𐑤𐑒𐑮𐑩𐑤 𐑮𐑦𐑐𐑤𐑲, “𐑞 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯𐑔 𐑢𐑫𐑛𐑩𐑯𐑑 𐑚𐑰 𐑑𐑵 𐑥𐑳𐑗, 𐑢𐑫𐑛 𐑦𐑑?—'𐑢𐑺 𐑥𐑪𐑔 𐑯 𐑮𐑳𐑕𐑑 𐑛𐑵 𐑒𐑼𐑳𐑐𐑑, 𐑚𐑳𐑑 𐑤𐑱—'”

𐑤𐑱, 𐑦𐑯𐑛𐑰𐑛, 𐑔𐑷𐑑 𐑲, 𐑯 𐑕𐑳𐑗 𐑩 𐑤𐑱! 𐑞 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯𐑔! 𐑢𐑧𐑤, 𐑴𐑤𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑿 𐑸 𐑛𐑦𐑑𐑻𐑥𐑦𐑯𐑛 𐑞𐑨𐑑 𐑲, 𐑓 𐑢𐑳𐑯, 𐑖𐑨𐑤 𐑯𐑪𐑑 𐑤𐑱 𐑳𐑐 𐑥𐑧𐑯𐑦 𐑤𐑱𐑟 𐑣𐑽 𐑚𐑦𐑤𐑴, 𐑢𐑺 𐑥𐑪𐑔 𐑯 𐑮𐑳𐑕𐑑 𐑛𐑵 𐑒𐑼𐑳𐑐𐑑. 𐑦𐑑 𐑢𐑪𐑟 𐑩𐑯 𐑦𐑒𐑕𐑰𐑛𐑦𐑙𐑤𐑦 𐑤𐑪𐑙 𐑤𐑱 𐑞𐑨𐑑, 𐑦𐑯𐑛𐑰𐑛; 𐑯 𐑞𐑴 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑨𐑜𐑯𐑦𐑑𐑿𐑛 𐑝 𐑞 𐑓𐑦𐑜𐑼 𐑦𐑑 𐑥𐑲𐑑 𐑨𐑑 𐑓𐑻𐑕𐑑 𐑛𐑦𐑕𐑰𐑝 𐑩 𐑤𐑨𐑯𐑛𐑟𐑥𐑩𐑯, 𐑘𐑧𐑑 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑱𐑖𐑩𐑯 𐑢𐑦𐑤 𐑖𐑴 𐑞𐑨𐑑 𐑞𐑴 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯 𐑦𐑟 𐑩 𐑐𐑮𐑦𐑑𐑦 𐑤𐑸𐑡 𐑯𐑳𐑥𐑚𐑼, 𐑘𐑧𐑑, 𐑢𐑧𐑯 𐑿 𐑒𐑳𐑥 𐑑 𐑥𐑱𐑒 𐑩 𐑑𐑰𐑯𐑔 𐑝 𐑦𐑑, 𐑿 𐑢𐑦𐑤 𐑞𐑧𐑯 𐑕𐑰, 𐑲 𐑕𐑱, 𐑞𐑨𐑑 𐑞 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯𐑔 𐑐𐑸𐑑 𐑝 𐑩 𐑓𐑸𐑞𐑦𐑙 𐑦𐑟 𐑩 𐑜𐑫𐑛 𐑛𐑰𐑤 𐑤𐑧𐑕 𐑞𐑨𐑯 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯 𐑜𐑴𐑤𐑛 𐑛𐑳𐑚𐑤𐑵𐑯𐑟; 𐑯 𐑕𐑴 𐑲 𐑔𐑷𐑑 𐑨𐑑 𐑞 𐑑𐑲𐑥.

“𐑢𐑲, 𐑚𐑤𐑭𐑕𐑑 𐑘𐑹 𐑲𐑟, ·𐑚𐑦𐑤𐑛𐑨𐑛,” 𐑒𐑮𐑲𐑛 ·𐑐𐑰𐑤𐑧𐑜, “𐑞𐑬 𐑛𐑳𐑕𐑑 𐑯𐑪𐑑 𐑢𐑪𐑯𐑑 𐑑 𐑕𐑢𐑦𐑯𐑛𐑩𐑤 𐑞𐑦𐑕 𐑘𐑳𐑙 𐑥𐑨𐑯! 𐑣𐑰 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑥𐑹 𐑞𐑨𐑯 𐑞𐑨𐑑.”

“𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯𐑔,” 𐑩𐑜𐑱𐑯 𐑕𐑧𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑢𐑦𐑞𐑬𐑑 𐑤𐑦𐑓𐑑𐑦𐑙 𐑣𐑦𐑟 𐑲𐑟; 𐑯 𐑞𐑧𐑯 𐑢𐑧𐑯𐑑 𐑪𐑯 𐑥𐑳𐑥𐑚𐑩𐑤𐑦𐑙—“𐑓 𐑢𐑺 𐑘𐑹 𐑑𐑮𐑧𐑠𐑼 𐑦𐑟, 𐑞𐑺 𐑢𐑦𐑤 𐑘𐑹 𐑣𐑸𐑑 𐑚𐑰 𐑷𐑤𐑕𐑴.”

“𐑲 𐑨𐑥 𐑜𐑴𐑦𐑙 𐑑 𐑐𐑫𐑑 𐑣𐑦𐑥 𐑛𐑬𐑯 𐑓 𐑞 𐑔𐑮𐑰 𐑣𐑳𐑯𐑛𐑮𐑩𐑛𐑔,” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜, “𐑛𐑵 𐑘𐑰 𐑣𐑽 𐑞𐑨𐑑, ·𐑚𐑦𐑤𐑛𐑨𐑛! 𐑞 𐑔𐑮𐑰 𐑣𐑳𐑯𐑛𐑮𐑩𐑛𐑔 𐑤𐑱, 𐑲 𐑕𐑱.”

·𐑚𐑦𐑤𐑛𐑨𐑛 𐑤𐑱𐑛 𐑛𐑬𐑯 𐑣𐑦𐑟 𐑚𐑫𐑒, 𐑯 𐑑𐑻𐑯𐑦𐑙 𐑕𐑪𐑤𐑩𐑥𐑤𐑦 𐑑𐑩𐑢𐑹𐑛𐑟 𐑣𐑦𐑥 𐑕𐑧𐑛, “𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜, 𐑞𐑬 𐑣𐑨𐑕𐑑 𐑩 𐑡𐑧𐑯𐑼𐑩𐑕 𐑣𐑸𐑑; 𐑚𐑳𐑑 𐑞𐑬 𐑥𐑳𐑕𐑑 𐑒𐑩𐑯𐑕𐑦𐑛𐑼 𐑞 𐑛𐑿𐑑𐑦 𐑞𐑬 𐑴𐑩𐑕𐑑 𐑑 𐑞 𐑳𐑞𐑼 𐑴𐑯𐑼𐑟 𐑝 𐑞𐑦𐑕 𐑖𐑦𐑐—𐑢𐑦𐑛𐑴𐑟 𐑯 𐑹𐑓𐑩𐑯𐑟, 𐑥𐑧𐑯𐑦 𐑝 𐑞𐑧𐑥—𐑯 𐑞𐑨𐑑 𐑦𐑓 𐑢𐑰 𐑑𐑵 𐑩𐑚𐑳𐑯𐑛𐑩𐑯𐑑𐑤𐑦 𐑮𐑦𐑢𐑹𐑛 𐑞 𐑤𐑱𐑚𐑼𐑟 𐑝 𐑞𐑦𐑕 𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑢𐑰 𐑥𐑱 𐑚𐑰 𐑑𐑱𐑒𐑦𐑙 𐑞 𐑚𐑮𐑧𐑛 𐑓𐑮𐑪𐑥 𐑞𐑴𐑟 𐑢𐑦𐑛𐑴𐑟 𐑯 𐑞𐑴𐑟 𐑹𐑓𐑩𐑯𐑟. 𐑞 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯𐑔 𐑤𐑱, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜.”

“𐑞𐑬 ·𐑚𐑦𐑤𐑛𐑨𐑛!” 𐑮𐑹𐑛 ·𐑐𐑰𐑤𐑧𐑜, 𐑕𐑑𐑸𐑑𐑦𐑙 𐑳𐑐 𐑯 𐑒𐑤𐑨𐑑𐑼𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞 𐑒𐑨𐑚𐑦𐑯. “𐑚𐑤𐑭𐑕𐑑 𐑘𐑰, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑦𐑓 𐑲 𐑣𐑨𐑛 𐑓𐑪𐑤𐑴𐑛 𐑞𐑲 𐑩𐑛𐑝𐑲𐑕 𐑦𐑯 𐑞𐑰𐑟 𐑥𐑨𐑑𐑼𐑟, 𐑲 𐑢𐑫𐑛 𐑩𐑓𐑹 𐑯𐑬 𐑣𐑨𐑛 𐑩 𐑒𐑪𐑯𐑖𐑩𐑯𐑕 𐑑 𐑤𐑳𐑜 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑 𐑢𐑫𐑛 𐑚𐑰 𐑣𐑧𐑝𐑦 𐑦𐑯𐑳𐑓 𐑑 𐑓𐑬𐑯𐑛𐑼 𐑞 𐑤𐑸𐑡𐑩𐑕𐑑 𐑖𐑦𐑐 𐑞𐑨𐑑 𐑧𐑝𐑼 𐑕𐑱𐑤𐑛 𐑮𐑬𐑯𐑛 ·𐑒𐑱𐑐 ·𐑣𐑹𐑯.”

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜,” 𐑕𐑧𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑕𐑑𐑧𐑛𐑦𐑤𐑦, “𐑞𐑲 𐑒𐑪𐑯𐑖𐑩𐑯𐑕 𐑥𐑱 𐑚𐑰 𐑛𐑮𐑷𐑦𐑙 𐑑𐑧𐑯 𐑦𐑯𐑗𐑩𐑟 𐑝 𐑢𐑷𐑑𐑼, 𐑹 𐑑𐑧𐑯 𐑓𐑨𐑞𐑩𐑥𐑟, 𐑲 𐑒𐑭𐑯𐑑 𐑑𐑧𐑤; 𐑚𐑳𐑑 𐑨𐑟 𐑞𐑬 𐑸𐑑 𐑕𐑑𐑦𐑤 𐑩𐑯 𐑦𐑥𐑐𐑧𐑯𐑦𐑑𐑩𐑯𐑑 𐑥𐑨𐑯, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜, 𐑲 𐑜𐑮𐑱𐑑𐑤𐑦 𐑓𐑽 𐑤𐑧𐑕𐑑 𐑞𐑲 𐑒𐑪𐑯𐑖𐑩𐑯𐑕 𐑚𐑰 𐑚𐑳𐑑 𐑩 𐑤𐑰𐑒𐑦 𐑢𐑳𐑯; 𐑯 𐑢𐑦𐑤 𐑦𐑯 𐑞 𐑧𐑯𐑛 𐑕𐑦𐑙𐑒 𐑞𐑰 𐑓𐑬𐑯𐑛𐑼𐑦𐑙 𐑛𐑬𐑯 𐑑 𐑞 𐑓𐑲𐑼𐑦 𐑐𐑦𐑑, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜.”

“𐑓𐑲𐑼𐑦 𐑐𐑦𐑑! 𐑓𐑲𐑼𐑦 𐑐𐑦𐑑! 𐑘𐑰 𐑦𐑯𐑕𐑳𐑤𐑑 𐑥𐑰, 𐑥𐑨𐑯; 𐑐𐑭𐑕𐑑 𐑷𐑤 𐑯𐑨𐑗𐑼𐑩𐑤 𐑚𐑺𐑦𐑙, 𐑘𐑰 𐑦𐑯𐑕𐑳𐑤𐑑 𐑥𐑰. 𐑦𐑑'𐑕 𐑩𐑯 𐑷𐑤-𐑓𐑲𐑼𐑛 𐑬𐑑𐑮𐑱𐑡 𐑑 𐑑𐑧𐑤 𐑧𐑯𐑦 𐑣𐑿𐑥𐑩𐑯 𐑒𐑮𐑰𐑗𐑼 𐑞𐑨𐑑 𐑣𐑰'𐑟 𐑚𐑬𐑯𐑛 𐑑 𐑣𐑧𐑤. 𐑓𐑤𐑵𐑒𐑕 𐑯 𐑓𐑤𐑱𐑥𐑟! ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑕𐑱 𐑞𐑨𐑑 𐑩𐑜𐑱𐑯 𐑑 𐑥𐑰, 𐑯 𐑕𐑑𐑸𐑑 𐑥𐑲 𐑕𐑴𐑤-𐑚𐑴𐑤𐑑𐑕, 𐑚𐑳𐑑 𐑲'𐑤—𐑲'𐑤—𐑘𐑧𐑕, 𐑲'𐑤 𐑕𐑢𐑪𐑤𐑴 𐑩 𐑤𐑲𐑝 𐑜𐑴𐑑 𐑢𐑦𐑞 𐑷𐑤 𐑣𐑦𐑟 𐑣𐑺 𐑯 𐑣𐑹𐑯𐑟 𐑪𐑯. 𐑬𐑑 𐑝 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑘𐑰 𐑒𐑨𐑯𐑑𐑦𐑙, 𐑛𐑮𐑨𐑚-𐑒𐑳𐑤𐑼𐑛 𐑕𐑳𐑯 𐑝 𐑩 𐑢𐑫𐑛𐑩𐑯 𐑜𐑳𐑯—𐑩 𐑕𐑑𐑮𐑱𐑑 𐑢𐑱𐑒 𐑢𐑦𐑞 𐑘𐑰!”

𐑨𐑟 𐑣𐑰 𐑔𐑳𐑯𐑛𐑼𐑛 𐑬𐑑 𐑞𐑦𐑕 𐑣𐑰 𐑥𐑱𐑛 𐑩 𐑮𐑳𐑖 𐑨𐑑 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑩 𐑥𐑸𐑝𐑩𐑤𐑩𐑕 𐑴𐑚𐑤𐑰𐑒, 𐑕𐑤𐑲𐑛𐑦𐑙 𐑕𐑩𐑤𐑧𐑮𐑦𐑑𐑦, ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑓 𐑞𐑨𐑑 𐑑𐑲𐑥 𐑦𐑤𐑵𐑛𐑩𐑛 𐑣𐑦𐑥.

𐑩𐑤𐑸𐑥𐑛 𐑨𐑑 𐑞𐑦𐑕 𐑑𐑧𐑮𐑦𐑚𐑩𐑤 𐑬𐑑𐑚𐑻𐑕𐑑 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑑𐑵 𐑐𐑮𐑦𐑯𐑕𐑦𐑐𐑩𐑤 𐑯 𐑮𐑦𐑕𐑐𐑪𐑯𐑕𐑩𐑚𐑩𐑤 𐑴𐑯𐑼𐑟 𐑝 𐑞 𐑖𐑦𐑐, 𐑯 𐑓𐑰𐑤𐑦𐑙 𐑣𐑭𐑓 𐑩 𐑥𐑲𐑯𐑛 𐑑 𐑜𐑦𐑝 𐑳𐑐 𐑷𐑤 𐑲𐑛𐑾 𐑝 𐑕𐑱𐑤𐑦𐑙 𐑦𐑯 𐑩 𐑝𐑧𐑕𐑩𐑤 𐑕𐑴 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑩𐑚𐑤𐑦 𐑴𐑯𐑛 𐑯 𐑑𐑧𐑥𐑐𐑼𐑺𐑦𐑤𐑦 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛, 𐑲 𐑕𐑑𐑧𐑐𐑑 𐑩𐑕𐑲𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑛𐑹 𐑑 𐑜𐑦𐑝 𐑰𐑜𐑮𐑧𐑕 𐑑 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑣𐑵, 𐑲 𐑥𐑱𐑛 𐑯𐑴 𐑛𐑬𐑑, 𐑢𐑪𐑟 𐑷𐑤 𐑰𐑜𐑼𐑯𐑩𐑕 𐑑 𐑝𐑨𐑯𐑦𐑖 𐑓𐑮𐑪𐑥 𐑚𐑦𐑓𐑹 𐑞 𐑩𐑢𐑱𐑒𐑩𐑯𐑛 𐑮𐑪𐑔 𐑝 ·𐑐𐑰𐑤𐑧𐑜. 𐑚𐑳𐑑 𐑑 𐑥𐑲 𐑩𐑕𐑑𐑪𐑯𐑦𐑖𐑥𐑩𐑯𐑑, 𐑣𐑰 𐑕𐑨𐑑 𐑛𐑬𐑯 𐑩𐑜𐑱𐑯 𐑪𐑯 𐑞 𐑑𐑮𐑨𐑯𐑕𐑩𐑥 𐑝𐑧𐑮𐑦 𐑒𐑢𐑲𐑩𐑑𐑤𐑦, 𐑯 𐑕𐑰𐑥𐑛 𐑑 𐑣𐑨𐑝 𐑯𐑪𐑑 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑦𐑯𐑑𐑧𐑯𐑖𐑩𐑯 𐑝 𐑢𐑦𐑞𐑛𐑮𐑷𐑦𐑙. 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑒𐑢𐑲𐑑 𐑿𐑟𐑛 𐑑 𐑦𐑥𐑐𐑧𐑯𐑦𐑑𐑩𐑯𐑑 ·𐑐𐑰𐑤𐑧𐑜 𐑯 𐑣𐑦𐑟 𐑢𐑱𐑟. 𐑨𐑟 𐑓 ·𐑐𐑰𐑤𐑧𐑜, 𐑭𐑓𐑑𐑼 𐑤𐑧𐑑𐑦𐑙 𐑪𐑓 𐑣𐑦𐑟 𐑮𐑱𐑡 𐑨𐑟 𐑣𐑰 𐑣𐑨𐑛, 𐑞𐑺 𐑕𐑰𐑥𐑛 𐑯𐑴 𐑥𐑹 𐑤𐑧𐑓𐑑 𐑦𐑯 𐑣𐑦𐑥, 𐑯 𐑣𐑰, 𐑑𐑵, 𐑕𐑨𐑑 𐑛𐑬𐑯 𐑤𐑲𐑒 𐑩 𐑤𐑨𐑥, 𐑞𐑴 𐑣𐑰 𐑑𐑢𐑦𐑗𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑨𐑟 𐑦𐑓 𐑕𐑑𐑦𐑤 𐑯𐑻𐑝𐑩𐑕𐑤𐑦 𐑨𐑡𐑦𐑑𐑱𐑑𐑩𐑛. “𐑓𐑿!” 𐑣𐑰 𐑢𐑦𐑕𐑩𐑤𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑—“𐑞 𐑕𐑒𐑢𐑷𐑤'𐑟 𐑜𐑪𐑯 𐑪𐑓 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑲 𐑔𐑦𐑙𐑒. ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑞𐑬 𐑿𐑟𐑛 𐑑 𐑚𐑰 𐑜𐑫𐑛 𐑨𐑑 𐑖𐑸𐑐𐑩𐑯𐑦𐑙 𐑩 𐑤𐑭𐑯𐑕, 𐑥𐑧𐑯𐑛 𐑞𐑨𐑑 𐑐𐑧𐑯, 𐑢𐑦𐑤 𐑘𐑰. 𐑥𐑲 𐑡𐑨𐑒-𐑯𐑲𐑓 𐑣𐑽 𐑯𐑰𐑛𐑟 𐑞 𐑜𐑮𐑲𐑯𐑛𐑕𐑑𐑴𐑯. 𐑞𐑨𐑑'𐑕 𐑣𐑰; 𐑔𐑨𐑙𐑒 𐑘𐑰, ·𐑚𐑦𐑤𐑛𐑨𐑛. 𐑯𐑬 𐑞𐑧𐑯, 𐑥𐑲 𐑘𐑳𐑙 𐑥𐑨𐑯, ·𐑦𐑖𐑥𐑱𐑩𐑤'𐑟 𐑞𐑲 𐑯𐑱𐑥, 𐑛𐑦𐑛𐑩𐑯𐑑 𐑘𐑰 𐑕𐑱? 𐑢𐑧𐑤 𐑞𐑧𐑯, 𐑛𐑬𐑯 𐑘𐑰 𐑜𐑴 𐑣𐑽, ·𐑦𐑖𐑥𐑱𐑩𐑤, 𐑓 𐑞 𐑔𐑮𐑰 𐑣𐑳𐑯𐑛𐑮𐑩𐑛𐑔 𐑤𐑱.”

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜,” 𐑕𐑧𐑛 𐑲, “𐑲 𐑣𐑨𐑝 𐑩 𐑓𐑮𐑧𐑯𐑛 𐑢𐑦𐑞 𐑥𐑰 𐑣𐑵 𐑢𐑪𐑯𐑑𐑕 𐑑 𐑖𐑦𐑐 𐑑𐑵—𐑖𐑨𐑤 𐑲 𐑚𐑮𐑦𐑙 𐑣𐑦𐑥 𐑛𐑬𐑯 𐑑-𐑥𐑪𐑮𐑴?”

“𐑑 𐑚𐑰 𐑖𐑫𐑼,” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜. “𐑓𐑧𐑗 𐑣𐑦𐑥 𐑩𐑤𐑪𐑙, 𐑯 𐑢𐑰'𐑤 𐑤𐑫𐑒 𐑨𐑑 𐑣𐑦𐑥.”

“𐑢𐑪𐑑 𐑤𐑱 𐑛𐑳𐑟 𐑣𐑰 𐑢𐑪𐑯𐑑?” 𐑜𐑮𐑴𐑯𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑳𐑐 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑫𐑒 𐑦𐑯 𐑢𐑦𐑗 𐑣𐑰 𐑣𐑨𐑛 𐑩𐑜𐑱𐑯 𐑚𐑰𐑯 𐑚𐑧𐑮𐑦𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓.

“𐑴! 𐑯𐑧𐑝𐑼 𐑞𐑰 𐑥𐑲𐑯𐑛 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑, ·𐑚𐑦𐑤𐑛𐑨𐑛,” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜. “𐑣𐑨𐑟 𐑣𐑰 𐑧𐑝𐑼 𐑢𐑱𐑤𐑛 𐑦𐑑 𐑧𐑯𐑦?” 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑥𐑰.

“𐑒𐑦𐑤𐑛 𐑥𐑹 𐑢𐑱𐑤𐑟 𐑞𐑨𐑯 𐑲 𐑒𐑨𐑯 𐑒𐑬𐑯𐑑, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜.”

“𐑢𐑧𐑤, 𐑚𐑮𐑦𐑙 𐑣𐑦𐑥 𐑩𐑤𐑪𐑙 𐑞𐑧𐑯.”

𐑯, 𐑭𐑓𐑑𐑼 𐑕𐑲𐑯𐑦𐑙 𐑞 𐑐𐑱𐑐𐑼𐑟, 𐑪𐑓 𐑲 𐑢𐑧𐑯𐑑; 𐑯𐑳𐑔𐑦𐑙 𐑛𐑬𐑑𐑦𐑙 𐑚𐑳𐑑 𐑞𐑨𐑑 𐑲 𐑣𐑨𐑛 𐑛𐑳𐑯 𐑩 𐑜𐑫𐑛 𐑥𐑹𐑯𐑦𐑙'𐑟 𐑢𐑻𐑒, 𐑯 𐑞𐑨𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑞 𐑲𐑛𐑧𐑯𐑑𐑦𐑒𐑩𐑤 𐑖𐑦𐑐 𐑞𐑨𐑑 ·𐑘𐑴𐑡𐑴 𐑣𐑨𐑛 𐑐𐑮𐑩𐑝𐑲𐑛𐑩𐑛 𐑑 𐑒𐑨𐑮𐑦 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑯 𐑥𐑰 𐑮𐑬𐑯𐑛 𐑞 ·𐑒𐑱𐑐.

𐑚𐑳𐑑 𐑲 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑐𐑮𐑴𐑕𐑰𐑛𐑩𐑛 𐑓𐑸, 𐑢𐑧𐑯 𐑲 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑚𐑦𐑔𐑦𐑙𐑒 𐑥𐑰 𐑞𐑨𐑑 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑢𐑦𐑞 𐑣𐑵𐑥 𐑲 𐑢𐑪𐑟 𐑑 𐑕𐑱𐑤 𐑘𐑧𐑑 𐑮𐑦𐑥𐑱𐑯𐑛 𐑳𐑯𐑕𐑰𐑯 𐑚𐑲 𐑥𐑰; 𐑞𐑴, 𐑦𐑯𐑛𐑰𐑛, 𐑦𐑯 𐑥𐑧𐑯𐑦 𐑒𐑱𐑕𐑩𐑟, 𐑩 𐑢𐑱𐑤-𐑖𐑦𐑐 𐑢𐑦𐑤 𐑚𐑰 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑓𐑦𐑑𐑩𐑛 𐑬𐑑, 𐑯 𐑮𐑦𐑕𐑰𐑝 𐑷𐑤 𐑣𐑻 𐑒𐑮𐑵 𐑪𐑯 𐑚𐑹𐑛, 𐑺 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑥𐑱𐑒𐑕 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑚𐑲 𐑼𐑲𐑝𐑦𐑙 𐑑 𐑑𐑱𐑒 𐑒𐑩𐑥𐑭𐑯𐑛; 𐑓 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑞𐑰𐑟 𐑝𐑶𐑦𐑡𐑩𐑟 𐑸 𐑕𐑴 𐑐𐑮𐑴𐑤𐑪𐑙𐑛, 𐑯 𐑞 𐑖𐑹 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑨𐑑 𐑣𐑴𐑥 𐑕𐑴 𐑦𐑒𐑕𐑰𐑛𐑦𐑙𐑤𐑦 𐑚𐑮𐑰𐑓, 𐑞𐑨𐑑 𐑦𐑓 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑣𐑨𐑝 𐑩 𐑓𐑨𐑥𐑤𐑦, 𐑹 𐑧𐑯𐑦 𐑩𐑚𐑕𐑹𐑚𐑦𐑙 𐑒𐑩𐑯𐑕𐑻𐑯𐑥𐑩𐑯𐑑 𐑝 𐑞𐑨𐑑 𐑕𐑹𐑑, 𐑣𐑰 𐑛𐑳𐑟 𐑯𐑪𐑑 𐑑𐑮𐑳𐑚𐑩𐑤 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑥𐑳𐑗 𐑩𐑚𐑬𐑑 𐑣𐑦𐑟 𐑖𐑦𐑐 𐑦𐑯 𐑐𐑹𐑑, 𐑚𐑳𐑑 𐑤𐑰𐑝𐑟 𐑣𐑻 𐑑 𐑞 𐑴𐑯𐑼𐑟 𐑑𐑦𐑤 𐑷𐑤 𐑦𐑟 𐑮𐑧𐑛𐑦 𐑓 𐑕𐑰. 𐑣𐑬𐑧𐑝𐑼, 𐑦𐑑 𐑦𐑟 𐑷𐑤𐑢𐑱𐑟 𐑨𐑟 𐑢𐑧𐑤 𐑑 𐑣𐑨𐑝 𐑩 𐑤𐑫𐑒 𐑨𐑑 𐑣𐑦𐑥 𐑚𐑦𐑓𐑹 𐑦𐑮𐑧𐑝𐑩𐑒𐑩𐑚𐑤𐑦 𐑒𐑩𐑥𐑦𐑑𐑦𐑙 𐑘𐑹𐑕𐑧𐑤𐑓 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛𐑟. 𐑑𐑻𐑯𐑦𐑙 𐑚𐑨𐑒 𐑲 𐑩𐑒𐑪𐑕𐑑𐑩𐑛 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜, 𐑦𐑯𐑒𐑢𐑲𐑼𐑦𐑙 𐑢𐑺 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑢𐑪𐑟 𐑑 𐑚𐑰 𐑓𐑬𐑯𐑛.

“𐑯 𐑢𐑪𐑑 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑢𐑪𐑯𐑑 𐑝 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚? 𐑦𐑑'𐑕 𐑷𐑤 𐑮𐑲𐑑 𐑦𐑯𐑳𐑓; 𐑞𐑬 𐑸𐑑 𐑖𐑦𐑐𐑑.”

“𐑘𐑧𐑕, 𐑚𐑳𐑑 𐑲 𐑖𐑵𐑛 𐑤𐑲𐑒 𐑑 𐑕𐑰 𐑣𐑦𐑥.”

“𐑚𐑳𐑑 𐑲 𐑛𐑴𐑯𐑑 𐑔𐑦𐑙𐑒 𐑞𐑬 𐑢𐑦𐑤𐑑 𐑚𐑰 𐑱𐑚𐑩𐑤 𐑑 𐑨𐑑 𐑐𐑮𐑧𐑟𐑩𐑯𐑑. 𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑦𐑜𐑟𐑨𐑒𐑑𐑤𐑦 𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑣𐑦𐑥; 𐑚𐑳𐑑 𐑣𐑰 𐑒𐑰𐑐𐑕 𐑒𐑤𐑴𐑕 𐑦𐑯𐑕𐑲𐑛 𐑞 𐑣𐑬𐑕; 𐑩 𐑕𐑹𐑑 𐑝 𐑕𐑦𐑒, 𐑯 𐑘𐑧𐑑 𐑣𐑰 𐑛𐑴𐑯𐑑 𐑤𐑫𐑒 𐑕𐑴. 𐑦𐑯 𐑓𐑨𐑒𐑑, 𐑣𐑰 𐑱𐑯𐑑 𐑕𐑦𐑒; 𐑚𐑳𐑑 𐑯𐑴, 𐑣𐑰 𐑦𐑟𐑩𐑯𐑑 𐑢𐑧𐑤 𐑲𐑞𐑼. 𐑧𐑯𐑦 𐑣𐑬, 𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑣𐑰 𐑢𐑴𐑯𐑑 𐑷𐑤𐑢𐑱𐑟 𐑕𐑰 𐑥𐑰, 𐑕𐑴 𐑲 𐑛𐑴𐑯𐑑 𐑕𐑩𐑐𐑴𐑟 𐑣𐑰 𐑢𐑦𐑤 𐑞𐑰. 𐑣𐑰'𐑟 𐑩 𐑒𐑢𐑽 𐑥𐑨𐑯, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚—𐑕𐑴 𐑕𐑳𐑥 𐑔𐑦𐑙𐑒—𐑚𐑳𐑑 𐑩 𐑜𐑫𐑛 𐑢𐑳𐑯. 𐑴, 𐑞𐑬'𐑤𐑑 𐑤𐑲𐑒 𐑣𐑦𐑥 𐑢𐑧𐑤 𐑦𐑯𐑳𐑓; 𐑯𐑴 𐑓𐑽, 𐑯𐑴 𐑓𐑽. 𐑣𐑰'𐑟 𐑩 𐑜𐑮𐑨𐑯𐑛, 𐑳𐑯𐑜𐑪𐑛𐑤𐑦, 𐑜𐑪𐑛-𐑤𐑲𐑒 𐑥𐑨𐑯, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚; 𐑛𐑳𐑟𐑩𐑯𐑑 𐑕𐑐𐑰𐑒 𐑥𐑳𐑗; 𐑚𐑳𐑑, 𐑢𐑧𐑯 𐑣𐑰 𐑛𐑳𐑟 𐑕𐑐𐑰𐑒, 𐑞𐑧𐑯 𐑿 𐑥𐑱 𐑢𐑧𐑤 𐑤𐑦𐑕𐑩𐑯. 𐑥𐑸𐑒 𐑘𐑰, 𐑚𐑰 𐑓𐑹𐑢𐑹𐑯𐑛; ·𐑱𐑣𐑨𐑚'𐑟 𐑩𐑚𐑳𐑝 𐑞 𐑒𐑪𐑥𐑩𐑯; ·𐑱𐑣𐑨𐑚'𐑟 𐑚𐑰𐑯 𐑦𐑯 𐑒𐑪𐑤𐑦𐑡𐑩𐑟, 𐑨𐑟 𐑢𐑧𐑤 𐑨𐑟 '𐑥𐑪𐑙 𐑞 𐑒𐑨𐑯𐑦𐑚𐑩𐑤𐑟; 𐑚𐑰𐑯 𐑿𐑟𐑛 𐑑 𐑛𐑰𐑐𐑼 𐑢𐑳𐑯𐑛𐑼𐑟 𐑞𐑨𐑯 𐑞 𐑢𐑱𐑝𐑟; 𐑓𐑦𐑒𐑕𐑑 𐑣𐑦𐑟 𐑓𐑲𐑼𐑦 𐑤𐑭𐑯𐑕 𐑦𐑯 𐑥𐑲𐑑𐑽, 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑓𐑴𐑟 𐑞𐑨𐑯 𐑢𐑱𐑤𐑟. 𐑣𐑦𐑟 𐑤𐑭𐑯𐑕! 𐑲, 𐑞 𐑒𐑰𐑯𐑩𐑕𐑑 𐑯 𐑞 𐑖𐑫𐑼𐑩𐑕𐑑 𐑞𐑨𐑑 𐑬𐑑 𐑝 𐑷𐑤 𐑬𐑼 𐑲𐑤! 𐑴! 𐑣𐑰 𐑱𐑯𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛; 𐑯𐑴, 𐑯 𐑣𐑰 𐑱𐑯𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜; 𐑣𐑰'𐑟 ·𐑱𐑣𐑨𐑚, 𐑚𐑶; 𐑯 ·𐑱𐑣𐑨𐑚 𐑝 𐑴𐑤𐑛, 𐑞𐑬 𐑯𐑴𐑩𐑕𐑑, 𐑢𐑪𐑟 𐑩 𐑒𐑮𐑬𐑯𐑛 𐑒𐑦𐑙!”
