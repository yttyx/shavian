·𐑗𐑨𐑐𐑑𐑼 74. 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟 ·𐑣𐑧𐑛—·𐑒𐑩𐑯𐑑𐑮𐑭𐑕𐑑𐑩𐑛 ·𐑝𐑿.

𐑣𐑽, 𐑯𐑬, 𐑸 𐑑𐑵 𐑜𐑮𐑱𐑑 𐑢𐑱𐑤𐑟, 𐑤𐑱𐑦𐑙 𐑞𐑺 𐑣𐑧𐑛𐑟 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑤𐑧𐑑 𐑳𐑕 𐑡𐑶𐑯 𐑞𐑧𐑥, 𐑯 𐑤𐑱 𐑑𐑩𐑜𐑧𐑞𐑼 𐑬𐑼 𐑴𐑯.

𐑝 𐑞 𐑜𐑮𐑨𐑯𐑛 𐑹𐑛𐑼 𐑝 𐑓𐑴𐑤𐑦𐑴 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯𐑟, 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑯 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑸 𐑚𐑲 𐑓𐑸 𐑞 𐑥𐑴𐑕𐑑 𐑯𐑴𐑑𐑢𐑻𐑞𐑦. 𐑞𐑱 𐑸 𐑞 𐑴𐑯𐑤𐑦 𐑢𐑱𐑤𐑟 𐑮𐑧𐑜𐑿𐑤𐑼𐑤𐑦 𐑣𐑳𐑯𐑑𐑩𐑛 𐑚𐑲 𐑥𐑨𐑯. 𐑑 𐑞 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼, 𐑞𐑱 𐑐𐑮𐑦𐑟𐑧𐑯𐑑 𐑞 𐑑𐑵 𐑧𐑒𐑕𐑑𐑮𐑰𐑥𐑟 𐑝 𐑷𐑤 𐑞 𐑯𐑴𐑯 𐑝𐑼𐑲𐑩𐑑𐑦𐑟 𐑝 𐑞 𐑢𐑱𐑤. 𐑨𐑟 𐑞 𐑧𐑒𐑕𐑑𐑼𐑯𐑩𐑤 𐑛𐑦𐑓𐑼𐑩𐑯𐑕 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞𐑧𐑥 𐑦𐑟 𐑥𐑱𐑯𐑤𐑦 𐑩𐑚𐑟𐑻𐑝𐑩𐑚𐑩𐑤 𐑦𐑯 𐑞𐑺 𐑣𐑧𐑛𐑟; 𐑯 𐑨𐑟 𐑩 𐑣𐑧𐑛 𐑝 𐑰𐑗 𐑦𐑟 𐑞𐑦𐑕 𐑥𐑴𐑥𐑧𐑯𐑑 𐑣𐑨𐑙𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑕𐑲𐑛; 𐑯 𐑨𐑟 𐑢𐑰 𐑥𐑱 𐑓𐑮𐑰𐑤𐑦 𐑜𐑴 𐑓𐑮𐑪𐑥 𐑢𐑳𐑯 𐑑 𐑞 𐑳𐑞𐑼, 𐑚𐑲 𐑥𐑽𐑤𐑦 𐑕𐑑𐑧𐑐𐑦𐑙 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑛𐑧𐑒:—𐑢𐑺, 𐑲 𐑖𐑵𐑛 𐑤𐑲𐑒 𐑑 𐑯𐑴, 𐑢𐑦𐑤 𐑿 𐑩𐑚𐑑𐑱𐑯 𐑩 𐑚𐑧𐑑𐑼 𐑗𐑭𐑯𐑕 𐑑 𐑕𐑑𐑳𐑛𐑦 𐑐𐑮𐑨𐑒𐑑𐑦𐑒𐑩𐑤 ·𐑕𐑧𐑑𐑪𐑤𐑩𐑡𐑦 𐑞𐑨𐑯 𐑣𐑽?

𐑦𐑯 𐑞 𐑓𐑻𐑕𐑑 𐑐𐑤𐑱𐑕, 𐑿 𐑸 𐑕𐑑𐑮𐑳𐑒 𐑚𐑲 𐑞 𐑡𐑧𐑯𐑼𐑩𐑤 𐑒𐑪𐑯𐑑𐑮𐑭𐑕𐑑 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞𐑰𐑟 𐑣𐑧𐑛𐑟. 𐑚𐑴𐑔 𐑸 𐑥𐑨𐑕𐑦𐑝 𐑦𐑯𐑳𐑓 𐑦𐑯 𐑷𐑤 𐑒𐑪𐑯𐑖𐑩𐑯𐑕; 𐑚𐑳𐑑 𐑞𐑺 𐑦𐑟 𐑩 𐑕𐑻𐑑𐑩𐑯 𐑥𐑨𐑔𐑩𐑥𐑨𐑑𐑦𐑒𐑩𐑤 𐑕𐑦𐑥𐑦𐑑𐑮𐑦 𐑦𐑯 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟 𐑢𐑦𐑗 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑕𐑨𐑛𐑤𐑦 𐑤𐑨𐑒𐑕. 𐑞𐑺 𐑦𐑟 𐑥𐑹 𐑒𐑨𐑮𐑩𐑒𐑑𐑼 𐑦𐑯 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛. 𐑨𐑟 𐑿 𐑚𐑦𐑣𐑴𐑤𐑛 𐑦𐑑, 𐑿 𐑦𐑯𐑝𐑪𐑤𐑩𐑯𐑑𐑺𐑦𐑤𐑦 𐑘𐑰𐑤𐑛 𐑞 𐑦𐑥𐑧𐑯𐑕 𐑕𐑵𐑐𐑽𐑦𐑪𐑮𐑦𐑑𐑦 𐑑 𐑣𐑦𐑥, 𐑦𐑯 𐑐𐑶𐑯𐑑 𐑝 𐑐𐑼𐑝𐑱𐑛𐑦𐑙 𐑛𐑦𐑜𐑯𐑦𐑑𐑦. 𐑦𐑯 𐑞 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑕, 𐑑𐑵, 𐑞𐑦𐑕 𐑛𐑦𐑜𐑯𐑦𐑑𐑦 𐑦𐑟 𐑣𐑲𐑑𐑩𐑯𐑛 𐑚𐑲 𐑞 𐑐𐑧𐑐𐑼 𐑯 𐑕𐑷𐑤𐑑 𐑒𐑳𐑤𐑼 𐑝 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑨𐑑 𐑞 𐑕𐑳𐑥𐑦𐑑, 𐑜𐑦𐑝𐑦𐑙 𐑑𐑴𐑒𐑩𐑯 𐑝 𐑩𐑛𐑝𐑭𐑯𐑕𐑑 𐑱𐑡 𐑯 𐑤𐑸𐑡 𐑦𐑒𐑕𐑐𐑽𐑾𐑯𐑕. 𐑦𐑯 𐑖𐑹𐑑, 𐑣𐑰 𐑦𐑟 𐑢𐑪𐑑 𐑞 𐑓𐑦𐑖𐑼𐑥𐑧𐑯 𐑑𐑧𐑒𐑯𐑦𐑒𐑤𐑦 𐑒𐑷𐑤 𐑩 “𐑜𐑮𐑱-𐑣𐑧𐑛𐑩𐑛 𐑢𐑱𐑤.”

𐑤𐑧𐑑 𐑳𐑕 𐑯𐑬 𐑯𐑴𐑑 𐑢𐑪𐑑 𐑦𐑟 𐑤𐑰𐑕𐑑 𐑛𐑦𐑕𐑦𐑥𐑦𐑤𐑼 𐑦𐑯 𐑞𐑰𐑟 𐑣𐑧𐑛𐑟—𐑯𐑱𐑥𐑤𐑦, 𐑞 𐑑𐑵 𐑥𐑴𐑕𐑑 𐑦𐑥𐑐𐑹𐑑𐑩𐑯𐑑 𐑹𐑜𐑩𐑯𐑟, 𐑞 𐑲 𐑯 𐑞 𐑽. 𐑓𐑸 𐑚𐑨𐑒 𐑪𐑯 𐑞 𐑕𐑲𐑛 𐑝 𐑞 𐑣𐑧𐑛, 𐑯 𐑤𐑴 𐑛𐑬𐑯, 𐑯𐑽 𐑞 𐑨𐑙𐑜𐑩𐑤 𐑝 𐑲𐑞𐑼 𐑢𐑱𐑤'𐑟 𐑡𐑷, 𐑦𐑓 𐑿 𐑯𐑼𐑴𐑤𐑦 𐑕𐑻𐑗, 𐑿 𐑢𐑦𐑤 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑕𐑰 𐑩 𐑤𐑨𐑖𐑤𐑩𐑕 𐑲, 𐑢𐑦𐑗 𐑿 𐑢𐑫𐑛 𐑓𐑨𐑯𐑕𐑦 𐑑 𐑚𐑰 𐑩 𐑘𐑳𐑙 𐑒𐑴𐑤𐑑'𐑕 𐑲; 𐑕𐑴 𐑬𐑑 𐑝 𐑷𐑤 𐑐𐑮𐑩𐑐𐑹𐑖𐑩𐑯 𐑦𐑟 𐑦𐑑 𐑑 𐑞 𐑥𐑨𐑜𐑯𐑦𐑑𐑿𐑛 𐑝 𐑞 𐑣𐑧𐑛.

𐑯𐑬, 𐑓𐑮𐑪𐑥 𐑞𐑦𐑕 𐑐𐑦𐑒𐑿𐑤𐑽 𐑕𐑲𐑛𐑢𐑱 𐑐𐑩𐑟𐑦𐑖𐑩𐑯 𐑝 𐑞 𐑢𐑱𐑤'𐑟 𐑲𐑟, 𐑦𐑑 𐑦𐑟 𐑐𐑤𐑱𐑯 𐑞𐑨𐑑 𐑣𐑰 𐑒𐑨𐑯 𐑯𐑧𐑝𐑼 𐑕𐑰 𐑩𐑯 𐑪𐑚𐑡𐑧𐑒𐑑 𐑢𐑦𐑗 𐑦𐑟 𐑦𐑜𐑟𐑨𐑒𐑑𐑤𐑦 𐑩𐑣𐑧𐑛, 𐑯𐑴 𐑥𐑹 𐑞𐑨𐑯 𐑣𐑰 𐑒𐑨𐑯 𐑢𐑳𐑯 𐑦𐑜𐑟𐑨𐑒𐑑𐑤𐑦 𐑩𐑕𐑑𐑻𐑯. 𐑦𐑯 𐑩 𐑢𐑻𐑛, 𐑞 𐑐𐑩𐑟𐑦𐑖𐑩𐑯 𐑝 𐑞 𐑢𐑱𐑤'𐑟 𐑲𐑟 𐑒𐑪𐑮𐑦𐑕𐑐𐑪𐑯𐑛𐑟 𐑑 𐑞𐑨𐑑 𐑝 𐑩 𐑥𐑨𐑯'𐑟 𐑽𐑟; 𐑯 𐑿 𐑥𐑱 𐑓𐑨𐑯𐑕𐑦, 𐑓 𐑘𐑹𐑕𐑧𐑤𐑓, 𐑣𐑬 𐑦𐑑 𐑢𐑫𐑛 𐑓𐑺 𐑢𐑦𐑞 𐑿, 𐑛𐑦𐑛 𐑿 𐑕𐑲𐑛𐑢𐑱𐑟 𐑕𐑻𐑝𐑱 𐑪𐑚𐑡𐑧𐑒𐑑𐑕 𐑔𐑮𐑵 𐑘𐑹 𐑽𐑟. 𐑿 𐑢𐑫𐑛 𐑓𐑲𐑯𐑛 𐑞𐑨𐑑 𐑿 𐑒𐑫𐑛 𐑴𐑯𐑤𐑦 𐑒𐑩𐑥𐑭𐑯𐑛 𐑕𐑳𐑥 𐑔𐑻𐑑𐑦 𐑛𐑦𐑜𐑮𐑰𐑟 𐑝 𐑝𐑦𐑠𐑩𐑯 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕 𐑝 𐑞 𐑕𐑑𐑮𐑱𐑑 𐑕𐑲𐑛-𐑤𐑲𐑯 𐑝 𐑕𐑲𐑑; 𐑯 𐑩𐑚𐑬𐑑 𐑔𐑻𐑑𐑦 𐑥𐑹 𐑚𐑦𐑣𐑲𐑯𐑛 𐑦𐑑. 𐑦𐑓 𐑘𐑹 𐑚𐑦𐑑𐑼𐑩𐑕𐑑 𐑓𐑴 𐑢𐑻 𐑢𐑷𐑒𐑦𐑙 𐑕𐑑𐑮𐑱𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 𐑿, 𐑢𐑦𐑞 𐑛𐑨𐑜𐑼 𐑳𐑐𐑤𐑦𐑓𐑑𐑩𐑛 𐑦𐑯 𐑚𐑮𐑷𐑛 𐑛𐑱, 𐑿 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑚𐑰 𐑱𐑚𐑩𐑤 𐑑 𐑕𐑰 𐑣𐑦𐑥, 𐑧𐑯𐑦 𐑥𐑹 𐑞𐑨𐑯 𐑦𐑓 𐑣𐑰 𐑢𐑻 𐑕𐑑𐑰𐑤𐑦𐑙 𐑩𐑐𐑪𐑯 𐑿 𐑓𐑮𐑪𐑥 𐑚𐑦𐑣𐑲𐑯𐑛. 𐑦𐑯 𐑩 𐑢𐑻𐑛, 𐑿 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑑𐑵 𐑚𐑨𐑒𐑕, 𐑕𐑴 𐑑 𐑕𐑐𐑰𐑒; 𐑚𐑳𐑑, 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥, 𐑷𐑤𐑕𐑴, 𐑑𐑵 𐑓𐑮𐑳𐑯𐑑𐑕 (𐑕𐑲𐑛 𐑓𐑮𐑳𐑯𐑑𐑕): 𐑓 𐑢𐑪𐑑 𐑦𐑟 𐑦𐑑 𐑞𐑨𐑑 𐑥𐑱𐑒𐑕 𐑞 𐑓𐑮𐑳𐑯𐑑 𐑝 𐑩 𐑥𐑨𐑯—𐑢𐑪𐑑, 𐑦𐑯𐑛𐑰𐑛, 𐑚𐑳𐑑 𐑣𐑦𐑟 𐑲𐑟?

𐑤𐑧𐑑 𐑳𐑕 𐑯𐑬 𐑢𐑦𐑞 𐑢𐑪𐑑𐑧𐑝𐑼 𐑤𐑰𐑝𐑼𐑟 𐑯 𐑕𐑑𐑰𐑥-𐑧𐑯𐑡𐑦𐑯𐑟 𐑢𐑰 𐑣𐑨𐑝 𐑨𐑑 𐑣𐑨𐑯𐑛, 𐑒𐑨𐑯𐑑 𐑴𐑝𐑼 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛, 𐑞𐑨𐑑 𐑦𐑑 𐑥𐑱 𐑤𐑲 𐑚𐑪𐑑𐑩𐑥 𐑳𐑐; 𐑞𐑧𐑯, 𐑩𐑕𐑧𐑯𐑛𐑦𐑙 𐑚𐑲 𐑩 𐑤𐑨𐑛𐑼 𐑑 𐑞 𐑕𐑳𐑥𐑦𐑑, 𐑣𐑨𐑝 𐑩 𐑐𐑰𐑐 𐑛𐑬𐑯 𐑞 𐑥𐑬𐑔; 𐑯 𐑢𐑻 𐑦𐑑 𐑯𐑪𐑑 𐑞𐑨𐑑 𐑞 𐑚𐑪𐑛𐑦 𐑦𐑟 𐑯𐑬 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑕𐑧𐑐𐑼𐑱𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑦𐑑, 𐑢𐑦𐑞 𐑩 𐑤𐑨𐑯𐑑𐑼𐑯 𐑢𐑰 𐑥𐑲𐑑 𐑛𐑦𐑕𐑧𐑯𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑜𐑮𐑱𐑑 ·𐑒𐑩𐑯𐑑𐑳𐑒𐑦 ·𐑥𐑨𐑥𐑩𐑔 ·𐑒𐑱𐑝 𐑝 𐑣𐑦𐑟 𐑕𐑑𐑳𐑥𐑩𐑒. 𐑚𐑳𐑑 𐑤𐑧𐑑 𐑳𐑕 𐑣𐑴𐑤𐑛 𐑪𐑯 𐑣𐑽 𐑚𐑲 𐑞𐑦𐑕 𐑑𐑵𐑔, 𐑯 𐑤𐑫𐑒 𐑩𐑚𐑬𐑑 𐑳𐑕 𐑢𐑺 𐑢𐑰 𐑸. 𐑢𐑪𐑑 𐑩 𐑮𐑾𐑤𐑦 𐑚𐑿𐑑𐑦𐑓𐑩𐑤 𐑯 𐑗𐑱𐑕𐑑-𐑤𐑫𐑒𐑦𐑙 𐑥𐑬𐑔! 𐑓𐑮𐑪𐑥 𐑓𐑤𐑹 𐑑 𐑕𐑰𐑤𐑦𐑙, 𐑤𐑲𐑯𐑛, 𐑹 𐑮𐑭𐑞𐑼 𐑐𐑱𐑐𐑼𐑛 𐑢𐑦𐑞 𐑩 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑢𐑲𐑑 𐑥𐑧𐑥𐑚𐑮𐑱𐑯, 𐑜𐑤𐑪𐑕𐑦 𐑨𐑟 𐑚𐑮𐑲𐑛𐑩𐑤 𐑕𐑨𐑑𐑦𐑯𐑟.

𐑚𐑳𐑑 𐑒𐑳𐑥 𐑬𐑑 𐑯𐑬, 𐑯 𐑤𐑫𐑒 𐑨𐑑 𐑞𐑦𐑕 𐑐𐑹𐑑𐑧𐑯𐑑𐑩𐑕 𐑤𐑴𐑼 𐑡𐑷, 𐑢𐑦𐑗 𐑕𐑰𐑥𐑟 𐑤𐑲𐑒 𐑞 𐑤𐑪𐑙 𐑯𐑼𐑴 𐑤𐑦𐑛 𐑝 𐑩𐑯 𐑦𐑥𐑧𐑯𐑕 𐑕𐑯𐑳𐑓-𐑚𐑪𐑒𐑕, 𐑢𐑦𐑞 𐑞 𐑣𐑦𐑯𐑡 𐑨𐑑 𐑢𐑳𐑯 𐑧𐑯𐑛, 𐑦𐑯𐑕𐑑𐑧𐑛 𐑝 𐑢𐑳𐑯 𐑕𐑲𐑛. 𐑦𐑓 𐑿 𐑐𐑮𐑲 𐑦𐑑 𐑳𐑐, 𐑕𐑴 𐑨𐑟 𐑑 𐑜𐑧𐑑 𐑦𐑑 𐑴𐑝𐑼𐑣𐑧𐑛, 𐑯 𐑦𐑒𐑕𐑐𐑴𐑟 𐑦𐑑𐑕 𐑮𐑴𐑟 𐑝 𐑑𐑰𐑔, 𐑦𐑑 𐑕𐑰𐑥𐑟 𐑩 𐑑𐑼𐑦𐑓𐑦𐑒 𐑐𐑹𐑑𐑒𐑳𐑤𐑦𐑕; 𐑯 𐑕𐑳𐑗, 𐑩𐑤𐑨𐑕! 𐑦𐑑 𐑐𐑮𐑵𐑝𐑟 𐑑 𐑥𐑧𐑯𐑦 𐑩 𐑐𐑫𐑼 𐑢𐑲𐑑 𐑦𐑯 𐑞 𐑓𐑦𐑖𐑼𐑦, 𐑩𐑐𐑪𐑯 𐑣𐑵𐑥 𐑞𐑰𐑟 𐑕𐑐𐑲𐑒𐑕 𐑓𐑷𐑤 𐑢𐑦𐑞 𐑦𐑥𐑐𐑱𐑤𐑦𐑙 𐑓𐑹𐑕. 𐑚𐑳𐑑 𐑓𐑸 𐑥𐑹 𐑑𐑧𐑮𐑦𐑚𐑩𐑤 𐑦𐑟 𐑦𐑑 𐑑 𐑚𐑦𐑣𐑴𐑤𐑛, 𐑢𐑧𐑯 𐑓𐑨𐑞𐑩𐑥𐑟 𐑛𐑬𐑯 𐑦𐑯 𐑞 𐑕𐑰, 𐑿 𐑕𐑰 𐑕𐑳𐑥 𐑕𐑳𐑤𐑒𐑦 𐑢𐑱𐑤, 𐑓𐑤𐑴𐑑𐑦𐑙 𐑞𐑺 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛, 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑐𐑮𐑩𐑛𐑦𐑡𐑩𐑕 𐑡𐑷, 𐑕𐑳𐑥 𐑓𐑦𐑓𐑑𐑰𐑯 𐑓𐑰𐑑 𐑤𐑪𐑙, 𐑣𐑨𐑙𐑦𐑙 𐑕𐑑𐑮𐑱𐑑 𐑛𐑬𐑯 𐑨𐑑 𐑮𐑲𐑑-𐑨𐑙𐑜𐑩𐑤𐑟 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑚𐑪𐑛𐑦, 𐑓 𐑷𐑤 𐑞 𐑢𐑻𐑤𐑛 𐑤𐑲𐑒 𐑩 𐑖𐑦𐑐'𐑕 𐑡𐑦𐑚-𐑚𐑵𐑥. 𐑞𐑦𐑕 𐑢𐑱𐑤 𐑦𐑟 𐑯𐑪𐑑 𐑛𐑧𐑛; 𐑣𐑰 𐑦𐑟 𐑴𐑯𐑤𐑦 𐑛𐑦𐑕𐑐𐑦𐑮𐑦𐑑𐑩𐑛; 𐑬𐑑 𐑝 𐑕𐑹𐑑𐑕, 𐑐𐑼𐑣𐑨𐑐𐑕; 𐑣𐑲𐑐𐑴𐑒𐑪𐑯𐑛𐑮𐑦𐑨𐑒; 𐑯 𐑕𐑴 𐑕𐑵𐑐𐑲𐑯, 𐑞𐑨𐑑 𐑞 𐑣𐑦𐑯𐑡𐑩𐑟 𐑝 𐑣𐑦𐑟 𐑡𐑷 𐑣𐑨𐑝 𐑮𐑦𐑤𐑨𐑒𐑕𐑑, 𐑤𐑰𐑝𐑦𐑙 𐑣𐑦𐑥 𐑞𐑺 𐑦𐑯 𐑞𐑨𐑑 𐑳𐑯𐑜𐑱𐑯𐑤𐑦 𐑕𐑹𐑑 𐑝 𐑐𐑤𐑲𐑑, 𐑩 𐑮𐑦𐑐𐑮𐑴𐑗 𐑑 𐑷𐑤 𐑣𐑦𐑟 𐑑𐑮𐑲𐑚, 𐑣𐑵 𐑥𐑳𐑕𐑑, 𐑯𐑴 𐑛𐑬𐑑, 𐑦𐑥𐑐𐑮𐑦𐑒𐑱𐑑 𐑤𐑪𐑒-𐑡𐑷𐑟 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥.

𐑦𐑯 𐑥𐑴𐑕𐑑 𐑒𐑱𐑕𐑩𐑟 𐑞𐑦𐑕 𐑤𐑴𐑼 𐑡𐑷—𐑚𐑰𐑦𐑙 𐑰𐑟𐑦𐑤𐑦 𐑳𐑯𐑣𐑦𐑯𐑡𐑛 𐑚𐑲 𐑩 𐑐𐑮𐑨𐑒𐑑𐑦𐑕𐑑 𐑸𐑑𐑦𐑕𐑑—𐑦𐑟 𐑛𐑦𐑕𐑦𐑯𐑜𐑱𐑡𐑛 𐑯 𐑣𐑶𐑕𐑑𐑩𐑛 𐑪𐑯 𐑛𐑧𐑒 𐑓 𐑞 𐑐𐑻𐑐𐑩𐑕 𐑝 𐑧𐑒𐑕𐑑𐑮𐑨𐑒𐑑𐑦𐑙 𐑞 𐑲𐑝𐑼𐑦 𐑑𐑰𐑔, 𐑯 𐑓𐑻𐑯𐑦𐑖𐑦𐑙 𐑩 𐑕𐑩𐑐𐑤𐑲 𐑝 𐑞𐑨𐑑 𐑣𐑸𐑛 𐑢𐑲𐑑 𐑢𐑱𐑤𐑚𐑴𐑯 𐑢𐑦𐑞 𐑢𐑦𐑗 𐑞 𐑓𐑦𐑖𐑼𐑥𐑧𐑯 𐑓𐑨𐑖𐑩𐑯 𐑷𐑤 𐑕𐑹𐑑𐑕 𐑝 𐑒𐑘𐑫𐑼𐑾𐑕 𐑸𐑑𐑦𐑒𐑩𐑤𐑟, 𐑦𐑯𐑒𐑤𐑵𐑛𐑦𐑙 𐑒𐑱𐑯𐑟, 𐑳𐑥𐑚𐑮𐑧𐑤𐑩-𐑕𐑑𐑪𐑒𐑕, 𐑯 𐑣𐑨𐑯𐑛𐑩𐑤𐑟 𐑑 𐑮𐑲𐑛𐑦𐑙-𐑢𐑦𐑐𐑕.

𐑢𐑦𐑞 𐑩 𐑤𐑪𐑙, 𐑢𐑽𐑦 𐑣𐑶𐑕𐑑 𐑞 𐑡𐑷 𐑦𐑟 𐑛𐑮𐑨𐑜𐑛 𐑪𐑯 𐑚𐑹𐑛, 𐑨𐑟 𐑦𐑓 𐑦𐑑 𐑢𐑻 𐑩𐑯 𐑨𐑙𐑒𐑼; 𐑯 𐑢𐑧𐑯 𐑞 𐑐𐑮𐑪𐑐𐑼 𐑑𐑲𐑥 𐑒𐑳𐑥𐑟—𐑕𐑳𐑥 𐑓𐑿 𐑛𐑱𐑟 𐑭𐑓𐑑𐑼 𐑞 𐑳𐑞𐑼 𐑢𐑻𐑒—·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, ·𐑛𐑨𐑜𐑵, 𐑯 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, 𐑚𐑰𐑦𐑙 𐑷𐑤 𐑩𐑒𐑳𐑥𐑐𐑤𐑦𐑖𐑑 𐑛𐑧𐑯𐑑𐑦𐑕𐑑𐑕, 𐑸 𐑕𐑧𐑑 𐑑 𐑛𐑮𐑷𐑦𐑙 𐑑𐑰𐑔. 𐑢𐑦𐑞 𐑩 𐑒𐑰𐑯 𐑒𐑳𐑑𐑦𐑙-𐑕𐑐𐑱𐑛, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑤𐑭𐑯𐑕𐑩𐑟 𐑞 𐑜𐑳𐑥𐑟; 𐑞𐑧𐑯 𐑞 𐑡𐑷 𐑦𐑟 𐑤𐑨𐑖𐑑 𐑛𐑬𐑯 𐑑 𐑮𐑦𐑙𐑚𐑴𐑤𐑑𐑕, 𐑯 𐑩 𐑑𐑨𐑒𐑩𐑤 𐑚𐑰𐑦𐑙 𐑮𐑦𐑜𐑛 𐑓𐑮𐑪𐑥 𐑩𐑤𐑪𐑓𐑑, 𐑞𐑱 𐑛𐑮𐑨𐑜 𐑬𐑑 𐑞𐑰𐑟 𐑑𐑰𐑔, 𐑨𐑟 ·𐑥𐑦𐑖𐑦𐑜𐑩𐑯 𐑪𐑒𐑕𐑩𐑯 𐑛𐑮𐑨𐑜 𐑕𐑑𐑳𐑥𐑐𐑕 𐑝 𐑴𐑤𐑛 𐑴𐑒𐑕 𐑬𐑑 𐑝 𐑢𐑲𐑤𐑛 𐑢𐑫𐑛 𐑤𐑨𐑯𐑛𐑟. 𐑞𐑺 𐑸 𐑡𐑧𐑯𐑼𐑩𐑤𐑦 𐑓𐑹𐑑𐑦-𐑑𐑵 𐑑𐑰𐑔 𐑦𐑯 𐑷𐑤; 𐑦𐑯 𐑴𐑤𐑛 𐑢𐑱𐑤𐑟, 𐑥𐑳𐑗 𐑢𐑹𐑯 𐑛𐑬𐑯, 𐑚𐑳𐑑 𐑳𐑯𐑛𐑦𐑒𐑱𐑛; 𐑯𐑹 𐑓𐑦𐑤𐑛 𐑭𐑓𐑑𐑼 𐑬𐑼 𐑸𐑑𐑦𐑓𐑦𐑖𐑩𐑤 𐑓𐑨𐑖𐑩𐑯. 𐑞 𐑡𐑷 𐑦𐑟 𐑭𐑓𐑑𐑼𐑢𐑼𐑛𐑟 𐑕𐑷𐑯 𐑦𐑯𐑑𐑵 𐑕𐑤𐑨𐑚𐑟, 𐑯 𐑐𐑲𐑤𐑛 𐑩𐑢𐑱 𐑤𐑲𐑒 𐑡𐑶𐑕𐑑𐑕 𐑓 𐑚𐑦𐑤𐑛𐑦𐑙 𐑣𐑬𐑟𐑩𐑟.
