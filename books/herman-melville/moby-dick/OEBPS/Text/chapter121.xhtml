·𐑗𐑨𐑐𐑑𐑼 121. 𐑥𐑦𐑛𐑯𐑲𐑑.—𐑞 ·𐑓𐑴𐑒𐑕𐑩𐑤 ·𐑚𐑫𐑤𐑢𐑼𐑒𐑕.

𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒 𐑥𐑬𐑯𐑑𐑩𐑛 𐑪𐑯 𐑞𐑧𐑥, 𐑯 𐑐𐑭𐑕𐑦𐑙 𐑩𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑤𐑨𐑖𐑦𐑙𐑟 𐑴𐑝𐑼 𐑞 𐑨𐑙𐑒𐑼𐑟 𐑞𐑺 𐑣𐑨𐑙𐑦𐑙.

“𐑯𐑴, ·𐑕𐑑𐑳𐑚; 𐑿 𐑥𐑱 𐑐𐑬𐑯𐑛 𐑞𐑨𐑑 𐑯𐑪𐑑 𐑞𐑺 𐑨𐑟 𐑥𐑳𐑗 𐑨𐑟 𐑿 𐑐𐑤𐑰𐑟, 𐑚𐑳𐑑 𐑿 𐑢𐑦𐑤 𐑯𐑧𐑝𐑼 𐑐𐑬𐑯𐑛 𐑦𐑯𐑑𐑵 𐑥𐑰 𐑢𐑪𐑑 𐑿 𐑢𐑻 𐑡𐑳𐑕𐑑 𐑯𐑬 𐑕𐑱𐑦𐑙. 𐑯 𐑣𐑬 𐑤𐑪𐑙 𐑩𐑜𐑴 𐑦𐑟 𐑦𐑑 𐑕𐑦𐑯𐑕 𐑿 𐑕𐑧𐑛 𐑞 𐑝𐑧𐑮𐑦 𐑒𐑪𐑯𐑑𐑮𐑺𐑦? 𐑛𐑦𐑛𐑩𐑯𐑑 𐑿 𐑢𐑳𐑯𐑕 𐑕𐑱 𐑞𐑨𐑑 𐑢𐑪𐑑𐑧𐑝𐑼 𐑖𐑦𐑐 ·𐑱𐑣𐑨𐑚 𐑕𐑱𐑤𐑟 𐑦𐑯, 𐑞𐑨𐑑 𐑖𐑦𐑐 𐑖𐑵𐑛 𐑐𐑱 𐑕𐑳𐑥𐑔𐑦𐑙 𐑧𐑒𐑕𐑑𐑮𐑩 𐑪𐑯 𐑦𐑑𐑕 𐑦𐑯𐑖𐑫𐑼𐑩𐑯𐑕 𐑐𐑪𐑤𐑩𐑕𐑦, 𐑡𐑳𐑕𐑑 𐑨𐑟 𐑞𐑴 𐑦𐑑 𐑢𐑻 𐑤𐑴𐑛𐑩𐑛 𐑢𐑦𐑞 𐑐𐑬𐑛𐑼 𐑚𐑨𐑮𐑩𐑤𐑟 𐑭𐑓𐑑 𐑯 𐑚𐑪𐑒𐑕𐑩𐑟 𐑝 𐑤𐑵𐑕𐑦𐑓𐑼𐑟 𐑓𐑹𐑢𐑼𐑛? 𐑕𐑑𐑪𐑐, 𐑯𐑬; 𐑛𐑦𐑛𐑩𐑯𐑑 𐑿 𐑕𐑱 𐑕𐑴?”

“𐑢𐑧𐑤, 𐑕𐑩𐑐𐑴𐑟 𐑲 𐑛𐑦𐑛? 𐑢𐑪𐑑 𐑞𐑧𐑯? 𐑲'𐑝 𐑐𐑸𐑑 𐑗𐑱𐑯𐑡𐑛 𐑥𐑲 𐑓𐑤𐑧𐑖 𐑕𐑦𐑯𐑕 𐑞𐑨𐑑 𐑑𐑲𐑥, 𐑢𐑲 𐑯𐑪𐑑 𐑥𐑲 𐑥𐑲𐑯𐑛? 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑕𐑩𐑐𐑴𐑟𐑦𐑙 𐑢𐑰 𐑸 𐑤𐑴𐑛𐑩𐑛 𐑢𐑦𐑞 𐑐𐑬𐑛𐑼 𐑚𐑨𐑮𐑩𐑤𐑟 𐑭𐑓𐑑 𐑯 𐑤𐑵𐑕𐑦𐑓𐑼𐑟 𐑓𐑹𐑢𐑼𐑛; 𐑣𐑬 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑒𐑫𐑛 𐑞 𐑤𐑵𐑕𐑦𐑓𐑼𐑟 𐑜𐑧𐑑 𐑩𐑓𐑲𐑼 𐑦𐑯 𐑞𐑦𐑕 𐑛𐑮𐑧𐑯𐑗𐑦𐑙 𐑕𐑐𐑮𐑱 𐑣𐑽? 𐑢𐑲, 𐑥𐑲 𐑤𐑦𐑑𐑩𐑤 𐑥𐑨𐑯, 𐑿 𐑣𐑨𐑝 𐑐𐑮𐑦𐑑𐑦 𐑮𐑧𐑛 𐑣𐑺, 𐑚𐑳𐑑 𐑿 𐑒𐑫𐑛𐑩𐑯𐑑 𐑜𐑧𐑑 𐑩𐑓𐑲𐑼 𐑯𐑬. 𐑖𐑱𐑒 𐑘𐑹𐑕𐑧𐑤𐑓; 𐑿'𐑼 ·𐑩𐑒𐑢𐑺𐑾𐑕, 𐑹 𐑞 𐑢𐑷𐑑𐑼-𐑚𐑺𐑼, ·𐑓𐑤𐑭𐑕𐑒; 𐑥𐑲𐑑 𐑓𐑦𐑤 𐑐𐑦𐑗𐑼𐑟 𐑨𐑑 𐑘𐑹 𐑒𐑴𐑑 𐑒𐑪𐑤𐑼. 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑰, 𐑞𐑧𐑯, 𐑞𐑨𐑑 𐑓 𐑞𐑰𐑟 𐑧𐑒𐑕𐑑𐑮𐑩 𐑮𐑦𐑕𐑒𐑕 𐑞 ·𐑥𐑼𐑰𐑯 ·𐑦𐑯𐑖𐑫𐑼𐑩𐑯𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑟 𐑣𐑨𐑝 𐑧𐑒𐑕𐑑𐑮𐑩 𐑜𐑨𐑮𐑩𐑯𐑑𐑰𐑟? 𐑣𐑽 𐑸 𐑣𐑲𐑛𐑮𐑩𐑯𐑑𐑕, ·𐑓𐑤𐑭𐑕𐑒. 𐑚𐑳𐑑 𐑣𐑸𐑒, 𐑩𐑜𐑱𐑯, 𐑯 𐑲'𐑤 𐑭𐑯𐑕𐑼 𐑘𐑰 𐑞 𐑳𐑞𐑼 𐑔𐑦𐑙. 𐑓𐑻𐑕𐑑 𐑑𐑱𐑒 𐑘𐑹 𐑤𐑧𐑜 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑮𐑬𐑯 𐑝 𐑞 𐑨𐑙𐑒𐑼 𐑣𐑽, 𐑞𐑴, 𐑕𐑴 𐑲 𐑒𐑨𐑯 𐑐𐑭𐑕 𐑞 𐑮𐑴𐑐; 𐑯𐑬 𐑤𐑦𐑕𐑩𐑯. 𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑲𐑑𐑦 𐑛𐑦𐑓𐑼𐑩𐑯𐑕 𐑚𐑦𐑑𐑢𐑰𐑯 𐑣𐑴𐑤𐑛𐑦𐑙 𐑩 𐑥𐑭𐑕𐑑'𐑕 𐑤𐑲𐑑𐑯𐑦𐑙-𐑮𐑪𐑛 𐑦𐑯 𐑞 𐑕𐑑𐑹𐑥, 𐑯 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑒𐑤𐑴𐑕 𐑚𐑲 𐑩 𐑥𐑭𐑕𐑑 𐑞𐑨𐑑 𐑣𐑨𐑟𐑩𐑯𐑑 𐑜𐑪𐑑 𐑧𐑯𐑦 𐑤𐑲𐑑𐑯𐑦𐑙-𐑮𐑪𐑛 𐑨𐑑 𐑷𐑤 𐑦𐑯 𐑩 𐑕𐑑𐑹𐑥? 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑰, 𐑿 𐑑𐑦𐑥𐑚𐑼-𐑣𐑧𐑛, 𐑞𐑨𐑑 𐑯𐑴 𐑣𐑸𐑥 𐑒𐑨𐑯 𐑒𐑳𐑥 𐑑 𐑞 𐑣𐑴𐑤𐑛𐑼 𐑝 𐑞 𐑮𐑪𐑛, 𐑩𐑯𐑤𐑧𐑕 𐑞 𐑥𐑭𐑕𐑑 𐑦𐑟 𐑓𐑻𐑕𐑑 𐑕𐑑𐑮𐑳𐑒? 𐑢𐑪𐑑 𐑸 𐑿 𐑑𐑷𐑒𐑦𐑙 𐑩𐑚𐑬𐑑, 𐑞𐑧𐑯? 𐑯𐑪𐑑 𐑢𐑳𐑯 𐑖𐑦𐑐 𐑦𐑯 𐑩 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑒𐑨𐑮𐑦𐑟 𐑮𐑪𐑛𐑟, 𐑯 ·𐑱𐑣𐑨𐑚,—𐑲, 𐑥𐑨𐑯, 𐑯 𐑷𐑤 𐑝 𐑳𐑕,—𐑢𐑻 𐑦𐑯 𐑯𐑴 𐑥𐑹 𐑛𐑱𐑯𐑡𐑼 𐑞𐑧𐑯, 𐑦𐑯 𐑥𐑲 𐑐𐑫𐑼 𐑩𐑐𐑦𐑯𐑘𐑩𐑯, 𐑞𐑨𐑯 𐑷𐑤 𐑞 𐑒𐑮𐑵𐑟 𐑦𐑯 𐑑𐑧𐑯 𐑔𐑬𐑟𐑩𐑯𐑛 𐑖𐑦𐑐𐑕 𐑯𐑬 𐑕𐑱𐑤𐑦𐑙 𐑞 𐑕𐑰𐑟. 𐑢𐑲, 𐑿 ·𐑒𐑦𐑙-·𐑐𐑴𐑕𐑑, 𐑿, 𐑲 𐑕𐑩𐑐𐑴𐑟 𐑿 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑧𐑝𐑮𐑦 𐑥𐑨𐑯 𐑦𐑯 𐑞 𐑢𐑻𐑤𐑛 𐑜𐑴 𐑩𐑚𐑬𐑑 𐑢𐑦𐑞 𐑩 𐑕𐑥𐑷𐑤 𐑤𐑲𐑑𐑯𐑦𐑙-𐑮𐑪𐑛 𐑮𐑳𐑯𐑦𐑙 𐑳𐑐 𐑞 𐑒𐑹𐑯𐑼 𐑝 𐑣𐑦𐑟 𐑣𐑨𐑑, 𐑤𐑲𐑒 𐑩 𐑥𐑦𐑤𐑦𐑖𐑩 𐑪𐑓𐑦𐑕𐑼'𐑟 𐑕𐑒𐑿𐑼𐑛 𐑓𐑧𐑞𐑼, 𐑯 𐑑𐑮𐑱𐑤𐑦𐑙 𐑚𐑦𐑣𐑲𐑯𐑛 𐑤𐑲𐑒 𐑣𐑦𐑟 𐑕𐑨𐑖. 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑘𐑰 𐑚𐑰 𐑕𐑧𐑯𐑕𐑩𐑚𐑩𐑤, ·𐑓𐑤𐑭𐑕𐑒? 𐑦𐑑'𐑕 𐑰𐑟𐑦 𐑑 𐑚𐑰 𐑕𐑧𐑯𐑕𐑩𐑚𐑩𐑤; 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑘𐑰, 𐑞𐑧𐑯? 𐑧𐑯𐑦 𐑥𐑨𐑯 𐑢𐑦𐑞 𐑣𐑭𐑓 𐑩𐑯 𐑲 𐑒𐑨𐑯 𐑚𐑰 𐑕𐑧𐑯𐑕𐑩𐑚𐑩𐑤.”

“𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑞𐑨𐑑, ·𐑕𐑑𐑳𐑚. 𐑿 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑓𐑲𐑯𐑛 𐑦𐑑 𐑮𐑭𐑞𐑼 𐑣𐑸𐑛.”

“𐑘𐑧𐑕, 𐑢𐑧𐑯 𐑩 𐑓𐑧𐑤𐑴'𐑟 𐑕𐑴𐑒𐑑 𐑔𐑮𐑵, 𐑦𐑑'𐑕 𐑣𐑸𐑛 𐑑 𐑚𐑰 𐑕𐑧𐑯𐑕𐑩𐑚𐑩𐑤, 𐑞𐑨𐑑'𐑕 𐑩 𐑓𐑨𐑒𐑑. 𐑯 𐑲 𐑨𐑥 𐑩𐑚𐑬𐑑 𐑛𐑮𐑧𐑯𐑗𐑑 𐑢𐑦𐑞 𐑞𐑦𐑕 𐑕𐑐𐑮𐑱. 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛; 𐑒𐑨𐑗 𐑞 𐑑𐑻𐑯 𐑞𐑺, 𐑯 𐑐𐑭𐑕 𐑦𐑑. 𐑕𐑰𐑥𐑟 𐑑 𐑥𐑰 𐑢𐑰 𐑸 𐑤𐑨𐑖𐑦𐑙 𐑛𐑬𐑯 𐑞𐑰𐑟 𐑨𐑙𐑒𐑼𐑟 𐑯𐑬 𐑨𐑟 𐑦𐑓 𐑞𐑱 𐑢𐑻 𐑯𐑧𐑝𐑼 𐑜𐑴𐑦𐑙 𐑑 𐑚𐑰 𐑿𐑟𐑛 𐑩𐑜𐑱𐑯. 𐑑𐑲𐑦𐑙 𐑞𐑰𐑟 𐑑𐑵 𐑨𐑙𐑒𐑼𐑟 𐑣𐑽, ·𐑓𐑤𐑭𐑕𐑒, 𐑕𐑰𐑥𐑟 𐑤𐑲𐑒 𐑑𐑲𐑦𐑙 𐑩 𐑥𐑨𐑯'𐑟 𐑣𐑨𐑯𐑛𐑟 𐑚𐑦𐑣𐑲𐑯𐑛 𐑣𐑦𐑥. 𐑯 𐑢𐑪𐑑 𐑚𐑦𐑜 𐑡𐑧𐑯𐑼𐑩𐑕 𐑣𐑨𐑯𐑛𐑟 𐑞𐑱 𐑸, 𐑑 𐑚𐑰 𐑖𐑫𐑼. 𐑞𐑰𐑟 𐑸 𐑘𐑹 𐑲𐑼𐑯 𐑓𐑦𐑕𐑑𐑕, 𐑣𐑱? 𐑢𐑪𐑑 𐑩 𐑣𐑴𐑤𐑛 𐑞𐑱 𐑣𐑨𐑝, 𐑑𐑵! 𐑲 𐑢𐑳𐑯𐑛𐑼, ·𐑓𐑤𐑭𐑕𐑒, 𐑢𐑧𐑞𐑼 𐑞 𐑢𐑻𐑤𐑛 𐑦𐑟 𐑨𐑙𐑒𐑼𐑛 𐑧𐑯𐑦𐑢𐑺; 𐑦𐑓 𐑖𐑰 𐑦𐑟, 𐑖𐑰 𐑕𐑢𐑦𐑙𐑟 𐑢𐑦𐑞 𐑩𐑯 𐑳𐑯𐑒𐑪𐑥𐑩𐑯 𐑤𐑪𐑙 𐑒𐑱𐑚𐑩𐑤, 𐑞𐑴. 𐑞𐑺, 𐑣𐑨𐑥𐑼 𐑞𐑨𐑑 𐑯𐑪𐑑 𐑛𐑬𐑯, 𐑯 𐑢𐑰'𐑝 𐑛𐑳𐑯. 𐑕𐑴; 𐑯𐑧𐑒𐑕𐑑 𐑑 𐑑𐑳𐑗𐑦𐑙 𐑤𐑨𐑯𐑛, 𐑤𐑲𐑑𐑦𐑙 𐑪𐑯 𐑛𐑧𐑒 𐑦𐑟 𐑞 𐑥𐑴𐑕𐑑 𐑕𐑨𐑑𐑦𐑕𐑓𐑨𐑒𐑑𐑼𐑦. 𐑲 𐑕𐑱, 𐑡𐑳𐑕𐑑 𐑮𐑦𐑙 𐑬𐑑 𐑥𐑲 𐑡𐑨𐑒𐑩𐑑 𐑕𐑒𐑻𐑑𐑕, 𐑢𐑦𐑤 𐑘𐑰? 𐑔𐑨𐑙𐑒 𐑘𐑰. 𐑞𐑱 𐑤𐑭𐑓 𐑨𐑑 𐑤𐑪𐑙-𐑑𐑪𐑜𐑟 𐑕𐑴, ·𐑓𐑤𐑭𐑕𐑒; 𐑚𐑳𐑑 𐑕𐑰𐑥𐑟 𐑑 𐑥𐑰, 𐑩 𐑤𐑪𐑙 𐑑𐑱𐑤𐑛 𐑒𐑴𐑑 𐑷𐑑 𐑷𐑤𐑢𐑱𐑟 𐑑 𐑚𐑰 𐑢𐑹𐑯 𐑦𐑯 𐑷𐑤 𐑕𐑑𐑹𐑥𐑟 𐑩𐑓𐑤𐑴𐑑. 𐑞 𐑑𐑱𐑤𐑟 𐑑𐑱𐑐𐑼𐑦𐑙 𐑛𐑬𐑯 𐑞𐑨𐑑 𐑢𐑱, 𐑕𐑻𐑝 𐑑 𐑒𐑨𐑮𐑦 𐑪𐑓 𐑞 𐑢𐑷𐑑𐑼, 𐑛'𐑘𐑰 𐑕𐑰. 𐑕𐑱𐑥 𐑢𐑦𐑞 𐑒𐑪𐑒𐑑 𐑣𐑨𐑑𐑕; 𐑞 𐑒𐑪𐑒𐑕 𐑓𐑹𐑥 𐑜𐑱𐑚𐑩𐑤-𐑧𐑯𐑛 𐑰𐑝-𐑑𐑮𐑪𐑓𐑕, ·𐑓𐑤𐑭𐑕𐑒. 𐑯𐑴 𐑥𐑹 𐑥𐑳𐑙𐑒𐑦-𐑡𐑨𐑒𐑩𐑑𐑕 𐑯 𐑑𐑸𐑐𐑷𐑤𐑦𐑯𐑟 𐑓 𐑥𐑰; 𐑲 𐑥𐑳𐑕𐑑 𐑥𐑬𐑯𐑑 𐑩 𐑕𐑢𐑪𐑤𐑴-𐑑𐑱𐑤, 𐑯 𐑛𐑮𐑲𐑝 𐑛𐑬𐑯 𐑩 𐑚𐑰𐑝𐑼; 𐑕𐑴. 𐑣𐑩𐑤𐑴! 𐑓𐑿! 𐑞𐑺 𐑜𐑴𐑟 𐑥𐑲 𐑑𐑸𐑐𐑷𐑤𐑦𐑯 𐑴𐑝𐑼𐑚𐑹𐑛; ·𐑤𐑹𐑛, ·𐑤𐑹𐑛, 𐑞𐑨𐑑 𐑞 𐑢𐑦𐑯𐑛𐑟 𐑞𐑨𐑑 𐑒𐑳𐑥 𐑓𐑮𐑪𐑥 𐑣𐑧𐑝𐑩𐑯 𐑖𐑵𐑛 𐑚𐑰 𐑕𐑴 𐑳𐑯𐑥𐑨𐑯𐑼𐑤𐑦! 𐑞𐑦𐑕 𐑦𐑟 𐑩 𐑯𐑭𐑕𐑑𐑦 𐑯𐑲𐑑, 𐑤𐑨𐑛.”
