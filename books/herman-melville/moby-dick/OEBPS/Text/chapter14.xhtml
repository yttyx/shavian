·𐑗𐑨𐑐𐑑𐑼 14. ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑.

𐑯𐑳𐑔𐑦𐑙 𐑥𐑹 𐑣𐑨𐑐𐑩𐑯𐑛 𐑪𐑯 𐑞 𐑐𐑨𐑕𐑦𐑡 𐑢𐑻𐑞𐑦 𐑞 𐑥𐑧𐑯𐑖𐑩𐑯𐑦𐑙; 𐑕𐑴, 𐑭𐑓𐑑𐑼 𐑩 𐑓𐑲𐑯 𐑮𐑳𐑯, 𐑢𐑰 𐑕𐑱𐑓𐑤𐑦 𐑼𐑲𐑝𐑛 𐑦𐑯 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑.

·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑! 𐑑𐑱𐑒 𐑬𐑑 𐑘𐑹 𐑥𐑨𐑐 𐑯 𐑤𐑫𐑒 𐑨𐑑 𐑦𐑑. 𐑕𐑰 𐑢𐑪𐑑 𐑩 𐑮𐑾𐑤 𐑒𐑹𐑯𐑼 𐑝 𐑞 𐑢𐑻𐑤𐑛 𐑦𐑑 𐑪𐑒𐑿𐑐𐑲𐑟; 𐑣𐑬 𐑦𐑑 𐑕𐑑𐑨𐑯𐑛𐑟 𐑞𐑺, 𐑩𐑢𐑱 𐑪𐑓 𐑖𐑹, 𐑥𐑹 𐑤𐑴𐑯𐑤𐑦 𐑞𐑨𐑯 𐑞 ·𐑧𐑛𐑦𐑕𐑑𐑴𐑯 𐑤𐑲𐑑𐑣𐑬𐑕. 𐑤𐑫𐑒 𐑨𐑑 𐑦𐑑—𐑩 𐑥𐑽 𐑣𐑦𐑤𐑩𐑒, 𐑯 𐑧𐑤𐑚𐑴 𐑝 𐑕𐑨𐑯𐑛; 𐑷𐑤 𐑚𐑰𐑗, 𐑢𐑦𐑞𐑬𐑑 𐑩 𐑚𐑨𐑒𐑜𐑮𐑬𐑯𐑛. 𐑞𐑺 𐑦𐑟 𐑥𐑹 𐑕𐑨𐑯𐑛 𐑞𐑺 𐑞𐑨𐑯 𐑿 𐑢𐑫𐑛 𐑿𐑟 𐑦𐑯 𐑑𐑢𐑧𐑯𐑑𐑦 𐑘𐑽𐑟 𐑨𐑟 𐑩 𐑕𐑳𐑚𐑕𐑑𐑦𐑑𐑿𐑑 𐑓 𐑚𐑤𐑪𐑑𐑦𐑙 𐑐𐑱𐑐𐑼. 𐑕𐑳𐑥 𐑜𐑱𐑥𐑕𐑩𐑥 𐑢𐑲𐑑𐑕 𐑢𐑦𐑤 𐑑𐑧𐑤 𐑿 𐑞𐑨𐑑 𐑞𐑱 𐑣𐑨𐑓 𐑑 𐑐𐑤𐑭𐑯𐑑 𐑢𐑰𐑛𐑟 𐑞𐑺, 𐑞𐑱 𐑛𐑴𐑯𐑑 𐑜𐑮𐑴 𐑯𐑨𐑗𐑼𐑩𐑤𐑦; 𐑞𐑨𐑑 𐑞𐑱 𐑦𐑥𐑐𐑹𐑑 ·𐑒𐑨𐑯𐑩𐑛𐑩 𐑔𐑦𐑕𐑩𐑤𐑟; 𐑞𐑨𐑑 𐑞𐑱 𐑣𐑨𐑓 𐑑 𐑕𐑧𐑯𐑛 𐑚𐑦𐑘𐑪𐑯𐑛 𐑕𐑰𐑟 𐑓 𐑩 𐑕𐑐𐑲𐑤 𐑑 𐑕𐑑𐑪𐑐 𐑩 𐑤𐑰𐑒 𐑦𐑯 𐑩𐑯 𐑶𐑤 𐑒𐑭𐑕𐑒; 𐑞𐑨𐑑 𐑐𐑰𐑕𐑩𐑟 𐑝 𐑢𐑫𐑛 𐑦𐑯 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑸 𐑒𐑨𐑮𐑦𐑛 𐑩𐑚𐑬𐑑 𐑤𐑲𐑒 𐑚𐑦𐑑𐑕 𐑝 𐑞 𐑑𐑮𐑵 𐑒𐑮𐑪𐑕 𐑦𐑯 ·𐑮𐑴𐑥; 𐑞𐑨𐑑 𐑐𐑰𐑐𐑩𐑤 𐑞𐑺 𐑐𐑤𐑭𐑯𐑑 𐑑𐑴𐑛𐑕𐑑𐑵𐑤𐑟 𐑚𐑦𐑓𐑹 𐑞𐑺 𐑣𐑬𐑟𐑩𐑟, 𐑑 𐑜𐑧𐑑 𐑳𐑯𐑛𐑼 𐑞 𐑖𐑱𐑛 𐑦𐑯 𐑕𐑳𐑥𐑼 𐑑𐑲𐑥; 𐑞𐑨𐑑 𐑢𐑳𐑯 𐑚𐑤𐑱𐑛 𐑝 𐑜𐑮𐑭𐑕 𐑥𐑱𐑒𐑕 𐑩𐑯 𐑴𐑱𐑕𐑦𐑕, 𐑔𐑮𐑰 𐑚𐑤𐑱𐑛𐑟 𐑦𐑯 𐑩 𐑛𐑱'𐑟 𐑢𐑷𐑒 𐑩 𐑐𐑮𐑺𐑦; 𐑞𐑨𐑑 𐑞𐑱 𐑢𐑺 𐑒𐑢𐑦𐑒𐑕𐑨𐑯𐑛 𐑖𐑵𐑟, 𐑕𐑳𐑥𐑔𐑦𐑙 𐑤𐑲𐑒 ·𐑤𐑨𐑐𐑤𐑩𐑯𐑛𐑼 𐑕𐑯𐑴-𐑖𐑵𐑟; 𐑞𐑨𐑑 𐑞𐑱 𐑸 𐑕𐑴 𐑖𐑳𐑑 𐑳𐑐, 𐑚𐑧𐑤𐑑𐑩𐑛 𐑩𐑚𐑬𐑑, 𐑧𐑝𐑮𐑦 𐑢𐑱 𐑦𐑯𐑒𐑤𐑴𐑟𐑛, 𐑕𐑼𐑬𐑯𐑛𐑩𐑛, 𐑯 𐑥𐑱𐑛 𐑩𐑯 𐑳𐑑𐑼 𐑲𐑤𐑩𐑯𐑛 𐑝 𐑚𐑲 𐑞 𐑴𐑖𐑩𐑯, 𐑞𐑨𐑑 𐑑 𐑞𐑺 𐑝𐑧𐑮𐑦 𐑗𐑺𐑟 𐑯 𐑑𐑱𐑚𐑩𐑤𐑟 𐑕𐑥𐑷𐑤 𐑒𐑤𐑨𐑥𐑟 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑚𐑰 𐑓𐑬𐑯𐑛 𐑨𐑛𐑣𐑽𐑦𐑙, 𐑨𐑟 𐑑 𐑞 𐑚𐑨𐑒𐑕 𐑝 𐑕𐑰 𐑑𐑻𐑑𐑩𐑤𐑟. 𐑚𐑳𐑑 𐑞𐑰𐑟 𐑧𐑒𐑕𐑑𐑮𐑨𐑝𐑩𐑜𐑨𐑯𐑟𐑩𐑟 𐑴𐑯𐑤𐑦 𐑖𐑴 𐑞𐑨𐑑 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑦𐑟 𐑯𐑴 ·𐑦𐑤𐑦𐑯𐑶.

𐑤𐑫𐑒 𐑯𐑬 𐑨𐑑 𐑞 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑑𐑮𐑩𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑕𐑑𐑹𐑦 𐑝 𐑣𐑬 𐑞𐑦𐑕 𐑲𐑤𐑩𐑯𐑛 𐑢𐑪𐑟 𐑕𐑧𐑑𐑩𐑤𐑛 𐑚𐑲 𐑞 𐑮𐑧𐑛-𐑥𐑧𐑯. 𐑞𐑳𐑕 𐑜𐑴𐑟 𐑞 𐑤𐑧𐑡𐑩𐑯𐑛. 𐑦𐑯 𐑴𐑤𐑛𐑩𐑯 𐑑𐑲𐑥𐑟 𐑩𐑯 𐑰𐑜𐑩𐑤 𐑕𐑢𐑵𐑐𐑑 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑞 ·𐑯𐑿 ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛 𐑒𐑴𐑕𐑑, 𐑯 𐑒𐑨𐑮𐑦𐑛 𐑪𐑓 𐑩𐑯 𐑦𐑯𐑓𐑩𐑯𐑑 ·𐑦𐑯𐑛𐑾𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑑𐑨𐑤𐑩𐑯𐑟. 𐑢𐑦𐑞 𐑤𐑬𐑛 𐑤𐑨𐑥𐑧𐑯𐑑 𐑞 𐑐𐑺𐑩𐑯𐑑𐑕 𐑕𐑷 𐑞𐑺 𐑗𐑲𐑤𐑛 𐑚𐑹𐑯 𐑬𐑑 𐑝 𐑕𐑲𐑑 𐑴𐑝𐑼 𐑞 𐑢𐑲𐑛 𐑢𐑷𐑑𐑼𐑟. 𐑞𐑱 𐑮𐑦𐑟𐑪𐑤𐑝𐑛 𐑑 𐑓𐑪𐑤𐑴 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑛𐑦𐑮𐑧𐑒𐑖𐑩𐑯. 𐑕𐑧𐑑𐑦𐑙 𐑬𐑑 𐑦𐑯 𐑞𐑺 𐑒𐑩𐑯𐑵𐑟, 𐑭𐑓𐑑𐑼 𐑩 𐑐𐑧𐑮𐑦𐑤𐑩𐑕 𐑐𐑨𐑕𐑦𐑡 𐑞𐑱 𐑛𐑦𐑕𐑒𐑳𐑝𐑼𐑛 𐑞 𐑲𐑤𐑩𐑯𐑛, 𐑯 𐑞𐑺 𐑞𐑱 𐑓𐑬𐑯𐑛 𐑩𐑯 𐑧𐑥𐑐𐑑𐑦 𐑲𐑝𐑼𐑦 𐑒𐑭𐑕𐑒𐑩𐑑,—𐑞 𐑐𐑫𐑼 𐑤𐑦𐑑𐑩𐑤 ·𐑦𐑯𐑛𐑾𐑯'𐑟 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯.

𐑢𐑪𐑑 𐑢𐑳𐑯𐑛𐑼, 𐑞𐑧𐑯, 𐑞𐑨𐑑 𐑞𐑰𐑟 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼𐑟, 𐑚𐑹𐑯 𐑪𐑯 𐑩 𐑚𐑰𐑗, 𐑖𐑵𐑛 𐑑𐑱𐑒 𐑑 𐑞 𐑕𐑰 𐑓 𐑩 𐑤𐑲𐑝𐑤𐑦𐑣𐑫𐑛! 𐑞𐑱 𐑓𐑻𐑕𐑑 𐑒𐑷𐑑 𐑒𐑮𐑨𐑚𐑟 𐑯 𐑒𐑢𐑴𐑙𐑪𐑜𐑕 𐑦𐑯 𐑞 𐑕𐑨𐑯𐑛; 𐑜𐑮𐑴𐑯 𐑚𐑴𐑤𐑛𐑼, 𐑞𐑱 𐑢𐑱𐑛𐑩𐑛 𐑬𐑑 𐑢𐑦𐑞 𐑯𐑧𐑑𐑕 𐑓 𐑥𐑨𐑒𐑼𐑩𐑤; 𐑥𐑹 𐑦𐑒𐑕𐑐𐑽𐑾𐑯𐑕𐑑, 𐑞𐑱 𐑐𐑫𐑖𐑑 𐑪𐑓 𐑦𐑯 𐑚𐑴𐑑𐑕 𐑯 𐑒𐑨𐑐𐑗𐑼𐑛 𐑒𐑪𐑛; 𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑤𐑷𐑯𐑗𐑦𐑙 𐑩 𐑯𐑱𐑝𐑦 𐑝 𐑜𐑮𐑱𐑑 𐑖𐑦𐑐𐑕 𐑪𐑯 𐑞 𐑕𐑰, 𐑦𐑒𐑕𐑐𐑤𐑹𐑛 𐑞𐑦𐑕 𐑢𐑷𐑑𐑼𐑦 𐑢𐑻𐑤𐑛; 𐑐𐑫𐑑 𐑩𐑯 𐑦𐑯𐑕𐑧𐑕𐑩𐑯𐑑 𐑚𐑧𐑤𐑑 𐑝 𐑕𐑻𐑒𐑩𐑥𐑯𐑨𐑝𐑦𐑜𐑱𐑖𐑩𐑯𐑟 𐑮𐑬𐑯𐑛 𐑦𐑑; 𐑐𐑰𐑐𐑑 𐑦𐑯 𐑨𐑑 ·𐑚𐑺𐑦𐑙'𐑟 ·𐑕𐑑𐑮𐑱𐑑𐑕; 𐑯 𐑦𐑯 𐑷𐑤 𐑕𐑰𐑟𐑩𐑯𐑟 𐑯 𐑷𐑤 𐑴𐑖𐑩𐑯𐑟 𐑛𐑦𐑒𐑤𐑺𐑛 𐑧𐑝𐑼𐑤𐑭𐑕𐑑𐑦𐑙 𐑢𐑹 𐑢𐑦𐑞 𐑞 𐑥𐑲𐑑𐑾𐑕𐑑 𐑨𐑯𐑦𐑥𐑱𐑑𐑩𐑛 𐑥𐑨𐑕 𐑞𐑨𐑑 𐑣𐑨𐑟 𐑕𐑼𐑝𐑲𐑝𐑛 𐑞 𐑓𐑤𐑳𐑛; 𐑥𐑴𐑕𐑑 𐑥𐑪𐑯𐑕𐑑𐑮𐑩𐑕 𐑯 𐑥𐑴𐑕𐑑 𐑥𐑬𐑯𐑑𐑩𐑯𐑩𐑕! 𐑞𐑨𐑑 ·𐑙𐑦𐑥𐑩𐑤𐑱𐑩𐑯, 𐑕𐑷𐑤𐑑-𐑕𐑰 ·𐑥𐑨𐑕𐑑𐑩𐑛𐑪𐑯, 𐑒𐑤𐑴𐑞𐑛 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑐𐑹𐑑𐑧𐑯𐑑𐑩𐑕𐑯𐑩𐑕 𐑝 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕 𐑐𐑬𐑼, 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑝𐑧𐑮𐑦 𐑐𐑨𐑯𐑦𐑒𐑕 𐑸 𐑥𐑹 𐑑 𐑚𐑰 𐑛𐑮𐑧𐑛𐑩𐑛 𐑞𐑨𐑯 𐑣𐑦𐑟 𐑥𐑴𐑕𐑑 𐑓𐑽𐑤𐑩𐑕 𐑯 𐑥𐑩𐑤𐑦𐑖𐑩𐑕 𐑩𐑕𐑷𐑤𐑑𐑕!
