·𐑗𐑨𐑐𐑑𐑼 38. 𐑛𐑳𐑕𐑒.

𐑚𐑲 𐑞 ·𐑥𐑱𐑯𐑥𐑭𐑕𐑑; ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑤𐑰𐑯𐑦𐑙 𐑩𐑜𐑱𐑯𐑕𐑑 𐑦𐑑.

𐑥𐑲 𐑕𐑴𐑤 𐑦𐑟 𐑥𐑹 𐑞𐑨𐑯 𐑥𐑨𐑗𐑑; 𐑖𐑰'𐑟 𐑴𐑝𐑼𐑥𐑨𐑯𐑛; 𐑯 𐑚𐑲 𐑩 𐑥𐑨𐑛𐑥𐑩𐑯! 𐑦𐑯𐑕𐑳𐑓𐑼𐑩𐑚𐑩𐑤 𐑕𐑑𐑦𐑙, 𐑞𐑨𐑑 𐑕𐑨𐑯𐑦𐑑𐑦 𐑖𐑵𐑛 𐑜𐑮𐑬𐑯𐑛 𐑸𐑥𐑟 𐑪𐑯 𐑕𐑳𐑗 𐑩 𐑓𐑰𐑤𐑛! 𐑚𐑳𐑑 𐑣𐑰 𐑛𐑮𐑦𐑤𐑛 𐑛𐑰𐑐 𐑛𐑬𐑯, 𐑯 𐑚𐑤𐑭𐑕𐑑𐑩𐑛 𐑷𐑤 𐑥𐑲 𐑮𐑰𐑟𐑩𐑯 𐑬𐑑 𐑝 𐑥𐑰! 𐑲 𐑔𐑦𐑙𐑒 𐑲 𐑕𐑰 𐑣𐑦𐑟 𐑦𐑥𐑐𐑲𐑩𐑕 𐑧𐑯𐑛; 𐑚𐑳𐑑 𐑓𐑰𐑤 𐑞𐑨𐑑 𐑲 𐑥𐑳𐑕𐑑 𐑣𐑧𐑤𐑐 𐑣𐑦𐑥 𐑑 𐑦𐑑. 𐑢𐑦𐑤 𐑲, 𐑯𐑦𐑤 𐑲, 𐑞 𐑦𐑯𐑧𐑓𐑩𐑚𐑩𐑤 𐑔𐑦𐑙 𐑣𐑨𐑟 𐑑𐑲𐑛 𐑥𐑰 𐑑 𐑣𐑦𐑥; 𐑑𐑴𐑟 𐑥𐑰 𐑢𐑦𐑞 𐑩 𐑒𐑱𐑚𐑩𐑤 𐑲 𐑣𐑨𐑝 𐑯𐑴 𐑯𐑲𐑓 𐑑 𐑒𐑳𐑑. 𐑣𐑪𐑮𐑦𐑚𐑩𐑤 𐑴𐑤𐑛 𐑥𐑨𐑯! 𐑣𐑵'𐑟 𐑴𐑝𐑼 𐑣𐑦𐑥, 𐑣𐑰 𐑒𐑮𐑲𐑟;—𐑲, 𐑣𐑰 𐑢𐑫𐑛 𐑚𐑰 𐑩 𐑛𐑧𐑥𐑩𐑒𐑮𐑨𐑑 𐑑 𐑷𐑤 𐑩𐑚𐑳𐑝; 𐑤𐑫𐑒, 𐑣𐑬 𐑣𐑰 𐑤𐑹𐑛𐑟 𐑦𐑑 𐑴𐑝𐑼 𐑷𐑤 𐑚𐑦𐑤𐑴! 𐑴! 𐑲 𐑐𐑤𐑱𐑯𐑤𐑦 𐑕𐑰 𐑥𐑲 𐑥𐑦𐑟𐑼𐑩𐑚𐑩𐑤 𐑪𐑓𐑦𐑕,—𐑑 𐑴𐑚𐑱, 𐑮𐑦𐑚𐑧𐑤𐑦𐑙; 𐑯 𐑢𐑻𐑕 𐑘𐑧𐑑, 𐑑 𐑣𐑱𐑑 𐑢𐑦𐑞 𐑑𐑳𐑗 𐑝 𐑐𐑦𐑑𐑦! 𐑓 𐑦𐑯 𐑣𐑦𐑟 𐑲𐑟 𐑲 𐑮𐑰𐑛 𐑕𐑳𐑥 𐑤𐑫𐑼𐑦𐑛 𐑢𐑴 𐑢𐑫𐑛 𐑖𐑮𐑦𐑝𐑩𐑤 𐑥𐑰 𐑳𐑐, 𐑣𐑨𐑛 𐑲 𐑦𐑑. 𐑘𐑧𐑑 𐑦𐑟 𐑞𐑺 𐑣𐑴𐑐. 𐑑𐑲𐑥 𐑯 𐑑𐑲𐑛 𐑓𐑤𐑴 𐑢𐑲𐑛. 𐑞 𐑣𐑱𐑑𐑩𐑛 𐑢𐑱𐑤 𐑣𐑨𐑟 𐑞 𐑮𐑬𐑯𐑛 𐑢𐑷𐑑𐑼𐑦 𐑢𐑻𐑤𐑛 𐑑 𐑕𐑢𐑦𐑥 𐑦𐑯, 𐑨𐑟 𐑞 𐑕𐑥𐑷𐑤 𐑜𐑴𐑤𐑛-𐑓𐑦𐑖 𐑣𐑨𐑟 𐑦𐑑𐑕 𐑜𐑤𐑭𐑕𐑦 𐑜𐑤𐑴𐑚. 𐑣𐑦𐑟 𐑣𐑧𐑝𐑩𐑯-𐑦𐑯𐑕𐑳𐑤𐑑𐑦𐑙 𐑐𐑻𐑐𐑩𐑕, ·𐑜𐑪𐑛 𐑥𐑱 𐑢𐑧𐑡 𐑩𐑕𐑲𐑛. 𐑲 𐑢𐑫𐑛 𐑳𐑐 𐑣𐑸𐑑, 𐑢𐑻 𐑦𐑑 𐑯𐑪𐑑 𐑤𐑲𐑒 𐑤𐑧𐑛. 𐑚𐑳𐑑 𐑥𐑲 𐑣𐑴𐑤 𐑒𐑤𐑪𐑒'𐑕 𐑮𐑳𐑯 𐑛𐑬𐑯; 𐑥𐑲 𐑣𐑸𐑑 𐑞 𐑷𐑤-𐑒𐑩𐑯𐑑𐑮𐑴𐑤𐑦𐑙 𐑢𐑱𐑑, 𐑲 𐑣𐑨𐑝 𐑯𐑴 𐑒𐑰 𐑑 𐑤𐑦𐑓𐑑 𐑩𐑜𐑱𐑯.

[𐑩 𐑚𐑻𐑕𐑑 𐑝 𐑮𐑧𐑝𐑩𐑤𐑮𐑦 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤.]

𐑴, ·𐑜𐑪𐑛! 𐑑 𐑕𐑱𐑤 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑩 𐑣𐑰𐑞𐑩𐑯 𐑒𐑮𐑵 𐑞𐑨𐑑 𐑣𐑨𐑝 𐑕𐑥𐑷𐑤 𐑑𐑳𐑗 𐑝 𐑣𐑿𐑥𐑩𐑯 𐑥𐑳𐑞𐑼𐑟 𐑦𐑯 𐑞𐑧𐑥! 𐑢𐑧𐑤𐑐𐑑 𐑕𐑳𐑥𐑢𐑺 𐑚𐑲 𐑞 𐑖𐑸𐑒𐑦𐑖 𐑕𐑰. 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑦𐑟 𐑞𐑺 𐑛𐑧𐑥𐑦𐑜𐑹𐑜𐑩𐑯. 𐑣𐑸𐑒! 𐑞 𐑦𐑯𐑓𐑻𐑯𐑩𐑤 𐑹𐑡𐑦𐑟! 𐑞𐑨𐑑 𐑮𐑧𐑝𐑩𐑤𐑮𐑦 𐑦𐑟 𐑓𐑹𐑢𐑼𐑛! 𐑥𐑸𐑒 𐑞 𐑳𐑯𐑓𐑷𐑤𐑑𐑼𐑦𐑙 𐑕𐑲𐑤𐑩𐑯𐑕 𐑭𐑓𐑑! 𐑥𐑦𐑔𐑦𐑙𐑒𐑕 𐑦𐑑 𐑐𐑦𐑒𐑗𐑼𐑟 𐑤𐑲𐑓. 𐑓𐑹𐑥𐑴𐑕𐑑 𐑔𐑮𐑵 𐑞 𐑕𐑐𐑸𐑒𐑩𐑤𐑦𐑙 𐑕𐑰 𐑖𐑵𐑑𐑕 𐑪𐑯 𐑞 𐑜𐑱, 𐑦𐑥𐑚𐑨𐑑𐑩𐑤𐑛, 𐑚𐑨𐑯𐑑𐑼𐑦𐑙 𐑚𐑬, 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑑 𐑛𐑮𐑨𐑜 𐑛𐑸𐑒 ·𐑱𐑣𐑨𐑚 𐑭𐑓𐑑𐑼 𐑦𐑑, 𐑢𐑺 𐑣𐑰 𐑚𐑮𐑵𐑛𐑟 𐑢𐑦𐑞𐑦𐑯 𐑣𐑦𐑟 𐑕𐑑𐑻𐑯𐑢𐑼𐑛 𐑒𐑨𐑚𐑦𐑯, 𐑚𐑦𐑤𐑛𐑩𐑛 𐑴𐑝𐑼 𐑞 𐑛𐑧𐑛 𐑢𐑷𐑑𐑼 𐑝 𐑞 𐑢𐑱𐑒, 𐑯 𐑓𐑻𐑞𐑼 𐑪𐑯, 𐑣𐑳𐑯𐑑𐑩𐑛 𐑚𐑲 𐑦𐑑𐑕 𐑢𐑫𐑤𐑓𐑦𐑖 𐑜𐑻𐑜𐑩𐑤𐑦𐑙𐑟. 𐑞 𐑤𐑪𐑙 𐑣𐑬𐑤 𐑔𐑮𐑦𐑤𐑟 𐑥𐑰 𐑔𐑮𐑵! 𐑐𐑰𐑕! 𐑘𐑰 𐑮𐑧𐑝𐑩𐑤𐑼𐑟, 𐑯 𐑕𐑧𐑑 𐑞 𐑢𐑪𐑗! 𐑴, 𐑤𐑲𐑓! '𐑑𐑦𐑟 𐑦𐑯 𐑩𐑯 𐑬𐑼 𐑤𐑲𐑒 𐑞𐑦𐑕, 𐑢𐑦𐑞 𐑕𐑴𐑤 𐑚𐑰𐑑 𐑛𐑬𐑯 𐑯 𐑣𐑧𐑤𐑛 𐑑 𐑯𐑪𐑤𐑦𐑡,—𐑨𐑟 𐑢𐑲𐑤𐑛, 𐑳𐑯𐑑𐑿𐑑𐑼𐑛 𐑔𐑦𐑙𐑟 𐑸 𐑓𐑹𐑕𐑑 𐑑 𐑓𐑰𐑛—𐑴, 𐑤𐑲𐑓! '𐑑𐑦𐑟 𐑯𐑬 𐑞𐑨𐑑 𐑲 𐑛𐑵 𐑓𐑰𐑤 𐑞 𐑤𐑱𐑑𐑩𐑯𐑑 𐑣𐑪𐑮𐑼 𐑦𐑯 𐑞𐑰! 𐑚𐑳𐑑 '𐑑𐑦𐑟 𐑯𐑪𐑑 𐑥𐑰! 𐑞𐑨𐑑 𐑣𐑪𐑮𐑼'𐑟 𐑬𐑑 𐑝 𐑥𐑰! 𐑯 𐑢𐑦𐑞 𐑞 𐑕𐑪𐑓𐑑 𐑓𐑰𐑤𐑦𐑙 𐑝 𐑞 𐑣𐑿𐑥𐑩𐑯 𐑦𐑯 𐑥𐑰, 𐑘𐑧𐑑 𐑢𐑦𐑤 𐑲 𐑑𐑮𐑲 𐑑 𐑓𐑲𐑑 𐑘𐑰, 𐑘𐑰 𐑜𐑮𐑦𐑥, 𐑓𐑨𐑯𐑑𐑩𐑥 𐑓𐑿𐑗𐑼𐑟! 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑥𐑰, 𐑣𐑴𐑤𐑛 𐑥𐑰, 𐑚𐑲𐑯𐑛 𐑥𐑰, 𐑴 𐑘𐑰 𐑚𐑤𐑧𐑕𐑑 𐑦𐑯𐑓𐑤𐑵𐑩𐑯𐑕𐑩𐑟!
