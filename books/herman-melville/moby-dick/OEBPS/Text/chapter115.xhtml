·𐑗𐑨𐑐𐑑𐑼 115. 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 ·𐑥𐑰𐑑𐑕 𐑞 ·𐑚𐑨𐑗𐑩𐑤𐑼.

𐑯 𐑡𐑪𐑤𐑦 𐑦𐑯𐑳𐑓 𐑢𐑻 𐑞 𐑕𐑲𐑑𐑕 𐑯 𐑞 𐑕𐑬𐑯𐑛𐑟 𐑞𐑨𐑑 𐑒𐑱𐑥 𐑚𐑺𐑦𐑙 𐑛𐑬𐑯 𐑚𐑦𐑓𐑹 𐑞 𐑢𐑦𐑯𐑛, 𐑕𐑳𐑥 𐑓𐑿 𐑢𐑰𐑒𐑕 𐑭𐑓𐑑𐑼 ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑸𐑐𐑵𐑯 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑢𐑧𐑤𐑛𐑩𐑛.

𐑦𐑑 𐑢𐑪𐑟 𐑩 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑖𐑦𐑐, 𐑞 ·𐑚𐑨𐑗𐑩𐑤𐑼, 𐑢𐑦𐑗 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑢𐑧𐑡𐑛 𐑦𐑯 𐑣𐑻 𐑤𐑭𐑕𐑑 𐑒𐑭𐑕𐑒 𐑝 𐑶𐑤, 𐑯 𐑚𐑴𐑤𐑑𐑩𐑛 𐑛𐑬𐑯 𐑣𐑻 𐑚𐑻𐑕𐑑𐑦𐑙 𐑣𐑨𐑗𐑩𐑟; 𐑯 𐑯𐑬, 𐑦𐑯 𐑜𐑤𐑨𐑛 𐑣𐑪𐑤𐑦𐑛𐑱 𐑩𐑐𐑨𐑮𐑩𐑤, 𐑢𐑪𐑟 𐑡𐑶𐑩𐑕𐑤𐑦, 𐑞𐑴 𐑕𐑳𐑥𐑢𐑪𐑑 𐑝𐑱𐑯-𐑜𐑤𐑹𐑾𐑕𐑤𐑦, 𐑕𐑱𐑤𐑦𐑙 𐑮𐑬𐑯𐑛 𐑩𐑥𐑳𐑙 𐑞 𐑢𐑲𐑛𐑤𐑦-𐑕𐑧𐑐𐑼𐑱𐑑𐑩𐑛 𐑖𐑦𐑐𐑕 𐑪𐑯 𐑞 𐑜𐑮𐑬𐑯𐑛, 𐑐𐑮𐑰𐑝𐑾𐑕 𐑑 𐑐𐑶𐑯𐑑𐑦𐑙 𐑣𐑻 𐑐𐑮𐑬 𐑓 𐑣𐑴𐑥.

𐑞 𐑔𐑮𐑰 𐑥𐑧𐑯 𐑨𐑑 𐑣𐑻 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛 𐑢𐑹 𐑤𐑪𐑙 𐑕𐑑𐑮𐑰𐑥𐑼𐑟 𐑝 𐑯𐑼𐑴 𐑮𐑧𐑛 𐑚𐑳𐑯𐑑𐑦𐑙 𐑨𐑑 𐑞𐑺 𐑣𐑨𐑑𐑕; 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑑𐑻𐑯, 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑 𐑢𐑪𐑟 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛, 𐑚𐑪𐑑𐑩𐑥 𐑛𐑬𐑯; 𐑯 𐑣𐑨𐑙𐑦𐑙 𐑒𐑨𐑐𐑑𐑦𐑝 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑕𐑐𐑮𐑦𐑑 𐑢𐑪𐑟 𐑕𐑰𐑯 𐑞 𐑤𐑪𐑙 𐑤𐑴𐑼 𐑡𐑷 𐑝 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑱𐑤 𐑞𐑱 𐑣𐑨𐑛 𐑕𐑤𐑱𐑯. 𐑕𐑦𐑜𐑯𐑩𐑤𐑟, 𐑧𐑯𐑕𐑩𐑯𐑟, 𐑯 𐑡𐑨𐑒𐑕 𐑝 𐑷𐑤 𐑒𐑳𐑤𐑼𐑟 𐑢𐑻 𐑓𐑤𐑲𐑦𐑙 𐑓𐑮𐑪𐑥 𐑣𐑻 𐑮𐑦𐑜𐑦𐑙, 𐑪𐑯 𐑧𐑝𐑮𐑦 𐑕𐑲𐑛. 𐑕𐑲𐑛𐑢𐑱𐑟 𐑤𐑨𐑖𐑑 𐑦𐑯 𐑰𐑗 𐑝 𐑣𐑻 𐑔𐑮𐑰 𐑚𐑭𐑕𐑒𐑩𐑑𐑩𐑛 𐑑𐑪𐑐𐑕 𐑢𐑻 𐑑𐑵 𐑚𐑨𐑮𐑩𐑤𐑟 𐑝 𐑕𐑐𐑻𐑥; 𐑩𐑚𐑳𐑝 𐑢𐑦𐑗, 𐑦𐑯 𐑣𐑻 𐑑𐑪𐑐-𐑥𐑭𐑕𐑑 𐑒𐑮𐑪𐑕-𐑑𐑮𐑰𐑟, 𐑿 𐑕𐑷 𐑕𐑤𐑧𐑯𐑛𐑼 𐑚𐑮𐑱𐑒𐑼𐑟 𐑝 𐑞 𐑕𐑱𐑥 𐑐𐑮𐑧𐑖𐑩𐑕 𐑓𐑤𐑵𐑦𐑛; 𐑯 𐑯𐑱𐑤𐑛 𐑑 𐑣𐑻 𐑥𐑱𐑯 𐑑𐑮𐑳𐑒 𐑢𐑪𐑟 𐑩 𐑚𐑮𐑱𐑟𐑩𐑯 𐑤𐑨𐑥𐑐.

𐑨𐑟 𐑢𐑪𐑟 𐑭𐑓𐑑𐑼𐑢𐑼𐑛𐑟 𐑤𐑻𐑯𐑛, 𐑞 ·𐑚𐑨𐑗𐑩𐑤𐑼 𐑣𐑨𐑛 𐑥𐑧𐑑 𐑢𐑦𐑞 𐑞 𐑥𐑴𐑕𐑑 𐑕𐑼𐑐𐑮𐑲𐑟𐑦𐑙 𐑕𐑩𐑒𐑕𐑧𐑕; 𐑷𐑤 𐑞 𐑥𐑹 𐑢𐑳𐑯𐑛𐑼𐑓𐑩𐑤, 𐑓 𐑞𐑨𐑑 𐑢𐑲𐑤 𐑒𐑮𐑵𐑟𐑦𐑙 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑕𐑰𐑟 𐑯𐑿𐑥𐑼𐑩𐑕 𐑳𐑞𐑼 𐑝𐑧𐑕𐑩𐑤𐑟 𐑣𐑨𐑛 𐑜𐑪𐑯 𐑦𐑯𐑑𐑲𐑼 𐑥𐑳𐑯𐑔𐑕 𐑢𐑦𐑞𐑬𐑑 𐑕𐑦𐑒𐑘𐑫𐑼𐑦𐑙 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑓𐑦𐑖. 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑣𐑨𐑛 𐑚𐑨𐑮𐑩𐑤𐑟 𐑝 𐑚𐑰𐑓 𐑯 𐑚𐑮𐑧𐑛 𐑚𐑰𐑯 𐑜𐑦𐑝𐑩𐑯 𐑩𐑢𐑱 𐑑 𐑥𐑱𐑒 𐑮𐑵𐑥 𐑓 𐑞 𐑓𐑸 𐑥𐑹 𐑝𐑨𐑤𐑿𐑩𐑚𐑩𐑤 𐑕𐑐𐑻𐑥, 𐑚𐑳𐑑 𐑩𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑕𐑳𐑐𐑤𐑦𐑥𐑩𐑯𐑑𐑩𐑤 𐑒𐑭𐑕𐑒𐑕 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑚𐑸𐑑𐑼𐑛 𐑓, 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐𐑕 𐑖𐑰 𐑣𐑨𐑛 𐑥𐑧𐑑; 𐑯 𐑞𐑰𐑟 𐑢𐑻 𐑕𐑑𐑴𐑛 𐑩𐑤𐑪𐑙 𐑞 𐑛𐑧𐑒, 𐑯 𐑦𐑯 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑯 𐑪𐑓𐑦𐑕𐑼𐑟' 𐑕𐑑𐑱𐑑-𐑮𐑵𐑥𐑟. 𐑰𐑝𐑩𐑯 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑑𐑱𐑚𐑩𐑤 𐑦𐑑𐑕𐑧𐑤𐑓 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑯𐑪𐑒𐑑 𐑦𐑯𐑑𐑵 𐑒𐑦𐑯𐑛𐑩𐑤𐑦𐑙-𐑢𐑫𐑛; 𐑯 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑥𐑧𐑕 𐑛𐑲𐑯𐑛 𐑪𐑓 𐑞 𐑚𐑮𐑷𐑛 𐑣𐑧𐑛 𐑝 𐑩𐑯 𐑶𐑤-𐑚𐑳𐑑, 𐑤𐑨𐑖𐑑 𐑛𐑬𐑯 𐑑 𐑞 𐑓𐑤𐑹 𐑓 𐑩 𐑕𐑧𐑯𐑑𐑼𐑐𐑰𐑕. 𐑦𐑯 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤, 𐑞 𐑕𐑱𐑤𐑼𐑟 𐑣𐑨𐑛 𐑨𐑒𐑗𐑵𐑩𐑤𐑦 𐑒𐑷𐑒𐑑 𐑯 𐑐𐑦𐑗𐑑 𐑞𐑺 𐑗𐑧𐑕𐑑𐑕, 𐑯 𐑓𐑦𐑤𐑛 𐑞𐑧𐑥; 𐑦𐑑 𐑢𐑪𐑟 𐑣𐑿𐑥𐑼𐑩𐑕𐑤𐑦 𐑨𐑛𐑩𐑛, 𐑞𐑨𐑑 𐑞 𐑒𐑫𐑒 𐑣𐑨𐑛 𐑒𐑤𐑨𐑐𐑑 𐑩 𐑣𐑧𐑛 𐑪𐑯 𐑣𐑦𐑟 𐑤𐑸𐑡𐑩𐑕𐑑 𐑚𐑶𐑤𐑼, 𐑯 𐑓𐑦𐑤𐑛 𐑦𐑑; 𐑞𐑨𐑑 𐑞 𐑕𐑑𐑿𐑼𐑛 𐑣𐑨𐑛 𐑐𐑤𐑳𐑜𐑛 𐑣𐑦𐑟 𐑕𐑐𐑺 𐑒𐑪𐑓𐑦-𐑐𐑪𐑑 𐑯 𐑓𐑦𐑤𐑛 𐑦𐑑; 𐑞𐑨𐑑 𐑞 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑣𐑨𐑛 𐑣𐑧𐑛𐑩𐑛 𐑞 𐑕𐑪𐑒𐑩𐑑𐑕 𐑝 𐑞𐑺 𐑲𐑼𐑯𐑟 𐑯 𐑓𐑦𐑤𐑛 𐑞𐑧𐑥; 𐑞𐑨𐑑 𐑦𐑯𐑛𐑰𐑛 𐑧𐑝𐑮𐑦𐑔𐑦𐑙 𐑢𐑪𐑟 𐑓𐑦𐑤𐑛 𐑢𐑦𐑞 𐑕𐑐𐑻𐑥, 𐑦𐑒𐑕𐑧𐑐𐑑 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑐𐑨𐑯𐑑𐑩𐑤𐑵𐑯𐑟 𐑐𐑪𐑒𐑩𐑑𐑕, 𐑯 𐑞𐑴𐑟 𐑣𐑰 𐑮𐑦𐑟𐑻𐑝𐑛 𐑑 𐑔𐑮𐑳𐑕𐑑 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛𐑟 𐑦𐑯𐑑𐑵, 𐑦𐑯 𐑕𐑧𐑤𐑓-𐑒𐑩𐑥𐑐𐑤𐑱𐑕𐑩𐑯𐑑 𐑑𐑧𐑕𐑑𐑦𐑥𐑩𐑯𐑦 𐑝 𐑣𐑦𐑟 𐑦𐑯𐑑𐑲𐑼 𐑕𐑨𐑑𐑦𐑕𐑓𐑨𐑒𐑖𐑩𐑯.

𐑨𐑟 𐑞𐑦𐑕 𐑜𐑤𐑨𐑛 𐑖𐑦𐑐 𐑝 𐑜𐑫𐑛 𐑤𐑳𐑒 𐑚𐑹 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑞 𐑥𐑵𐑛𐑦 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑞 𐑚𐑸𐑚𐑼𐑾𐑯 𐑕𐑬𐑯𐑛 𐑝 𐑦𐑯𐑹𐑥𐑩𐑕 𐑛𐑮𐑳𐑥𐑟 𐑒𐑱𐑥 𐑓𐑮𐑪𐑥 𐑣𐑻 𐑓𐑴𐑒𐑕𐑩𐑤; 𐑯 𐑛𐑮𐑷𐑦𐑙 𐑕𐑑𐑦𐑤 𐑯𐑽𐑼, 𐑩 𐑒𐑮𐑬𐑛 𐑝 𐑣𐑻 𐑥𐑧𐑯 𐑢𐑻 𐑕𐑰𐑯 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑮𐑬𐑯𐑛 𐑣𐑻 𐑣𐑿𐑡 𐑑𐑮𐑲-𐑐𐑪𐑑𐑕, 𐑢𐑦𐑗, 𐑒𐑳𐑝𐑼𐑛 𐑢𐑦𐑞 𐑞 𐑐𐑸𐑗𐑥𐑩𐑯𐑑-𐑤𐑲𐑒 𐑐𐑴𐑒 𐑹 𐑕𐑑𐑳𐑥𐑩𐑒 𐑕𐑒𐑦𐑯 𐑝 𐑞 𐑚𐑤𐑨𐑒 𐑓𐑦𐑖, 𐑜𐑱𐑝 𐑓𐑹𐑔 𐑩 𐑤𐑬𐑛 𐑮𐑹 𐑑 𐑧𐑝𐑮𐑦 𐑕𐑑𐑮𐑴𐑒 𐑝 𐑞 𐑒𐑤𐑧𐑯𐑗𐑑 𐑣𐑨𐑯𐑛𐑟 𐑝 𐑞 𐑒𐑮𐑵. 𐑪𐑯 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒, 𐑞 𐑥𐑱𐑑𐑕 𐑯 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑢𐑻 𐑛𐑭𐑯𐑕𐑦𐑙 𐑢𐑦𐑞 𐑞 𐑪𐑤𐑦𐑝-𐑣𐑿𐑛 𐑜𐑻𐑤𐑟 𐑣𐑵 𐑣𐑨𐑛 𐑦𐑤𐑴𐑐𐑑 𐑢𐑦𐑞 𐑞𐑧𐑥 𐑓𐑮𐑪𐑥 𐑞 ·𐑐𐑪𐑤𐑦𐑯𐑰𐑟𐑾𐑯 ·𐑲𐑤𐑟; 𐑢𐑲𐑤 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑦𐑯 𐑩𐑯 𐑹𐑯𐑩𐑥𐑩𐑯𐑑𐑩𐑛 𐑚𐑴𐑑, 𐑓𐑻𐑥𐑤𐑦 𐑕𐑦𐑒𐑘𐑫𐑼𐑛 𐑩𐑤𐑪𐑓𐑑 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑓𐑹𐑥𐑭𐑕𐑑 𐑯 𐑥𐑱𐑯𐑥𐑭𐑕𐑑, 𐑔𐑮𐑰 ·𐑤𐑪𐑙 ·𐑲𐑤𐑩𐑯𐑛 𐑯𐑰𐑜𐑮𐑴𐑟, 𐑢𐑦𐑞 𐑜𐑤𐑦𐑑𐑼𐑦𐑙 𐑓𐑦𐑛𐑩𐑤-𐑚𐑴𐑟 𐑝 𐑢𐑱𐑤 𐑲𐑝𐑼𐑦, 𐑢𐑻 𐑐𐑮𐑦𐑟𐑲𐑛𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑣𐑦𐑤𐑺𐑾𐑕 𐑡𐑦𐑜. 𐑥𐑰𐑯𐑢𐑲𐑤, 𐑳𐑞𐑼𐑟 𐑝 𐑞 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑢𐑻 𐑑𐑿𐑥𐑳𐑤𐑗𐑵𐑩𐑕𐑤𐑦 𐑚𐑦𐑟𐑦 𐑨𐑑 𐑞 𐑥𐑱𐑕𐑩𐑯𐑼𐑦 𐑝 𐑞 𐑑𐑮𐑲-𐑢𐑻𐑒𐑕, 𐑓𐑮𐑪𐑥 𐑢𐑦𐑗 𐑞 𐑣𐑿𐑡 𐑐𐑪𐑑𐑕 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑮𐑦𐑥𐑵𐑝𐑛. 𐑿 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑷𐑤𐑥𐑴𐑕𐑑 𐑔𐑷𐑑 𐑞𐑱 𐑢𐑻 𐑐𐑫𐑤𐑦𐑙 𐑛𐑬𐑯 𐑞 𐑒𐑻𐑕𐑑 ·𐑚𐑨𐑕𐑑𐑰𐑤, 𐑕𐑳𐑗 𐑢𐑲𐑤𐑛 𐑒𐑮𐑲𐑟 𐑞𐑱 𐑮𐑱𐑟𐑛, 𐑨𐑟 𐑞 𐑯𐑬 𐑿𐑕𐑤𐑩𐑕 𐑚𐑮𐑦𐑒 𐑯 𐑥𐑹𐑑𐑼 𐑢𐑻 𐑚𐑰𐑦𐑙 𐑣𐑻𐑤𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰.

𐑤𐑹𐑛 𐑯 𐑥𐑭𐑕𐑑𐑼 𐑴𐑝𐑼 𐑷𐑤 𐑞𐑦𐑕 𐑕𐑰𐑯, 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑕𐑑𐑫𐑛 𐑦𐑮𐑧𐑒𐑑 𐑪𐑯 𐑞 𐑖𐑦𐑐'𐑕 𐑧𐑤𐑦𐑝𐑱𐑑𐑩𐑛 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒, 𐑕𐑴 𐑞𐑨𐑑 𐑞 𐑣𐑴𐑤 𐑮𐑦𐑡𐑶𐑕𐑦𐑙 𐑛𐑮𐑭𐑥𐑩 𐑢𐑪𐑟 𐑓𐑫𐑤 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥, 𐑯 𐑕𐑰𐑥𐑛 𐑥𐑽𐑤𐑦 𐑒𐑩𐑯𐑑𐑮𐑲𐑝𐑛 𐑓 𐑣𐑦𐑟 𐑴𐑯 𐑦𐑯𐑛𐑦𐑝𐑦𐑡𐑵𐑩𐑤 𐑛𐑲𐑝𐑻𐑖𐑩𐑯.

𐑯 ·𐑱𐑣𐑨𐑚, 𐑣𐑰 𐑑𐑵 𐑢𐑪𐑟 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑪𐑯 𐑣𐑦𐑟 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒, 𐑖𐑨𐑜𐑦 𐑯 𐑚𐑤𐑨𐑒, 𐑢𐑦𐑞 𐑩 𐑕𐑑𐑳𐑚𐑼𐑯 𐑜𐑤𐑵𐑥; 𐑯 𐑨𐑟 𐑞 𐑑𐑵 𐑖𐑦𐑐𐑕 𐑒𐑮𐑪𐑕𐑑 𐑰𐑗 𐑳𐑞𐑼'𐑟 𐑢𐑱𐑒𐑕—𐑢𐑳𐑯 𐑷𐑤 𐑡𐑵𐑚𐑦𐑤𐑱𐑖𐑩𐑯𐑟 𐑓 𐑔𐑦𐑙𐑟 𐑐𐑭𐑕𐑑, 𐑞 𐑳𐑞𐑼 𐑷𐑤 𐑓𐑹𐑚𐑴𐑛𐑦𐑙𐑟 𐑨𐑟 𐑑 𐑔𐑦𐑙𐑟 𐑑 𐑒𐑳𐑥—𐑞𐑺 𐑑𐑵 𐑒𐑨𐑐𐑑𐑦𐑯𐑟 𐑦𐑯 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑦𐑥𐑐𐑻𐑕𐑩𐑯𐑱𐑑𐑩𐑛 𐑞 𐑣𐑴𐑤 𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑒𐑪𐑯𐑑𐑮𐑭𐑕𐑑 𐑝 𐑞 𐑕𐑰𐑯.

“𐑒𐑳𐑥 𐑩𐑚𐑹𐑛, 𐑒𐑳𐑥 𐑩𐑚𐑹𐑛!” 𐑒𐑮𐑲𐑛 𐑞 𐑜𐑱 ·𐑚𐑨𐑗𐑩𐑤𐑼'𐑟 𐑒𐑩𐑥𐑭𐑯𐑛𐑼, 𐑤𐑦𐑓𐑑𐑦𐑙 𐑩 𐑜𐑤𐑭𐑕 𐑯 𐑩 𐑚𐑪𐑑𐑩𐑤 𐑦𐑯 𐑞 𐑺.

“𐑣𐑨𐑕𐑑 𐑕𐑰𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?” 𐑜𐑮𐑦𐑑𐑩𐑛 ·𐑱𐑣𐑨𐑚 𐑦𐑯 𐑮𐑦𐑐𐑤𐑲.

“𐑯𐑴; 𐑴𐑯𐑤𐑦 𐑣𐑻𐑛 𐑝 𐑣𐑦𐑥; 𐑚𐑳𐑑 𐑛𐑴𐑯𐑑 𐑚𐑦𐑤𐑰𐑝 𐑦𐑯 𐑣𐑦𐑥 𐑨𐑑 𐑷𐑤,” 𐑕𐑧𐑛 𐑞 𐑳𐑞𐑼 𐑜𐑫𐑛-𐑣𐑿𐑥𐑼𐑩𐑛𐑤𐑦. “𐑒𐑳𐑥 𐑩𐑚𐑹𐑛!”

“𐑞𐑬 𐑸𐑑 𐑑𐑵 𐑛𐑨𐑥𐑛 𐑡𐑪𐑤𐑦. 𐑕𐑱𐑤 𐑪𐑯. 𐑣𐑨𐑕𐑑 𐑤𐑪𐑕𐑑 𐑧𐑯𐑦 𐑥𐑧𐑯?”

“𐑯𐑪𐑑 𐑦𐑯𐑳𐑓 𐑑 𐑕𐑐𐑰𐑒 𐑝—𐑑𐑵 𐑲𐑕𐑤𐑭𐑯𐑛𐑼𐑟, 𐑞𐑨𐑑'𐑕 𐑷𐑤;—𐑚𐑳𐑑 𐑒𐑳𐑥 𐑩𐑚𐑹𐑛, 𐑴𐑤𐑛 𐑣𐑸𐑑𐑦, 𐑒𐑳𐑥 𐑩𐑤𐑪𐑙. 𐑲'𐑤 𐑕𐑵𐑯 𐑑𐑱𐑒 𐑞𐑨𐑑 𐑚𐑤𐑨𐑒 𐑓𐑮𐑪𐑥 𐑘𐑹 𐑚𐑮𐑬. 𐑒𐑳𐑥 𐑩𐑤𐑪𐑙, 𐑢𐑦𐑤 𐑘𐑰 (𐑥𐑧𐑮𐑦'𐑟 𐑞 𐑐𐑤𐑱); 𐑩 𐑓𐑫𐑤 𐑖𐑦𐑐 𐑯 𐑣𐑴𐑥𐑢𐑼𐑛-𐑚𐑬𐑯𐑛.”

“𐑣𐑬 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑓𐑩𐑥𐑦𐑤𐑽 𐑦𐑟 𐑩 𐑓𐑵𐑤!” 𐑥𐑳𐑑𐑼𐑛 ·𐑱𐑣𐑨𐑚; 𐑞𐑧𐑯 𐑩𐑤𐑬𐑛, “𐑞𐑬 𐑸𐑑 𐑩 𐑓𐑫𐑤 𐑖𐑦𐑐 𐑯 𐑣𐑴𐑥𐑢𐑼𐑛 𐑚𐑬𐑯𐑛, 𐑞𐑬 𐑕𐑱'𐑕𐑑; 𐑢𐑧𐑤, 𐑞𐑧𐑯, 𐑒𐑷𐑤 𐑥𐑰 𐑩𐑯 𐑧𐑥𐑐𐑑𐑦 𐑖𐑦𐑐, 𐑯 𐑬𐑑𐑢𐑼𐑛-𐑚𐑬𐑯𐑛. 𐑕𐑴 𐑜𐑴 𐑞𐑲 𐑢𐑱𐑟, 𐑯 𐑲 𐑢𐑦𐑤 𐑥𐑲𐑯. 𐑓𐑹𐑢𐑼𐑛 𐑞𐑺! 𐑕𐑧𐑑 𐑷𐑤 𐑕𐑱𐑤, 𐑯 𐑒𐑰𐑐 𐑣𐑻 𐑑 𐑞 𐑢𐑦𐑯𐑛!”

𐑯 𐑞𐑳𐑕, 𐑢𐑲𐑤 𐑞 𐑢𐑳𐑯 𐑖𐑦𐑐 𐑢𐑧𐑯𐑑 𐑗𐑽𐑦𐑤𐑦 𐑚𐑦𐑓𐑹 𐑞 𐑚𐑮𐑰𐑟, 𐑞 𐑳𐑞𐑼 𐑕𐑑𐑳𐑚𐑼𐑯𐑤𐑦 𐑓𐑷𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑦𐑑; 𐑯 𐑕𐑴 𐑞 𐑑𐑵 𐑝𐑧𐑕𐑩𐑤𐑟 𐑐𐑸𐑑𐑩𐑛; 𐑞 𐑒𐑮𐑵 𐑝 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑤𐑫𐑒𐑦𐑙 𐑢𐑦𐑞 𐑜𐑮𐑱𐑝, 𐑤𐑦𐑙𐑜𐑼𐑦𐑙 𐑜𐑤𐑭𐑯𐑕𐑩𐑟 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑮𐑦𐑕𐑰𐑛𐑦𐑙 ·𐑚𐑨𐑗𐑩𐑤𐑼; 𐑚𐑳𐑑 𐑞 ·𐑚𐑨𐑗𐑩𐑤𐑼'𐑟 𐑥𐑧𐑯 𐑯𐑧𐑝𐑼 𐑣𐑰𐑛𐑦𐑙 𐑞𐑺 𐑜𐑱𐑟 𐑓 𐑞 𐑤𐑲𐑝𐑤𐑦 𐑮𐑧𐑝𐑩𐑤𐑮𐑦 𐑞𐑱 𐑢𐑻 𐑦𐑯. 𐑯 𐑨𐑟 ·𐑱𐑣𐑨𐑚, 𐑤𐑰𐑯𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑑𐑨𐑓𐑮𐑱𐑤, 𐑲𐑛 𐑞 𐑣𐑴𐑥𐑢𐑼𐑛-𐑚𐑬𐑯𐑛 𐑒𐑮𐑭𐑓𐑑, 𐑣𐑰 𐑑𐑫𐑒 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑐𐑪𐑒𐑩𐑑 𐑩 𐑕𐑥𐑷𐑤 𐑝𐑲𐑩𐑤 𐑝 𐑕𐑨𐑯𐑛, 𐑯 𐑞𐑧𐑯 𐑤𐑫𐑒𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐 𐑑 𐑞 𐑝𐑲𐑩𐑤, 𐑕𐑰𐑥𐑛 𐑞𐑺𐑚𐑲 𐑚𐑮𐑦𐑙𐑦𐑙 𐑑𐑵 𐑮𐑦𐑥𐑴𐑑 𐑩𐑕𐑴𐑖𐑦𐑱𐑖𐑩𐑯𐑟 𐑑𐑩𐑜𐑧𐑞𐑼, 𐑓 𐑞𐑨𐑑 𐑝𐑲𐑩𐑤 𐑢𐑪𐑟 𐑓𐑦𐑤𐑛 𐑢𐑦𐑞 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑕𐑬𐑯𐑛𐑦𐑙𐑟.
