·𐑗𐑨𐑐𐑑𐑼 22. 𐑥𐑧𐑮𐑦 ·𐑒𐑮𐑦𐑕𐑥𐑩𐑕.

𐑨𐑑 𐑤𐑧𐑙𐑔, 𐑑𐑩𐑢𐑹𐑛𐑟 𐑯𐑵𐑯, 𐑩𐑐𐑪𐑯 𐑞 𐑓𐑲𐑯𐑩𐑤 𐑛𐑦𐑕𐑥𐑦𐑕𐑩𐑤 𐑝 𐑞 𐑖𐑦𐑐'𐑕 𐑮𐑦𐑜𐑼𐑟, 𐑯 𐑭𐑓𐑑𐑼 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑣𐑷𐑤𐑛 𐑬𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑹𐑓, 𐑯 𐑭𐑓𐑑𐑼 𐑞 𐑧𐑝𐑼-𐑔𐑷𐑑𐑓𐑩𐑤 ·𐑗𐑨𐑮𐑦𐑑𐑦 𐑣𐑨𐑛 𐑒𐑳𐑥 𐑪𐑓 𐑦𐑯 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑, 𐑢𐑦𐑞 𐑣𐑻 𐑤𐑭𐑕𐑑 𐑜𐑦𐑓𐑑—𐑩 𐑯𐑲𐑑-𐑒𐑨𐑐 𐑓 ·𐑕𐑑𐑳𐑚, 𐑞 𐑕𐑧𐑒𐑩𐑯𐑛 𐑥𐑱𐑑, 𐑣𐑻 𐑚𐑮𐑳𐑞𐑼-𐑦𐑯-𐑤𐑷, 𐑯 𐑩 𐑕𐑐𐑺 ·𐑚𐑲𐑚𐑩𐑤 𐑓 𐑞 𐑕𐑑𐑿𐑼𐑛—𐑭𐑓𐑑𐑼 𐑷𐑤 𐑞𐑦𐑕, 𐑞 𐑑𐑵 ·𐑒𐑨𐑐𐑑𐑦𐑯𐑟, ·𐑐𐑰𐑤𐑧𐑜 𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑦𐑖𐑿𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑯 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑞 𐑗𐑰𐑓 𐑥𐑱𐑑, ·𐑐𐑰𐑤𐑧𐑜 𐑕𐑧𐑛:

“𐑯𐑬, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑸 𐑿 𐑖𐑫𐑼 𐑧𐑝𐑮𐑦𐑔𐑦𐑙 𐑦𐑟 𐑮𐑲𐑑? 𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑷𐑤 𐑮𐑧𐑛𐑦—𐑡𐑳𐑕𐑑 𐑕𐑐𐑴𐑒 𐑑 𐑣𐑦𐑥—𐑯𐑳𐑔𐑦𐑙 𐑥𐑹 𐑑 𐑚𐑰 𐑜𐑪𐑑 𐑓𐑮𐑪𐑥 𐑖𐑹, 𐑱? 𐑢𐑧𐑤, 𐑒𐑷𐑤 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟, 𐑞𐑧𐑯. 𐑥𐑳𐑕𐑑𐑼 '𐑩𐑥 𐑭𐑓𐑑 𐑣𐑽—𐑚𐑤𐑭𐑕𐑑 '𐑩𐑥!”

“𐑯𐑴 𐑯𐑰𐑛 𐑝 𐑐𐑮𐑴𐑓𐑱𐑯 𐑢𐑻𐑛𐑟, 𐑣𐑬𐑧𐑝𐑼 𐑜𐑮𐑱𐑑 𐑞 𐑣𐑳𐑮𐑦, ·𐑐𐑰𐑤𐑧𐑜,” 𐑕𐑧𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, “𐑚𐑳𐑑 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑞𐑰, 𐑓𐑮𐑧𐑯𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑯 𐑛𐑵 𐑬𐑼 𐑚𐑦𐑛𐑦𐑙.”

𐑣𐑬 𐑯𐑬! 𐑣𐑽 𐑩𐑐𐑪𐑯 𐑞 𐑝𐑧𐑮𐑦 𐑐𐑶𐑯𐑑 𐑝 𐑕𐑑𐑸𐑑𐑦𐑙 𐑓 𐑞 𐑝𐑶𐑦𐑡, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑢𐑻 𐑜𐑴𐑦𐑙 𐑦𐑑 𐑢𐑦𐑞 𐑩 𐑣𐑲 𐑣𐑨𐑯𐑛 𐑪𐑯 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒, 𐑡𐑳𐑕𐑑 𐑨𐑟 𐑦𐑓 𐑞𐑱 𐑢𐑻 𐑑 𐑚𐑰 𐑡𐑶𐑯𐑑-𐑒𐑩𐑥𐑭𐑯𐑛𐑼𐑟 𐑨𐑑 𐑕𐑰, 𐑨𐑟 𐑢𐑧𐑤 𐑨𐑟 𐑑 𐑷𐑤 𐑩𐑐𐑽𐑩𐑯𐑕𐑩𐑟 𐑦𐑯 𐑐𐑹𐑑. 𐑯, 𐑨𐑟 𐑓 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑯𐑴 𐑕𐑲𐑯 𐑝 𐑣𐑦𐑥 𐑢𐑪𐑟 𐑘𐑧𐑑 𐑑 𐑚𐑰 𐑕𐑰𐑯; 𐑴𐑯𐑤𐑦, 𐑞𐑱 𐑕𐑧𐑛 𐑣𐑰 𐑢𐑪𐑟 𐑦𐑯 𐑞 𐑒𐑨𐑚𐑦𐑯. 𐑚𐑳𐑑 𐑞𐑧𐑯, 𐑞 𐑲𐑛𐑾 𐑢𐑪𐑟, 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑐𐑮𐑧𐑟𐑩𐑯𐑕 𐑢𐑪𐑟 𐑚𐑲 𐑯𐑴 𐑥𐑰𐑯𐑟 𐑯𐑧𐑕𐑦𐑕𐑺𐑦 𐑦𐑯 𐑜𐑧𐑑𐑦𐑙 𐑞 𐑖𐑦𐑐 𐑳𐑯𐑛𐑼 𐑢𐑱, 𐑯 𐑕𐑑𐑽𐑦𐑙 𐑣𐑻 𐑢𐑧𐑤 𐑬𐑑 𐑑 𐑕𐑰. 𐑦𐑯𐑛𐑰𐑛, 𐑨𐑟 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑨𐑑 𐑷𐑤 𐑣𐑦𐑟 𐑐𐑮𐑪𐑐𐑼 𐑚𐑦𐑟𐑯𐑩𐑕, 𐑚𐑳𐑑 𐑞 𐑐𐑲𐑤𐑩𐑑'𐑕; 𐑯 𐑨𐑟 𐑣𐑰 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑘𐑧𐑑 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑮𐑦𐑒𐑳𐑝𐑼𐑛—𐑕𐑴 𐑞𐑱 𐑕𐑧𐑛—𐑞𐑺𐑓𐑹, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑱𐑛 𐑚𐑦𐑤𐑴. 𐑯 𐑷𐑤 𐑞𐑦𐑕 𐑕𐑰𐑥𐑛 𐑯𐑨𐑗𐑼𐑩𐑤 𐑦𐑯𐑳𐑓; 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑨𐑟 𐑦𐑯 𐑞 𐑥𐑻𐑗𐑩𐑯𐑑 𐑕𐑻𐑝𐑦𐑕 𐑥𐑧𐑯𐑦 𐑒𐑨𐑐𐑑𐑦𐑯𐑟 𐑯𐑧𐑝𐑼 𐑖𐑴 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑪𐑯 𐑛𐑧𐑒 𐑓 𐑩 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑑𐑲𐑥 𐑭𐑓𐑑𐑼 𐑣𐑰𐑝𐑦𐑙 𐑳𐑐 𐑞 𐑨𐑙𐑒𐑼, 𐑚𐑳𐑑 𐑮𐑦𐑥𐑱𐑯 𐑴𐑝𐑼 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑑𐑱𐑚𐑩𐑤, 𐑣𐑨𐑝𐑦𐑙 𐑩 𐑓𐑺𐑢𐑧𐑤 𐑥𐑧𐑮𐑦-𐑥𐑱𐑒𐑦𐑙 𐑢𐑦𐑞 𐑞𐑺 𐑖𐑹 𐑓𐑮𐑧𐑯𐑛𐑟, 𐑚𐑦𐑓𐑹 𐑞𐑱 𐑒𐑢𐑦𐑑 𐑞 𐑖𐑦𐑐 𐑓 𐑜𐑫𐑛 𐑢𐑦𐑞 𐑞 𐑐𐑲𐑤𐑩𐑑.

𐑚𐑳𐑑 𐑞𐑺 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑥𐑳𐑗 𐑗𐑭𐑯𐑕 𐑑 𐑔𐑦𐑙𐑒 𐑴𐑝𐑼 𐑞 𐑥𐑨𐑑𐑼, 𐑓 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑢𐑪𐑟 𐑯𐑬 𐑷𐑤 𐑩𐑤𐑲𐑝. 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑑 𐑛𐑵 𐑥𐑴𐑕𐑑 𐑝 𐑞 𐑑𐑷𐑒𐑦𐑙 𐑯 𐑒𐑩𐑥𐑭𐑯𐑛𐑦𐑙, 𐑯 𐑯𐑪𐑑 ·𐑚𐑦𐑤𐑛𐑨𐑛.

“𐑭𐑓𐑑 𐑣𐑽, 𐑘𐑰 𐑕𐑳𐑯𐑟 𐑝 𐑚𐑨𐑗𐑩𐑤𐑼𐑟,” 𐑣𐑰 𐑒𐑮𐑲𐑛, 𐑨𐑟 𐑞 𐑕𐑱𐑤𐑼𐑟 𐑤𐑦𐑙𐑜𐑼𐑛 𐑨𐑑 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑. “·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑛𐑮𐑲𐑝 '𐑩𐑥 𐑭𐑓𐑑.”

“𐑕𐑑𐑮𐑲𐑒 𐑞 𐑑𐑧𐑯𐑑 𐑞𐑺!”—𐑢𐑪𐑟 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑹𐑛𐑼. 𐑨𐑟 𐑲 𐑣𐑦𐑯𐑑𐑩𐑛 𐑚𐑦𐑓𐑹, 𐑞𐑦𐑕 𐑢𐑱𐑤𐑚𐑴𐑯 𐑥𐑸𐑒𐑰 𐑢𐑪𐑟 𐑯𐑧𐑝𐑼 𐑐𐑦𐑗𐑑 𐑦𐑒𐑕𐑧𐑐𐑑 𐑦𐑯 𐑐𐑹𐑑; 𐑯 𐑪𐑯 𐑚𐑹𐑛 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑓 𐑔𐑻𐑑𐑦 𐑘𐑽𐑟, 𐑞 𐑹𐑛𐑼 𐑑 𐑕𐑑𐑮𐑲𐑒 𐑞 𐑑𐑧𐑯𐑑 𐑢𐑪𐑟 𐑢𐑧𐑤 𐑯𐑴𐑯 𐑑 𐑚𐑰 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑔𐑦𐑙 𐑑 𐑣𐑰𐑝𐑦𐑙 𐑳𐑐 𐑞 𐑨𐑙𐑒𐑼.

“𐑥𐑨𐑯 𐑞 𐑒𐑨𐑐𐑕𐑑𐑩𐑯! 𐑚𐑤𐑳𐑛 𐑯 𐑔𐑳𐑯𐑛𐑼!—𐑡𐑳𐑥𐑐!”—𐑢𐑪𐑟 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑒𐑩𐑥𐑭𐑯𐑛, 𐑯 𐑞 𐑒𐑮𐑵 𐑕𐑐𐑮𐑨𐑙 𐑓 𐑞 𐑣𐑨𐑯𐑛𐑕𐑐𐑲𐑒𐑕.

𐑯𐑬 𐑦𐑯 𐑜𐑧𐑑𐑦𐑙 𐑳𐑯𐑛𐑼 𐑢𐑱, 𐑞 𐑕𐑑𐑱𐑖𐑩𐑯 𐑡𐑧𐑯𐑼𐑩𐑤𐑦 𐑪𐑒𐑿𐑐𐑲𐑛 𐑚𐑲 𐑞 𐑐𐑲𐑤𐑩𐑑 𐑦𐑟 𐑞 𐑓𐑹𐑢𐑼𐑛 𐑐𐑸𐑑 𐑝 𐑞 𐑖𐑦𐑐. 𐑯 𐑣𐑽 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑣𐑵, 𐑢𐑦𐑞 ·𐑐𐑰𐑤𐑧𐑜, 𐑚𐑰 𐑦𐑑 𐑯𐑴𐑯, 𐑦𐑯 𐑩𐑛𐑦𐑖𐑩𐑯 𐑑 𐑣𐑦𐑟 𐑳𐑞𐑼 𐑪𐑓𐑦𐑕𐑼𐑟, 𐑢𐑪𐑟 𐑢𐑳𐑯 𐑝 𐑞 𐑤𐑲𐑕𐑩𐑯𐑕𐑑 𐑐𐑲𐑤𐑩𐑑𐑕 𐑝 𐑞 𐑐𐑹𐑑—𐑣𐑰 𐑚𐑰𐑦𐑙 𐑕𐑩𐑕𐑐𐑧𐑒𐑑𐑩𐑛 𐑑 𐑣𐑨𐑝 𐑜𐑪𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑥𐑱𐑛 𐑩 𐑐𐑲𐑤𐑩𐑑 𐑦𐑯 𐑹𐑛𐑼 𐑑 𐑕𐑱𐑝 𐑞 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑐𐑲𐑤𐑩𐑑-𐑓𐑰 𐑑 𐑷𐑤 𐑞 𐑖𐑦𐑐𐑕 𐑣𐑰 𐑢𐑪𐑟 𐑒𐑩𐑯𐑕𐑻𐑯𐑛 𐑦𐑯, 𐑓 𐑣𐑰 𐑯𐑧𐑝𐑼 𐑐𐑲𐑤𐑩𐑑𐑩𐑛 𐑧𐑯𐑦 𐑳𐑞𐑼 𐑒𐑮𐑭𐑓𐑑—·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑲 𐑕𐑱, 𐑥𐑲𐑑 𐑯𐑬 𐑚𐑰 𐑕𐑰𐑯 𐑨𐑒𐑑𐑦𐑝𐑤𐑦 𐑦𐑯𐑜𐑱𐑡𐑛 𐑦𐑯 𐑤𐑫𐑒𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑚𐑬𐑟 𐑓 𐑞 𐑩𐑐𐑮𐑴𐑗𐑦𐑙 𐑨𐑙𐑒𐑼, 𐑯 𐑨𐑑 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑕𐑦𐑙𐑦𐑙 𐑢𐑪𐑑 𐑕𐑰𐑥𐑛 𐑩 𐑛𐑦𐑟𐑥𐑩𐑤 𐑕𐑑𐑱𐑝 𐑝 𐑕𐑭𐑥𐑩𐑛𐑦, 𐑑 𐑗𐑽 𐑞 𐑣𐑨𐑯𐑛𐑟 𐑨𐑑 𐑞 𐑢𐑦𐑯𐑛𐑤𐑩𐑕, 𐑣𐑵 𐑮𐑹𐑛 𐑓𐑹𐑔 𐑕𐑳𐑥 𐑕𐑹𐑑 𐑝 𐑩 𐑒𐑹𐑩𐑕 𐑩𐑚𐑬𐑑 𐑞 𐑜𐑻𐑤𐑟 𐑦𐑯 ·𐑚𐑵𐑚𐑩𐑤 ·𐑨𐑤𐑦, 𐑢𐑦𐑞 𐑣𐑸𐑑𐑦 𐑜𐑫𐑛 𐑢𐑦𐑤. 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕, 𐑯𐑪𐑑 𐑔𐑮𐑰 𐑛𐑱𐑟 𐑐𐑮𐑰𐑝𐑾𐑕, ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑣𐑨𐑛 𐑑𐑴𐑤𐑛 𐑞𐑧𐑥 𐑞𐑨𐑑 𐑯𐑴 𐑐𐑮𐑴𐑓𐑱𐑯 𐑕𐑪𐑙𐑟 𐑢𐑫𐑛 𐑚𐑰 𐑩𐑤𐑬𐑛 𐑪𐑯 𐑚𐑹𐑛 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼𐑤𐑦 𐑦𐑯 𐑜𐑧𐑑𐑦𐑙 𐑳𐑯𐑛𐑼 𐑢𐑱; 𐑯 ·𐑗𐑨𐑮𐑦𐑑𐑦, 𐑣𐑦𐑟 𐑕𐑦𐑕𐑑𐑼, 𐑣𐑨𐑛 𐑐𐑤𐑱𐑕𐑑 𐑩 𐑕𐑥𐑷𐑤 𐑗𐑶𐑕 𐑒𐑪𐑐𐑦 𐑝 ·𐑢𐑪𐑑𐑕 𐑦𐑯 𐑰𐑗 𐑕𐑰𐑥𐑩𐑯'𐑟 𐑚𐑻𐑔.

𐑥𐑰𐑯𐑑𐑲𐑥, 𐑴𐑝𐑼𐑕𐑰𐑦𐑙 𐑞 𐑳𐑞𐑼 𐑐𐑸𐑑 𐑝 𐑞 𐑖𐑦𐑐, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑮𐑦𐑐𐑑 𐑯 𐑕𐑢𐑹 𐑩𐑕𐑑𐑻𐑯 𐑦𐑯 𐑞 𐑥𐑴𐑕𐑑 𐑓𐑮𐑲𐑑𐑓𐑩𐑤 𐑥𐑨𐑯𐑼. 𐑲 𐑷𐑤𐑥𐑴𐑕𐑑 𐑔𐑷𐑑 𐑣𐑰 𐑢𐑫𐑛 𐑕𐑦𐑙𐑒 𐑞 𐑖𐑦𐑐 𐑚𐑦𐑓𐑹 𐑞 𐑨𐑙𐑒𐑼 𐑒𐑫𐑛 𐑚𐑰 𐑜𐑪𐑑 𐑳𐑐; 𐑦𐑯𐑝𐑪𐑤𐑩𐑯𐑑𐑺𐑦𐑤𐑦 𐑲 𐑐𐑷𐑟𐑛 𐑪𐑯 𐑥𐑲 𐑣𐑨𐑯𐑛𐑕𐑐𐑲𐑒, 𐑯 𐑑𐑴𐑤𐑛 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑑 𐑛𐑵 𐑞 𐑕𐑱𐑥, 𐑔𐑦𐑙𐑒𐑦𐑙 𐑝 𐑞 𐑐𐑧𐑮𐑦𐑤𐑟 𐑢𐑰 𐑚𐑴𐑔 𐑮𐑨𐑯, 𐑦𐑯 𐑕𐑑𐑸𐑑𐑦𐑙 𐑪𐑯 𐑞 𐑝𐑶𐑦𐑡 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑩 𐑛𐑧𐑝𐑩𐑤 𐑓 𐑩 𐑐𐑲𐑤𐑩𐑑. 𐑲 𐑢𐑪𐑟 𐑒𐑳𐑥𐑓𐑼𐑑𐑦𐑙 𐑥𐑲𐑕𐑧𐑤𐑓, 𐑣𐑬𐑧𐑝𐑼, 𐑢𐑦𐑞 𐑞 𐑔𐑷𐑑 𐑞𐑨𐑑 𐑦𐑯 𐑐𐑲𐑩𐑕 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑥𐑲𐑑 𐑚𐑰 𐑓𐑬𐑯𐑛 𐑕𐑳𐑥 𐑕𐑨𐑤𐑝𐑱𐑖𐑩𐑯, 𐑕𐑐𐑲𐑑 𐑝 𐑣𐑦𐑟 𐑕𐑧𐑝𐑩𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑕𐑧𐑝𐑩𐑯𐑔 𐑤𐑱; 𐑢𐑧𐑯 𐑲 𐑓𐑧𐑤𐑑 𐑩 𐑕𐑳𐑛𐑩𐑯 𐑖𐑸𐑐 𐑐𐑴𐑒 𐑦𐑯 𐑥𐑲 𐑮𐑽, 𐑯 𐑑𐑻𐑯𐑦𐑙 𐑮𐑬𐑯𐑛, 𐑢𐑪𐑟 𐑣𐑪𐑮𐑦𐑓𐑲𐑛 𐑨𐑑 𐑞 𐑨𐑐𐑼𐑦𐑖𐑩𐑯 𐑝 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑦𐑯 𐑞 𐑨𐑒𐑑 𐑝 𐑢𐑦𐑞𐑛𐑮𐑷𐑦𐑙 𐑣𐑦𐑟 𐑤𐑧𐑜 𐑓𐑮𐑪𐑥 𐑥𐑲 𐑦𐑥𐑰𐑛𐑾𐑑 𐑝𐑦𐑕𐑦𐑯𐑦𐑑𐑦. 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑥𐑲 𐑓𐑻𐑕𐑑 𐑒𐑦𐑒.

“𐑦𐑟 𐑞𐑨𐑑 𐑞 𐑢𐑱 𐑞𐑱 𐑣𐑰𐑝 𐑦𐑯 𐑞 𐑥𐑸𐑗𐑩𐑯𐑑 𐑕𐑻𐑝𐑦𐑕?” 𐑣𐑰 𐑮𐑹𐑛. “𐑕𐑐𐑮𐑦𐑙, 𐑞𐑬 𐑖𐑰𐑐-𐑣𐑧𐑛; 𐑕𐑐𐑮𐑦𐑙, 𐑯 𐑚𐑮𐑱𐑒 𐑞𐑲 𐑚𐑨𐑒𐑚𐑴𐑯! 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑘𐑰 𐑕𐑐𐑮𐑦𐑙, 𐑲 𐑕𐑱, 𐑷𐑤 𐑝 𐑘𐑰—𐑕𐑐𐑮𐑦𐑙! ·𐑒𐑢𐑴𐑙𐑪𐑜! 𐑕𐑐𐑮𐑦𐑙, 𐑞𐑬 𐑗𐑨𐑐 𐑢𐑦𐑞 𐑞 𐑮𐑧𐑛 𐑢𐑦𐑕𐑒𐑼𐑟; 𐑕𐑐𐑮𐑦𐑙 𐑞𐑺, ·𐑕𐑒𐑪𐑗-𐑒𐑨𐑐; 𐑕𐑐𐑮𐑦𐑙, 𐑞𐑬 𐑜𐑮𐑰𐑯 𐑐𐑨𐑯𐑑𐑕. 𐑕𐑐𐑮𐑦𐑙, 𐑲 𐑕𐑱, 𐑷𐑤 𐑝 𐑘𐑰, 𐑯 𐑕𐑐𐑮𐑦𐑙 𐑘𐑹 𐑲𐑟 𐑬𐑑!” 𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙, 𐑣𐑰 𐑥𐑵𐑝𐑛 𐑩𐑤𐑪𐑙 𐑞 𐑢𐑦𐑯𐑛𐑤𐑩𐑕, 𐑣𐑽 𐑯 𐑞𐑺 𐑿𐑟𐑦𐑙 𐑣𐑦𐑟 𐑤𐑧𐑜 𐑝𐑧𐑮𐑦 𐑓𐑮𐑰𐑤𐑦, 𐑢𐑲𐑤 𐑦𐑥𐑐𐑼𐑑𐑻𐑚𐑩𐑚𐑩𐑤 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑒𐑧𐑐𐑑 𐑤𐑰𐑛𐑦𐑙 𐑪𐑓 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑕𐑭𐑥𐑩𐑛𐑦. 𐑔𐑦𐑙𐑒𐑕 𐑲, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑛𐑮𐑦𐑙𐑒𐑦𐑙 𐑕𐑳𐑥𐑔𐑦𐑙 𐑑-𐑛𐑱.

𐑨𐑑 𐑤𐑭𐑕𐑑 𐑞 𐑨𐑙𐑒𐑼 𐑢𐑪𐑟 𐑳𐑐, 𐑞 𐑕𐑱𐑤𐑟 𐑢𐑻 𐑕𐑧𐑑, 𐑯 𐑪𐑓 𐑢𐑰 𐑜𐑤𐑲𐑛𐑩𐑛. 𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑖𐑹𐑑, 𐑒𐑴𐑤𐑛 ·𐑒𐑮𐑦𐑕𐑥𐑩𐑕; 𐑯 𐑨𐑟 𐑞 𐑖𐑹𐑑 𐑯𐑹𐑞𐑼𐑯 𐑛𐑱 𐑥𐑻𐑡𐑛 𐑦𐑯𐑑𐑵 𐑯𐑲𐑑, 𐑢𐑰 𐑓𐑬𐑯𐑛 𐑬𐑼𐑕𐑧𐑤𐑝𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑚𐑮𐑷𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑢𐑦𐑯𐑑𐑮𐑦 𐑴𐑖𐑩𐑯, 𐑣𐑵𐑟 𐑓𐑮𐑰𐑟𐑦𐑙 𐑕𐑐𐑮𐑱 𐑒𐑱𐑕𐑑 𐑳𐑕 𐑦𐑯 𐑲𐑕, 𐑨𐑟 𐑦𐑯 𐑐𐑪𐑤𐑦𐑖𐑑 𐑸𐑥𐑼. 𐑞 𐑤𐑪𐑙 𐑮𐑴𐑟 𐑝 𐑑𐑰𐑔 𐑪𐑯 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕 𐑜𐑤𐑦𐑕𐑩𐑯𐑛 𐑦𐑯 𐑞 𐑥𐑵𐑯𐑤𐑲𐑑; 𐑯 𐑤𐑲𐑒 𐑞 𐑢𐑲𐑑 𐑲𐑝𐑼𐑦 𐑑𐑳𐑕𐑒𐑕 𐑝 𐑕𐑳𐑥 𐑣𐑿𐑡 𐑧𐑤𐑦𐑓𐑩𐑯𐑑, 𐑝𐑭𐑕𐑑 𐑒𐑻𐑝𐑦𐑙 𐑲𐑕𐑦𐑒𐑩𐑤𐑟 𐑛𐑦𐑐𐑧𐑯𐑛𐑩𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑬𐑟.

·𐑤𐑨𐑙𐑒 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑨𐑟 𐑐𐑲𐑤𐑩𐑑, 𐑣𐑧𐑛𐑩𐑛 𐑞 𐑓𐑻𐑕𐑑 𐑢𐑪𐑗, 𐑯 𐑧𐑝𐑼 𐑯 𐑩𐑯𐑪𐑯, 𐑨𐑟 𐑞 𐑴𐑤𐑛 𐑒𐑮𐑭𐑓𐑑 𐑛𐑰𐑐 𐑛𐑲𐑝𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑜𐑮𐑰𐑯 𐑕𐑰𐑟, 𐑯 𐑕𐑧𐑯𐑑 𐑞 𐑖𐑦𐑝𐑼𐑦𐑙 𐑓𐑮𐑪𐑕𐑑 𐑷𐑤 𐑴𐑝𐑼 𐑣𐑻, 𐑯 𐑞 𐑢𐑦𐑯𐑛𐑟 𐑣𐑬𐑤𐑛, 𐑯 𐑞 𐑒𐑹𐑛𐑦𐑡 𐑮𐑨𐑙, 𐑣𐑦𐑟 𐑕𐑑𐑧𐑛𐑦 𐑯𐑴𐑑𐑕 𐑢𐑻 𐑣𐑻𐑛,—

“𐑕𐑢𐑰𐑑 𐑓𐑰𐑤𐑛𐑟 𐑚𐑦𐑘𐑪𐑯𐑛 𐑞 𐑕𐑢𐑧𐑤𐑦𐑙 𐑓𐑤𐑳𐑛,

·𐑕𐑑𐑨𐑯𐑛 𐑛𐑮𐑧𐑕𐑑 𐑦𐑯 𐑤𐑦𐑝𐑦𐑙 𐑜𐑮𐑰𐑯.

𐑕𐑴 𐑑 𐑞 ·𐑡𐑵𐑟 𐑴𐑤𐑛 ·𐑒𐑱𐑯𐑩𐑯 𐑕𐑑𐑫𐑛,

·𐑢𐑲𐑤 ·𐑡𐑹𐑛𐑩𐑯 𐑮𐑴𐑤𐑛 𐑚𐑦𐑑𐑢𐑰𐑯.”

𐑯𐑧𐑝𐑼 𐑛𐑦𐑛 𐑞𐑴𐑟 𐑕𐑢𐑰𐑑 𐑢𐑻𐑛𐑟 𐑕𐑬𐑯𐑛 𐑥𐑹 𐑕𐑢𐑰𐑑𐑤𐑦 𐑑 𐑥𐑰 𐑞𐑨𐑯 𐑞𐑧𐑯. 𐑞𐑱 𐑢𐑻 𐑓𐑫𐑤 𐑝 𐑣𐑴𐑐 𐑯 𐑓𐑮𐑵𐑦𐑖𐑩𐑯. 𐑕𐑐𐑲𐑑 𐑝 𐑞𐑦𐑕 𐑓𐑮𐑦𐑡𐑦𐑛 𐑢𐑦𐑯𐑑𐑼 𐑯𐑲𐑑 𐑦𐑯 𐑞 𐑚𐑶𐑕𐑑𐑼𐑩𐑕 ·𐑩𐑑𐑤𐑨𐑯𐑑𐑦𐑒, 𐑕𐑐𐑲𐑑 𐑝 𐑥𐑲 𐑢𐑧𐑑 𐑓𐑰𐑑 𐑯 𐑢𐑧𐑑𐑼 𐑡𐑨𐑒𐑩𐑑, 𐑞𐑺 𐑢𐑪𐑟 𐑘𐑧𐑑, 𐑦𐑑 𐑞𐑧𐑯 𐑕𐑰𐑥𐑛 𐑑 𐑥𐑰, 𐑥𐑧𐑯𐑦 𐑩 𐑐𐑤𐑧𐑟𐑩𐑯𐑑 𐑣𐑱𐑝𐑩𐑯 𐑦𐑯 𐑕𐑑𐑹; 𐑯 𐑥𐑰𐑛𐑟 𐑯 𐑜𐑤𐑱𐑛𐑟 𐑕𐑴 𐑦𐑑𐑻𐑯𐑩𐑤𐑦 𐑝𐑻𐑯𐑩𐑤, 𐑞𐑨𐑑 𐑞 𐑜𐑮𐑭𐑕 𐑖𐑪𐑑 𐑳𐑐 𐑚𐑲 𐑞 𐑕𐑐𐑮𐑦𐑙, 𐑳𐑯𐑑𐑮𐑪𐑛𐑩𐑯, 𐑳𐑯𐑢𐑦𐑤𐑑𐑩𐑛, 𐑮𐑦𐑥𐑱𐑯𐑟 𐑨𐑑 𐑥𐑦𐑛𐑕𐑳𐑥𐑼.

𐑨𐑑 𐑤𐑭𐑕𐑑 𐑢𐑰 𐑜𐑱𐑯𐑛 𐑕𐑳𐑗 𐑩𐑯 𐑪𐑓𐑦𐑙, 𐑞𐑨𐑑 𐑞 𐑑𐑵 𐑐𐑲𐑤𐑩𐑑𐑕 𐑢𐑻 𐑯𐑰𐑛𐑩𐑛 𐑯𐑴 𐑤𐑪𐑙𐑜𐑼. 𐑞 𐑕𐑑𐑬𐑑 𐑕𐑱𐑤-𐑚𐑴𐑑 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑛 𐑳𐑕 𐑚𐑦𐑜𐑨𐑯 𐑮𐑱𐑯𐑡𐑦𐑙 𐑩𐑤𐑪𐑙𐑕𐑲𐑛.

𐑦𐑑 𐑢𐑪𐑟 𐑒𐑘𐑫𐑼𐑾𐑕 𐑯 𐑯𐑪𐑑 𐑳𐑯𐑐𐑤𐑰𐑟𐑦𐑙, 𐑣𐑬 ·𐑐𐑰𐑤𐑧𐑜 𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑢𐑻 𐑩𐑓𐑧𐑒𐑑𐑩𐑛 𐑨𐑑 𐑞𐑦𐑕 𐑡𐑳𐑙𐑒𐑗𐑼, 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛. 𐑓 𐑤𐑴𐑞 𐑑 𐑛𐑦𐑐𐑸𐑑, 𐑘𐑧𐑑; 𐑝𐑧𐑮𐑦 𐑤𐑴𐑞 𐑑 𐑤𐑰𐑝, 𐑓 𐑜𐑫𐑛, 𐑩 𐑖𐑦𐑐 𐑚𐑬𐑯𐑛 𐑪𐑯 𐑕𐑴 𐑤𐑪𐑙 𐑯 𐑐𐑧𐑮𐑦𐑤𐑩𐑕 𐑩 𐑝𐑶𐑦𐑡—𐑚𐑦𐑘𐑪𐑯𐑛 𐑚𐑴𐑔 𐑕𐑑𐑹𐑥𐑦 ·𐑒𐑱𐑐𐑕; 𐑩 𐑖𐑦𐑐 𐑦𐑯 𐑢𐑦𐑗 𐑕𐑳𐑥 𐑔𐑬𐑟𐑩𐑯𐑛𐑟 𐑝 𐑣𐑦𐑟 𐑣𐑸𐑛 𐑻𐑯𐑛 𐑛𐑪𐑤𐑼𐑟 𐑢𐑻 𐑦𐑯𐑝𐑧𐑕𐑑𐑩𐑛; 𐑩 𐑖𐑦𐑐, 𐑦𐑯 𐑢𐑦𐑗 𐑩𐑯 𐑴𐑤𐑛 𐑖𐑦𐑐𐑥𐑱𐑑 𐑕𐑱𐑤𐑛 𐑨𐑟 𐑒𐑨𐑐𐑑𐑦𐑯; 𐑩 𐑥𐑨𐑯 𐑷𐑤𐑥𐑴𐑕𐑑 𐑨𐑟 𐑴𐑤𐑛 𐑨𐑟 𐑣𐑰, 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑕𐑑𐑸𐑑𐑦𐑙 𐑑 𐑦𐑯𐑒𐑬𐑯𐑑𐑼 𐑷𐑤 𐑞 𐑑𐑧𐑮𐑼𐑟 𐑝 𐑞 𐑐𐑦𐑑𐑦𐑤𐑩𐑕 𐑡𐑷; 𐑤𐑴𐑞 𐑑 𐑕𐑱 𐑜𐑫𐑛-𐑚𐑲 𐑑 𐑩 𐑔𐑦𐑙 𐑕𐑴 𐑧𐑝𐑮𐑦 𐑢𐑱 𐑚𐑮𐑦𐑥𐑓𐑫𐑤 𐑝 𐑧𐑝𐑮𐑦 𐑦𐑯𐑑𐑼𐑩𐑕𐑑 𐑑 𐑣𐑦𐑥,—𐑐𐑫𐑼 𐑴𐑤𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑤𐑦𐑙𐑜𐑼𐑛 𐑤𐑪𐑙; 𐑐𐑱𐑕𐑑 𐑞 𐑛𐑧𐑒 𐑢𐑦𐑞 𐑨𐑙𐑒𐑖𐑩𐑕 𐑕𐑑𐑮𐑲𐑛𐑟; 𐑮𐑨𐑯 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑑 𐑕𐑐𐑰𐑒 𐑩𐑯𐑳𐑞𐑼 𐑓𐑺𐑢𐑧𐑤 𐑢𐑻𐑛 𐑞𐑺; 𐑩𐑜𐑱𐑯 𐑒𐑱𐑥 𐑪𐑯 𐑛𐑧𐑒, 𐑯 𐑤𐑫𐑒𐑑 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛; 𐑤𐑫𐑒𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑢𐑲𐑛 𐑯 𐑧𐑯𐑛𐑤𐑩𐑕 𐑢𐑷𐑑𐑼𐑟, 𐑴𐑯𐑤𐑦 𐑚𐑬𐑯𐑛𐑩𐑛 𐑚𐑲 𐑞 𐑓𐑸-𐑪𐑓 𐑳𐑯𐑕𐑰𐑯 ·𐑰𐑕𐑑𐑼𐑯 ·𐑒𐑪𐑯𐑑𐑦𐑯𐑩𐑯𐑑𐑕; 𐑤𐑫𐑒𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑤𐑨𐑯𐑛; 𐑤𐑫𐑒𐑑 𐑩𐑤𐑪𐑓𐑑; 𐑤𐑫𐑒𐑑 𐑮𐑲𐑑 𐑯 𐑤𐑧𐑓𐑑; 𐑤𐑫𐑒𐑑 𐑧𐑝𐑮𐑦𐑢𐑺 𐑯 𐑯𐑴𐑢𐑺; 𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑥𐑦𐑒𐑨𐑯𐑦𐑒𐑤𐑦 𐑒𐑶𐑤𐑦𐑙 𐑩 𐑮𐑴𐑐 𐑩𐑐𐑪𐑯 𐑦𐑑𐑕 𐑐𐑦𐑯, 𐑒𐑩𐑯𐑝𐑳𐑤𐑕𐑦𐑝𐑤𐑦 𐑜𐑮𐑭𐑕𐑐𐑑 𐑕𐑑𐑬𐑑 ·𐑐𐑰𐑤𐑧𐑜 𐑚𐑲 𐑞 𐑣𐑨𐑯𐑛, 𐑯 𐑣𐑴𐑤𐑛𐑦𐑙 𐑳𐑐 𐑩 𐑤𐑨𐑯𐑑𐑼𐑯, 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑕𐑑𐑫𐑛 𐑜𐑱𐑟𐑦𐑙 𐑣𐑦𐑮𐑴𐑦𐑒𐑤𐑦 𐑦𐑯 𐑣𐑦𐑟 𐑓𐑱𐑕, 𐑨𐑟 𐑥𐑳𐑗 𐑨𐑟 𐑑 𐑕𐑱, “𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕, 𐑓𐑮𐑧𐑯𐑛 ·𐑐𐑰𐑤𐑧𐑜, 𐑲 𐑒𐑨𐑯 𐑕𐑑𐑨𐑯𐑛 𐑦𐑑; 𐑘𐑧𐑕, 𐑲 𐑒𐑨𐑯.”

𐑨𐑟 𐑓 ·𐑐𐑰𐑤𐑧𐑜 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑣𐑰 𐑑𐑫𐑒 𐑦𐑑 𐑥𐑹 𐑤𐑲𐑒 𐑩 𐑓𐑦𐑤𐑪𐑕𐑩𐑓𐑼; 𐑚𐑳𐑑 𐑓 𐑷𐑤 𐑣𐑦𐑟 𐑓𐑦𐑤𐑪𐑕𐑩𐑓𐑦, 𐑞𐑺 𐑢𐑪𐑟 𐑩 𐑑𐑺 𐑑𐑢𐑦𐑙𐑒𐑩𐑤𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑲, 𐑢𐑧𐑯 𐑞 𐑤𐑨𐑯𐑑𐑼𐑯 𐑒𐑱𐑥 𐑑𐑵 𐑯𐑽. 𐑯 𐑣𐑰, 𐑑𐑵, 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑮𐑳𐑯 𐑓𐑮𐑪𐑥 𐑒𐑨𐑚𐑦𐑯 𐑑 𐑛𐑧𐑒—𐑯𐑬 𐑩 𐑢𐑻𐑛 𐑚𐑦𐑤𐑴, 𐑯 𐑯𐑬 𐑩 𐑢𐑻𐑛 𐑢𐑦𐑞 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑞 𐑗𐑰𐑓 𐑥𐑱𐑑.

𐑚𐑳𐑑, 𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑣𐑰 𐑑𐑻𐑯𐑛 𐑑 𐑣𐑦𐑟 𐑒𐑪𐑥𐑮𐑱𐑛, 𐑢𐑦𐑞 𐑩 𐑓𐑲𐑯𐑩𐑤 𐑕𐑹𐑑 𐑝 𐑤𐑫𐑒 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥,—“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛—𐑒𐑳𐑥, 𐑴𐑤𐑛 𐑖𐑦𐑐𐑥𐑱𐑑, 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑜𐑴. 𐑚𐑨𐑒 𐑞 𐑥𐑱𐑯-𐑘𐑸𐑛 𐑞𐑺! 𐑚𐑴𐑑 𐑩𐑣𐑶! 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑒𐑳𐑥 𐑒𐑤𐑴𐑕 𐑩𐑤𐑪𐑙𐑕𐑲𐑛, 𐑯𐑬! 𐑒𐑺𐑓𐑩𐑤, 𐑒𐑺𐑓𐑩𐑤!—𐑒𐑳𐑥, ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑚𐑶—𐑕𐑱 𐑘𐑹 𐑤𐑭𐑕𐑑. 𐑤𐑳𐑒 𐑑 𐑘𐑰, ·𐑕𐑑𐑸𐑚𐑳𐑒—𐑤𐑳𐑒 𐑑 𐑘𐑰, ·𐑥𐑼 ·𐑕𐑑𐑳𐑚—𐑤𐑳𐑒 𐑑 𐑘𐑰, ·𐑥𐑼 ·𐑓𐑤𐑭𐑕𐑒—𐑜𐑫𐑛-𐑚𐑲 𐑯 𐑜𐑫𐑛 𐑤𐑳𐑒 𐑑 𐑘𐑰 𐑷𐑤—𐑯 𐑞𐑦𐑕 𐑛𐑱 𐑔𐑮𐑰 𐑘𐑽𐑟 𐑲'𐑤 𐑣𐑨𐑝 𐑩 𐑣𐑪𐑑 𐑕𐑳𐑐𐑼 𐑕𐑥𐑴𐑒𐑦𐑙 𐑓 𐑘𐑰 𐑦𐑯 𐑴𐑤𐑛 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑. 𐑣𐑼𐑭 𐑯 𐑩𐑢𐑱!”

“𐑜𐑪𐑛 𐑚𐑤𐑧𐑕 𐑘𐑰, 𐑯 𐑣𐑨𐑝 𐑘𐑰 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑴𐑤𐑦 𐑒𐑰𐑐𐑦𐑙, 𐑥𐑧𐑯,” 𐑥𐑻𐑥𐑼𐑛 𐑴𐑤𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑯𐑒𐑴𐑣𐑽𐑩𐑯𐑑𐑤𐑦. “𐑲 𐑣𐑴𐑐 𐑘𐑰'𐑤 𐑣𐑨𐑝 𐑓𐑲𐑯 𐑢𐑧𐑞𐑼 𐑯𐑬, 𐑕𐑴 𐑞𐑨𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑥𐑱 𐑕𐑵𐑯 𐑚𐑰 𐑥𐑵𐑝𐑦𐑙 𐑩𐑥𐑳𐑙 𐑘𐑰—𐑩 𐑐𐑤𐑧𐑟𐑩𐑯𐑑 𐑕𐑳𐑯 𐑦𐑟 𐑷𐑤 𐑣𐑰 𐑯𐑰𐑛𐑟, 𐑯 𐑘𐑰'𐑤 𐑣𐑨𐑝 𐑐𐑤𐑧𐑯𐑑𐑦 𐑝 𐑞𐑧𐑥 𐑦𐑯 𐑞 𐑑𐑮𐑪𐑐𐑦𐑒 𐑝𐑶𐑦𐑡 𐑘𐑰 𐑜𐑴. 𐑚𐑰 𐑒𐑺𐑓𐑩𐑤 𐑦𐑯 𐑞 𐑣𐑳𐑯𐑑, 𐑘𐑰 𐑥𐑱𐑑𐑕. 𐑛𐑴𐑯𐑑 𐑕𐑑𐑱𐑝 𐑞 𐑚𐑴𐑑𐑕 𐑯𐑰𐑛𐑤𐑩𐑕𐑤𐑦, 𐑘𐑰 𐑣𐑸𐑐𐑵𐑯𐑽𐑟; 𐑜𐑫𐑛 𐑢𐑲𐑑 𐑕𐑰𐑛𐑼 𐑐𐑤𐑨𐑙𐑒 𐑦𐑟 𐑮𐑱𐑟𐑛 𐑓𐑫𐑤 𐑔𐑮𐑰 𐑐𐑻 𐑕𐑧𐑯𐑑 𐑢𐑦𐑞𐑦𐑯 𐑞 𐑘𐑽. 𐑛𐑴𐑯𐑑 𐑓𐑼𐑜𐑧𐑑 𐑘𐑹 𐑐𐑮𐑱𐑼𐑟, 𐑲𐑞𐑼. ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑥𐑲𐑯𐑛 𐑞𐑨𐑑 𐑒𐑵𐑐𐑼 𐑛𐑴𐑯𐑑 𐑢𐑱𐑕𐑑 𐑞 𐑕𐑐𐑺 𐑕𐑑𐑱𐑝𐑟. 𐑴! 𐑞 𐑕𐑱𐑤-𐑯𐑰𐑛𐑩𐑤𐑟 𐑸 𐑦𐑯 𐑞 𐑜𐑮𐑰𐑯 𐑤𐑪𐑒𐑼! 𐑛𐑴𐑯𐑑 𐑢𐑱𐑤 𐑦𐑑 𐑑𐑵 𐑥𐑳𐑗 𐑩' ·𐑤𐑹𐑛'𐑟 𐑛𐑱𐑟, 𐑥𐑧𐑯; 𐑚𐑳𐑑 𐑛𐑴𐑯𐑑 𐑥𐑦𐑕 𐑩 𐑓𐑺 𐑗𐑭𐑯𐑕 𐑲𐑞𐑼, 𐑞𐑨𐑑'𐑕 𐑮𐑦𐑡𐑧𐑒𐑑𐑦𐑙 ·𐑣𐑧𐑝𐑩𐑯'𐑟 𐑜𐑫𐑛 𐑜𐑦𐑓𐑑𐑕. 𐑣𐑨𐑝 𐑩𐑯 𐑲 𐑑 𐑞 𐑥𐑩𐑤𐑨𐑕𐑩𐑟 𐑑𐑽𐑕, ·𐑥𐑼 ·𐑕𐑑𐑳𐑚; 𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑤𐑰𐑒𐑦, 𐑲 𐑔𐑷𐑑. 𐑦𐑓 𐑘𐑰 𐑑𐑳𐑗 𐑨𐑑 𐑞 𐑲𐑤𐑩𐑯𐑛𐑟, ·𐑥𐑼 ·𐑓𐑤𐑭𐑕𐑒, 𐑚𐑦𐑢𐑺 𐑝 𐑓𐑹𐑯𐑦𐑒𐑱𐑖𐑩𐑯. 𐑜𐑫𐑛-𐑚𐑲, 𐑜𐑫𐑛-𐑚𐑲! 𐑛𐑴𐑯𐑑 𐑒𐑰𐑐 𐑞𐑨𐑑 𐑗𐑰𐑟 𐑑𐑵 𐑤𐑪𐑙 𐑛𐑬𐑯 𐑦𐑯 𐑞 𐑣𐑴𐑤𐑛, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑦𐑑'𐑤 𐑕𐑐𐑶𐑤. 𐑚𐑰 𐑒𐑺𐑓𐑩𐑤 𐑢𐑦𐑞 𐑞 𐑚𐑳𐑑𐑼—𐑑𐑢𐑧𐑯𐑑𐑦 𐑕𐑧𐑯𐑑𐑕 𐑞 𐑐𐑬𐑯𐑛 𐑦𐑑 𐑢𐑪𐑟, 𐑯 𐑥𐑲𐑯𐑛 𐑘𐑰, 𐑦𐑓—”

“𐑒𐑳𐑥, 𐑒𐑳𐑥, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛; 𐑕𐑑𐑪𐑐 𐑐𐑩𐑤𐑭𐑝𐑼𐑦𐑙,—𐑩𐑢𐑱!” 𐑯 𐑢𐑦𐑞 𐑞𐑨𐑑, ·𐑐𐑰𐑤𐑧𐑜 𐑣𐑳𐑮𐑦𐑛 𐑣𐑦𐑥 𐑴𐑝𐑼 𐑞 𐑕𐑲𐑛, 𐑯 𐑚𐑴𐑔 𐑛𐑮𐑪𐑐𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑚𐑴𐑑.

𐑖𐑦𐑐 𐑯 𐑚𐑴𐑑 𐑛𐑲𐑝𐑻𐑡𐑛; 𐑞 𐑒𐑴𐑤𐑛, 𐑛𐑨𐑥𐑐 𐑯𐑲𐑑 𐑚𐑮𐑰𐑟 𐑚𐑤𐑵 𐑚𐑦𐑑𐑢𐑰𐑯; 𐑩 𐑕𐑒𐑮𐑰𐑥𐑦𐑙 𐑜𐑳𐑤 𐑓𐑤𐑵 𐑴𐑝𐑼𐑣𐑧𐑛; 𐑞 𐑑𐑵 𐑣𐑳𐑤𐑟 𐑢𐑲𐑤𐑛𐑤𐑦 𐑮𐑴𐑤𐑛; 𐑢𐑰 𐑜𐑱𐑝 𐑔𐑮𐑰 𐑣𐑧𐑝𐑦-𐑣𐑸𐑑𐑩𐑛 𐑗𐑽𐑟, 𐑯 𐑚𐑤𐑲𐑯𐑛𐑤𐑦 𐑐𐑤𐑳𐑯𐑡𐑛 𐑤𐑲𐑒 𐑓𐑱𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑤𐑴𐑯 ·𐑩𐑑𐑤𐑨𐑯𐑑𐑦𐑒.
