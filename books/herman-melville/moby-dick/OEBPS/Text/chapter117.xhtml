·𐑗𐑨𐑐𐑑𐑼 117. 𐑞 ·𐑢𐑱𐑤 ·𐑢𐑪𐑗.

𐑞 𐑓𐑹 𐑢𐑱𐑤𐑟 𐑕𐑤𐑱𐑯 𐑞𐑨𐑑 𐑰𐑝𐑯𐑦𐑙 𐑣𐑨𐑛 𐑛𐑲𐑛 𐑢𐑲𐑛 𐑩𐑐𐑸𐑑; 𐑢𐑳𐑯, 𐑓𐑸 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛; 𐑢𐑳𐑯, 𐑤𐑧𐑕 𐑛𐑦𐑕𐑑𐑩𐑯𐑑, 𐑑 𐑤𐑰𐑢𐑼𐑛; 𐑢𐑳𐑯 𐑩𐑣𐑧𐑛; 𐑢𐑳𐑯 𐑩𐑕𐑑𐑻𐑯. 𐑞𐑰𐑟 𐑤𐑭𐑕𐑑 𐑔𐑮𐑰 𐑢𐑻 𐑚𐑮𐑷𐑑 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑺 𐑯𐑲𐑑𐑓𐑷𐑤; 𐑚𐑳𐑑 𐑞 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑢𐑳𐑯 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑚𐑰 𐑮𐑰𐑗𐑑 𐑑𐑦𐑤 𐑥𐑹𐑯𐑦𐑙; 𐑯 𐑞 𐑚𐑴𐑑 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑒𐑦𐑤𐑛 𐑦𐑑 𐑤𐑱 𐑚𐑲 𐑦𐑑𐑕 𐑕𐑲𐑛 𐑷𐑤 𐑯𐑲𐑑; 𐑯 𐑞𐑨𐑑 𐑚𐑴𐑑 𐑢𐑪𐑟 ·𐑱𐑣𐑨𐑚'𐑟.

𐑞 𐑢𐑱𐑓-𐑐𐑴𐑤 𐑢𐑪𐑟 𐑔𐑮𐑳𐑕𐑑 𐑳𐑐𐑮𐑲𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑛𐑧𐑛 𐑢𐑱𐑤'𐑟 𐑕𐑐𐑬𐑑-𐑣𐑴𐑤; 𐑯 𐑞 𐑤𐑨𐑯𐑑𐑼𐑯 𐑣𐑨𐑙𐑦𐑙 𐑓𐑮𐑪𐑥 𐑦𐑑𐑕 𐑑𐑪𐑐, 𐑒𐑭𐑕𐑑 𐑩 𐑑𐑮𐑳𐑚𐑩𐑤𐑛 𐑓𐑤𐑦𐑒𐑼𐑦𐑙 𐑜𐑤𐑺 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑤𐑨𐑒, 𐑜𐑤𐑪𐑕𐑦 𐑚𐑨𐑒, 𐑯 𐑓𐑸 𐑬𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑥𐑦𐑛𐑯𐑲𐑑 𐑢𐑱𐑝𐑟, 𐑢𐑦𐑗 𐑡𐑧𐑯𐑑𐑤𐑦 𐑗𐑱𐑓𐑑 𐑞 𐑢𐑱𐑤'𐑟 𐑚𐑮𐑷𐑛 𐑓𐑤𐑨𐑙𐑒, 𐑤𐑲𐑒 𐑕𐑪𐑓𐑑 𐑕𐑻𐑓 𐑩𐑐𐑪𐑯 𐑩 𐑚𐑰𐑗.

𐑱𐑣𐑨𐑚 𐑯 𐑷𐑤 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵 𐑕𐑰𐑥𐑛 𐑩𐑕𐑤𐑰𐑐 𐑚𐑳𐑑 𐑞 ·𐑐𐑸𐑟𐑰; 𐑣𐑵 𐑒𐑮𐑬𐑗𐑦𐑙 𐑦𐑯 𐑞 𐑚𐑬, 𐑕𐑨𐑑 𐑢𐑪𐑗𐑦𐑙 𐑞 𐑖𐑸𐑒𐑕, 𐑞𐑨𐑑 𐑕𐑐𐑧𐑒𐑑𐑮𐑩𐑤𐑦 𐑐𐑤𐑱𐑛 𐑮𐑬𐑯𐑛 𐑞 𐑢𐑱𐑤, 𐑯 𐑑𐑨𐑐𐑑 𐑞 𐑤𐑲𐑑 𐑕𐑰𐑛𐑼 𐑐𐑤𐑨𐑙𐑒𐑕 𐑢𐑦𐑞 𐑞𐑺 𐑑𐑱𐑤𐑟. 𐑩 𐑕𐑬𐑯𐑛 𐑤𐑲𐑒 𐑞 𐑥𐑴𐑯𐑦𐑙 𐑦𐑯 𐑕𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑟 𐑴𐑝𐑼 ·𐑨𐑕𐑓𐑨𐑤𐑑𐑲𐑑𐑕 𐑝 𐑳𐑯𐑓𐑼𐑜𐑦𐑝𐑩𐑯 𐑜𐑴𐑕𐑑𐑕 𐑝 ·𐑜𐑩𐑥𐑹𐑩, 𐑮𐑨𐑯 𐑖𐑳𐑛𐑼𐑦𐑙 𐑔𐑮𐑵 𐑞 𐑺.

𐑕𐑑𐑸𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑕𐑤𐑳𐑥𐑚𐑼𐑟, ·𐑱𐑣𐑨𐑚, 𐑓𐑱𐑕 𐑑 𐑓𐑱𐑕, 𐑕𐑷 𐑞 ·𐑐𐑸𐑟𐑰; 𐑯 𐑣𐑵𐑐𐑑 𐑮𐑬𐑯𐑛 𐑚𐑲 𐑞 𐑜𐑤𐑵𐑥 𐑝 𐑞 𐑯𐑲𐑑 𐑞𐑱 𐑕𐑰𐑥𐑛 𐑞 𐑤𐑭𐑕𐑑 𐑥𐑧𐑯 𐑦𐑯 𐑩 𐑓𐑤𐑳𐑛𐑩𐑛 𐑢𐑻𐑤𐑛. “𐑲 𐑣𐑨𐑝 𐑛𐑮𐑰𐑥𐑛 𐑦𐑑 𐑩𐑜𐑱𐑯,” 𐑕𐑧𐑛 𐑣𐑰.

“𐑝 𐑞 𐑣𐑻𐑕𐑩𐑟? 𐑣𐑨𐑝 𐑲 𐑯𐑪𐑑 𐑕𐑧𐑛, 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑞𐑨𐑑 𐑯𐑲𐑞𐑼 𐑣𐑻𐑕 𐑯𐑹 𐑒𐑪𐑓𐑦𐑯 𐑒𐑨𐑯 𐑚𐑰 𐑞𐑲𐑯?”

“𐑯 𐑣𐑵 𐑸 𐑣𐑻𐑕𐑑 𐑞𐑨𐑑 𐑛𐑲 𐑪𐑯 𐑞 𐑕𐑰?”

“𐑚𐑳𐑑 𐑲 𐑕𐑧𐑛, 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑞𐑨𐑑 𐑺 𐑞𐑬 𐑒𐑫𐑛𐑕𐑑 𐑛𐑲 𐑪𐑯 𐑞𐑦𐑕 𐑝𐑶𐑦𐑡, 𐑑𐑵 𐑣𐑻𐑕𐑩𐑟 𐑥𐑳𐑕𐑑 𐑝𐑧𐑮𐑦𐑤𐑦 𐑚𐑰 𐑕𐑰𐑯 𐑚𐑲 𐑞𐑰 𐑪𐑯 𐑞 𐑕𐑰; 𐑞 𐑓𐑻𐑕𐑑 𐑯𐑪𐑑 𐑥𐑱𐑛 𐑚𐑲 𐑥𐑹𐑑𐑩𐑤 𐑣𐑨𐑯𐑛𐑟; 𐑯 𐑞 𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑢𐑫𐑛 𐑝 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑳𐑯 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑜𐑮𐑴𐑯 𐑦𐑯 ·𐑩𐑥𐑧𐑮𐑦𐑒𐑩.”

“𐑲, 𐑲! 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑲𐑑 𐑞𐑨𐑑, ·𐑐𐑸𐑟𐑰:—𐑩 𐑣𐑻𐑕 𐑯 𐑦𐑑𐑕 𐑐𐑤𐑵𐑥𐑟 𐑓𐑤𐑴𐑑𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑴𐑖𐑩𐑯 𐑢𐑦𐑞 𐑞 𐑢𐑱𐑝𐑟 𐑓 𐑞 𐑐𐑷𐑤-𐑚𐑺𐑼𐑟. 𐑣𐑭! 𐑕𐑳𐑗 𐑩 𐑕𐑲𐑑 𐑢𐑰 𐑖𐑨𐑤 𐑯𐑪𐑑 𐑕𐑵𐑯 𐑕𐑰.”

“𐑚𐑦𐑤𐑰𐑝 𐑦𐑑 𐑹 𐑯𐑪𐑑, 𐑞𐑬 𐑒𐑨𐑯𐑕𐑑 𐑯𐑪𐑑 𐑛𐑲 𐑑𐑦𐑤 𐑦𐑑 𐑚𐑰 𐑕𐑰𐑯, 𐑴𐑤𐑛 𐑥𐑨𐑯.”

“𐑯 𐑢𐑪𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑕𐑱𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞𐑲𐑕𐑧𐑤𐑓?”

“𐑞𐑴 𐑦𐑑 𐑒𐑳𐑥 𐑑 𐑞 𐑤𐑭𐑕𐑑, 𐑲 𐑖𐑨𐑤 𐑕𐑑𐑦𐑤 𐑜𐑴 𐑚𐑦𐑓𐑹 𐑞𐑰 𐑞𐑲 𐑐𐑲𐑤𐑩𐑑.”

“𐑯 𐑢𐑧𐑯 𐑞𐑬 𐑸𐑑 𐑕𐑴 𐑜𐑪𐑯 𐑚𐑦𐑓𐑹—𐑦𐑓 𐑞𐑨𐑑 𐑧𐑝𐑼 𐑚𐑦𐑓𐑷𐑤—𐑞𐑧𐑯 𐑺 𐑲 𐑒𐑨𐑯 𐑓𐑪𐑤𐑴, 𐑞𐑬 𐑥𐑳𐑕𐑑 𐑕𐑑𐑦𐑤 𐑩𐑐𐑽 𐑑 𐑥𐑰, 𐑑 𐑐𐑲𐑤𐑩𐑑 𐑥𐑰 𐑕𐑑𐑦𐑤?—𐑢𐑪𐑟 𐑦𐑑 𐑯𐑪𐑑 𐑕𐑴? 𐑢𐑧𐑤, 𐑞𐑧𐑯, 𐑛𐑦𐑛 𐑲 𐑚𐑦𐑤𐑰𐑝 𐑷𐑤 𐑘𐑰 𐑕𐑱, 𐑴 𐑥𐑲 𐑐𐑲𐑤𐑩𐑑! 𐑲 𐑣𐑨𐑝 𐑣𐑽 𐑑𐑵 𐑐𐑤𐑧𐑡𐑩𐑟 𐑞𐑨𐑑 𐑲 𐑖𐑨𐑤 𐑘𐑧𐑑 𐑕𐑤𐑱 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑯 𐑕𐑼𐑝𐑲𐑝 𐑦𐑑.”

“𐑑𐑱𐑒 𐑩𐑯𐑳𐑞𐑼 𐑐𐑤𐑧𐑡, 𐑴𐑤𐑛 𐑥𐑨𐑯,” 𐑕𐑧𐑛 𐑞 ·𐑐𐑸𐑟𐑰, 𐑨𐑟 𐑣𐑦𐑟 𐑲𐑟 𐑤𐑲𐑑𐑩𐑛 𐑳𐑐 𐑤𐑲𐑒 𐑓𐑲𐑼-𐑓𐑤𐑲𐑟 𐑦𐑯 𐑞 𐑜𐑤𐑵𐑥—“𐑣𐑧𐑥𐑐 𐑴𐑯𐑤𐑦 𐑒𐑨𐑯 𐑒𐑦𐑤 𐑞𐑰.”

“𐑞 𐑜𐑨𐑤𐑴𐑟, 𐑘𐑰 𐑥𐑰𐑯.—𐑲 𐑨𐑥 𐑦𐑥𐑹𐑑𐑩𐑤 𐑞𐑧𐑯, 𐑪𐑯 𐑤𐑨𐑯𐑛 𐑯 𐑪𐑯 𐑕𐑰,” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑢𐑦𐑞 𐑩 𐑤𐑭𐑓 𐑝 𐑛𐑦𐑮𐑦𐑠𐑩𐑯;—“𐑦𐑥𐑹𐑑𐑩𐑤 𐑪𐑯 𐑤𐑨𐑯𐑛 𐑯 𐑪𐑯 𐑕𐑰!”

𐑚𐑴𐑔 𐑢𐑻 𐑕𐑲𐑤𐑩𐑯𐑑 𐑩𐑜𐑱𐑯, 𐑨𐑟 𐑢𐑳𐑯 𐑥𐑨𐑯. 𐑞 𐑜𐑮𐑱 𐑛𐑷𐑯 𐑒𐑱𐑥 𐑪𐑯, 𐑯 𐑞 𐑕𐑤𐑳𐑥𐑚𐑼𐑦𐑙 𐑒𐑮𐑵 𐑼𐑴𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑑'𐑕 𐑚𐑪𐑑𐑩𐑥, 𐑯 𐑺 𐑯𐑵𐑯 𐑞 𐑛𐑧𐑛 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑚𐑮𐑷𐑑 𐑑 𐑞 𐑖𐑦𐑐.
