·𐑗𐑨𐑐𐑑𐑼 83. ·𐑡𐑴𐑯𐑩 ·𐑣𐑦𐑕𐑑𐑪𐑮𐑦𐑒𐑤𐑦 ·𐑮𐑦𐑜𐑸𐑛𐑩𐑛.

𐑮𐑧𐑓𐑼𐑩𐑯𐑕 𐑢𐑪𐑟 𐑥𐑱𐑛 𐑑 𐑞 𐑣𐑦𐑕𐑑𐑪𐑮𐑦𐑒𐑩𐑤 𐑕𐑑𐑹𐑦 𐑝 ·𐑡𐑴𐑯𐑩 𐑯 𐑞 𐑢𐑱𐑤 𐑦𐑯 𐑞 𐑐𐑮𐑦𐑕𐑰𐑛𐑦𐑙 𐑗𐑨𐑐𐑑𐑼. 𐑯𐑬 𐑕𐑳𐑥 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼𐑟 𐑮𐑭𐑞𐑼 𐑛𐑦𐑕𐑑𐑮𐑳𐑕𐑑 𐑞𐑦𐑕 𐑣𐑦𐑕𐑑𐑪𐑮𐑦𐑒𐑩𐑤 𐑕𐑑𐑹𐑦 𐑝 ·𐑡𐑴𐑯𐑩 𐑯 𐑞 𐑢𐑱𐑤. 𐑚𐑳𐑑 𐑞𐑧𐑯 𐑞𐑺 𐑢𐑻 𐑕𐑳𐑥 𐑕𐑒𐑧𐑐𐑑𐑦𐑒𐑩𐑤 ·𐑜𐑮𐑰𐑒𐑕 𐑯 ·𐑮𐑴𐑥𐑩𐑯𐑟, 𐑣𐑵, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑬𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑹𐑔𐑩𐑛𐑪𐑒𐑕 𐑐𐑱𐑜𐑩𐑯𐑟 𐑝 𐑞𐑺 𐑑𐑲𐑥𐑟, 𐑰𐑒𐑢𐑩𐑤𐑦 𐑛𐑬𐑑𐑩𐑛 𐑞 𐑕𐑑𐑹𐑦 𐑝 ·𐑣𐑻𐑒𐑿𐑤𐑰𐑟 𐑯 𐑞 𐑢𐑱𐑤, 𐑯 ·𐑸𐑾𐑯 𐑯 𐑞 𐑛𐑪𐑤𐑓𐑦𐑯; 𐑯 𐑘𐑧𐑑 𐑞𐑺 𐑛𐑬𐑑𐑦𐑙 𐑞𐑴𐑟 𐑑𐑮𐑩𐑛𐑦𐑖𐑩𐑯𐑟 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑥𐑱𐑒 𐑞𐑴𐑟 𐑑𐑮𐑩𐑛𐑦𐑖𐑩𐑯𐑟 𐑢𐑳𐑯 𐑢𐑦𐑑 𐑞 𐑤𐑧𐑕 𐑓𐑨𐑒𐑑𐑕, 𐑓 𐑷𐑤 𐑞𐑨𐑑.

𐑢𐑳𐑯 𐑴𐑤𐑛 ·𐑕𐑨𐑜-·𐑣𐑸𐑚𐑼 𐑢𐑱𐑤𐑥𐑩𐑯'𐑟 𐑗𐑰𐑓 𐑮𐑰𐑟𐑩𐑯 𐑓 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑦𐑙 𐑞 ·𐑣𐑰𐑚𐑮𐑵 𐑕𐑑𐑹𐑦 𐑢𐑪𐑟 𐑞𐑦𐑕:—𐑣𐑰 𐑣𐑨𐑛 𐑢𐑳𐑯 𐑝 𐑞𐑴𐑟 𐑒𐑢𐑱𐑯𐑑 𐑴𐑤𐑛-𐑓𐑨𐑖𐑩𐑯𐑛 ·𐑚𐑲𐑚𐑩𐑤𐑟, 𐑦𐑥𐑚𐑧𐑤𐑦𐑖𐑑 𐑢𐑦𐑞 𐑒𐑘𐑫𐑼𐑾𐑕, 𐑳𐑯𐑕𐑲𐑩𐑯𐑑𐑦𐑓𐑦𐑒 𐑐𐑤𐑱𐑑𐑕; 𐑢𐑳𐑯 𐑝 𐑢𐑦𐑗 𐑮𐑧𐑐𐑮𐑦𐑟𐑧𐑯𐑑𐑩𐑛 ·𐑡𐑴𐑯𐑩'𐑟 𐑢𐑱𐑤 𐑢𐑦𐑞 𐑑𐑵 𐑕𐑐𐑬𐑑𐑕 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛—𐑩 𐑐𐑦𐑒𐑿𐑤𐑦𐑨𐑮𐑦𐑑𐑦 𐑴𐑯𐑤𐑦 𐑑𐑮𐑵 𐑢𐑦𐑞 𐑮𐑦𐑕𐑐𐑧𐑒𐑑 𐑑 𐑩 𐑕𐑐𐑰𐑖𐑰𐑟 𐑝 𐑞 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 (𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤, 𐑯 𐑞 𐑝𐑼𐑲𐑩𐑑𐑦𐑟 𐑝 𐑞𐑨𐑑 𐑹𐑛𐑼), 𐑒𐑩𐑯𐑕𐑻𐑯𐑦𐑙 𐑢𐑦𐑗 𐑞 𐑓𐑦𐑖𐑼𐑥𐑧𐑯 𐑣𐑨𐑝 𐑞𐑦𐑕 𐑕𐑱𐑦𐑙, “𐑩 𐑐𐑧𐑯𐑦 𐑮𐑴𐑤 𐑢𐑫𐑛 𐑗𐑴𐑒 𐑣𐑦𐑥”; 𐑣𐑦𐑟 𐑕𐑢𐑪𐑤𐑴 𐑦𐑟 𐑕𐑴 𐑝𐑧𐑮𐑦 𐑕𐑥𐑷𐑤. 𐑚𐑳𐑑, 𐑑 𐑞𐑦𐑕, ·𐑚𐑦𐑖𐑩𐑐 ·𐑡𐑧𐑚'𐑟 𐑨𐑯𐑑𐑦𐑕𐑦𐑐𐑩𐑑𐑦𐑝 𐑭𐑯𐑕𐑼 𐑦𐑟 𐑮𐑧𐑛𐑦. 𐑦𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑯𐑧𐑕𐑦𐑕𐑺𐑦, 𐑣𐑦𐑯𐑑𐑕 𐑞 ·𐑚𐑦𐑖𐑩𐑐, 𐑞𐑨𐑑 𐑢𐑰 𐑒𐑩𐑯𐑕𐑦𐑛𐑼 ·𐑡𐑴𐑯𐑩 𐑨𐑟 𐑑𐑵𐑥𐑛 𐑦𐑯 𐑞 𐑢𐑱𐑤'𐑟 𐑚𐑧𐑤𐑦, 𐑚𐑳𐑑 𐑨𐑟 𐑑𐑧𐑥𐑐𐑼𐑺𐑦𐑤𐑦 𐑤𐑪𐑡𐑛 𐑦𐑯 𐑕𐑳𐑥 𐑐𐑸𐑑 𐑝 𐑣𐑦𐑟 𐑥𐑬𐑔. 𐑯 𐑞𐑦𐑕 𐑕𐑰𐑥𐑟 𐑮𐑰𐑟𐑩𐑯𐑩𐑚𐑩𐑤 𐑦𐑯𐑳𐑓 𐑦𐑯 𐑞 𐑜𐑫𐑛 ·𐑚𐑦𐑖𐑩𐑐. 𐑓 𐑑𐑮𐑵𐑤𐑦, 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑥𐑬𐑔 𐑢𐑫𐑛 𐑩𐑒𐑪𐑥𐑩𐑛𐑱𐑑 𐑩 𐑒𐑳𐑐𐑩𐑤 𐑝 𐑢𐑦𐑕𐑑-𐑑𐑱𐑚𐑩𐑤𐑟, 𐑯 𐑒𐑳𐑥𐑓𐑼𐑑𐑩𐑚𐑤𐑦 𐑕𐑰𐑑 𐑷𐑤 𐑞 𐑐𐑤𐑱𐑼𐑟. 𐑐𐑪𐑕𐑩𐑚𐑤𐑦, 𐑑𐑵, ·𐑡𐑴𐑯𐑩 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑦𐑯𐑕𐑒𐑪𐑯𐑕𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑦𐑯 𐑩 𐑣𐑪𐑤𐑴 𐑑𐑵𐑔; 𐑚𐑳𐑑, 𐑪𐑯 𐑕𐑧𐑒𐑩𐑯𐑛 𐑔𐑷𐑑𐑕, 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑦𐑟 𐑑𐑵𐑔𐑤𐑩𐑕.

𐑩𐑯𐑳𐑞𐑼 𐑮𐑰𐑟𐑩𐑯 𐑢𐑦𐑗 ·𐑕𐑨𐑜-·𐑣𐑸𐑚𐑼 (𐑣𐑰 𐑢𐑧𐑯𐑑 𐑚𐑲 𐑞𐑨𐑑 𐑯𐑱𐑥) 𐑻𐑡𐑛 𐑓 𐑣𐑦𐑟 𐑢𐑪𐑯𐑑 𐑝 𐑓𐑱𐑔 𐑦𐑯 𐑞𐑦𐑕 𐑥𐑨𐑑𐑼 𐑝 𐑞 𐑐𐑮𐑪𐑓𐑩𐑑, 𐑢𐑪𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑩𐑚𐑕𐑒𐑘𐑫𐑼𐑤𐑦 𐑦𐑯 𐑮𐑧𐑓𐑼𐑩𐑯𐑕 𐑑 𐑣𐑦𐑟 𐑦𐑯𐑒𐑸𐑕𐑼𐑱𐑑𐑩𐑛 𐑚𐑪𐑛𐑦 𐑯 𐑞 𐑢𐑱𐑤'𐑟 𐑜𐑨𐑕𐑑𐑮𐑦𐑒 𐑡𐑵𐑕𐑩𐑟. 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑩𐑚𐑡𐑧𐑒𐑖𐑩𐑯 𐑤𐑲𐑒𐑢𐑲𐑟 𐑓𐑷𐑤𐑟 𐑑 𐑞 𐑜𐑮𐑬𐑯𐑛, 𐑚𐑦𐑒𐑪𐑟 𐑩 ·𐑡𐑻𐑥𐑩𐑯 𐑧𐑒𐑕𐑦𐑡𐑰𐑑𐑦𐑕𐑑 𐑕𐑩𐑐𐑴𐑟𐑩𐑟 𐑞𐑨𐑑 ·𐑡𐑴𐑯𐑩 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑑𐑱𐑒𐑩𐑯 𐑮𐑧𐑓𐑿𐑡 𐑦𐑯 𐑞 𐑓𐑤𐑴𐑑𐑦𐑙 𐑚𐑪𐑛𐑦 𐑝 𐑩 𐑛𐑧𐑛 𐑢𐑱𐑤—𐑰𐑝𐑩𐑯 𐑨𐑟 𐑞 ·𐑓𐑮𐑧𐑯𐑗 𐑕𐑴𐑤𐑡𐑼𐑟 𐑦𐑯 𐑞 ·𐑮𐑳𐑖𐑩𐑯 𐑒𐑨𐑥𐑐𐑱𐑯 𐑑𐑻𐑯𐑛 𐑞𐑺 𐑛𐑧𐑛 𐑣𐑹𐑕𐑩𐑟 𐑦𐑯𐑑𐑵 𐑑𐑧𐑯𐑑𐑕, 𐑯 𐑒𐑮𐑷𐑤𐑛 𐑦𐑯𐑑𐑵 𐑞𐑧𐑥. 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑦𐑑 𐑣𐑨𐑟 𐑚𐑰𐑯 𐑛𐑦𐑝𐑲𐑯𐑛 𐑚𐑲 𐑳𐑞𐑼 𐑒𐑪𐑯𐑑𐑦𐑯𐑧𐑯𐑑𐑩𐑤 𐑒𐑪𐑥𐑩𐑯𐑑𐑱𐑑𐑼𐑟, 𐑞𐑨𐑑 𐑢𐑧𐑯 ·𐑡𐑴𐑯𐑩 𐑢𐑪𐑟 𐑔𐑮𐑴𐑯 𐑴𐑝𐑼𐑚𐑹𐑛 𐑓𐑮𐑪𐑥 𐑞 ·𐑡𐑪𐑐𐑩 𐑖𐑦𐑐, 𐑣𐑰 𐑕𐑑𐑮𐑱𐑑𐑢𐑱 𐑦𐑓𐑧𐑒𐑑𐑩𐑛 𐑣𐑦𐑟 𐑦𐑕𐑒𐑱𐑐 𐑑 𐑩𐑯𐑳𐑞𐑼 𐑝𐑧𐑕𐑩𐑤 𐑯𐑽 𐑚𐑲, 𐑕𐑳𐑥 𐑝𐑧𐑕𐑩𐑤 𐑢𐑦𐑞 𐑩 𐑢𐑱𐑤 𐑓 𐑩 𐑓𐑦𐑜𐑼-𐑣𐑧𐑛; 𐑯, 𐑲 𐑢𐑫𐑛 𐑨𐑛, 𐑐𐑪𐑕𐑩𐑚𐑤𐑦 𐑒𐑷𐑤𐑛 “𐑞 ·𐑢𐑱𐑤,” 𐑨𐑟 𐑕𐑳𐑥 𐑒𐑮𐑭𐑓𐑑 𐑸 𐑯𐑬𐑩𐑛𐑱𐑟 𐑒𐑮𐑦𐑕𐑩𐑯𐑛 𐑞 “𐑖𐑸𐑒,” 𐑞 “𐑜𐑳𐑤,” 𐑞 “𐑰𐑜𐑩𐑤.” 𐑯𐑹 𐑣𐑨𐑝 𐑞𐑺 𐑚𐑰𐑯 𐑢𐑪𐑯𐑑𐑦𐑙 𐑤𐑻𐑯𐑛 𐑧𐑒𐑕𐑦𐑡𐑰𐑑𐑦𐑕𐑑𐑕 𐑣𐑵 𐑣𐑨𐑝 𐑩𐑐𐑲𐑯𐑛 𐑞𐑨𐑑 𐑞 𐑢𐑱𐑤 𐑥𐑧𐑯𐑖𐑩𐑯𐑛 𐑦𐑯 𐑞 𐑚𐑫𐑒 𐑝 ·𐑡𐑴𐑯𐑩 𐑥𐑽𐑤𐑦 𐑥𐑧𐑯𐑑 𐑩 𐑤𐑲𐑓-𐑐𐑮𐑦𐑟𐑻𐑝𐑼—𐑩𐑯 𐑦𐑯𐑓𐑤𐑱𐑑𐑩𐑛 𐑚𐑨𐑜 𐑝 𐑢𐑦𐑯𐑛—𐑢𐑦𐑗 𐑞 𐑦𐑯𐑛𐑱𐑯𐑡𐑼𐑛 𐑐𐑮𐑪𐑓𐑩𐑑 𐑕𐑢𐑨𐑥 𐑑, 𐑯 𐑕𐑴 𐑢𐑪𐑟 𐑕𐑱𐑝𐑛 𐑓𐑮𐑪𐑥 𐑩 𐑢𐑷𐑑𐑼𐑦 𐑛𐑵𐑥. 𐑐𐑫𐑼 ·𐑕𐑨𐑜-·𐑣𐑸𐑚𐑼, 𐑞𐑺𐑓𐑹, 𐑕𐑰𐑥𐑟 𐑢𐑫𐑕𐑑𐑩𐑛 𐑷𐑤 𐑮𐑬𐑯𐑛. 𐑚𐑳𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑕𐑑𐑦𐑤 𐑩𐑯𐑳𐑞𐑼 𐑮𐑰𐑟𐑩𐑯 𐑓 𐑣𐑦𐑟 𐑢𐑪𐑯𐑑 𐑝 𐑓𐑱𐑔. 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑦𐑕, 𐑦𐑓 𐑲 𐑮𐑦𐑥𐑧𐑥𐑚𐑼 𐑮𐑲𐑑: ·𐑡𐑴𐑯𐑩 𐑢𐑪𐑟 𐑕𐑢𐑪𐑤𐑴𐑛 𐑚𐑲 𐑞 𐑢𐑱𐑤 𐑦𐑯 𐑞 ·𐑥𐑧𐑛𐑦𐑑𐑼𐑱𐑯𐑾𐑯 ·𐑕𐑰, 𐑯 𐑭𐑓𐑑𐑼 𐑔𐑮𐑰 𐑛𐑱𐑟 𐑣𐑰 𐑢𐑪𐑟 𐑝𐑪𐑥𐑦𐑑𐑩𐑛 𐑳𐑐 𐑕𐑳𐑥𐑢𐑺 𐑢𐑦𐑞𐑦𐑯 𐑔𐑮𐑰 𐑛𐑱𐑟' 𐑡𐑻𐑯𐑦 𐑝 ·𐑯𐑦𐑯𐑩𐑝𐑩, 𐑩 𐑕𐑦𐑑𐑦 𐑪𐑯 𐑞 ·𐑑𐑲𐑜𐑮𐑦𐑕, 𐑝𐑧𐑮𐑦 𐑥𐑳𐑗 𐑥𐑹 𐑞𐑨𐑯 𐑔𐑮𐑰 𐑛𐑱𐑟' 𐑡𐑻𐑯𐑦 𐑩𐑒𐑮𐑪𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑯𐑽𐑩𐑕𐑑 𐑐𐑶𐑯𐑑 𐑝 𐑞 ·𐑥𐑧𐑛𐑦𐑑𐑼𐑱𐑯𐑾𐑯 𐑒𐑴𐑕𐑑. 𐑣𐑬 𐑦𐑟 𐑞𐑨𐑑?

𐑚𐑳𐑑 𐑢𐑪𐑟 𐑞𐑺 𐑯𐑴 𐑳𐑞𐑼 𐑢𐑱 𐑓 𐑞 𐑢𐑱𐑤 𐑑 𐑤𐑨𐑯𐑛 𐑞 𐑐𐑮𐑪𐑓𐑩𐑑 𐑢𐑦𐑞𐑦𐑯 𐑞𐑨𐑑 𐑖𐑹𐑑 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑝 ·𐑯𐑦𐑯𐑩𐑝𐑩? 𐑘𐑧𐑕. 𐑣𐑰 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑒𐑨𐑮𐑦𐑛 𐑣𐑦𐑥 𐑮𐑬𐑯𐑛 𐑚𐑲 𐑞 𐑢𐑱 𐑝 𐑞 ·𐑒𐑱𐑐 𐑝 𐑜𐑫𐑛 ·𐑣𐑴𐑐. 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑑 𐑕𐑐𐑰𐑒 𐑝 𐑞 𐑐𐑨𐑕𐑦𐑡 𐑔𐑮𐑵 𐑞 𐑣𐑴𐑤 𐑤𐑧𐑙𐑔 𐑝 𐑞 ·𐑥𐑧𐑛𐑦𐑑𐑼𐑱𐑯𐑾𐑯, 𐑯 𐑩𐑯𐑳𐑞𐑼 𐑐𐑨𐑕𐑦𐑡 𐑳𐑐 𐑞 ·𐑐𐑻𐑖𐑩𐑯 ·𐑜𐑳𐑤𐑓 𐑯 ·𐑮𐑧𐑛 ·𐑕𐑰, 𐑕𐑳𐑗 𐑩 𐑕𐑳𐑐𐑩𐑟𐑦𐑖𐑩𐑯 𐑢𐑫𐑛 𐑦𐑯𐑝𐑪𐑤𐑝 𐑞 𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑕𐑻𐑒𐑩𐑥𐑯𐑨𐑝𐑦𐑜𐑱𐑖𐑩𐑯 𐑝 𐑷𐑤 ·𐑨𐑓𐑮𐑦𐑒𐑩 𐑦𐑯 𐑔𐑮𐑰 𐑛𐑱𐑟, 𐑯𐑪𐑑 𐑑 𐑕𐑐𐑰𐑒 𐑝 𐑞 ·𐑑𐑲𐑜𐑮𐑦𐑕 𐑢𐑷𐑑𐑼𐑟, 𐑯𐑽 𐑞 𐑕𐑲𐑑 𐑝 ·𐑯𐑦𐑯𐑩𐑝𐑩, 𐑚𐑰𐑦𐑙 𐑑𐑵 𐑖𐑨𐑤𐑴 𐑓 𐑧𐑯𐑦 𐑢𐑱𐑤 𐑑 𐑕𐑢𐑦𐑥 𐑦𐑯. 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑞𐑦𐑕 𐑲𐑛𐑾 𐑝 ·𐑡𐑴𐑯𐑩'𐑟 𐑢𐑧𐑞𐑼𐑦𐑙 𐑞 ·𐑒𐑱𐑐 𐑝 𐑜𐑫𐑛 ·𐑣𐑴𐑐 𐑨𐑑 𐑕𐑴 𐑻𐑤𐑦 𐑩 𐑛𐑱 𐑢𐑫𐑛 𐑮𐑧𐑕𐑑 𐑞 𐑪𐑯𐑼 𐑝 𐑞 𐑛𐑦𐑕𐑒𐑳𐑝𐑼𐑦 𐑝 𐑞𐑨𐑑 𐑜𐑮𐑱𐑑 𐑣𐑧𐑛𐑤𐑩𐑯𐑛 𐑓𐑮𐑪𐑥 ·𐑚𐑸𐑔𐑪𐑤𐑩𐑥𐑿 ·𐑛𐑾𐑟, 𐑦𐑑𐑕 𐑮𐑦𐑐𐑿𐑑𐑩𐑛 𐑛𐑦𐑕𐑒𐑳𐑝𐑼𐑼, 𐑯 𐑕𐑴 𐑥𐑱𐑒 𐑥𐑪𐑛𐑼𐑯 𐑣𐑦𐑕𐑑𐑼𐑦 𐑩 𐑤𐑲𐑼.
