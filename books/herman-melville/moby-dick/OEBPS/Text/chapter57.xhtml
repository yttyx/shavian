·𐑗𐑨𐑐𐑑𐑼 57. 𐑝 ·𐑢𐑱𐑤𐑟 𐑦𐑯 ·𐑐𐑱𐑯𐑑; 𐑦𐑯 ·𐑑𐑰𐑔; 𐑦𐑯 ·𐑢𐑫𐑛; 𐑦𐑯 ·𐑖𐑰𐑑-·𐑲𐑼𐑯; 𐑦𐑯 ·𐑕𐑑𐑴𐑯; 𐑦𐑯 ·𐑥𐑬𐑯𐑑𐑩𐑯𐑟; 𐑦𐑯 ·𐑕𐑑𐑸𐑟.

𐑪𐑯 ·𐑑𐑬𐑼-𐑣𐑦𐑤, 𐑨𐑟 𐑿 𐑜𐑴 𐑛𐑬𐑯 𐑑 𐑞 ·𐑤𐑳𐑯𐑛𐑩𐑯 𐑛𐑪𐑒𐑕, 𐑿 𐑥𐑱 𐑣𐑨𐑝 𐑕𐑰𐑯 𐑩 𐑒𐑮𐑦𐑐𐑩𐑤𐑛 𐑚𐑧𐑜𐑼 (𐑹 𐑒𐑧𐑡𐑩, 𐑨𐑟 𐑞 𐑕𐑱𐑤𐑼𐑟 𐑕𐑱) 𐑣𐑴𐑤𐑛𐑦𐑙 𐑩 𐑐𐑱𐑯𐑑𐑩𐑛 𐑚𐑹𐑛 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥, 𐑮𐑧𐑐𐑮𐑦𐑟𐑧𐑯𐑑𐑦𐑙 𐑞 𐑑𐑮𐑨𐑡𐑦𐑒 𐑕𐑰𐑯 𐑦𐑯 𐑢𐑦𐑗 𐑣𐑰 𐑤𐑪𐑕𐑑 𐑣𐑦𐑟 𐑤𐑧𐑜. 𐑞𐑺 𐑸 𐑔𐑮𐑰 𐑢𐑱𐑤𐑟 𐑯 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕; 𐑯 𐑢𐑳𐑯 𐑝 𐑞 𐑚𐑴𐑑𐑕 (𐑐𐑮𐑦𐑟𐑿𐑥𐑛 𐑑 𐑒𐑩𐑯𐑑𐑱𐑯 𐑞 𐑥𐑦𐑕𐑦𐑙 𐑤𐑧𐑜 𐑦𐑯 𐑷𐑤 𐑦𐑑𐑕 𐑼𐑦𐑡𐑦𐑯𐑩𐑤 𐑦𐑯𐑑𐑧𐑜𐑮𐑦𐑑𐑦) 𐑦𐑟 𐑚𐑰𐑦𐑙 𐑒𐑮𐑳𐑯𐑗𐑑 𐑚𐑲 𐑞 𐑡𐑷𐑟 𐑝 𐑞 𐑓𐑹𐑥𐑴𐑕𐑑 𐑢𐑱𐑤. 𐑧𐑯𐑦 𐑑𐑲𐑥 𐑞𐑰𐑟 𐑑𐑧𐑯 𐑘𐑽𐑟, 𐑞𐑱 𐑑𐑧𐑤 𐑥𐑰, 𐑣𐑨𐑟 𐑞𐑨𐑑 𐑥𐑨𐑯 𐑣𐑧𐑤𐑛 𐑳𐑐 𐑞𐑨𐑑 𐑐𐑦𐑒𐑗𐑼, 𐑯 𐑦𐑜𐑟𐑦𐑚𐑦𐑑𐑩𐑛 𐑞𐑨𐑑 𐑕𐑑𐑳𐑥𐑐 𐑑 𐑩𐑯 𐑦𐑯𐑒𐑮𐑧𐑡𐑵𐑤𐑩𐑕 𐑢𐑻𐑤𐑛. 𐑚𐑳𐑑 𐑞 𐑑𐑲𐑥 𐑝 𐑣𐑦𐑟 𐑡𐑳𐑕𐑑𐑦𐑓𐑦𐑒𐑱𐑖𐑩𐑯 𐑣𐑨𐑟 𐑯𐑬 𐑒𐑳𐑥. 𐑣𐑦𐑟 𐑔𐑮𐑰 𐑢𐑱𐑤𐑟 𐑸 𐑨𐑟 𐑜𐑫𐑛 𐑢𐑱𐑤𐑟 𐑨𐑟 𐑢𐑻 𐑧𐑝𐑼 𐑐𐑳𐑚𐑤𐑦𐑖𐑑 𐑦𐑯 ·𐑢𐑪𐑐𐑦𐑙, 𐑨𐑑 𐑧𐑯𐑦 𐑮𐑱𐑑; 𐑯 𐑣𐑦𐑟 𐑕𐑑𐑳𐑥𐑐 𐑨𐑟 𐑳𐑯𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑩𐑚𐑩𐑤 𐑩 𐑕𐑑𐑳𐑥𐑐 𐑨𐑟 𐑧𐑯𐑦 𐑿 𐑢𐑦𐑤 𐑓𐑲𐑯𐑛 𐑦𐑯 𐑞 𐑢𐑧𐑕𐑑𐑼𐑯 𐑒𐑤𐑽𐑦𐑙𐑟. 𐑚𐑳𐑑, 𐑞𐑴 𐑓 𐑧𐑝𐑼 𐑥𐑬𐑯𐑑𐑩𐑛 𐑪𐑯 𐑞𐑨𐑑 𐑕𐑑𐑳𐑥𐑐, 𐑯𐑧𐑝𐑼 𐑩 𐑕𐑑𐑳𐑥𐑐-𐑕𐑐𐑰𐑗 𐑛𐑳𐑟 𐑞 𐑐𐑫𐑼 𐑢𐑱𐑤𐑥𐑩𐑯 𐑥𐑱𐑒; 𐑚𐑳𐑑, 𐑢𐑦𐑞 𐑛𐑬𐑯𐑒𐑭𐑕𐑑 𐑲𐑟, 𐑕𐑑𐑨𐑯𐑛𐑟 𐑮𐑵𐑓𐑩𐑤𐑦 𐑒𐑪𐑯𐑑𐑩𐑥𐑐𐑤𐑱𐑑𐑦𐑙 𐑣𐑦𐑟 𐑴𐑯 𐑨𐑥𐑐𐑿𐑑𐑱𐑖𐑩𐑯.

𐑔𐑮𐑵𐑬𐑑 𐑞 ·𐑐𐑩𐑕𐑦𐑓𐑦𐑒, 𐑯 𐑷𐑤𐑕𐑴 𐑦𐑯 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑, 𐑯 ·𐑯𐑿 ·𐑚𐑧𐑛𐑓𐑼𐑛, 𐑯 ·𐑕𐑨𐑜 ·𐑣𐑸𐑚𐑼, 𐑿 𐑢𐑦𐑤 𐑒𐑳𐑥 𐑩𐑒𐑮𐑪𐑕 𐑤𐑲𐑝𐑤𐑦 𐑕𐑒𐑧𐑗𐑩𐑟 𐑝 𐑢𐑱𐑤𐑟 𐑯 𐑢𐑱𐑤𐑦𐑙-𐑕𐑰𐑯𐑟, 𐑜𐑮𐑱𐑝𐑩𐑯 𐑚𐑲 𐑞 𐑓𐑦𐑖𐑼𐑥𐑧𐑯 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑪𐑯 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤-𐑑𐑰𐑔, 𐑹 𐑤𐑱𐑛𐑦𐑟' 𐑚𐑳𐑕𐑒𐑕 𐑮𐑷𐑑 𐑬𐑑 𐑝 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤-𐑚𐑴𐑯, 𐑯 𐑳𐑞𐑼 𐑤𐑲𐑒 𐑕𐑒𐑮𐑦𐑥𐑟𐑣𐑨𐑯𐑛𐑼 𐑸𐑑𐑦𐑒𐑩𐑤𐑟, 𐑨𐑟 𐑞 𐑢𐑱𐑤𐑥𐑧𐑯 𐑒𐑷𐑤 𐑞 𐑯𐑿𐑥𐑼𐑩𐑕 𐑤𐑦𐑑𐑩𐑤 𐑦𐑯𐑡𐑰𐑯𐑾𐑕 𐑒𐑩𐑯𐑑𐑮𐑲𐑝𐑩𐑯𐑕𐑩𐑟 𐑞𐑱 𐑦𐑤𐑨𐑚𐑼𐑩𐑑𐑤𐑦 𐑒𐑸𐑝 𐑬𐑑 𐑝 𐑞 𐑮𐑳𐑓 𐑥𐑩𐑑𐑽𐑾𐑤, 𐑦𐑯 𐑞𐑺 𐑬𐑼𐑟 𐑝 𐑴𐑖𐑩𐑯 𐑤𐑧𐑠𐑼. 𐑕𐑳𐑥 𐑝 𐑞𐑧𐑥 𐑣𐑨𐑝 𐑤𐑦𐑑𐑩𐑤 𐑚𐑪𐑒𐑕𐑩𐑟 𐑝 𐑛𐑧𐑯𐑑𐑦𐑕𐑑𐑦𐑒𐑩𐑤-𐑤𐑫𐑒𐑦𐑙 𐑦𐑥𐑐𐑤𐑦𐑥𐑧𐑯𐑑𐑕, 𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑦𐑯𐑑𐑧𐑯𐑛𐑩𐑛 𐑓 𐑞 𐑕𐑒𐑮𐑦𐑥𐑟𐑣𐑨𐑯𐑛𐑼𐑦𐑙 𐑚𐑦𐑟𐑯𐑩𐑕. 𐑚𐑳𐑑, 𐑦𐑯 𐑡𐑧𐑯𐑼𐑩𐑤, 𐑞𐑱 𐑑𐑶𐑤 𐑢𐑦𐑞 𐑞𐑺 𐑡𐑨𐑒-𐑯𐑲𐑝𐑟 𐑩𐑤𐑴𐑯; 𐑯, 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑪𐑥𐑯𐑦𐑐𐑩𐑑𐑩𐑯𐑑 𐑑𐑵𐑤 𐑝 𐑞 𐑕𐑱𐑤𐑼, 𐑞𐑱 𐑢𐑦𐑤 𐑑𐑻𐑯 𐑿 𐑬𐑑 𐑧𐑯𐑦𐑔𐑦𐑙 𐑿 𐑐𐑤𐑰𐑟, 𐑦𐑯 𐑞 𐑢𐑱 𐑝 𐑩 𐑥𐑨𐑮𐑦𐑯𐑼'𐑟 𐑓𐑨𐑯𐑕𐑦.

𐑤𐑪𐑙 𐑧𐑒𐑕𐑲𐑤 𐑓𐑮𐑪𐑥 ·𐑒𐑮𐑦𐑕𐑩𐑯𐑛𐑩𐑥 𐑯 𐑕𐑦𐑝𐑦𐑤𐑲𐑟𐑱𐑖𐑩𐑯 𐑦𐑯𐑧𐑝𐑦𐑑𐑩𐑚𐑤𐑦 𐑮𐑦𐑕𐑑𐑹𐑟 𐑩 𐑥𐑨𐑯 𐑑 𐑞𐑨𐑑 𐑒𐑩𐑯𐑛𐑦𐑖𐑩𐑯 𐑦𐑯 𐑢𐑦𐑗 ·𐑜𐑪𐑛 𐑐𐑤𐑱𐑕𐑑 𐑣𐑦𐑥, 𐑲𐑰 𐑢𐑪𐑑 𐑦𐑟 𐑒𐑷𐑤𐑛 𐑕𐑨𐑝𐑦𐑡𐑮𐑦. 𐑘𐑹 𐑑𐑮𐑵 𐑢𐑱𐑤-𐑣𐑳𐑯𐑑𐑼 𐑦𐑟 𐑨𐑟 𐑥𐑳𐑗 𐑩 𐑕𐑨𐑝𐑦𐑡 𐑨𐑟 𐑩𐑯 ·𐑽𐑩𐑒𐑢𐑭. 𐑲 𐑥𐑲𐑕𐑧𐑤𐑓 𐑨𐑥 𐑩 𐑕𐑨𐑝𐑦𐑡, 𐑴𐑯𐑦𐑙 𐑯𐑴 𐑩𐑤𐑰𐑡𐑩𐑯𐑕 𐑚𐑳𐑑 𐑑 𐑞 ·𐑒𐑦𐑙 𐑝 𐑞 ·𐑒𐑨𐑯𐑦𐑚𐑩𐑤𐑟; 𐑯 𐑮𐑧𐑛𐑦 𐑨𐑑 𐑧𐑯𐑦 𐑥𐑴𐑥𐑧𐑯𐑑 𐑑 𐑮𐑦𐑚𐑧𐑤 𐑩𐑜𐑱𐑯𐑕𐑑 𐑣𐑦𐑥.

𐑯𐑬, 𐑢𐑳𐑯 𐑝 𐑞 𐑐𐑦𐑒𐑿𐑤𐑽 𐑒𐑨𐑮𐑩𐑒𐑑𐑼𐑦𐑕𐑑𐑦𐑒𐑕 𐑝 𐑞 𐑕𐑨𐑝𐑦𐑡 𐑦𐑯 𐑣𐑦𐑟 𐑛𐑩𐑥𐑧𐑕𐑑𐑦𐑒 𐑬𐑼𐑟, 𐑦𐑟 𐑣𐑦𐑟 𐑢𐑳𐑯𐑛𐑼𐑓𐑩𐑤 𐑐𐑱𐑖𐑩𐑯𐑕 𐑝 𐑦𐑯𐑛𐑳𐑕𐑑𐑮𐑦. 𐑩𐑯 𐑱𐑯𐑗𐑩𐑯𐑑 ·𐑣𐑩𐑢𐑲𐑩𐑯 𐑢𐑹-𐑒𐑤𐑳𐑚 𐑹 𐑕𐑐𐑽-𐑐𐑨𐑛𐑩𐑤, 𐑦𐑯 𐑦𐑑𐑕 𐑓𐑫𐑤 𐑥𐑳𐑤𐑑𐑦𐑐𐑤𐑦𐑕𐑦𐑑𐑦 𐑯 𐑦𐑤𐑨𐑚𐑼𐑱𐑖𐑩𐑯 𐑝 𐑒𐑸𐑝𐑦𐑙, 𐑦𐑟 𐑨𐑟 𐑜𐑮𐑱𐑑 𐑩 𐑑𐑮𐑴𐑓𐑦 𐑝 𐑣𐑿𐑥𐑩𐑯 𐑐𐑻𐑕𐑩𐑝𐑽𐑩𐑯𐑕 𐑨𐑟 𐑩 ·𐑤𐑨𐑑𐑦𐑯 𐑤𐑧𐑒𐑕𐑦𐑒𐑪𐑯. 𐑓, 𐑢𐑦𐑞 𐑚𐑳𐑑 𐑩 𐑚𐑦𐑑 𐑝 𐑚𐑮𐑴𐑒𐑩𐑯 𐑕𐑰-𐑖𐑧𐑤 𐑹 𐑩 𐑖𐑸𐑒'𐑕 𐑑𐑵𐑔, 𐑞𐑨𐑑 𐑥𐑦𐑮𐑨𐑒𐑿𐑤𐑩𐑕 𐑦𐑯𐑑𐑮𐑦𐑒𐑩𐑕𐑦 𐑝 𐑢𐑫𐑛𐑩𐑯 𐑯𐑧𐑑-𐑢𐑻𐑒 𐑣𐑨𐑟 𐑚𐑰𐑯 𐑩𐑗𐑰𐑝𐑛; 𐑯 𐑦𐑑 𐑣𐑨𐑟 𐑒𐑪𐑕𐑑 𐑕𐑑𐑧𐑛𐑦 𐑘𐑽𐑟 𐑝 𐑕𐑑𐑧𐑛𐑦 𐑨𐑐𐑤𐑦𐑒𐑱𐑖𐑩𐑯.

𐑨𐑟 𐑢𐑦𐑞 𐑞 ·𐑣𐑩𐑢𐑲𐑩𐑯 𐑕𐑨𐑝𐑦𐑡, 𐑕𐑴 𐑢𐑦𐑞 𐑞 𐑢𐑲𐑑 𐑕𐑱𐑤𐑼-𐑕𐑨𐑝𐑦𐑡. 𐑢𐑦𐑞 𐑞 𐑕𐑱𐑥 𐑥𐑸𐑝𐑩𐑤𐑩𐑕 𐑐𐑱𐑖𐑩𐑯𐑕, 𐑯 𐑢𐑦𐑞 𐑞 𐑕𐑱𐑥 𐑕𐑦𐑙𐑜𐑩𐑤 𐑖𐑸𐑒'𐑕 𐑑𐑵𐑔, 𐑝 𐑣𐑦𐑟 𐑢𐑳𐑯 𐑐𐑫𐑼 𐑡𐑨𐑒-𐑯𐑲𐑓, 𐑣𐑰 𐑢𐑦𐑤 𐑒𐑸𐑝 𐑿 𐑩 𐑚𐑦𐑑 𐑝 𐑚𐑴𐑯 𐑕𐑒𐑳𐑤𐑐𐑗𐑼, 𐑯𐑪𐑑 𐑒𐑢𐑲𐑑 𐑨𐑟 𐑢𐑻𐑒𐑥𐑩𐑯𐑤𐑲𐑒, 𐑚𐑳𐑑 𐑨𐑟 𐑒𐑤𐑴𐑕 𐑐𐑨𐑒𐑑 𐑦𐑯 𐑦𐑑𐑕 𐑥𐑱𐑟𐑦𐑯𐑩𐑕 𐑝 𐑛𐑦𐑟𐑲𐑯, 𐑨𐑟 𐑞 ·𐑜𐑮𐑰𐑒 𐑕𐑨𐑝𐑦𐑡, ·𐑩𐑒𐑦𐑤𐑰𐑟'𐑩𐑟 𐑖𐑰𐑤𐑛; 𐑯 𐑓𐑫𐑤 𐑝 𐑚𐑸𐑚𐑨𐑮𐑦𐑒 𐑕𐑐𐑦𐑮𐑦𐑑 𐑯 𐑕𐑩𐑡𐑧𐑕𐑑𐑦𐑝𐑯𐑩𐑕, 𐑨𐑟 𐑞 𐑐𐑮𐑦𐑯𐑑𐑕 𐑝 𐑞𐑨𐑑 𐑓𐑲𐑯 𐑴𐑤𐑛 ·𐑛𐑳𐑗 𐑕𐑨𐑝𐑦𐑡, ·𐑨𐑤𐑚𐑼𐑑 ·𐑛𐑘𐑫𐑼𐑼.

𐑢𐑫𐑛𐑩𐑯 𐑢𐑱𐑤𐑟, 𐑹 𐑢𐑱𐑤𐑟 𐑒𐑳𐑑 𐑦𐑯 𐑐𐑮𐑴𐑓𐑲𐑤 𐑬𐑑 𐑝 𐑞 𐑕𐑥𐑷𐑤 𐑛𐑸𐑒 𐑕𐑤𐑨𐑚𐑟 𐑝 𐑞 𐑯𐑴𐑚𐑩𐑤 ·𐑕𐑬𐑔 ·𐑕𐑰 𐑢𐑹-𐑢𐑫𐑛, 𐑸 𐑓𐑮𐑰𐑒𐑢𐑩𐑯𐑑𐑤𐑦 𐑥𐑧𐑑 𐑢𐑦𐑞 𐑦𐑯 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤𐑟 𐑝 ·𐑩𐑥𐑧𐑮𐑦𐑒𐑩𐑯 𐑢𐑱𐑤𐑼𐑟. 𐑕𐑳𐑥 𐑝 𐑞𐑧𐑥 𐑸 𐑛𐑳𐑯 𐑢𐑦𐑞 𐑥𐑳𐑗 𐑨𐑒𐑘𐑼𐑩𐑕𐑦.

𐑨𐑑 𐑕𐑳𐑥 𐑴𐑤𐑛 𐑜𐑱𐑚𐑩𐑤-𐑮𐑵𐑓𐑑 𐑒𐑳𐑯𐑑𐑮𐑦 𐑣𐑬𐑟𐑩𐑟 𐑿 𐑢𐑦𐑤 𐑕𐑰 𐑚𐑮𐑭𐑕 𐑢𐑱𐑤𐑟 𐑣𐑳𐑙 𐑚𐑲 𐑞 𐑑𐑱𐑤 𐑓 𐑯𐑪𐑒𐑼𐑟 𐑑 𐑞 𐑮𐑴𐑛-𐑕𐑲𐑛 𐑛𐑹. 𐑢𐑧𐑯 𐑞 𐑐𐑹𐑑𐑼 𐑦𐑟 𐑕𐑤𐑰𐑐𐑦, 𐑞 𐑨𐑯𐑝𐑦𐑤-𐑣𐑧𐑛𐑩𐑛 𐑢𐑱𐑤 𐑢𐑫𐑛 𐑚𐑰 𐑚𐑧𐑕𐑑. 𐑚𐑳𐑑 𐑞𐑰𐑟 𐑯𐑪𐑒𐑦𐑙 𐑢𐑱𐑤𐑟 𐑸 𐑕𐑧𐑤𐑛𐑩𐑥 𐑮𐑦𐑥𐑸𐑒𐑩𐑚𐑩𐑤 𐑨𐑟 𐑓𐑱𐑔𐑓𐑩𐑤 𐑧𐑕𐑱𐑟. 𐑪𐑯 𐑞 𐑕𐑐𐑲𐑼𐑟 𐑝 𐑕𐑳𐑥 𐑴𐑤𐑛-𐑓𐑨𐑖𐑩𐑯𐑛 𐑗𐑻𐑗𐑩𐑟 𐑿 𐑢𐑦𐑤 𐑕𐑰 𐑖𐑰𐑑-𐑲𐑼𐑯 𐑢𐑱𐑤𐑟 𐑐𐑤𐑱𐑕𐑑 𐑞𐑺 𐑓 𐑢𐑧𐑞𐑼-𐑒𐑪𐑒𐑕; 𐑚𐑳𐑑 𐑞𐑱 𐑸 𐑕𐑴 𐑧𐑤𐑦𐑝𐑱𐑑𐑩𐑛, 𐑯 𐑚𐑦𐑕𐑲𐑛𐑟 𐑞𐑨𐑑 𐑸 𐑑 𐑷𐑤 𐑦𐑯𐑑𐑧𐑯𐑑𐑕 𐑯 𐑐𐑻𐑐𐑩𐑕𐑩𐑟 𐑕𐑴 𐑤𐑱𐑚𐑩𐑤𐑛 𐑢𐑦𐑞 “𐑣𐑨𐑯𐑛𐑟 𐑪𐑓!” 𐑿 𐑒𐑨𐑯𐑪𐑑 𐑦𐑜𐑟𐑨𐑥𐑦𐑯 𐑞𐑧𐑥 𐑒𐑤𐑴𐑕𐑤𐑦 𐑦𐑯𐑳𐑓 𐑑 𐑛𐑦𐑕𐑲𐑛 𐑩𐑐𐑪𐑯 𐑞𐑺 𐑥𐑧𐑮𐑦𐑑.

𐑦𐑯 𐑚𐑴𐑯𐑦, 𐑮𐑦𐑚𐑦 𐑮𐑰𐑡𐑩𐑯𐑟 𐑝 𐑞 𐑻𐑔, 𐑢𐑺 𐑨𐑑 𐑞 𐑚𐑱𐑕 𐑝 𐑣𐑲 𐑚𐑮𐑴𐑒𐑩𐑯 𐑒𐑤𐑦𐑓𐑕 𐑥𐑨𐑕𐑩𐑟 𐑝 𐑮𐑪𐑒 𐑤𐑲 𐑕𐑑𐑮𐑵𐑯 𐑦𐑯 𐑓𐑨𐑯𐑑𐑨𐑕𐑑𐑦𐑒 𐑜𐑮𐑵𐑐𐑦𐑙𐑟 𐑩𐑐𐑪𐑯 𐑞 𐑐𐑤𐑱𐑯, 𐑿 𐑢𐑦𐑤 𐑪𐑓𐑩𐑯 𐑛𐑦𐑕𐑒𐑳𐑝𐑼 𐑦𐑥𐑦𐑡𐑩𐑟 𐑨𐑟 𐑝 𐑞 𐑐𐑧𐑑𐑮𐑦𐑓𐑲𐑛 𐑓𐑹𐑥𐑟 𐑝 𐑞 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑐𐑸𐑑𐑤𐑦 𐑥𐑻𐑡𐑛 𐑦𐑯 𐑜𐑮𐑭𐑕, 𐑢𐑦𐑗 𐑝 𐑩 𐑢𐑦𐑯𐑛𐑦 𐑛𐑱 𐑚𐑮𐑱𐑒𐑕 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞𐑧𐑥 𐑦𐑯 𐑩 𐑕𐑻𐑓 𐑝 𐑜𐑮𐑰𐑯 𐑕𐑻𐑡𐑩𐑟.

𐑞𐑧𐑯, 𐑩𐑜𐑱𐑯, 𐑦𐑯 𐑥𐑬𐑯𐑑𐑩𐑯𐑩𐑕 𐑒𐑳𐑯𐑑𐑮𐑦𐑟 𐑢𐑺 𐑞 𐑑𐑮𐑨𐑝𐑩𐑤𐑼 𐑦𐑟 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤𐑦 𐑜𐑻𐑛𐑩𐑤𐑛 𐑚𐑲 𐑨𐑥𐑓𐑦𐑔𐑦𐑨𐑑𐑮𐑦𐑒𐑩𐑤 𐑣𐑲𐑑𐑕; 𐑣𐑽 𐑯 𐑞𐑺 𐑓𐑮𐑪𐑥 𐑕𐑳𐑥 𐑤𐑳𐑒𐑦 𐑐𐑶𐑯𐑑 𐑝 𐑝𐑿 𐑿 𐑢𐑦𐑤 𐑒𐑨𐑗 𐑐𐑭𐑕𐑦𐑙 𐑜𐑤𐑦𐑥𐑐𐑕𐑩𐑟 𐑝 𐑞 𐑐𐑮𐑴𐑓𐑲𐑤𐑟 𐑝 𐑢𐑱𐑤𐑟 𐑛𐑦𐑓𐑲𐑯𐑛 𐑩𐑤𐑪𐑙 𐑞 𐑳𐑯𐑡𐑩𐑤𐑱𐑑𐑦𐑙 𐑮𐑦𐑡𐑩𐑟. 𐑚𐑳𐑑 𐑿 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑩 𐑔𐑳𐑮𐑴 𐑢𐑱𐑤𐑥𐑩𐑯, 𐑑 𐑕𐑰 𐑞𐑰𐑟 𐑕𐑲𐑑𐑕; 𐑯 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑞𐑨𐑑, 𐑚𐑳𐑑 𐑦𐑓 𐑿 𐑢𐑦𐑖 𐑑 𐑮𐑦𐑑𐑻𐑯 𐑑 𐑕𐑳𐑗 𐑩 𐑕𐑲𐑑 𐑩𐑜𐑱𐑯, 𐑿 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑖𐑫𐑼 𐑯 𐑑𐑱𐑒 𐑞 𐑦𐑜𐑟𐑨𐑒𐑑 𐑦𐑯𐑑𐑼𐑕𐑧𐑒𐑑𐑦𐑙 𐑤𐑨𐑑𐑦𐑑𐑿𐑛 𐑯 𐑤𐑪𐑯𐑜𐑦𐑑𐑿𐑛 𐑝 𐑘𐑹 𐑓𐑻𐑕𐑑 𐑕𐑑𐑨𐑯𐑛-𐑐𐑶𐑯𐑑, 𐑧𐑤𐑕 𐑕𐑴 𐑗𐑭𐑯𐑕-𐑤𐑲𐑒 𐑸 𐑕𐑳𐑗 𐑪𐑚𐑟𐑼𐑝𐑱𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑣𐑦𐑤𐑟, 𐑞𐑨𐑑 𐑘𐑹 𐑐𐑮𐑦𐑕𐑲𐑕, 𐑐𐑮𐑰𐑝𐑾𐑕 𐑕𐑑𐑨𐑯𐑛-𐑐𐑶𐑯𐑑 𐑢𐑫𐑛 𐑮𐑦𐑒𐑢𐑲𐑼 𐑩 𐑤𐑩𐑚𐑹𐑾𐑕 𐑮𐑰-𐑛𐑦𐑕𐑒𐑳𐑝𐑼𐑦; 𐑤𐑲𐑒 𐑞 ·𐑕𐑴𐑤𐑴𐑥𐑩 ·𐑲𐑤𐑩𐑯𐑛𐑟, 𐑢𐑦𐑗 𐑕𐑑𐑦𐑤 𐑮𐑦𐑥𐑱𐑯 𐑦𐑯𐑒𐑪𐑜𐑯𐑰𐑑𐑩, 𐑞𐑴 𐑢𐑳𐑯𐑕 𐑣𐑲-𐑮𐑳𐑓𐑑 ·𐑥𐑧𐑯𐑛𐑩𐑯𐑩 𐑑𐑮𐑪𐑛 𐑞𐑧𐑥 𐑯 𐑴𐑤𐑛 ·𐑝𐑦𐑜𐑻𐑮𐑩 𐑒𐑮𐑪𐑯𐑦𐑒𐑩𐑤𐑛 𐑞𐑧𐑥.
