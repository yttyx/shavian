·𐑗𐑨𐑐𐑑𐑼 5. 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑.

𐑲 𐑒𐑢𐑦𐑒𐑤𐑦 𐑓𐑪𐑤𐑴𐑛 𐑕𐑵𐑑, 𐑯 𐑛𐑦𐑕𐑧𐑯𐑛𐑦𐑙 𐑦𐑯𐑑𐑵 𐑞 𐑚𐑸-𐑮𐑵𐑥 𐑩𐑒𐑪𐑕𐑑𐑩𐑛 𐑞 𐑜𐑮𐑦𐑯𐑦𐑙 𐑤𐑨𐑯𐑛𐑤𐑹𐑛 𐑝𐑧𐑮𐑦 𐑐𐑤𐑧𐑟𐑩𐑯𐑑𐑤𐑦. 𐑲 𐑗𐑧𐑮𐑦𐑖𐑑 𐑯𐑴 𐑥𐑨𐑤𐑦𐑕 𐑑𐑩𐑢𐑹𐑛𐑟 𐑣𐑦𐑥, 𐑞𐑴 𐑣𐑰 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑒𐑲𐑤𐑸𐑒𐑦𐑙 𐑢𐑦𐑞 𐑥𐑰 𐑯𐑪𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑦𐑯 𐑞 𐑥𐑨𐑑𐑼 𐑝 𐑥𐑲 𐑚𐑧𐑛𐑓𐑧𐑤𐑴.

𐑣𐑬𐑧𐑝𐑼, 𐑩 𐑜𐑫𐑛 𐑤𐑭𐑓 𐑦𐑟 𐑩 𐑥𐑲𐑑𐑦 𐑜𐑫𐑛 𐑔𐑦𐑙, 𐑯 𐑮𐑭𐑞𐑼 𐑑𐑵 𐑕𐑒𐑺𐑕 𐑩 𐑜𐑫𐑛 𐑔𐑦𐑙; 𐑞 𐑥𐑹'𐑟 𐑞 𐑐𐑦𐑑𐑦. 𐑕𐑴, 𐑦𐑓 𐑧𐑯𐑦 𐑢𐑳𐑯 𐑥𐑨𐑯, 𐑦𐑯 𐑣𐑦𐑟 𐑴𐑯 𐑐𐑮𐑪𐑐𐑼 𐑐𐑻𐑕𐑩𐑯, 𐑩𐑓𐑹𐑛 𐑕𐑑𐑳𐑓 𐑓 𐑩 𐑜𐑫𐑛 𐑡𐑴𐑒 𐑑 𐑧𐑯𐑦𐑚𐑪𐑛𐑦, 𐑤𐑧𐑑 𐑣𐑦𐑥 𐑯𐑪𐑑 𐑚𐑰 𐑚𐑨𐑒𐑢𐑼𐑛, 𐑚𐑳𐑑 𐑤𐑧𐑑 𐑣𐑦𐑥 𐑗𐑽𐑓𐑩𐑤𐑦 𐑩𐑤𐑬 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑑 𐑕𐑐𐑧𐑯𐑛 𐑯 𐑚𐑰 𐑕𐑐𐑧𐑯𐑑 𐑦𐑯 𐑞𐑨𐑑 𐑢𐑱. 𐑯 𐑞 𐑥𐑨𐑯 𐑞𐑨𐑑 𐑣𐑨𐑟 𐑧𐑯𐑦𐑔𐑦𐑙 𐑚𐑬𐑯𐑑𐑦𐑓𐑩𐑤𐑦 𐑤𐑭𐑓𐑩𐑚𐑩𐑤 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥, 𐑚𐑰 𐑖𐑫𐑼 𐑞𐑺 𐑦𐑟 𐑥𐑹 𐑦𐑯 𐑞𐑨𐑑 𐑥𐑨𐑯 𐑞𐑨𐑯 𐑿 𐑐𐑼𐑣𐑨𐑐𐑕 𐑔𐑦𐑙𐑒 𐑓.

𐑞 𐑚𐑸-𐑮𐑵𐑥 𐑢𐑪𐑟 𐑯𐑬 𐑓𐑫𐑤 𐑝 𐑞 𐑚𐑹𐑛𐑼𐑟 𐑣𐑵 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑛𐑮𐑪𐑐𐑦𐑙 𐑦𐑯 𐑞 𐑯𐑲𐑑 𐑐𐑮𐑰𐑝𐑾𐑕, 𐑯 𐑣𐑵𐑥 𐑲 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑨𐑟 𐑘𐑧𐑑 𐑣𐑨𐑛 𐑩 𐑜𐑫𐑛 𐑤𐑫𐑒 𐑨𐑑 𐑞𐑱 𐑢𐑻 𐑯𐑽𐑤𐑦 𐑷𐑤 𐑢𐑱𐑤𐑥𐑧𐑯; 𐑗𐑰𐑓 𐑥𐑱𐑑𐑕, 𐑯 𐑕𐑧𐑒𐑩𐑯𐑛 𐑥𐑱𐑑𐑕, 𐑯 𐑔𐑻𐑛 𐑥𐑱𐑑𐑕, 𐑯 𐑕𐑰 𐑒𐑸𐑐𐑩𐑯𐑑𐑼𐑟, 𐑯 𐑕𐑰 𐑒𐑵𐑐𐑼𐑟, 𐑯 𐑕𐑰 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔𐑕, 𐑯 𐑣𐑸𐑐𐑵𐑯𐑽𐑟, 𐑯 𐑖𐑦𐑐 𐑒𐑰𐑐𐑼𐑟; 𐑩 𐑚𐑮𐑬𐑯 𐑯 𐑚𐑮𐑷𐑯𐑦 𐑒𐑳𐑥𐑐𐑩𐑯𐑦, 𐑢𐑦𐑞 𐑚𐑪𐑕𐑒𐑦 𐑚𐑽𐑛𐑟; 𐑩𐑯 𐑳𐑯𐑖𐑹𐑯, 𐑖𐑨𐑜𐑦 𐑕𐑧𐑑, 𐑷𐑤 𐑢𐑺𐑦𐑙 𐑥𐑳𐑙𐑒𐑦 𐑡𐑨𐑒𐑩𐑑𐑕 𐑓 𐑥𐑹𐑯𐑦𐑙 𐑜𐑬𐑯𐑟.

𐑿 𐑒𐑫𐑛 𐑐𐑮𐑦𐑑𐑦 𐑐𐑤𐑱𐑯𐑤𐑦 𐑑𐑧𐑤 𐑣𐑬 𐑤𐑪𐑙 𐑰𐑗 𐑢𐑳𐑯 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑩𐑖𐑹. 𐑞𐑦𐑕 𐑘𐑳𐑙 𐑓𐑧𐑤𐑴'𐑟 𐑣𐑧𐑤𐑔𐑦 𐑗𐑰𐑒 𐑦𐑟 𐑤𐑲𐑒 𐑩 𐑕𐑳𐑯-𐑑𐑴𐑕𐑑𐑩𐑛 𐑐𐑺 𐑦𐑯 𐑣𐑿, 𐑯 𐑢𐑫𐑛 𐑕𐑰𐑥 𐑑 𐑕𐑥𐑧𐑤 𐑷𐑤𐑥𐑴𐑕𐑑 𐑨𐑟 𐑥𐑳𐑕𐑒𐑦; 𐑣𐑰 𐑒𐑨𐑯𐑪𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑔𐑮𐑰 𐑛𐑱𐑟 𐑤𐑨𐑯𐑛𐑩𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 ·𐑦𐑯𐑛𐑾𐑯 𐑝𐑶𐑦𐑡. 𐑞𐑨𐑑 𐑥𐑨𐑯 𐑯𐑧𐑒𐑕𐑑 𐑣𐑦𐑥 𐑤𐑫𐑒𐑕 𐑩 𐑓𐑿 𐑖𐑱𐑛𐑟 𐑤𐑲𐑑𐑼; 𐑿 𐑥𐑲𐑑 𐑕𐑱 𐑩 𐑑𐑳𐑗 𐑝 𐑕𐑨𐑑𐑦𐑯 𐑢𐑫𐑛 𐑦𐑟 𐑦𐑯 𐑣𐑦𐑥. 𐑦𐑯 𐑞 𐑒𐑩𐑥𐑐𐑤𐑧𐑒𐑖𐑩𐑯 𐑝 𐑩 𐑔𐑻𐑛 𐑕𐑑𐑦𐑤 𐑤𐑦𐑙𐑜𐑼𐑟 𐑩 𐑑𐑮𐑪𐑐𐑦𐑒 𐑑𐑷𐑯, 𐑚𐑳𐑑 𐑕𐑤𐑲𐑑𐑤𐑦 𐑚𐑤𐑰𐑗𐑑 𐑢𐑦𐑞𐑷𐑤; 𐑣𐑰 𐑛𐑬𐑑𐑤𐑩𐑕 𐑣𐑨𐑟 𐑑𐑨𐑮𐑦𐑛 𐑣𐑴𐑤 𐑢𐑰𐑒𐑕 𐑩𐑖𐑹. 𐑚𐑳𐑑 𐑣𐑵 𐑒𐑫𐑛 𐑖𐑴 𐑩 𐑗𐑰𐑒 𐑤𐑲𐑒 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜? 𐑢𐑦𐑗, 𐑚𐑸𐑛 𐑢𐑦𐑞 𐑝𐑺𐑾𐑕 𐑑𐑦𐑯𐑑𐑕, 𐑕𐑰𐑥𐑛 𐑤𐑲𐑒 𐑞 ·𐑨𐑯𐑛𐑰𐑟' 𐑢𐑧𐑕𐑑𐑼𐑯 𐑕𐑤𐑴𐑐, 𐑑 𐑖𐑴 𐑓𐑹𐑔 𐑦𐑯 𐑢𐑳𐑯 𐑼𐑱, 𐑒𐑩𐑯𐑑𐑮𐑭𐑕𐑑𐑦𐑙 𐑒𐑤𐑲𐑥𐑩𐑑𐑕, 𐑟𐑴𐑯 𐑚𐑲 𐑟𐑴𐑯.

“𐑜𐑮𐑳𐑚, 𐑣𐑴!” 𐑯𐑬 𐑒𐑮𐑲𐑛 𐑞 𐑤𐑨𐑯𐑛𐑤𐑹𐑛, 𐑓𐑤𐑦𐑙𐑦𐑙 𐑴𐑐𐑩𐑯 𐑩 𐑛𐑹, 𐑯 𐑦𐑯 𐑢𐑰 𐑢𐑧𐑯𐑑 𐑑 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑.

𐑞𐑱 𐑕𐑱 𐑞𐑨𐑑 𐑥𐑧𐑯 𐑣𐑵 𐑣𐑨𐑝 𐑕𐑰𐑯 𐑞 𐑢𐑻𐑤𐑛, 𐑞𐑺𐑚𐑲 𐑚𐑦𐑒𐑳𐑥 𐑒𐑢𐑲𐑑 𐑨𐑑 𐑰𐑟 𐑦𐑯 𐑥𐑨𐑯𐑼, 𐑒𐑢𐑲𐑑 𐑕𐑧𐑤𐑓-𐑐𐑩𐑟𐑧𐑕𐑑 𐑦𐑯 𐑒𐑳𐑥𐑐𐑩𐑯𐑦. 𐑯𐑪𐑑 𐑷𐑤𐑢𐑱𐑟, 𐑞𐑴: 𐑤𐑰𐑛𐑘𐑸𐑛, 𐑞 𐑜𐑮𐑱𐑑 ·𐑯𐑿 ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛 𐑑𐑮𐑨𐑝𐑩𐑤𐑼, 𐑯 ·𐑥𐑳𐑙𐑜𐑴 ·𐑐𐑸𐑒, 𐑞 ·𐑕𐑒𐑪𐑗 𐑢𐑳𐑯; 𐑝 𐑷𐑤 𐑥𐑧𐑯, 𐑞𐑱 𐑐𐑩𐑟𐑧𐑕𐑑 𐑞 𐑤𐑰𐑕𐑑 𐑩𐑖𐑫𐑼𐑩𐑯𐑕 𐑦𐑯 𐑞 𐑐𐑸𐑤𐑼. 𐑚𐑳𐑑 𐑐𐑼𐑣𐑨𐑐𐑕 𐑞 𐑥𐑽 𐑒𐑮𐑪𐑕𐑦𐑙 𐑝 ·𐑕𐑲𐑚𐑽𐑾 𐑦𐑯 𐑩 𐑕𐑤𐑧𐑡 𐑛𐑮𐑷𐑯 𐑚𐑲 𐑛𐑪𐑜𐑟 𐑨𐑟 ·𐑤𐑰𐑛𐑘𐑸𐑛 𐑛𐑦𐑛, 𐑹 𐑞 𐑑𐑱𐑒𐑦𐑙 𐑩 𐑤𐑪𐑙 𐑕𐑪𐑤𐑦𐑑𐑼𐑦 𐑢𐑷𐑒 𐑪𐑯 𐑩𐑯 𐑧𐑥𐑐𐑑𐑦 𐑕𐑑𐑳𐑥𐑩𐑒, 𐑦𐑯 𐑞 𐑯𐑰𐑜𐑮𐑴 𐑣𐑸𐑑 𐑝 ·𐑨𐑓𐑮𐑦𐑒𐑩, 𐑢𐑦𐑗 𐑢𐑪𐑟 𐑞 𐑕𐑳𐑥 𐑝 𐑐𐑫𐑼 ·𐑥𐑳𐑙𐑜𐑴'𐑟 𐑐𐑼𐑓𐑹𐑥𐑩𐑯𐑕𐑩𐑟—𐑞𐑦𐑕 𐑒𐑲𐑯𐑛 𐑝 𐑑𐑮𐑨𐑝𐑩𐑤, 𐑲 𐑕𐑱, 𐑥𐑱 𐑯𐑪𐑑 𐑚𐑰 𐑞 𐑝𐑧𐑮𐑦 𐑚𐑧𐑕𐑑 𐑥𐑴𐑛 𐑝 𐑩𐑑𐑱𐑯𐑦𐑙 𐑩 𐑣𐑲 𐑕𐑴𐑖𐑩𐑤 𐑐𐑪𐑤𐑦𐑖. 𐑕𐑑𐑦𐑤, 𐑓 𐑞 𐑥𐑴𐑕𐑑 𐑐𐑸𐑑, 𐑞𐑨𐑑 𐑕𐑹𐑑 𐑝 𐑔𐑦𐑙 𐑦𐑟 𐑑 𐑚𐑰 𐑣𐑨𐑛 𐑧𐑯𐑦𐑢𐑺.

𐑞𐑰𐑟 𐑮𐑦𐑓𐑤𐑧𐑒𐑖𐑩𐑯𐑟 𐑡𐑳𐑕𐑑 𐑣𐑽 𐑸 𐑩𐑒𐑱𐑠𐑩𐑯𐑛 𐑚𐑲 𐑞 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕 𐑞𐑨𐑑 𐑭𐑓𐑑𐑼 𐑢𐑰 𐑢𐑻 𐑷𐑤 𐑕𐑰𐑑𐑩𐑛 𐑨𐑑 𐑞 𐑑𐑱𐑚𐑩𐑤, 𐑯 𐑲 𐑢𐑪𐑟 𐑐𐑮𐑦𐑐𐑺𐑦𐑙 𐑑 𐑣𐑽 𐑕𐑳𐑥 𐑜𐑫𐑛 𐑕𐑑𐑹𐑦𐑟 𐑩𐑚𐑬𐑑 𐑢𐑱𐑤𐑦𐑙; 𐑑 𐑥𐑲 𐑯𐑴 𐑕𐑥𐑷𐑤 𐑕𐑼𐑐𐑮𐑲𐑟, 𐑯𐑽𐑤𐑦 𐑧𐑝𐑮𐑦 𐑥𐑨𐑯 𐑥𐑱𐑯𐑑𐑱𐑯𐑛 𐑩 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑕𐑲𐑤𐑩𐑯𐑕. 𐑯 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑞𐑨𐑑, 𐑚𐑳𐑑 𐑞𐑱 𐑤𐑫𐑒𐑑 𐑦𐑥𐑚𐑨𐑮𐑩𐑕𐑑. 𐑘𐑧𐑕, 𐑣𐑽 𐑢𐑻 𐑩 𐑕𐑧𐑑 𐑝 𐑕𐑰-𐑛𐑪𐑜𐑟, 𐑥𐑧𐑯𐑦 𐑝 𐑣𐑵𐑥 𐑢𐑦𐑞𐑬𐑑 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑚𐑨𐑖𐑓𐑩𐑤𐑯𐑩𐑕 𐑣𐑨𐑛 𐑚𐑹𐑛𐑩𐑛 𐑜𐑮𐑱𐑑 𐑢𐑱𐑤𐑟 𐑪𐑯 𐑞 𐑣𐑲 𐑕𐑰𐑟—𐑦𐑯𐑑𐑲𐑼 𐑕𐑑𐑮𐑱𐑯𐑡𐑼𐑟 𐑑 𐑞𐑧𐑥—𐑯 𐑛𐑿𐑩𐑤𐑛 𐑞𐑧𐑥 𐑛𐑧𐑛 𐑢𐑦𐑞𐑬𐑑 𐑢𐑦𐑙𐑒𐑦𐑙; 𐑯 𐑘𐑧𐑑, 𐑣𐑽 𐑞𐑱 𐑕𐑨𐑑 𐑨𐑑 𐑩 𐑕𐑴𐑖𐑩𐑤 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑 𐑑𐑱𐑚𐑩𐑤—𐑷𐑤 𐑝 𐑞 𐑕𐑱𐑥 𐑒𐑷𐑤𐑦𐑙, 𐑷𐑤 𐑝 𐑒𐑦𐑯𐑛𐑮𐑩𐑛 𐑑𐑱𐑕𐑑𐑕—𐑤𐑫𐑒𐑦𐑙 𐑮𐑬𐑯𐑛 𐑨𐑟 𐑖𐑰𐑐𐑦𐑖𐑤𐑦 𐑨𐑑 𐑰𐑗 𐑳𐑞𐑼 𐑨𐑟 𐑞𐑴 𐑞𐑱 𐑣𐑨𐑛 𐑯𐑧𐑝𐑼 𐑚𐑰𐑯 𐑬𐑑 𐑝 𐑕𐑲𐑑 𐑝 𐑕𐑳𐑥 𐑖𐑰𐑐𐑓𐑴𐑤𐑛 𐑩𐑥𐑳𐑙 𐑞 ·𐑜𐑮𐑰𐑯 ·𐑥𐑬𐑯𐑑𐑩𐑯𐑟. 𐑩 𐑒𐑘𐑫𐑼𐑾𐑕 𐑕𐑲𐑑; 𐑞𐑰𐑟 𐑚𐑨𐑖𐑓𐑩𐑤 𐑚𐑺𐑟, 𐑞𐑰𐑟 𐑑𐑦𐑥𐑦𐑛 𐑢𐑹𐑽 𐑢𐑱𐑤𐑥𐑧𐑯!

𐑚𐑳𐑑 𐑨𐑟 𐑓 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜—𐑢𐑲, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑕𐑨𐑑 𐑞𐑺 𐑩𐑥𐑳𐑙 𐑞𐑧𐑥—𐑨𐑑 𐑞 𐑣𐑧𐑛 𐑝 𐑞 𐑑𐑱𐑚𐑩𐑤, 𐑑𐑵, 𐑦𐑑 𐑕𐑴 𐑗𐑭𐑯𐑕𐑑; 𐑨𐑟 𐑒𐑵𐑤 𐑨𐑟 𐑩𐑯 𐑲𐑕𐑦𐑒𐑩𐑤. 𐑑 𐑚𐑰 𐑖𐑫𐑼 𐑲 𐑒𐑨𐑯𐑪𐑑 𐑕𐑱 𐑥𐑳𐑗 𐑓 𐑣𐑦𐑟 𐑚𐑮𐑰𐑛𐑦𐑙. 𐑣𐑦𐑟 𐑜𐑮𐑱𐑑𐑩𐑕𐑑 𐑩𐑛𐑥𐑲𐑼𐑼 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑣𐑨𐑝 𐑒𐑹𐑛𐑾𐑤𐑦 𐑡𐑳𐑕𐑑𐑦𐑓𐑲𐑛 𐑣𐑦𐑟 𐑚𐑮𐑦𐑙𐑦𐑙 𐑣𐑦𐑟 𐑣𐑸𐑐𐑵𐑯 𐑦𐑯𐑑𐑵 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑯 𐑿𐑟𐑦𐑙 𐑦𐑑 𐑞𐑺 𐑢𐑦𐑞𐑬𐑑 𐑕𐑧𐑮𐑦𐑥𐑩𐑯𐑦; 𐑮𐑰𐑗𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑑𐑱𐑚𐑩𐑤 𐑢𐑦𐑞 𐑦𐑑, 𐑑 𐑞 𐑦𐑥𐑦𐑯𐑩𐑯𐑑 𐑡𐑧𐑐𐑼𐑛𐑦 𐑝 𐑥𐑧𐑯𐑦 𐑣𐑧𐑛𐑟, 𐑯 𐑜𐑮𐑨𐑐𐑩𐑤𐑦𐑙 𐑞 𐑚𐑰𐑓𐑕𐑑𐑱𐑒𐑕 𐑑𐑩𐑢𐑹𐑛𐑟 𐑣𐑦𐑥. 𐑚𐑳𐑑 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑕𐑻𐑑𐑩𐑯𐑤𐑦 𐑝𐑧𐑮𐑦 𐑒𐑵𐑤𐑦 𐑛𐑳𐑯 𐑚𐑲 𐑣𐑦𐑥, 𐑯 𐑧𐑝𐑮𐑦 𐑢𐑳𐑯 𐑯𐑴𐑟 𐑞𐑨𐑑 𐑦𐑯 𐑥𐑴𐑕𐑑 𐑐𐑰𐑐𐑩𐑤'𐑟 𐑧𐑕𐑑𐑦𐑥𐑱𐑖𐑩𐑯, 𐑑 𐑛𐑵 𐑧𐑯𐑦𐑔𐑦𐑙 𐑒𐑵𐑤𐑦 𐑦𐑟 𐑑 𐑛𐑵 𐑦𐑑 𐑡𐑧𐑯𐑑𐑰𐑤𐑦.

𐑢𐑰 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑕𐑐𐑰𐑒 𐑝 𐑷𐑤 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑐𐑦𐑒𐑿𐑤𐑦𐑨𐑮𐑦𐑑𐑦𐑟 𐑣𐑽; 𐑣𐑬 𐑣𐑰 𐑦𐑕𐑗𐑵𐑛 𐑒𐑪𐑓𐑦 𐑯 𐑣𐑪𐑑 𐑮𐑴𐑤𐑟, 𐑯 𐑩𐑐𐑤𐑲𐑛 𐑣𐑦𐑟 𐑳𐑯𐑛𐑦𐑝𐑲𐑛𐑩𐑛 𐑩𐑑𐑧𐑯𐑖𐑩𐑯 𐑑 𐑚𐑰𐑓𐑕𐑑𐑱𐑒𐑕, 𐑛𐑳𐑯 𐑮𐑺. 𐑦𐑯𐑳𐑓, 𐑞𐑨𐑑 𐑢𐑧𐑯 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑 𐑢𐑪𐑟 𐑴𐑝𐑼 𐑣𐑰 𐑢𐑦𐑞𐑛𐑮𐑵 𐑤𐑲𐑒 𐑞 𐑮𐑧𐑕𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑐𐑳𐑚𐑤𐑦𐑒 𐑮𐑵𐑥, 𐑤𐑲𐑑𐑩𐑛 𐑣𐑦𐑟 𐑑𐑪𐑥𐑩𐑣𐑷𐑒-𐑐𐑲𐑐, 𐑯 𐑢𐑪𐑟 𐑕𐑦𐑑𐑦𐑙 𐑞𐑺 𐑒𐑢𐑲𐑩𐑑𐑤𐑦 𐑛𐑲𐑡𐑧𐑕𐑑𐑦𐑙 𐑯 𐑕𐑥𐑴𐑒𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑦𐑯𐑕𐑧𐑐𐑼𐑩𐑚𐑩𐑤 𐑣𐑨𐑑 𐑪𐑯, 𐑢𐑧𐑯 𐑲 𐑕𐑨𐑤𐑦𐑛 𐑬𐑑 𐑓 𐑩 𐑕𐑑𐑮𐑴𐑤.
