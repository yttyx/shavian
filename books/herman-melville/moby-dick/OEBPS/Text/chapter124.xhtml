·𐑗𐑨𐑐𐑑𐑼 124. 𐑞 ·𐑯𐑰𐑛𐑩𐑤.

𐑯𐑧𐑒𐑕𐑑 𐑥𐑹𐑯𐑦𐑙 𐑞 𐑯𐑪𐑑-𐑘𐑧𐑑-𐑕𐑩𐑚𐑕𐑲𐑛𐑩𐑛 𐑕𐑰 𐑮𐑴𐑤𐑛 𐑦𐑯 𐑤𐑪𐑙 𐑕𐑤𐑴 𐑚𐑦𐑤𐑴𐑟 𐑝 𐑥𐑲𐑑𐑦 𐑚𐑳𐑤𐑒, 𐑯 𐑕𐑑𐑮𐑲𐑝𐑦𐑙 𐑦𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑜𐑻𐑜𐑩𐑤𐑦𐑙 𐑑𐑮𐑨𐑒, 𐑐𐑫𐑖𐑑 𐑣𐑻 𐑪𐑯 𐑤𐑲𐑒 𐑡𐑲𐑩𐑯𐑑𐑕' 𐑐𐑭𐑥𐑟 𐑬𐑑𐑕𐑐𐑮𐑧𐑛. 𐑞 𐑕𐑑𐑮𐑪𐑙, 𐑳𐑯𐑕𐑑𐑨𐑜𐑼𐑦𐑙 𐑚𐑮𐑰𐑟 𐑩𐑚𐑬𐑯𐑛𐑩𐑛 𐑕𐑴, 𐑞𐑨𐑑 𐑕𐑒𐑲 𐑯 𐑺 𐑕𐑰𐑥𐑛 𐑝𐑭𐑕𐑑 𐑬𐑑𐑚𐑧𐑤𐑦𐑦𐑙 𐑕𐑱𐑤𐑟; 𐑞 𐑣𐑴𐑤 𐑢𐑻𐑤𐑛 𐑚𐑵𐑥𐑛 𐑚𐑦𐑓𐑹 𐑞 𐑢𐑦𐑯𐑛. 𐑥𐑳𐑓𐑩𐑤𐑛 𐑦𐑯 𐑞 𐑓𐑫𐑤 𐑥𐑹𐑯𐑦𐑙 𐑤𐑲𐑑, 𐑞 𐑦𐑯𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑕𐑳𐑯 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑯𐑴𐑯 𐑚𐑲 𐑞 𐑕𐑐𐑮𐑧𐑛 𐑦𐑯𐑑𐑧𐑯𐑕𐑦𐑑𐑦 𐑝 𐑣𐑦𐑟 𐑐𐑤𐑱𐑕; 𐑢𐑺 𐑣𐑦𐑟 𐑚𐑱𐑩𐑯𐑧𐑑 𐑮𐑱𐑟 𐑥𐑵𐑝𐑛 𐑪𐑯 𐑦𐑯 𐑕𐑑𐑨𐑒𐑕. 𐑦𐑥𐑚𐑤𐑱𐑟𐑩𐑯𐑦𐑙𐑟, 𐑨𐑟 𐑝 𐑒𐑮𐑬𐑯𐑛 ·𐑚𐑨𐑚𐑦𐑤𐑴𐑯𐑾𐑯 𐑒𐑦𐑙𐑟 𐑯 𐑒𐑢𐑰𐑯𐑟, 𐑮𐑱𐑯𐑛 𐑴𐑝𐑼 𐑧𐑝𐑮𐑦𐑔𐑦𐑙. 𐑞 𐑕𐑰 𐑢𐑪𐑟 𐑨𐑟 𐑩 𐑒𐑮𐑵𐑕𐑩𐑚𐑩𐑤 𐑝 𐑥𐑴𐑤𐑑𐑩𐑯 𐑜𐑴𐑤𐑛, 𐑞𐑨𐑑 𐑚𐑳𐑚𐑩𐑤𐑦𐑙𐑤𐑦 𐑤𐑰𐑐𐑕 𐑢𐑦𐑞 𐑤𐑲𐑑 𐑯 𐑣𐑰𐑑.

𐑤𐑪𐑙 𐑥𐑱𐑯𐑑𐑱𐑯𐑦𐑙 𐑩𐑯 𐑦𐑯𐑗𐑭𐑯𐑑𐑩𐑛 𐑕𐑲𐑤𐑩𐑯𐑕, ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑫𐑛 𐑩𐑐𐑸𐑑; 𐑯 𐑧𐑝𐑮𐑦 𐑑𐑲𐑥 𐑞 ·𐑑𐑰𐑑𐑼𐑦𐑙 𐑖𐑦𐑐 𐑤𐑴𐑼𐑦𐑙𐑤𐑦 𐑐𐑦𐑗𐑑 𐑛𐑬𐑯 𐑣𐑻 𐑚𐑴𐑕𐑐𐑮𐑦𐑑, 𐑣𐑰 𐑑𐑻𐑯𐑛 𐑑 𐑲 𐑞 𐑚𐑮𐑲𐑑 𐑕𐑳𐑯'𐑟 𐑮𐑱𐑟 𐑐𐑮𐑩𐑛𐑿𐑕𐑑 𐑩𐑣𐑧𐑛; 𐑯 𐑢𐑧𐑯 𐑖𐑰 𐑐𐑮𐑩𐑓𐑬𐑯𐑛𐑤𐑦 𐑕𐑧𐑑𐑩𐑤𐑛 𐑚𐑲 𐑞 𐑕𐑑𐑻𐑯, 𐑣𐑰 𐑑𐑻𐑯𐑛 𐑚𐑦𐑣𐑲𐑯𐑛, 𐑯 𐑕𐑷 𐑞 𐑕𐑳𐑯'𐑟 𐑮𐑽𐑢𐑼𐑛 𐑐𐑤𐑱𐑕, 𐑯 𐑣𐑬 𐑞 𐑕𐑱𐑥 𐑘𐑧𐑤𐑴 𐑮𐑱𐑟 𐑢𐑻 𐑚𐑤𐑧𐑯𐑛𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑳𐑯𐑛𐑰𐑝𐑦𐑱𐑑𐑦𐑙 𐑢𐑱𐑒.

“𐑣𐑭, 𐑣𐑭, 𐑥𐑲 𐑖𐑦𐑐! 𐑞𐑬 𐑥𐑲𐑑𐑩𐑕𐑑 𐑢𐑧𐑤 𐑚𐑰 𐑑𐑱𐑒𐑩𐑯 𐑯𐑬 𐑓 𐑞 𐑕𐑰-𐑗𐑨𐑮𐑾𐑑 𐑝 𐑞 𐑕𐑳𐑯. 𐑣𐑴, 𐑣𐑴! 𐑷𐑤 𐑘𐑰 𐑯𐑱𐑖𐑩𐑯𐑟 𐑚𐑦𐑓𐑹 𐑥𐑲 𐑐𐑮𐑬, 𐑲 𐑚𐑮𐑦𐑙 𐑞 𐑕𐑳𐑯 𐑑 𐑘𐑰! 𐑘𐑴𐑒 𐑪𐑯 𐑞 𐑓𐑻𐑞𐑼 𐑚𐑦𐑤𐑴𐑟; 𐑣𐑩𐑤𐑴! 𐑩 𐑑𐑨𐑯𐑛𐑩𐑥, 𐑲 𐑛𐑮𐑲𐑝 𐑞 𐑕𐑰!”

𐑚𐑳𐑑 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑮𐑱𐑯𐑛 𐑚𐑨𐑒 𐑚𐑲 𐑕𐑳𐑥 𐑒𐑬𐑯𐑑𐑼 𐑔𐑷𐑑, 𐑣𐑰 𐑣𐑳𐑮𐑦𐑛 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑣𐑧𐑤𐑥, 𐑣𐑳𐑕𐑒𐑦𐑤𐑦 𐑛𐑦𐑥𐑭𐑯𐑛𐑦𐑙 𐑣𐑬 𐑞 𐑖𐑦𐑐 𐑢𐑪𐑟 𐑣𐑧𐑛𐑦𐑙.

“𐑰𐑕𐑑-𐑕𐑵-𐑰𐑕𐑑, 𐑕𐑻,” 𐑕𐑧𐑛 𐑞 𐑓𐑮𐑲𐑑𐑩𐑯𐑛 𐑕𐑑𐑽𐑟𐑥𐑩𐑯.

“𐑞𐑬 𐑤𐑲𐑩𐑕𐑑!” 𐑕𐑥𐑲𐑑𐑦𐑙 𐑣𐑦𐑥 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑒𐑤𐑧𐑯𐑗𐑑 𐑓𐑦𐑕𐑑. “𐑣𐑧𐑛𐑦𐑙 ·𐑰𐑕𐑑 𐑨𐑑 𐑞𐑦𐑕 𐑬𐑼 𐑦𐑯 𐑞 𐑥𐑹𐑯𐑦𐑙, 𐑯 𐑞 𐑕𐑳𐑯 𐑩𐑕𐑑𐑻𐑯?”

𐑩𐑐𐑪𐑯 𐑞𐑦𐑕 𐑧𐑝𐑮𐑦 𐑕𐑴𐑤 𐑢𐑪𐑟 𐑒𐑩𐑯𐑓𐑬𐑯𐑛𐑩𐑛; 𐑓 𐑞 𐑓𐑦𐑯𐑪𐑥𐑦𐑯𐑪𐑯 𐑡𐑳𐑕𐑑 𐑞𐑧𐑯 𐑩𐑚𐑟𐑻𐑝𐑛 𐑚𐑲 ·𐑱𐑣𐑨𐑚 𐑣𐑨𐑛 𐑳𐑯𐑩𐑒𐑬𐑯𐑑𐑩𐑚𐑤𐑦 𐑦𐑕𐑒𐑱𐑐𐑑 𐑧𐑝𐑮𐑦 𐑢𐑳𐑯 𐑧𐑤𐑕; 𐑚𐑳𐑑 𐑦𐑑𐑕 𐑝𐑧𐑮𐑦 𐑚𐑤𐑲𐑯𐑛𐑦𐑙 𐑐𐑨𐑤𐑐𐑩𐑚𐑩𐑤𐑯𐑩𐑕 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑞 𐑒𐑷𐑟.

𐑔𐑮𐑳𐑕𐑑𐑦𐑙 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑣𐑭𐑓 𐑢𐑱 𐑦𐑯𐑑𐑵 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤, ·𐑱𐑣𐑨𐑚 𐑒𐑷𐑑 𐑢𐑳𐑯 𐑜𐑤𐑦𐑥𐑐𐑕 𐑝 𐑞 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟; 𐑣𐑦𐑟 𐑳𐑐𐑤𐑦𐑓𐑑𐑩𐑛 𐑸𐑥 𐑕𐑤𐑴𐑤𐑦 𐑓𐑧𐑤; 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑣𐑰 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰𐑥𐑛 𐑑 𐑕𐑑𐑨𐑜𐑼. 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑦𐑣𐑲𐑯𐑛 𐑣𐑦𐑥 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑤𐑫𐑒𐑑, 𐑯 𐑤𐑴! 𐑞 𐑑𐑵 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟 𐑐𐑶𐑯𐑑𐑩𐑛 ·𐑰𐑕𐑑, 𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑨𐑟 𐑦𐑯𐑓𐑨𐑤𐑩𐑚𐑤𐑦 𐑜𐑴𐑦𐑙 ·𐑢𐑧𐑕𐑑.

𐑚𐑳𐑑 𐑺 𐑞 𐑓𐑻𐑕𐑑 𐑢𐑲𐑤𐑛 𐑩𐑤𐑸𐑥 𐑒𐑫𐑛 𐑜𐑧𐑑 𐑬𐑑 𐑩𐑚𐑮𐑷𐑛 𐑩𐑥𐑳𐑙 𐑞 𐑒𐑮𐑵, 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑢𐑦𐑞 𐑩 𐑮𐑦𐑡𐑦𐑛 𐑤𐑭𐑓 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛, “𐑲 𐑣𐑨𐑝 𐑦𐑑! 𐑦𐑑 𐑣𐑨𐑟 𐑣𐑨𐑐𐑩𐑯𐑛 𐑚𐑦𐑓𐑹. ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑤𐑭𐑕𐑑 𐑯𐑲𐑑'𐑕 𐑔𐑳𐑯𐑛𐑼 𐑑𐑻𐑯𐑛 𐑬𐑼 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟—𐑞𐑨𐑑'𐑕 𐑷𐑤. 𐑞𐑬 𐑣𐑨𐑕𐑑 𐑚𐑦𐑓𐑹 𐑯𐑬 𐑣𐑻𐑛 𐑝 𐑕𐑳𐑗 𐑩 𐑔𐑦𐑙, 𐑲 𐑑𐑱𐑒 𐑦𐑑.”

“𐑲; 𐑚𐑳𐑑 𐑯𐑧𐑝𐑼 𐑚𐑦𐑓𐑹 𐑣𐑨𐑟 𐑦𐑑 𐑣𐑨𐑐𐑩𐑯𐑛 𐑑 𐑥𐑰, 𐑕𐑻,” 𐑕𐑧𐑛 𐑞 𐑐𐑱𐑤 𐑥𐑱𐑑, 𐑜𐑤𐑵𐑥𐑦𐑤𐑦.

𐑣𐑽, 𐑦𐑑 𐑥𐑳𐑕𐑑 𐑯𐑰𐑛𐑟 𐑚𐑰 𐑕𐑧𐑛, 𐑞𐑨𐑑 𐑨𐑒𐑕𐑦𐑛𐑩𐑯𐑑𐑕 𐑤𐑲𐑒 𐑞𐑦𐑕 𐑣𐑨𐑝 𐑦𐑯 𐑥𐑹 𐑞𐑨𐑯 𐑢𐑳𐑯 𐑒𐑱𐑕 𐑩𐑒𐑻𐑛 𐑑 𐑖𐑦𐑐𐑕 𐑦𐑯 𐑝𐑲𐑩𐑤𐑩𐑯𐑑 𐑕𐑑𐑹𐑥𐑟. 𐑞 𐑥𐑨𐑜𐑯𐑧𐑑𐑦𐑒 𐑧𐑯𐑼𐑡𐑦, 𐑨𐑟 𐑛𐑦𐑝𐑧𐑤𐑩𐑐𐑑 𐑦𐑯 𐑞 𐑥𐑨𐑮𐑦𐑯𐑼'𐑟 𐑯𐑰𐑛𐑩𐑤, 𐑦𐑟, 𐑨𐑟 𐑷𐑤 𐑯𐑴, 𐑦𐑕𐑧𐑯𐑖𐑩𐑤𐑦 𐑢𐑳𐑯 𐑢𐑦𐑞 𐑞 𐑦𐑤𐑧𐑒𐑑𐑮𐑦𐑕𐑦𐑑𐑦 𐑚𐑦𐑣𐑧𐑤𐑛 𐑦𐑯 𐑣𐑧𐑝𐑩𐑯; 𐑣𐑧𐑯𐑕 𐑦𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑑 𐑚𐑰 𐑥𐑳𐑗 𐑥𐑸𐑝𐑩𐑤𐑛 𐑨𐑑, 𐑞𐑨𐑑 𐑕𐑳𐑗 𐑔𐑦𐑙𐑟 𐑖𐑵𐑛 𐑚𐑰. 𐑦𐑯𐑕𐑑𐑩𐑯𐑕𐑩𐑟 𐑢𐑺 𐑞 𐑤𐑲𐑑𐑯𐑦𐑙 𐑣𐑨𐑟 𐑨𐑒𐑗𐑵𐑩𐑤𐑦 𐑕𐑑𐑮𐑳𐑒 𐑞 𐑝𐑧𐑕𐑩𐑤, 𐑕𐑴 𐑨𐑟 𐑑 𐑕𐑥𐑲𐑑 𐑛𐑬𐑯 𐑕𐑳𐑥 𐑝 𐑞 𐑕𐑐𐑸𐑟 𐑯 𐑮𐑦𐑜𐑦𐑙, 𐑞 𐑦𐑓𐑧𐑒𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑯𐑰𐑛𐑩𐑤 𐑣𐑨𐑟 𐑨𐑑 𐑑𐑲𐑥𐑟 𐑚𐑰𐑯 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑓𐑱𐑑𐑩𐑤; 𐑷𐑤 𐑦𐑑𐑕 𐑤𐑴𐑛𐑕𐑑𐑴𐑯 𐑝𐑻𐑗𐑵 𐑚𐑰𐑦𐑙 𐑩𐑯𐑲𐑩𐑤𐑱𐑑𐑩𐑛, 𐑕𐑴 𐑞𐑨𐑑 𐑞 𐑚𐑦𐑓𐑹 𐑥𐑨𐑜𐑯𐑧𐑑𐑦𐑒 𐑕𐑑𐑰𐑤 𐑢𐑪𐑟 𐑝 𐑯𐑴 𐑥𐑹 𐑿𐑕 𐑞𐑨𐑯 𐑩𐑯 𐑴𐑤𐑛 𐑢𐑲𐑓'𐑕 𐑯𐑦𐑑𐑦𐑙 𐑯𐑰𐑛𐑩𐑤. 𐑚𐑳𐑑 𐑦𐑯 𐑲𐑞𐑼 𐑒𐑱𐑕, 𐑞 𐑯𐑰𐑛𐑩𐑤 𐑯𐑧𐑝𐑼 𐑩𐑜𐑱𐑯, 𐑝 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑮𐑦𐑒𐑳𐑝𐑼𐑟 𐑞 𐑼𐑦𐑡𐑦𐑯𐑩𐑤 𐑝𐑻𐑗𐑵 𐑞𐑳𐑕 𐑥𐑸𐑛 𐑹 𐑤𐑪𐑕𐑑; 𐑯 𐑦𐑓 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟 𐑚𐑰 𐑩𐑓𐑧𐑒𐑑𐑩𐑛, 𐑞 𐑕𐑱𐑥 𐑓𐑱𐑑 𐑮𐑰𐑗𐑩𐑟 𐑷𐑤 𐑞 𐑳𐑞𐑼𐑟 𐑞𐑨𐑑 𐑥𐑱 𐑚𐑰 𐑦𐑯 𐑞 𐑖𐑦𐑐; 𐑰𐑝𐑩𐑯 𐑢𐑻 𐑞 𐑤𐑴𐑼𐑥𐑴𐑕𐑑 𐑢𐑳𐑯 𐑦𐑯𐑕𐑻𐑑𐑩𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑧𐑤𐑕𐑩𐑯.

𐑛𐑦𐑤𐑦𐑚𐑼𐑩𐑑𐑤𐑦 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑦𐑓𐑹 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤, 𐑯 𐑲𐑦𐑙 𐑞 𐑑𐑮𐑨𐑯𐑟𐑐𐑶𐑯𐑑𐑩𐑛 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟, 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑢𐑦𐑞 𐑞 𐑖𐑸𐑐 𐑝 𐑣𐑦𐑟 𐑧𐑒𐑕𐑑𐑧𐑯𐑛𐑩𐑛 𐑣𐑨𐑯𐑛, 𐑯𐑬 𐑑𐑫𐑒 𐑞 𐑐𐑮𐑦𐑕𐑲𐑕 𐑚𐑺𐑦𐑙 𐑝 𐑞 𐑕𐑳𐑯, 𐑯 𐑕𐑨𐑑𐑦𐑕𐑓𐑲𐑛 𐑞𐑨𐑑 𐑞 𐑯𐑰𐑛𐑩𐑤𐑟 𐑢𐑻 𐑦𐑜𐑟𐑨𐑒𐑑𐑤𐑦 𐑦𐑯𐑝𐑻𐑑𐑩𐑛, 𐑖𐑬𐑑𐑩𐑛 𐑬𐑑 𐑣𐑦𐑟 𐑹𐑛𐑼𐑟 𐑓 𐑞 𐑖𐑦𐑐'𐑕 𐑒𐑹𐑕 𐑑 𐑚𐑰 𐑗𐑱𐑯𐑡𐑛 𐑩𐑒𐑹𐑛𐑦𐑙𐑤𐑦. 𐑞 𐑘𐑸𐑛𐑟 𐑢𐑻 𐑣𐑸𐑛 𐑳𐑐; 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑔𐑮𐑳𐑕𐑑 𐑣𐑻 𐑳𐑯𐑛𐑷𐑯𐑑𐑩𐑛 𐑚𐑬𐑟 𐑦𐑯𐑑𐑵 𐑞 𐑩𐑐𐑴𐑟𐑦𐑙 𐑢𐑦𐑯𐑛, 𐑓 𐑞 𐑕𐑩𐑐𐑴𐑟𐑛 𐑓𐑺 𐑢𐑳𐑯 𐑣𐑨𐑛 𐑴𐑯𐑤𐑦 𐑚𐑰𐑯 𐑡𐑳𐑜𐑩𐑤𐑦𐑙 𐑣𐑻.

𐑥𐑰𐑯𐑢𐑲𐑤, 𐑢𐑪𐑑𐑧𐑝𐑼 𐑢𐑻 𐑣𐑦𐑟 𐑴𐑯 𐑕𐑰𐑒𐑮𐑩𐑑 𐑔𐑷𐑑𐑕, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑕𐑧𐑛 𐑯𐑳𐑔𐑦𐑙, 𐑚𐑳𐑑 𐑒𐑢𐑲𐑩𐑑𐑤𐑦 𐑣𐑰 𐑦𐑖𐑿𐑛 𐑷𐑤 𐑮𐑧𐑒𐑢𐑦𐑟𐑦𐑑 𐑹𐑛𐑼𐑟; 𐑢𐑲𐑤 ·𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒—𐑣𐑵 𐑦𐑯 𐑕𐑳𐑥 𐑕𐑥𐑷𐑤 𐑛𐑦𐑜𐑮𐑰 𐑕𐑰𐑥𐑛 𐑞𐑧𐑯 𐑑 𐑚𐑰 𐑖𐑺𐑦𐑙 𐑣𐑦𐑟 𐑓𐑰𐑤𐑦𐑙𐑟—𐑤𐑲𐑒𐑢𐑲𐑟 𐑳𐑯𐑥𐑻𐑥𐑼𐑦𐑙𐑤𐑦 𐑨𐑒𐑢𐑦𐑧𐑕𐑑. 𐑨𐑟 𐑓 𐑞 𐑥𐑧𐑯, 𐑞𐑴 𐑕𐑳𐑥 𐑝 𐑞𐑧𐑥 𐑤𐑴𐑤𐑦 𐑮𐑳𐑥𐑚𐑩𐑤𐑛, 𐑞𐑺 𐑓𐑽 𐑝 ·𐑱𐑣𐑨𐑚 𐑢𐑪𐑟 𐑜𐑮𐑱𐑑𐑼 𐑞𐑨𐑯 𐑞𐑺 𐑓𐑽 𐑝 ·𐑓𐑱𐑑. 𐑚𐑳𐑑 𐑨𐑟 𐑧𐑝𐑼 𐑚𐑦𐑓𐑹, 𐑞 𐑐𐑱𐑜𐑩𐑯 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑮𐑦𐑥𐑱𐑯𐑛 𐑷𐑤𐑥𐑴𐑕𐑑 𐑣𐑴𐑤𐑦 𐑳𐑯𐑦𐑥𐑐𐑮𐑧𐑕𐑑; 𐑹 𐑦𐑓 𐑦𐑥𐑐𐑮𐑧𐑕𐑑, 𐑦𐑑 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑢𐑦𐑞 𐑩 𐑕𐑻𐑑𐑩𐑯 𐑥𐑨𐑜𐑯𐑩𐑑𐑦𐑟𐑩𐑥 𐑖𐑪𐑑 𐑦𐑯𐑑𐑵 𐑞𐑺 𐑒𐑩𐑯𐑡𐑰𐑯𐑾𐑤 𐑣𐑸𐑑𐑕 𐑓𐑮𐑪𐑥 𐑦𐑯𐑓𐑤𐑧𐑒𐑕𐑩𐑚𐑩𐑤 ·𐑱𐑣𐑨𐑚'𐑟.

𐑓 𐑩 𐑕𐑐𐑱𐑕 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑢𐑷𐑒𐑑 𐑞 𐑛𐑧𐑒 𐑦𐑯 𐑮𐑴𐑤𐑦𐑙 𐑮𐑧𐑝𐑼𐑦𐑟. 𐑚𐑳𐑑 𐑗𐑭𐑯𐑕𐑦𐑙 𐑑 𐑕𐑤𐑦𐑐 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑲𐑝𐑼𐑦 𐑣𐑰𐑤, 𐑣𐑰 𐑕𐑷 𐑞 𐑒𐑮𐑳𐑖𐑑 𐑒𐑪𐑐𐑼 𐑕𐑲𐑑-𐑑𐑿𐑚𐑟 𐑝 𐑞 𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑 𐑣𐑰 𐑣𐑨𐑛 𐑞 𐑛𐑱 𐑚𐑦𐑓𐑹 𐑛𐑨𐑖𐑑 𐑑 𐑞 𐑛𐑧𐑒.

“𐑞𐑬 𐑐𐑫𐑼, 𐑐𐑮𐑬𐑛 𐑣𐑧𐑝𐑩𐑯-𐑜𐑱𐑟𐑼 𐑯 𐑕𐑳𐑯'𐑟 𐑐𐑲𐑤𐑩𐑑! 𐑘𐑧𐑕𐑑𐑼𐑛𐑱 𐑲 𐑮𐑧𐑒𐑑 𐑞𐑰, 𐑯 𐑑-𐑛𐑱 𐑞 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟 𐑢𐑫𐑛 𐑓𐑱𐑯 𐑣𐑨𐑝 𐑮𐑧𐑒𐑑 𐑥𐑰. 𐑕𐑴, 𐑕𐑴. 𐑚𐑳𐑑 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑤𐑹𐑛 𐑴𐑝𐑼 𐑞 𐑤𐑧𐑝𐑩𐑤 𐑤𐑴𐑛𐑕𐑑𐑴𐑯 𐑘𐑧𐑑. ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒—𐑩 𐑤𐑭𐑯𐑕 𐑢𐑦𐑞𐑬𐑑 𐑩 𐑐𐑴𐑤; 𐑩 𐑑𐑪𐑐-𐑥𐑷𐑤, 𐑯 𐑞 𐑕𐑥𐑷𐑤𐑩𐑕𐑑 𐑝 𐑞 𐑕𐑱𐑤-𐑥𐑱𐑒𐑼'𐑟 𐑯𐑰𐑛𐑩𐑤𐑟. 𐑒𐑢𐑦𐑒!”

𐑩𐑒𐑕𐑧𐑕𐑼𐑦, 𐑐𐑼𐑣𐑨𐑐𐑕, 𐑑 𐑞 𐑦𐑥𐑐𐑳𐑤𐑕 𐑛𐑦𐑒𐑑𐑱𐑑𐑦𐑙 𐑞 𐑔𐑦𐑙 𐑣𐑰 𐑢𐑪𐑟 𐑯𐑬 𐑩𐑚𐑬𐑑 𐑑 𐑛𐑵, 𐑢𐑻 𐑕𐑻𐑑𐑩𐑯 𐑐𐑮𐑵𐑛𐑧𐑯𐑖𐑩𐑤 𐑥𐑴𐑑𐑦𐑝𐑟, 𐑣𐑵𐑟 𐑪𐑚𐑡𐑧𐑒𐑑 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑑 𐑮𐑦𐑝𐑲𐑝 𐑞 𐑕𐑐𐑦𐑮𐑦𐑑𐑕 𐑝 𐑣𐑦𐑟 𐑒𐑮𐑵 𐑚𐑲 𐑩 𐑕𐑑𐑮𐑴𐑒 𐑝 𐑣𐑦𐑟 𐑕𐑳𐑚𐑑𐑲𐑤 𐑕𐑒𐑦𐑤, 𐑦𐑯 𐑩 𐑥𐑨𐑑𐑼 𐑕𐑴 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑨𐑟 𐑞𐑨𐑑 𐑝 𐑞 𐑦𐑯𐑝𐑻𐑑𐑩𐑛 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟. 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑢𐑧𐑤 𐑯𐑿 𐑞𐑨𐑑 𐑑 𐑕𐑑𐑽 𐑚𐑲 𐑑𐑮𐑨𐑯𐑟𐑐𐑶𐑯𐑑𐑩𐑛 𐑯𐑰𐑛𐑩𐑤𐑟, 𐑞𐑴 𐑒𐑤𐑳𐑥𐑟𐑦𐑤𐑦 𐑐𐑮𐑨𐑒𐑑𐑦𐑒𐑩𐑚𐑩𐑤, 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑩 𐑔𐑦𐑙 𐑑 𐑚𐑰 𐑐𐑭𐑕𐑑 𐑴𐑝𐑼 𐑚𐑲 𐑕𐑵𐑐𐑼𐑕𐑑𐑦𐑖𐑩𐑕 𐑕𐑱𐑤𐑼𐑟, 𐑢𐑦𐑞𐑬𐑑 𐑕𐑳𐑥 𐑖𐑳𐑛𐑼𐑦𐑙𐑟 𐑯 𐑰𐑝𐑩𐑤 𐑐𐑹𐑑𐑧𐑯𐑑𐑕.

“𐑥𐑧𐑯,” 𐑕𐑧𐑛 𐑣𐑰, 𐑕𐑑𐑧𐑛𐑦𐑤𐑦 𐑑𐑻𐑯𐑦𐑙 𐑩𐑐𐑪𐑯 𐑞 𐑒𐑮𐑵, 𐑨𐑟 𐑞 𐑥𐑱𐑑 𐑣𐑨𐑯𐑛𐑩𐑛 𐑣𐑦𐑥 𐑞 𐑔𐑦𐑙𐑟 𐑣𐑰 𐑣𐑨𐑛 𐑛𐑦𐑥𐑭𐑯𐑛𐑩𐑛, “𐑥𐑲 𐑥𐑧𐑯, 𐑞 𐑔𐑳𐑯𐑛𐑼 𐑑𐑻𐑯𐑛 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚'𐑟 𐑯𐑰𐑛𐑩𐑤𐑟; 𐑚𐑳𐑑 𐑬𐑑 𐑝 𐑞𐑦𐑕 𐑚𐑦𐑑 𐑝 𐑕𐑑𐑰𐑤 ·𐑱𐑣𐑨𐑚 𐑒𐑨𐑯 𐑥𐑱𐑒 𐑢𐑳𐑯 𐑝 𐑣𐑦𐑟 𐑴𐑯, 𐑞𐑨𐑑 𐑢𐑦𐑤 𐑐𐑶𐑯𐑑 𐑨𐑟 𐑑𐑮𐑵 𐑨𐑟 𐑧𐑯𐑦.”

𐑩𐑚𐑨𐑖𐑑 𐑜𐑤𐑭𐑯𐑕𐑩𐑟 𐑝 𐑕𐑻𐑝𐑲𐑤 𐑢𐑳𐑯𐑛𐑼 𐑢𐑻 𐑦𐑒𐑕𐑗𐑱𐑯𐑡𐑛 𐑚𐑲 𐑞 𐑕𐑱𐑤𐑼𐑟, 𐑨𐑟 𐑞𐑦𐑕 𐑢𐑪𐑟 𐑕𐑧𐑛; 𐑯 𐑢𐑦𐑞 𐑓𐑨𐑕𐑦𐑯𐑱𐑑𐑩𐑛 𐑲𐑟 𐑞𐑱 𐑩𐑢𐑱𐑑𐑩𐑛 𐑢𐑪𐑑𐑧𐑝𐑼 𐑥𐑨𐑡𐑦𐑒 𐑥𐑲𐑑 𐑓𐑪𐑤𐑴. 𐑚𐑳𐑑 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑤𐑫𐑒𐑑 𐑩𐑢𐑱.

𐑢𐑦𐑞 𐑩 𐑚𐑤𐑴 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑪𐑐-𐑥𐑷𐑤 ·𐑱𐑣𐑨𐑚 𐑯𐑪𐑒𐑑 𐑪𐑓 𐑞 𐑕𐑑𐑰𐑤 𐑣𐑧𐑛 𐑝 𐑞 𐑤𐑭𐑯𐑕, 𐑯 𐑞𐑧𐑯 𐑣𐑨𐑯𐑛𐑦𐑙 𐑑 𐑞 𐑥𐑱𐑑 𐑞 𐑤𐑪𐑙 𐑲𐑼𐑯 𐑮𐑪𐑛 𐑮𐑦𐑥𐑱𐑯𐑦𐑙, 𐑚𐑨𐑛 𐑣𐑦𐑥 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑳𐑐𐑮𐑲𐑑, 𐑢𐑦𐑞𐑬𐑑 𐑦𐑑𐑕 𐑑𐑳𐑗𐑦𐑙 𐑞 𐑛𐑧𐑒. 𐑞𐑧𐑯, 𐑢𐑦𐑞 𐑞 𐑥𐑷𐑤, 𐑭𐑓𐑑𐑼 𐑮𐑦𐑐𐑰𐑑𐑩𐑛𐑤𐑦 𐑕𐑥𐑲𐑑𐑦𐑙 𐑞 𐑳𐑐𐑼 𐑧𐑯𐑛 𐑝 𐑞𐑦𐑕 𐑲𐑼𐑯 𐑮𐑪𐑛, 𐑣𐑰 𐑐𐑤𐑱𐑕𐑑 𐑞 𐑚𐑤𐑳𐑯𐑑𐑩𐑛 𐑯𐑰𐑛𐑩𐑤 𐑧𐑯𐑛𐑢𐑲𐑟 𐑪𐑯 𐑞 𐑑𐑪𐑐 𐑝 𐑦𐑑, 𐑯 𐑤𐑧𐑕 𐑕𐑑𐑮𐑪𐑙𐑤𐑦 𐑣𐑨𐑥𐑼𐑛 𐑞𐑨𐑑, 𐑕𐑧𐑝𐑼𐑩𐑤 𐑑𐑲𐑥𐑟, 𐑞 𐑥𐑱𐑑 𐑕𐑑𐑦𐑤 𐑣𐑴𐑤𐑛𐑦𐑙 𐑞 𐑮𐑪𐑛 𐑨𐑟 𐑚𐑦𐑓𐑹. 𐑞𐑧𐑯 𐑜𐑴𐑦𐑙 𐑔𐑮𐑵 𐑕𐑳𐑥 𐑕𐑥𐑷𐑤 𐑕𐑑𐑮𐑱𐑯𐑡 𐑥𐑴𐑖𐑩𐑯𐑟 𐑢𐑦𐑞 𐑦𐑑—𐑢𐑧𐑞𐑼 𐑦𐑯𐑛𐑦𐑕𐑐𐑧𐑯𐑕𐑩𐑚𐑩𐑤 𐑑 𐑞 𐑥𐑨𐑜𐑯𐑩𐑑𐑲𐑟𐑦𐑙 𐑝 𐑞 𐑕𐑑𐑰𐑤, 𐑹 𐑥𐑽𐑤𐑦 𐑦𐑯𐑑𐑧𐑯𐑛𐑩𐑛 𐑑 𐑷𐑜𐑥𐑧𐑯𐑑 𐑞 𐑷 𐑝 𐑞 𐑒𐑮𐑵, 𐑦𐑟 𐑳𐑯𐑕𐑻𐑑𐑩𐑯—𐑣𐑰 𐑒𐑷𐑤𐑛 𐑓 𐑤𐑦𐑯𐑦𐑯 𐑔𐑮𐑧𐑛; 𐑯 𐑥𐑵𐑝𐑦𐑙 𐑑 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤, 𐑕𐑤𐑦𐑐𐑑 𐑬𐑑 𐑞 𐑑𐑵 𐑮𐑦𐑝𐑻𐑕𐑑 𐑯𐑰𐑛𐑩𐑤𐑟 𐑞𐑺, 𐑯 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤𐑦 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑞 𐑕𐑱𐑤-𐑯𐑰𐑛𐑩𐑤 𐑚𐑲 𐑦𐑑𐑕 𐑥𐑦𐑛𐑩𐑤, 𐑴𐑝𐑼 𐑢𐑳𐑯 𐑝 𐑞 𐑒𐑳𐑥𐑐𐑩𐑕-𐑒𐑸𐑛𐑟. 𐑨𐑑 𐑓𐑻𐑕𐑑, 𐑞 𐑕𐑑𐑰𐑤 𐑢𐑧𐑯𐑑 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛, 𐑒𐑢𐑦𐑝𐑼𐑦𐑙 𐑯 𐑝𐑲𐑚𐑮𐑱𐑑𐑦𐑙 𐑨𐑑 𐑲𐑞𐑼 𐑧𐑯𐑛; 𐑚𐑳𐑑 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑦𐑑 𐑕𐑧𐑑𐑩𐑤𐑛 𐑑 𐑦𐑑𐑕 𐑐𐑤𐑱𐑕, 𐑢𐑧𐑯 ·𐑱𐑣𐑨𐑚, 𐑣𐑵 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑦𐑯𐑑𐑧𐑯𐑑𐑤𐑦 𐑢𐑪𐑗𐑦𐑙 𐑓 𐑞𐑦𐑕 𐑮𐑦𐑟𐑳𐑤𐑑, 𐑕𐑑𐑧𐑐𐑑 𐑓𐑮𐑨𐑙𐑒𐑤𐑦 𐑚𐑨𐑒 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤, 𐑯 𐑐𐑶𐑯𐑑𐑦𐑙 𐑣𐑦𐑟 𐑕𐑑𐑮𐑧𐑗𐑑 𐑸𐑥 𐑑𐑩𐑢𐑹𐑛𐑟 𐑦𐑑, 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛,—“𐑤𐑫𐑒 𐑘𐑰, 𐑓 𐑘𐑹𐑕𐑧𐑤𐑝𐑟, 𐑦𐑓 ·𐑱𐑣𐑨𐑚 𐑚𐑰 𐑯𐑪𐑑 𐑤𐑹𐑛 𐑝 𐑞 𐑤𐑧𐑝𐑩𐑤 𐑤𐑴𐑛𐑕𐑑𐑴𐑯! 𐑞 𐑕𐑳𐑯 𐑦𐑟 ·𐑰𐑕𐑑, 𐑯 𐑞𐑨𐑑 𐑒𐑳𐑥𐑐𐑩𐑕 𐑕𐑢𐑺𐑟 𐑦𐑑!”

𐑢𐑳𐑯 𐑭𐑓𐑑𐑼 𐑩𐑯𐑳𐑞𐑼 𐑞𐑱 𐑐𐑽𐑛 𐑦𐑯, 𐑓 𐑯𐑳𐑔𐑦𐑙 𐑚𐑳𐑑 𐑞𐑺 𐑴𐑯 𐑲𐑟 𐑒𐑫𐑛 𐑐𐑼𐑕𐑢𐑱𐑛 𐑕𐑳𐑗 𐑦𐑜𐑯𐑼𐑩𐑯𐑕 𐑨𐑟 𐑞𐑺𐑟, 𐑯 𐑢𐑳𐑯 𐑭𐑓𐑑𐑼 𐑩𐑯𐑳𐑞𐑼 𐑞𐑱 𐑕𐑤𐑳𐑙𐑒 𐑩𐑢𐑱.

𐑦𐑯 𐑣𐑦𐑟 𐑓𐑲𐑼𐑦 𐑲𐑟 𐑝 𐑕𐑒𐑹𐑯 𐑯 𐑑𐑮𐑲𐑳𐑥𐑓, 𐑿 𐑞𐑧𐑯 𐑕𐑷 ·𐑱𐑣𐑨𐑚 𐑦𐑯 𐑷𐑤 𐑣𐑦𐑟 𐑓𐑱𐑑𐑩𐑤 𐑐𐑮𐑲𐑛.
