·𐑗𐑨𐑐𐑑𐑼 21. 𐑜𐑴𐑦𐑙 ·𐑩𐑚𐑹𐑛.

𐑦𐑑 𐑢𐑪𐑟 𐑯𐑽𐑤𐑦 𐑕𐑦𐑒𐑕 𐑴'𐑒𐑤𐑪𐑒, 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑜𐑮𐑱 𐑦𐑥𐑐𐑻𐑓𐑦𐑒𐑑 𐑥𐑦𐑕𐑑𐑦 𐑛𐑷𐑯, 𐑢𐑧𐑯 𐑢𐑰 𐑛𐑮𐑵 𐑯𐑲 𐑞 𐑢𐑹𐑓.

“𐑞𐑺 𐑸 𐑕𐑳𐑥 𐑕𐑱𐑤𐑼𐑟 𐑮𐑳𐑯𐑦𐑙 𐑩𐑣𐑧𐑛 𐑞𐑺, 𐑦𐑓 𐑲 𐑕𐑰 𐑮𐑲𐑑,” 𐑕𐑧𐑛 𐑲 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, “𐑦𐑑 𐑒𐑭𐑯𐑑 𐑚𐑰 𐑖𐑨𐑛𐑴𐑟; 𐑖𐑰'𐑟 𐑪𐑓 𐑚𐑲 𐑕𐑳𐑯𐑮𐑲𐑟, 𐑲 𐑜𐑧𐑕; 𐑒𐑳𐑥 𐑪𐑯!”

“𐑩𐑝𐑭𐑕𐑑!” 𐑒𐑮𐑲𐑛 𐑩 𐑝𐑶𐑕, 𐑣𐑵𐑟 𐑴𐑯𐑼 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥 𐑒𐑳𐑥𐑦𐑙 𐑒𐑤𐑴𐑕 𐑚𐑦𐑣𐑲𐑯𐑛 𐑳𐑕, 𐑤𐑱𐑛 𐑩 𐑣𐑨𐑯𐑛 𐑩𐑐𐑪𐑯 𐑚𐑴𐑔 𐑬𐑼 𐑖𐑴𐑤𐑛𐑼𐑟, 𐑯 𐑞𐑧𐑯 𐑦𐑯𐑕𐑦𐑯𐑿𐑱𐑑𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑚𐑦𐑑𐑢𐑰𐑯 𐑳𐑕, 𐑕𐑑𐑫𐑛 𐑕𐑑𐑵𐑐𐑦𐑙 𐑓𐑹𐑢𐑼𐑛 𐑩 𐑤𐑦𐑑𐑩𐑤, 𐑦𐑯 𐑞 𐑳𐑯𐑕𐑻𐑑𐑩𐑯 𐑑𐑢𐑲𐑤𐑲𐑑, 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑐𐑽𐑦𐑙 𐑓𐑮𐑪𐑥 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑑 𐑥𐑰. 𐑦𐑑 𐑢𐑪𐑟 ·𐑦𐑤𐑲𐑡𐑩.

“𐑜𐑴𐑦𐑙 𐑩𐑚𐑹𐑛?”

“𐑣𐑨𐑯𐑛𐑟 𐑪𐑓, 𐑢𐑦𐑤 𐑿,” 𐑕𐑧𐑛 𐑲.

“𐑤𐑫𐑒𐑰 𐑣𐑽,” 𐑕𐑧𐑛 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑖𐑱𐑒𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, “𐑜𐑴 '𐑢𐑱!”

“𐑱𐑯𐑑 𐑜𐑴𐑦𐑙 𐑩𐑚𐑹𐑛, 𐑞𐑧𐑯?”

“𐑘𐑧𐑕, 𐑢𐑰 𐑸,” 𐑕𐑧𐑛 𐑲, “𐑚𐑳𐑑 𐑢𐑪𐑑 𐑚𐑦𐑟𐑯𐑩𐑕 𐑦𐑟 𐑞𐑨𐑑 𐑝 𐑘𐑹𐑟? 𐑛𐑵 𐑿 𐑯𐑴, ·𐑥𐑼 ·𐑦𐑤𐑲𐑡𐑩, 𐑞𐑨𐑑 𐑲 𐑒𐑩𐑯𐑕𐑦𐑛𐑼 𐑿 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑦𐑥𐑐𐑻𐑑𐑦𐑯𐑩𐑯𐑑?”

“𐑯𐑴, 𐑯𐑴, 𐑯𐑴; 𐑲 𐑢𐑪𐑟𐑩𐑯𐑑 𐑩𐑢𐑺 𐑝 𐑞𐑨𐑑,” 𐑕𐑧𐑛 ·𐑦𐑤𐑲𐑡𐑩, 𐑕𐑤𐑴𐑤𐑦 𐑯 𐑢𐑳𐑯𐑛𐑼𐑦𐑙𐑤𐑦 𐑤𐑫𐑒𐑦𐑙 𐑓𐑮𐑪𐑥 𐑥𐑰 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑢𐑦𐑞 𐑞 𐑥𐑴𐑕𐑑 𐑳𐑯𐑩𐑒𐑬𐑯𐑑𐑩𐑚𐑩𐑤 𐑜𐑤𐑭𐑯𐑕𐑩𐑟.

“·𐑦𐑤𐑲𐑡𐑩,” 𐑕𐑧𐑛 𐑲, “𐑿 𐑢𐑦𐑤 𐑩𐑚𐑤𐑲𐑡 𐑥𐑲 𐑓𐑮𐑧𐑯𐑛 𐑯 𐑥𐑰 𐑚𐑲 𐑢𐑦𐑞𐑛𐑮𐑷𐑦𐑙. 𐑢𐑰 𐑸 𐑜𐑴𐑦𐑙 𐑑 𐑞 ·𐑦𐑯𐑛𐑾𐑯 𐑯 ·𐑐𐑩𐑕𐑦𐑓𐑦𐑒 ·𐑴𐑖𐑩𐑯𐑟, 𐑯 𐑢𐑫𐑛 𐑐𐑮𐑦𐑓𐑻 𐑯𐑪𐑑 𐑑 𐑚𐑰 𐑛𐑦𐑑𐑱𐑯𐑛.”

“𐑘𐑰 𐑚𐑰, 𐑚𐑰 𐑘𐑰? 𐑒𐑳𐑥𐑦𐑙 𐑚𐑨𐑒 𐑩𐑓𐑹 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑?”

“𐑣𐑰'𐑟 𐑒𐑮𐑨𐑒𐑑, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜,” 𐑕𐑧𐑛 𐑲, “𐑒𐑳𐑥 𐑪𐑯.”

“𐑣𐑪𐑤𐑴!” 𐑒𐑮𐑲𐑛 𐑕𐑑𐑱𐑖𐑩𐑯𐑼𐑦 ·𐑦𐑤𐑲𐑡𐑩, 𐑣𐑱𐑤𐑦𐑙 𐑳𐑕 𐑢𐑧𐑯 𐑢𐑰 𐑣𐑨𐑛 𐑮𐑦𐑥𐑵𐑝𐑛 𐑩 𐑓𐑿 𐑐𐑱𐑕𐑩𐑟.

“𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛 𐑣𐑦𐑥,” 𐑕𐑧𐑛 𐑲, “·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑒𐑳𐑥 𐑪𐑯.”

𐑚𐑳𐑑 𐑣𐑰 𐑕𐑑𐑴𐑤 𐑳𐑐 𐑑 𐑳𐑕 𐑩𐑜𐑱𐑯, 𐑯 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑒𐑤𐑨𐑐𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑪𐑯 𐑥𐑲 𐑖𐑴𐑤𐑛𐑼, 𐑕𐑧𐑛—“𐑛𐑦𐑛 𐑘𐑰 𐑕𐑰 𐑧𐑯𐑦𐑔𐑦𐑙 𐑤𐑫𐑒𐑦𐑙 𐑤𐑲𐑒 𐑥𐑧𐑯 𐑜𐑴𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞𐑨𐑑 𐑖𐑦𐑐 𐑩 𐑢𐑲𐑤 𐑩𐑜𐑴?”

𐑕𐑑𐑮𐑳𐑒 𐑚𐑲 𐑞𐑦𐑕 𐑐𐑤𐑱𐑯 𐑥𐑨𐑑𐑼-𐑝-𐑓𐑨𐑒𐑑 𐑒𐑢𐑧𐑕𐑗𐑩𐑯, 𐑲 𐑭𐑯𐑕𐑼𐑛, 𐑕𐑱𐑦𐑙, “𐑘𐑧𐑕, 𐑲 𐑔𐑷𐑑 𐑲 𐑛𐑦𐑛 𐑕𐑰 𐑓𐑹 𐑹 𐑓𐑲𐑝 𐑥𐑧𐑯; 𐑚𐑳𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑑𐑵 𐑛𐑦𐑥 𐑑 𐑚𐑰 𐑖𐑫𐑼.”

“𐑝𐑧𐑮𐑦 𐑛𐑦𐑥, 𐑝𐑧𐑮𐑦 𐑛𐑦𐑥,” 𐑕𐑧𐑛 ·𐑦𐑤𐑲𐑡𐑩. “𐑥𐑹𐑯𐑦𐑙 𐑑 𐑘𐑰.”

𐑢𐑳𐑯𐑕 𐑥𐑹 𐑢𐑰 𐑒𐑢𐑦𐑑𐑩𐑛 𐑣𐑦𐑥; 𐑚𐑳𐑑 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑣𐑰 𐑒𐑱𐑥 𐑕𐑪𐑓𐑑𐑤𐑦 𐑭𐑓𐑑𐑼 𐑳𐑕; 𐑯 𐑑𐑳𐑗𐑦𐑙 𐑥𐑲 𐑖𐑴𐑤𐑛𐑼 𐑩𐑜𐑱𐑯, 𐑕𐑧𐑛, “𐑕𐑰 𐑦𐑓 𐑿 𐑒𐑨𐑯 𐑓𐑲𐑯𐑛 '𐑩𐑥 𐑯𐑬, 𐑢𐑦𐑤 𐑘𐑰?

“𐑓𐑲𐑯𐑛 𐑣𐑵?”

“𐑥𐑹𐑯𐑦𐑙 𐑑 𐑘𐑰! 𐑥𐑹𐑯𐑦𐑙 𐑑 𐑘𐑰!” 𐑣𐑰 𐑮𐑰𐑡𐑶𐑯𐑛, 𐑩𐑜𐑱𐑯 𐑥𐑵𐑝𐑦𐑙 𐑪𐑓. “𐑴! 𐑲 𐑢𐑪𐑟 𐑜𐑴𐑦𐑙 𐑑 𐑢𐑹𐑯 𐑘𐑰 𐑩𐑜𐑱𐑯𐑕𐑑—𐑚𐑳𐑑 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛, 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛—𐑦𐑑'𐑕 𐑷𐑤 𐑢𐑳𐑯, 𐑷𐑤 𐑦𐑯 𐑞 𐑓𐑨𐑥𐑤𐑦 𐑑𐑵;—𐑖𐑸𐑐 𐑓𐑮𐑪𐑕𐑑 𐑞𐑦𐑕 𐑥𐑹𐑯𐑦𐑙, 𐑱𐑯𐑑 𐑦𐑑? 𐑜𐑫𐑛-𐑚𐑲 𐑑 𐑘𐑰. 𐑖𐑭𐑯𐑑 𐑕𐑰 𐑘𐑰 𐑩𐑜𐑱𐑯 𐑝𐑧𐑮𐑦 𐑕𐑵𐑯, 𐑲 𐑜𐑧𐑕; 𐑩𐑯𐑤𐑧𐑕 𐑦𐑑'𐑕 𐑚𐑦𐑓𐑹 𐑞 ·𐑜𐑮𐑨𐑯𐑛 ·𐑡𐑫𐑼𐑦.” 𐑯 𐑢𐑦𐑞 𐑞𐑰𐑟 𐑒𐑮𐑨𐑒𐑑 𐑢𐑻𐑛𐑟 𐑣𐑰 𐑓𐑲𐑯𐑩𐑤𐑦 𐑛𐑦𐑐𐑸𐑑𐑩𐑛, 𐑤𐑰𐑝𐑦𐑙 𐑥𐑰, 𐑓 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑦𐑯 𐑯𐑴 𐑕𐑥𐑷𐑤 𐑢𐑳𐑯𐑛𐑼𐑥𐑩𐑯𐑑 𐑨𐑑 𐑣𐑦𐑟 𐑓𐑮𐑨𐑯𐑑𐑦𐑒 𐑦𐑥𐑐𐑿𐑛𐑩𐑯𐑕.

𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑕𐑑𐑧𐑐𐑦𐑙 𐑪𐑯 𐑚𐑹𐑛 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑢𐑰 𐑓𐑬𐑯𐑛 𐑧𐑝𐑮𐑦𐑔𐑦𐑙 𐑦𐑯 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑒𐑢𐑲𐑩𐑑, 𐑯𐑪𐑑 𐑩 𐑕𐑴𐑤 𐑥𐑵𐑝𐑦𐑙. 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑧𐑯𐑑𐑮𐑩𐑯𐑕 𐑢𐑪𐑟 𐑤𐑪𐑒𐑑 𐑢𐑦𐑞𐑦𐑯; 𐑞 𐑣𐑨𐑗𐑩𐑟 𐑢𐑻 𐑷𐑤 𐑪𐑯, 𐑯 𐑤𐑳𐑥𐑚𐑼𐑛 𐑢𐑦𐑞 𐑒𐑶𐑤𐑟 𐑝 𐑮𐑦𐑜𐑦𐑙. 𐑜𐑴𐑦𐑙 𐑓𐑹𐑢𐑼𐑛 𐑑 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤, 𐑢𐑰 𐑓𐑬𐑯𐑛 𐑞 𐑕𐑤𐑲𐑛 𐑝 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤 𐑴𐑐𐑩𐑯. 𐑕𐑰𐑦𐑙 𐑩 𐑤𐑲𐑑, 𐑢𐑰 𐑢𐑧𐑯𐑑 𐑛𐑬𐑯, 𐑯 𐑓𐑬𐑯𐑛 𐑴𐑯𐑤𐑦 𐑩𐑯 𐑴𐑤𐑛 𐑮𐑦𐑜𐑼 𐑞𐑺, 𐑮𐑨𐑐𐑑 𐑦𐑯 𐑩 𐑑𐑨𐑑𐑼𐑛 𐑐𐑰-𐑡𐑨𐑒𐑩𐑑. 𐑣𐑰 𐑢𐑪𐑟 𐑔𐑮𐑴𐑯 𐑨𐑑 𐑣𐑴𐑤 𐑤𐑧𐑙𐑔 𐑩𐑐𐑪𐑯 𐑑𐑵 𐑗𐑧𐑕𐑑𐑕, 𐑣𐑦𐑟 𐑓𐑱𐑕 𐑛𐑬𐑯𐑢𐑼𐑛𐑟 𐑯 𐑦𐑯𐑒𐑤𐑴𐑟𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑓𐑴𐑤𐑛𐑩𐑛 𐑸𐑥𐑟. 𐑞 𐑐𐑮𐑩𐑓𐑬𐑯𐑛𐑩𐑕𐑑 𐑕𐑤𐑳𐑥𐑚𐑼 𐑕𐑤𐑧𐑐𐑑 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥.

“𐑞𐑴𐑟 𐑕𐑱𐑤𐑼𐑟 𐑢𐑰 𐑕𐑷, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑢𐑺 𐑒𐑨𐑯 𐑞𐑱 𐑣𐑨𐑝 𐑜𐑪𐑯 𐑑?” 𐑕𐑧𐑛 𐑲, 𐑤𐑫𐑒𐑦𐑙 𐑛𐑿𐑚𐑾𐑕𐑤𐑦 𐑨𐑑 𐑞 𐑕𐑤𐑰𐑐𐑼. 𐑚𐑳𐑑 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑞𐑨𐑑, 𐑢𐑧𐑯 𐑪𐑯 𐑞 𐑢𐑹𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑨𐑑 𐑷𐑤 𐑯𐑴𐑑𐑦𐑕𐑑 𐑢𐑪𐑑 𐑲 𐑯𐑬 𐑩𐑤𐑵𐑛𐑩𐑛 𐑑; 𐑣𐑧𐑯𐑕 𐑲 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑔𐑷𐑑 𐑥𐑲𐑕𐑧𐑤𐑓 𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑪𐑐𐑑𐑦𐑒𐑤𐑦 𐑛𐑦𐑕𐑰𐑝𐑛 𐑦𐑯 𐑞𐑨𐑑 𐑥𐑨𐑑𐑼, 𐑢𐑻 𐑦𐑑 𐑯𐑪𐑑 𐑓 ·𐑦𐑤𐑲𐑡𐑩'𐑟 𐑳𐑞𐑼𐑢𐑲𐑟 𐑦𐑯𐑦𐑒𐑕𐑐𐑤𐑦𐑒𐑩𐑚𐑩𐑤 𐑒𐑢𐑧𐑕𐑗𐑩𐑯. 𐑚𐑳𐑑 𐑲 𐑚𐑰𐑑 𐑞 𐑔𐑦𐑙 𐑛𐑬𐑯; 𐑯 𐑩𐑜𐑱𐑯 𐑥𐑸𐑒𐑦𐑙 𐑞 𐑕𐑤𐑰𐑐𐑼, 𐑡𐑪𐑒𐑿𐑤𐑼𐑤𐑦 𐑣𐑦𐑯𐑑𐑩𐑛 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑞𐑨𐑑 𐑐𐑼𐑣𐑨𐑐𐑕 𐑢𐑰 𐑣𐑨𐑛 𐑚𐑧𐑕𐑑 𐑕𐑦𐑑 𐑳𐑐 𐑢𐑦𐑞 𐑞 𐑚𐑪𐑛𐑦; 𐑑𐑧𐑤𐑦𐑙 𐑣𐑦𐑥 𐑑 𐑦𐑕𐑑𐑨𐑚𐑤𐑦𐑖 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑩𐑒𐑹𐑛𐑦𐑙𐑤𐑦. 𐑣𐑰 𐑐𐑫𐑑 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑤𐑰𐑐𐑼'𐑟 𐑮𐑽, 𐑨𐑟 𐑞𐑴 𐑓𐑰𐑤𐑦𐑙 𐑦𐑓 𐑦𐑑 𐑢𐑪𐑟 𐑕𐑪𐑓𐑑 𐑦𐑯𐑳𐑓; 𐑯 𐑞𐑧𐑯, 𐑢𐑦𐑞𐑬𐑑 𐑥𐑹 𐑩𐑛𐑵, 𐑕𐑨𐑑 𐑒𐑢𐑲𐑩𐑑𐑤𐑦 𐑛𐑬𐑯 𐑞𐑺.

“𐑜𐑮𐑱𐑖𐑩𐑕! ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑛𐑴𐑯𐑑 𐑕𐑦𐑑 𐑞𐑺,” 𐑕𐑧𐑛 𐑲.

“𐑴! ·𐑐𐑧𐑮𐑦 𐑛𐑵𐑛 𐑕𐑰𐑑,” 𐑕𐑧𐑛 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, “𐑥𐑲 𐑒𐑳𐑯𐑑𐑮𐑦 𐑢𐑱; 𐑢𐑴𐑯𐑑 𐑣𐑻𐑑 𐑣𐑦𐑥 𐑓𐑱𐑕.”

“𐑓𐑱𐑕!” 𐑕𐑧𐑛 𐑲, “𐑒𐑷𐑤 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑓𐑱𐑕? 𐑝𐑧𐑮𐑦 𐑚𐑦𐑯𐑧𐑝𐑩𐑤𐑩𐑯𐑑 𐑒𐑬𐑯𐑑𐑩𐑯𐑩𐑯𐑕 𐑞𐑧𐑯; 𐑚𐑳𐑑 𐑣𐑬 𐑣𐑸𐑛 𐑣𐑰 𐑚𐑮𐑰𐑞𐑟, 𐑣𐑰'𐑟 𐑣𐑰𐑝𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓; 𐑜𐑧𐑑 𐑪𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑿 𐑸 𐑣𐑧𐑝𐑦, 𐑦𐑑'𐑕 𐑜𐑮𐑲𐑯𐑛𐑦𐑙 𐑞 𐑓𐑱𐑕 𐑝 𐑞 𐑐𐑫𐑼. 𐑜𐑧𐑑 𐑪𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜! 𐑤𐑫𐑒, 𐑣𐑰'𐑤 𐑑𐑢𐑦𐑗 𐑿 𐑪𐑓 𐑕𐑵𐑯. 𐑲 𐑢𐑳𐑯𐑛𐑼 𐑣𐑰 𐑛𐑴𐑯𐑑 𐑢𐑱𐑒.”

·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑮𐑦𐑥𐑵𐑝𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑑 𐑡𐑳𐑕𐑑 𐑚𐑦𐑘𐑪𐑯𐑛 𐑞 𐑣𐑧𐑛 𐑝 𐑞 𐑕𐑤𐑰𐑐𐑼, 𐑯 𐑤𐑲𐑑𐑩𐑛 𐑣𐑦𐑟 𐑑𐑪𐑥𐑩𐑣𐑷𐑒 𐑐𐑲𐑐. 𐑲 𐑕𐑨𐑑 𐑨𐑑 𐑞 𐑓𐑰𐑑. 𐑢𐑰 𐑒𐑧𐑐𐑑 𐑞 𐑐𐑲𐑐 𐑐𐑭𐑕𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑕𐑤𐑰𐑐𐑼, 𐑓𐑮𐑪𐑥 𐑢𐑳𐑯 𐑑 𐑞 𐑳𐑞𐑼. 𐑥𐑰𐑯𐑢𐑲𐑤, 𐑩𐑐𐑪𐑯 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑦𐑙 𐑣𐑦𐑥 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑮𐑴𐑒𐑩𐑯 𐑓𐑨𐑖𐑩𐑯, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑜𐑱𐑝 𐑥𐑰 𐑑 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑞𐑨𐑑, 𐑦𐑯 𐑣𐑦𐑟 𐑤𐑨𐑯𐑛, 𐑴𐑦𐑙 𐑑 𐑞 𐑨𐑚𐑕𐑩𐑯𐑕 𐑝 𐑕𐑧𐑑𐑰𐑟 𐑯 𐑕𐑴𐑓𐑩𐑟 𐑝 𐑷𐑤 𐑕𐑹𐑑𐑕, 𐑞 𐑒𐑦𐑙, 𐑗𐑰𐑓𐑕, 𐑯 𐑜𐑮𐑱𐑑 𐑐𐑰𐑐𐑩𐑤 𐑡𐑧𐑯𐑼𐑩𐑤𐑦, 𐑢𐑻 𐑦𐑯 𐑞 𐑒𐑳𐑕𐑑𐑩𐑥 𐑝 𐑓𐑨𐑑𐑩𐑯𐑦𐑙 𐑕𐑳𐑥 𐑝 𐑞 𐑤𐑴𐑼 𐑹𐑛𐑼𐑟 𐑓 𐑪𐑑𐑩𐑥𐑩𐑯𐑟; 𐑯 𐑑 𐑓𐑻𐑯𐑦𐑖 𐑩 𐑣𐑬𐑕 𐑒𐑳𐑥𐑓𐑼𐑑𐑩𐑚𐑤𐑦 𐑦𐑯 𐑞𐑨𐑑 𐑮𐑦𐑕𐑐𐑧𐑒𐑑, 𐑿 𐑣𐑨𐑛 𐑴𐑯𐑤𐑦 𐑑 𐑚𐑲 𐑳𐑐 𐑱𐑑 𐑹 𐑑𐑧𐑯 𐑤𐑱𐑟𐑦 𐑓𐑧𐑤𐑴𐑟, 𐑯 𐑤𐑱 𐑞𐑧𐑥 𐑮𐑬𐑯𐑛 𐑦𐑯 𐑞 𐑐𐑽𐑟 𐑯 𐑨𐑤𐑒𐑴𐑝𐑟. 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑦𐑑 𐑢𐑪𐑟 𐑝𐑧𐑮𐑦 𐑒𐑩𐑯𐑝𐑰𐑯𐑾𐑯𐑑 𐑪𐑯 𐑩𐑯 𐑦𐑒𐑕𐑒𐑻𐑖𐑩𐑯; 𐑥𐑳𐑗 𐑚𐑧𐑑𐑼 𐑞𐑨𐑯 𐑞𐑴𐑟 𐑜𐑸𐑛𐑩𐑯-𐑗𐑺𐑟 𐑢𐑦𐑗 𐑸 𐑒𐑩𐑯𐑝𐑻𐑑𐑩𐑚𐑩𐑤 𐑦𐑯𐑑𐑵 𐑢𐑷𐑒𐑦𐑙-𐑕𐑑𐑦𐑒𐑕; 𐑩𐑐𐑪𐑯 𐑩𐑒𐑱𐑠𐑩𐑯, 𐑩 𐑗𐑰𐑓 𐑒𐑷𐑤𐑦𐑙 𐑣𐑦𐑟 𐑩𐑑𐑧𐑯𐑛𐑩𐑯𐑑, 𐑯 𐑛𐑦𐑟𐑲𐑼𐑦𐑙 𐑣𐑦𐑥 𐑑 𐑥𐑱𐑒 𐑩 𐑕𐑧𐑑𐑰 𐑝 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑳𐑯𐑛𐑼 𐑩 𐑕𐑐𐑮𐑧𐑛𐑦𐑙 𐑑𐑮𐑰, 𐑐𐑼𐑣𐑨𐑐𐑕 𐑦𐑯 𐑕𐑳𐑥 𐑛𐑨𐑥𐑐 𐑥𐑸𐑖𐑦 𐑐𐑤𐑱𐑕.

𐑢𐑲𐑤 𐑯𐑼𐑱𐑑𐑦𐑙 𐑞𐑰𐑟 𐑔𐑦𐑙𐑟, 𐑧𐑝𐑮𐑦 𐑑𐑲𐑥 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑮𐑦𐑕𐑰𐑝𐑛 𐑞 𐑑𐑪𐑥𐑩𐑣𐑷𐑒 𐑓𐑮𐑪𐑥 𐑥𐑰, 𐑣𐑰 𐑓𐑤𐑳𐑮𐑦𐑖𐑑 𐑞 𐑣𐑨𐑗𐑩𐑑-𐑕𐑲𐑛 𐑝 𐑦𐑑 𐑴𐑝𐑼 𐑞 𐑕𐑤𐑰𐑐𐑼'𐑟 𐑣𐑧𐑛.

“𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜?”

“·𐑐𐑧𐑮𐑦 𐑰𐑟𐑦, 𐑒𐑦𐑤-𐑰; 𐑴! ·𐑐𐑧𐑮𐑦 𐑰𐑟𐑦!”

𐑣𐑰 𐑢𐑪𐑟 𐑜𐑴𐑦𐑙 𐑪𐑯 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑢𐑲𐑤𐑛 𐑮𐑧𐑥𐑦𐑯𐑦𐑕𐑩𐑯𐑕𐑩𐑟 𐑩𐑚𐑬𐑑 𐑣𐑦𐑟 𐑑𐑪𐑥𐑩𐑣𐑷𐑒-𐑐𐑲𐑐, 𐑢𐑦𐑗, 𐑦𐑑 𐑕𐑰𐑥𐑛, 𐑣𐑨𐑛 𐑦𐑯 𐑦𐑑𐑕 𐑑𐑵 𐑿𐑕𐑩𐑟 𐑚𐑴𐑔 𐑚𐑮𐑱𐑯𐑛 𐑣𐑦𐑟 𐑓𐑴𐑟 𐑯 𐑕𐑵𐑞𐑛 𐑣𐑦𐑟 𐑕𐑴𐑤, 𐑢𐑧𐑯 𐑢𐑰 𐑢𐑻 𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑩𐑑𐑮𐑨𐑒𐑑𐑩𐑛 𐑑 𐑞 𐑕𐑤𐑰𐑐𐑦𐑙 𐑮𐑦𐑜𐑼. 𐑞 𐑕𐑑𐑮𐑪𐑙 𐑝𐑱𐑐𐑼 𐑯𐑬 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑓𐑦𐑤𐑦𐑙 𐑞 𐑒𐑩𐑯𐑑𐑮𐑨𐑒𐑑𐑩𐑛 𐑣𐑴𐑤, 𐑦𐑑 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑑𐑧𐑤 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥. 𐑣𐑰 𐑚𐑮𐑰𐑞𐑛 𐑢𐑦𐑞 𐑩 𐑕𐑹𐑑 𐑝 𐑥𐑳𐑓𐑩𐑤𐑩𐑛𐑯𐑩𐑕; 𐑞𐑧𐑯 𐑕𐑰𐑥𐑛 𐑑𐑮𐑳𐑚𐑩𐑤𐑛 𐑦𐑯 𐑞 𐑯𐑴𐑟; 𐑞𐑧𐑯 𐑮𐑦𐑝𐑪𐑤𐑝𐑛 𐑴𐑝𐑼 𐑢𐑳𐑯𐑕 𐑹 𐑑𐑢𐑲𐑕; 𐑞𐑧𐑯 𐑕𐑨𐑑 𐑳𐑐 𐑯 𐑮𐑳𐑚𐑛 𐑣𐑦𐑟 𐑲𐑟.

“𐑣𐑪𐑤𐑴!” 𐑣𐑰 𐑚𐑮𐑰𐑞𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑, “𐑣𐑵 𐑚𐑰 𐑘𐑰 𐑕𐑥𐑴𐑒𐑼𐑟?”

“𐑖𐑦𐑐𐑑 𐑥𐑧𐑯,” 𐑭𐑯𐑕𐑼𐑛 𐑲, “𐑢𐑧𐑯 𐑛𐑳𐑟 𐑖𐑰 𐑕𐑱𐑤?”

“𐑲, 𐑲, 𐑘𐑰 𐑸 𐑜𐑴𐑦𐑙 𐑦𐑯 𐑣𐑻, 𐑚𐑰 𐑘𐑰? 𐑖𐑰 𐑕𐑱𐑤𐑟 𐑑-𐑛𐑱. 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑒𐑱𐑥 𐑩𐑚𐑹𐑛 𐑤𐑭𐑕𐑑 𐑯𐑲𐑑.”

“𐑢𐑪𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯?—𐑱𐑣𐑨𐑚?”

“𐑣𐑵 𐑚𐑳𐑑 𐑣𐑦𐑥 𐑦𐑯𐑛𐑰𐑛?”

𐑲 𐑢𐑪𐑟 𐑜𐑴𐑦𐑙 𐑑 𐑭𐑕𐑒 𐑣𐑦𐑥 𐑕𐑳𐑥 𐑓𐑻𐑞𐑼 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑟 𐑒𐑩𐑯𐑕𐑻𐑯𐑦𐑙 ·𐑱𐑣𐑨𐑚, 𐑢𐑧𐑯 𐑢𐑰 𐑣𐑻𐑛 𐑩 𐑯𐑶𐑟 𐑪𐑯 𐑛𐑧𐑒.

“𐑣𐑪𐑤𐑴! 𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑩𐑕𐑑𐑻,” 𐑕𐑧𐑛 𐑞 𐑮𐑦𐑜𐑼. “𐑣𐑰'𐑟 𐑩 𐑤𐑲𐑝𐑤𐑦 𐑗𐑰𐑓 𐑥𐑱𐑑, 𐑞𐑨𐑑; 𐑜𐑫𐑛 𐑥𐑨𐑯, 𐑯 𐑩 𐑐𐑲𐑩𐑕; 𐑚𐑳𐑑 𐑷𐑤 𐑩𐑤𐑲𐑝 𐑯𐑬, 𐑲 𐑥𐑳𐑕𐑑 𐑑𐑻𐑯 𐑑.” 𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙 𐑣𐑰 𐑢𐑧𐑯𐑑 𐑪𐑯 𐑛𐑧𐑒, 𐑯 𐑢𐑰 𐑓𐑪𐑤𐑴𐑛.

𐑦𐑑 𐑢𐑪𐑟 𐑯𐑬 𐑒𐑤𐑽 𐑕𐑳𐑯𐑮𐑲𐑟. 𐑕𐑵𐑯 𐑞 𐑒𐑮𐑵 𐑒𐑱𐑥 𐑪𐑯 𐑚𐑹𐑛 𐑦𐑯 𐑑𐑵𐑟 𐑯 𐑔𐑮𐑰𐑟; 𐑞 𐑮𐑦𐑜𐑼𐑟 𐑚𐑦𐑕𐑑𐑻𐑛 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟; 𐑞 𐑥𐑱𐑑𐑕 𐑢𐑻 𐑨𐑒𐑑𐑦𐑝𐑤𐑦 𐑦𐑯𐑜𐑱𐑡𐑛; 𐑯 𐑕𐑧𐑝𐑼𐑩𐑤 𐑝 𐑞 𐑖𐑹 𐑐𐑰𐑐𐑩𐑤 𐑢𐑻 𐑚𐑦𐑟𐑦 𐑦𐑯 𐑚𐑮𐑦𐑙𐑦𐑙 𐑝𐑺𐑾𐑕 𐑤𐑭𐑕𐑑 𐑔𐑦𐑙𐑟 𐑪𐑯 𐑚𐑹𐑛. 𐑥𐑰𐑯𐑢𐑲𐑤 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑮𐑦𐑥𐑱𐑯𐑛 𐑦𐑯𐑝𐑦𐑟𐑩𐑚𐑤𐑦 𐑦𐑯𐑖𐑮𐑲𐑯𐑛 𐑢𐑦𐑞𐑦𐑯 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯.
