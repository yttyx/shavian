𐑯𐑬 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑧𐑒𐑩𐑯𐑛 𐑤𐑴𐑼𐑦𐑙, 𐑞 𐑚𐑴𐑑 𐑐𐑨𐑛𐑩𐑤𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑢𐑱𐑤; 𐑯 𐑨𐑟 𐑞 𐑓𐑦𐑖 𐑮𐑦𐑕𐑰𐑝𐑛 𐑞 𐑛𐑸𐑑𐑩𐑛 𐑲𐑼𐑯, 𐑦𐑑 𐑜𐑱𐑝 𐑦𐑑𐑕 𐑒𐑳𐑕𐑑𐑩𐑥𐑼𐑦 𐑮𐑨𐑐, 𐑢𐑦𐑗 𐑣𐑨𐑐𐑩𐑯𐑛, 𐑦𐑯 𐑞𐑦𐑕 𐑦𐑯𐑕𐑑𐑩𐑯𐑕, 𐑑 𐑚𐑰 𐑮𐑲𐑑 𐑳𐑯𐑛𐑼 𐑐𐑫𐑼 ·𐑐𐑦𐑐'𐑕 𐑕𐑰𐑑. 𐑞 𐑦𐑯𐑝𐑪𐑤𐑩𐑯𐑑𐑼𐑦 𐑒𐑪𐑯𐑕𐑑𐑼𐑯𐑱𐑖𐑩𐑯 𐑝 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑 𐑒𐑷𐑟𐑛 𐑣𐑦𐑥 𐑑 𐑤𐑰𐑐, 𐑐𐑨𐑛𐑩𐑤 𐑦𐑯 𐑣𐑨𐑯𐑛, 𐑬𐑑 𐑝 𐑞 𐑚𐑴𐑑; 𐑯 𐑦𐑯 𐑕𐑳𐑗 𐑩 𐑢𐑱, 𐑞𐑨𐑑 𐑐𐑸𐑑 𐑝 𐑞 𐑕𐑤𐑨𐑒 𐑢𐑱𐑤 𐑤𐑲𐑯 𐑒𐑳𐑥𐑦𐑙 𐑩𐑜𐑱𐑯𐑕𐑑 𐑣𐑦𐑟 𐑗𐑧𐑕𐑑, 𐑣𐑰 𐑚𐑮𐑧𐑕𐑑𐑩𐑛 𐑦𐑑 𐑴𐑝𐑼𐑚𐑹𐑛 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑕𐑴 𐑨𐑟 𐑑 𐑚𐑦𐑒𐑳𐑥 𐑦𐑯𐑑𐑨𐑙𐑜𐑩𐑤𐑛 𐑦𐑯 𐑦𐑑, 𐑢𐑧𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑐𐑤𐑳𐑥𐑐𐑦𐑙 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑷𐑑𐑼. 𐑞𐑨𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑞 𐑕𐑑𐑮𐑦𐑒𐑩𐑯 𐑢𐑱𐑤 𐑕𐑑𐑸𐑑𐑩𐑛 𐑪𐑯 𐑩 𐑓𐑽𐑕 𐑮𐑳𐑯, 𐑞 𐑤𐑲𐑯 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑕𐑑𐑮𐑱𐑑𐑩𐑯𐑛; 𐑯 𐑐𐑮𐑧𐑕𐑑𐑴! 𐑐𐑫𐑼 ·𐑐𐑦𐑐 𐑒𐑱𐑥 𐑷𐑤 𐑓𐑴𐑥𐑦𐑙 𐑳𐑐 𐑑 𐑞 𐑗𐑪𐑒𐑕 𐑝 𐑞 𐑚𐑴𐑑, 𐑮𐑦𐑥𐑹𐑕𐑤𐑩𐑕𐑤𐑦 𐑛𐑮𐑨𐑜𐑛 𐑞𐑺 𐑚𐑲 𐑞 𐑤𐑲𐑯, 𐑢𐑦𐑗 𐑣𐑨𐑛 𐑑𐑱𐑒𐑩𐑯 𐑕𐑧𐑝𐑼𐑩𐑤 𐑑𐑻𐑯𐑟 𐑼𐑬𐑯𐑛 𐑣𐑦𐑟 𐑗𐑧𐑕𐑑 𐑯 𐑯𐑧𐑒.

·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑕𐑑𐑫𐑛 𐑦𐑯 𐑞 𐑚𐑬𐑟. 𐑣𐑰 𐑢𐑪𐑟 𐑓𐑫𐑤 𐑝 𐑞 𐑓𐑲𐑼 𐑝 𐑞 𐑣𐑳𐑯𐑑. 𐑣𐑰 𐑣𐑱𐑑𐑩𐑛 ·𐑐𐑦𐑐 𐑓 𐑩 𐑐𐑪𐑤𐑑𐑮𐑵𐑯. 𐑕𐑯𐑨𐑗𐑦𐑙 𐑞 𐑚𐑴𐑑-𐑯𐑲𐑓 𐑓𐑮𐑪𐑥 𐑦𐑑𐑕 𐑖𐑰𐑔, 𐑣𐑰 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑦𐑑𐑕 𐑖𐑸𐑐 𐑧𐑡 𐑴𐑝𐑼 𐑞 𐑤𐑲𐑯, 𐑯 𐑑𐑻𐑯𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 ·𐑕𐑑𐑳𐑚, 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛 𐑦𐑯𐑑𐑼𐑪𐑜𐑩𐑑𐑦𐑝𐑤𐑦, “𐑒𐑳𐑑?” 𐑥𐑰𐑯𐑑𐑲𐑥 ·𐑐𐑦𐑐'𐑕 𐑚𐑤𐑵, 𐑗𐑴𐑒𐑑 𐑓𐑱𐑕 𐑐𐑤𐑱𐑯𐑤𐑦 𐑤𐑫𐑒𐑑, 𐑛𐑵, 𐑓 ·𐑜𐑪𐑛'𐑟 𐑕𐑱𐑒! 𐑷𐑤 𐑐𐑭𐑕𐑑 𐑦𐑯 𐑩 𐑓𐑤𐑨𐑖. 𐑦𐑯 𐑤𐑧𐑕 𐑞𐑨𐑯 𐑣𐑭𐑓 𐑩 𐑥𐑦𐑯𐑦𐑑, 𐑞𐑦𐑕 𐑦𐑯𐑑𐑲𐑼 𐑔𐑦𐑙 𐑣𐑨𐑐𐑩𐑯𐑛.

“𐑛𐑨𐑥 𐑣𐑦𐑥, 𐑒𐑳𐑑!” 𐑮𐑹𐑛 ·𐑕𐑑𐑳𐑚; 𐑯 𐑕𐑴 𐑞 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑤𐑪𐑕𐑑 𐑯 ·𐑐𐑦𐑐 𐑢𐑪𐑟 𐑕𐑱𐑝𐑛.

𐑕𐑴 𐑕𐑵𐑯 𐑨𐑟 𐑣𐑰 𐑮𐑦𐑒𐑳𐑝𐑼𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑞 𐑐𐑫𐑼 𐑤𐑦𐑑𐑩𐑤 𐑯𐑰𐑜𐑮𐑴 𐑢𐑪𐑟 𐑩𐑕𐑱𐑤𐑛 𐑚𐑲 𐑘𐑧𐑤𐑟 𐑯 𐑧𐑒𐑕𐑦𐑒𐑮𐑱𐑖𐑩𐑯𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑮𐑵. 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤𐑦 𐑐𐑼𐑥𐑦𐑑𐑦𐑙 𐑞𐑰𐑟 𐑦𐑮𐑧𐑜𐑿𐑤𐑼 𐑒𐑻𐑕𐑦𐑙𐑟 𐑑 𐑦𐑝𐑨𐑐𐑼𐑱𐑑, ·𐑕𐑑𐑳𐑚 𐑞𐑧𐑯 𐑦𐑯 𐑩 𐑐𐑤𐑱𐑯, 𐑚𐑦𐑟𐑯𐑩𐑕-𐑤𐑲𐑒, 𐑚𐑳𐑑 𐑕𐑑𐑦𐑤 𐑣𐑭𐑓 𐑣𐑿𐑥𐑼𐑩𐑕 𐑥𐑨𐑯𐑼, 𐑒𐑻𐑕𐑑 ·𐑐𐑦𐑐 𐑩𐑓𐑦𐑖𐑩𐑤𐑦; 𐑯 𐑞𐑨𐑑 𐑛𐑳𐑯, 𐑳𐑯𐑩𐑓𐑦𐑖𐑩𐑤𐑦 𐑜𐑱𐑝 𐑣𐑦𐑥 𐑥𐑳𐑗 𐑣𐑴𐑤𐑕𐑩𐑥 𐑩𐑛𐑝𐑲𐑕. 𐑞 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕 𐑢𐑪𐑟, ·𐑯𐑧𐑝𐑼 𐑡𐑳𐑥𐑐 𐑓𐑮𐑪𐑥 𐑩 𐑚𐑴𐑑, ·𐑐𐑦𐑐, 𐑦𐑒𐑕𐑧𐑐𐑑—𐑚𐑳𐑑 𐑷𐑤 𐑞 𐑮𐑧𐑕𐑑 𐑢𐑪𐑟 𐑦𐑯𐑛𐑧𐑓𐑦𐑯𐑦𐑑, 𐑨𐑟 𐑞 𐑕𐑬𐑯𐑛𐑩𐑕𐑑 𐑩𐑛𐑝𐑲𐑕 𐑧𐑝𐑼 𐑦𐑟. 𐑯𐑬, 𐑦𐑯 𐑡𐑧𐑯𐑼𐑩𐑤, ·𐑕𐑑𐑦𐑒 𐑑 𐑞 𐑚𐑴𐑑, 𐑦𐑟 𐑘𐑹 𐑑𐑮𐑵 𐑥𐑪𐑑𐑴 𐑦𐑯 𐑢𐑱𐑤𐑦𐑙; 𐑚𐑳𐑑 𐑒𐑱𐑕𐑩𐑟 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑣𐑨𐑐𐑩𐑯 𐑢𐑧𐑯 ·𐑤𐑰𐑐 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑑, 𐑦𐑟 𐑕𐑑𐑦𐑤 𐑚𐑧𐑑𐑼. 𐑥𐑹𐑴𐑝𐑼, 𐑨𐑟 𐑦𐑓 𐑐𐑼𐑕𐑰𐑝𐑦𐑙 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑞𐑨𐑑 𐑦𐑓 𐑣𐑰 𐑖𐑵𐑛 𐑜𐑦𐑝 𐑳𐑯𐑛𐑲𐑤𐑵𐑑𐑩𐑛 𐑒𐑪𐑯𐑖𐑦𐑧𐑯𐑖𐑩𐑕 𐑩𐑛𐑝𐑲𐑕 𐑑 ·𐑐𐑦𐑐, 𐑣𐑰 𐑢𐑫𐑛 𐑚𐑰 𐑤𐑰𐑝𐑦𐑙 𐑣𐑦𐑥 𐑑𐑵 𐑢𐑲𐑛 𐑩 𐑥𐑸𐑡𐑦𐑯 𐑑 𐑡𐑳𐑥𐑐 𐑦𐑯 𐑓 𐑞 𐑓𐑿𐑗𐑼; ·𐑕𐑑𐑳𐑚 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑛𐑮𐑪𐑐𐑑 𐑷𐑤 𐑩𐑛𐑝𐑲𐑕, 𐑯 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑢𐑦𐑞 𐑩 𐑐𐑼𐑧𐑥𐑐𐑑𐑼𐑦 𐑒𐑩𐑥𐑭𐑯𐑛, “𐑕𐑑𐑦𐑒 𐑑 𐑞 𐑚𐑴𐑑, ·𐑐𐑦𐑐, 𐑹 𐑚𐑲 𐑞 ·𐑤𐑹𐑛, 𐑲 𐑢𐑴𐑯𐑑 𐑐𐑦𐑒 𐑿 𐑳𐑐 𐑦𐑓 𐑿 𐑡𐑳𐑥𐑐; 𐑥𐑲𐑯𐑛 𐑞𐑨𐑑. 𐑢𐑰 𐑒𐑭𐑯𐑑 𐑩𐑓𐑹𐑛 𐑑 𐑤𐑵𐑟 𐑢𐑱𐑤𐑟 𐑚𐑲 𐑞 𐑤𐑲𐑒𐑕 𐑝 𐑿; 𐑩 𐑢𐑱𐑤 𐑢𐑫𐑛 𐑕𐑧𐑤 𐑓 𐑔𐑻𐑑𐑦 𐑑𐑲𐑥𐑟 𐑢𐑪𐑑 𐑿 𐑢𐑫𐑛, ·𐑐𐑦𐑐, 𐑦𐑯 ·𐑨𐑤𐑩𐑚𐑨𐑥𐑩. 𐑚𐑺 𐑞𐑨𐑑 𐑦𐑯 𐑥𐑲𐑯𐑛, 𐑯 𐑛𐑴𐑯𐑑 𐑡𐑳𐑥𐑐 𐑧𐑯𐑦 𐑥𐑹.” 𐑣𐑽𐑚𐑲 𐑐𐑼𐑣𐑨𐑐𐑕 ·𐑕𐑑𐑳𐑚 𐑦𐑯𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑣𐑦𐑯𐑑𐑩𐑛, 𐑞𐑨𐑑 𐑞𐑴 𐑥𐑨𐑯 𐑤𐑳𐑝𐑛 𐑣𐑦𐑟 𐑓𐑧𐑤𐑴, 𐑘𐑧𐑑 𐑥𐑨𐑯 𐑦𐑟 𐑩 𐑥𐑳𐑯𐑦-𐑥𐑱𐑒𐑦𐑙 𐑨𐑯𐑦𐑥𐑩𐑤, 𐑢𐑦𐑗 𐑐𐑮𐑩𐑐𐑧𐑯𐑕𐑦𐑑𐑦 𐑑𐑵 𐑪𐑓𐑩𐑯 𐑦𐑯𐑑𐑼𐑓𐑽𐑟 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑚𐑦𐑯𐑧𐑝𐑩𐑤𐑩𐑯𐑕.
