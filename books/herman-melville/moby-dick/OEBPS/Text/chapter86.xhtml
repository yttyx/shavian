·𐑗𐑨𐑐𐑑𐑼 86. 𐑞 ·𐑑𐑱𐑤.

𐑳𐑞𐑼 𐑐𐑴𐑩𐑑𐑕 𐑣𐑨𐑝 𐑢𐑹𐑚𐑩𐑤𐑛 𐑞 𐑐𐑮𐑱𐑟𐑩𐑟 𐑝 𐑞 𐑕𐑪𐑓𐑑 𐑲 𐑝 𐑞 𐑨𐑯𐑑𐑦𐑤𐑴𐑐, 𐑯 𐑞 𐑤𐑳𐑝𐑤𐑦 𐑐𐑤𐑵𐑥𐑦𐑡 𐑝 𐑞 𐑚𐑻𐑛 𐑞𐑨𐑑 𐑯𐑧𐑝𐑼 𐑩𐑤𐑲𐑑𐑕; 𐑤𐑧𐑕 𐑕𐑦𐑤𐑧𐑕𐑑𐑾𐑤, 𐑲 𐑕𐑧𐑤𐑩𐑚𐑮𐑱𐑑 𐑩 𐑑𐑱𐑤.

𐑮𐑧𐑒𐑩𐑯𐑦𐑙 𐑞 𐑤𐑸𐑡𐑩𐑕𐑑 𐑕𐑲𐑟𐑛 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟 𐑑𐑱𐑤 𐑑 𐑚𐑦𐑜𐑦𐑯 𐑨𐑑 𐑞𐑨𐑑 𐑐𐑶𐑯𐑑 𐑝 𐑞 𐑑𐑮𐑳𐑙𐑒 𐑢𐑺 𐑦𐑑 𐑑𐑱𐑐𐑼𐑟 𐑑 𐑩𐑚𐑬𐑑 𐑞 𐑜𐑻𐑔 𐑝 𐑩 𐑥𐑨𐑯, 𐑦𐑑 𐑒𐑩𐑥𐑐𐑮𐑲𐑟𐑩𐑟 𐑩𐑐𐑪𐑯 𐑦𐑑𐑕 𐑳𐑐𐑼 𐑕𐑻𐑓𐑦𐑕 𐑩𐑤𐑴𐑯, 𐑩𐑯 𐑺𐑾 𐑝 𐑨𐑑 𐑤𐑰𐑕𐑑 𐑓𐑦𐑓𐑑𐑦 𐑕𐑒𐑢𐑺 𐑓𐑰𐑑. 𐑞 𐑒𐑩𐑥𐑐𐑨𐑒𐑑 𐑮𐑬𐑯𐑛 𐑚𐑪𐑛𐑦 𐑝 𐑦𐑑𐑕 𐑮𐑵𐑑 𐑦𐑒𐑕𐑐𐑨𐑯𐑛𐑟 𐑦𐑯𐑑𐑵 𐑑𐑵 𐑚𐑮𐑷𐑛, 𐑓𐑻𐑥, 𐑓𐑤𐑨𐑑 𐑐𐑭𐑥𐑟 𐑹 𐑓𐑤𐑵𐑒𐑕, 𐑜𐑮𐑨𐑡𐑵𐑩𐑤𐑦 𐑖𐑴𐑤𐑦𐑙 𐑩𐑢𐑱 𐑑 𐑤𐑧𐑕 𐑞𐑨𐑯 𐑩𐑯 𐑦𐑯𐑗 𐑦𐑯 𐑔𐑦𐑒𐑯𐑩𐑕. 𐑨𐑑 𐑞 𐑒𐑮𐑪𐑗 𐑹 𐑡𐑳𐑙𐑒𐑖𐑩𐑯, 𐑞𐑰𐑟 𐑓𐑤𐑵𐑒𐑕 𐑕𐑤𐑲𐑑𐑤𐑦 𐑴𐑝𐑼𐑤𐑨𐑐, 𐑞𐑧𐑯 𐑕𐑲𐑛𐑢𐑱𐑟 𐑮𐑦𐑕𐑰𐑛 𐑓𐑮𐑪𐑥 𐑰𐑗 𐑳𐑞𐑼 𐑤𐑲𐑒 𐑢𐑦𐑙𐑟, 𐑤𐑰𐑝𐑦𐑙 𐑩 𐑢𐑲𐑛 𐑝𐑱𐑒𐑩𐑯𐑕𐑦 𐑚𐑦𐑑𐑢𐑰𐑯. 𐑦𐑯 𐑯𐑴 𐑤𐑦𐑝𐑦𐑙 𐑔𐑦𐑙 𐑸 𐑞 𐑤𐑲𐑯𐑟 𐑝 𐑚𐑿𐑑𐑦 𐑥𐑹 𐑦𐑒𐑕𐑒𐑢𐑦𐑟𐑦𐑑𐑤𐑦 𐑛𐑦𐑓𐑲𐑯𐑛 𐑞𐑨𐑯 𐑦𐑯 𐑞 𐑒𐑮𐑧𐑟𐑩𐑯𐑑𐑦𐑒 𐑚𐑹𐑛𐑼𐑟 𐑝 𐑞𐑰𐑟 𐑓𐑤𐑵𐑒𐑕. 𐑨𐑑 𐑦𐑑𐑕 𐑳𐑑𐑥𐑴𐑕𐑑 𐑦𐑒𐑕𐑐𐑨𐑯𐑖𐑩𐑯 𐑦𐑯 𐑞 𐑓𐑫𐑤 𐑜𐑮𐑴𐑯 𐑢𐑱𐑤, 𐑞 𐑑𐑱𐑤 𐑢𐑦𐑤 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑤𐑦 𐑦𐑒𐑕𐑰𐑛 𐑑𐑢𐑧𐑯𐑑𐑦 𐑓𐑰𐑑 𐑩𐑒𐑮𐑪𐑕.

𐑞 𐑦𐑯𐑑𐑲𐑼 𐑥𐑧𐑥𐑚𐑼 𐑕𐑰𐑥𐑟 𐑩 𐑛𐑧𐑯𐑕 𐑢𐑧𐑚𐑛 𐑚𐑧𐑛 𐑝 𐑢𐑧𐑤𐑛𐑩𐑛 𐑕𐑦𐑯𐑿𐑟; 𐑚𐑳𐑑 𐑒𐑳𐑑 𐑦𐑯𐑑𐑵 𐑦𐑑, 𐑯 𐑿 𐑓𐑲𐑯𐑛 𐑞𐑨𐑑 𐑔𐑮𐑰 𐑛𐑦𐑕𐑑𐑦𐑙𐑒𐑑 𐑕𐑑𐑮𐑭𐑑𐑩 𐑒𐑩𐑥𐑐𐑴𐑟 𐑦𐑑:—𐑳𐑐𐑼, 𐑥𐑦𐑛𐑩𐑤, 𐑯 𐑤𐑴𐑼. 𐑞 𐑓𐑲𐑚𐑼𐑟 𐑦𐑯 𐑞 𐑳𐑐𐑼 𐑯 𐑤𐑴𐑼 𐑤𐑱𐑼𐑟, 𐑸 𐑤𐑪𐑙 𐑯 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤; 𐑞𐑴𐑟 𐑝 𐑞 𐑥𐑦𐑛𐑩𐑤 𐑢𐑳𐑯, 𐑝𐑧𐑮𐑦 𐑖𐑹𐑑, 𐑯 𐑮𐑳𐑯𐑦𐑙 𐑒𐑮𐑪𐑕𐑢𐑲𐑟 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑬𐑑𐑕𐑲𐑛 𐑤𐑱𐑼𐑟. 𐑞𐑦𐑕 𐑑𐑮𐑲𐑿𐑯 𐑕𐑑𐑮𐑳𐑒𐑗𐑼, 𐑨𐑟 𐑥𐑳𐑗 𐑨𐑟 𐑧𐑯𐑦𐑔𐑦𐑙 𐑧𐑤𐑕, 𐑦𐑥𐑐𐑸𐑑𐑕 𐑐𐑬𐑼 𐑑 𐑞 𐑑𐑱𐑤. 𐑑 𐑞 𐑕𐑑𐑿𐑛𐑩𐑯𐑑 𐑝 𐑴𐑤𐑛 ·𐑮𐑴𐑥𐑩𐑯 𐑢𐑷𐑤𐑟, 𐑞 𐑥𐑦𐑛𐑩𐑤 𐑤𐑱𐑼 𐑢𐑦𐑤 𐑓𐑻𐑯𐑦𐑖 𐑩 𐑒𐑘𐑫𐑼𐑾𐑕 𐑐𐑨𐑮𐑩𐑤𐑧𐑤 𐑑 𐑞 𐑔𐑦𐑯 𐑒𐑹𐑕 𐑝 𐑑𐑲𐑤𐑟 𐑷𐑤𐑢𐑱𐑟 𐑷𐑤𐑑𐑼𐑯𐑱𐑑𐑦𐑙 𐑢𐑦𐑞 𐑞 𐑕𐑑𐑴𐑯 𐑦𐑯 𐑞𐑴𐑟 𐑢𐑳𐑯𐑛𐑼𐑓𐑩𐑤 𐑮𐑧𐑤𐑦𐑒𐑕 𐑝 𐑞 𐑨𐑯𐑑𐑰𐑒, 𐑯 𐑢𐑦𐑗 𐑳𐑯𐑛𐑬𐑑𐑩𐑛𐑤𐑦 𐑒𐑩𐑯𐑑𐑮𐑦𐑚𐑿𐑑 𐑕𐑴 𐑥𐑳𐑗 𐑑 𐑞 𐑜𐑮𐑱𐑑 𐑕𐑑𐑮𐑧𐑙𐑔 𐑝 𐑞 𐑥𐑱𐑕𐑩𐑯𐑼𐑦.

𐑚𐑳𐑑 𐑨𐑟 𐑦𐑓 𐑞𐑦𐑕 𐑝𐑭𐑕𐑑 𐑤𐑴𐑒𐑩𐑤 𐑐𐑬𐑼 𐑦𐑯 𐑞 𐑑𐑧𐑯𐑛𐑦𐑯𐑩𐑕 𐑑𐑱𐑤 𐑢𐑻 𐑯𐑪𐑑 𐑦𐑯𐑳𐑓, 𐑞 𐑣𐑴𐑤 𐑚𐑳𐑤𐑒 𐑝 𐑞 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑦𐑟 𐑯𐑦𐑑 𐑴𐑝𐑼 𐑢𐑦𐑞 𐑩 𐑢𐑹𐑐 𐑯 𐑢𐑵𐑓 𐑝 𐑥𐑳𐑕𐑒𐑿𐑤𐑼 𐑓𐑲𐑚𐑼𐑟 𐑯 𐑓𐑦𐑤𐑩𐑥𐑩𐑯𐑑𐑕, 𐑢𐑦𐑗 𐑐𐑭𐑕𐑦𐑙 𐑪𐑯 𐑲𐑞𐑼 𐑕𐑲𐑛 𐑞 𐑤𐑶𐑯𐑟 𐑯 𐑮𐑳𐑯𐑦𐑙 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑓𐑤𐑵𐑒𐑕, 𐑦𐑯𐑕𐑧𐑯𐑕𐑩𐑚𐑤𐑦 𐑚𐑤𐑧𐑯𐑛 𐑢𐑦𐑞 𐑞𐑧𐑥, 𐑯 𐑤𐑸𐑡𐑤𐑦 𐑒𐑩𐑯𐑑𐑮𐑦𐑚𐑿𐑑 𐑑 𐑞𐑺 𐑥𐑲𐑑; 𐑕𐑴 𐑞𐑨𐑑 𐑦𐑯 𐑞 𐑑𐑱𐑤 𐑞 𐑒𐑪𐑯𐑓𐑤𐑵𐑩𐑯𐑑 𐑥𐑧𐑠𐑼𐑤𐑩𐑕 𐑓𐑹𐑕 𐑝 𐑞 𐑣𐑴𐑤 𐑢𐑱𐑤 𐑕𐑰𐑥𐑟 𐑒𐑪𐑯𐑕𐑩𐑯𐑑𐑮𐑱𐑑𐑩𐑛 𐑑 𐑩 𐑐𐑶𐑯𐑑. 𐑒𐑫𐑛 𐑩𐑯𐑲𐑩𐑤𐑱𐑖𐑩𐑯 𐑩𐑒𐑻 𐑑 𐑥𐑨𐑑𐑼, 𐑞𐑦𐑕 𐑢𐑻 𐑞 𐑔𐑦𐑙 𐑑 𐑛𐑵 𐑦𐑑.

𐑯𐑹 𐑛𐑳𐑟 𐑞𐑦𐑕—𐑦𐑑𐑕 𐑩𐑥𐑱𐑟𐑦𐑙 𐑕𐑑𐑮𐑧𐑙𐑔, 𐑨𐑑 𐑷𐑤 𐑑𐑧𐑯𐑛 𐑑 𐑒𐑮𐑦𐑐𐑩𐑤 𐑞 𐑜𐑮𐑱𐑕𐑓𐑩𐑤 𐑓𐑤𐑧𐑒𐑖𐑩𐑯 𐑝 𐑦𐑑𐑕 𐑥𐑴𐑖𐑩𐑯𐑟; 𐑢𐑺 𐑦𐑯𐑓𐑨𐑯𐑑𐑦𐑤𐑩𐑯𐑧𐑕 𐑝 𐑰𐑟 𐑳𐑯𐑡𐑩𐑤𐑱𐑑𐑕 𐑔𐑮𐑵 𐑩 ·𐑑𐑲𐑑𐑱𐑯𐑦𐑟𐑩𐑥 𐑝 𐑐𐑬𐑼. 𐑪𐑯 𐑞 𐑒𐑪𐑯𐑑𐑮𐑺𐑦, 𐑞𐑴𐑟 𐑥𐑴𐑖𐑩𐑯𐑟 𐑛𐑦𐑮𐑲𐑝 𐑞𐑺 𐑥𐑴𐑕𐑑 𐑩𐑐𐑷𐑤𐑦𐑙 𐑚𐑿𐑑𐑦 𐑓𐑮𐑪𐑥 𐑦𐑑. 𐑮𐑾𐑤 𐑕𐑑𐑮𐑧𐑙𐑔 𐑯𐑧𐑝𐑼 𐑦𐑥𐑐𐑺𐑟 𐑚𐑿𐑑𐑦 𐑹 𐑣𐑸𐑥𐑩𐑯𐑦, 𐑚𐑳𐑑 𐑦𐑑 𐑪𐑓𐑩𐑯 𐑚𐑦𐑕𐑑𐑴𐑟 𐑦𐑑; 𐑯 𐑦𐑯 𐑧𐑝𐑮𐑦𐑔𐑦𐑙 𐑦𐑥𐑐𐑴𐑟𐑦𐑙𐑤𐑦 𐑚𐑿𐑑𐑦𐑓𐑩𐑤, 𐑕𐑑𐑮𐑧𐑙𐑔 𐑣𐑨𐑟 𐑥𐑳𐑗 𐑑 𐑛𐑵 𐑢𐑦𐑞 𐑞 𐑥𐑨𐑡𐑦𐑒. 𐑑𐑱𐑒 𐑩𐑢𐑱 𐑞 𐑑𐑲𐑛 𐑑𐑧𐑯𐑛𐑩𐑯𐑟 𐑞𐑨𐑑 𐑷𐑤 𐑴𐑝𐑼 𐑕𐑰𐑥 𐑚𐑻𐑕𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑸𐑚𐑩𐑤 𐑦𐑯 𐑞 𐑒𐑸𐑝𐑛 ·𐑣𐑻𐑒𐑿𐑤𐑰𐑟, 𐑯 𐑦𐑑𐑕 𐑗𐑸𐑥 𐑢𐑫𐑛 𐑚𐑰 𐑜𐑪𐑯. 𐑨𐑟 𐑛𐑦𐑝𐑬𐑑 ·𐑧𐑒𐑼𐑥𐑩𐑯 𐑤𐑦𐑓𐑑𐑩𐑛 𐑞 𐑤𐑦𐑯𐑦𐑯 𐑖𐑰𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑯𐑱𐑒𐑩𐑛 𐑒𐑹𐑐𐑕 𐑝 ·𐑜𐑻𐑑𐑩, 𐑣𐑰 𐑢𐑪𐑟 𐑴𐑝𐑼𐑢𐑧𐑤𐑥𐑛 𐑢𐑦𐑞 𐑞 𐑥𐑨𐑕𐑦𐑝 𐑗𐑧𐑕𐑑 𐑝 𐑞 𐑥𐑨𐑯, 𐑞𐑨𐑑 𐑕𐑰𐑥𐑛 𐑨𐑟 𐑩 ·𐑮𐑴𐑥𐑩𐑯 𐑑𐑮𐑲𐑳𐑥𐑓𐑩𐑤 𐑸𐑗. 𐑢𐑧𐑯 ·𐑨𐑯𐑡𐑩𐑤𐑴 𐑐𐑱𐑯𐑑𐑕 𐑰𐑝𐑩𐑯 ·𐑜𐑪𐑛 𐑞 ·𐑓𐑭𐑞𐑼 𐑦𐑯 𐑣𐑿𐑥𐑩𐑯 𐑓𐑹𐑥, 𐑥𐑸𐑒 𐑢𐑪𐑑 𐑮𐑴𐑚𐑳𐑕𐑑𐑯𐑩𐑕 𐑦𐑟 𐑞𐑺. 𐑯 𐑢𐑪𐑑𐑧𐑝𐑼 𐑞𐑱 𐑥𐑱 𐑮𐑦𐑝𐑰𐑤 𐑝 𐑞 𐑛𐑦𐑝𐑲𐑯 𐑤𐑳𐑝 𐑦𐑯 𐑞 ·𐑕𐑳𐑯, 𐑞 𐑕𐑪𐑓𐑑, 𐑒𐑻𐑤𐑛, 𐑣𐑻𐑥𐑨𐑓𐑮𐑩𐑛𐑲𐑑𐑦𐑒𐑩𐑤 ·𐑦𐑑𐑨𐑤𐑾𐑯 𐑐𐑦𐑒𐑗𐑼𐑟, 𐑦𐑯 𐑢𐑦𐑗 𐑣𐑦𐑟 𐑲𐑛𐑾 𐑣𐑨𐑟 𐑚𐑰𐑯 𐑥𐑴𐑕𐑑 𐑕𐑩𐑒𐑕𐑧𐑕𐑓𐑩𐑤𐑦 𐑦𐑥𐑚𐑪𐑛𐑦𐑛; 𐑞𐑰𐑟 𐑐𐑦𐑒𐑗𐑼𐑟, 𐑕𐑴 𐑛𐑧𐑕𐑑𐑦𐑑𐑿𐑑 𐑨𐑟 𐑞𐑱 𐑸 𐑝 𐑷𐑤 𐑚𐑮𐑷𐑯𐑦𐑯𐑩𐑕, 𐑣𐑦𐑯𐑑 𐑯𐑳𐑔𐑦𐑙 𐑝 𐑧𐑯𐑦 𐑐𐑬𐑼, 𐑚𐑳𐑑 𐑞 𐑥𐑽 𐑯𐑧𐑜𐑩𐑑𐑦𐑝, 𐑓𐑧𐑥𐑦𐑯𐑦𐑯 𐑢𐑳𐑯 𐑝 𐑕𐑩𐑚𐑥𐑦𐑖𐑩𐑯 𐑯 𐑦𐑯𐑛𐑻𐑩𐑯𐑕, 𐑢𐑦𐑗 𐑪𐑯 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟 𐑦𐑑 𐑦𐑟 𐑒𐑩𐑯𐑕𐑰𐑛𐑩𐑛, 𐑓𐑹𐑥 𐑞 𐑐𐑦𐑒𐑿𐑤𐑽 𐑐𐑮𐑨𐑒𐑑𐑦𐑒𐑩𐑤 𐑝𐑻𐑗𐑵𐑟 𐑝 𐑣𐑦𐑟 𐑑𐑰𐑗𐑦𐑙𐑟.
