·𐑗𐑨𐑐𐑑𐑼 130. 𐑞 ·𐑣𐑨𐑑.

𐑯 𐑯𐑬 𐑞𐑨𐑑 𐑨𐑑 𐑞 𐑐𐑮𐑪𐑐𐑼 𐑑𐑲𐑥 𐑯 𐑐𐑤𐑱𐑕, 𐑭𐑓𐑑𐑼 𐑕𐑴 𐑤𐑪𐑙 𐑯 𐑢𐑲𐑛 𐑩 𐑐𐑮𐑦𐑤𐑦𐑥𐑦𐑯𐑼𐑦 𐑒𐑮𐑵𐑟, ·𐑱𐑣𐑨𐑚,—𐑷𐑤 𐑳𐑞𐑼 𐑢𐑱𐑤𐑦𐑙 𐑢𐑷𐑑𐑼𐑟 𐑕𐑢𐑧𐑐𐑑—𐑕𐑰𐑥𐑛 𐑑 𐑣𐑨𐑝 𐑗𐑱𐑕𐑑 𐑣𐑦𐑟 𐑓𐑴 𐑦𐑯𐑑𐑵 𐑩𐑯 𐑴𐑖𐑩𐑯-𐑓𐑴𐑤𐑛, 𐑑 𐑕𐑤𐑱 𐑣𐑦𐑥 𐑞 𐑥𐑹 𐑕𐑦𐑒𐑘𐑫𐑼𐑤𐑦 𐑞𐑺; 𐑯𐑬, 𐑞𐑨𐑑 𐑣𐑰 𐑓𐑬𐑯𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑣𐑸𐑛 𐑚𐑲 𐑞 𐑝𐑧𐑮𐑦 𐑤𐑨𐑑𐑦𐑑𐑿𐑛 𐑯 𐑤𐑪𐑯𐑜𐑦𐑑𐑿𐑛 𐑢𐑺 𐑣𐑦𐑟 𐑑𐑹𐑥𐑧𐑯𐑑𐑦𐑙 𐑢𐑵𐑯𐑛 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑦𐑯𐑓𐑤𐑦𐑒𐑑𐑩𐑛; 𐑯𐑬 𐑞𐑨𐑑 𐑩 𐑝𐑧𐑕𐑩𐑤 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑐𐑴𐑒𐑩𐑯 𐑢𐑦𐑗 𐑪𐑯 𐑞 𐑝𐑧𐑮𐑦 𐑛𐑱 𐑐𐑮𐑦𐑕𐑰𐑛𐑦𐑙 𐑣𐑨𐑛 𐑨𐑒𐑗𐑵𐑩𐑤𐑦 𐑦𐑯𐑒𐑬𐑯𐑑𐑼𐑛 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒;—𐑯 𐑯𐑬 𐑞𐑨𐑑 𐑷𐑤 𐑣𐑦𐑟 𐑕𐑩𐑒𐑕𐑧𐑕𐑦𐑝 𐑥𐑰𐑑𐑦𐑙𐑟 𐑢𐑦𐑞 𐑝𐑺𐑾𐑕 𐑖𐑦𐑐𐑕 𐑒𐑩𐑯𐑑𐑮𐑭𐑕𐑑𐑦𐑙𐑤𐑦 𐑒𐑩𐑯𐑒𐑻𐑛 𐑑 𐑖𐑴 𐑞 𐑛𐑦𐑥𐑴𐑯𐑦𐑨𐑒 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑕 𐑢𐑦𐑞 𐑢𐑦𐑗 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑑𐑹 𐑣𐑦𐑟 𐑣𐑳𐑯𐑑𐑼𐑟, 𐑢𐑧𐑞𐑼 𐑕𐑦𐑯𐑦𐑙 𐑹 𐑕𐑦𐑯𐑛 𐑩𐑜𐑱𐑯𐑕𐑑; 𐑯𐑬 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑞𐑺 𐑤𐑻𐑒𐑑 𐑩 𐑕𐑳𐑥𐑔𐑦𐑙 𐑦𐑯 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑲𐑟, 𐑢𐑦𐑗 𐑦𐑑 𐑢𐑪𐑟 𐑣𐑸𐑛𐑤𐑦 𐑕𐑳𐑓𐑼𐑩𐑚𐑩𐑤 𐑓 𐑓𐑰𐑚𐑩𐑤 𐑕𐑴𐑤𐑟 𐑑 𐑕𐑰. 𐑨𐑟 𐑞 𐑳𐑯𐑕𐑧𐑑𐑦𐑙 𐑐𐑴𐑤𐑼 𐑕𐑑𐑸, 𐑢𐑦𐑗 𐑔𐑮𐑵 𐑞 𐑤𐑦𐑝𐑤𐑪𐑙, 𐑸𐑒𐑑𐑦𐑒, 𐑕𐑦𐑒𐑕 𐑥𐑳𐑯𐑔𐑕' 𐑯𐑲𐑑 𐑕𐑩𐑕𐑑𐑱𐑯𐑟 𐑦𐑑𐑕 𐑐𐑽𐑕𐑦𐑙, 𐑕𐑑𐑧𐑛𐑦, 𐑕𐑧𐑯𐑑𐑮𐑩𐑤 𐑜𐑱𐑟; 𐑕𐑴 ·𐑱𐑣𐑨𐑚'𐑟 𐑐𐑻𐑐𐑩𐑕 𐑯𐑬 𐑓𐑦𐑒𐑕𐑩𐑛𐑤𐑦 𐑜𐑤𐑰𐑥𐑛 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑞 𐑒𐑪𐑯𐑕𐑑𐑩𐑯𐑑 𐑥𐑦𐑛𐑯𐑲𐑑 𐑝 𐑞 𐑜𐑤𐑵𐑥𐑦 𐑒𐑮𐑵. 𐑦𐑑 𐑛𐑪𐑥𐑦𐑯𐑽𐑛 𐑩𐑚𐑳𐑝 𐑞𐑧𐑥 𐑕𐑴, 𐑞𐑨𐑑 𐑷𐑤 𐑞𐑺 𐑚𐑴𐑛𐑦𐑙𐑟, 𐑛𐑬𐑑𐑕, 𐑥𐑦𐑕𐑜𐑦𐑝𐑦𐑙𐑟, 𐑓𐑽𐑟, 𐑢𐑻 𐑓𐑱𐑯 𐑑 𐑣𐑲𐑛 𐑚𐑦𐑯𐑰𐑔 𐑞𐑺 𐑕𐑴𐑤𐑟, 𐑯 𐑯𐑪𐑑 𐑕𐑐𐑮𐑬𐑑 𐑓𐑹𐑔 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑕𐑐𐑽 𐑹 𐑤𐑰𐑓.

𐑦𐑯 𐑞𐑦𐑕 𐑓𐑹𐑖𐑨𐑛𐑴𐑦𐑙 𐑦𐑯𐑑𐑼𐑝𐑩𐑤 𐑑𐑵, 𐑷𐑤 𐑣𐑿𐑥𐑼, 𐑓𐑹𐑕𐑑 𐑹 𐑯𐑨𐑗𐑼𐑩𐑤, 𐑝𐑨𐑯𐑦𐑖𐑑. 𐑕𐑑𐑳𐑚 𐑯𐑴 𐑥𐑹 𐑕𐑑𐑮𐑴𐑝 𐑑 𐑮𐑱𐑟 𐑩 𐑕𐑥𐑲𐑤; ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑯𐑴 𐑥𐑹 𐑕𐑑𐑮𐑴𐑝 𐑑 𐑗𐑧𐑒 𐑢𐑳𐑯. 𐑩𐑤𐑲𐑒, 𐑡𐑶 𐑯 𐑕𐑪𐑮𐑴, 𐑣𐑴𐑐 𐑯 𐑓𐑽, 𐑕𐑰𐑥𐑛 𐑜𐑮𐑬𐑯𐑛 𐑑 𐑓𐑲𐑯𐑩𐑕𐑑 𐑛𐑳𐑕𐑑, 𐑯 𐑐𐑬𐑛𐑼𐑛, 𐑓 𐑞 𐑑𐑲𐑥, 𐑦𐑯 𐑞 𐑒𐑤𐑨𐑥𐑐𐑑 𐑥𐑹𐑑𐑼 𐑝 ·𐑱𐑣𐑨𐑚'𐑟 𐑲𐑼𐑯 𐑕𐑴𐑤. 𐑤𐑲𐑒 𐑥𐑩𐑖𐑰𐑯𐑟, 𐑞𐑱 𐑛𐑳𐑥𐑤𐑦 𐑥𐑵𐑝𐑛 𐑩𐑚𐑬𐑑 𐑞 𐑛𐑧𐑒, 𐑧𐑝𐑼 𐑒𐑪𐑯𐑖𐑩𐑕 𐑞𐑨𐑑 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑛𐑧𐑕𐑐𐑪𐑑 𐑲 𐑢𐑪𐑟 𐑪𐑯 𐑞𐑧𐑥.

𐑯 𐑘𐑧𐑑, 𐑕𐑳𐑥𐑣𐑬, 𐑛𐑦𐑛 ·𐑱𐑣𐑨𐑚—𐑦𐑯 𐑣𐑦𐑟 𐑴𐑯 𐑐𐑮𐑪𐑐𐑼 𐑕𐑧𐑤𐑓, 𐑨𐑟 𐑛𐑱𐑤𐑦, 𐑬𐑼𐑤𐑦, 𐑯 𐑧𐑝𐑮𐑦 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑒𐑩𐑥𐑭𐑯𐑛𐑦𐑙𐑤𐑦 𐑮𐑦𐑝𐑰𐑤𐑛 𐑑 𐑣𐑦𐑟 𐑕𐑩𐑚𐑹𐑛𐑦𐑯𐑩𐑑𐑕,—·𐑱𐑣𐑨𐑚 𐑕𐑰𐑥𐑛 𐑩𐑯 𐑦𐑯𐑛𐑦𐑐𐑧𐑯𐑛𐑩𐑯𐑑 𐑤𐑹𐑛; 𐑞 ·𐑐𐑸𐑟𐑰 𐑚𐑳𐑑 𐑣𐑦𐑟 𐑕𐑤𐑱𐑝. 𐑕𐑑𐑦𐑤 𐑩𐑜𐑱𐑯 𐑚𐑴𐑔 𐑕𐑰𐑥𐑛 𐑘𐑴𐑒𐑑 𐑑𐑩𐑜𐑧𐑞𐑼, 𐑯 𐑩𐑯 𐑳𐑯𐑕𐑰𐑯 𐑑𐑲𐑮𐑩𐑯𐑑 𐑛𐑮𐑲𐑝𐑦𐑙 𐑞𐑧𐑥; 𐑞 𐑤𐑰𐑯 𐑖𐑱𐑛 𐑕𐑲𐑛𐑦𐑙 𐑞 𐑕𐑪𐑤𐑦𐑛 𐑮𐑦𐑚. 𐑓 𐑚𐑰 𐑞𐑦𐑕 ·𐑐𐑸𐑟𐑰 𐑢𐑪𐑑 𐑣𐑰 𐑥𐑱, 𐑷𐑤 𐑮𐑦𐑚 𐑯 𐑒𐑰𐑤 𐑢𐑪𐑟 𐑕𐑪𐑤𐑦𐑛 ·𐑱𐑣𐑨𐑚.

𐑨𐑑 𐑞 𐑓𐑻𐑕𐑑 𐑓𐑱𐑯𐑑𐑩𐑕𐑑 𐑜𐑤𐑦𐑥𐑼𐑦𐑙 𐑝 𐑞 𐑛𐑷𐑯, 𐑣𐑦𐑟 𐑲𐑼𐑯 𐑝𐑶𐑕 𐑢𐑪𐑟 𐑣𐑻𐑛 𐑓𐑮𐑪𐑥 𐑭𐑓𐑑,—“𐑥𐑨𐑯 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟!”—𐑯 𐑷𐑤 𐑔𐑮𐑵 𐑞 𐑛𐑱, 𐑑𐑦𐑤 𐑭𐑓𐑑𐑼 𐑕𐑳𐑯𐑕𐑧𐑑 𐑯 𐑭𐑓𐑑𐑼 𐑑𐑢𐑲𐑤𐑲𐑑, 𐑞 𐑕𐑱𐑥 𐑝𐑶𐑕 𐑧𐑝𐑮𐑦 𐑬𐑼, 𐑨𐑑 𐑞 𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑝 𐑞 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯'𐑟 𐑚𐑧𐑤, 𐑢𐑪𐑟 𐑣𐑻𐑛—“𐑢𐑪𐑑 𐑛'𐑘𐑰 𐑕𐑰?—𐑖𐑸𐑐! 𐑖𐑸𐑐!”

𐑚𐑳𐑑 𐑢𐑧𐑯 𐑔𐑮𐑰 𐑹 𐑓𐑹 𐑛𐑱𐑟 𐑣𐑨𐑛 𐑕𐑤𐑲𐑛𐑩𐑛 𐑚𐑲, 𐑭𐑓𐑑𐑼 𐑥𐑰𐑑𐑦𐑙 𐑞 𐑗𐑦𐑤𐑛𐑮𐑩𐑯-𐑕𐑰𐑒𐑦𐑙 ·𐑮𐑱𐑗𐑩𐑤; 𐑯 𐑯𐑴 𐑕𐑐𐑬𐑑 𐑣𐑨𐑛 𐑘𐑧𐑑 𐑚𐑰𐑯 𐑕𐑰𐑯; 𐑞 𐑥𐑪𐑯𐑩𐑥𐑱𐑯𐑦𐑨𐑒 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑕𐑰𐑥𐑛 𐑛𐑦𐑕𐑑𐑮𐑳𐑕𐑑𐑓𐑩𐑤 𐑝 𐑣𐑦𐑟 𐑒𐑮𐑵'𐑟 𐑓𐑦𐑛𐑧𐑤𐑦𐑑𐑦; 𐑨𐑑 𐑤𐑰𐑕𐑑, 𐑝 𐑯𐑽𐑤𐑦 𐑷𐑤 𐑦𐑒𐑕𐑧𐑐𐑑 𐑞 ·𐑐𐑱𐑜𐑩𐑯 𐑣𐑸𐑐𐑵𐑯𐑽𐑟; 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑑 𐑛𐑬𐑑, 𐑰𐑝𐑩𐑯, 𐑢𐑧𐑞𐑼 ·𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒 𐑥𐑲𐑑 𐑯𐑪𐑑 𐑢𐑦𐑤𐑦𐑙𐑤𐑦 𐑴𐑝𐑼𐑤𐑫𐑒 𐑞 𐑕𐑲𐑑 𐑣𐑰 𐑕𐑷𐑑. 𐑚𐑳𐑑 𐑦𐑓 𐑞𐑰𐑟 𐑕𐑩𐑕𐑐𐑦𐑖𐑩𐑯𐑟 𐑢𐑻 𐑮𐑾𐑤𐑦 𐑣𐑦𐑟, 𐑣𐑰 𐑕𐑩𐑜𐑱𐑖𐑩𐑕𐑤𐑦 𐑮𐑦𐑓𐑮𐑱𐑯𐑛 𐑓𐑮𐑪𐑥 𐑝𐑻𐑚𐑩𐑤𐑦 𐑦𐑒𐑕𐑐𐑮𐑧𐑕𐑦𐑙 𐑞𐑧𐑥, 𐑣𐑬𐑧𐑝𐑼 𐑣𐑦𐑟 𐑨𐑒𐑖𐑩𐑯𐑟 𐑥𐑲𐑑 𐑕𐑰𐑥 𐑑 𐑣𐑦𐑯𐑑 𐑞𐑧𐑥.

“𐑲 𐑢𐑦𐑤 𐑣𐑨𐑝 𐑞 𐑓𐑻𐑕𐑑 𐑕𐑲𐑑 𐑝 𐑞 𐑢𐑱𐑤 𐑥𐑲𐑕𐑧𐑤𐑓,”—𐑣𐑰 𐑕𐑧𐑛. “𐑲! 𐑱𐑣𐑨𐑚 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑞 𐑛𐑳𐑚𐑤𐑵𐑯!” 𐑯 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑴𐑯 𐑣𐑨𐑯𐑛𐑟 𐑣𐑰 𐑮𐑦𐑜𐑛 𐑩 𐑯𐑧𐑕𐑑 𐑝 𐑚𐑭𐑕𐑒𐑩𐑑𐑩𐑛 𐑚𐑴𐑤𐑦𐑯𐑟; 𐑯 𐑕𐑧𐑯𐑛𐑦𐑙 𐑩 𐑣𐑨𐑯𐑛 𐑩𐑤𐑪𐑓𐑑, 𐑢𐑦𐑞 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑖𐑰𐑝𐑛 𐑚𐑤𐑪𐑒, 𐑑 𐑕𐑦𐑒𐑘𐑫𐑼 𐑑 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑 𐑣𐑧𐑛, 𐑣𐑰 𐑮𐑦𐑕𐑰𐑝𐑛 𐑞 𐑑𐑵 𐑧𐑯𐑛𐑟 𐑝 𐑞 𐑛𐑬𐑯𐑢𐑼𐑛-𐑮𐑰𐑝𐑛 𐑮𐑴𐑐; 𐑯 𐑩𐑑𐑨𐑗𐑦𐑙 𐑢𐑳𐑯 𐑑 𐑣𐑦𐑟 𐑚𐑭𐑕𐑒𐑩𐑑 𐑐𐑮𐑦𐑐𐑺𐑛 𐑩 𐑐𐑦𐑯 𐑓 𐑞 𐑳𐑞𐑼 𐑧𐑯𐑛, 𐑦𐑯 𐑹𐑛𐑼 𐑑 𐑓𐑭𐑕𐑩𐑯 𐑦𐑑 𐑨𐑑 𐑞 𐑮𐑱𐑤. 𐑞𐑦𐑕 𐑛𐑳𐑯, 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑧𐑯𐑛 𐑘𐑧𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑯 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑦𐑕𐑲𐑛 𐑞 𐑐𐑦𐑯, 𐑣𐑰 𐑤𐑫𐑒𐑑 𐑮𐑬𐑯𐑛 𐑩𐑐𐑪𐑯 𐑣𐑦𐑟 𐑒𐑮𐑵, 𐑕𐑢𐑰𐑐𐑦𐑙 𐑓𐑮𐑪𐑥 𐑢𐑳𐑯 𐑑 𐑞 𐑳𐑞𐑼; 𐑐𐑷𐑟𐑦𐑙 𐑣𐑦𐑟 𐑜𐑤𐑭𐑯𐑕 𐑤𐑪𐑙 𐑩𐑐𐑪𐑯 ·𐑛𐑨𐑜𐑵, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴; 𐑚𐑳𐑑 𐑖𐑳𐑯𐑦𐑙 ·𐑓𐑧𐑛𐑩𐑤𐑩; 𐑯 𐑞𐑧𐑯 𐑕𐑧𐑑𐑩𐑤𐑦𐑙 𐑣𐑦𐑟 𐑓𐑻𐑥 𐑮𐑦𐑤𐑲𐑦𐑙 𐑲 𐑩𐑐𐑪𐑯 𐑞 𐑗𐑰𐑓 𐑥𐑱𐑑, 𐑕𐑧𐑛,—“𐑑𐑱𐑒 𐑞 𐑮𐑴𐑐, 𐑕𐑻—𐑲 𐑜𐑦𐑝 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑞𐑲 𐑣𐑨𐑯𐑛𐑟, ·𐑕𐑑𐑸𐑚𐑳𐑒.” 𐑞𐑧𐑯 𐑼𐑱𐑯𐑡𐑦𐑙 𐑣𐑦𐑟 𐑐𐑻𐑕𐑩𐑯 𐑦𐑯 𐑞 𐑚𐑭𐑕𐑒𐑩𐑑, 𐑣𐑰 𐑜𐑱𐑝 𐑞 𐑢𐑻𐑛 𐑓 𐑞𐑧𐑥 𐑑 𐑣𐑶𐑕𐑑 𐑣𐑦𐑥 𐑑 𐑣𐑦𐑟 𐑐𐑻𐑗, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑚𐑰𐑦𐑙 𐑞 𐑢𐑳𐑯 𐑣𐑵 𐑕𐑦𐑒𐑘𐑫𐑼𐑛 𐑞 𐑮𐑴𐑐 𐑨𐑑 𐑤𐑭𐑕𐑑; 𐑯 𐑭𐑓𐑑𐑼𐑢𐑼𐑛𐑟 𐑕𐑑𐑫𐑛 𐑯𐑽 𐑦𐑑. 𐑯 𐑞𐑳𐑕, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑒𐑤𐑦𐑙𐑦𐑙 𐑮𐑬𐑯𐑛 𐑞 𐑮𐑶𐑩𐑤 𐑥𐑭𐑕𐑑, ·𐑱𐑣𐑨𐑚 𐑜𐑱𐑟𐑛 𐑩𐑚𐑮𐑷𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑰 𐑓 𐑥𐑲𐑤𐑟 𐑯 𐑥𐑲𐑤𐑟,—𐑩𐑣𐑧𐑛, 𐑩𐑕𐑑𐑻𐑯, 𐑞𐑦𐑕 𐑕𐑲𐑛, 𐑯 𐑞𐑨𐑑,—𐑢𐑦𐑞𐑦𐑯 𐑞 𐑢𐑲𐑛 𐑦𐑒𐑕𐑐𐑨𐑯𐑛𐑩𐑛 𐑕𐑻𐑒𐑩𐑤 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛 𐑨𐑑 𐑕𐑴 𐑜𐑮𐑱𐑑 𐑩 𐑣𐑲𐑑.

𐑯𐑬, 𐑞 𐑓𐑻𐑕𐑑 𐑑𐑲𐑥 ·𐑱𐑣𐑨𐑚 𐑢𐑪𐑟 𐑐𐑻𐑗𐑑 𐑩𐑤𐑪𐑓𐑑; 𐑺 𐑣𐑰 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑞𐑺 𐑑𐑧𐑯 𐑥𐑦𐑯𐑦𐑑𐑕; 𐑢𐑳𐑯 𐑝 𐑞𐑴𐑟 𐑮𐑧𐑛-𐑚𐑦𐑤𐑛 𐑕𐑨𐑝𐑦𐑡 𐑕𐑰-𐑣𐑷𐑒𐑕 𐑢𐑦𐑗 𐑕𐑴 𐑪𐑓𐑩𐑯 𐑓𐑤𐑲 𐑦𐑯𐑒𐑩𐑥𐑴𐑛𐑾𐑕𐑤𐑦 𐑒𐑤𐑴𐑕 𐑮𐑬𐑯𐑛 𐑞 𐑥𐑨𐑯𐑛 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑝 𐑢𐑱𐑤𐑥𐑧𐑯 𐑦𐑯 𐑞𐑰𐑟 𐑤𐑨𐑑𐑦𐑑𐑿𐑛𐑟; 𐑢𐑳𐑯 𐑝 𐑞𐑰𐑟 𐑚𐑻𐑛𐑟 𐑒𐑱𐑥 𐑢𐑰𐑤𐑦𐑙 𐑯 𐑕𐑒𐑮𐑰𐑥𐑦𐑙 𐑮𐑬𐑯𐑛 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑦𐑯 𐑩 𐑥𐑱𐑟 𐑝 𐑳𐑯𐑑𐑮𐑨𐑒𐑩𐑚𐑤𐑦 𐑕𐑢𐑦𐑓𐑑 𐑕𐑻𐑒𐑩𐑤𐑦𐑙𐑟. 𐑞𐑧𐑯 𐑦𐑑 𐑛𐑸𐑑𐑩𐑛 𐑩 𐑔𐑬𐑟𐑩𐑯𐑛 𐑓𐑰𐑑 𐑕𐑑𐑮𐑱𐑑 𐑳𐑐 𐑦𐑯𐑑𐑵 𐑞 𐑺; 𐑞𐑧𐑯 𐑕𐑐𐑲𐑮𐑩𐑤𐑲𐑟𐑛 𐑛𐑬𐑯𐑢𐑼𐑛𐑟, 𐑯 𐑢𐑧𐑯𐑑 𐑧𐑛𐑦𐑦𐑙 𐑩𐑜𐑱𐑯 𐑮𐑬𐑯𐑛 𐑣𐑦𐑟 𐑣𐑧𐑛.

𐑚𐑳𐑑 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑜𐑱𐑟 𐑓𐑦𐑒𐑕𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑛𐑦𐑥 𐑯 𐑛𐑦𐑕𐑑𐑩𐑯𐑑 𐑣𐑼𐑲𐑟𐑩𐑯, ·𐑱𐑣𐑨𐑚 𐑕𐑰𐑥𐑛 𐑯𐑪𐑑 𐑑 𐑥𐑸𐑒 𐑞𐑦𐑕 𐑢𐑲𐑤𐑛 𐑚𐑻𐑛; 𐑯𐑹, 𐑦𐑯𐑛𐑰𐑛, 𐑢𐑫𐑛 𐑧𐑯𐑦 𐑢𐑳𐑯 𐑧𐑤𐑕 𐑣𐑨𐑝 𐑥𐑸𐑒𐑑 𐑦𐑑 𐑥𐑳𐑗, 𐑦𐑑 𐑚𐑰𐑦𐑙 𐑯𐑴 𐑳𐑯𐑒𐑪𐑥𐑩𐑯 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕; 𐑴𐑯𐑤𐑦 𐑯𐑬 𐑷𐑤𐑥𐑴𐑕𐑑 𐑞 𐑤𐑰𐑕𐑑 𐑣𐑰𐑛𐑓𐑩𐑤 𐑲 𐑕𐑰𐑥𐑛 𐑑 𐑕𐑰 𐑕𐑳𐑥 𐑕𐑹𐑑 𐑝 𐑒𐑳𐑯𐑦𐑙 𐑥𐑰𐑯𐑦𐑙 𐑦𐑯 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑝𐑮𐑦 𐑕𐑲𐑑.

“𐑘𐑹 𐑣𐑨𐑑, 𐑘𐑹 𐑣𐑨𐑑, 𐑕𐑻!” 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑒𐑮𐑲𐑛 𐑞 ·𐑕𐑦𐑕𐑦𐑤𐑾𐑯 𐑕𐑰𐑥𐑩𐑯, 𐑣𐑵 𐑚𐑰𐑦𐑙 𐑐𐑴𐑕𐑑𐑩𐑛 𐑨𐑑 𐑞 𐑥𐑦𐑟𐑩𐑯-𐑥𐑭𐑕𐑑-𐑣𐑧𐑛, 𐑕𐑑𐑫𐑛 𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑚𐑦𐑣𐑲𐑯𐑛 ·𐑱𐑣𐑨𐑚, 𐑞𐑴 𐑕𐑳𐑥𐑢𐑪𐑑 𐑤𐑴𐑼 𐑞𐑨𐑯 𐑣𐑦𐑟 𐑤𐑧𐑝𐑩𐑤, 𐑯 𐑢𐑦𐑞 𐑩 𐑛𐑰𐑐 𐑜𐑳𐑤𐑓 𐑝 𐑺 𐑛𐑦𐑝𐑲𐑛𐑦𐑙 𐑞𐑧𐑥.

𐑚𐑳𐑑 𐑷𐑤𐑮𐑧𐑛𐑦 𐑞 𐑕𐑱𐑚𐑩𐑤 𐑢𐑦𐑙 𐑢𐑪𐑟 𐑚𐑦𐑓𐑹 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑲𐑟; 𐑞 𐑤𐑪𐑙 𐑣𐑫𐑒𐑑 𐑚𐑦𐑤 𐑨𐑑 𐑣𐑦𐑟 𐑣𐑧𐑛: 𐑢𐑦𐑞 𐑩 𐑕𐑒𐑮𐑰𐑥, 𐑞 𐑚𐑤𐑨𐑒 𐑣𐑷𐑒 𐑛𐑸𐑑𐑩𐑛 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑐𐑮𐑲𐑟.

𐑩𐑯 𐑰𐑜𐑩𐑤 𐑓𐑤𐑵 𐑔𐑮𐑲𐑕 𐑮𐑬𐑯𐑛 ·𐑑𐑸𐑒𐑢𐑦𐑯'𐑟 𐑣𐑧𐑛, 𐑮𐑦𐑥𐑵𐑝𐑦𐑙 𐑣𐑦𐑟 𐑒𐑨𐑐 𐑑 𐑮𐑦𐑐𐑤𐑱𐑕 𐑦𐑑, 𐑯 𐑞𐑺𐑩𐑐𐑪𐑯 ·𐑑𐑨𐑯𐑨𐑒𐑢𐑦𐑤, 𐑣𐑦𐑟 𐑢𐑲𐑓, 𐑛𐑦𐑒𐑤𐑺𐑛 𐑞𐑨𐑑 ·𐑑𐑸𐑒𐑢𐑦𐑯 𐑢𐑫𐑛 𐑚𐑰 𐑒𐑦𐑙 𐑝 ·𐑮𐑴𐑥. 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑚𐑲 𐑞 𐑮𐑦𐑐𐑤𐑱𐑕𐑦𐑙 𐑝 𐑞 𐑒𐑨𐑐 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑴𐑥𐑩𐑯 𐑩𐑒𐑬𐑯𐑑𐑩𐑛 𐑜𐑫𐑛. 𐑱𐑣𐑨𐑚'𐑟 𐑣𐑨𐑑 𐑢𐑪𐑟 𐑯𐑧𐑝𐑼 𐑮𐑦𐑕𐑑𐑹𐑛; 𐑞 𐑢𐑲𐑤𐑛 𐑣𐑷𐑒 𐑓𐑤𐑵 𐑪𐑯 𐑯 𐑪𐑯 𐑢𐑦𐑞 𐑦𐑑; 𐑓𐑸 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕 𐑝 𐑞 𐑐𐑮𐑬: 𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑛𐑦𐑕𐑩𐑐𐑽𐑛; 𐑢𐑲𐑤 𐑓𐑮𐑪𐑥 𐑞 𐑐𐑶𐑯𐑑 𐑝 𐑞𐑨𐑑 𐑛𐑦𐑕𐑩𐑐𐑽𐑩𐑯𐑕, 𐑩 𐑥𐑲𐑯𐑿𐑑 𐑚𐑤𐑨𐑒 𐑕𐑐𐑪𐑑 𐑢𐑪𐑟 𐑛𐑦𐑥𐑤𐑦 𐑛𐑦𐑕𐑻𐑯𐑛, 𐑓𐑷𐑤𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞𐑨𐑑 𐑝𐑭𐑕𐑑 𐑣𐑲𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰.
