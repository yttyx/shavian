·𐑗𐑨𐑐𐑑𐑼 48. 𐑞 𐑓𐑻𐑕𐑑 ·𐑤𐑴𐑼𐑦𐑙.

𐑞 𐑓𐑨𐑯𐑑𐑩𐑥𐑟, 𐑓 𐑕𐑴 𐑞𐑱 𐑞𐑧𐑯 𐑕𐑰𐑥𐑛, 𐑢𐑻 𐑓𐑤𐑦𐑑𐑦𐑙 𐑪𐑯 𐑞 𐑳𐑞𐑼 𐑕𐑲𐑛 𐑝 𐑞 𐑛𐑧𐑒, 𐑯, 𐑢𐑦𐑞 𐑩 𐑯𐑶𐑟𐑤𐑩𐑕 𐑕𐑩𐑤𐑧𐑮𐑦𐑑𐑦, 𐑢𐑻 𐑒𐑭𐑕𐑑𐑦𐑙 𐑤𐑵𐑕 𐑞 𐑑𐑨𐑒𐑩𐑤𐑟 𐑯 𐑚𐑨𐑯𐑛𐑟 𐑝 𐑞 𐑚𐑴𐑑 𐑢𐑦𐑗 𐑕𐑢𐑳𐑙 𐑞𐑺. 𐑞𐑦𐑕 𐑚𐑴𐑑 𐑣𐑨𐑛 𐑷𐑤𐑢𐑱𐑟 𐑚𐑰𐑯 𐑛𐑰𐑥𐑛 𐑢𐑳𐑯 𐑝 𐑞 𐑕𐑐𐑺 𐑚𐑴𐑑𐑕, 𐑞𐑴 𐑑𐑧𐑒𐑯𐑦𐑒𐑤𐑦 𐑒𐑷𐑤𐑛 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯'𐑟, 𐑪𐑯 𐑩𐑒𐑬𐑯𐑑 𐑝 𐑦𐑑𐑕 𐑣𐑨𐑙𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑑𐑸𐑚𐑼𐑛 𐑒𐑢𐑹𐑑𐑼. 𐑞 𐑓𐑦𐑜𐑼 𐑞𐑨𐑑 𐑯𐑬 𐑕𐑑𐑫𐑛 𐑚𐑲 𐑦𐑑𐑕 𐑚𐑬𐑟 𐑢𐑪𐑟 𐑑𐑷𐑤 𐑯 𐑕𐑢𐑸𐑑, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑢𐑲𐑑 𐑑𐑵𐑔 𐑰𐑝𐑩𐑤𐑦 𐑐𐑮𐑴𐑑𐑮𐑵𐑛𐑦𐑙 𐑓𐑮𐑪𐑥 𐑦𐑑𐑕 𐑕𐑑𐑰𐑤-𐑤𐑲𐑒 𐑤𐑦𐑐𐑕. 𐑩 𐑮𐑳𐑥𐑐𐑩𐑤𐑛 ·𐑗𐑲𐑯𐑰𐑟 𐑡𐑨𐑒𐑩𐑑 𐑝 𐑚𐑤𐑨𐑒 𐑒𐑪𐑑𐑩𐑯 𐑓𐑿𐑯𐑽𐑾𐑤𐑦 𐑦𐑯𐑝𐑧𐑕𐑑𐑩𐑛 𐑣𐑦𐑥, 𐑢𐑦𐑞 𐑢𐑲𐑛 𐑚𐑤𐑨𐑒 𐑑𐑮𐑴𐑟𐑼𐑟 𐑝 𐑞 𐑕𐑱𐑥 𐑛𐑸𐑒 𐑕𐑑𐑳𐑓. 𐑚𐑳𐑑 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑒𐑮𐑬𐑯𐑦𐑙 𐑞𐑦𐑕 𐑦𐑚𐑪𐑯𐑯𐑩𐑕 𐑢𐑪𐑟 𐑩 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑢𐑲𐑑 𐑐𐑤𐑨𐑑𐑩𐑛 𐑑𐑻𐑚𐑩𐑯, 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑣𐑺 𐑚𐑮𐑱𐑛𐑩𐑛 𐑯 𐑒𐑶𐑤𐑛 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛 𐑩𐑐𐑪𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛. 𐑤𐑧𐑕 𐑕𐑢𐑸𐑑 𐑦𐑯 𐑨𐑕𐑐𐑧𐑒𐑑, 𐑞 𐑒𐑩𐑥𐑐𐑨𐑯𐑘𐑩𐑯𐑟 𐑝 𐑞𐑦𐑕 𐑓𐑦𐑜𐑼 𐑢𐑻 𐑝 𐑞𐑨𐑑 𐑝𐑦𐑝𐑦𐑛, 𐑑𐑲𐑜𐑼-𐑘𐑧𐑤𐑴 𐑒𐑩𐑥𐑐𐑤𐑧𐑒𐑖𐑩𐑯 𐑐𐑦𐑒𐑿𐑤𐑽 𐑑 𐑕𐑳𐑥 𐑝 𐑞 𐑨𐑚𐑼𐑦𐑡𐑦𐑯𐑩𐑤 𐑯𐑱𐑑𐑦𐑝𐑟 𐑝 𐑞 ·𐑥𐑩𐑯𐑦𐑤𐑩𐑟;—𐑩 𐑮𐑱𐑕 𐑯𐑴𐑑𐑹𐑾𐑕 𐑓 𐑩 𐑕𐑻𐑑𐑩𐑯 𐑛𐑲𐑩𐑚𐑪𐑤𐑦𐑟𐑩𐑥 𐑝 𐑕𐑳𐑚𐑑𐑦𐑤𐑑𐑦, 𐑯 𐑚𐑲 𐑕𐑳𐑥 𐑪𐑯𐑩𐑕𐑑 𐑢𐑲𐑑 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑕𐑩𐑐𐑴𐑕𐑑 𐑑 𐑚𐑰 𐑞 𐑐𐑱𐑛 𐑕𐑐𐑲𐑟 𐑯 𐑕𐑰𐑒𐑮𐑩𐑑 𐑒𐑪𐑯𐑓𐑦𐑛𐑧𐑯𐑖𐑩𐑤 𐑱𐑡𐑩𐑯𐑑𐑕 𐑪𐑯 𐑞 𐑢𐑷𐑑𐑼 𐑝 𐑞 𐑛𐑧𐑝𐑩𐑤, 𐑞𐑺 𐑤𐑹𐑛, 𐑣𐑵𐑟 𐑒𐑬𐑯𐑑𐑦𐑙-𐑮𐑵𐑥 𐑞𐑱 𐑕𐑩𐑐𐑴𐑟 𐑑 𐑚𐑰 𐑧𐑤𐑕𐑢𐑺.

𐑢𐑲𐑤 𐑘𐑧𐑑 𐑞 𐑢𐑳𐑯𐑛𐑼𐑦𐑙 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑢𐑻 𐑜𐑱𐑟𐑦𐑙 𐑩𐑐𐑪𐑯 𐑞𐑰𐑟 𐑕𐑑𐑮𐑱𐑯𐑡𐑼𐑟, ·𐑱𐑣𐑨𐑚 𐑒𐑮𐑲𐑛 𐑬𐑑 𐑑 𐑞 𐑢𐑲𐑑-𐑑𐑻𐑚𐑩𐑯𐑛 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑨𐑑 𐑞𐑺 𐑣𐑧𐑛, “𐑷𐑤 𐑮𐑧𐑛𐑦 𐑞𐑺, ·𐑓𐑧𐑛𐑩𐑤𐑩?”

“𐑮𐑧𐑛𐑦,” 𐑢𐑪𐑟 𐑞 𐑣𐑭𐑓-𐑣𐑦𐑕𐑑 𐑮𐑦𐑐𐑤𐑲.

“𐑤𐑴𐑼 𐑩𐑢𐑱 𐑞𐑧𐑯; 𐑛'𐑘𐑰 𐑣𐑽?” 𐑖𐑬𐑑𐑦𐑙 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑛𐑧𐑒. “𐑤𐑴𐑼 𐑩𐑢𐑱 𐑞𐑺, 𐑲 𐑕𐑱.”

𐑕𐑳𐑗 𐑢𐑪𐑟 𐑞 𐑔𐑳𐑯𐑛𐑼 𐑝 𐑣𐑦𐑟 𐑝𐑶𐑕, 𐑞𐑨𐑑 𐑕𐑐𐑲𐑑 𐑝 𐑞𐑺 𐑩𐑥𐑱𐑟𐑥𐑩𐑯𐑑 𐑞 𐑥𐑧𐑯 𐑕𐑐𐑮𐑨𐑙 𐑴𐑝𐑼 𐑞 𐑮𐑱𐑤; 𐑞 𐑖𐑰𐑝𐑟 𐑢𐑻𐑤𐑛 𐑮𐑬𐑯𐑛 𐑦𐑯 𐑞 𐑚𐑤𐑪𐑒𐑕; 𐑢𐑦𐑞 𐑩 𐑢𐑪𐑤𐑴, 𐑞 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕 𐑛𐑮𐑪𐑐𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰; 𐑢𐑲𐑤, 𐑢𐑦𐑞 𐑩 𐑛𐑧𐑒𐑕𐑑𐑼𐑩𐑕, 𐑪𐑓-𐑣𐑨𐑯𐑛𐑩𐑛 𐑛𐑺𐑦𐑙, 𐑳𐑯𐑯𐑴𐑯 𐑦𐑯 𐑧𐑯𐑦 𐑳𐑞𐑼 𐑝𐑴𐑒𐑱𐑖𐑩𐑯, 𐑞 𐑕𐑱𐑤𐑼𐑟, 𐑜𐑴𐑑-𐑤𐑲𐑒, 𐑤𐑰𐑐𐑑 𐑛𐑬𐑯 𐑞 𐑮𐑴𐑤𐑦𐑙 𐑖𐑦𐑐'𐑕 𐑕𐑲𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑑𐑪𐑕𐑑 𐑚𐑴𐑑𐑕 𐑚𐑦𐑤𐑴.

𐑣𐑸𐑛𐑤𐑦 𐑣𐑨𐑛 𐑞𐑱 𐑐𐑫𐑤𐑛 𐑬𐑑 𐑓𐑮𐑪𐑥 𐑳𐑯𐑛𐑼 𐑞 𐑖𐑦𐑐'𐑕 𐑤𐑰, 𐑢𐑧𐑯 𐑩 𐑓𐑹𐑔 𐑒𐑰𐑤, 𐑒𐑳𐑥𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑕𐑲𐑛, 𐑐𐑫𐑤𐑛 𐑮𐑬𐑯𐑛 𐑳𐑯𐑛𐑼 𐑞 𐑕𐑑𐑻𐑯, 𐑯 𐑖𐑴𐑛 𐑞 𐑓𐑲𐑝 𐑕𐑑𐑮𐑱𐑯𐑡𐑼𐑟 𐑮𐑴𐑦𐑙 ·𐑱𐑣𐑨𐑚, 𐑣𐑵, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑮𐑧𐑒𐑑 𐑦𐑯 𐑞 𐑕𐑑𐑻𐑯, 𐑤𐑬𐑛𐑤𐑦 𐑣𐑱𐑤𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, ·𐑕𐑑𐑳𐑚, 𐑯 ·𐑓𐑤𐑭𐑕𐑒, 𐑑 𐑕𐑐𐑮𐑧𐑛 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑢𐑲𐑛𐑤𐑦, 𐑕𐑴 𐑨𐑟 𐑑 𐑒𐑳𐑝𐑼 𐑩 𐑤𐑸𐑡 𐑦𐑒𐑕𐑐𐑨𐑯𐑕 𐑝 𐑢𐑷𐑑𐑼. 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑷𐑤 𐑞𐑺 𐑲𐑟 𐑩𐑜𐑱𐑯 𐑮𐑦𐑝𐑩𐑑𐑩𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑢𐑸𐑑 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑯 𐑣𐑦𐑟 𐑒𐑮𐑵, 𐑞 𐑦𐑯𐑥𐑱𐑑𐑕 𐑝 𐑞 𐑳𐑞𐑼 𐑚𐑴𐑑𐑕 𐑴𐑚𐑱𐑛 𐑯𐑪𐑑 𐑞 𐑒𐑩𐑥𐑭𐑯𐑛.

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚?—” 𐑕𐑧𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒.

“𐑕𐑐𐑮𐑧𐑛 𐑘𐑹𐑕𐑧𐑤𐑝𐑟,” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚; “𐑜𐑦𐑝 𐑢𐑱, 𐑷𐑤 𐑓𐑹 𐑚𐑴𐑑𐑕. 𐑞𐑬, ·𐑓𐑤𐑭𐑕𐑒, 𐑐𐑫𐑤 𐑬𐑑 𐑥𐑹 𐑑 𐑤𐑰𐑢𐑼𐑛!”

“𐑲, 𐑲, 𐑕𐑻,” 𐑗𐑽𐑦𐑤𐑦 𐑒𐑮𐑲𐑛 𐑤𐑦𐑑𐑩𐑤 ·𐑒𐑦𐑙-·𐑐𐑴𐑕𐑑, 𐑕𐑢𐑰𐑐𐑦𐑙 𐑮𐑬𐑯𐑛 𐑣𐑦𐑟 𐑜𐑮𐑱𐑑 𐑕𐑑𐑽𐑦𐑙 𐑹. “𐑤𐑱 𐑚𐑨𐑒!” 𐑩𐑛𐑮𐑧𐑕𐑦𐑙 𐑣𐑦𐑟 𐑒𐑮𐑵. “𐑞𐑺!—𐑞𐑺!—𐑞𐑺 𐑩𐑜𐑱𐑯! 𐑞𐑺 𐑖𐑰 𐑚𐑤𐑴𐑟 𐑮𐑲𐑑 𐑩𐑣𐑧𐑛, 𐑚𐑶𐑟!—𐑤𐑱 𐑚𐑨𐑒!”

“𐑯𐑧𐑝𐑼 𐑣𐑰𐑛 𐑘𐑪𐑯𐑛𐑼 𐑘𐑧𐑤𐑴 𐑚𐑶𐑟, ·𐑸𐑒𐑦.”

“𐑴, 𐑲 𐑛𐑴𐑯𐑑 𐑥𐑲𐑯𐑛 '𐑩𐑥, 𐑕𐑻,” 𐑕𐑧𐑛 ·𐑸𐑒𐑦; “𐑲 𐑯𐑿 𐑦𐑑 𐑷𐑤 𐑚𐑦𐑓𐑹 𐑯𐑬. 𐑛𐑦𐑛𐑩𐑯𐑑 𐑲 𐑣𐑽 '𐑩𐑥 𐑦𐑯 𐑞 𐑣𐑴𐑤𐑛? 𐑯 𐑛𐑦𐑛𐑩𐑯𐑑 𐑲 𐑑𐑧𐑤 ·𐑒𐑱𐑚𐑩𐑒𐑴 𐑣𐑽 𐑝 𐑦𐑑? 𐑢𐑪𐑑 𐑕𐑱 𐑘𐑰, ·𐑒𐑱𐑚𐑩𐑒𐑴? 𐑞𐑱 𐑸 𐑕𐑑𐑴𐑩𐑢𐑱𐑟, ·𐑥𐑼 ·𐑓𐑤𐑭𐑕𐑒.”

𐑢𐑺𐑩𐑐𐑪𐑯 𐑐𐑤𐑭𐑯𐑑𐑦𐑙 𐑣𐑦𐑟 𐑓𐑰𐑑 𐑓𐑻𐑥𐑤𐑦 𐑩𐑜𐑱𐑯𐑕𐑑 𐑑𐑵 𐑪𐑐𐑩𐑟𐑦𐑑 𐑐𐑤𐑨𐑙𐑒𐑕 𐑝 𐑞 𐑚𐑴𐑑, 𐑞 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑯𐑰𐑜𐑮𐑴, 𐑕𐑑𐑵𐑐𐑦𐑙 𐑩 𐑤𐑦𐑑𐑩𐑤, 𐑐𐑮𐑦𐑟𐑧𐑯𐑑𐑩𐑛 𐑣𐑦𐑟 𐑓𐑤𐑨𐑑 𐑐𐑭𐑥 𐑑 ·𐑓𐑤𐑭𐑕𐑒'𐑕 𐑓𐑫𐑑, 𐑯 𐑞𐑧𐑯 𐑐𐑫𐑑𐑦𐑙 ·𐑓𐑤𐑭𐑕𐑒'𐑕 𐑣𐑨𐑯𐑛 𐑪𐑯 𐑣𐑦𐑟 𐑣𐑻𐑕-𐑐𐑤𐑵𐑥𐑛 𐑣𐑧𐑛 𐑯 𐑚𐑦𐑛𐑦𐑙 𐑣𐑦𐑥 𐑕𐑐𐑮𐑦𐑙 𐑨𐑟 𐑣𐑰 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑖𐑵𐑛 𐑑𐑪𐑕, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑛𐑧𐑒𐑕𐑑𐑼𐑩𐑕 𐑓𐑤𐑦𐑙 𐑤𐑨𐑯𐑛𐑩𐑛 𐑞 𐑤𐑦𐑑𐑩𐑤 𐑥𐑨𐑯 𐑣𐑲 𐑯 𐑛𐑮𐑲 𐑪𐑯 𐑣𐑦𐑟 𐑖𐑴𐑤𐑛𐑼𐑟. 𐑯 𐑣𐑽 𐑢𐑪𐑟 ·𐑓𐑤𐑭𐑕𐑒 𐑯𐑬 𐑕𐑑𐑨𐑯𐑛𐑦𐑙, ·𐑛𐑨𐑜𐑵 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑤𐑦𐑓𐑑𐑩𐑛 𐑸𐑥 𐑓𐑻𐑯𐑦𐑖𐑦𐑙 𐑣𐑦𐑥 𐑢𐑦𐑞 𐑩 𐑚𐑮𐑧𐑕𐑑𐑚𐑨𐑯𐑛 𐑑 𐑤𐑰𐑯 𐑩𐑜𐑱𐑯𐑕𐑑 𐑯 𐑕𐑑𐑧𐑛𐑦 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑚𐑲.

𐑨𐑑 𐑧𐑯𐑦 𐑑𐑲𐑥 𐑦𐑑 𐑦𐑟 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑲𐑑 𐑑 𐑞 𐑑𐑲𐑮𐑴 𐑑 𐑕𐑰 𐑢𐑦𐑞 𐑢𐑪𐑑 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑣𐑨𐑚𐑦𐑑𐑿𐑛 𐑝 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕 𐑕𐑒𐑦𐑤 𐑞 𐑢𐑱𐑤𐑥𐑩𐑯 𐑢𐑦𐑤 𐑥𐑱𐑯𐑑𐑱𐑯 𐑩𐑯 𐑦𐑮𐑧𐑒𐑑 𐑐𐑪𐑕𐑗𐑼 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑴𐑑, 𐑰𐑝𐑩𐑯 𐑢𐑧𐑯 𐑐𐑦𐑗𐑑 𐑩𐑚𐑬𐑑 𐑚𐑲 𐑞 𐑥𐑴𐑕𐑑 𐑮𐑲𐑩𐑑𐑩𐑕𐑤𐑦 𐑐𐑼𐑝𐑻𐑕 𐑯 𐑒𐑮𐑪𐑕-𐑮𐑳𐑯𐑦𐑙 𐑕𐑰𐑟. 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑕𐑑𐑮𐑱𐑯𐑡 𐑑 𐑕𐑰 𐑣𐑦𐑥 𐑜𐑦𐑛𐑦𐑤𐑦 𐑐𐑻𐑗𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑤𐑪𐑜𐑼𐑣𐑧𐑛 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑳𐑯𐑛𐑼 𐑕𐑳𐑗 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕𐑩𐑟. 𐑚𐑳𐑑 𐑞 𐑕𐑲𐑑 𐑝 𐑤𐑦𐑑𐑩𐑤 ·𐑓𐑤𐑭𐑕𐑒 𐑥𐑬𐑯𐑑𐑩𐑛 𐑩𐑐𐑪𐑯 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 ·𐑛𐑨𐑜𐑵 𐑢𐑪𐑟 𐑘𐑧𐑑 𐑥𐑹 𐑒𐑘𐑫𐑼𐑾𐑕; 𐑓 𐑕𐑩𐑕𐑑𐑱𐑯𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑢𐑦𐑞 𐑩 𐑒𐑵𐑤, 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑑, 𐑰𐑟𐑦, 𐑳𐑯𐑔𐑷𐑑 𐑝, 𐑚𐑸𐑚𐑨𐑮𐑦𐑒 𐑥𐑨𐑡𐑩𐑕𐑑𐑦, 𐑞 𐑯𐑴𐑚𐑩𐑤 𐑯𐑰𐑜𐑮𐑴 𐑑 𐑧𐑝𐑮𐑦 𐑮𐑴𐑤 𐑝 𐑞 𐑕𐑰 𐑣𐑸𐑥𐑴𐑯𐑾𐑕𐑤𐑦 𐑮𐑴𐑤𐑛 𐑣𐑦𐑟 𐑓𐑲𐑯 𐑓𐑹𐑥. 𐑪𐑯 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛 𐑚𐑨𐑒, 𐑓𐑤𐑨𐑒𐑕𐑩𐑯-𐑣𐑺𐑛 ·𐑓𐑤𐑭𐑕𐑒 𐑕𐑰𐑥𐑛 𐑩 𐑕𐑯𐑴-𐑓𐑤𐑱𐑒. 𐑞 𐑚𐑺𐑼 𐑤𐑫𐑒𐑑 𐑯𐑴𐑚𐑩𐑤𐑼 𐑞𐑨𐑯 𐑞 𐑮𐑲𐑛𐑼. 𐑞𐑴 𐑑𐑮𐑵𐑤𐑦 𐑝𐑦𐑝𐑱𐑖𐑩𐑕, 𐑑𐑿𐑥𐑳𐑤𐑗𐑵𐑩𐑕, 𐑪𐑕𐑑𐑧𐑯𐑑𐑱𐑖𐑩𐑕 𐑤𐑦𐑑𐑩𐑤 ·𐑓𐑤𐑭𐑕𐑒 𐑢𐑫𐑛 𐑯𐑬 𐑯 𐑞𐑧𐑯 𐑕𐑑𐑨𐑥𐑐 𐑢𐑦𐑞 𐑦𐑥𐑐𐑱𐑖𐑩𐑯𐑕; 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑢𐑳𐑯 𐑨𐑛𐑩𐑛 𐑣𐑰𐑝 𐑛𐑦𐑛 𐑣𐑰 𐑞𐑺𐑚𐑲 𐑜𐑦𐑝 𐑑 𐑞 𐑯𐑰𐑜𐑮𐑴'𐑟 𐑤𐑹𐑛𐑤𐑦 𐑗𐑧𐑕𐑑. 𐑕𐑴 𐑣𐑨𐑝 𐑲 𐑕𐑰𐑯 ·𐑐𐑭𐑖𐑩𐑯 𐑯 ·𐑝𐑨𐑯𐑦𐑑𐑦 𐑕𐑑𐑨𐑥𐑐𐑦𐑙 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑥𐑨𐑜𐑯𐑨𐑯𐑦𐑥𐑩𐑕 𐑻𐑔, 𐑚𐑳𐑑 𐑞 𐑻𐑔 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑷𐑤𐑑𐑼 𐑣𐑻 𐑑𐑲𐑛𐑟 𐑯 𐑣𐑻 𐑕𐑰𐑟𐑩𐑯𐑟 𐑓 𐑞𐑨𐑑.

𐑥𐑰𐑯𐑢𐑲𐑤 ·𐑕𐑑𐑳𐑚, 𐑞 𐑔𐑻𐑛 𐑥𐑱𐑑, 𐑚𐑦𐑑𐑮𐑱𐑛 𐑯𐑴 𐑕𐑳𐑗 𐑓𐑸-𐑜𐑱𐑟𐑦𐑙 𐑕𐑩𐑤𐑦𐑕𐑦𐑑𐑿𐑛𐑟. 𐑞 𐑢𐑱𐑤𐑟 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑥𐑱𐑛 𐑢𐑳𐑯 𐑝 𐑞𐑺 𐑮𐑧𐑜𐑿𐑤𐑼 𐑕𐑬𐑯𐑛𐑦𐑙𐑟, 𐑯𐑪𐑑 𐑩 𐑑𐑧𐑥𐑐𐑼𐑼𐑦 𐑛𐑲𐑝 𐑓𐑮𐑪𐑥 𐑥𐑽 𐑓𐑮𐑲𐑑; 𐑯 𐑦𐑓 𐑞𐑨𐑑 𐑢𐑻 𐑞 𐑒𐑱𐑕, ·𐑕𐑑𐑳𐑚, 𐑨𐑟 𐑣𐑦𐑟 𐑢𐑴𐑯𐑑 𐑦𐑯 𐑕𐑳𐑗 𐑒𐑱𐑕𐑩𐑟, 𐑦𐑑 𐑕𐑰𐑥𐑟, 𐑢𐑪𐑟 𐑮𐑦𐑟𐑪𐑤𐑝𐑛 𐑑 𐑕𐑪𐑤𐑦𐑕 𐑞 𐑤𐑨𐑙𐑜𐑢𐑦𐑖𐑦𐑙 𐑦𐑯𐑑𐑼𐑝𐑩𐑤 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑐𐑲𐑐. 𐑣𐑰 𐑢𐑦𐑞𐑛𐑮𐑵 𐑦𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑣𐑨𐑑𐑚𐑨𐑯𐑛, 𐑢𐑺 𐑣𐑰 𐑷𐑤𐑢𐑱𐑟 𐑢𐑹 𐑦𐑑 𐑩𐑕𐑤𐑭𐑯𐑑 𐑤𐑲𐑒 𐑩 𐑓𐑧𐑞𐑼. 𐑣𐑰 𐑤𐑴𐑛𐑩𐑛 𐑦𐑑, 𐑯 𐑮𐑨𐑥𐑛 𐑣𐑴𐑥 𐑞 𐑤𐑴𐑛𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑔𐑳𐑥-𐑧𐑯𐑛; 𐑚𐑳𐑑 𐑣𐑸𐑛𐑤𐑦 𐑣𐑨𐑛 𐑣𐑰 𐑦𐑜𐑯𐑲𐑑𐑩𐑛 𐑣𐑦𐑟 𐑥𐑨𐑗 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑮𐑳𐑓 𐑕𐑨𐑯𐑛𐑐𐑱𐑐𐑼 𐑝 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛, 𐑢𐑧𐑯 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, 𐑣𐑦𐑟 𐑣𐑸𐑐𐑵𐑯𐑽, 𐑣𐑵𐑟 𐑲𐑟 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑧𐑑𐑦𐑙 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑤𐑲𐑒 𐑑𐑵 𐑓𐑦𐑒𐑕𐑑 𐑕𐑑𐑸𐑟, 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑛𐑮𐑪𐑐𐑑 𐑤𐑲𐑒 𐑤𐑲𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑦𐑮𐑧𐑒𐑑 𐑨𐑑𐑦𐑑𐑿𐑛 𐑑 𐑣𐑦𐑟 𐑕𐑰𐑑, 𐑒𐑮𐑲𐑦𐑙 𐑬𐑑 𐑦𐑯 𐑩 𐑒𐑢𐑦𐑒 𐑓𐑮𐑧𐑯𐑟𐑦 𐑝 𐑣𐑳𐑮𐑦, “𐑛𐑬𐑯, 𐑛𐑬𐑯 𐑷𐑤, 𐑯 𐑜𐑦𐑝 𐑢𐑱!—𐑞𐑺 𐑞𐑱 𐑸!”

𐑑 𐑩 𐑤𐑨𐑯𐑛𐑟𐑥𐑩𐑯, 𐑯𐑴 𐑢𐑱𐑤, 𐑯𐑹 𐑧𐑯𐑦 𐑕𐑲𐑯 𐑝 𐑩 𐑣𐑧𐑮𐑦𐑙, 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑨𐑑 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑; 𐑯𐑳𐑔𐑦𐑙 𐑚𐑳𐑑 𐑩 𐑑𐑮𐑳𐑚𐑩𐑤𐑛 𐑚𐑦𐑑 𐑝 𐑜𐑮𐑰𐑯𐑦𐑖 𐑢𐑲𐑑 𐑢𐑷𐑑𐑼, 𐑯 𐑔𐑦𐑯 𐑕𐑒𐑨𐑑𐑼𐑛 𐑐𐑳𐑓𐑕 𐑝 𐑝𐑱𐑐𐑼 𐑣𐑪𐑝𐑼𐑦𐑙 𐑴𐑝𐑼 𐑦𐑑, 𐑯 𐑕𐑩𐑓𐑿𐑟𐑦𐑙𐑤𐑦 𐑚𐑤𐑴𐑦𐑙 𐑪𐑓 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑤𐑲𐑒 𐑞 𐑒𐑩𐑯𐑓𐑿𐑟𐑛 𐑕𐑒𐑳𐑛 𐑓𐑮𐑪𐑥 𐑢𐑲𐑑 𐑮𐑴𐑤𐑦𐑙 𐑚𐑦𐑤𐑴𐑟. 𐑞 𐑺 𐑼𐑬𐑯𐑛 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑝𐑲𐑚𐑮𐑱𐑑𐑩𐑛 𐑯 𐑑𐑦𐑙𐑜𐑩𐑤𐑛, 𐑨𐑟 𐑦𐑑 𐑢𐑻, 𐑤𐑲𐑒 𐑞 𐑺 𐑴𐑝𐑼 𐑦𐑯𐑑𐑧𐑯𐑕𐑤𐑦 𐑣𐑰𐑑𐑩𐑛 𐑐𐑤𐑱𐑑𐑕 𐑝 𐑲𐑼𐑯. 𐑚𐑦𐑯𐑰𐑔 𐑞𐑦𐑕 𐑨𐑑𐑥𐑩𐑕𐑓𐑧𐑮𐑦𐑒 𐑢𐑱𐑝𐑦𐑙 𐑯 𐑒𐑻𐑤𐑦𐑙, 𐑯 𐑐𐑸𐑖𐑩𐑤𐑦 𐑚𐑦𐑯𐑰𐑔 𐑩 𐑔𐑦𐑯 𐑤𐑱𐑼 𐑝 𐑢𐑷𐑑𐑼, 𐑷𐑤𐑕𐑴, 𐑞 𐑢𐑱𐑤𐑟 𐑢𐑻 𐑕𐑢𐑦𐑥𐑦𐑙. 𐑕𐑰𐑯 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕 𐑝 𐑷𐑤 𐑞 𐑳𐑞𐑼 𐑦𐑯𐑛𐑦𐑒𐑱𐑖𐑩𐑯𐑟, 𐑞 𐑐𐑳𐑓𐑕 𐑝 𐑝𐑱𐑐𐑼 𐑞𐑱 𐑕𐑐𐑬𐑑𐑩𐑛, 𐑕𐑰𐑥𐑛 𐑞𐑺 𐑓𐑹𐑮𐑳𐑯𐑦𐑙 𐑒𐑫𐑮𐑽𐑟 𐑯 𐑛𐑦𐑑𐑨𐑗𐑑 𐑓𐑤𐑲𐑦𐑙 𐑬𐑑𐑮𐑲𐑛𐑼𐑟.

𐑷𐑤 𐑓𐑹 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑯𐑬 𐑦𐑯 𐑒𐑰𐑯 𐑐𐑼𐑕𐑿𐑑 𐑝 𐑞𐑨𐑑 𐑢𐑳𐑯 𐑕𐑐𐑪𐑑 𐑝 𐑑𐑮𐑳𐑚𐑩𐑤𐑛 𐑢𐑷𐑑𐑼 𐑯 𐑺. 𐑚𐑳𐑑 𐑦𐑑 𐑚𐑨𐑛 𐑓𐑺 𐑑 𐑬𐑑𐑕𐑑𐑮𐑦𐑐 𐑞𐑧𐑥; 𐑦𐑑 𐑓𐑤𐑵 𐑪𐑯 𐑯 𐑪𐑯, 𐑨𐑟 𐑩 𐑥𐑨𐑕 𐑝 𐑦𐑯𐑑𐑼𐑚𐑤𐑧𐑯𐑛𐑦𐑙 𐑚𐑳𐑚𐑩𐑤𐑟 𐑚𐑹𐑯 𐑛𐑬𐑯 𐑩 𐑮𐑨𐑐𐑦𐑛 𐑕𐑑𐑮𐑰𐑥 𐑓𐑮𐑪𐑥 𐑞 𐑣𐑦𐑤𐑟.

“𐑐𐑫𐑤, 𐑐𐑫𐑤, 𐑥𐑲 𐑜𐑫𐑛 𐑚𐑶𐑟,” 𐑕𐑧𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑦𐑯 𐑞 𐑤𐑴𐑩𐑕𐑑 𐑐𐑪𐑕𐑩𐑚𐑩𐑤 𐑚𐑳𐑑 𐑦𐑯𐑑𐑧𐑯𐑕𐑩𐑕𐑑 𐑒𐑪𐑯𐑕𐑩𐑯𐑑𐑮𐑱𐑑𐑩𐑛 𐑢𐑦𐑕𐑐𐑼 𐑑 𐑣𐑦𐑟 𐑥𐑧𐑯; 𐑢𐑲𐑤 𐑞 𐑖𐑸𐑐 𐑓𐑦𐑒𐑕𐑑 𐑜𐑤𐑭𐑯𐑕 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑲𐑟 𐑛𐑸𐑑𐑩𐑛 𐑕𐑑𐑮𐑱𐑑 𐑩𐑣𐑧𐑛 𐑝 𐑞 𐑚𐑬, 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰𐑥𐑛 𐑨𐑟 𐑑𐑵 𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑯𐑰𐑛𐑩𐑤𐑟 𐑦𐑯 𐑑𐑵 𐑳𐑯𐑻𐑦𐑙 𐑚𐑦𐑯𐑩𐑒𐑩𐑤 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟. 𐑣𐑰 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑕𐑱 𐑥𐑳𐑗 𐑑 𐑣𐑦𐑟 𐑒𐑮𐑵, 𐑞𐑴, 𐑯𐑹 𐑛𐑦𐑛 𐑣𐑦𐑟 𐑒𐑮𐑵 𐑕𐑱 𐑧𐑯𐑦𐑔𐑦𐑙 𐑑 𐑣𐑦𐑥. 𐑴𐑯𐑤𐑦 𐑞 𐑕𐑲𐑤𐑩𐑯𐑕 𐑝 𐑞 𐑚𐑴𐑑 𐑢𐑪𐑟 𐑨𐑑 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑕𐑑𐑸𐑑𐑩𐑤𐑦𐑙𐑤𐑦 𐑐𐑽𐑕𐑑 𐑚𐑲 𐑢𐑳𐑯 𐑝 𐑣𐑦𐑟 𐑐𐑦𐑒𐑿𐑤𐑽 𐑢𐑦𐑕𐑐𐑼𐑟, 𐑯𐑬 𐑣𐑸𐑖 𐑢𐑦𐑞 𐑒𐑩𐑥𐑭𐑯𐑛, 𐑯𐑬 𐑕𐑪𐑓𐑑 𐑢𐑦𐑞 𐑦𐑯𐑑𐑮𐑰𐑑𐑦.

𐑣𐑬 𐑛𐑦𐑓𐑼𐑩𐑯𐑑 𐑞 𐑤𐑬𐑛 𐑤𐑦𐑑𐑩𐑤 ·𐑒𐑦𐑙-·𐑐𐑴𐑕𐑑. “𐑕𐑦𐑙 𐑬𐑑 𐑯 𐑕𐑱 𐑕𐑳𐑥𐑔𐑦𐑙, 𐑥𐑲 𐑣𐑸𐑑𐑦𐑟. 𐑮𐑹 𐑯 𐑐𐑫𐑤, 𐑥𐑲 𐑔𐑳𐑯𐑛𐑼𐑚𐑴𐑤𐑑𐑕! 𐑚𐑰𐑗 𐑥𐑰, 𐑚𐑰𐑗 𐑥𐑰 𐑪𐑯 𐑞𐑺 𐑚𐑤𐑨𐑒 𐑚𐑨𐑒𐑕, 𐑚𐑶𐑟; 𐑴𐑯𐑤𐑦 𐑛𐑵 𐑞𐑨𐑑 𐑓 𐑥𐑰, 𐑯 𐑲'𐑤 𐑕𐑲𐑯 𐑴𐑝𐑼 𐑑 𐑿 𐑥𐑲 ·𐑥𐑸𐑔𐑩'𐑟 ·𐑝𐑦𐑯𐑘𐑼𐑛 𐑐𐑤𐑭𐑯𐑑𐑱𐑖𐑩𐑯, 𐑚𐑶𐑟; 𐑦𐑯𐑒𐑤𐑵𐑛𐑦𐑙 𐑢𐑲𐑓 𐑯 𐑗𐑦𐑤𐑛𐑮𐑩𐑯, 𐑚𐑶𐑟. 𐑤𐑱 𐑥𐑰 𐑪𐑯—𐑤𐑱 𐑥𐑰 𐑪𐑯! 𐑴 ·𐑤𐑹𐑛, ·𐑤𐑹𐑛! 𐑚𐑳𐑑 𐑲 𐑖𐑨𐑤 𐑜𐑴 𐑕𐑑𐑸𐑒, 𐑕𐑑𐑺𐑦𐑙 𐑥𐑨𐑛! 𐑕𐑰! 𐑕𐑰 𐑞𐑨𐑑 𐑢𐑲𐑑 𐑢𐑷𐑑𐑼!” 𐑯 𐑕𐑴 𐑖𐑬𐑑𐑦𐑙, 𐑣𐑰 𐑐𐑫𐑤𐑛 𐑣𐑦𐑟 𐑣𐑨𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑯 𐑕𐑑𐑨𐑥𐑐𐑑 𐑳𐑐 𐑯 𐑛𐑬𐑯 𐑪𐑯 𐑦𐑑; 𐑞𐑧𐑯 𐑐𐑦𐑒𐑦𐑙 𐑦𐑑 𐑳𐑐, 𐑓𐑤𐑻𐑑𐑩𐑛 𐑦𐑑 𐑓𐑸 𐑪𐑓 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑰; 𐑯 𐑓𐑲𐑯𐑩𐑤𐑦 𐑓𐑧𐑤 𐑑 𐑮𐑽𐑦𐑙 𐑯 𐑐𐑤𐑳𐑯𐑡𐑦𐑙 𐑦𐑯 𐑞 𐑚𐑴𐑑'𐑕 𐑕𐑑𐑻𐑯 𐑤𐑲𐑒 𐑩 𐑒𐑮𐑱𐑟𐑛 𐑒𐑴𐑤𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑐𐑮𐑺𐑦.

“𐑤𐑫𐑒 𐑨𐑑 𐑞𐑨𐑑 𐑗𐑨𐑐 𐑯𐑬,” 𐑓𐑦𐑤𐑩𐑕𐑪𐑓𐑦𐑒𐑤𐑦 𐑛𐑮𐑷𐑤𐑛 ·𐑕𐑑𐑳𐑚, 𐑣𐑵, 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑳𐑯𐑤𐑲𐑑𐑩𐑛 𐑖𐑹𐑑 𐑐𐑲𐑐, 𐑥𐑦𐑒𐑨𐑯𐑦𐑒𐑤𐑦 𐑮𐑦𐑑𐑱𐑯𐑛 𐑚𐑦𐑑𐑢𐑰𐑯 𐑣𐑦𐑟 𐑑𐑰𐑔, 𐑨𐑑 𐑩 𐑖𐑹𐑑 𐑛𐑦𐑕𐑑𐑩𐑯𐑕, 𐑓𐑪𐑤𐑴𐑛 𐑭𐑓𐑑𐑼—“𐑣𐑰'𐑟 𐑜𐑪𐑑 𐑓𐑦𐑑𐑕, 𐑞𐑨𐑑 ·𐑓𐑤𐑭𐑕𐑒 𐑣𐑨𐑟. 𐑓𐑦𐑑𐑕? 𐑘𐑧𐑕, 𐑜𐑦𐑝 𐑣𐑦𐑥 𐑓𐑦𐑑𐑕—𐑞𐑨𐑑'𐑕 𐑞 𐑝𐑧𐑮𐑦 𐑢𐑻𐑛—𐑐𐑦𐑗 𐑓𐑦𐑑𐑕 𐑦𐑯𐑑𐑵 '𐑩𐑥. 𐑥𐑧𐑮𐑦𐑤𐑦, 𐑥𐑧𐑮𐑦𐑤𐑦, 𐑣𐑸𐑑𐑕-𐑩𐑤𐑲𐑝. 𐑐𐑫𐑛𐑦𐑙 𐑓 𐑕𐑳𐑐𐑼, 𐑿 𐑯𐑴;—𐑥𐑧𐑮𐑦'𐑟 𐑞 𐑢𐑻𐑛. 𐑐𐑫𐑤, 𐑚𐑱𐑚𐑟—𐑐𐑫𐑤, 𐑕𐑳𐑒𐑩𐑤𐑦𐑙𐑟—𐑐𐑫𐑤, 𐑷𐑤. 𐑚𐑳𐑑 𐑢𐑪𐑑 𐑞 𐑛𐑧𐑝𐑩𐑤 𐑸 𐑿 𐑣𐑳𐑮𐑦𐑦𐑙 𐑩𐑚𐑬𐑑? 𐑕𐑪𐑓𐑑𐑤𐑦, 𐑕𐑪𐑓𐑑𐑤𐑦, 𐑯 𐑕𐑑𐑧𐑛𐑦𐑤𐑦, 𐑥𐑲 𐑥𐑧𐑯. 𐑴𐑯𐑤𐑦 𐑐𐑫𐑤, 𐑯 𐑒𐑰𐑐 𐑐𐑫𐑤𐑦𐑙; 𐑯𐑳𐑔𐑦𐑙 𐑥𐑹. 𐑒𐑮𐑨𐑒 𐑷𐑤 𐑘𐑹 𐑚𐑨𐑒𐑚𐑴𐑯𐑟, 𐑯 𐑚𐑲𐑑 𐑘𐑹 𐑯𐑲𐑝𐑟 𐑦𐑯 𐑑𐑵—𐑞𐑨𐑑'𐑕 𐑷𐑤. 𐑑𐑱𐑒 𐑦𐑑 𐑰𐑟𐑦—𐑢𐑲 𐑛𐑴𐑯𐑑 𐑘𐑰 𐑑𐑱𐑒 𐑦𐑑 𐑰𐑟𐑦, 𐑲 𐑕𐑱, 𐑯 𐑚𐑻𐑕𐑑 𐑷𐑤 𐑘𐑹 𐑤𐑦𐑝𐑼𐑟 𐑯 𐑤𐑳𐑙𐑟!”

𐑚𐑳𐑑 𐑢𐑪𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑦𐑯𐑕𐑒𐑮𐑵𐑑𐑩𐑚𐑩𐑤 ·𐑱𐑣𐑨𐑚 𐑕𐑧𐑛 𐑑 𐑞𐑨𐑑 𐑑𐑲𐑜𐑼-𐑘𐑧𐑤𐑴 𐑒𐑮𐑵 𐑝 𐑣𐑦𐑟—𐑞𐑰𐑟 𐑢𐑻 𐑢𐑻𐑛𐑟 𐑚𐑧𐑕𐑑 𐑴𐑥𐑦𐑑𐑩𐑛 𐑣𐑽; 𐑓 𐑿 𐑤𐑦𐑝 𐑳𐑯𐑛𐑼 𐑞 𐑚𐑤𐑧𐑕𐑩𐑛 𐑤𐑲𐑑 𐑝 𐑞 𐑰𐑝𐑨𐑯𐑡𐑧𐑤𐑦𐑒𐑩𐑤 𐑤𐑨𐑯𐑛. 𐑴𐑯𐑤𐑦 𐑞 𐑦𐑯𐑓𐑦𐑛𐑩𐑤 𐑖𐑸𐑒𐑕 𐑦𐑯 𐑞 𐑷𐑛𐑱𐑖𐑩𐑕 𐑕𐑰𐑟 𐑥𐑱 𐑜𐑦𐑝 𐑽 𐑑 𐑕𐑳𐑗 𐑢𐑻𐑛𐑟, 𐑢𐑧𐑯, 𐑢𐑦𐑞 𐑑𐑹𐑯𐑱𐑛𐑴 𐑚𐑮𐑬, 𐑯 𐑲𐑟 𐑝 𐑮𐑧𐑛 𐑥𐑻𐑛𐑼, 𐑯 𐑓𐑴𐑥-𐑜𐑤𐑵𐑛 𐑤𐑦𐑐𐑕, ·𐑱𐑣𐑨𐑚 𐑤𐑰𐑐𐑑 𐑭𐑓𐑑𐑼 𐑣𐑦𐑟 𐑐𐑮𐑱.

𐑥𐑰𐑯𐑢𐑲𐑤, 𐑷𐑤 𐑞 𐑚𐑴𐑑𐑕 𐑑𐑹 𐑪𐑯. 𐑞 𐑮𐑦𐑐𐑰𐑑𐑩𐑛 𐑕𐑐𐑩𐑕𐑦𐑓𐑦𐑒 𐑩𐑤𐑵𐑠𐑩𐑯𐑟 𐑝 ·𐑓𐑤𐑭𐑕𐑒 𐑑 “𐑞𐑨𐑑 𐑢𐑱𐑤,” 𐑨𐑟 𐑣𐑰 𐑒𐑷𐑤𐑛 𐑞 𐑓𐑦𐑒𐑑𐑦𐑖𐑩𐑕 𐑥𐑪𐑯𐑕𐑑𐑼 𐑢𐑦𐑗 𐑣𐑰 𐑛𐑦𐑒𐑤𐑺𐑛 𐑑 𐑚𐑰 𐑦𐑯𐑕𐑧𐑕𐑩𐑯𐑑𐑤𐑦 𐑑𐑨𐑯𐑑𐑩𐑤𐑲𐑟𐑦𐑙 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑚𐑬 𐑢𐑦𐑞 𐑦𐑑𐑕 𐑑𐑱𐑤—𐑞𐑰𐑟 𐑩𐑤𐑵𐑠𐑩𐑯𐑟 𐑝 𐑣𐑦𐑟 𐑢𐑻 𐑨𐑑 𐑑𐑲𐑥𐑟 𐑕𐑴 𐑝𐑦𐑝𐑦𐑛 𐑯 𐑤𐑲𐑓-𐑤𐑲𐑒, 𐑞𐑨𐑑 𐑞𐑱 𐑢𐑫𐑛 𐑒𐑷𐑟 𐑕𐑳𐑥 𐑢𐑳𐑯 𐑹 𐑑𐑵 𐑝 𐑣𐑦𐑟 𐑥𐑧𐑯 𐑑 𐑕𐑯𐑨𐑗 𐑩 𐑓𐑽𐑓𐑩𐑤 𐑤𐑫𐑒 𐑴𐑝𐑼 𐑞 𐑖𐑴𐑤𐑛𐑼. 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑢𐑪𐑟 𐑩𐑜𐑱𐑯𐑕𐑑 𐑷𐑤 𐑮𐑵𐑤; 𐑓 𐑞 𐑹𐑟𐑥𐑧𐑯 𐑥𐑳𐑕𐑑 𐑐𐑫𐑑 𐑬𐑑 𐑞𐑺 𐑲𐑟, 𐑯 𐑮𐑨𐑥 𐑩 𐑕𐑒𐑿𐑼 𐑔𐑮𐑵 𐑞𐑺 𐑯𐑧𐑒𐑕; 𐑿𐑕𐑦𐑡 𐑐𐑮𐑩𐑯𐑬𐑯𐑕𐑦𐑙 𐑞𐑨𐑑 𐑞𐑱 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑯𐑴 𐑹𐑜𐑩𐑯𐑟 𐑚𐑳𐑑 𐑽𐑟, 𐑯 𐑯𐑴 𐑤𐑦𐑥𐑟 𐑚𐑳𐑑 𐑸𐑥𐑟, 𐑦𐑯 𐑞𐑰𐑟 𐑒𐑮𐑦𐑑𐑦𐑒𐑩𐑤 𐑥𐑴𐑥𐑧𐑯𐑑𐑕.

𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑕𐑲𐑑 𐑓𐑫𐑤 𐑝 𐑒𐑢𐑦𐑒 𐑢𐑳𐑯𐑛𐑼 𐑯 𐑷! 𐑞 𐑝𐑭𐑕𐑑 𐑕𐑢𐑧𐑤𐑟 𐑝 𐑞 𐑪𐑥𐑯𐑦𐑐𐑩𐑑𐑩𐑯𐑑 𐑕𐑰; 𐑞 𐑕𐑻𐑡𐑦𐑙, 𐑣𐑪𐑤𐑴 𐑮𐑹 𐑞𐑱 𐑥𐑱𐑛, 𐑨𐑟 𐑞𐑱 𐑮𐑴𐑤𐑛 𐑩𐑤𐑪𐑙 𐑞 𐑱𐑑 𐑜𐑳𐑯𐑩𐑤𐑟, 𐑤𐑲𐑒 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑚𐑴𐑤𐑟 𐑦𐑯 𐑩 𐑚𐑬𐑯𐑛𐑤𐑩𐑕 𐑚𐑴𐑤𐑦𐑙-𐑜𐑮𐑰𐑯; 𐑞 𐑚𐑮𐑰𐑓 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑨𐑜𐑩𐑯𐑦 𐑝 𐑞 𐑚𐑴𐑑, 𐑨𐑟 𐑦𐑑 𐑢𐑫𐑛 𐑑𐑦𐑐 𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑪𐑯 𐑞 𐑯𐑲𐑓-𐑤𐑲𐑒 𐑧𐑡 𐑝 𐑞 𐑖𐑸𐑐𐑼 𐑢𐑱𐑝𐑟, 𐑞𐑨𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰𐑥𐑛 𐑔𐑮𐑧𐑑𐑩𐑯𐑦𐑙 𐑑 𐑒𐑳𐑑 𐑦𐑑 𐑦𐑯 𐑑𐑵; 𐑞 𐑕𐑳𐑛𐑩𐑯 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑛𐑦𐑐 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑷𐑑𐑼𐑦 𐑜𐑤𐑧𐑯𐑟 𐑯 𐑣𐑪𐑤𐑴𐑟; 𐑞 𐑒𐑰𐑯 𐑕𐑐𐑻𐑦𐑙𐑟 𐑯 𐑜𐑴𐑛𐑦𐑙𐑟 𐑑 𐑜𐑱𐑯 𐑞 𐑑𐑪𐑐 𐑝 𐑞 𐑪𐑐𐑩𐑟𐑦𐑑 𐑣𐑦𐑤; 𐑞 𐑣𐑧𐑛𐑤𐑪𐑙, 𐑕𐑤𐑧𐑛-𐑤𐑲𐑒 𐑕𐑤𐑲𐑛 𐑛𐑬𐑯 𐑦𐑑𐑕 𐑳𐑞𐑼 𐑕𐑲𐑛;—𐑷𐑤 𐑞𐑰𐑟, 𐑢𐑦𐑞 𐑞 𐑒𐑮𐑲𐑟 𐑝 𐑞 𐑣𐑧𐑛𐑟𐑥𐑧𐑯 𐑯 𐑣𐑸𐑐𐑵𐑯𐑽𐑟, 𐑯 𐑞 𐑖𐑳𐑛𐑼𐑦𐑙 𐑜𐑭𐑕𐑐𐑕 𐑝 𐑞 𐑹𐑟𐑥𐑧𐑯, 𐑢𐑦𐑞 𐑞 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑕𐑲𐑑 𐑝 𐑞 𐑲𐑝𐑼𐑦 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑚𐑺𐑦𐑙 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑣𐑻 𐑚𐑴𐑑𐑕 𐑢𐑦𐑞 𐑬𐑑𐑕𐑑𐑮𐑧𐑗𐑑 𐑕𐑱𐑤𐑟, 𐑤𐑲𐑒 𐑩 𐑢𐑲𐑤𐑛 𐑣𐑧𐑯 𐑭𐑓𐑑𐑼 𐑣𐑻 𐑕𐑒𐑮𐑰𐑥𐑦𐑙 𐑚𐑮𐑵𐑛;—𐑷𐑤 𐑞𐑦𐑕 𐑢𐑪𐑟 𐑔𐑮𐑦𐑤𐑦𐑙.

𐑕𐑵𐑯 𐑢𐑰 𐑢𐑻 𐑮𐑳𐑯𐑦𐑙 𐑔𐑮𐑵 𐑩 𐑕𐑩𐑓𐑿𐑟𐑦𐑙 𐑢𐑲𐑛 𐑝𐑱𐑤 𐑝 𐑥𐑦𐑕𐑑; 𐑯𐑲𐑞𐑼 𐑖𐑦𐑐 𐑯𐑹 𐑚𐑴𐑑 𐑑 𐑚𐑰 𐑕𐑰𐑯.

“𐑜𐑦𐑝 𐑢𐑱, 𐑥𐑧𐑯,” 𐑢𐑦𐑕𐑐𐑼𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑛𐑮𐑷𐑦𐑙 𐑕𐑑𐑦𐑤 𐑓𐑻𐑞𐑼 𐑭𐑓𐑑 𐑞 𐑖𐑰𐑑 𐑝 𐑣𐑦𐑟 𐑕𐑱𐑤; “𐑞𐑺 𐑦𐑟 𐑑𐑲𐑥 𐑑 𐑒𐑦𐑤 𐑩 𐑓𐑦𐑖 𐑘𐑧𐑑 𐑚𐑦𐑓𐑹 𐑞 𐑕𐑒𐑢𐑷𐑤 𐑒𐑳𐑥𐑟. 𐑞𐑺'𐑟 𐑢𐑲𐑑 𐑢𐑷𐑑𐑼 𐑩𐑜𐑱𐑯!—𐑒𐑤𐑴𐑕 𐑑! 𐑕𐑐𐑮𐑦𐑙!”

𐑕𐑵𐑯 𐑭𐑓𐑑𐑼, 𐑑𐑵 𐑒𐑮𐑲𐑟 𐑦𐑯 𐑒𐑢𐑦𐑒 𐑕𐑩𐑒𐑕𐑧𐑖𐑩𐑯 𐑪𐑯 𐑰𐑗 𐑕𐑲𐑛 𐑝 𐑳𐑕 𐑛𐑦𐑯𐑴𐑑𐑩𐑛 𐑞𐑨𐑑 𐑞 𐑳𐑞𐑼 𐑚𐑴𐑑𐑕 𐑣𐑨𐑛 𐑜𐑪𐑑 𐑓𐑭𐑕𐑑; 𐑚𐑳𐑑 𐑣𐑸𐑛𐑤𐑦 𐑢𐑻 𐑞𐑱 𐑴𐑝𐑼𐑣𐑻𐑛, 𐑢𐑧𐑯 𐑢𐑦𐑞 𐑩 𐑤𐑲𐑑𐑯𐑦𐑙-𐑤𐑲𐑒 𐑣𐑻𐑑𐑩𐑤𐑦𐑙 𐑢𐑦𐑕𐑐𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑕𐑧𐑛: “𐑕𐑑𐑨𐑯𐑛 𐑳𐑐!” 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑣𐑸𐑐𐑵𐑯 𐑦𐑯 𐑣𐑨𐑯𐑛, 𐑕𐑐𐑮𐑨𐑙 𐑑 𐑣𐑦𐑟 𐑓𐑰𐑑.

𐑞𐑴 𐑯𐑪𐑑 𐑢𐑳𐑯 𐑝 𐑞 𐑹𐑟𐑥𐑧𐑯 𐑢𐑪𐑟 𐑞𐑧𐑯 𐑓𐑱𐑕𐑦𐑙 𐑞 𐑤𐑲𐑓 𐑯 𐑛𐑧𐑔 𐑐𐑧𐑮𐑦𐑤 𐑕𐑴 𐑒𐑤𐑴𐑕 𐑑 𐑞𐑧𐑥 𐑩𐑣𐑧𐑛, 𐑘𐑧𐑑 𐑢𐑦𐑞 𐑞𐑺 𐑲𐑟 𐑪𐑯 𐑞 𐑦𐑯𐑑𐑧𐑯𐑕 𐑒𐑬𐑯𐑑𐑩𐑯𐑩𐑯𐑕 𐑝 𐑞 𐑥𐑱𐑑 𐑦𐑯 𐑞 𐑕𐑑𐑻𐑯 𐑝 𐑞 𐑚𐑴𐑑, 𐑞𐑱 𐑯𐑿 𐑞𐑨𐑑 𐑞 𐑦𐑥𐑦𐑯𐑩𐑯𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑣𐑨𐑛 𐑒𐑳𐑥; 𐑞𐑱 𐑣𐑻𐑛, 𐑑𐑵, 𐑩𐑯 𐑦𐑯𐑹𐑥𐑩𐑕 𐑢𐑪𐑤𐑴𐑦𐑙 𐑕𐑬𐑯𐑛 𐑨𐑟 𐑝 𐑓𐑦𐑓𐑑𐑦 𐑧𐑤𐑦𐑓𐑩𐑯𐑑𐑕 𐑕𐑑𐑻𐑦𐑙 𐑦𐑯 𐑞𐑺 𐑤𐑦𐑑𐑼. 𐑥𐑰𐑯𐑢𐑲𐑤 𐑞 𐑚𐑴𐑑 𐑢𐑪𐑟 𐑕𐑑𐑦𐑤 𐑚𐑵𐑥𐑦𐑙 𐑔𐑮𐑵 𐑞 𐑥𐑦𐑕𐑑, 𐑞 𐑢𐑱𐑝𐑟 𐑒𐑻𐑤𐑦𐑙 𐑯 𐑣𐑦𐑕𐑦𐑙 𐑼𐑬𐑯𐑛 𐑳𐑕 𐑤𐑲𐑒 𐑞 𐑦𐑮𐑧𐑒𐑑𐑩𐑛 𐑒𐑮𐑧𐑕𐑑𐑕 𐑝 𐑦𐑯𐑮𐑱𐑡𐑛 𐑕𐑻𐑐𐑩𐑯𐑑𐑕.

“𐑞𐑨𐑑'𐑕 𐑣𐑦𐑟 𐑣𐑳𐑥𐑐. 𐑞𐑺, 𐑞𐑺, 𐑜𐑦𐑝 𐑦𐑑 𐑑 𐑣𐑦𐑥!” 𐑢𐑦𐑕𐑐𐑼𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒.

𐑩 𐑖𐑹𐑑 𐑮𐑳𐑖𐑦𐑙 𐑕𐑬𐑯𐑛 𐑤𐑰𐑐𐑑 𐑬𐑑 𐑝 𐑞 𐑚𐑴𐑑; 𐑦𐑑 𐑢𐑪𐑟 𐑞 𐑛𐑸𐑑𐑩𐑛 𐑲𐑼𐑯 𐑝 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜. 𐑞𐑧𐑯 𐑷𐑤 𐑦𐑯 𐑢𐑳𐑯 𐑢𐑧𐑤𐑛𐑩𐑛 𐑒𐑩𐑥𐑴𐑖𐑩𐑯 𐑒𐑱𐑥 𐑩𐑯 𐑦𐑯𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑐𐑫𐑖 𐑓𐑮𐑪𐑥 𐑩𐑕𐑑𐑻𐑯, 𐑢𐑲𐑤 𐑓𐑹𐑢𐑼𐑛 𐑞 𐑚𐑴𐑑 𐑕𐑰𐑥𐑛 𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑪𐑯 𐑩 𐑤𐑧𐑡; 𐑞 𐑕𐑱𐑤 𐑒𐑩𐑤𐑨𐑐𐑕𐑑 𐑯 𐑦𐑒𐑕𐑐𐑤𐑴𐑛𐑩𐑛; 𐑩 𐑜𐑳𐑖 𐑝 𐑕𐑒𐑷𐑤𐑛𐑦𐑙 𐑝𐑱𐑐𐑼 𐑖𐑪𐑑 𐑳𐑐 𐑯𐑽 𐑚𐑲; 𐑕𐑳𐑥𐑔𐑦𐑙 𐑮𐑴𐑤𐑛 𐑯 𐑑𐑳𐑥𐑚𐑩𐑤𐑛 𐑤𐑲𐑒 𐑩𐑯 𐑻𐑔𐑒𐑢𐑱𐑒 𐑚𐑦𐑯𐑰𐑔 𐑳𐑕. 𐑞 𐑣𐑴𐑤 𐑒𐑮𐑵 𐑢𐑻 𐑣𐑭𐑓 𐑕𐑳𐑓𐑩𐑒𐑱𐑑𐑩𐑛 𐑨𐑟 𐑞𐑱 𐑢𐑻 𐑑𐑪𐑕𐑑 𐑣𐑧𐑤𐑑𐑼-𐑕𐑒𐑧𐑤𐑑𐑼 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑲𐑑 𐑒𐑻𐑛𐑩𐑤𐑦𐑙 𐑒𐑮𐑰𐑥 𐑝 𐑞 𐑕𐑒𐑢𐑷𐑤. 𐑕𐑒𐑢𐑷𐑤, 𐑢𐑱𐑤, 𐑯 𐑣𐑸𐑐𐑵𐑯 𐑣𐑨𐑛 𐑷𐑤 𐑚𐑤𐑧𐑯𐑛𐑩𐑛 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑯 𐑞 𐑢𐑱𐑤, 𐑥𐑽𐑤𐑦 𐑜𐑮𐑱𐑟𐑛 𐑚𐑲 𐑞 𐑲𐑼𐑯, 𐑦𐑕𐑒𐑱𐑐𐑑.

𐑞 𐑢𐑦𐑯𐑛 𐑦𐑯𐑒𐑮𐑰𐑕𐑑 𐑑 𐑩 𐑣𐑬𐑤; 𐑞 𐑢𐑱𐑝𐑟 𐑛𐑨𐑖𐑑 𐑞𐑺 𐑚𐑳𐑒𐑩𐑤𐑼𐑟 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑞 𐑣𐑴𐑤 𐑕𐑒𐑢𐑷𐑤 𐑮𐑹𐑛, 𐑓𐑹𐑒𐑑, 𐑯 𐑒𐑮𐑨𐑒𐑩𐑤𐑛 𐑼𐑬𐑯𐑛 𐑳𐑕 𐑤𐑲𐑒 𐑩 𐑢𐑲𐑑 𐑓𐑲𐑼 𐑩𐑐𐑪𐑯 𐑞 𐑐𐑮𐑺𐑦, 𐑦𐑯 𐑢𐑦𐑗, 𐑳𐑯𐑒𐑩𐑯𐑕𐑿𐑥𐑛, 𐑢𐑰 𐑢𐑻 𐑚𐑻𐑯𐑦𐑙; 𐑦𐑥𐑹𐑑𐑩𐑤 𐑦𐑯 𐑞𐑰𐑟 𐑡𐑷𐑟 𐑝 𐑛𐑧𐑔! 𐑦𐑯 𐑝𐑱𐑯 𐑢𐑰 𐑣𐑱𐑤𐑛 𐑞 𐑳𐑞𐑼 𐑚𐑴𐑑𐑕; 𐑨𐑟 𐑢𐑧𐑤 𐑮𐑹 𐑑 𐑞 𐑤𐑲𐑝 𐑒𐑴𐑤𐑟 𐑛𐑬𐑯 𐑞 𐑗𐑦𐑥𐑯𐑦 𐑝 𐑩 𐑓𐑤𐑱𐑥𐑦𐑙 𐑓𐑻𐑯𐑦𐑕 𐑨𐑟 𐑣𐑱𐑤 𐑞𐑴𐑟 𐑚𐑴𐑑𐑕 𐑦𐑯 𐑞𐑨𐑑 𐑕𐑑𐑹𐑥. 𐑥𐑰𐑯𐑢𐑲𐑤 𐑞 𐑛𐑮𐑲𐑝𐑦𐑙 𐑕𐑒𐑳𐑛, 𐑮𐑨𐑒, 𐑯 𐑥𐑦𐑕𐑑, 𐑜𐑮𐑵 𐑛𐑸𐑒𐑼 𐑢𐑦𐑞 𐑞 𐑖𐑨𐑛𐑴𐑟 𐑝 𐑯𐑲𐑑; 𐑯𐑴 𐑕𐑲𐑯 𐑝 𐑞 𐑖𐑦𐑐 𐑒𐑫𐑛 𐑚𐑰 𐑕𐑰𐑯. 𐑞 𐑮𐑲𐑟𐑦𐑙 𐑕𐑰 𐑓𐑼𐑚𐑨𐑛 𐑷𐑤 𐑩𐑑𐑧𐑥𐑐𐑑𐑕 𐑑 𐑚𐑱𐑤 𐑬𐑑 𐑞 𐑚𐑴𐑑. 𐑞 𐑹𐑟 𐑢𐑻 𐑿𐑕𐑤𐑩𐑕 𐑨𐑟 𐑐𐑮𐑩𐑐𐑧𐑤𐑼𐑟, 𐑐𐑼𐑓𐑹𐑥𐑦𐑙 𐑯𐑬 𐑞 𐑪𐑓𐑦𐑕 𐑝 𐑤𐑲𐑓-𐑐𐑮𐑦𐑟𐑻𐑝𐑼𐑟. 𐑕𐑴, 𐑒𐑳𐑑𐑦𐑙 𐑞 𐑤𐑨𐑖𐑦𐑙 𐑝 𐑞 𐑢𐑷𐑑𐑼𐑐𐑮𐑵𐑓 𐑥𐑨𐑗 𐑒𐑧𐑜, 𐑭𐑓𐑑𐑼 𐑥𐑧𐑯𐑦 𐑓𐑱𐑤𐑘𐑼𐑟 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑒𐑩𐑯𐑑𐑮𐑲𐑝𐑛 𐑑 𐑦𐑜𐑯𐑲𐑑 𐑞 𐑤𐑨𐑥𐑐 𐑦𐑯 𐑞 𐑤𐑨𐑯𐑑𐑼𐑯; 𐑞𐑧𐑯 𐑕𐑑𐑮𐑧𐑗𐑦𐑙 𐑦𐑑 𐑪𐑯 𐑩 𐑢𐑱𐑓 𐑐𐑴𐑤, 𐑣𐑨𐑯𐑛𐑩𐑛 𐑦𐑑 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑨𐑟 𐑞 𐑕𐑑𐑨𐑯𐑛𐑼𐑛-𐑚𐑺𐑼 𐑝 𐑞𐑦𐑕 𐑓𐑼𐑤𐑹𐑯 𐑣𐑴𐑐. 𐑞𐑺, 𐑞𐑧𐑯, 𐑣𐑰 𐑕𐑨𐑑, 𐑣𐑴𐑤𐑛𐑦𐑙 𐑳𐑐 𐑞𐑨𐑑 𐑦𐑥𐑚𐑩𐑕𐑰𐑤 𐑒𐑨𐑯𐑛𐑩𐑤 𐑦𐑯 𐑞 𐑣𐑸𐑑 𐑝 𐑞𐑨𐑑 𐑷𐑤𐑥𐑲𐑑𐑦 𐑓𐑼𐑤𐑹𐑯𐑯𐑩𐑕. 𐑞𐑺, 𐑞𐑧𐑯, 𐑣𐑰 𐑕𐑨𐑑, 𐑞 𐑕𐑲𐑯 𐑯 𐑕𐑦𐑥𐑚𐑩𐑤 𐑝 𐑩 𐑥𐑨𐑯 𐑢𐑦𐑞𐑬𐑑 𐑓𐑱𐑔, 𐑣𐑴𐑐𐑤𐑩𐑕𐑤𐑦 𐑣𐑴𐑤𐑛𐑦𐑙 𐑳𐑐 𐑣𐑴𐑐 𐑦𐑯 𐑞 𐑥𐑦𐑛𐑕𐑑 𐑝 𐑛𐑦𐑕𐑐𐑺.
