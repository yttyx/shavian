·𐑗𐑨𐑐𐑑𐑼 114. 𐑞 ·𐑜𐑦𐑤𐑛𐑼.

𐑐𐑧𐑯𐑦𐑑𐑮𐑱𐑑𐑦𐑙 𐑓𐑻𐑞𐑼 𐑯 𐑓𐑻𐑞𐑼 𐑦𐑯𐑑𐑵 𐑞 𐑣𐑸𐑑 𐑝 𐑞 ·𐑡𐑨𐑐𐑩𐑯𐑰𐑟 𐑒𐑮𐑵𐑟𐑦𐑙 𐑜𐑮𐑬𐑯𐑛, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑕𐑵𐑯 𐑷𐑤 𐑩𐑕𐑑𐑻 𐑦𐑯 𐑞 𐑓𐑦𐑖𐑼𐑦. 𐑪𐑓𐑩𐑯, 𐑦𐑯 𐑥𐑲𐑤𐑛, 𐑐𐑤𐑧𐑟𐑩𐑯𐑑 𐑢𐑧𐑞𐑼, 𐑓 𐑑𐑢𐑧𐑤𐑝, 𐑓𐑦𐑓𐑑𐑰𐑯, 𐑱𐑑𐑰𐑯, 𐑯 𐑑𐑢𐑧𐑯𐑑𐑦 𐑬𐑼𐑟 𐑪𐑯 𐑞 𐑕𐑑𐑮𐑧𐑗, 𐑞𐑱 𐑢𐑻 𐑦𐑯𐑜𐑱𐑡𐑛 𐑦𐑯 𐑞 𐑚𐑴𐑑𐑕, 𐑕𐑑𐑧𐑛𐑦𐑤𐑦 𐑐𐑫𐑤𐑦𐑙, 𐑹 𐑕𐑱𐑤𐑦𐑙, 𐑹 𐑐𐑨𐑛𐑩𐑤𐑦𐑙 𐑭𐑓𐑑𐑼 𐑞 𐑢𐑱𐑤𐑟, 𐑹 𐑓 𐑩𐑯 𐑦𐑯𐑑𐑼𐑤𐑵𐑛 𐑝 𐑕𐑦𐑒𐑕𐑑𐑦 𐑹 𐑕𐑧𐑝𐑩𐑯𐑑𐑦 𐑥𐑦𐑯𐑦𐑑𐑕 𐑒𐑭𐑥𐑤𐑦 𐑩𐑢𐑱𐑑𐑦𐑙 𐑞𐑺 𐑳𐑐𐑮𐑲𐑟𐑦𐑙; 𐑞𐑴 𐑢𐑦𐑞 𐑚𐑳𐑑 𐑕𐑥𐑷𐑤 𐑕𐑩𐑒𐑕𐑧𐑕 𐑓 𐑞𐑺 𐑐𐑱𐑯𐑟.

𐑨𐑑 𐑕𐑳𐑗 𐑑𐑲𐑥𐑟, 𐑳𐑯𐑛𐑼 𐑩𐑯 𐑩𐑚𐑱𐑑𐑩𐑛 𐑕𐑳𐑯; 𐑩𐑓𐑤𐑴𐑑 𐑷𐑤 𐑛𐑱 𐑩𐑐𐑪𐑯 𐑕𐑥𐑵𐑞, 𐑕𐑤𐑴 𐑣𐑰𐑝𐑦𐑙 𐑕𐑢𐑧𐑤𐑟; 𐑕𐑰𐑑𐑩𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑴𐑑, 𐑤𐑲𐑑 𐑨𐑟 𐑩 𐑚𐑻𐑗 𐑒𐑩𐑯𐑵; 𐑯 𐑕𐑴 𐑕𐑴𐑖𐑩𐑚𐑤𐑦 𐑥𐑦𐑒𐑕𐑦𐑙 𐑢𐑦𐑞 𐑞 𐑕𐑪𐑓𐑑 𐑢𐑱𐑝𐑟 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟, 𐑞𐑨𐑑 𐑤𐑲𐑒 𐑣𐑸𐑔-𐑕𐑑𐑴𐑯 𐑒𐑨𐑑𐑕 𐑞𐑱 𐑐𐑻 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑜𐑳𐑯𐑩𐑤; 𐑞𐑰𐑟 𐑸 𐑞 𐑑𐑲𐑥𐑟 𐑝 𐑛𐑮𐑰𐑥𐑦 𐑒𐑢𐑲𐑩𐑑𐑵𐑛, 𐑢𐑧𐑯 𐑚𐑦𐑣𐑴𐑤𐑛𐑦𐑙 𐑞 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤 𐑚𐑿𐑑𐑦 𐑯 𐑚𐑮𐑦𐑤𐑾𐑯𐑕𐑦 𐑝 𐑞 𐑴𐑖𐑩𐑯'𐑟 𐑕𐑒𐑦𐑯, 𐑢𐑳𐑯 𐑓𐑼𐑜𐑧𐑑𐑕 𐑞 𐑑𐑲𐑜𐑼 𐑣𐑸𐑑 𐑞𐑨𐑑 𐑐𐑨𐑯𐑑𐑕 𐑚𐑦𐑯𐑰𐑔 𐑦𐑑; 𐑯 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑢𐑦𐑤𐑦𐑙𐑤𐑦 𐑮𐑦𐑥𐑧𐑥𐑚𐑼, 𐑞𐑨𐑑 𐑞𐑦𐑕 𐑝𐑧𐑤𐑝𐑩𐑑 𐑐𐑷 𐑚𐑳𐑑 𐑒𐑩𐑯𐑕𐑰𐑤𐑟 𐑩 𐑮𐑦𐑥𐑹𐑕𐑤𐑩𐑕 𐑓𐑨𐑙.

𐑞𐑰𐑟 𐑸 𐑞 𐑑𐑲𐑥𐑟, 𐑢𐑧𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑢𐑱𐑤-𐑚𐑴𐑑 𐑞 𐑮𐑴𐑝𐑼 𐑕𐑪𐑓𐑑𐑤𐑦 𐑓𐑰𐑤𐑟 𐑩 𐑕𐑻𐑑𐑩𐑯 𐑓𐑦𐑤𐑾𐑤, 𐑒𐑪𐑯𐑓𐑦𐑛𐑩𐑯𐑑, 𐑤𐑨𐑯𐑛-𐑤𐑲𐑒 𐑓𐑰𐑤𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑕𐑰; 𐑞𐑨𐑑 𐑣𐑰 𐑮𐑦𐑜𐑸𐑛𐑟 𐑦𐑑 𐑨𐑟 𐑕𐑴 𐑥𐑳𐑗 𐑓𐑤𐑬𐑼𐑦 𐑻𐑔; 𐑯 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑑 𐑖𐑦𐑐 𐑮𐑦𐑝𐑰𐑤𐑦𐑙 𐑴𐑯𐑤𐑦 𐑞 𐑑𐑪𐑐𐑕 𐑝 𐑣𐑻 𐑥𐑭𐑕𐑑𐑕, 𐑕𐑰𐑥𐑟 𐑕𐑑𐑮𐑳𐑜𐑩𐑤𐑦𐑙 𐑓𐑹𐑢𐑼𐑛, 𐑯𐑪𐑑 𐑔𐑮𐑵 𐑣𐑲 𐑮𐑴𐑤𐑦𐑙 𐑢𐑱𐑝𐑟, 𐑚𐑳𐑑 𐑔𐑮𐑵 𐑞 𐑑𐑷𐑤 𐑜𐑮𐑭𐑕 𐑝 𐑩 𐑮𐑴𐑤𐑦𐑙 𐑐𐑮𐑺𐑦: 𐑨𐑟 𐑢𐑧𐑯 𐑞 𐑢𐑧𐑕𐑑𐑼𐑯 𐑧𐑥𐑦𐑜𐑮𐑩𐑯𐑑𐑕' 𐑣𐑹𐑕𐑩𐑟 𐑴𐑯𐑤𐑦 𐑖𐑴 𐑞𐑺 𐑦𐑮𐑧𐑒𐑑𐑩𐑛 𐑽𐑟, 𐑢𐑲𐑤 𐑞𐑺 𐑣𐑦𐑛𐑩𐑯 𐑚𐑪𐑛𐑦𐑟 𐑢𐑲𐑛𐑤𐑦 𐑢𐑱𐑛 𐑔𐑮𐑵 𐑞 𐑩𐑥𐑱𐑟𐑦𐑙 𐑝𐑻𐑡𐑼.

𐑞 𐑤𐑪𐑙-𐑛𐑮𐑷𐑯 𐑝𐑻𐑡𐑦𐑯 𐑝𐑱𐑤𐑟; 𐑞 𐑥𐑲𐑤𐑛 𐑚𐑤𐑵 𐑣𐑦𐑤-𐑕𐑲𐑛𐑟; 𐑨𐑟 𐑴𐑝𐑼 𐑞𐑰𐑟 𐑞𐑺 𐑕𐑑𐑰𐑤𐑟 𐑞 𐑣𐑳𐑖, 𐑞 𐑣𐑳𐑥; 𐑿 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑢𐑺 𐑞𐑨𐑑 𐑐𐑤𐑱-𐑢𐑽𐑦𐑛 𐑗𐑦𐑤𐑛𐑮𐑩𐑯 𐑤𐑲 𐑕𐑤𐑰𐑐𐑦𐑙 𐑦𐑯 𐑞𐑰𐑟 𐑕𐑪𐑤𐑦𐑑𐑿𐑛𐑟, 𐑦𐑯 𐑕𐑳𐑥 𐑜𐑤𐑨𐑛 𐑥𐑱-𐑑𐑲𐑥, 𐑢𐑧𐑯 𐑞 𐑓𐑤𐑬𐑼𐑟 𐑝 𐑞 𐑢𐑫𐑛𐑟 𐑸 𐑐𐑤𐑳𐑒𐑑. 𐑯 𐑷𐑤 𐑞𐑦𐑕 𐑥𐑦𐑒𐑕𐑩𐑟 𐑢𐑦𐑞 𐑘𐑹 𐑥𐑴𐑕𐑑 𐑥𐑦𐑕𐑑𐑦𐑒 𐑥𐑵𐑛; 𐑕𐑴 𐑞𐑨𐑑 𐑓𐑨𐑒𐑑 𐑯 𐑓𐑨𐑯𐑕𐑦, 𐑣𐑭𐑓-𐑢𐑱 𐑥𐑰𐑑𐑦𐑙, 𐑦𐑯𐑑𐑻𐑐𐑧𐑯𐑦𐑑𐑮𐑱𐑑, 𐑯 𐑓𐑹𐑥 𐑢𐑳𐑯 𐑕𐑰𐑥𐑤𐑩𐑕 𐑣𐑴𐑤.

𐑯𐑹 𐑛𐑦𐑛 𐑕𐑳𐑗 𐑕𐑵𐑞𐑦𐑙 𐑕𐑰𐑯𐑟, 𐑣𐑬𐑧𐑝𐑼 𐑑𐑧𐑥𐑐𐑼𐑼𐑦, 𐑓𐑱𐑤 𐑝 𐑨𐑑 𐑤𐑰𐑕𐑑 𐑨𐑟 𐑑𐑧𐑥𐑐𐑼𐑼𐑦 𐑩𐑯 𐑦𐑓𐑧𐑒𐑑 𐑪𐑯 ·𐑱𐑣𐑨𐑚. 𐑚𐑳𐑑 𐑦𐑓 𐑞𐑰𐑟 𐑕𐑰𐑒𐑮𐑩𐑑 𐑜𐑴𐑤𐑛𐑩𐑯 𐑒𐑰𐑟 𐑛𐑦𐑛 𐑕𐑰𐑥 𐑑 𐑴𐑐𐑩𐑯 𐑦𐑯 𐑣𐑦𐑥 𐑣𐑦𐑟 𐑴𐑯 𐑕𐑰𐑒𐑮𐑩𐑑 𐑜𐑴𐑤𐑛𐑩𐑯 𐑑𐑮𐑧𐑠𐑼𐑦𐑟, 𐑘𐑧𐑑 𐑛𐑦𐑛 𐑣𐑦𐑟 𐑚𐑮𐑧𐑔 𐑩𐑐𐑪𐑯 𐑞𐑧𐑥 𐑐𐑮𐑵𐑝 𐑚𐑳𐑑 𐑑𐑸𐑯𐑦𐑖𐑦𐑙.

𐑴, 𐑜𐑮𐑭𐑕𐑦 𐑜𐑤𐑱𐑛𐑟! 𐑴, 𐑧𐑝𐑼 𐑝𐑻𐑯𐑩𐑤 𐑧𐑯𐑛𐑤𐑩𐑕 𐑤𐑨𐑯𐑛𐑕𐑒𐑱𐑐𐑕 𐑦𐑯 𐑞 𐑕𐑴𐑤; 𐑦𐑯 𐑘𐑰,—𐑞𐑴 𐑤𐑪𐑙 𐑐𐑸𐑗𐑑 𐑚𐑲 𐑞 𐑛𐑧𐑛 𐑛𐑮𐑬𐑑 𐑝 𐑞 𐑻𐑔𐑦 𐑤𐑲𐑓,—𐑦𐑯 𐑘𐑰, 𐑥𐑧𐑯 𐑘𐑧𐑑 𐑥𐑱 𐑮𐑴𐑤, 𐑤𐑲𐑒 𐑘𐑳𐑙 𐑣𐑹𐑕𐑩𐑟 𐑦𐑯 𐑯𐑿 𐑥𐑹𐑯𐑦𐑙 𐑒𐑤𐑴𐑝𐑼; 𐑯 𐑓 𐑕𐑳𐑥 𐑓𐑿 𐑓𐑤𐑰𐑑𐑦𐑙 𐑥𐑴𐑥𐑧𐑯𐑑𐑕, 𐑓𐑰𐑤 𐑞 𐑒𐑵𐑤 𐑛𐑿 𐑝 𐑞 𐑤𐑲𐑓 𐑦𐑥𐑹𐑑𐑩𐑤 𐑪𐑯 𐑞𐑧𐑥. 𐑢𐑫𐑛 𐑑 ·𐑜𐑪𐑛 𐑞𐑰𐑟 𐑚𐑤𐑧𐑕𐑑 𐑒𐑭𐑥𐑟 𐑢𐑫𐑛 𐑤𐑭𐑕𐑑. 𐑚𐑳𐑑 𐑞 𐑥𐑦𐑙𐑜𐑩𐑤𐑛, 𐑥𐑦𐑙𐑜𐑩𐑤𐑦𐑙 𐑔𐑮𐑧𐑛𐑟 𐑝 𐑤𐑲𐑓 𐑸 𐑢𐑴𐑝𐑩𐑯 𐑚𐑲 𐑢𐑹𐑐 𐑯 𐑢𐑵𐑓: 𐑒𐑭𐑥𐑟 𐑒𐑮𐑪𐑕𐑑 𐑚𐑲 𐑕𐑑𐑹𐑥𐑟, 𐑩 𐑕𐑑𐑹𐑥 𐑓 𐑧𐑝𐑮𐑦 𐑒𐑭𐑥. 𐑞𐑺 𐑦𐑟 𐑯𐑴 𐑕𐑑𐑧𐑛𐑦 𐑳𐑯𐑮𐑰𐑑𐑮𐑱𐑕𐑦𐑙 𐑐𐑮𐑴𐑜𐑮𐑧𐑕 𐑦𐑯 𐑞𐑦𐑕 𐑤𐑲𐑓; 𐑢𐑰 𐑛𐑵 𐑯𐑪𐑑 𐑩𐑛𐑝𐑭𐑯𐑕 𐑔𐑮𐑵 𐑓𐑦𐑒𐑕𐑑 𐑜𐑮𐑱𐑛𐑱𐑖𐑩𐑯𐑟, 𐑯 𐑨𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑳𐑯 𐑐𐑷𐑟:—𐑔𐑮𐑵 𐑦𐑯𐑓𐑩𐑯𐑕𐑦'𐑟 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕 𐑕𐑐𐑧𐑤, 𐑚𐑶𐑣𐑫𐑛'𐑟 𐑔𐑷𐑑𐑤𐑩𐑕 𐑓𐑱𐑔, 𐑨𐑛𐑩𐑤𐑧𐑕𐑩𐑯𐑕' 𐑛𐑬𐑑 (𐑞 𐑒𐑪𐑥𐑩𐑯 𐑛𐑵𐑥), 𐑞𐑧𐑯 𐑕𐑒𐑧𐑐𐑑𐑦𐑕𐑦𐑟𐑩𐑥, 𐑞𐑧𐑯 𐑛𐑦𐑕𐑚𐑦𐑤𐑰𐑓, 𐑮𐑧𐑕𐑑𐑦𐑙 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑦𐑯 𐑥𐑨𐑯𐑣𐑫𐑛'𐑟 𐑐𐑪𐑯𐑛𐑼𐑦𐑙 𐑮𐑦𐑐𐑴𐑟 𐑝 𐑦𐑓. 𐑚𐑳𐑑 𐑢𐑳𐑯𐑕 𐑜𐑪𐑯 𐑔𐑮𐑵, 𐑢𐑰 𐑑𐑮𐑱𐑕 𐑞 𐑮𐑬𐑯𐑛 𐑩𐑜𐑱𐑯; 𐑯 𐑸 𐑦𐑯𐑓𐑩𐑯𐑑𐑕, 𐑚𐑶𐑟, 𐑯 𐑥𐑧𐑯, 𐑯 ·𐑦𐑓𐑕 𐑦𐑑𐑻𐑯𐑩𐑤𐑦. 𐑢𐑺 𐑤𐑲𐑟 𐑞 𐑓𐑲𐑯𐑩𐑤 𐑣𐑸𐑚𐑼, 𐑢𐑧𐑯𐑕 𐑢𐑰 𐑳𐑯𐑥𐑫𐑼 𐑯𐑴 𐑥𐑹? 𐑦𐑯 𐑢𐑪𐑑 𐑮𐑨𐑐𐑑 𐑰𐑔𐑼 𐑕𐑱𐑤𐑟 𐑞 𐑢𐑻𐑤𐑛, 𐑝 𐑢𐑦𐑗 𐑞 𐑢𐑽𐑾𐑕𐑑 𐑢𐑦𐑤 𐑯𐑧𐑝𐑼 𐑢𐑽𐑦? 𐑢𐑺 𐑦𐑟 𐑞 𐑓𐑬𐑯𐑛𐑤𐑦𐑙'𐑟 𐑓𐑭𐑞𐑼 𐑣𐑦𐑛𐑩𐑯? 𐑬𐑼 𐑕𐑴𐑤𐑟 𐑸 𐑤𐑲𐑒 𐑞𐑴𐑟 𐑹𐑓𐑩𐑯𐑟 𐑣𐑵𐑟 𐑳𐑯𐑢𐑧𐑛𐑩𐑛 𐑥𐑳𐑞𐑼𐑟 𐑛𐑲 𐑦𐑯 𐑚𐑺𐑦𐑙 𐑞𐑧𐑥: 𐑞 𐑕𐑰𐑒𐑮𐑩𐑑 𐑝 𐑬𐑼 𐑐𐑩𐑑𐑻𐑯𐑦𐑑𐑦 𐑤𐑲𐑟 𐑦𐑯 𐑞𐑺 𐑜𐑮𐑱𐑝, 𐑯 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑞𐑺 𐑑 𐑤𐑻𐑯 𐑦𐑑.

𐑯 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑛𐑱, 𐑑𐑵, 𐑜𐑱𐑟𐑦𐑙 𐑓𐑸 𐑛𐑬𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑕𐑲𐑛 𐑦𐑯𐑑𐑵 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑜𐑴𐑤𐑛𐑩𐑯 𐑕𐑰, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑤𐑴𐑤𐑦 𐑥𐑻𐑥𐑼𐑛:—

“𐑤𐑳𐑝𐑤𐑦𐑯𐑩𐑕 𐑳𐑯𐑓𐑨𐑞𐑩𐑥𐑩𐑚𐑩𐑤, 𐑨𐑟 𐑧𐑝𐑼 𐑤𐑳𐑝𐑼 𐑕𐑷 𐑦𐑯 𐑣𐑦𐑟 𐑘𐑳𐑙 𐑚𐑮𐑲𐑛'𐑟 𐑲!—𐑑𐑧𐑤 𐑥𐑰 𐑯𐑪𐑑 𐑝 𐑞𐑲 𐑑𐑰𐑔-𐑑𐑽𐑛 𐑖𐑸𐑒𐑕, 𐑯 𐑞𐑲 𐑒𐑦𐑛𐑯𐑨𐑐𐑦𐑙 𐑒𐑨𐑯𐑦𐑚𐑩𐑤 𐑢𐑱𐑟. 𐑤𐑧𐑑 𐑓𐑱𐑔 𐑬𐑕𐑑 𐑓𐑨𐑒𐑑; 𐑤𐑧𐑑 𐑓𐑨𐑯𐑕𐑦 𐑬𐑕𐑑 𐑥𐑧𐑥𐑼𐑦; 𐑲 𐑤𐑫𐑒 𐑛𐑰𐑐 𐑛𐑬𐑯 𐑯 𐑛𐑵 𐑚𐑦𐑤𐑰𐑝.”

𐑯 ·𐑕𐑑𐑳𐑚, 𐑓𐑦𐑖-𐑤𐑲𐑒, 𐑢𐑦𐑞 𐑕𐑐𐑸𐑒𐑩𐑤𐑦𐑙 𐑕𐑒𐑱𐑤𐑟, 𐑤𐑰𐑐𐑑 𐑳𐑐 𐑦𐑯 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑜𐑴𐑤𐑛𐑩𐑯 𐑤𐑲𐑑:—

“𐑲 𐑨𐑥 ·𐑕𐑑𐑳𐑚, 𐑯 ·𐑕𐑑𐑳𐑚 𐑣𐑨𐑟 𐑣𐑦𐑟 𐑣𐑦𐑕𐑑𐑼𐑦; 𐑚𐑳𐑑 𐑣𐑽 ·𐑕𐑑𐑳𐑚 𐑑𐑱𐑒𐑕 𐑴𐑞𐑟 𐑞𐑨𐑑 𐑣𐑰 𐑣𐑨𐑟 𐑷𐑤𐑢𐑱𐑟 𐑚𐑰𐑯 𐑡𐑪𐑤𐑦!”
