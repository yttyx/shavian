·𐑗𐑨𐑐𐑑𐑼 59. 𐑕𐑒𐑢𐑦𐑛.

𐑕𐑤𐑴𐑤𐑦 𐑢𐑱𐑛𐑦𐑙 𐑔𐑮𐑵 𐑞 𐑥𐑧𐑛𐑴𐑟 𐑝 ·𐑚𐑮𐑦𐑑, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑕𐑑𐑦𐑤 𐑣𐑧𐑤𐑛 𐑪𐑯 𐑣𐑻 𐑢𐑱 𐑯𐑹𐑔-𐑰𐑕𐑑𐑢𐑼𐑛 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑲𐑤𐑩𐑯𐑛 𐑝 ·𐑡𐑭𐑝𐑩; 𐑩 𐑡𐑧𐑯𐑑𐑩𐑤 𐑺 𐑦𐑥𐑐𐑧𐑤𐑦𐑙 𐑣𐑻 𐑒𐑰𐑤, 𐑕𐑴 𐑞𐑨𐑑 𐑦𐑯 𐑞 𐑕𐑼𐑬𐑯𐑛𐑦𐑙 𐑕𐑼𐑧𐑯𐑦𐑑𐑦 𐑣𐑻 𐑔𐑮𐑰 𐑑𐑷𐑤 𐑑𐑱𐑐𐑼𐑦𐑙 𐑥𐑭𐑕𐑑𐑕 𐑥𐑲𐑤𐑛𐑤𐑦 𐑢𐑱𐑝𐑛 𐑑 𐑞𐑨𐑑 𐑤𐑨𐑙𐑜𐑢𐑦𐑛 𐑚𐑮𐑰𐑟, 𐑨𐑟 𐑔𐑮𐑰 𐑥𐑲𐑤𐑛 𐑐𐑭𐑥𐑟 𐑪𐑯 𐑩 𐑐𐑤𐑱𐑯. 𐑯 𐑕𐑑𐑦𐑤, 𐑨𐑑 𐑢𐑲𐑛 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑦𐑯 𐑞 𐑕𐑦𐑤𐑝𐑼𐑦 𐑯𐑲𐑑, 𐑞 𐑤𐑴𐑯𐑤𐑦, 𐑩𐑤𐑫𐑼𐑦𐑙 𐑡𐑧𐑑 𐑢𐑫𐑛 𐑚𐑰 𐑕𐑰𐑯.

𐑚𐑳𐑑 𐑢𐑳𐑯 𐑑𐑮𐑨𐑯𐑕𐑐𐑨𐑮𐑩𐑯𐑑 𐑚𐑤𐑵 𐑥𐑹𐑯𐑦𐑙, 𐑢𐑧𐑯 𐑩 𐑕𐑑𐑦𐑤𐑯𐑩𐑕 𐑷𐑤𐑥𐑴𐑕𐑑 𐑐𐑮𐑰𐑑𐑼𐑯𐑨𐑗𐑼𐑩𐑤 𐑕𐑐𐑮𐑧𐑛 𐑴𐑝𐑼 𐑞 𐑕𐑰, 𐑣𐑬𐑧𐑝𐑼 𐑳𐑯𐑩𐑑𐑧𐑯𐑛𐑩𐑛 𐑢𐑦𐑞 𐑧𐑯𐑦 𐑕𐑑𐑨𐑜𐑯𐑩𐑯𐑑 𐑒𐑭𐑥; 𐑢𐑧𐑯 𐑞 𐑤𐑪𐑙 𐑚𐑻𐑯𐑦𐑖𐑑 𐑕𐑳𐑯-𐑜𐑤𐑱𐑛 𐑪𐑯 𐑞 𐑢𐑷𐑑𐑼𐑟 𐑕𐑰𐑥𐑛 𐑩 𐑜𐑴𐑤𐑛𐑩𐑯 𐑓𐑦𐑙𐑜𐑼 𐑤𐑱𐑛 𐑩𐑒𐑮𐑪𐑕 𐑞𐑧𐑥, 𐑦𐑯𐑡𐑶𐑯𐑦𐑙 𐑕𐑳𐑥 𐑕𐑰𐑒𐑮𐑩𐑕𐑦; 𐑢𐑧𐑯 𐑞 𐑕𐑤𐑦𐑐𐑼𐑛 𐑢𐑱𐑝𐑟 𐑢𐑦𐑕𐑐𐑼𐑛 𐑑𐑩𐑜𐑧𐑞𐑼 𐑨𐑟 𐑞𐑱 𐑕𐑪𐑓𐑑𐑤𐑦 𐑮𐑨𐑯 𐑪𐑯; 𐑦𐑯 𐑞𐑦𐑕 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑣𐑳𐑖 𐑝 𐑞 𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑕𐑓𐑽 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑐𐑧𐑒𐑑𐑼 𐑢𐑪𐑟 𐑕𐑰𐑯 𐑚𐑲 ·𐑛𐑨𐑜𐑵 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑-𐑣𐑧𐑛.

𐑦𐑯 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑕, 𐑩 𐑜𐑮𐑱𐑑 𐑢𐑲𐑑 𐑥𐑨𐑕 𐑤𐑱𐑟𐑦𐑤𐑦 𐑮𐑴𐑟, 𐑯 𐑮𐑲𐑟𐑦𐑙 𐑣𐑲𐑼 𐑯 𐑣𐑲𐑼, 𐑯 𐑛𐑦𐑕𐑦𐑯𐑑𐑨𐑙𐑜𐑩𐑤𐑦𐑙 𐑦𐑑𐑕𐑧𐑤𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑨𐑠𐑼, 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑜𐑤𐑰𐑥𐑛 𐑚𐑦𐑓𐑹 𐑬𐑼 𐑐𐑮𐑬 𐑤𐑲𐑒 𐑩 𐑕𐑯𐑴-𐑕𐑤𐑲𐑛, 𐑯𐑿 𐑕𐑤𐑦𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑣𐑦𐑤𐑟. 𐑞𐑳𐑕 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑨𐑟 𐑕𐑤𐑴𐑤𐑦 𐑦𐑑 𐑕𐑩𐑚𐑕𐑲𐑛𐑩𐑛, 𐑯 𐑕𐑨𐑙𐑒. 𐑞𐑧𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑼𐑴𐑟, 𐑯 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑜𐑤𐑰𐑥𐑛. 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑯𐑪𐑑 𐑩 𐑢𐑱𐑤; 𐑯 𐑘𐑧𐑑 𐑦𐑟 𐑞𐑦𐑕 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒? 𐑔𐑷𐑑 ·𐑛𐑨𐑜𐑵. 𐑩𐑜𐑱𐑯 𐑞 𐑓𐑨𐑯𐑑𐑩𐑥 𐑢𐑧𐑯𐑑 𐑛𐑬𐑯, 𐑚𐑳𐑑 𐑪𐑯 𐑮𐑰-𐑩𐑐𐑽𐑦𐑙 𐑢𐑳𐑯𐑕 𐑥𐑹, 𐑢𐑦𐑞 𐑩 𐑕𐑑𐑦𐑤𐑧𐑑𐑴-𐑤𐑲𐑒 𐑒𐑮𐑲 𐑞𐑨𐑑 𐑕𐑑𐑸𐑑𐑩𐑤𐑛 𐑧𐑝𐑮𐑦 𐑥𐑨𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑯𐑪𐑛, 𐑞 𐑯𐑰𐑜𐑮𐑴 𐑘𐑧𐑤𐑛 𐑬𐑑—“𐑞𐑺! 𐑞𐑺 𐑩𐑜𐑱𐑯! 𐑞𐑺 𐑖𐑰 𐑚𐑮𐑰𐑗𐑩𐑟! 𐑮𐑲𐑑 𐑩𐑣𐑧𐑛! 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤, 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤!”

𐑩𐑐𐑪𐑯 𐑞𐑦𐑕, 𐑞 𐑕𐑰𐑥𐑧𐑯 𐑮𐑳𐑖𐑑 𐑑 𐑞 𐑘𐑸𐑛-𐑸𐑥𐑟, 𐑨𐑟 𐑦𐑯 𐑕𐑢𐑹𐑥𐑦𐑙-𐑑𐑲𐑥 𐑞 𐑚𐑰𐑟 𐑮𐑳𐑖 𐑑 𐑞 𐑚𐑬𐑟. 𐑚𐑺-𐑣𐑧𐑛𐑩𐑛 𐑦𐑯 𐑞 𐑕𐑳𐑤𐑑𐑮𐑦 𐑕𐑳𐑯, ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑫𐑛 𐑪𐑯 𐑞 𐑚𐑴𐑕𐑐𐑮𐑦𐑑, 𐑯 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑐𐑫𐑖𐑑 𐑓𐑸 𐑚𐑦𐑣𐑲𐑯𐑛 𐑦𐑯 𐑮𐑧𐑛𐑦𐑯𐑩𐑕 𐑑 𐑢𐑱𐑝 𐑣𐑦𐑟 𐑹𐑛𐑼𐑟 𐑑 𐑞 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯, 𐑒𐑭𐑕𐑑 𐑣𐑦𐑟 𐑰𐑜𐑼 𐑜𐑤𐑭𐑯𐑕 𐑦𐑯 𐑞 𐑛𐑦𐑮𐑧𐑒𐑖𐑩𐑯 𐑦𐑯𐑛𐑦𐑒𐑱𐑑𐑩𐑛 𐑩𐑤𐑪𐑓𐑑 𐑚𐑲 𐑞 𐑬𐑑𐑕𐑑𐑮𐑧𐑗𐑑 𐑥𐑴𐑖𐑩𐑯𐑤𐑩𐑕 𐑸𐑥 𐑝 ·𐑛𐑨𐑜𐑵.

𐑢𐑧𐑞𐑼 𐑞 𐑓𐑤𐑦𐑑𐑦𐑙 𐑩𐑑𐑧𐑯𐑛𐑩𐑯𐑕 𐑝 𐑞 𐑢𐑳𐑯 𐑕𐑑𐑦𐑤 𐑯 𐑕𐑪𐑤𐑦𐑑𐑼𐑦 𐑡𐑧𐑑 𐑣𐑨𐑛 𐑜𐑮𐑨𐑡𐑵𐑩𐑤𐑦 𐑢𐑻𐑒𐑑 𐑩𐑐𐑪𐑯 ·𐑱𐑣𐑨𐑚, 𐑕𐑴 𐑞𐑨𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑯𐑬 𐑐𐑮𐑦𐑐𐑺𐑛 𐑑 𐑒𐑩𐑯𐑧𐑒𐑑 𐑞 𐑲𐑛𐑾𐑟 𐑝 𐑥𐑲𐑤𐑛𐑯𐑩𐑕 𐑯 𐑮𐑦𐑐𐑴𐑟 𐑢𐑦𐑞 𐑞 𐑓𐑻𐑕𐑑 𐑕𐑲𐑑 𐑝 𐑞 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑢𐑱𐑤 𐑣𐑰 𐑐𐑼𐑕𐑿𐑛; 𐑣𐑬𐑧𐑝𐑼 𐑞𐑦𐑕 𐑢𐑪𐑟, 𐑹 𐑢𐑧𐑞𐑼 𐑣𐑦𐑟 𐑰𐑜𐑼𐑯𐑩𐑕 𐑚𐑦𐑑𐑮𐑱𐑛 𐑣𐑦𐑥; 𐑢𐑦𐑗𐑧𐑝𐑼 𐑢𐑱 𐑦𐑑 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯, 𐑯𐑴 𐑕𐑵𐑯𐑼 𐑛𐑦𐑛 𐑣𐑰 𐑛𐑦𐑕𐑑𐑦𐑙𐑒𐑑𐑤𐑦 𐑐𐑼𐑕𐑰𐑝 𐑞 𐑢𐑲𐑑 𐑥𐑨𐑕, 𐑞𐑨𐑯 𐑢𐑦𐑞 𐑩 𐑒𐑢𐑦𐑒 𐑦𐑯𐑑𐑧𐑯𐑕𐑦𐑑𐑦 𐑣𐑰 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑜𐑱𐑝 𐑹𐑛𐑼𐑟 𐑓 𐑤𐑴𐑼𐑦𐑙.

𐑞 𐑓𐑹 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑕𐑵𐑯 𐑪𐑯 𐑞 𐑢𐑷𐑑𐑼; ·𐑱𐑣𐑨𐑚'𐑟 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕, 𐑯 𐑷𐑤 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑐𐑫𐑤𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞𐑺 𐑐𐑮𐑱. 𐑕𐑵𐑯 𐑦𐑑 𐑢𐑧𐑯𐑑 𐑛𐑬𐑯, 𐑯 𐑢𐑲𐑤, 𐑢𐑦𐑞 𐑹𐑟 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛, 𐑢𐑰 𐑢𐑻 𐑩𐑢𐑱𐑑𐑦𐑙 𐑦𐑑𐑕 𐑮𐑰𐑩𐑐𐑽𐑩𐑯𐑕, 𐑤𐑴! 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑕𐑐𐑪𐑑 𐑢𐑺 𐑦𐑑 𐑕𐑨𐑙𐑒, 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑦𐑑 𐑕𐑤𐑴𐑤𐑦 𐑮𐑴𐑟. 𐑷𐑤𐑥𐑴𐑕𐑑 𐑓𐑼𐑜𐑧𐑑𐑦𐑙 𐑓 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑 𐑷𐑤 𐑔𐑷𐑑𐑕 𐑝 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒, 𐑢𐑰 𐑯𐑬 𐑜𐑱𐑟𐑛 𐑨𐑑 𐑞 𐑥𐑴𐑕𐑑 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑓𐑦𐑯𐑪𐑥𐑦𐑯𐑪𐑯 𐑢𐑦𐑗 𐑞 𐑕𐑰𐑒𐑮𐑩𐑑 𐑕𐑰𐑟 𐑣𐑨𐑝 𐑣𐑦𐑞𐑼𐑑𐑵 𐑮𐑦𐑝𐑰𐑤𐑛 𐑑 𐑥𐑨𐑯𐑒𐑲𐑯𐑛. 𐑩 𐑝𐑭𐑕𐑑 𐑐𐑳𐑤𐑐𐑦 𐑥𐑨𐑕, 𐑓𐑻𐑤𐑪𐑙𐑟 𐑦𐑯 𐑤𐑧𐑙𐑔 𐑯 𐑚𐑮𐑧𐑛𐑔, 𐑝 𐑩 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑒𐑮𐑰𐑥-𐑒𐑳𐑤𐑼, 𐑤𐑱 𐑓𐑤𐑴𐑑𐑦𐑙 𐑪𐑯 𐑞 𐑢𐑷𐑑𐑼, 𐑦𐑯𐑿𐑥𐑼𐑩𐑚𐑩𐑤 𐑤𐑪𐑙 𐑸𐑥𐑟 𐑮𐑱𐑛𐑦𐑱𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑦𐑑𐑕 𐑕𐑧𐑯𐑑𐑼, 𐑯 𐑒𐑻𐑤𐑦𐑙 𐑯 𐑑𐑢𐑦𐑕𐑑𐑦𐑙 𐑤𐑲𐑒 𐑩 𐑯𐑧𐑕𐑑 𐑝 𐑨𐑯𐑩𐑒𐑪𐑯𐑛𐑩𐑟, 𐑨𐑟 𐑦𐑓 𐑚𐑤𐑲𐑯𐑛𐑤𐑦 𐑑 𐑒𐑤𐑳𐑗 𐑨𐑑 𐑧𐑯𐑦 𐑣𐑨𐑐𐑤𐑩𐑕 𐑪𐑚𐑡𐑧𐑒𐑑 𐑢𐑦𐑞𐑦𐑯 𐑮𐑰𐑗. 𐑯𐑴 𐑐𐑼𐑕𐑧𐑐𐑑𐑩𐑚𐑩𐑤 𐑓𐑱𐑕 𐑹 𐑓𐑮𐑳𐑯𐑑 𐑛𐑦𐑛 𐑦𐑑 𐑣𐑨𐑝; 𐑯𐑴 𐑒𐑩𐑯𐑕𐑰𐑝𐑩𐑚𐑩𐑤 𐑑𐑴𐑒𐑩𐑯 𐑝 𐑲𐑞𐑼 𐑕𐑧𐑯𐑕𐑱𐑖𐑩𐑯 𐑹 𐑦𐑯𐑕𐑑𐑦𐑙𐑒𐑑; 𐑚𐑳𐑑 𐑳𐑯𐑡𐑩𐑤𐑱𐑑𐑩𐑛 𐑞𐑺 𐑪𐑯 𐑞 𐑚𐑦𐑤𐑴𐑟, 𐑩𐑯 𐑳𐑯𐑻𐑔𐑤𐑦, 𐑓𐑹𐑥𐑤𐑩𐑕, 𐑗𐑭𐑯𐑕-𐑤𐑲𐑒 𐑨𐑐𐑼𐑦𐑖𐑩𐑯 𐑝 𐑤𐑲𐑓.

𐑨𐑟 𐑢𐑦𐑞 𐑩 𐑤𐑴 𐑕𐑳𐑒𐑦𐑙 𐑕𐑬𐑯𐑛 𐑦𐑑 𐑕𐑤𐑴𐑤𐑦 𐑛𐑦𐑕𐑩𐑐𐑽𐑛 𐑩𐑜𐑱𐑯, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑕𐑑𐑦𐑤 𐑜𐑱𐑟𐑦𐑙 𐑨𐑑 𐑞 𐑨𐑡𐑦𐑑𐑱𐑑𐑩𐑛 𐑢𐑷𐑑𐑼𐑟 𐑢𐑺 𐑦𐑑 𐑣𐑨𐑛 𐑕𐑳𐑙𐑒, 𐑢𐑦𐑞 𐑩 𐑢𐑲𐑤𐑛 𐑝𐑶𐑕 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛—“𐑷𐑤𐑥𐑴𐑕𐑑 𐑮𐑭𐑞𐑼 𐑣𐑨𐑛 𐑲 𐑕𐑰𐑯 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑯 𐑓𐑷𐑑 𐑣𐑦𐑥, 𐑞𐑨𐑯 𐑑 𐑣𐑨𐑝 𐑕𐑰𐑯 𐑞𐑰, 𐑞𐑬 𐑢𐑲𐑑 𐑜𐑴𐑕𐑑!”

“𐑢𐑪𐑑 𐑢𐑪𐑟 𐑦𐑑, ·𐑕𐑻?” 𐑕𐑧𐑛 ·𐑓𐑤𐑭𐑕𐑒.

“𐑞 𐑜𐑮𐑱𐑑 𐑤𐑲𐑝 𐑕𐑒𐑢𐑦𐑛, 𐑢𐑦𐑗, 𐑞𐑱 𐑕𐑱, 𐑓𐑿 𐑢𐑱𐑤-𐑖𐑦𐑐𐑕 𐑧𐑝𐑼 𐑚𐑦𐑣𐑧𐑤𐑛, 𐑯 𐑮𐑩𐑑𐑻𐑯𐑛 𐑑 𐑞𐑺 𐑐𐑹𐑑𐑕 𐑑 𐑑𐑧𐑤 𐑝 𐑦𐑑.”

𐑚𐑳𐑑 ·𐑱𐑣𐑨𐑚 𐑕𐑧𐑛 𐑯𐑳𐑔𐑦𐑙; 𐑑𐑻𐑯𐑦𐑙 𐑣𐑦𐑟 𐑚𐑴𐑑, 𐑣𐑰 𐑕𐑱𐑤𐑛 𐑚𐑨𐑒 𐑑 𐑞 𐑝𐑧𐑕𐑩𐑤; 𐑞 𐑮𐑧𐑕𐑑 𐑨𐑟 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑓𐑪𐑤𐑴𐑦𐑙.

𐑢𐑪𐑑𐑧𐑝𐑼 𐑕𐑵𐑐𐑼𐑕𐑑𐑦𐑖𐑩𐑯𐑟 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤𐑥𐑧𐑯 𐑦𐑯 𐑡𐑧𐑯𐑼𐑩𐑤 𐑣𐑨𐑝 𐑒𐑩𐑯𐑧𐑒𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞 𐑕𐑲𐑑 𐑝 𐑞𐑦𐑕 𐑪𐑚𐑡𐑧𐑒𐑑, 𐑕𐑻𐑑𐑩𐑯 𐑦𐑑 𐑦𐑟, 𐑞𐑨𐑑 𐑩 𐑜𐑤𐑦𐑥𐑐𐑕 𐑝 𐑦𐑑 𐑚𐑰𐑦𐑙 𐑕𐑴 𐑝𐑧𐑮𐑦 𐑳𐑯𐑿𐑠𐑵𐑩𐑤, 𐑞𐑨𐑑 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕 𐑣𐑨𐑟 𐑜𐑪𐑯 𐑓𐑸 𐑑 𐑦𐑯𐑝𐑧𐑕𐑑 𐑦𐑑 𐑢𐑦𐑞 𐑐𐑹𐑑𐑧𐑯𐑑𐑩𐑕𐑯𐑩𐑕. 𐑕𐑴 𐑮𐑺𐑤𐑦 𐑦𐑟 𐑦𐑑 𐑚𐑦𐑣𐑧𐑤𐑛, 𐑞𐑨𐑑 𐑞𐑴 𐑢𐑳𐑯 𐑯 𐑷𐑤 𐑝 𐑞𐑧𐑥 𐑛𐑦𐑒𐑤𐑺 𐑦𐑑 𐑑 𐑚𐑰 𐑞 𐑤𐑸𐑡𐑩𐑕𐑑 𐑨𐑯𐑦𐑥𐑱𐑑𐑩𐑛 𐑔𐑦𐑙 𐑦𐑯 𐑞 𐑴𐑖𐑩𐑯, 𐑘𐑧𐑑 𐑝𐑧𐑮𐑦 𐑓𐑿 𐑝 𐑞𐑧𐑥 𐑣𐑨𐑝 𐑧𐑯𐑦 𐑚𐑳𐑑 𐑞 𐑥𐑴𐑕𐑑 𐑝𐑱𐑜 𐑲𐑛𐑾𐑟 𐑒𐑩𐑯𐑕𐑻𐑯𐑦𐑙 𐑦𐑑𐑕 𐑑𐑮𐑵 𐑯𐑱𐑗𐑼 𐑯 𐑓𐑹𐑥; 𐑯𐑪𐑑𐑢𐑦𐑞𐑕𐑑𐑨𐑯𐑛𐑦𐑙, 𐑞𐑱 𐑚𐑦𐑤𐑰𐑝 𐑦𐑑 𐑑 𐑓𐑻𐑯𐑦𐑖 𐑑 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤 𐑣𐑦𐑟 𐑴𐑯𐑤𐑦 𐑓𐑵𐑛. 𐑓 𐑞𐑴 𐑳𐑞𐑼 𐑕𐑐𐑰𐑖𐑰𐑟 𐑝 𐑢𐑱𐑤𐑟 𐑓𐑲𐑯𐑛 𐑞𐑺 𐑓𐑵𐑛 𐑩𐑚𐑳𐑝 𐑢𐑷𐑑𐑼, 𐑯 𐑥𐑱 𐑚𐑰 𐑕𐑰𐑯 𐑚𐑲 𐑥𐑨𐑯 𐑦𐑯 𐑞 𐑨𐑒𐑑 𐑝 𐑓𐑰𐑛𐑦𐑙, 𐑞 𐑕𐑐𐑻𐑥𐑩𐑕𐑧𐑑𐑦 𐑢𐑱𐑤 𐑩𐑚𐑑𐑱𐑯𐑟 𐑣𐑦𐑟 𐑣𐑴𐑤 𐑓𐑵𐑛 𐑦𐑯 𐑳𐑯𐑯𐑴𐑯 𐑟𐑴𐑯𐑟 𐑚𐑦𐑤𐑴 𐑞 𐑕𐑻𐑓𐑦𐑕; 𐑯 𐑴𐑯𐑤𐑦 𐑚𐑲 𐑦𐑯𐑓𐑼𐑩𐑯𐑕 𐑦𐑟 𐑦𐑑 𐑞𐑨𐑑 𐑧𐑯𐑦 𐑢𐑳𐑯 𐑒𐑨𐑯 𐑑𐑧𐑤 𐑝 𐑢𐑪𐑑, 𐑐𐑮𐑦𐑕𐑲𐑕𐑤𐑦, 𐑞𐑨𐑑 𐑓𐑵𐑛 𐑒𐑩𐑯𐑕𐑦𐑕𐑑𐑕. 𐑨𐑑 𐑑𐑲𐑥𐑟, 𐑢𐑧𐑯 𐑒𐑤𐑴𐑕𐑤𐑦 𐑐𐑼𐑕𐑿𐑛, 𐑣𐑰 𐑢𐑦𐑤 𐑛𐑦𐑕𐑜𐑹𐑡 𐑢𐑪𐑑 𐑸 𐑕𐑩𐑐𐑴𐑕𐑑 𐑑 𐑚𐑰 𐑞 𐑛𐑦𐑑𐑨𐑗𐑑 𐑸𐑥𐑟 𐑝 𐑞 𐑕𐑒𐑢𐑦𐑛; 𐑕𐑳𐑥 𐑝 𐑞𐑧𐑥 𐑞𐑳𐑕 𐑦𐑜𐑟𐑦𐑚𐑦𐑑𐑩𐑛 𐑦𐑒𐑕𐑰𐑛𐑦𐑙 𐑑𐑢𐑧𐑯𐑑𐑦 𐑯 𐑔𐑻𐑑𐑦 𐑓𐑰𐑑 𐑦𐑯 𐑤𐑧𐑙𐑔. 𐑞𐑱 𐑓𐑨𐑯𐑕𐑦 𐑞𐑨𐑑 𐑞 𐑥𐑪𐑯𐑕𐑑𐑼 𐑑 𐑢𐑦𐑗 𐑞𐑰𐑟 𐑸𐑥𐑟 𐑚𐑦𐑤𐑪𐑙𐑛 𐑹𐑛𐑦𐑯𐑺𐑦𐑤𐑦 𐑒𐑤𐑦𐑙𐑟 𐑚𐑲 𐑞𐑧𐑥 𐑑 𐑞 𐑚𐑧𐑛 𐑝 𐑞 𐑴𐑖𐑩𐑯; 𐑯 𐑞𐑨𐑑 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤, 𐑳𐑯𐑤𐑲𐑒 𐑳𐑞𐑼 𐑕𐑐𐑰𐑖𐑰𐑟, 𐑦𐑟 𐑕𐑩𐑐𐑤𐑲𐑛 𐑢𐑦𐑞 𐑑𐑰𐑔 𐑦𐑯 𐑹𐑛𐑼 𐑑 𐑩𐑑𐑨𐑒 𐑯 𐑑𐑺 𐑦𐑑.

𐑞𐑺 𐑕𐑰𐑥𐑟 𐑕𐑳𐑥 𐑜𐑮𐑬𐑯𐑛 𐑑 𐑦𐑥𐑨𐑡𐑦𐑯 𐑞𐑨𐑑 𐑞 𐑜𐑮𐑱𐑑 ·𐑒𐑮𐑭𐑒𐑧𐑯 𐑝 ·𐑚𐑦𐑖𐑩𐑐 ·𐑐𐑪𐑯𐑑𐑪𐑐𐑐𐑪𐑛𐑩𐑯 𐑥𐑱 𐑳𐑤𐑑𐑦𐑥𐑩𐑑𐑤𐑦 𐑮𐑦𐑟𐑪𐑤𐑝 𐑦𐑑𐑕𐑧𐑤𐑓 𐑦𐑯𐑑𐑵 ·𐑕𐑒𐑢𐑦𐑛. 𐑞 𐑥𐑨𐑯𐑼 𐑦𐑯 𐑢𐑦𐑗 𐑞 ·𐑚𐑦𐑖𐑩𐑐 𐑛𐑦𐑕𐑒𐑮𐑲𐑚𐑟 𐑦𐑑, 𐑨𐑟 𐑷𐑤𐑑𐑻𐑯𐑩𐑑𐑤𐑦 𐑮𐑲𐑟𐑦𐑙 𐑯 𐑕𐑦𐑙𐑒𐑦𐑙, 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼𐑟 𐑣𐑰 𐑯𐑼𐑱𐑑𐑕, 𐑦𐑯 𐑷𐑤 𐑞𐑦𐑕 𐑞 𐑑𐑵 𐑒𐑪𐑮𐑦𐑕𐑐𐑪𐑯𐑛. 𐑚𐑳𐑑 𐑥𐑳𐑗 𐑩𐑚𐑱𐑑𐑥𐑩𐑯𐑑 𐑦𐑟 𐑯𐑧𐑕𐑦𐑕𐑺𐑦 𐑢𐑦𐑞 𐑮𐑦𐑕𐑐𐑧𐑒𐑑 𐑑 𐑞 𐑦𐑯𐑒𐑮𐑧𐑛𐑩𐑚𐑩𐑤 𐑚𐑳𐑤𐑒 𐑣𐑰 𐑩𐑕𐑲𐑯𐑟 𐑦𐑑.

𐑚𐑲 𐑕𐑳𐑥 𐑯𐑨𐑗𐑼𐑩𐑤𐑦𐑕𐑑𐑕 𐑣𐑵 𐑣𐑨𐑝 𐑝𐑱𐑜𐑤𐑦 𐑣𐑻𐑛 𐑮𐑵𐑥𐑼𐑟 𐑝 𐑞 𐑥𐑦𐑕𐑑𐑽𐑾𐑕 𐑒𐑮𐑰𐑗𐑼, 𐑣𐑽 𐑕𐑐𐑴𐑒𐑩𐑯 𐑝, 𐑦𐑑 𐑦𐑟 𐑦𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑩𐑥𐑳𐑙 𐑞 𐑒𐑤𐑭𐑕 𐑝 𐑒𐑳𐑑𐑩𐑤-𐑓𐑦𐑖, 𐑑 𐑢𐑦𐑗, 𐑦𐑯𐑛𐑰𐑛, 𐑦𐑯 𐑕𐑻𐑑𐑩𐑯 𐑧𐑒𐑕𐑑𐑼𐑯𐑩𐑤 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑕 𐑦𐑑 𐑢𐑫𐑛 𐑕𐑰𐑥 𐑑 𐑚𐑦𐑤𐑪𐑙, 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑨𐑟 𐑞 ·𐑱𐑯𐑨𐑒 𐑝 𐑞 𐑑𐑮𐑲𐑚.
