𐑚𐑳𐑑 𐑣𐑬 𐑯𐑬, ·𐑦𐑖𐑥𐑱𐑩𐑤? 𐑣𐑬 𐑦𐑟 𐑦𐑑, 𐑞𐑨𐑑 𐑿, 𐑩 𐑥𐑽 𐑹𐑟𐑥𐑩𐑯 𐑦𐑯 𐑞 𐑓𐑦𐑖𐑼𐑦, 𐑐𐑮𐑦𐑑𐑧𐑯𐑛 𐑑 𐑯𐑴 𐑷𐑑 𐑩𐑚𐑬𐑑 𐑞 𐑕𐑳𐑚𐑑𐑼𐑱𐑯𐑾𐑯 𐑐𐑸𐑑𐑕 𐑝 𐑞 𐑢𐑱𐑤? 𐑛𐑦𐑛 𐑧𐑮𐑫𐑛𐑲𐑑 ·𐑕𐑑𐑳𐑚, 𐑥𐑬𐑯𐑑𐑩𐑛 𐑩𐑐𐑪𐑯 𐑘𐑹 𐑒𐑨𐑐𐑕𐑑𐑩𐑯, 𐑛𐑦𐑤𐑦𐑝𐑼 𐑤𐑧𐑒𐑗𐑼𐑟 𐑪𐑯 𐑞 𐑩𐑯𐑨𐑑𐑩𐑥𐑦 𐑝 𐑞 ·𐑕𐑧𐑑𐑱𐑖𐑩; 𐑯 𐑚𐑲 𐑣𐑧𐑤𐑐 𐑝 𐑞 𐑢𐑦𐑯𐑛𐑤𐑩𐑕, 𐑣𐑴𐑤𐑛 𐑳𐑐 𐑩 𐑕𐑐𐑧𐑕𐑩𐑥𐑩𐑯 𐑮𐑦𐑚 𐑓 𐑧𐑒𐑕𐑦𐑚𐑦𐑖𐑩𐑯? 𐑦𐑒𐑕𐑐𐑤𐑱𐑯 𐑞𐑲𐑕𐑧𐑤𐑓, ·𐑦𐑖𐑥𐑱𐑩𐑤. 𐑒𐑨𐑯 𐑿 𐑤𐑨𐑯𐑛 𐑩 𐑓𐑫𐑤-𐑜𐑮𐑴𐑯 𐑢𐑱𐑤 𐑪𐑯 𐑘𐑹 𐑛𐑧𐑒 𐑓 𐑦𐑜𐑟𐑨𐑥𐑦𐑯𐑱𐑖𐑩𐑯, 𐑨𐑟 𐑩 𐑒𐑫𐑒 𐑛𐑦𐑖𐑩𐑟 𐑩 𐑮𐑴𐑕𐑑-𐑐𐑦𐑜? 𐑖𐑫𐑼𐑤𐑦 𐑯𐑪𐑑. 𐑩 𐑝𐑧𐑮𐑦𐑑𐑩𐑚𐑩𐑤 𐑢𐑦𐑑𐑯𐑩𐑕 𐑣𐑨𐑝 𐑿 𐑣𐑦𐑞𐑼𐑑𐑵 𐑚𐑰𐑯, ·𐑦𐑖𐑥𐑱𐑩𐑤; 𐑚𐑳𐑑 𐑣𐑨𐑝 𐑩 𐑒𐑺 𐑣𐑬 𐑿 𐑕𐑰𐑟 𐑞 𐑐𐑮𐑦𐑝𐑩𐑤𐑦𐑡 𐑝 ·𐑡𐑴𐑯𐑩 𐑩𐑤𐑴𐑯; 𐑞 𐑐𐑮𐑦𐑝𐑩𐑤𐑦𐑡 𐑝 𐑛𐑦𐑕𐑒𐑹𐑕𐑦𐑙 𐑩𐑐𐑪𐑯 𐑞 𐑡𐑶𐑕𐑑𐑕 𐑯 𐑚𐑰𐑥𐑟; 𐑞 𐑮𐑭𐑓𐑑𐑼𐑟, 𐑮𐑦𐑡-𐑐𐑴𐑤, 𐑕𐑤𐑰𐑐𐑼𐑟, 𐑯 𐑳𐑯𐑛𐑼-𐑐𐑦𐑯𐑦𐑙𐑟, 𐑥𐑱𐑒𐑦𐑙 𐑳𐑐 𐑞 𐑓𐑮𐑱𐑥-𐑢𐑻𐑒 𐑝 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯; 𐑯 𐑚𐑦𐑤𐑲𐑒 𐑝 𐑞 𐑑𐑨𐑤𐑴-𐑝𐑨𐑑𐑕, 𐑛𐑺𐑦-𐑮𐑵𐑥𐑟, 𐑚𐑳𐑑𐑼𐑦𐑟, 𐑯 𐑗𐑰𐑟𐑼𐑦𐑟 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑬𐑩𐑤𐑟.

𐑲 𐑒𐑩𐑯𐑓𐑧𐑕, 𐑞𐑨𐑑 𐑕𐑦𐑯𐑕 ·𐑡𐑴𐑯𐑩, 𐑓𐑿 𐑢𐑱𐑤𐑥𐑧𐑯 𐑣𐑨𐑝 𐑐𐑧𐑯𐑦𐑑𐑮𐑱𐑑𐑩𐑛 𐑝𐑧𐑮𐑦 𐑓𐑸 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑕𐑒𐑦𐑯 𐑝 𐑞 𐑨𐑛𐑳𐑤𐑑 𐑢𐑱𐑤; 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕, 𐑲 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑚𐑤𐑧𐑕𐑑 𐑢𐑦𐑞 𐑩𐑯 𐑪𐑐𐑼𐑑𐑿𐑯𐑦𐑑𐑦 𐑑 𐑛𐑦𐑕𐑧𐑒𐑑 𐑣𐑦𐑥 𐑦𐑯 𐑥𐑦𐑯𐑦𐑗𐑼. 𐑦𐑯 𐑩 𐑖𐑦𐑐 𐑲 𐑚𐑦𐑤𐑪𐑙𐑛 𐑑, 𐑩 𐑕𐑥𐑷𐑤 𐑒𐑳𐑚 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑢𐑪𐑟 𐑢𐑳𐑯𐑕 𐑚𐑪𐑛𐑦𐑤𐑦 𐑣𐑶𐑕𐑑𐑩𐑛 𐑑 𐑞 𐑛𐑧𐑒 𐑓 𐑣𐑦𐑟 𐑐𐑴𐑒 𐑹 𐑚𐑨𐑜, 𐑑 𐑥𐑱𐑒 𐑖𐑰𐑞𐑟 𐑓 𐑞 𐑚𐑸𐑚𐑟 𐑝 𐑞 𐑣𐑸𐑐𐑵𐑯𐑟, 𐑯 𐑓 𐑞 𐑣𐑧𐑛𐑟 𐑝 𐑞 𐑤𐑭𐑯𐑕𐑩𐑟. 𐑔𐑦𐑙𐑒 𐑿 𐑲 𐑤𐑧𐑑 𐑞𐑨𐑑 𐑗𐑭𐑯𐑕 𐑜𐑴, 𐑢𐑦𐑞𐑬𐑑 𐑿𐑟𐑦𐑙 𐑥𐑲 𐑚𐑴𐑑-𐑣𐑨𐑗𐑩𐑑 𐑯 𐑡𐑨𐑒-𐑯𐑲𐑓, 𐑯 𐑚𐑮𐑱𐑒𐑦𐑙 𐑞 𐑕𐑰𐑤 𐑯 𐑮𐑰𐑛𐑦𐑙 𐑷𐑤 𐑞 𐑒𐑪𐑯𐑑𐑧𐑯𐑑𐑕 𐑝 𐑞𐑨𐑑 𐑘𐑳𐑙 𐑒𐑳𐑚?

𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑕𐑲𐑑. 𐑞 𐑢𐑫𐑛 𐑢𐑪𐑟 𐑜𐑮𐑰𐑯 𐑨𐑟 𐑥𐑪𐑕𐑩𐑟 𐑝 𐑞 ·𐑲𐑕𐑦 ·𐑜𐑤𐑧𐑯; 𐑞 𐑑𐑮𐑰𐑟 𐑕𐑑𐑫𐑛 𐑣𐑲 𐑯 𐑣𐑷𐑑𐑦, 𐑓𐑰𐑤𐑦𐑙 𐑞𐑺 𐑤𐑦𐑝𐑦𐑙 𐑕𐑨𐑐; 𐑞 𐑦𐑯𐑛𐑳𐑕𐑑𐑮𐑾𐑕 𐑻𐑔 𐑚𐑦𐑯𐑰𐑔 𐑢𐑪𐑟 𐑨𐑟 𐑩 𐑢𐑰𐑝𐑼'𐑟 𐑤𐑵𐑥, 𐑢𐑦𐑞 𐑩 𐑜𐑹𐑡𐑩𐑕 𐑒𐑸𐑐𐑩𐑑 𐑪𐑯 𐑦𐑑, 𐑢𐑺𐑪𐑝 𐑞 𐑜𐑮𐑬𐑯𐑛-𐑝𐑲𐑯 𐑑𐑧𐑯𐑛𐑮𐑩𐑤𐑟 𐑓𐑹𐑥𐑛 𐑞 𐑢𐑹𐑐 𐑯 𐑢𐑵𐑓, 𐑯 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑓𐑤𐑬𐑼𐑟 𐑞 𐑓𐑦𐑜𐑼𐑟. 𐑷𐑤 𐑞 𐑑𐑮𐑰𐑟, 𐑢𐑦𐑞 𐑷𐑤 𐑞𐑺 𐑤𐑱𐑛𐑩𐑯 𐑚𐑮𐑭𐑯𐑗𐑩𐑟; 𐑷𐑤 𐑞 𐑖𐑮𐑳𐑚𐑟, 𐑯 𐑓𐑻𐑯𐑟, 𐑯 𐑜𐑮𐑭𐑕𐑩𐑟; 𐑞 𐑥𐑧𐑕𐑦𐑡-𐑒𐑨𐑮𐑦𐑦𐑙 𐑺; 𐑷𐑤 𐑞𐑰𐑟 𐑳𐑯𐑕𐑰𐑕𐑦𐑙𐑤𐑦 𐑢𐑻 𐑨𐑒𐑑𐑦𐑝. 𐑔𐑮𐑵 𐑞 𐑤𐑱𐑕𐑦𐑙𐑟 𐑝 𐑞 𐑤𐑰𐑝𐑟, 𐑞 𐑜𐑮𐑱𐑑 𐑕𐑳𐑯 𐑕𐑰𐑥𐑛 𐑩 𐑓𐑤𐑲𐑦𐑙 𐑖𐑳𐑑𐑩𐑤 𐑢𐑰𐑝𐑦𐑙 𐑞 𐑳𐑯𐑢𐑽𐑦𐑛 𐑝𐑻𐑡𐑼. 𐑴, 𐑚𐑦𐑟𐑦 𐑢𐑰𐑝𐑼! 𐑳𐑯𐑕𐑰𐑯 𐑢𐑰𐑝𐑼!—𐑐𐑷𐑟!—𐑢𐑳𐑯 𐑢𐑻𐑛!—𐑢𐑦𐑞𐑼 𐑓𐑤𐑴𐑟 𐑞 𐑓𐑨𐑚𐑮𐑦𐑒? 𐑢𐑪𐑑 𐑐𐑨𐑤𐑩𐑕 𐑥𐑱 𐑦𐑑 𐑛𐑧𐑒? 𐑢𐑺𐑓𐑹 𐑷𐑤 𐑞𐑰𐑟 𐑕𐑰𐑕𐑤𐑩𐑕 𐑑𐑶𐑤𐑦𐑙𐑟? 𐑕𐑐𐑰𐑒, 𐑢𐑰𐑝𐑼!—𐑕𐑑𐑱 𐑞𐑲 𐑣𐑨𐑯𐑛!—𐑚𐑳𐑑 𐑢𐑳𐑯 𐑕𐑦𐑙𐑜𐑩𐑤 𐑢𐑻𐑛 𐑢𐑦𐑞 𐑞𐑰! 𐑯𐑱—𐑞 𐑖𐑳𐑑𐑩𐑤 𐑓𐑤𐑲𐑟—𐑞 𐑓𐑦𐑜𐑼𐑟 𐑓𐑤𐑴𐑑 𐑓𐑮𐑪𐑥 𐑓𐑹𐑔 𐑞 𐑤𐑵𐑥; 𐑞 𐑓𐑮𐑧𐑖𐑩𐑑-𐑮𐑳𐑖𐑦𐑙 𐑒𐑸𐑐𐑩𐑑 𐑓 𐑧𐑝𐑼 𐑕𐑤𐑲𐑛𐑟 𐑩𐑢𐑱. 𐑞 𐑢𐑰𐑝𐑼-𐑜𐑪𐑛, 𐑣𐑰 𐑢𐑰𐑝𐑟; 𐑯 𐑚𐑲 𐑞𐑨𐑑 𐑢𐑰𐑝𐑦𐑙 𐑦𐑟 𐑣𐑰 𐑛𐑧𐑓𐑩𐑯𐑛, 𐑞𐑨𐑑 𐑣𐑰 𐑣𐑽𐑟 𐑯𐑴 𐑥𐑹𐑑𐑩𐑤 𐑝𐑶𐑕; 𐑯 𐑚𐑲 𐑞𐑨𐑑 𐑣𐑳𐑥𐑦𐑙, 𐑢𐑰, 𐑑𐑵, 𐑣𐑵 𐑤𐑫𐑒 𐑪𐑯 𐑞 𐑤𐑵𐑥 𐑸 𐑛𐑧𐑓𐑩𐑯𐑛; 𐑯 𐑴𐑯𐑤𐑦 𐑢𐑧𐑯 𐑢𐑰 𐑦𐑕𐑒𐑱𐑐 𐑦𐑑 𐑖𐑨𐑤 𐑢𐑰 𐑣𐑽 𐑞 𐑔𐑬𐑟𐑩𐑯𐑛 𐑝𐑶𐑕𐑩𐑟 𐑞𐑨𐑑 𐑕𐑐𐑰𐑒 𐑔𐑮𐑵 𐑦𐑑. 𐑓 𐑰𐑝𐑩𐑯 𐑕𐑴 𐑦𐑑 𐑦𐑟 𐑦𐑯 𐑷𐑤 𐑥𐑩𐑑𐑽𐑾𐑤 𐑓𐑨𐑒𐑑𐑼𐑦𐑟. 𐑞 𐑕𐑐𐑴𐑒𐑩𐑯 𐑢𐑻𐑛𐑟 𐑞𐑨𐑑 𐑸 𐑦𐑯𐑷𐑛𐑩𐑚𐑩𐑤 𐑩𐑥𐑳𐑙 𐑞 𐑓𐑤𐑲𐑦𐑙 𐑕𐑐𐑦𐑯𐑛𐑩𐑤𐑟; 𐑞𐑴𐑟 𐑕𐑱𐑥 𐑢𐑻𐑛𐑟 𐑸 𐑐𐑤𐑱𐑯𐑤𐑦 𐑣𐑻𐑛 𐑢𐑦𐑞𐑬𐑑 𐑞 𐑢𐑷𐑤𐑟, 𐑚𐑻𐑕𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑴𐑐𐑩𐑯𐑛 𐑒𐑱𐑕𐑥𐑩𐑯𐑑𐑕. 𐑞𐑺𐑚𐑲 𐑣𐑨𐑝 𐑝𐑦𐑤𐑩𐑯𐑦𐑟 𐑚𐑰𐑯 𐑛𐑦𐑑𐑧𐑒𐑑𐑩𐑛. 𐑭, 𐑥𐑹𐑑𐑩𐑤! 𐑞𐑧𐑯, 𐑚𐑰 𐑣𐑰𐑛𐑓𐑩𐑤; 𐑓 𐑕𐑴, 𐑦𐑯 𐑷𐑤 𐑞𐑦𐑕 𐑛𐑦𐑯 𐑝 𐑞 𐑜𐑮𐑱𐑑 𐑢𐑻𐑤𐑛'𐑟 𐑤𐑵𐑥, 𐑞𐑲 𐑕𐑳𐑑𐑩𐑤𐑩𐑕𐑑 𐑔𐑦𐑙𐑒𐑦𐑙𐑟 𐑥𐑱 𐑚𐑰 𐑴𐑝𐑼𐑣𐑻𐑛 𐑩𐑓𐑸.

𐑯𐑬, 𐑩𐑥𐑦𐑛 𐑞 𐑜𐑮𐑰𐑯, 𐑤𐑲𐑓-𐑮𐑧𐑕𐑑𐑤𐑩𐑕 𐑤𐑵𐑥 𐑝 𐑞𐑨𐑑 ·𐑸𐑕𐑩𐑕𐑲𐑛𐑾𐑯 𐑢𐑫𐑛, 𐑞 𐑜𐑮𐑱𐑑, 𐑢𐑲𐑑, 𐑢𐑻𐑖𐑦𐑐𐑑 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯 𐑤𐑱 𐑤𐑬𐑯𐑡𐑦𐑙—𐑩 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑲𐑛𐑩𐑤𐑼! 𐑘𐑧𐑑, 𐑨𐑟 𐑞 𐑧𐑝𐑼-𐑢𐑴𐑝𐑩𐑯 𐑝𐑻𐑛𐑩𐑯𐑑 𐑢𐑹𐑐 𐑯 𐑢𐑵𐑓 𐑦𐑯𐑑𐑼𐑥𐑦𐑒𐑕𐑑 𐑯 𐑣𐑳𐑥𐑛 𐑼𐑬𐑯𐑛 𐑣𐑦𐑥, 𐑞 𐑥𐑲𐑑𐑦 𐑲𐑛𐑩𐑤𐑼 𐑕𐑰𐑥𐑛 𐑞 𐑒𐑳𐑯𐑦𐑙 𐑢𐑰𐑝𐑼; 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑷𐑤 𐑢𐑴𐑝𐑩𐑯 𐑴𐑝𐑼 𐑢𐑦𐑞 𐑞 𐑝𐑲𐑯𐑟; 𐑧𐑝𐑮𐑦 𐑥𐑳𐑯𐑔 𐑩𐑕𐑿𐑥𐑦𐑙 𐑜𐑮𐑰𐑯𐑼, 𐑓𐑮𐑧𐑖𐑼 𐑝𐑻𐑡𐑼; 𐑚𐑳𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑩 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯. 𐑤𐑲𐑓 𐑓𐑴𐑤𐑛𐑩𐑛 ·𐑛𐑧𐑔; ·𐑛𐑧𐑔 𐑑𐑮𐑧𐑤𐑦𐑕𐑑 ·𐑤𐑲𐑓; 𐑞 𐑜𐑮𐑦𐑥 𐑜𐑪𐑛 𐑢𐑲𐑝𐑛 𐑢𐑦𐑞 𐑿𐑔𐑓𐑩𐑤 ·𐑤𐑲𐑓, 𐑯 𐑚𐑦𐑜𐑨𐑑 𐑣𐑦𐑥 𐑒𐑻𐑤𐑦-𐑣𐑧𐑛𐑩𐑛 𐑜𐑤𐑹𐑦𐑟.

𐑯𐑬, 𐑢𐑧𐑯 𐑢𐑦𐑞 𐑮𐑶𐑩𐑤 ·𐑑𐑮𐑨𐑯𐑒𐑢𐑴 𐑲 𐑝𐑦𐑟𐑦𐑑𐑩𐑛 𐑞𐑦𐑕 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑢𐑱𐑤, 𐑯 𐑕𐑷 𐑞 𐑕𐑒𐑳𐑤 𐑩𐑯 𐑷𐑤𐑑𐑼, 𐑯 𐑞 𐑸𐑑𐑦𐑓𐑦𐑖𐑩𐑤 𐑕𐑥𐑴𐑒 𐑩𐑕𐑧𐑯𐑛𐑦𐑙 𐑓𐑮𐑪𐑥 𐑢𐑺 𐑞 𐑮𐑾𐑤 𐑡𐑧𐑑 𐑣𐑨𐑛 𐑦𐑖𐑿𐑛, 𐑲 𐑥𐑸𐑝𐑩𐑤𐑛 𐑞𐑨𐑑 𐑞 𐑒𐑦𐑙 𐑖𐑵𐑛 𐑮𐑦𐑜𐑸𐑛 𐑩 𐑗𐑨𐑐𐑩𐑤 𐑨𐑟 𐑩𐑯 𐑪𐑚𐑡𐑧𐑒𐑑 𐑝 ·𐑝𐑻𐑑𐑵. 𐑣𐑰 𐑤𐑭𐑓𐑑. 𐑚𐑳𐑑 𐑥𐑹 𐑲 𐑥𐑸𐑝𐑩𐑤𐑛 𐑞𐑨𐑑 𐑞 𐑐𐑮𐑰𐑕𐑑𐑕 𐑖𐑵𐑛 𐑕𐑢𐑺 𐑞𐑨𐑑 𐑕𐑥𐑴𐑒𐑦 𐑡𐑧𐑑 𐑝 𐑣𐑦𐑟 𐑢𐑪𐑟 𐑡𐑧𐑯𐑿𐑦𐑯. 𐑑 𐑯 𐑓𐑮𐑴 𐑲 𐑐𐑱𐑕𐑑 𐑚𐑦𐑓𐑹 𐑞𐑦𐑕 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯—𐑚𐑮𐑳𐑖𐑑 𐑞 𐑝𐑲𐑯𐑟 𐑩𐑕𐑲𐑛—𐑚𐑮𐑴𐑒 𐑔𐑮𐑵 𐑞 𐑮𐑦𐑚𐑟—𐑯 𐑢𐑦𐑞 𐑩 𐑚𐑷𐑤 𐑝 ·𐑸𐑕𐑩𐑕𐑲𐑛𐑾𐑯 𐑑𐑢𐑲𐑯, 𐑢𐑪𐑯𐑛𐑼𐑛, 𐑧𐑛𐑦𐑛 𐑤𐑪𐑙 𐑩𐑥𐑦𐑛 𐑦𐑑𐑕 𐑥𐑧𐑯𐑦 𐑢𐑲𐑯𐑛𐑦𐑙, 𐑖𐑱𐑛𐑩𐑛 𐑒𐑪𐑤𐑩𐑯𐑱𐑛𐑟 𐑯 𐑸𐑚𐑼𐑟. 𐑚𐑳𐑑 𐑕𐑵𐑯 𐑥𐑲 𐑤𐑲𐑯 𐑢𐑪𐑟 𐑬𐑑; 𐑯 𐑓𐑪𐑤𐑴𐑦𐑙 𐑦𐑑 𐑚𐑨𐑒, 𐑲 𐑦𐑥𐑻𐑡𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑴𐑐𐑩𐑯𐑦𐑙 𐑢𐑺 𐑲 𐑧𐑯𐑑𐑼𐑛. 𐑲 𐑕𐑷 𐑯𐑴 𐑤𐑦𐑝𐑦𐑙 𐑔𐑦𐑙 𐑢𐑦𐑞𐑦𐑯; 𐑯𐑷𐑑 𐑢𐑪𐑟 𐑞𐑺 𐑚𐑳𐑑 𐑚𐑴𐑯𐑟.

𐑒𐑳𐑑𐑦𐑙 𐑥𐑰 𐑩 𐑜𐑮𐑰𐑯 𐑥𐑧𐑠𐑼𐑦𐑙-𐑮𐑪𐑛, 𐑲 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑛𐑲𐑝𐑛 𐑢𐑦𐑞𐑦𐑯 𐑞 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯. 𐑓𐑮𐑪𐑥 𐑞𐑺 𐑨𐑮𐑴-𐑕𐑤𐑦𐑑 𐑦𐑯 𐑞 𐑕𐑒𐑳𐑤, 𐑞 𐑐𐑮𐑰𐑕𐑑𐑕 𐑐𐑼𐑕𐑰𐑝𐑛 𐑥𐑰 𐑑𐑱𐑒𐑦𐑙 𐑞 𐑨𐑤𐑑𐑦𐑑𐑿𐑛 𐑝 𐑞 𐑓𐑲𐑯𐑩𐑤 𐑮𐑦𐑚, “𐑣𐑬 𐑯𐑬!” 𐑞𐑱 𐑖𐑬𐑑𐑩𐑛; “·𐑛𐑺𐑕𐑑 𐑞𐑬 𐑥𐑧𐑠𐑼 𐑞𐑦𐑕 𐑬𐑼 𐑜𐑪𐑛! 𐑞𐑨𐑑'𐑕 𐑓 𐑳𐑕.” “𐑲, 𐑐𐑮𐑰𐑕𐑑𐑕—𐑢𐑧𐑤, 𐑣𐑬 𐑤𐑪𐑙 𐑛𐑵 𐑘𐑰 𐑥𐑱𐑒 𐑣𐑦𐑥, 𐑞𐑧𐑯?” 𐑚𐑳𐑑 𐑣𐑽𐑩𐑐𐑪𐑯 𐑩 𐑓𐑽𐑕 𐑒𐑪𐑯𐑑𐑧𐑕𐑑 𐑮𐑴𐑟 𐑩𐑥𐑳𐑙 𐑞𐑧𐑥, 𐑒𐑩𐑯𐑕𐑻𐑯𐑦𐑙 𐑓𐑰𐑑 𐑯 𐑦𐑯𐑗𐑩𐑟; 𐑞𐑱 𐑒𐑮𐑨𐑒𐑑 𐑰𐑗 𐑳𐑞𐑼'𐑟 𐑕𐑒𐑪𐑯𐑕𐑩𐑟 𐑢𐑦𐑞 𐑞𐑺 𐑘𐑸𐑛-𐑕𐑑𐑦𐑒𐑕—𐑞 𐑜𐑮𐑱𐑑 𐑕𐑒𐑳𐑤 𐑧𐑒𐑴𐑛—𐑯 𐑕𐑰𐑟𐑦𐑙 𐑞𐑨𐑑 𐑤𐑳𐑒𐑦 𐑗𐑭𐑯𐑕, 𐑲 𐑒𐑢𐑦𐑒𐑤𐑦 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑥𐑲 𐑴𐑯 𐑩𐑛𐑥𐑧𐑖𐑩𐑥𐑧𐑯𐑑𐑕.

𐑞𐑰𐑟 𐑩𐑛𐑥𐑧𐑖𐑩𐑥𐑧𐑯𐑑𐑕 𐑲 𐑯𐑬 𐑐𐑮𐑩𐑐𐑴𐑟 𐑑 𐑕𐑧𐑑 𐑚𐑦𐑓𐑹 𐑿. 𐑚𐑳𐑑 𐑓𐑻𐑕𐑑, 𐑚𐑰 𐑦𐑑 𐑮𐑦𐑒𐑹𐑛𐑩𐑛, 𐑞𐑨𐑑, 𐑦𐑯 𐑞𐑦𐑕 𐑥𐑨𐑑𐑼, 𐑲 𐑨𐑥 𐑯𐑪𐑑 𐑓𐑮𐑰 𐑑 𐑳𐑑𐑼 𐑧𐑯𐑦 𐑓𐑨𐑯𐑕𐑦𐑛 𐑥𐑧𐑠𐑼𐑥𐑩𐑯𐑑 𐑲 𐑐𐑤𐑰𐑟. 𐑚𐑦𐑒𐑪𐑟 𐑞𐑺 𐑸 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯 𐑷𐑔𐑪𐑮𐑦𐑑𐑦𐑟 𐑿 𐑒𐑨𐑯 𐑮𐑦𐑓𐑻 𐑑, 𐑑 𐑑𐑧𐑕𐑑 𐑥𐑲 𐑨𐑒𐑘𐑼𐑩𐑕𐑦. 𐑞𐑺 𐑦𐑟 𐑩 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯𐑦𐑒 ·𐑥𐑿𐑟𐑾𐑥, 𐑞𐑱 𐑑𐑧𐑤 𐑥𐑰, 𐑦𐑯 ·𐑣𐑳𐑤, ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛, 𐑢𐑳𐑯 𐑝 𐑞 𐑢𐑱𐑤𐑦𐑙 𐑐𐑹𐑑𐑕 𐑝 𐑞𐑨𐑑 𐑒𐑳𐑯𐑑𐑮𐑦, 𐑢𐑺 𐑞𐑱 𐑣𐑨𐑝 𐑕𐑳𐑥 𐑓𐑲𐑯 𐑕𐑐𐑧𐑕𐑩𐑥𐑩𐑯𐑟 𐑝 𐑓𐑦𐑯-𐑚𐑨𐑒𐑕 𐑯 𐑳𐑞𐑼 𐑢𐑱𐑤𐑟. 𐑤𐑲𐑒𐑢𐑲𐑟, 𐑲 𐑣𐑨𐑝 𐑣𐑻𐑛 𐑞𐑨𐑑 𐑦𐑯 𐑞 𐑥𐑿𐑟𐑾𐑥 𐑝 ·𐑥𐑨𐑯𐑗𐑧𐑕𐑑𐑼, 𐑦𐑯 ·𐑯𐑿 ·𐑣𐑨𐑥𐑐𐑖𐑼, 𐑞𐑱 𐑣𐑨𐑝 𐑢𐑪𐑑 𐑞 𐑐𐑮𐑩𐑐𐑮𐑲𐑩𐑑𐑼𐑟 𐑒𐑷𐑤 “𐑞 𐑴𐑯𐑤𐑦 𐑐𐑻𐑓𐑦𐑒𐑑 𐑕𐑐𐑧𐑕𐑩𐑥𐑩𐑯 𐑝 𐑩 ·𐑜𐑮𐑰𐑯𐑤𐑩𐑯𐑛 𐑹 ·𐑮𐑦𐑝𐑼 ·𐑢𐑱𐑤 𐑦𐑯 𐑞 ·𐑿𐑯𐑲𐑑𐑩𐑛 ·𐑕𐑑𐑱𐑑𐑕.” 𐑥𐑹𐑴𐑝𐑼, 𐑨𐑑 𐑩 𐑐𐑤𐑱𐑕 𐑦𐑯 ·𐑘𐑹𐑒𐑖𐑼, ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛, ·𐑚𐑻𐑑𐑩𐑯 ·𐑒𐑳𐑯𐑕𐑑𐑩𐑚𐑩𐑤 𐑚𐑲 𐑯𐑱𐑥, 𐑩 𐑕𐑻𐑑𐑩𐑯 ·𐑕𐑻 ·𐑒𐑤𐑦𐑓𐑼𐑛 ·𐑒𐑳𐑯𐑕𐑑𐑩𐑚𐑩𐑤 𐑣𐑨𐑟 𐑦𐑯 𐑣𐑦𐑟 𐑐𐑩𐑟𐑧𐑖𐑩𐑯 𐑞 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯 𐑝 𐑩 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤, 𐑚𐑳𐑑 𐑝 𐑥𐑪𐑛𐑼𐑩𐑑 𐑕𐑲𐑟, 𐑚𐑲 𐑯𐑴 𐑥𐑰𐑯𐑟 𐑝 𐑞 𐑓𐑫𐑤-𐑜𐑮𐑴𐑯 𐑥𐑨𐑜𐑯𐑦𐑑𐑿𐑛 𐑝 𐑥𐑲 𐑓𐑮𐑧𐑯𐑛 ·𐑒𐑦𐑙 ·𐑑𐑮𐑨𐑯𐑒𐑢𐑴'𐑟.
