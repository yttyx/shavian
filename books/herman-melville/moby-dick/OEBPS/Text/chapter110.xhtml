·𐑗𐑨𐑐𐑑𐑼 110. ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑦𐑯 𐑣𐑦𐑟 ·𐑒𐑪𐑓𐑦𐑯.

𐑩𐑐𐑪𐑯 𐑕𐑻𐑗𐑦𐑙, 𐑦𐑑 𐑢𐑪𐑟 𐑓𐑬𐑯𐑛 𐑞𐑨𐑑 𐑞 𐑒𐑭𐑕𐑒𐑕 𐑤𐑭𐑕𐑑 𐑕𐑑𐑮𐑳𐑒 𐑦𐑯𐑑𐑵 𐑞 𐑣𐑴𐑤𐑛 𐑢𐑻 𐑐𐑻𐑓𐑦𐑒𐑑𐑤𐑦 𐑕𐑬𐑯𐑛, 𐑯 𐑞𐑨𐑑 𐑞 𐑤𐑰𐑒 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑓𐑻𐑞𐑼 𐑪𐑓. 𐑕𐑴, 𐑦𐑑 𐑚𐑰𐑦𐑙 𐑒𐑭𐑥 𐑢𐑧𐑞𐑼, 𐑞𐑱 𐑚𐑮𐑴𐑒 𐑬𐑑 𐑛𐑰𐑐𐑼 𐑯 𐑛𐑰𐑐𐑼, 𐑛𐑦𐑕𐑑𐑻𐑚𐑦𐑙 𐑞 𐑕𐑤𐑳𐑥𐑚𐑼𐑟 𐑝 𐑞 𐑣𐑿𐑡 𐑜𐑮𐑬𐑯𐑛-𐑑𐑽 𐑚𐑳𐑑𐑕; 𐑯 𐑓𐑮𐑪𐑥 𐑞𐑨𐑑 𐑚𐑤𐑨𐑒 𐑥𐑦𐑛𐑯𐑲𐑑 𐑕𐑧𐑯𐑛𐑦𐑙 𐑞𐑴𐑟 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑥𐑴𐑤𐑟 𐑦𐑯𐑑𐑵 𐑞 𐑛𐑱𐑤𐑲𐑑 𐑩𐑚𐑳𐑝. 𐑕𐑴 𐑛𐑰𐑐 𐑛𐑦𐑛 𐑞𐑱 𐑜𐑴; 𐑯 𐑕𐑴 𐑱𐑯𐑗𐑩𐑯𐑑, 𐑯 𐑒𐑼𐑴𐑛𐑩𐑛, 𐑯 𐑢𐑰𐑛𐑦 𐑞 𐑨𐑕𐑐𐑧𐑒𐑑 𐑝 𐑞 𐑤𐑴𐑼𐑥𐑴𐑕𐑑 𐑐𐑳𐑯𐑗𐑩𐑯𐑕, 𐑞𐑨𐑑 𐑿 𐑷𐑤𐑥𐑴𐑕𐑑 𐑤𐑫𐑒𐑑 𐑯𐑧𐑒𐑕𐑑 𐑓 𐑕𐑳𐑥 𐑥𐑴𐑤𐑛𐑦 𐑒𐑹𐑯𐑼-𐑕𐑑𐑴𐑯 𐑒𐑭𐑕𐑒 𐑒𐑩𐑯𐑑𐑱𐑯𐑦𐑙 𐑒𐑶𐑯𐑟 𐑝 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑯𐑴𐑩, 𐑢𐑦𐑞 𐑒𐑪𐑐𐑦𐑟 𐑝 𐑞 𐑐𐑴𐑕𐑑𐑩𐑛 𐑐𐑤𐑨𐑒𐑸𐑛𐑟, 𐑝𐑱𐑯𐑤𐑦 𐑢𐑹𐑯𐑦𐑙 𐑞 𐑦𐑯𐑓𐑨𐑗𐑵𐑱𐑑𐑩𐑛 𐑴𐑤𐑛 𐑢𐑻𐑤𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑤𐑳𐑛. 𐑑𐑽𐑕 𐑭𐑓𐑑𐑼 𐑑𐑽𐑕, 𐑑𐑵, 𐑝 𐑢𐑷𐑑𐑼, 𐑯 𐑚𐑮𐑧𐑛, 𐑯 𐑚𐑰𐑓, 𐑯 𐑖𐑫𐑒𐑕 𐑝 𐑕𐑑𐑱𐑝𐑟, 𐑯 𐑲𐑼𐑯 𐑚𐑳𐑯𐑛𐑩𐑤𐑟 𐑝 𐑣𐑵𐑐𐑕, 𐑢𐑻 𐑣𐑶𐑕𐑑𐑩𐑛 𐑬𐑑, 𐑑𐑦𐑤 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑞 𐑐𐑲𐑤𐑛 𐑛𐑧𐑒𐑕 𐑢𐑻 𐑣𐑸𐑛 𐑑 𐑜𐑧𐑑 𐑩𐑚𐑬𐑑; 𐑯 𐑞 𐑣𐑪𐑤𐑴 𐑣𐑳𐑤 𐑧𐑒𐑴𐑛 𐑳𐑯𐑛𐑼 𐑓𐑫𐑑, 𐑨𐑟 𐑦𐑓 𐑿 𐑢𐑻 𐑑𐑮𐑧𐑛𐑦𐑙 𐑴𐑝𐑼 𐑧𐑥𐑐𐑑𐑦 𐑒𐑨𐑑𐑩𐑒𐑵𐑥𐑟, 𐑯 𐑮𐑰𐑤𐑛 𐑯 𐑮𐑴𐑤𐑛 𐑦𐑯 𐑞 𐑕𐑰 𐑤𐑲𐑒 𐑩𐑯 𐑺-𐑓𐑮𐑱𐑑𐑩𐑛 𐑛𐑧𐑥𐑦𐑡𐑪𐑯. 𐑑𐑪𐑐-𐑣𐑧𐑝𐑦 𐑢𐑪𐑟 𐑞 𐑖𐑦𐑐 𐑨𐑟 𐑩 𐑛𐑦𐑯𐑼𐑤𐑩𐑕 𐑕𐑑𐑿𐑛𐑩𐑯𐑑 𐑢𐑦𐑞 𐑷𐑤 ·𐑨𐑮𐑦𐑕𐑑𐑪𐑑𐑩𐑤 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛. 𐑢𐑧𐑤 𐑢𐑪𐑟 𐑦𐑑 𐑞𐑨𐑑 𐑞 ·𐑑𐑲𐑓𐑵𐑯𐑟 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑝𐑦𐑟𐑦𐑑 𐑞𐑧𐑥 𐑞𐑧𐑯.

𐑯𐑬, 𐑢𐑧𐑯 𐑞𐑦𐑕 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕 𐑢𐑪𐑟 𐑥𐑱𐑛 𐑯𐑴𐑯 𐑭𐑓𐑑, 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑢𐑪𐑟 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛 𐑑 𐑛𐑵 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑚𐑦𐑛𐑦𐑙, 𐑢𐑪𐑑𐑧𐑝𐑼 𐑦𐑑 𐑥𐑲𐑑 𐑦𐑯𐑒𐑤𐑵𐑛. 𐑞𐑺 𐑢𐑪𐑟 𐑕𐑳𐑥 𐑣𐑰𐑞𐑩𐑯𐑦𐑖, 𐑒𐑪𐑓𐑦𐑯-𐑒𐑳𐑤𐑼𐑛 𐑴𐑤𐑛 𐑤𐑳𐑥𐑚𐑼 𐑩𐑚𐑹𐑛, 𐑢𐑦𐑗, 𐑩𐑐𐑪𐑯 𐑩 𐑤𐑪𐑙 𐑐𐑮𐑰𐑝𐑾𐑕 𐑝𐑶𐑦𐑡, 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑒𐑳𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑨𐑚𐑼𐑦𐑡𐑦𐑯𐑩𐑤 𐑜𐑮𐑴𐑝𐑟 𐑝 𐑞 ·𐑤𐑨𐑒𐑩𐑛𐑱 𐑲𐑤𐑩𐑯𐑛𐑟, 𐑯 𐑓𐑮𐑪𐑥 𐑞𐑰𐑟 𐑛𐑸𐑒 𐑐𐑤𐑨𐑙𐑒𐑕 𐑞 𐑒𐑪𐑓𐑦𐑯 𐑢𐑪𐑟 𐑮𐑧𐑒𐑩𐑥𐑧𐑯𐑛𐑩𐑛 𐑑 𐑚𐑰 𐑥𐑱𐑛. 𐑯𐑴 𐑕𐑵𐑯𐑼 𐑢𐑪𐑟 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑩𐑐𐑮𐑲𐑟𐑛 𐑝 𐑞 𐑹𐑛𐑼, 𐑞𐑨𐑯 𐑑𐑱𐑒𐑦𐑙 𐑣𐑦𐑟 𐑮𐑵𐑤, 𐑣𐑰 𐑓𐑹𐑔𐑢𐑦𐑔 𐑢𐑦𐑞 𐑷𐑤 𐑞 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑑 𐑐𐑮𐑪𐑥𐑐𐑑𐑦𐑑𐑿𐑛 𐑝 𐑣𐑦𐑟 𐑒𐑨𐑮𐑩𐑒𐑑𐑼, 𐑐𐑮𐑴𐑕𐑰𐑛𐑩𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤 𐑯 𐑑𐑫𐑒 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑥𐑧𐑠𐑼 𐑢𐑦𐑞 𐑜𐑮𐑱𐑑 𐑨𐑒𐑘𐑼𐑩𐑕𐑦, 𐑮𐑧𐑜𐑿𐑤𐑼𐑤𐑦 𐑗𐑷𐑒𐑦𐑙 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑐𐑻𐑕𐑩𐑯 𐑨𐑟 𐑣𐑰 𐑖𐑦𐑓𐑑𐑩𐑛 𐑞 𐑮𐑵𐑤.

“𐑭! 𐑐𐑫𐑼 𐑓𐑧𐑤𐑴! 𐑣𐑰'𐑤 𐑣𐑨𐑓 𐑑 𐑛𐑲 𐑯𐑬,” 𐑦𐑡𐑨𐑒𐑿𐑤𐑱𐑑𐑩𐑛 𐑞 ·𐑤𐑪𐑙 ·𐑲𐑤𐑩𐑯𐑛 𐑕𐑱𐑤𐑼.

𐑜𐑴𐑦𐑙 𐑑 𐑣𐑦𐑟 𐑝𐑲𐑕-𐑚𐑧𐑯𐑗, 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑓 𐑒𐑩𐑯𐑝𐑰𐑯𐑾𐑯𐑕 𐑕𐑱𐑒 𐑯 𐑡𐑧𐑯𐑼𐑩𐑤 𐑮𐑧𐑓𐑼𐑩𐑯𐑕, 𐑯𐑬 𐑑𐑮𐑨𐑯𐑕𐑓𐑻𐑦𐑙𐑤𐑦 𐑥𐑧𐑠𐑼𐑛 𐑪𐑯 𐑦𐑑 𐑞 𐑦𐑜𐑟𐑨𐑒𐑑 𐑤𐑧𐑙𐑔 𐑞 𐑒𐑪𐑓𐑦𐑯 𐑢𐑪𐑟 𐑑 𐑚𐑰, 𐑯 𐑞𐑧𐑯 𐑥𐑱𐑛 𐑞 𐑑𐑮𐑨𐑯𐑕𐑓𐑻 𐑐𐑻𐑥𐑩𐑯𐑩𐑯𐑑 𐑚𐑲 𐑒𐑳𐑑𐑦𐑙 𐑑𐑵 𐑯𐑪𐑗𐑩𐑟 𐑨𐑑 𐑦𐑑𐑕 𐑧𐑒𐑕𐑑𐑮𐑧𐑥𐑦𐑑𐑦𐑟. 𐑞𐑦𐑕 𐑛𐑳𐑯, 𐑣𐑰 𐑥𐑸𐑖𐑩𐑤𐑛 𐑞 𐑐𐑤𐑨𐑙𐑒𐑕 𐑯 𐑣𐑦𐑟 𐑑𐑵𐑤𐑟, 𐑯 𐑑 𐑢𐑻𐑒.

𐑚𐑳𐑑 𐑺 𐑞𐑦𐑕 𐑢𐑪𐑟 𐑛𐑳𐑯, ·𐑐𐑦𐑐, 𐑣𐑵 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑤𐑲𐑤𐑦 𐑣𐑪𐑝𐑼𐑦𐑙 𐑯𐑽 𐑚𐑲 𐑷𐑤 𐑞𐑦𐑕 𐑢𐑲𐑤, 𐑛𐑮𐑵 𐑯𐑲 𐑑 𐑣𐑦𐑥 𐑢𐑺 𐑣𐑰 𐑤𐑱, 𐑯 𐑢𐑦𐑞 𐑕𐑪𐑓𐑑 𐑕𐑪𐑚𐑦𐑙𐑟, 𐑑𐑫𐑒 𐑣𐑦𐑥 𐑚𐑲 𐑞 𐑣𐑨𐑯𐑛; 𐑦𐑯 𐑞 𐑳𐑞𐑼, 𐑣𐑴𐑤𐑛𐑦𐑙 𐑣𐑦𐑟 𐑑𐑨𐑥𐑚𐑼𐑰𐑯.

“𐑲 𐑣𐑨𐑝 𐑣𐑻𐑛,” 𐑥𐑻𐑥𐑼𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑜𐑱𐑟𐑦𐑙 𐑛𐑬𐑯 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤, “𐑞𐑨𐑑 𐑦𐑯 𐑝𐑲𐑩𐑤𐑩𐑯𐑑 𐑓𐑰𐑝𐑼𐑟, 𐑥𐑧𐑯, 𐑷𐑤 𐑦𐑜𐑯𐑼𐑩𐑯𐑕, 𐑣𐑨𐑝 𐑑𐑷𐑒𐑑 𐑦𐑯 𐑱𐑯𐑗𐑩𐑯𐑑 𐑑𐑳𐑙𐑟; 𐑯 𐑞𐑨𐑑 𐑢𐑧𐑯 𐑞 𐑥𐑦𐑕𐑑𐑼𐑦 𐑦𐑟 𐑐𐑮𐑴𐑚𐑛, 𐑦𐑑 𐑑𐑻𐑯𐑟 𐑬𐑑 𐑷𐑤𐑢𐑱𐑟 𐑞𐑨𐑑 𐑦𐑯 𐑞𐑺 𐑣𐑴𐑤𐑦 𐑓𐑼𐑜𐑪𐑑𐑩𐑯 𐑗𐑲𐑤𐑛𐑣𐑫𐑛 𐑞𐑴𐑟 𐑱𐑯𐑗𐑩𐑯𐑑 𐑑𐑳𐑙𐑟 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑮𐑾𐑤𐑦 𐑕𐑐𐑴𐑒𐑩𐑯 𐑦𐑯 𐑞𐑺 𐑣𐑽𐑦𐑙 𐑚𐑲 𐑕𐑳𐑥 𐑤𐑪𐑓𐑑𐑦 𐑕𐑒𐑪𐑤𐑼𐑟. 𐑕𐑴, 𐑑 𐑥𐑲 𐑓𐑪𐑯𐑛 𐑓𐑱𐑔, 𐑐𐑫𐑼 ·𐑐𐑦𐑐, 𐑦𐑯 𐑞𐑦𐑕 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑢𐑰𐑑𐑯𐑩𐑕 𐑝 𐑣𐑦𐑟 𐑤𐑵𐑯𐑩𐑕𐑦, 𐑚𐑮𐑦𐑙𐑟 𐑣𐑧𐑝𐑩𐑯𐑤𐑦 𐑝𐑬𐑗𐑼𐑟 𐑝 𐑷𐑤 𐑬𐑼 𐑣𐑧𐑝𐑩𐑯𐑤𐑦 𐑣𐑴𐑥𐑟. 𐑢𐑺 𐑤𐑻𐑯𐑛 𐑣𐑰 𐑞𐑨𐑑, 𐑚𐑳𐑑 𐑞𐑺?—𐑣𐑸𐑒! 𐑣𐑰 𐑕𐑐𐑰𐑒𐑕 𐑩𐑜𐑱𐑯: 𐑚𐑳𐑑 𐑥𐑹 𐑢𐑲𐑤𐑛𐑤𐑦 𐑯𐑬.”

“𐑓𐑹𐑥 𐑑𐑵 𐑯 𐑑𐑵! 𐑤𐑧𐑑'𐑕 𐑥𐑱𐑒 𐑩 ·𐑡𐑧𐑯𐑼𐑩𐑤 𐑝 𐑣𐑦𐑥! 𐑣𐑴, 𐑢𐑺'𐑟 𐑣𐑦𐑟 𐑣𐑸𐑐𐑵𐑯? 𐑤𐑱 𐑦𐑑 𐑩𐑒𐑮𐑪𐑕 𐑣𐑽.—𐑮𐑦𐑜-𐑩-𐑛𐑦𐑜, 𐑛𐑦𐑜, 𐑛𐑦𐑜! 𐑣𐑩𐑟𐑭! 𐑴 𐑓 𐑩 𐑜𐑱𐑥 𐑒𐑪𐑒 𐑯𐑬 𐑑 𐑕𐑦𐑑 𐑩𐑐𐑪𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑯 𐑒𐑮𐑴! ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑛𐑲𐑟 𐑜𐑱𐑥!—𐑥𐑲𐑯𐑛 𐑘𐑰 𐑞𐑨𐑑; ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑛𐑲𐑟 𐑜𐑱𐑥!—𐑑𐑱𐑒 𐑘𐑰 𐑜𐑫𐑛 𐑣𐑰𐑛 𐑝 𐑞𐑨𐑑; ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑛𐑲𐑟 𐑜𐑱𐑥! 𐑲 𐑕𐑱; 𐑜𐑱𐑥, 𐑜𐑱𐑥, 𐑜𐑱𐑥! 𐑚𐑳𐑑 𐑚𐑱𐑕 𐑤𐑦𐑑𐑩𐑤 ·𐑐𐑦𐑐, 𐑣𐑰 𐑛𐑲𐑛 𐑩 𐑒𐑬𐑼𐑛; 𐑛𐑲𐑛 𐑷𐑤 𐑩'𐑖𐑦𐑝𐑼;—𐑬𐑑 𐑩𐑐𐑪𐑯 ·𐑐𐑦𐑐! 𐑣𐑸𐑒 𐑘𐑰; 𐑦𐑓 𐑘𐑰 𐑓𐑲𐑯𐑛 ·𐑐𐑦𐑐, 𐑑𐑧𐑤 𐑷𐑤 𐑞 ·𐑨𐑯𐑑𐑦𐑤𐑦𐑟 𐑣𐑰'𐑟 𐑩 𐑮𐑳𐑯𐑩𐑢𐑱; 𐑩 𐑒𐑬𐑼𐑛, 𐑩 𐑒𐑬𐑼𐑛, 𐑩 𐑒𐑬𐑼𐑛! 𐑑𐑧𐑤 𐑞𐑧𐑥 𐑣𐑰 𐑡𐑳𐑥𐑐𐑑 𐑓𐑮𐑪𐑥 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑! 𐑲'𐑛 𐑯𐑧𐑝𐑼 𐑚𐑰𐑑 𐑥𐑲 𐑑𐑨𐑥𐑚𐑼𐑰𐑯 𐑴𐑝𐑼 𐑚𐑱𐑕 ·𐑐𐑦𐑐, 𐑯 𐑣𐑱𐑤 𐑣𐑦𐑥 ·𐑡𐑧𐑯𐑼𐑩𐑤, 𐑦𐑓 𐑣𐑰 𐑢𐑻 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑛𐑲𐑦𐑙 𐑣𐑽. 𐑯𐑴, 𐑯𐑴! 𐑖𐑱𐑥 𐑩𐑐𐑪𐑯 𐑷𐑤 𐑒𐑬𐑼𐑛𐑟—𐑖𐑱𐑥 𐑩𐑐𐑪𐑯 𐑞𐑧𐑥! 𐑤𐑧𐑑 '𐑩𐑥 𐑜𐑴 𐑛𐑮𐑬𐑯 𐑤𐑲𐑒 ·𐑐𐑦𐑐, 𐑞𐑨𐑑 𐑡𐑳𐑥𐑐𐑑 𐑓𐑮𐑪𐑥 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑. 𐑖𐑱𐑥! 𐑖𐑱𐑥!”
