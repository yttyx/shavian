·𐑗𐑨𐑐𐑑𐑼 119. 𐑞 ·𐑒𐑨𐑯𐑛𐑩𐑤𐑟.

𐑢𐑹𐑥𐑩𐑕𐑑 𐑒𐑤𐑲𐑥𐑟 𐑚𐑳𐑑 𐑯𐑻𐑕 𐑞 𐑒𐑮𐑵𐑩𐑤𐑩𐑕𐑑 𐑓𐑨𐑙𐑟: 𐑞 𐑑𐑲𐑜𐑼 𐑝 ·𐑚𐑧𐑯𐑜𐑨𐑤 𐑒𐑮𐑬𐑗𐑩𐑟 𐑦𐑯 𐑕𐑐𐑲𐑕𐑑 𐑜𐑮𐑴𐑝𐑟 𐑝 𐑕𐑰𐑕𐑤𐑩𐑕 𐑝𐑻𐑡𐑼. 𐑕𐑒𐑲𐑟 𐑞 𐑥𐑴𐑕𐑑 𐑦𐑓𐑳𐑤𐑡𐑩𐑯𐑑 𐑚𐑳𐑑 𐑚𐑭𐑕𐑒𐑩𐑑 𐑞 𐑛𐑧𐑛𐑤𐑾𐑕𐑑 𐑔𐑳𐑯𐑛𐑼𐑟: 𐑜𐑹𐑡𐑩𐑕 ·𐑒𐑿𐑚𐑩 𐑯𐑴𐑟 𐑑𐑹𐑯𐑱𐑛𐑴𐑟 𐑞𐑨𐑑 𐑯𐑧𐑝𐑼 𐑕𐑢𐑧𐑐𐑑 𐑑𐑱𐑥 𐑯𐑹𐑞𐑼𐑯 𐑤𐑨𐑯𐑛𐑟. 𐑕𐑴, 𐑑𐑵, 𐑦𐑑 𐑦𐑟, 𐑞𐑨𐑑 𐑦𐑯 𐑞𐑰𐑟 𐑮𐑦𐑕𐑐𐑤𐑧𐑯𐑛𐑩𐑯𐑑 ·𐑡𐑨𐑐𐑩𐑯𐑰𐑟 𐑕𐑰𐑟 𐑞 𐑥𐑨𐑮𐑦𐑯𐑼 𐑦𐑯𐑒𐑬𐑯𐑑𐑼𐑟 𐑞 𐑛𐑲𐑼𐑩𐑕𐑑 𐑝 𐑷𐑤 𐑕𐑑𐑹𐑥𐑟, 𐑞 ·𐑑𐑲𐑓𐑵𐑯. 𐑦𐑑 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑚𐑻𐑕𐑑 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑞𐑨𐑑 𐑒𐑤𐑬𐑛𐑤𐑩𐑕 𐑕𐑒𐑲, 𐑤𐑲𐑒 𐑩𐑯 𐑦𐑒𐑕𐑐𐑤𐑴𐑛𐑦𐑙 𐑚𐑪𐑥 𐑩𐑐𐑪𐑯 𐑩 𐑛𐑱𐑟𐑛 𐑯 𐑕𐑤𐑰𐑐𐑦 𐑑𐑬𐑯.

𐑑𐑩𐑢𐑹𐑛𐑟 𐑰𐑝𐑯𐑦𐑙 𐑝 𐑞𐑨𐑑 𐑛𐑱, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑑𐑹𐑯 𐑝 𐑣𐑻 𐑒𐑨𐑯𐑝𐑩𐑕, 𐑯 𐑚𐑺-𐑐𐑴𐑤𐑛 𐑢𐑪𐑟 𐑤𐑧𐑓𐑑 𐑑 𐑓𐑲𐑑 𐑩 ·𐑑𐑲𐑓𐑵𐑯 𐑢𐑦𐑗 𐑣𐑨𐑛 𐑕𐑑𐑮𐑳𐑒 𐑣𐑻 𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑩𐑣𐑧𐑛. 𐑢𐑧𐑯 𐑛𐑸𐑒𐑯𐑩𐑕 𐑒𐑱𐑥 𐑪𐑯, 𐑕𐑒𐑲 𐑯 𐑕𐑰 𐑮𐑹𐑛 𐑯 𐑕𐑐𐑤𐑦𐑑 𐑢𐑦𐑞 𐑞 𐑔𐑳𐑯𐑛𐑼, 𐑯 𐑚𐑤𐑱𐑟𐑛 𐑢𐑦𐑞 𐑞 𐑤𐑲𐑑𐑯𐑦𐑙, 𐑞𐑨𐑑 𐑖𐑴𐑛 𐑞 𐑛𐑦𐑕𐑱𐑚𐑩𐑤𐑛 𐑥𐑭𐑕𐑑𐑕 𐑓𐑤𐑳𐑑𐑼𐑦𐑙 𐑣𐑽 𐑯 𐑞𐑺 𐑢𐑦𐑞 𐑞 𐑮𐑨𐑜𐑟 𐑢𐑦𐑗 𐑞 𐑓𐑻𐑕𐑑 𐑓𐑘𐑫𐑼𐑦 𐑝 𐑞 𐑑𐑧𐑥𐑐𐑦𐑕𐑑 𐑣𐑨𐑛 𐑤𐑧𐑓𐑑 𐑓 𐑦𐑑𐑕 𐑭𐑓𐑑𐑼 𐑕𐑐𐑹𐑑.

𐑣𐑴𐑤𐑛𐑦𐑙 𐑚𐑲 𐑩 𐑖𐑮𐑬𐑛, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑢𐑪𐑟 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑪𐑯 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒; 𐑨𐑑 𐑧𐑝𐑮𐑦 𐑓𐑤𐑨𐑖 𐑝 𐑞 𐑤𐑲𐑑𐑯𐑦𐑙 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑩𐑤𐑪𐑓𐑑, 𐑑 𐑕𐑰 𐑢𐑪𐑑 𐑩𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑛𐑦𐑟𐑭𐑕𐑑𐑼 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑚𐑦𐑓𐑷𐑤𐑩𐑯 𐑞 𐑦𐑯𐑑𐑮𐑦𐑒𐑩𐑑 𐑣𐑨𐑥𐑐𐑼 𐑞𐑺; 𐑢𐑲𐑤 ·𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒 𐑢𐑻 𐑛𐑦𐑮𐑧𐑒𐑑𐑦𐑙 𐑞 𐑥𐑧𐑯 𐑦𐑯 𐑞 𐑣𐑲𐑼 𐑣𐑶𐑕𐑑𐑦𐑙 𐑯 𐑓𐑻𐑥𐑼 𐑤𐑨𐑖𐑦𐑙 𐑝 𐑞 𐑚𐑴𐑑𐑕. 𐑚𐑳𐑑 𐑷𐑤 𐑞𐑺 𐑐𐑱𐑯𐑟 𐑕𐑰𐑥𐑛 𐑯𐑷𐑑. 𐑞𐑴 𐑤𐑦𐑓𐑑𐑩𐑛 𐑑 𐑞 𐑝𐑧𐑮𐑦 𐑑𐑪𐑐 𐑝 𐑞 𐑒𐑮𐑱𐑯𐑟, 𐑞 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑒𐑢𐑹𐑑𐑼 𐑚𐑴𐑑 (·𐑱𐑣𐑨𐑚'𐑟) 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑦𐑕𐑒𐑱𐑐. 𐑩 𐑜𐑮𐑱𐑑 𐑮𐑴𐑤𐑦𐑙 𐑕𐑰, 𐑛𐑨𐑖𐑦𐑙 𐑣𐑲 𐑳𐑐 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑮𐑰𐑤𐑦𐑙 𐑖𐑦𐑐'𐑕 𐑣𐑲 𐑑𐑰𐑑𐑼𐑦𐑙 𐑕𐑲𐑛, 𐑕𐑑𐑴𐑝 𐑦𐑯 𐑞 𐑚𐑴𐑑'𐑕 𐑚𐑪𐑑𐑩𐑥 𐑨𐑑 𐑞 𐑕𐑑𐑻𐑯, 𐑯 𐑤𐑧𐑓𐑑 𐑦𐑑 𐑩𐑜𐑱𐑯, 𐑷𐑤 𐑛𐑮𐑦𐑐𐑦𐑙 𐑔𐑮𐑵 𐑤𐑲𐑒 𐑩 𐑕𐑦𐑝.

“𐑚𐑨𐑛 𐑢𐑻𐑒, 𐑚𐑨𐑛 𐑢𐑻𐑒! ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑮𐑦𐑜𐑸𐑛𐑦𐑙 𐑞 𐑮𐑧𐑒, “𐑚𐑳𐑑 𐑞 𐑕𐑰 𐑢𐑦𐑤 𐑣𐑨𐑝 𐑦𐑑𐑕 𐑢𐑱. 𐑕𐑑𐑳𐑚, 𐑓 𐑢𐑳𐑯, 𐑒𐑭𐑯𐑑 𐑓𐑲𐑑 𐑦𐑑. 𐑿 𐑕𐑰, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑩 𐑢𐑱𐑝 𐑣𐑨𐑟 𐑕𐑳𐑗 𐑩 𐑜𐑮𐑱𐑑 𐑤𐑪𐑙 𐑕𐑑𐑸𐑑 𐑚𐑦𐑓𐑹 𐑦𐑑 𐑤𐑰𐑐𐑕, 𐑷𐑤 𐑮𐑬𐑯𐑛 𐑞 𐑢𐑻𐑤𐑛 𐑦𐑑 𐑮𐑳𐑯𐑟, 𐑯 𐑞𐑧𐑯 𐑒𐑳𐑥𐑟 𐑞 𐑕𐑐𐑮𐑦𐑙! 𐑚𐑳𐑑 𐑨𐑟 𐑓 𐑥𐑰, 𐑷𐑤 𐑞 𐑕𐑑𐑸𐑑 𐑲 𐑣𐑨𐑓 𐑑 𐑥𐑰𐑑 𐑦𐑑, 𐑦𐑟 𐑡𐑳𐑕𐑑 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑛𐑧𐑒 𐑣𐑽. 𐑚𐑳𐑑 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛; 𐑦𐑑'𐑕 𐑷𐑤 𐑦𐑯 𐑓𐑳𐑯: 𐑕𐑴 𐑞 𐑴𐑤𐑛 𐑕𐑪𐑙 𐑕𐑧𐑟;”—(𐑕𐑦𐑙𐑟.)

“𐑩𐑝𐑭𐑕𐑑 ·𐑕𐑑𐑳𐑚,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, “𐑤𐑧𐑑 𐑞 ·𐑑𐑲𐑓𐑵𐑯 𐑕𐑦𐑙, 𐑯 𐑕𐑑𐑮𐑲𐑒 𐑣𐑦𐑟 𐑣𐑸𐑐 𐑣𐑽 𐑦𐑯 𐑬𐑼 𐑮𐑦𐑜𐑦𐑙; 𐑚𐑳𐑑 𐑦𐑓 𐑞𐑬 𐑸𐑑 𐑩 𐑚𐑮𐑱𐑝 𐑥𐑨𐑯 𐑞𐑬 𐑢𐑦𐑤𐑑 𐑣𐑴𐑤𐑛 𐑞𐑲 𐑐𐑰𐑕.”

“𐑚𐑳𐑑 𐑲 𐑨𐑥 𐑯𐑪𐑑 𐑩 𐑚𐑮𐑱𐑝 𐑥𐑨𐑯; 𐑯𐑧𐑝𐑼 𐑕𐑧𐑛 𐑲 𐑢𐑪𐑟 𐑩 𐑚𐑮𐑱𐑝 𐑥𐑨𐑯; 𐑲 𐑨𐑥 𐑩 𐑒𐑬𐑼𐑛; 𐑯 𐑲 𐑕𐑦𐑙 𐑑 𐑒𐑰𐑐 𐑳𐑐 𐑥𐑲 𐑕𐑐𐑦𐑮𐑦𐑑𐑕. 𐑯 𐑲 𐑑𐑧𐑤 𐑿 𐑢𐑪𐑑 𐑦𐑑 𐑦𐑟, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑞𐑺'𐑟 𐑯𐑴 𐑢𐑱 𐑑 𐑕𐑑𐑪𐑐 𐑥𐑲 𐑕𐑦𐑙𐑦𐑙 𐑦𐑯 𐑞𐑦𐑕 𐑢𐑻𐑤𐑛 𐑚𐑳𐑑 𐑑 𐑒𐑳𐑑 𐑥𐑲 𐑔𐑮𐑴𐑑. 𐑯 𐑢𐑧𐑯 𐑞𐑨𐑑'𐑕 𐑛𐑳𐑯, 𐑑𐑧𐑯 𐑑 𐑢𐑳𐑯 𐑲 𐑕𐑦𐑙 𐑘𐑰 𐑞 𐑛𐑪𐑒𐑕𐑪𐑤𐑩𐑡𐑦 𐑓 𐑩 𐑢𐑦𐑯𐑛-𐑳𐑐.”

“𐑥𐑨𐑛𐑥𐑩𐑯! 𐑤𐑫𐑒 𐑔𐑮𐑵 𐑥𐑲 𐑲𐑟 𐑦𐑓 𐑞𐑬 𐑣𐑨𐑕𐑑 𐑯𐑳𐑯 𐑝 𐑞𐑲𐑯 𐑴𐑯.”

“𐑢𐑪𐑑! 𐑣𐑬 𐑒𐑨𐑯 𐑿 𐑕𐑰 𐑚𐑧𐑑𐑼 𐑝 𐑩 𐑛𐑸𐑒 𐑯𐑲𐑑 𐑞𐑨𐑯 𐑧𐑯𐑦𐑚𐑪𐑛𐑦 𐑧𐑤𐑕, 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛 𐑣𐑬 𐑓𐑵𐑤𐑦𐑖?”

“𐑣𐑽!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑕𐑰𐑟𐑦𐑙 ·𐑕𐑑𐑳𐑚 𐑚𐑲 𐑞 𐑖𐑴𐑤𐑛𐑼, 𐑯 𐑐𐑶𐑯𐑑𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑢𐑧𐑞𐑼 𐑚𐑬, “𐑥𐑸𐑒𐑩𐑕𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑞𐑨𐑑 𐑞 𐑜𐑱𐑤 𐑒𐑳𐑥𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑰𐑕𐑑𐑢𐑼𐑛, 𐑞 𐑝𐑧𐑮𐑦 𐑒𐑹𐑕 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑑 𐑮𐑳𐑯 𐑓 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒? 𐑞 𐑝𐑧𐑮𐑦 𐑒𐑹𐑕 𐑣𐑰 𐑕𐑢𐑳𐑙 𐑑 𐑞𐑦𐑕 𐑛𐑱 𐑯𐑵𐑯? 𐑯𐑬 𐑥𐑸𐑒 𐑣𐑦𐑟 𐑚𐑴𐑑 𐑞𐑺; 𐑢𐑺 𐑦𐑟 𐑞𐑨𐑑 𐑕𐑑𐑴𐑝? 𐑦𐑯 𐑞 𐑕𐑑𐑻𐑯-𐑖𐑰𐑑𐑕, 𐑥𐑨𐑯; 𐑢𐑺 𐑣𐑰 𐑦𐑟 𐑢𐑴𐑯𐑑 𐑑 𐑕𐑑𐑨𐑯𐑛—𐑣𐑦𐑟 𐑕𐑑𐑨𐑯𐑛-𐑐𐑶𐑯𐑑 𐑦𐑟 𐑕𐑑𐑴𐑝, 𐑥𐑨𐑯! 𐑯𐑬 𐑡𐑳𐑥𐑐 𐑴𐑝𐑼𐑚𐑹𐑛, 𐑯 𐑕𐑦𐑙 𐑩𐑢𐑱, 𐑦𐑓 𐑞𐑬 𐑥𐑳𐑕𐑑!

“𐑲 𐑛𐑴𐑯𐑑 𐑣𐑭𐑓 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑘𐑰: 𐑢𐑪𐑑'𐑕 𐑦𐑯 𐑞 𐑢𐑦𐑯𐑛?”

“𐑘𐑧𐑕, 𐑘𐑧𐑕, 𐑮𐑬𐑯𐑛 𐑞 ·𐑒𐑱𐑐 𐑝 𐑜𐑫𐑛 ·𐑣𐑴𐑐 𐑦𐑟 𐑞 𐑖𐑹𐑑𐑩𐑕𐑑 𐑢𐑱 𐑑 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑,” 𐑕𐑩𐑤𐑦𐑤𐑩𐑒𐑢𐑲𐑟𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑕𐑳𐑛𐑩𐑯𐑤𐑦, 𐑣𐑰𐑛𐑤𐑩𐑕 𐑝 ·𐑕𐑑𐑳𐑚'𐑟 𐑒𐑢𐑧𐑕𐑗𐑩𐑯. “𐑞 𐑜𐑱𐑤 𐑞𐑨𐑑 𐑯𐑬 𐑣𐑨𐑥𐑼𐑟 𐑨𐑑 𐑳𐑕 𐑑 𐑕𐑑𐑱𐑝 𐑳𐑕, 𐑢𐑰 𐑒𐑨𐑯 𐑑𐑻𐑯 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑩 𐑓𐑺 𐑢𐑦𐑯𐑛 𐑞𐑨𐑑 𐑢𐑦𐑤 𐑛𐑮𐑲𐑝 𐑳𐑕 𐑑𐑩𐑢𐑹𐑛𐑟 𐑣𐑴𐑥. 𐑘𐑪𐑯𐑛𐑼, 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛, 𐑷𐑤 𐑦𐑟 𐑚𐑤𐑨𐑒𐑯𐑩𐑕 𐑝 𐑛𐑵𐑥; 𐑚𐑳𐑑 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑣𐑴𐑥𐑢𐑼𐑛—𐑲 𐑕𐑰 𐑦𐑑 𐑤𐑲𐑑𐑩𐑯𐑟 𐑳𐑐 𐑞𐑺; 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑢𐑦𐑞 𐑞 𐑤𐑲𐑑𐑯𐑦𐑙.”

𐑨𐑑 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑦𐑯 𐑢𐑳𐑯 𐑝 𐑞 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑝 𐑐𐑮𐑩𐑓𐑬𐑯𐑛 𐑛𐑸𐑒𐑯𐑩𐑕, 𐑓𐑪𐑤𐑴𐑦𐑙 𐑞 𐑓𐑤𐑨𐑖𐑩𐑟, 𐑩 𐑝𐑶𐑕 𐑢𐑪𐑟 𐑣𐑻𐑛 𐑨𐑑 𐑣𐑦𐑟 𐑕𐑲𐑛; 𐑯 𐑷𐑤𐑥𐑴𐑕𐑑 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑩 𐑝𐑪𐑤𐑦 𐑝 𐑔𐑳𐑯𐑛𐑼 𐑐𐑰𐑤𐑟 𐑮𐑴𐑤𐑛 𐑴𐑝𐑼𐑣𐑧𐑛.

“𐑣𐑵'𐑟 𐑞𐑺?”

“𐑴𐑤𐑛 ·𐑔𐑳𐑯𐑛𐑼!” 𐑕𐑧𐑛 ·𐑱𐑣𐑨𐑚, 𐑜𐑮𐑴𐑐𐑦𐑙 𐑣𐑦𐑟 𐑢𐑱 𐑩𐑤𐑪𐑙 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕 𐑑 𐑣𐑦𐑟 𐑐𐑦𐑝𐑩𐑑-𐑣𐑴𐑤; 𐑚𐑳𐑑 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑓𐑲𐑯𐑛𐑦𐑙 𐑣𐑦𐑟 𐑐𐑭𐑔 𐑥𐑱𐑛 𐑐𐑤𐑱𐑯 𐑑 𐑣𐑦𐑥 𐑚𐑲 𐑧𐑤𐑚𐑴𐑛 𐑤𐑭𐑯𐑕𐑩𐑟 𐑝 𐑓𐑲𐑼.

𐑯𐑬, 𐑨𐑟 𐑞 𐑤𐑲𐑑𐑯𐑦𐑙 𐑮𐑪𐑛 𐑑 𐑩 𐑕𐑐𐑲𐑼 𐑪𐑯 𐑖𐑹 𐑦𐑟 𐑦𐑯𐑑𐑧𐑯𐑛𐑩𐑛 𐑑 𐑒𐑨𐑮𐑦 𐑪𐑓 𐑞 𐑐𐑧𐑮𐑦𐑤𐑩𐑕 𐑓𐑤𐑵𐑦𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑶𐑤; 𐑕𐑴 𐑞 𐑒𐑦𐑯𐑛𐑮𐑩𐑛 𐑮𐑪𐑛 𐑢𐑦𐑗 𐑨𐑑 𐑕𐑰 𐑕𐑳𐑥 𐑖𐑦𐑐𐑕 𐑒𐑨𐑮𐑦 𐑑 𐑰𐑗 𐑥𐑭𐑕𐑑, 𐑦𐑟 𐑦𐑯𐑑𐑧𐑯𐑛𐑩𐑛 𐑑 𐑒𐑩𐑯𐑛𐑳𐑒𐑑 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑷𐑑𐑼. 𐑚𐑳𐑑 𐑨𐑟 𐑞𐑦𐑕 𐑒𐑪𐑯𐑛𐑳𐑒𐑑𐑼 𐑥𐑳𐑕𐑑 𐑛𐑦𐑕𐑧𐑯𐑛 𐑑 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑛𐑧𐑐𐑔, 𐑞𐑨𐑑 𐑦𐑑𐑕 𐑧𐑯𐑛 𐑥𐑱 𐑩𐑝𐑶𐑛 𐑷𐑤 𐑒𐑪𐑯𐑑𐑨𐑒𐑑 𐑢𐑦𐑞 𐑞 𐑣𐑳𐑤; 𐑯 𐑨𐑟 𐑥𐑹𐑴𐑝𐑼, 𐑦𐑓 𐑒𐑧𐑐𐑑 𐑒𐑪𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑑𐑴𐑦𐑙 𐑞𐑺, 𐑦𐑑 𐑢𐑫𐑛 𐑚𐑰 𐑤𐑲𐑩𐑚𐑩𐑤 𐑑 𐑥𐑧𐑯𐑦 𐑥𐑦𐑕𐑣𐑨𐑐𐑕, 𐑚𐑦𐑕𐑲𐑛𐑟 𐑦𐑯𐑑𐑼𐑓𐑽𐑦𐑙 𐑯𐑪𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑝 𐑞 𐑮𐑦𐑜𐑦𐑙, 𐑯 𐑥𐑹 𐑹 𐑤𐑧𐑕 𐑦𐑥𐑐𐑰𐑛𐑦𐑙 𐑞 𐑝𐑧𐑕𐑩𐑤'𐑟 𐑢𐑱 𐑦𐑯 𐑞 𐑢𐑷𐑑𐑼; 𐑚𐑦𐑒𐑪𐑟 𐑝 𐑷𐑤 𐑞𐑦𐑕, 𐑞 𐑤𐑴𐑼 𐑐𐑸𐑑𐑕 𐑝 𐑩 𐑖𐑦𐑐'𐑕 𐑤𐑲𐑑𐑯𐑦𐑙-𐑮𐑪𐑛𐑟 𐑸 𐑯𐑪𐑑 𐑷𐑤𐑢𐑱𐑟 𐑴𐑝𐑼𐑚𐑹𐑛; 𐑚𐑳𐑑 𐑸 𐑡𐑧𐑯𐑼𐑩𐑤𐑦 𐑥𐑱𐑛 𐑦𐑯 𐑤𐑪𐑙 𐑕𐑤𐑧𐑯𐑛𐑼 𐑤𐑦𐑙𐑒𐑕, 𐑕𐑴 𐑨𐑟 𐑑 𐑚𐑰 𐑞 𐑥𐑹 𐑮𐑧𐑛𐑦𐑤𐑦 𐑣𐑷𐑤𐑛 𐑳𐑐 𐑦𐑯𐑑𐑵 𐑞 𐑗𐑱𐑯𐑟 𐑬𐑑𐑕𐑲𐑛, 𐑹 𐑔𐑮𐑴𐑯 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰, 𐑨𐑟 𐑩𐑒𐑱𐑠𐑩𐑯 𐑥𐑱 𐑮𐑦𐑒𐑢𐑲𐑼.

“𐑞 𐑮𐑪𐑛𐑟! 𐑞 𐑮𐑪𐑛𐑟!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑑 𐑞 𐑒𐑮𐑵, 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑨𐑛𐑥𐑪𐑯𐑦𐑖𐑑 𐑑 𐑝𐑦𐑡𐑦𐑤𐑩𐑯𐑕 𐑚𐑲 𐑞 𐑝𐑦𐑝𐑦𐑛 𐑤𐑲𐑑𐑯𐑦𐑙 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑚𐑰𐑯 𐑛𐑸𐑑𐑦𐑙 𐑓𐑤𐑨𐑥𐑚𐑴𐑟, 𐑑 𐑤𐑲𐑑 ·𐑱𐑣𐑨𐑚 𐑑 𐑣𐑦𐑟 𐑐𐑴𐑕𐑑. “𐑸 𐑞𐑱 𐑴𐑝𐑼𐑚𐑹𐑛? 𐑛𐑮𐑪𐑐 𐑞𐑧𐑥 𐑴𐑝𐑼, 𐑓𐑹 𐑯 𐑭𐑓𐑑. 𐑒𐑢𐑦𐑒!”

“𐑩𐑝𐑭𐑕𐑑!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚; “𐑤𐑧𐑑'𐑕 𐑣𐑨𐑝 𐑓𐑺 𐑐𐑤𐑱 𐑣𐑽, 𐑞𐑴 𐑢𐑰 𐑚𐑰 𐑞 𐑢𐑰𐑒𐑼 𐑕𐑲𐑛. 𐑘𐑧𐑑 𐑲'𐑤 𐑒𐑩𐑯𐑑𐑮𐑦𐑚𐑿𐑑 𐑑 𐑮𐑱𐑟 𐑮𐑪𐑛𐑟 𐑪𐑯 𐑞 ·𐑙𐑦𐑥𐑩𐑤𐑱𐑟 𐑯 ·𐑨𐑯𐑛𐑰𐑟, 𐑞𐑨𐑑 𐑷𐑤 𐑞 𐑢𐑻𐑤𐑛 𐑥𐑱 𐑚𐑰 𐑕𐑦𐑒𐑘𐑫𐑼𐑛; 𐑚𐑳𐑑 𐑬𐑑 𐑪𐑯 𐑐𐑮𐑦𐑝𐑩𐑤𐑦𐑡𐑩𐑟! 𐑤𐑧𐑑 𐑞𐑧𐑥 𐑚𐑰, 𐑕𐑻.”

“𐑤𐑫𐑒 𐑩𐑤𐑪𐑓𐑑!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒. “𐑞 𐑒𐑹𐑐𐑩𐑕𐑩𐑯𐑑𐑕! 𐑞 𐑒𐑹𐑐𐑩𐑕𐑩𐑯𐑑𐑕!”

𐑷𐑤 𐑞 𐑘𐑸𐑛-𐑸𐑥𐑟 𐑢𐑻 𐑑𐑦𐑐𐑑 𐑢𐑦𐑞 𐑩 𐑐𐑨𐑤𐑦𐑛 𐑓𐑲𐑼; 𐑯 𐑑𐑳𐑗𐑑 𐑨𐑑 𐑰𐑗 𐑑𐑮𐑲-𐑐𐑶𐑯𐑑𐑩𐑛 𐑤𐑲𐑑𐑯𐑦𐑙-𐑮𐑪𐑛-𐑧𐑯𐑛 𐑢𐑦𐑞 𐑔𐑮𐑰 𐑑𐑱𐑐𐑼𐑦𐑙 𐑢𐑲𐑑 𐑓𐑤𐑱𐑥𐑟, 𐑰𐑗 𐑝 𐑞 𐑔𐑮𐑰 𐑑𐑷𐑤 𐑥𐑭𐑕𐑑𐑕 𐑢𐑪𐑟 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑚𐑻𐑯𐑦𐑙 𐑦𐑯 𐑞𐑨𐑑 𐑕𐑳𐑤𐑓𐑼𐑩𐑕 𐑺, 𐑤𐑲𐑒 𐑔𐑮𐑰 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑢𐑨𐑒𐑕 𐑑𐑱𐑐𐑼𐑟 𐑚𐑦𐑓𐑹 𐑩𐑯 𐑷𐑤𐑑𐑼.

“𐑚𐑤𐑭𐑕𐑑 𐑞 𐑚𐑴𐑑! 𐑤𐑧𐑑 𐑦𐑑 𐑜𐑴!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚 𐑨𐑑 𐑞𐑦𐑕 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑨𐑟 𐑩 𐑕𐑢𐑭𐑖𐑦𐑙 𐑕𐑰 𐑣𐑰𐑝𐑛 𐑳𐑐 𐑳𐑯𐑛𐑼 𐑣𐑦𐑟 𐑴𐑯 𐑤𐑦𐑑𐑩𐑤 𐑒𐑮𐑭𐑓𐑑, 𐑕𐑴 𐑞𐑨𐑑 𐑦𐑑𐑕 𐑜𐑳𐑯𐑩𐑤 𐑝𐑲𐑩𐑤𐑩𐑯𐑑𐑤𐑦 𐑡𐑨𐑥𐑛 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛, 𐑨𐑟 𐑣𐑰 𐑢𐑪𐑟 𐑐𐑭𐑕𐑦𐑙 𐑩 𐑤𐑨𐑖𐑦𐑙. “𐑚𐑤𐑭𐑕𐑑 𐑦𐑑!”—𐑚𐑳𐑑 𐑕𐑤𐑦𐑐𐑦𐑙 𐑚𐑨𐑒𐑢𐑼𐑛 𐑪𐑯 𐑞 𐑛𐑧𐑒, 𐑣𐑦𐑟 𐑳𐑐𐑤𐑦𐑓𐑑𐑩𐑛 𐑲𐑟 𐑒𐑷𐑑 𐑞 𐑓𐑤𐑱𐑥𐑟; 𐑯 𐑦𐑥𐑰𐑛𐑾𐑑𐑤𐑦 𐑖𐑦𐑓𐑑𐑦𐑙 𐑣𐑦𐑟 𐑑𐑴𐑯 𐑣𐑰 𐑒𐑮𐑲𐑛—“𐑞 𐑒𐑹𐑐𐑩𐑕𐑩𐑯𐑑𐑕 𐑣𐑨𐑝 𐑥𐑻𐑕𐑦 𐑪𐑯 𐑳𐑕 𐑷𐑤!”

𐑑 𐑕𐑱𐑤𐑼𐑟, 𐑴𐑞𐑟 𐑸 𐑣𐑬𐑕𐑣𐑴𐑤𐑛 𐑢𐑻𐑛𐑟; 𐑞𐑱 𐑢𐑦𐑤 𐑕𐑢𐑺 𐑦𐑯 𐑞 𐑑𐑮𐑭𐑯𐑕 𐑝 𐑞 𐑒𐑭𐑥, 𐑯 𐑦𐑯 𐑞 𐑑𐑰𐑔 𐑝 𐑞 𐑑𐑧𐑥𐑐𐑦𐑕𐑑; 𐑞𐑱 𐑢𐑦𐑤 𐑦𐑥𐑐𐑮𐑦𐑒𐑱𐑑 𐑒𐑻𐑕𐑩𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑪𐑐𐑕𐑱𐑤-𐑘𐑸𐑛-𐑸𐑥𐑟, 𐑢𐑧𐑯 𐑥𐑴𐑕𐑑 𐑞𐑱 𐑑𐑰𐑑𐑼 𐑴𐑝𐑼 𐑑 𐑩 𐑕𐑰𐑞𐑦𐑙 𐑕𐑰; 𐑚𐑳𐑑 𐑦𐑯 𐑷𐑤 𐑥𐑲 𐑝𐑶𐑦𐑡𐑦𐑙𐑟, 𐑕𐑧𐑤𐑛𐑩𐑥 𐑣𐑨𐑝 𐑲 𐑣𐑻𐑛 𐑩 𐑒𐑪𐑥𐑩𐑯 𐑴𐑔 𐑢𐑧𐑯 ·𐑜𐑪𐑛'𐑟 𐑚𐑻𐑯𐑦𐑙 𐑓𐑦𐑙𐑜𐑼 𐑣𐑨𐑟 𐑚𐑰𐑯 𐑤𐑱𐑛 𐑪𐑯 𐑞 𐑖𐑦𐑐; 𐑢𐑧𐑯 𐑣𐑦𐑟 “𐑥𐑧𐑯𐑧, ·𐑥𐑰𐑯𐑦, ·𐑑𐑧𐑒𐑩𐑤 ·𐑿𐑓𐑸𐑟𐑦𐑯” 𐑣𐑨𐑟 𐑚𐑰𐑯 𐑢𐑴𐑝𐑩𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑖𐑮𐑬𐑛𐑟 𐑯 𐑞 𐑒𐑹𐑛𐑦𐑡.

𐑢𐑲𐑤 𐑞𐑦𐑕 𐑐𐑨𐑤𐑦𐑛𐑯𐑩𐑕 𐑢𐑪𐑟 𐑚𐑻𐑯𐑦𐑙 𐑩𐑤𐑪𐑓𐑑, 𐑓𐑿 𐑢𐑻𐑛𐑟 𐑢𐑻 𐑣𐑻𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑦𐑯𐑗𐑭𐑯𐑑𐑩𐑛 𐑒𐑮𐑵; 𐑣𐑵 𐑦𐑯 𐑢𐑳𐑯 𐑔𐑦𐑒 𐑒𐑤𐑳𐑕𐑑𐑼 𐑕𐑑𐑫𐑛 𐑪𐑯 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤, 𐑷𐑤 𐑞𐑺 𐑲𐑟 𐑜𐑤𐑰𐑥𐑦𐑙 𐑦𐑯 𐑞𐑨𐑑 𐑐𐑱𐑤 𐑓𐑪𐑕𐑓𐑼𐑧𐑕𐑩𐑯𐑕, 𐑤𐑲𐑒 𐑩 𐑓𐑸 𐑩𐑢𐑱 𐑒𐑪𐑯𐑕𐑑𐑩𐑤𐑱𐑖𐑩𐑯 𐑝 𐑕𐑑𐑸𐑟. 𐑮𐑦𐑤𐑰𐑝𐑛 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑜𐑴𐑕𐑑𐑤𐑦 𐑤𐑲𐑑, 𐑞 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑡𐑧𐑑 𐑯𐑰𐑜𐑮𐑴, ·𐑛𐑨𐑜𐑵, 𐑤𐑵𐑥𐑛 𐑳𐑐 𐑑 𐑔𐑮𐑲𐑕 𐑣𐑦𐑟 𐑮𐑾𐑤 𐑕𐑑𐑨𐑗𐑼, 𐑯 𐑕𐑰𐑥𐑛 𐑞 𐑚𐑤𐑨𐑒 𐑒𐑤𐑬𐑛 𐑓𐑮𐑪𐑥 𐑢𐑦𐑗 𐑞 𐑔𐑳𐑯𐑛𐑼 𐑣𐑨𐑛 𐑒𐑳𐑥. 𐑞 𐑐𐑸𐑑𐑩𐑛 𐑥𐑬𐑔 𐑝 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑮𐑦𐑝𐑰𐑤𐑛 𐑣𐑦𐑟 𐑖𐑸𐑒-𐑢𐑲𐑑 𐑑𐑰𐑔, 𐑢𐑦𐑗 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑜𐑤𐑰𐑥𐑛 𐑨𐑟 𐑦𐑓 𐑞𐑱 𐑑𐑵 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑑𐑦𐑐𐑑 𐑚𐑲 𐑒𐑹𐑐𐑩𐑕𐑩𐑯𐑑𐑕; 𐑢𐑲𐑤 𐑤𐑦𐑑 𐑳𐑐 𐑚𐑲 𐑞 𐑐𐑮𐑰𐑑𐑼𐑯𐑨𐑗𐑼𐑩𐑤 𐑤𐑲𐑑, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑑𐑩𐑑𐑵𐑦𐑙 𐑚𐑻𐑯𐑛 𐑤𐑲𐑒 ·𐑕𐑩𐑑𐑨𐑯𐑦𐑒 𐑚𐑤𐑵 𐑓𐑤𐑱𐑥𐑟 𐑪𐑯 𐑣𐑦𐑟 𐑚𐑪𐑛𐑦.

𐑞 𐑑𐑨𐑚𐑤𐑴 𐑷𐑤 𐑢𐑱𐑯𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑢𐑦𐑞 𐑞 𐑐𐑨𐑤𐑦𐑛𐑯𐑩𐑕 𐑩𐑤𐑪𐑓𐑑; 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑯 𐑧𐑝𐑮𐑦 𐑕𐑴𐑤 𐑪𐑯 𐑣𐑻 𐑛𐑧𐑒𐑕 𐑢𐑻 𐑮𐑨𐑐𐑑 𐑦𐑯 𐑩 𐑐𐑷𐑤. 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑹 𐑑𐑵 𐑐𐑭𐑕𐑑, 𐑢𐑧𐑯 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑜𐑴𐑦𐑙 𐑓𐑹𐑢𐑼𐑛, 𐑐𐑫𐑖𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑕𐑳𐑥 𐑢𐑳𐑯. 𐑦𐑑 𐑢𐑪𐑟 ·𐑕𐑑𐑳𐑚. “𐑢𐑪𐑑 𐑔𐑦𐑙𐑒𐑩𐑕𐑑 𐑞𐑬 𐑯𐑬, 𐑥𐑨𐑯; 𐑲 𐑣𐑻𐑛 𐑞𐑲 𐑒𐑮𐑲; 𐑦𐑑 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑞 𐑕𐑱𐑥 𐑦𐑯 𐑞 𐑕𐑪𐑙.”

“𐑯𐑴, 𐑯𐑴, 𐑦𐑑 𐑢𐑪𐑟𐑩𐑯𐑑; 𐑲 𐑕𐑧𐑛 𐑞 𐑒𐑹𐑐𐑩𐑕𐑩𐑯𐑑𐑕 𐑣𐑨𐑝 𐑥𐑻𐑕𐑦 𐑪𐑯 𐑳𐑕 𐑷𐑤; 𐑯 𐑲 𐑣𐑴𐑐 𐑞𐑱 𐑢𐑦𐑤, 𐑕𐑑𐑦𐑤. 𐑚𐑳𐑑 𐑛𐑵 𐑞𐑱 𐑴𐑯𐑤𐑦 𐑣𐑨𐑝 𐑥𐑻𐑕𐑦 𐑪𐑯 𐑤𐑪𐑙 𐑓𐑱𐑕𐑩𐑟?—𐑣𐑨𐑝 𐑞𐑱 𐑯𐑴 𐑚𐑬𐑩𐑤𐑟 𐑓 𐑩 𐑤𐑭𐑓? 𐑯 𐑤𐑫𐑒 𐑘𐑰, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒—𐑚𐑳𐑑 𐑦𐑑'𐑕 𐑑𐑵 𐑛𐑸𐑒 𐑑 𐑤𐑫𐑒. 𐑣𐑽 𐑥𐑰, 𐑞𐑧𐑯: 𐑲 𐑑𐑱𐑒 𐑞𐑨𐑑 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛 𐑓𐑤𐑱𐑥 𐑢𐑰 𐑕𐑷 𐑓 𐑩 𐑕𐑲𐑯 𐑝 𐑜𐑫𐑛 𐑤𐑳𐑒; 𐑓 𐑞𐑴𐑟 𐑥𐑭𐑕𐑑𐑕 𐑸 𐑮𐑵𐑑𐑩𐑛 𐑦𐑯 𐑩 𐑣𐑴𐑤𐑛 𐑞𐑨𐑑 𐑦𐑟 𐑜𐑴𐑦𐑙 𐑑 𐑚𐑰 𐑗𐑪𐑒 𐑩' 𐑚𐑤𐑪𐑒 𐑢𐑦𐑞 𐑕𐑐𐑻𐑥-𐑶𐑤, 𐑛'𐑘𐑰 𐑕𐑰; 𐑯 𐑕𐑴, 𐑷𐑤 𐑞𐑨𐑑 𐑕𐑐𐑻𐑥 𐑢𐑦𐑤 𐑢𐑻𐑒 𐑳𐑐 𐑦𐑯𐑑𐑵 𐑞 𐑥𐑭𐑕𐑑𐑕, 𐑤𐑲𐑒 𐑕𐑨𐑐 𐑦𐑯 𐑩 𐑑𐑮𐑰. 𐑘𐑧𐑕, 𐑬𐑼 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑𐑕 𐑢𐑦𐑤 𐑘𐑧𐑑 𐑚𐑰 𐑨𐑟 𐑔𐑮𐑰 𐑕𐑐𐑻𐑥𐑩𐑕𐑧𐑑𐑦 𐑒𐑨𐑯𐑛𐑩𐑤𐑟—𐑞𐑨𐑑'𐑕 𐑞 𐑜𐑫𐑛 𐑐𐑮𐑪𐑥𐑦𐑕 𐑢𐑰 𐑕𐑷.”

𐑨𐑑 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑒𐑷𐑑 𐑕𐑲𐑑 𐑝 ·𐑕𐑑𐑳𐑚'𐑟 𐑓𐑱𐑕 𐑕𐑤𐑴𐑤𐑦 𐑚𐑦𐑜𐑦𐑯𐑦𐑙 𐑑 𐑜𐑤𐑦𐑥𐑼 𐑦𐑯𐑑𐑵 𐑕𐑲𐑑. 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑳𐑐𐑢𐑼𐑛𐑟, 𐑣𐑰 𐑒𐑮𐑲𐑛: “𐑕𐑰! 𐑕𐑰!” 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑞 𐑣𐑲 𐑑𐑱𐑐𐑼𐑦𐑙 𐑓𐑤𐑱𐑥𐑟 𐑢𐑻 𐑚𐑦𐑣𐑧𐑤𐑛 𐑢𐑦𐑞 𐑢𐑪𐑑 𐑕𐑰𐑥𐑛 𐑮𐑦𐑛𐑳𐑚𐑩𐑤𐑛 𐑕𐑵𐑐𐑼𐑯𐑨𐑗𐑼𐑩𐑤𐑯𐑩𐑕 𐑦𐑯 𐑞𐑺 𐑐𐑨𐑤𐑼.

“𐑞 𐑒𐑹𐑐𐑩𐑕𐑩𐑯𐑑𐑕 𐑣𐑨𐑝 𐑥𐑻𐑕𐑦 𐑪𐑯 𐑳𐑕 𐑷𐑤,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, 𐑩𐑜𐑱𐑯.

𐑨𐑑 𐑞 𐑚𐑱𐑕 𐑝 𐑞 𐑥𐑱𐑯𐑥𐑭𐑕𐑑, 𐑓𐑫𐑤 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑛𐑳𐑚𐑤𐑵𐑯 𐑯 𐑞 𐑓𐑤𐑱𐑥, 𐑞 ·𐑐𐑸𐑟𐑰 𐑢𐑪𐑟 𐑯𐑰𐑤𐑦𐑙 𐑦𐑯 ·𐑱𐑣𐑨𐑚'𐑟 𐑓𐑮𐑳𐑯𐑑, 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑚𐑬𐑛 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥; 𐑢𐑲𐑤 𐑯𐑽 𐑚𐑲, 𐑓𐑮𐑪𐑥 𐑞 𐑸𐑗𐑑 𐑯 𐑴𐑝𐑼𐑣𐑨𐑙𐑦𐑙 𐑮𐑦𐑜𐑦𐑙, 𐑢𐑺 𐑞𐑱 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑚𐑰𐑯 𐑦𐑯𐑜𐑱𐑡𐑛 𐑕𐑦𐑒𐑘𐑫𐑼𐑦𐑙 𐑩 𐑕𐑐𐑸, 𐑩 𐑯𐑳𐑥𐑚𐑼 𐑝 𐑞 𐑕𐑰𐑥𐑧𐑯, 𐑼𐑧𐑕𐑑𐑩𐑛 𐑚𐑲 𐑞 𐑜𐑤𐑺, 𐑯𐑬 𐑒𐑴𐑣𐑽𐑛 𐑑𐑩𐑜𐑧𐑞𐑼, 𐑯 𐑣𐑳𐑙 𐑐𐑧𐑯𐑡𐑩𐑤𐑩𐑕, 𐑤𐑲𐑒 𐑩 𐑯𐑪𐑑 𐑝 𐑯𐑳𐑥𐑛 𐑢𐑪𐑕𐑐𐑕 𐑓𐑮𐑪𐑥 𐑩 𐑛𐑮𐑵𐑐𐑦𐑙, 𐑹𐑗𐑼𐑛 𐑑𐑢𐑦𐑜. 𐑦𐑯 𐑝𐑺𐑾𐑕 𐑦𐑯𐑗𐑭𐑯𐑑𐑩𐑛 𐑨𐑑𐑦𐑑𐑿𐑛𐑟, 𐑤𐑲𐑒 𐑞 𐑕𐑑𐑨𐑯𐑛𐑦𐑙, 𐑹 𐑕𐑑𐑧𐑐𐑦𐑙, 𐑹 𐑮𐑳𐑯𐑦𐑙 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯𐑟 𐑦𐑯 ·𐑣𐑼𐑒𐑘𐑩𐑤𐑱𐑯𐑾𐑥, 𐑳𐑞𐑼𐑟 𐑮𐑦𐑥𐑱𐑯𐑛 𐑮𐑵𐑑𐑩𐑛 𐑑 𐑞 𐑛𐑧𐑒; 𐑚𐑳𐑑 𐑷𐑤 𐑞𐑺 𐑲𐑟 𐑳𐑐𐑒𐑭𐑕𐑑.

“𐑲, 𐑲, 𐑥𐑧𐑯!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚. “𐑤𐑫𐑒 𐑳𐑐 𐑨𐑑 𐑦𐑑; 𐑥𐑸𐑒 𐑦𐑑 𐑢𐑧𐑤; 𐑞 𐑢𐑲𐑑 𐑓𐑤𐑱𐑥 𐑚𐑳𐑑 𐑤𐑲𐑑𐑕 𐑞 𐑢𐑱 𐑑 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤! 𐑣𐑨𐑯𐑛 𐑥𐑰 𐑞𐑴𐑟 𐑥𐑱𐑯𐑥𐑭𐑕𐑑 𐑤𐑦𐑙𐑒𐑕 𐑞𐑺; 𐑲 𐑢𐑫𐑛 𐑓𐑱𐑯 𐑓𐑰𐑤 𐑞𐑦𐑕 𐑐𐑳𐑤𐑕, 𐑯 𐑤𐑧𐑑 𐑥𐑲𐑯 𐑚𐑰𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑦𐑑; 𐑚𐑤𐑳𐑛 𐑩𐑜𐑱𐑯𐑕𐑑 𐑓𐑲𐑼! 𐑕𐑴.”

𐑞𐑧𐑯 𐑑𐑻𐑯𐑦𐑙—𐑞 𐑤𐑭𐑕𐑑 𐑤𐑦𐑙𐑒 𐑣𐑧𐑤𐑛 𐑓𐑭𐑕𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑤𐑧𐑓𐑑 𐑣𐑨𐑯𐑛, 𐑣𐑰 𐑐𐑫𐑑 𐑣𐑦𐑟 𐑓𐑫𐑑 𐑩𐑐𐑪𐑯 𐑞 ·𐑐𐑸𐑟𐑰; 𐑯 𐑢𐑦𐑞 𐑓𐑦𐑒𐑕𐑑 𐑳𐑐𐑢𐑼𐑛 𐑲, 𐑯 𐑣𐑲-𐑓𐑤𐑳𐑙 𐑮𐑲𐑑 𐑸𐑥, 𐑣𐑰 𐑕𐑑𐑫𐑛 𐑦𐑮𐑧𐑒𐑑 𐑚𐑦𐑓𐑹 𐑞 𐑤𐑪𐑓𐑑𐑦 𐑑𐑮𐑲-𐑐𐑶𐑯𐑑𐑩𐑛 𐑑𐑮𐑦𐑯𐑦𐑑𐑦 𐑝 𐑓𐑤𐑱𐑥𐑟.

“𐑴! 𐑞𐑬 𐑒𐑤𐑽 𐑕𐑐𐑦𐑮𐑦𐑑 𐑝 𐑒𐑤𐑽 𐑓𐑲𐑼, 𐑣𐑵𐑥 𐑪𐑯 𐑞𐑰𐑟 𐑕𐑰𐑟 𐑲 𐑨𐑟 ·𐑐𐑻𐑖𐑩𐑯 𐑢𐑳𐑯𐑕 𐑛𐑦𐑛 𐑢𐑻𐑖𐑦𐑐, 𐑑𐑦𐑤 𐑦𐑯 𐑞 𐑕𐑨𐑒𐑮𐑩𐑥𐑧𐑯𐑑𐑩𐑤 𐑨𐑒𐑑 𐑕𐑴 𐑚𐑻𐑯𐑛 𐑚𐑲 𐑞𐑰, 𐑞𐑨𐑑 𐑑 𐑞𐑦𐑕 𐑬𐑼 𐑲 𐑚𐑺 𐑞 𐑕𐑒𐑸; 𐑲 𐑯𐑬 𐑯𐑴 𐑞𐑰, 𐑞𐑬 𐑒𐑤𐑽 𐑕𐑐𐑦𐑮𐑦𐑑, 𐑯 𐑲 𐑯𐑬 𐑯𐑴 𐑞𐑨𐑑 𐑞𐑲 𐑮𐑲𐑑 𐑢𐑻𐑖𐑦𐑐 𐑦𐑟 𐑛𐑦𐑓𐑲𐑩𐑯𐑕. 𐑑 𐑯𐑲𐑞𐑼 𐑤𐑳𐑝 𐑯𐑹 𐑮𐑧𐑝𐑼𐑩𐑯𐑕 𐑢𐑦𐑤𐑑 𐑞𐑬 𐑚𐑰 𐑒𐑲𐑯𐑛; 𐑯 𐑰'𐑩𐑯 𐑓 𐑣𐑱𐑑 𐑞𐑬 𐑒𐑨𐑯𐑕𐑑 𐑚𐑳𐑑 𐑒𐑦𐑤; 𐑯 𐑷𐑤 𐑸 𐑒𐑦𐑤𐑛. 𐑯𐑴 𐑓𐑽𐑤𐑩𐑕 𐑓𐑵𐑤 𐑯𐑬 𐑓𐑮𐑳𐑯𐑑𐑕 𐑞𐑰. 𐑲 𐑴𐑯 𐑞𐑲 𐑕𐑐𐑰𐑗𐑤𐑩𐑕, 𐑐𐑤𐑱𐑕𐑤𐑩𐑕 𐑐𐑬𐑼; 𐑚𐑳𐑑 𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑜𐑭𐑕𐑐 𐑝 𐑥𐑲 𐑻𐑔𐑒𐑢𐑱𐑒 𐑤𐑲𐑓 𐑢𐑦𐑤 𐑛𐑦𐑕𐑐𐑿𐑑 𐑦𐑑𐑕 𐑳𐑯𐑒𐑩𐑯𐑛𐑦𐑖𐑩𐑯𐑩𐑤, 𐑳𐑯𐑦𐑯𐑑𐑦𐑜𐑮𐑩𐑤 𐑥𐑭𐑕𐑑𐑼𐑦 𐑦𐑯 𐑥𐑰. 𐑦𐑯 𐑞 𐑥𐑦𐑛𐑕𐑑 𐑝 𐑞 𐑐𐑼𐑕𐑪𐑯𐑦𐑓𐑲𐑛 𐑦𐑥𐑐𐑻𐑕𐑩𐑯𐑩𐑤, 𐑩 𐑐𐑻𐑕𐑩𐑯𐑨𐑤𐑦𐑑𐑦 𐑕𐑑𐑨𐑯𐑛𐑟 𐑣𐑽. 𐑞𐑴 𐑚𐑳𐑑 𐑩 𐑐𐑶𐑯𐑑 𐑨𐑑 𐑚𐑧𐑕𐑑; 𐑢𐑧𐑯𐑟𐑴'𐑼 𐑲 𐑒𐑱𐑥; 𐑢𐑺𐑕𐑴𐑺 𐑲 𐑜𐑴; 𐑘𐑧𐑑 𐑢𐑲𐑤 𐑲 𐑻𐑔𐑤𐑦 𐑤𐑲𐑝, 𐑞 𐑒𐑢𐑰𐑯𐑤𐑦 𐑐𐑻𐑕𐑩𐑯𐑨𐑤𐑦𐑑𐑦 𐑤𐑦𐑝𐑟 𐑦𐑯 𐑥𐑰, 𐑯 𐑓𐑰𐑤𐑟 𐑣𐑻 𐑮𐑶𐑩𐑤 𐑮𐑲𐑑𐑕. 𐑚𐑳𐑑 𐑢𐑹 𐑦𐑟 𐑐𐑱𐑯, 𐑯 𐑣𐑱𐑑 𐑦𐑟 𐑢𐑴. 𐑒𐑳𐑥 𐑦𐑯 𐑞𐑲 𐑤𐑴𐑩𐑕𐑑 𐑓𐑹𐑥 𐑝 𐑤𐑳𐑝, 𐑯 𐑲 𐑢𐑦𐑤 𐑯𐑰𐑤 𐑯 𐑒𐑦𐑕 𐑞𐑰; 𐑚𐑳𐑑 𐑨𐑑 𐑞𐑲 𐑣𐑲𐑩𐑕𐑑, 𐑒𐑳𐑥 𐑨𐑟 𐑥𐑽 𐑕𐑵𐑐𐑻𐑯𐑩𐑤 𐑐𐑬𐑼; 𐑯 𐑞𐑴 𐑞𐑬 𐑤𐑷𐑯𐑗𐑩𐑕𐑑 𐑯𐑱𐑝𐑦𐑟 𐑝 𐑓𐑫𐑤-𐑓𐑮𐑱𐑑𐑩𐑛 𐑢𐑻𐑤𐑛𐑟, 𐑞𐑺'𐑟 𐑞𐑨𐑑 𐑦𐑯 𐑣𐑽 𐑞𐑨𐑑 𐑕𐑑𐑦𐑤 𐑮𐑦𐑥𐑱𐑯𐑟 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑑. 𐑴, 𐑞𐑬 𐑒𐑤𐑽 𐑕𐑐𐑦𐑮𐑦𐑑, 𐑝 𐑞𐑲 𐑓𐑲𐑼 𐑞𐑬 𐑥𐑱𐑛𐑩𐑕𐑑 𐑥𐑰, 𐑯 𐑤𐑲𐑒 𐑩 𐑑𐑮𐑵 𐑗𐑲𐑤𐑛 𐑝 𐑓𐑲𐑼, 𐑲 𐑚𐑮𐑰𐑞 𐑦𐑑 𐑚𐑨𐑒 𐑑 𐑞𐑰.”

𐑱𐑣𐑨𐑚'𐑟 𐑣𐑸𐑐𐑵𐑯, 𐑞 𐑢𐑳𐑯 𐑓𐑹𐑡𐑛 𐑨𐑑 ·𐑐𐑻𐑔'𐑕 𐑓𐑲𐑼, 𐑮𐑦𐑥𐑱𐑯𐑛 𐑓𐑻𐑥𐑤𐑦 𐑤𐑨𐑖𐑑 𐑦𐑯 𐑦𐑑𐑕 𐑒𐑩𐑯𐑕𐑐𐑦𐑒𐑿𐑩𐑕 𐑒𐑮𐑪𐑗, 𐑕𐑴 𐑞𐑨𐑑 𐑦𐑑 𐑐𐑮𐑩𐑡𐑧𐑒𐑑𐑩𐑛 𐑚𐑦𐑘𐑪𐑯𐑛 𐑣𐑦𐑟 𐑢𐑱𐑤-𐑚𐑴𐑑'𐑕 𐑚𐑬; 𐑚𐑳𐑑 𐑞 𐑕𐑰 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑕𐑑𐑴𐑝 𐑦𐑑𐑕 𐑚𐑪𐑑𐑩𐑥 𐑣𐑨𐑛 𐑒𐑷𐑟𐑛 𐑞 𐑤𐑵𐑕 𐑤𐑧𐑞𐑼 𐑖𐑰𐑔 𐑑 𐑛𐑮𐑪𐑐 𐑪𐑓; 𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑰𐑯 𐑕𐑑𐑰𐑤 𐑚𐑸𐑚 𐑞𐑺 𐑯𐑬 𐑒𐑱𐑥 𐑩 𐑤𐑧𐑝𐑩𐑤𐑛 𐑓𐑤𐑱𐑥 𐑝 𐑐𐑱𐑤, 𐑓𐑹𐑒𐑑 𐑓𐑲𐑼. 𐑨𐑟 𐑞 𐑕𐑲𐑤𐑩𐑯𐑑 𐑣𐑸𐑐𐑵𐑯 𐑚𐑻𐑯𐑛 𐑞𐑺 𐑤𐑲𐑒 𐑩 𐑕𐑻𐑐𐑩𐑯𐑑'𐑕 𐑑𐑳𐑙, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑜𐑮𐑭𐑕𐑐𐑑 ·𐑱𐑣𐑨𐑚 𐑚𐑲 𐑞 𐑸𐑥—“𐑜𐑪𐑛, ·𐑜𐑪𐑛 𐑦𐑟 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞𐑰, 𐑴𐑤𐑛 𐑥𐑨𐑯; 𐑓𐑹𐑚𐑺! '𐑑𐑦𐑟 𐑩𐑯 𐑦𐑤 𐑝𐑶𐑦𐑡! 𐑦𐑤 𐑚𐑦𐑜𐑳𐑯, 𐑦𐑤 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑛; 𐑤𐑧𐑑 𐑥𐑰 𐑕𐑒𐑢𐑺 𐑞 𐑘𐑸𐑛𐑟, 𐑢𐑲𐑤 𐑢𐑰 𐑥𐑱, 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑯 𐑥𐑱𐑒 𐑩 𐑓𐑺 𐑢𐑦𐑯𐑛 𐑝 𐑦𐑑 𐑣𐑴𐑥𐑢𐑼𐑛𐑟, 𐑑 𐑜𐑴 𐑪𐑯 𐑩 𐑚𐑧𐑑𐑼 𐑝𐑶𐑦𐑡 𐑞𐑨𐑯 𐑞𐑦𐑕.”

𐑴𐑝𐑼𐑣𐑽𐑦𐑙 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑞 𐑐𐑨𐑯𐑦𐑒-𐑕𐑑𐑮𐑦𐑒𐑩𐑯 𐑒𐑮𐑵 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑮𐑨𐑯 𐑑 𐑞 𐑚𐑮𐑱𐑕𐑩𐑟—𐑞𐑴 𐑯𐑪𐑑 𐑩 𐑕𐑱𐑤 𐑢𐑪𐑟 𐑤𐑧𐑓𐑑 𐑩𐑤𐑪𐑓𐑑. 𐑓 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑 𐑷𐑤 𐑞 𐑩𐑜𐑭𐑕𐑑 𐑥𐑱𐑑'𐑕 𐑔𐑷𐑑𐑕 𐑕𐑰𐑥𐑛 𐑞𐑺𐑟; 𐑞𐑱 𐑮𐑱𐑟𐑛 𐑩 𐑣𐑭𐑓 𐑥𐑿𐑑𐑦𐑯𐑩𐑕 𐑒𐑮𐑲. 𐑚𐑳𐑑 𐑛𐑨𐑖𐑦𐑙 𐑞 𐑮𐑨𐑑𐑩𐑤𐑦𐑙 𐑤𐑲𐑑𐑯𐑦𐑙 𐑤𐑦𐑙𐑒𐑕 𐑑 𐑞 𐑛𐑧𐑒, 𐑯 𐑕𐑯𐑨𐑗𐑦𐑙 𐑞 𐑚𐑻𐑯𐑦𐑙 𐑣𐑸𐑐𐑵𐑯, ·𐑱𐑣𐑨𐑚 𐑢𐑱𐑝𐑛 𐑦𐑑 𐑤𐑲𐑒 𐑩 𐑑𐑹𐑗 𐑩𐑥𐑳𐑙 𐑞𐑧𐑥; 𐑕𐑢𐑺𐑦𐑙 𐑑 𐑑𐑮𐑨𐑯𐑕𐑓𐑦𐑒𐑕 𐑢𐑦𐑞 𐑦𐑑 𐑞 𐑓𐑻𐑕𐑑 𐑕𐑱𐑤𐑼 𐑞𐑨𐑑 𐑚𐑳𐑑 𐑒𐑭𐑕𐑑 𐑤𐑵𐑕 𐑩 𐑮𐑴𐑐'𐑕 𐑧𐑯𐑛. 𐑐𐑧𐑑𐑮𐑦𐑓𐑲𐑛 𐑚𐑲 𐑣𐑦𐑟 𐑨𐑕𐑐𐑧𐑒𐑑, 𐑯 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑖𐑮𐑦𐑙𐑒𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑲𐑼𐑦 𐑛𐑸𐑑 𐑞𐑨𐑑 𐑣𐑰 𐑣𐑧𐑤𐑛, 𐑞 𐑥𐑧𐑯 𐑓𐑧𐑤 𐑚𐑨𐑒 𐑦𐑯 𐑛𐑦𐑕𐑥𐑱, 𐑯 ·𐑱𐑣𐑨𐑚 𐑩𐑜𐑱𐑯 𐑕𐑐𐑴𐑒:—

“𐑷𐑤 𐑘𐑹 𐑴𐑞𐑟 𐑑 𐑣𐑳𐑯𐑑 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑸 𐑨𐑟 𐑚𐑲𐑯𐑛𐑦𐑙 𐑨𐑟 𐑥𐑲𐑯; 𐑯 𐑣𐑸𐑑, 𐑕𐑴𐑤, 𐑯 𐑚𐑪𐑛𐑦, 𐑤𐑳𐑙𐑟 𐑯 𐑤𐑲𐑓, 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑚𐑬𐑯𐑛. 𐑯 𐑞𐑨𐑑 𐑘𐑰 𐑥𐑱 𐑯𐑴 𐑑 𐑢𐑪𐑑 𐑑𐑿𐑯 𐑞𐑦𐑕 𐑣𐑸𐑑 𐑚𐑰𐑑𐑕; 𐑤𐑫𐑒 𐑘𐑰 𐑣𐑽; 𐑞𐑳𐑕 𐑲 𐑚𐑤𐑴 𐑬𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑓𐑽!” 𐑯 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑚𐑤𐑭𐑕𐑑 𐑝 𐑣𐑦𐑟 𐑚𐑮𐑧𐑔 𐑣𐑰 𐑧𐑒𐑕𐑑𐑦𐑙𐑜𐑢𐑦𐑖𐑑 𐑞 𐑓𐑤𐑱𐑥.
