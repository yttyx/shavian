·𐑗𐑨𐑐𐑑𐑼 18. 𐑣𐑦𐑟 ·𐑥𐑸𐑒.

𐑨𐑟 𐑢𐑰 𐑢𐑻 𐑢𐑷𐑒𐑦𐑙 𐑛𐑬𐑯 𐑞 𐑧𐑯𐑛 𐑝 𐑞 𐑢𐑹𐑓 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑖𐑦𐑐, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑒𐑨𐑮𐑦𐑦𐑙 𐑣𐑦𐑟 𐑣𐑸𐑐𐑵𐑯, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑦𐑯 𐑣𐑦𐑟 𐑜𐑮𐑳𐑓 𐑝𐑶𐑕 𐑤𐑬𐑛𐑤𐑦 𐑣𐑱𐑤𐑛 𐑳𐑕 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑢𐑦𐑜𐑢𐑭𐑥, 𐑕𐑱𐑦𐑙 𐑣𐑰 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑕𐑩𐑕𐑐𐑧𐑒𐑑𐑩𐑛 𐑥𐑲 𐑓𐑮𐑧𐑯𐑛 𐑢𐑪𐑟 𐑩 𐑒𐑨𐑯𐑦𐑚𐑩𐑤, 𐑯 𐑓𐑻𐑞𐑼𐑥𐑹 𐑩𐑯𐑬𐑯𐑕𐑦𐑙 𐑞𐑨𐑑 𐑣𐑰 𐑤𐑧𐑑 𐑯𐑴 𐑒𐑨𐑯𐑦𐑚𐑩𐑤𐑟 𐑪𐑯 𐑚𐑹𐑛 𐑞𐑨𐑑 𐑒𐑮𐑭𐑓𐑑, 𐑩𐑯𐑤𐑧𐑕 𐑞𐑱 𐑐𐑮𐑰𐑝𐑾𐑕𐑤𐑦 𐑐𐑮𐑩𐑛𐑿𐑕𐑑 𐑞𐑺 𐑐𐑱𐑐𐑼𐑟.

“𐑢𐑪𐑑 𐑛𐑵 𐑿 𐑥𐑰𐑯 𐑚𐑲 𐑞𐑨𐑑, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜?” 𐑕𐑧𐑛 𐑲, 𐑯𐑬 𐑡𐑳𐑥𐑐𐑦𐑙 𐑪𐑯 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑯 𐑤𐑰𐑝𐑦𐑙 𐑥𐑲 𐑒𐑪𐑥𐑮𐑱𐑛 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑪𐑯 𐑞 𐑢𐑹𐑓.

“𐑲 𐑥𐑰𐑯,” 𐑣𐑰 𐑮𐑦𐑐𐑤𐑲𐑛, “𐑣𐑰 𐑥𐑳𐑕𐑑 𐑖𐑴 𐑣𐑦𐑟 𐑐𐑱𐑐𐑼𐑟.”

“𐑘𐑧𐑕,” 𐑕𐑧𐑛 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑪𐑤𐑴 𐑝𐑶𐑕, 𐑕𐑑𐑦𐑒𐑦𐑙 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑓𐑮𐑪𐑥 𐑚𐑦𐑣𐑲𐑯𐑛 ·𐑐𐑰𐑤𐑧𐑜'𐑟, 𐑬𐑑 𐑝 𐑞 𐑢𐑦𐑜𐑢𐑭𐑥. “𐑣𐑰 𐑥𐑳𐑕𐑑 𐑖𐑴 𐑞𐑨𐑑 𐑣𐑰'𐑟 𐑒𐑩𐑯𐑝𐑻𐑑𐑩𐑛. 𐑕𐑳𐑯 𐑝 𐑛𐑸𐑒𐑯𐑩𐑕,” 𐑣𐑰 𐑨𐑛𐑩𐑛, 𐑑𐑻𐑯𐑦𐑙 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, “𐑸𐑑 𐑞𐑬 𐑨𐑑 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑦𐑯 𐑒𐑩𐑥𐑿𐑯𐑾𐑯 𐑢𐑦𐑞 𐑧𐑯𐑦 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯 𐑗𐑻𐑗?”

“𐑢𐑲,” 𐑕𐑧𐑛 𐑲, “𐑣𐑰'𐑟 𐑩 𐑥𐑧𐑥𐑚𐑼 𐑝 𐑞 𐑓𐑻𐑕𐑑 ·𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑖𐑩𐑯𐑩𐑤 ·𐑗𐑻𐑗.” 𐑣𐑽 𐑚𐑰 𐑦𐑑 𐑕𐑧𐑛, 𐑞𐑨𐑑 𐑥𐑧𐑯𐑦 𐑑𐑩𐑑𐑵𐑛 𐑕𐑨𐑝𐑦𐑡𐑩𐑟 𐑕𐑱𐑤𐑦𐑙 𐑦𐑯 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑖𐑦𐑐𐑕 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑒𐑳𐑥 𐑑 𐑚𐑰 𐑒𐑩𐑯𐑝𐑻𐑑𐑩𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑗𐑻𐑗𐑩𐑟.

“𐑓𐑻𐑕𐑑 ·𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑖𐑩𐑯𐑩𐑤 ·𐑗𐑻𐑗,” 𐑒𐑮𐑲𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, “𐑢𐑪𐑑! 𐑞𐑨𐑑 𐑢𐑻𐑖𐑦𐑐𐑕 𐑦𐑯 ·𐑛𐑰𐑒𐑩𐑯 ·𐑛𐑵𐑑𐑼𐑪𐑯𐑩𐑥𐑦 ·𐑒𐑴𐑤𐑥𐑩𐑯'𐑟 𐑥𐑰𐑑𐑦𐑙-𐑣𐑬𐑕?” 𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙, 𐑑𐑱𐑒𐑦𐑙 𐑬𐑑 𐑣𐑦𐑟 𐑕𐑐𐑧𐑒𐑑𐑩𐑒𐑩𐑤𐑟, 𐑣𐑰 𐑮𐑳𐑚𐑛 𐑞𐑧𐑥 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑜𐑮𐑱𐑑 𐑘𐑧𐑤𐑴 𐑚𐑨𐑯𐑛𐑨𐑯𐑩 𐑣𐑨𐑙𐑒𐑼𐑗𐑦𐑓, 𐑯 𐑐𐑫𐑑𐑦𐑙 𐑞𐑧𐑥 𐑪𐑯 𐑝𐑧𐑮𐑦 𐑒𐑺𐑓𐑩𐑤𐑦, 𐑒𐑱𐑥 𐑬𐑑 𐑝 𐑞 𐑢𐑦𐑜𐑢𐑭𐑥, 𐑯 𐑤𐑰𐑯𐑦𐑙 𐑕𐑑𐑦𐑓𐑤𐑦 𐑴𐑝𐑼 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑑𐑫𐑒 𐑩 𐑜𐑫𐑛 𐑤𐑪𐑙 𐑤𐑫𐑒 𐑨𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜.

“𐑣𐑬 𐑤𐑪𐑙 𐑣𐑨𐑔 𐑣𐑰 𐑚𐑰𐑯 𐑩 𐑥𐑧𐑥𐑚𐑼?” 𐑣𐑰 𐑞𐑧𐑯 𐑕𐑧𐑛, 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑥𐑰; “𐑯𐑪𐑑 𐑝𐑧𐑮𐑦 𐑤𐑪𐑙, 𐑲 𐑮𐑭𐑞𐑼 𐑜𐑧𐑕, 𐑘𐑳𐑙 𐑥𐑨𐑯.”

“𐑯𐑴,” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜, “𐑯 𐑣𐑰 𐑣𐑨𐑟𐑩𐑯𐑑 𐑚𐑰𐑯 𐑚𐑨𐑐𐑑𐑲𐑟𐑛 𐑮𐑲𐑑 𐑲𐑞𐑼, 𐑹 𐑦𐑑 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑢𐑪𐑖𐑑 𐑕𐑳𐑥 𐑝 𐑞𐑨𐑑 𐑛𐑧𐑝𐑩𐑤'𐑟 𐑚𐑤𐑵 𐑪𐑓 𐑣𐑦𐑟 𐑓𐑱𐑕.”

“𐑛𐑵 𐑑𐑧𐑤, 𐑯𐑬,” 𐑒𐑮𐑲𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, “𐑦𐑟 𐑞𐑦𐑕 ·𐑓𐑦𐑤𐑦𐑕𐑑𐑲𐑯 𐑩 𐑮𐑧𐑜𐑿𐑤𐑼 𐑥𐑧𐑥𐑚𐑼 𐑝 ·𐑛𐑰𐑒𐑩𐑯 ·𐑛𐑵𐑑𐑼𐑪𐑯𐑩𐑥𐑦'𐑟 𐑥𐑰𐑑𐑦𐑙? 𐑲 𐑯𐑧𐑝𐑼 𐑕𐑷 𐑣𐑦𐑥 𐑜𐑴𐑦𐑙 𐑞𐑺, 𐑯 𐑲 𐑐𐑭𐑕 𐑦𐑑 𐑧𐑝𐑮𐑦 ·𐑤𐑹𐑛'𐑟 𐑛𐑱.”

“𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑧𐑯𐑦𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 ·𐑛𐑰𐑒𐑩𐑯 ·𐑛𐑵𐑑𐑼𐑪𐑯𐑩𐑥𐑦 𐑹 𐑣𐑦𐑟 𐑥𐑰𐑑𐑦𐑙,” 𐑕𐑧𐑛 𐑲; “𐑷𐑤 𐑲 𐑯𐑴 𐑦𐑟, 𐑞𐑨𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑣𐑽 𐑦𐑟 𐑩 𐑚𐑹𐑯 𐑥𐑧𐑥𐑚𐑼 𐑝 𐑞 𐑓𐑻𐑕𐑑 ·𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑖𐑩𐑯𐑩𐑤 ·𐑗𐑻𐑗. 𐑣𐑰 𐑦𐑟 𐑩 𐑛𐑰𐑒𐑩𐑯 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑦𐑟.”

“𐑘𐑳𐑙 𐑥𐑨𐑯,” 𐑕𐑧𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑕𐑑𐑻𐑯𐑤𐑦, “𐑞𐑬 𐑸𐑑 𐑕𐑒𐑲𐑤𐑸𐑒𐑦𐑙 𐑢𐑦𐑞 𐑥𐑰—𐑦𐑒𐑕𐑐𐑤𐑱𐑯 𐑞𐑲𐑕𐑧𐑤𐑓, 𐑞𐑬 𐑘𐑳𐑙 ·𐑣𐑦𐑑𐑲𐑑. 𐑢𐑪𐑑 𐑗𐑻𐑗 𐑛𐑳𐑕𐑑 𐑞𐑰 𐑥𐑰𐑯? 𐑭𐑯𐑕𐑼 𐑥𐑰.”

𐑓𐑲𐑯𐑛𐑦𐑙 𐑥𐑲𐑕𐑧𐑤𐑓 𐑞𐑳𐑕 𐑣𐑸𐑛 𐑐𐑫𐑖𐑑, 𐑲 𐑮𐑦𐑐𐑤𐑲𐑛. “𐑲 𐑥𐑰𐑯, 𐑕𐑻, 𐑞 𐑕𐑱𐑥 𐑱𐑯𐑗𐑩𐑯𐑑 ·𐑒𐑨𐑔𐑩𐑤𐑦𐑒 ·𐑗𐑻𐑗 𐑑 𐑢𐑦𐑗 𐑿 𐑯 𐑲, 𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑞𐑺, 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑣𐑽, 𐑯 𐑷𐑤 𐑝 𐑳𐑕, 𐑯 𐑧𐑝𐑮𐑦 𐑥𐑳𐑞𐑼'𐑟 𐑕𐑳𐑯 𐑯 𐑕𐑴𐑤 𐑝 𐑳𐑕 𐑚𐑦𐑤𐑪𐑙; 𐑞 𐑜𐑮𐑱𐑑 𐑯 𐑧𐑝𐑼𐑤𐑭𐑕𐑑𐑦𐑙 𐑓𐑻𐑕𐑑 ·𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑖𐑩𐑯 𐑝 𐑞𐑦𐑕 𐑣𐑴𐑤 𐑢𐑻𐑖𐑦𐑐𐑦𐑙 𐑢𐑻𐑤𐑛; 𐑢𐑰 𐑷𐑤 𐑚𐑦𐑤𐑪𐑙 𐑑 𐑞𐑨𐑑; 𐑴𐑯𐑤𐑦 𐑕𐑳𐑥 𐑝 𐑳𐑕 𐑗𐑧𐑮𐑦𐑖 𐑕𐑳𐑥 𐑒𐑢𐑽 𐑒𐑮𐑪𐑗𐑩𐑑𐑕 𐑯𐑴 𐑢𐑱𐑟 𐑑𐑳𐑗𐑦𐑙 𐑞 𐑜𐑮𐑨𐑯𐑛 𐑚𐑦𐑤𐑰𐑓; 𐑦𐑯 𐑞𐑨𐑑 𐑢𐑰 𐑷𐑤 𐑡𐑶𐑯 𐑣𐑨𐑯𐑛𐑟.”

“𐑕𐑐𐑤𐑲𐑕, 𐑞𐑬 𐑥𐑰𐑯'𐑕𐑑 𐑕𐑐𐑤𐑲𐑕 𐑣𐑨𐑯𐑛𐑟,” 𐑒𐑮𐑲𐑛 ·𐑐𐑰𐑤𐑧𐑜, 𐑛𐑮𐑷𐑦𐑙 𐑯𐑽𐑼. “𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑿'𐑛 𐑚𐑧𐑑𐑼 𐑖𐑦𐑐 𐑓 𐑩 𐑥𐑦𐑖𐑩𐑯𐑼𐑦, 𐑦𐑯𐑕𐑑𐑧𐑛 𐑝 𐑩 𐑓𐑹-𐑥𐑭𐑕𐑑 𐑣𐑨𐑯𐑛; 𐑲 𐑯𐑧𐑝𐑼 𐑣𐑻𐑛 𐑩 𐑚𐑧𐑑𐑼 𐑕𐑻𐑥𐑩𐑯. 𐑛𐑰𐑒𐑩𐑯 ·𐑛𐑵𐑑𐑼𐑪𐑯𐑩𐑥𐑦—𐑢𐑲 ·𐑓𐑭𐑞𐑼 ·𐑥𐑱𐑐𐑩𐑤 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑒𐑫𐑛𐑩𐑯𐑑 𐑚𐑰𐑑 𐑦𐑑, 𐑯 𐑣𐑰'𐑟 𐑮𐑧𐑒𐑩𐑯𐑛 𐑕𐑳𐑥𐑔𐑦𐑙. 𐑒𐑳𐑥 𐑩𐑚𐑹𐑛, 𐑒𐑳𐑥 𐑩𐑚𐑹𐑛; 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛 𐑩𐑚𐑬𐑑 𐑞 𐑐𐑱𐑐𐑼𐑟. 𐑲 𐑕𐑱, 𐑑𐑧𐑤 ·𐑒𐑢𐑴𐑙𐑪𐑜 𐑞𐑺—𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑿 𐑒𐑷𐑤 𐑣𐑦𐑥? 𐑑𐑧𐑤 ·𐑒𐑢𐑴𐑙𐑪𐑜 𐑑 𐑕𐑑𐑧𐑐 𐑩𐑤𐑪𐑙. 𐑚𐑲 𐑞 𐑜𐑮𐑱𐑑 𐑨𐑙𐑒𐑼, 𐑢𐑪𐑑 𐑩 𐑣𐑸𐑐𐑵𐑯 𐑣𐑰'𐑟 𐑜𐑪𐑑 𐑞𐑺! 𐑤𐑫𐑒𐑕 𐑤𐑲𐑒 𐑜𐑫𐑛 𐑕𐑑𐑳𐑓 𐑞𐑨𐑑; 𐑯 𐑣𐑰 𐑣𐑨𐑯𐑛𐑩𐑤𐑟 𐑦𐑑 𐑩𐑚𐑬𐑑 𐑮𐑲𐑑. 𐑲 𐑕𐑱, ·𐑒𐑢𐑴𐑙𐑪𐑜, 𐑹 𐑢𐑪𐑑𐑧𐑝𐑼 𐑘𐑹 𐑯𐑱𐑥 𐑦𐑟, 𐑛𐑦𐑛 𐑿 𐑧𐑝𐑼 𐑕𐑑𐑨𐑯𐑛 𐑦𐑯 𐑞 𐑣𐑧𐑛 𐑝 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑? 𐑛𐑦𐑛 𐑿 𐑧𐑝𐑼 𐑕𐑑𐑮𐑲𐑒 𐑩 𐑓𐑦𐑖?”

𐑢𐑦𐑞𐑬𐑑 𐑕𐑱𐑦𐑙 𐑩 𐑢𐑻𐑛, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑦𐑯 𐑣𐑦𐑟 𐑢𐑲𐑤𐑛 𐑕𐑹𐑑 𐑝 𐑢𐑱, 𐑡𐑳𐑥𐑐𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑓𐑮𐑪𐑥 𐑞𐑧𐑯𐑕 𐑦𐑯𐑑𐑵 𐑞 𐑚𐑬𐑟 𐑝 𐑢𐑳𐑯 𐑝 𐑞 𐑢𐑱𐑤-𐑚𐑴𐑑𐑕 𐑣𐑨𐑙𐑦𐑙 𐑑 𐑞 𐑕𐑲𐑛; 𐑯 𐑞𐑧𐑯 𐑚𐑮𐑱𐑕𐑦𐑙 𐑣𐑦𐑟 𐑤𐑧𐑓𐑑 𐑯𐑰, 𐑯 𐑐𐑶𐑟𐑦𐑙 𐑣𐑦𐑟 𐑣𐑸𐑐𐑵𐑯, 𐑒𐑮𐑲𐑛 𐑬𐑑 𐑦𐑯 𐑕𐑳𐑥 𐑕𐑳𐑗 𐑢𐑱 𐑨𐑟 𐑞𐑦𐑕:—

“𐑒𐑨𐑐'𐑱𐑯, 𐑿 𐑕𐑰 𐑣𐑦𐑥 𐑕𐑥𐑷𐑤 𐑛𐑮𐑪𐑐 𐑑𐑸 𐑪𐑯 𐑢𐑷𐑑𐑼 𐑛𐑸? 𐑿 𐑕𐑰 𐑣𐑦𐑥? 𐑢𐑧𐑤, 𐑕𐑐𐑴𐑟 𐑣𐑦𐑥 𐑢𐑳𐑯 𐑢𐑱𐑤 𐑲, 𐑢𐑧𐑤, 𐑛𐑧𐑯!” 𐑯 𐑑𐑱𐑒𐑦𐑙 𐑖𐑸𐑐 𐑱𐑥 𐑨𐑑 𐑦𐑑, 𐑣𐑰 𐑛𐑸𐑑𐑩𐑛 𐑞 𐑲𐑼𐑯 𐑮𐑲𐑑 𐑴𐑝𐑼 𐑴𐑤𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛'𐑟 𐑚𐑮𐑷𐑛 𐑚𐑮𐑦𐑥, 𐑒𐑤𐑰𐑯 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑖𐑦𐑐'𐑕 𐑛𐑧𐑒𐑕, 𐑯 𐑕𐑑𐑮𐑳𐑒 𐑞 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑑𐑸 𐑕𐑐𐑪𐑑 𐑬𐑑 𐑝 𐑕𐑲𐑑.

“𐑯𐑬,” 𐑕𐑧𐑛 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑒𐑢𐑲𐑩𐑑𐑤𐑦 𐑣𐑷𐑤𐑦𐑙 𐑦𐑯 𐑞 𐑤𐑲𐑯, “𐑕𐑐𐑴𐑟-ee 𐑣𐑦𐑥 𐑢𐑱𐑤-𐑰 𐑲; 𐑢𐑲, 𐑛𐑨𐑛 𐑢𐑱𐑤 𐑛𐑧𐑛.”

“𐑒𐑢𐑦𐑒, ·𐑚𐑦𐑤𐑛𐑨𐑛,” 𐑕𐑧𐑛 ·𐑐𐑰𐑤𐑧𐑜, 𐑣𐑦𐑟 𐑐𐑸𐑑𐑯𐑼, 𐑣𐑵, 𐑩𐑜𐑭𐑕𐑑 𐑨𐑑 𐑞 𐑒𐑤𐑴𐑕 𐑝𐑦𐑕𐑦𐑯𐑦𐑑𐑦 𐑝 𐑞 𐑓𐑤𐑲𐑦𐑙 𐑣𐑸𐑐𐑵𐑯, 𐑣𐑨𐑛 𐑮𐑦𐑑𐑮𐑰𐑑𐑩𐑛 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑜𐑨𐑙𐑢𐑱. “𐑒𐑢𐑦𐑒, 𐑲 𐑕𐑱, 𐑿 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑯 𐑜𐑧𐑑 𐑞 𐑖𐑦𐑐'𐑕 𐑐𐑱𐑐𐑼𐑟. 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 ·𐑣𐑧𐑡𐑣𐑪𐑜 𐑞𐑺, 𐑲 𐑥𐑰𐑯 ·𐑒𐑢𐑴𐑙𐑪𐑜, 𐑦𐑯 𐑢𐑳𐑯 𐑝 𐑬𐑼 𐑚𐑴𐑑𐑕. 𐑤𐑫𐑒 𐑘𐑰, ·𐑒𐑢𐑴𐑙𐑪𐑜, 𐑢𐑰'𐑤 𐑜𐑦𐑝 𐑘𐑰 𐑞 𐑯𐑲𐑯𐑑𐑾𐑔 𐑤𐑱, 𐑯 𐑞𐑨𐑑'𐑕 𐑥𐑹 𐑞𐑨𐑯 𐑧𐑝𐑼 𐑢𐑪𐑟 𐑜𐑦𐑝𐑩𐑯 𐑩 𐑣𐑸𐑐𐑵𐑯𐑽 𐑘𐑧𐑑 𐑬𐑑 𐑝 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑.”

𐑕𐑴 𐑛𐑬𐑯 𐑢𐑰 𐑢𐑧𐑯𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑯 𐑑 𐑥𐑲 𐑜𐑮𐑱𐑑 𐑡𐑶 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑢𐑪𐑟 𐑕𐑵𐑯 𐑦𐑯𐑮𐑴𐑤𐑛 𐑩𐑥𐑳𐑙 𐑞 𐑕𐑱𐑥 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑑 𐑢𐑦𐑗 𐑲 𐑥𐑲𐑕𐑧𐑤𐑓 𐑚𐑦𐑤𐑪𐑙𐑛.

𐑢𐑧𐑯 𐑷𐑤 𐑐𐑮𐑦𐑤𐑦𐑥𐑦𐑯𐑼𐑦𐑟 𐑢𐑻 𐑴𐑝𐑼 𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑣𐑨𐑛 𐑜𐑪𐑑 𐑧𐑝𐑮𐑦𐑔𐑦𐑙 𐑮𐑧𐑛𐑦 𐑓 𐑕𐑲𐑯𐑦𐑙, 𐑣𐑰 𐑑𐑻𐑯𐑛 𐑑 𐑥𐑰 𐑯 𐑕𐑧𐑛, “𐑲 𐑜𐑧𐑕, ·𐑒𐑢𐑴𐑙𐑪𐑜 𐑞𐑺 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑣𐑬 𐑑 𐑮𐑲𐑑, 𐑛𐑳𐑟 𐑣𐑰? 𐑲 𐑕𐑱, ·𐑒𐑢𐑴𐑙𐑪𐑜, 𐑚𐑤𐑭𐑕𐑑 𐑘𐑰! 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑕𐑲𐑯 𐑞𐑲 𐑯𐑱𐑥 𐑹 𐑥𐑱𐑒 𐑞𐑲 𐑥𐑸𐑒?”

𐑚𐑳𐑑 𐑨𐑑 𐑞𐑦𐑕 𐑒𐑢𐑧𐑕𐑗𐑩𐑯, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑣𐑵 𐑣𐑨𐑛 𐑑𐑢𐑲𐑕 𐑹 𐑔𐑮𐑲𐑕 𐑚𐑦𐑓𐑹 𐑑𐑱𐑒𐑩𐑯 𐑐𐑸𐑑 𐑦𐑯 𐑕𐑦𐑥𐑦𐑤𐑼 𐑕𐑧𐑮𐑦𐑥𐑩𐑯𐑦𐑟, 𐑤𐑫𐑒𐑑 𐑯𐑴 𐑢𐑱𐑟 𐑩𐑚𐑨𐑖𐑑; 𐑚𐑳𐑑 𐑑𐑱𐑒𐑦𐑙 𐑞 𐑪𐑓𐑼𐑛 𐑐𐑧𐑯, 𐑒𐑪𐑐𐑦𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑐𐑱𐑐𐑼, 𐑦𐑯 𐑞 𐑐𐑮𐑪𐑐𐑼 𐑐𐑤𐑱𐑕, 𐑩𐑯 𐑦𐑜𐑟𐑨𐑒𐑑 𐑒𐑬𐑯𐑑𐑼𐑐𐑸𐑑 𐑝 𐑩 𐑒𐑢𐑽 𐑮𐑬𐑯𐑛 𐑓𐑦𐑜𐑼 𐑢𐑦𐑗 𐑢𐑪𐑟 𐑑𐑩𐑑𐑵𐑛 𐑩𐑐𐑪𐑯 𐑣𐑦𐑟 𐑸𐑥; 𐑕𐑴 𐑞𐑨𐑑 𐑔𐑮𐑵 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜'𐑟 𐑪𐑚𐑕𐑑𐑦𐑯𐑩𐑑 𐑥𐑦𐑕𐑑𐑱𐑒 𐑑𐑳𐑗𐑦𐑙 𐑣𐑦𐑟 𐑩𐑐𐑧𐑤𐑧𐑑𐑦𐑝, 𐑦𐑑 𐑕𐑑𐑫𐑛 𐑕𐑳𐑥𐑔𐑦𐑙 𐑤𐑲𐑒 𐑞𐑦𐑕:—

·𐑒𐑢𐑴𐑙𐑪𐑜. 𐑣𐑦𐑟 X 𐑥𐑸𐑒.

𐑥𐑰𐑯𐑢𐑲𐑤 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑦𐑤𐑛𐑨𐑛 𐑕𐑨𐑑 𐑻𐑯𐑩𐑕𐑑𐑤𐑦 𐑯 𐑕𐑑𐑧𐑛𐑓𐑭𐑕𐑑𐑤𐑦 𐑲𐑦𐑙 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑮𐑲𐑟𐑦𐑙 𐑕𐑪𐑤𐑩𐑥𐑤𐑦 𐑯 𐑓𐑳𐑥𐑚𐑩𐑤𐑦𐑙 𐑦𐑯 𐑞 𐑣𐑿𐑡 𐑐𐑪𐑒𐑩𐑑𐑕 𐑝 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛-𐑕𐑒𐑻𐑑𐑩𐑛 𐑛𐑮𐑨𐑚 𐑒𐑴𐑑, 𐑑𐑫𐑒 𐑬𐑑 𐑩 𐑚𐑳𐑯𐑛𐑩𐑤 𐑝 𐑑𐑮𐑨𐑒𐑑𐑕, 𐑯 𐑕𐑦𐑤𐑧𐑒𐑑𐑦𐑙 𐑢𐑳𐑯 𐑦𐑯𐑑𐑲𐑑𐑩𐑤𐑛 “𐑞 ·𐑤𐑨𐑑𐑼 𐑛𐑱 ·𐑒𐑳𐑥𐑦𐑙; 𐑹 𐑯𐑴 ·𐑑𐑲𐑥 𐑑 ·𐑤𐑵𐑟,” 𐑐𐑤𐑱𐑕𐑑 𐑦𐑑 𐑦𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑣𐑨𐑯𐑛𐑟, 𐑯 𐑞𐑧𐑯 𐑜𐑮𐑭𐑕𐑐𐑦𐑙 𐑞𐑧𐑥 𐑯 𐑞 𐑚𐑫𐑒 𐑢𐑦𐑞 𐑚𐑴𐑔 𐑣𐑦𐑟, 𐑤𐑫𐑒𐑑 𐑻𐑯𐑩𐑕𐑑𐑤𐑦 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑲𐑟, 𐑯 𐑕𐑧𐑛, “𐑕𐑳𐑯 𐑝 𐑛𐑸𐑒𐑯𐑩𐑕, 𐑲 𐑥𐑳𐑕𐑑 𐑛𐑵 𐑥𐑲 𐑛𐑿𐑑𐑦 𐑚𐑲 𐑞𐑰; 𐑲 𐑨𐑥 𐑐𐑸𐑑 𐑴𐑯𐑼 𐑝 𐑞𐑦𐑕 𐑖𐑦𐑐, 𐑯 𐑓𐑰𐑤 𐑒𐑩𐑯𐑕𐑻𐑯𐑛 𐑓 𐑞 𐑕𐑴𐑤𐑟 𐑝 𐑷𐑤 𐑦𐑑𐑕 𐑒𐑮𐑵; 𐑦𐑓 𐑞𐑬 𐑕𐑑𐑦𐑤 𐑒𐑤𐑦𐑙𐑩𐑕𐑑 𐑑 𐑞𐑲 ·𐑐𐑱𐑜𐑩𐑯 𐑢𐑱𐑟, 𐑢𐑦𐑗 𐑲 𐑕𐑨𐑛𐑤𐑦 𐑓𐑽, 𐑲 𐑚𐑦𐑕𐑰𐑗 𐑞𐑰, 𐑮𐑦𐑥𐑱𐑯 𐑯𐑪𐑑 𐑓 𐑲 𐑩 ·𐑚𐑧𐑤𐑾𐑤 𐑚𐑪𐑯𐑛𐑟𐑥𐑩𐑯. 𐑕𐑐𐑻𐑯 𐑞 𐑲𐑛𐑩𐑤 ·𐑚𐑧𐑤, 𐑯 𐑞 𐑣𐑦𐑛𐑾𐑕 𐑛𐑮𐑨𐑜𐑩𐑯; 𐑑𐑻𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑮𐑪𐑔 𐑑 𐑒𐑳𐑥; 𐑥𐑲𐑯𐑛 𐑞𐑲𐑯 𐑲, 𐑲 𐑕𐑱; 𐑴! 𐑜𐑫𐑛𐑯𐑩𐑕 𐑜𐑮𐑱𐑖𐑩𐑕! 𐑕𐑑𐑽 𐑒𐑤𐑽 𐑝 𐑞 𐑓𐑲𐑼𐑦 𐑐𐑦𐑑!”

𐑕𐑳𐑥𐑔𐑦𐑙 𐑝 𐑞 𐑕𐑷𐑤𐑑 𐑕𐑰 𐑘𐑧𐑑 𐑤𐑦𐑙𐑜𐑼𐑛 𐑦𐑯 𐑴𐑤𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛'𐑟 𐑤𐑨𐑙𐑜𐑢𐑦𐑡, 𐑣𐑧𐑑𐑼𐑩𐑡𐑰𐑯𐑾𐑕𐑤𐑦 𐑥𐑦𐑒𐑕𐑑 𐑢𐑦𐑞 ·𐑕𐑒𐑮𐑦𐑐𐑗𐑼𐑩𐑤 𐑯 𐑛𐑩𐑥𐑧𐑕𐑑𐑦𐑒 𐑓𐑮𐑱𐑟𐑩𐑟.

“𐑩𐑝𐑭𐑕𐑑 𐑞𐑺, 𐑩𐑝𐑭𐑕𐑑 𐑞𐑺, ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑩𐑝𐑭𐑕𐑑 𐑯𐑬 𐑕𐑐𐑶𐑤𐑦𐑙 𐑬𐑼 𐑣𐑸𐑐𐑵𐑯𐑽,” 𐑒𐑮𐑲𐑛 ·𐑐𐑰𐑤𐑧𐑜. “𐑐𐑲𐑩𐑕 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑯𐑧𐑝𐑼 𐑥𐑱𐑒 𐑜𐑫𐑛 𐑝𐑶𐑦𐑡𐑼𐑟—𐑦𐑑 𐑑𐑱𐑒𐑕 𐑞 𐑖𐑸𐑒 𐑬𐑑 𐑝 '𐑩𐑥; 𐑯𐑴 𐑣𐑸𐑐𐑵𐑯𐑽 𐑦𐑟 𐑢𐑻𐑔 𐑩 𐑕𐑑𐑮𐑷 𐑣𐑵 𐑱𐑯𐑑 𐑐𐑮𐑦𐑑𐑦 𐑖𐑸𐑒𐑦𐑖. 𐑞𐑺 𐑢𐑪𐑟 𐑘𐑳𐑙 ·𐑯𐑨𐑑 ·𐑕𐑢𐑱𐑯, 𐑢𐑳𐑯𐑕 𐑞 𐑚𐑮𐑱𐑝𐑩𐑕𐑑 𐑚𐑴𐑑-𐑣𐑧𐑛𐑼 𐑬𐑑 𐑝 𐑷𐑤 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑯 𐑞 ·𐑝𐑦𐑯𐑘𐑼𐑛; 𐑣𐑰 𐑡𐑶𐑯𐑛 𐑞 𐑥𐑰𐑑𐑦𐑙, 𐑯 𐑯𐑧𐑝𐑼 𐑒𐑱𐑥 𐑑 𐑜𐑫𐑛. 𐑣𐑰 𐑜𐑪𐑑 𐑕𐑴 𐑓𐑮𐑲𐑑𐑩𐑯𐑛 𐑩𐑚𐑬𐑑 𐑣𐑦𐑟 𐑐𐑤𐑱𐑜𐑦 𐑕𐑴𐑤, 𐑞𐑨𐑑 𐑣𐑰 𐑖𐑮𐑦𐑙𐑒𐑑 𐑯 𐑖𐑽𐑛 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑢𐑱𐑤𐑟, 𐑓 𐑓𐑽 𐑝 𐑭𐑓𐑑𐑼-𐑒𐑤𐑨𐑐𐑕, 𐑦𐑯 𐑒𐑱𐑕 𐑣𐑰 𐑜𐑪𐑑 𐑕𐑑𐑴𐑝 𐑯 𐑢𐑧𐑯𐑑 𐑑 ·𐑛𐑱𐑝𐑦 ·𐑡𐑴𐑯𐑟.”

“·𐑐𐑰𐑤𐑧𐑜! ·𐑐𐑰𐑤𐑧𐑜!” 𐑕𐑧𐑛 ·𐑚𐑦𐑤𐑛𐑨𐑛, 𐑤𐑦𐑓𐑑𐑦𐑙 𐑣𐑦𐑟 𐑲𐑟 𐑯 𐑣𐑨𐑯𐑛𐑟, “𐑞𐑬 𐑞𐑲𐑕𐑧𐑤𐑓, 𐑨𐑟 𐑲 𐑥𐑲𐑕𐑧𐑤𐑓, 𐑣𐑨𐑕𐑑 𐑕𐑰𐑯 𐑥𐑧𐑯𐑦 𐑩 𐑐𐑧𐑮𐑦𐑤𐑩𐑕 𐑑𐑲𐑥; 𐑞𐑬 𐑯𐑴𐑩𐑕𐑑, ·𐑐𐑰𐑤𐑧𐑜, 𐑢𐑪𐑑 𐑦𐑑 𐑦𐑟 𐑑 𐑣𐑨𐑝 𐑞 𐑓𐑽 𐑝 𐑛𐑧𐑔; 𐑣𐑬, 𐑞𐑧𐑯, 𐑒𐑨𐑯'𐑕𐑑 𐑞𐑬 𐑐𐑮𐑱𐑑 𐑦𐑯 𐑞𐑦𐑕 𐑳𐑯𐑜𐑪𐑛𐑤𐑦 𐑜𐑲𐑟. 𐑞𐑬 𐑚𐑧𐑤𐑾𐑕𐑑 𐑞𐑲𐑯 𐑴𐑯 𐑣𐑸𐑑, ·𐑐𐑰𐑤𐑧𐑜. 𐑑𐑧𐑤 𐑥𐑰, 𐑢𐑧𐑯 𐑞𐑦𐑕 𐑕𐑱𐑥 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑣𐑽 𐑣𐑨𐑛 𐑣𐑻 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑𐑕 𐑴𐑝𐑼𐑚𐑹𐑛 𐑦𐑯 𐑞𐑨𐑑 𐑑𐑲𐑓𐑵𐑯 𐑪𐑯 ·𐑡𐑩𐑐𐑨𐑯, 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑝𐑶𐑦𐑡 𐑢𐑧𐑯 𐑞𐑬 𐑢𐑧𐑯𐑑 𐑥𐑱𐑑 𐑢𐑦𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑛𐑲𐑛'𐑕𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑔𐑦𐑙𐑒 𐑝 ·𐑛𐑧𐑔 𐑯 𐑞 ·𐑡𐑳𐑡𐑥𐑩𐑯𐑑 𐑞𐑧𐑯?”

“𐑣𐑽 𐑣𐑦𐑥, 𐑣𐑽 𐑣𐑦𐑥 𐑯𐑬,” 𐑒𐑮𐑲𐑛 ·𐑐𐑰𐑤𐑧𐑜, 𐑥𐑸𐑗𐑦𐑙 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑯 𐑔𐑮𐑳𐑕𐑑𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛𐑟 𐑓𐑸 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑐𐑪𐑒𐑩𐑑𐑕,—“𐑣𐑽 𐑣𐑦𐑥, 𐑷𐑤 𐑝 𐑘𐑰. 𐑔𐑦𐑙𐑒 𐑝 𐑞𐑨𐑑! 𐑢𐑧𐑯 𐑧𐑝𐑮𐑦 𐑥𐑴𐑥𐑧𐑯𐑑 𐑢𐑰 𐑔𐑷𐑑 𐑞 𐑖𐑦𐑐 𐑢𐑫𐑛 𐑕𐑦𐑙𐑒! 𐑛𐑧𐑔 𐑯 𐑞 ·𐑡𐑳𐑡𐑥𐑩𐑯𐑑 𐑞𐑧𐑯? 𐑢𐑪𐑑? 𐑢𐑦𐑞 𐑷𐑤 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑𐑕 𐑥𐑱𐑒𐑦𐑙 𐑕𐑳𐑗 𐑩𐑯 𐑧𐑝𐑼𐑤𐑭𐑕𐑑𐑦𐑙 𐑔𐑳𐑯𐑛𐑼𐑦𐑙 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑕𐑲𐑛; 𐑯 𐑧𐑝𐑮𐑦 𐑕𐑰 𐑚𐑮𐑱𐑒𐑦𐑙 𐑴𐑝𐑼 𐑳𐑕, 𐑓𐑹 𐑯 𐑭𐑓𐑑. 𐑔𐑦𐑙𐑒 𐑝 ·𐑛𐑧𐑔 𐑯 𐑞 ·𐑡𐑳𐑡𐑥𐑩𐑯𐑑 𐑞𐑧𐑯? 𐑯𐑴! 𐑯𐑴 𐑑𐑲𐑥 𐑑 𐑔𐑦𐑙𐑒 𐑩𐑚𐑬𐑑 ·𐑛𐑧𐑔 𐑞𐑧𐑯. 𐑤𐑲𐑓 𐑢𐑪𐑟 𐑢𐑪𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑯 𐑲 𐑢𐑪𐑟 𐑔𐑦𐑙𐑒𐑦𐑙 𐑝; 𐑯 𐑣𐑬 𐑑 𐑕𐑱𐑝 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟—𐑣𐑬 𐑑 𐑮𐑦𐑜 𐑡𐑫𐑼𐑦-𐑥𐑭𐑕𐑑𐑕—𐑣𐑬 𐑑 𐑜𐑧𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑯𐑽𐑩𐑕𐑑 𐑐𐑹𐑑; 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑢𐑪𐑑 𐑲 𐑢𐑪𐑟 𐑔𐑦𐑙𐑒𐑦𐑙 𐑝.”

·𐑚𐑦𐑤𐑛𐑨𐑛 𐑕𐑧𐑛 𐑯𐑴 𐑥𐑹, 𐑚𐑳𐑑 𐑚𐑳𐑑𐑩𐑯𐑦𐑙 𐑳𐑐 𐑣𐑦𐑟 𐑒𐑴𐑑, 𐑕𐑑𐑷𐑒𐑑 𐑪𐑯 𐑛𐑧𐑒, 𐑢𐑺 𐑢𐑰 𐑓𐑪𐑤𐑴𐑛 𐑣𐑦𐑥. 𐑞𐑺 𐑣𐑰 𐑕𐑑𐑫𐑛, 𐑝𐑧𐑮𐑦 𐑒𐑢𐑲𐑩𐑑𐑤𐑦 𐑴𐑝𐑼𐑤𐑫𐑒𐑦𐑙 𐑕𐑳𐑥 𐑕𐑱𐑤𐑥𐑱𐑒𐑼𐑟 𐑣𐑵 𐑢𐑻 𐑥𐑧𐑯𐑛𐑦𐑙 𐑩 𐑑𐑪𐑐-𐑕𐑱𐑤 𐑦𐑯 𐑞 𐑢𐑱𐑕𐑑. 𐑯𐑬 𐑯 𐑞𐑧𐑯 𐑣𐑰 𐑕𐑑𐑵𐑐𐑑 𐑑 𐑐𐑦𐑒 𐑳𐑐 𐑩 𐑐𐑨𐑗, 𐑹 𐑕𐑱𐑝 𐑩𐑯 𐑧𐑯𐑛 𐑝 𐑑𐑸𐑛 𐑑𐑢𐑲𐑯, 𐑢𐑦𐑗 𐑳𐑞𐑼𐑢𐑲𐑟 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑢𐑱𐑕𐑑𐑩𐑛.
