“𐑥𐑨𐑯 𐑴𐑝𐑼𐑚𐑹𐑛!” 𐑒𐑮𐑲𐑛 ·𐑛𐑨𐑜𐑵, 𐑣𐑵 𐑩𐑥𐑦𐑛 𐑞 𐑡𐑧𐑯𐑼𐑩𐑤 𐑒𐑪𐑯𐑕𐑑𐑼𐑯𐑱𐑖𐑩𐑯 𐑓𐑻𐑕𐑑 𐑒𐑱𐑥 𐑑 𐑣𐑦𐑟 𐑕𐑧𐑯𐑕𐑩𐑟. “𐑕𐑢𐑦𐑙 𐑞 𐑚𐑳𐑒𐑩𐑑 𐑞𐑦𐑕 𐑢𐑱!” 𐑯 𐑐𐑫𐑑𐑦𐑙 𐑢𐑳𐑯 𐑓𐑫𐑑 𐑦𐑯𐑑𐑵 𐑦𐑑, 𐑕𐑴 𐑨𐑟 𐑞 𐑚𐑧𐑑𐑼 𐑑 𐑕𐑦𐑒𐑘𐑫𐑼 𐑣𐑦𐑟 𐑕𐑤𐑦𐑐𐑼𐑦 𐑣𐑨𐑯𐑛-𐑣𐑴𐑤𐑛 𐑪𐑯 𐑞 𐑢𐑦𐑐 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑞 𐑣𐑶𐑕𐑑𐑼𐑟 𐑮𐑨𐑯 𐑣𐑦𐑥 𐑣𐑲 𐑳𐑐 𐑑 𐑞 𐑑𐑪𐑐 𐑝 𐑞 𐑣𐑧𐑛, 𐑷𐑤𐑥𐑴𐑕𐑑 𐑚𐑦𐑓𐑹 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑒𐑫𐑛 𐑣𐑨𐑝 𐑮𐑰𐑗𐑑 𐑦𐑑𐑕 𐑦𐑯𐑑𐑽𐑽 𐑚𐑪𐑑𐑩𐑥. 𐑥𐑰𐑯𐑑𐑲𐑥, 𐑞𐑺 𐑢𐑪𐑟 𐑩 𐑑𐑧𐑮𐑦𐑚𐑩𐑤 𐑑𐑿𐑥𐑳𐑤𐑑. 𐑤𐑫𐑒𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑕𐑲𐑛, 𐑞𐑱 𐑕𐑷 𐑞 𐑚𐑦𐑓𐑹 𐑤𐑲𐑓𐑤𐑩𐑕 𐑣𐑧𐑛 𐑔𐑮𐑪𐑚𐑦𐑙 𐑯 𐑣𐑰𐑝𐑦𐑙 𐑡𐑳𐑕𐑑 𐑚𐑦𐑤𐑴 𐑞 𐑕𐑻𐑓𐑦𐑕 𐑝 𐑞 𐑕𐑰, 𐑨𐑟 𐑦𐑓 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑕𐑰𐑟𐑛 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑥𐑴𐑥𐑧𐑯𐑑𐑩𐑕 𐑲𐑛𐑾; 𐑢𐑺𐑨𐑟 𐑦𐑑 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑞 𐑐𐑫𐑼 ·𐑦𐑯𐑛𐑾𐑯 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕𐑤𐑦 𐑮𐑦𐑝𐑰𐑤𐑦𐑙 𐑚𐑲 𐑞𐑴𐑟 𐑕𐑑𐑮𐑳𐑜𐑩𐑤𐑟 𐑞 𐑐𐑧𐑮𐑦𐑤𐑩𐑕 𐑛𐑧𐑐𐑔 𐑑 𐑢𐑦𐑗 𐑣𐑰 𐑣𐑨𐑛 𐑕𐑳𐑙𐑒.

𐑨𐑑 𐑞𐑦𐑕 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑢𐑲𐑤 ·𐑛𐑨𐑜𐑵, 𐑪𐑯 𐑞 𐑕𐑳𐑥𐑦𐑑 𐑝 𐑞 𐑣𐑧𐑛, 𐑢𐑪𐑟 𐑒𐑤𐑽𐑦𐑙 𐑞 𐑢𐑦𐑐—𐑢𐑦𐑗 𐑣𐑨𐑛 𐑕𐑳𐑥𐑣𐑬 𐑜𐑪𐑑 𐑓𐑬𐑤 𐑝 𐑞 𐑜𐑮𐑱𐑑 𐑒𐑳𐑑𐑦𐑙 𐑑𐑨𐑒𐑩𐑤𐑟—𐑩 𐑖𐑸𐑐 𐑒𐑮𐑨𐑒𐑦𐑙 𐑯𐑶𐑟 𐑢𐑪𐑟 𐑣𐑻𐑛; 𐑯 𐑑 𐑞 𐑳𐑯𐑕𐑐𐑰𐑒𐑩𐑚𐑩𐑤 𐑣𐑪𐑮𐑼 𐑝 𐑷𐑤, 𐑢𐑳𐑯 𐑝 𐑞 𐑑𐑵 𐑦𐑯𐑹𐑥𐑩𐑕 𐑣𐑫𐑒𐑕 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑦𐑙 𐑞 𐑣𐑧𐑛 𐑑𐑹 𐑬𐑑, 𐑯 𐑢𐑦𐑞 𐑩 𐑝𐑭𐑕𐑑 𐑝𐑲𐑚𐑮𐑱𐑖𐑩𐑯 𐑞 𐑦𐑯𐑹𐑥𐑩𐑕 𐑥𐑨𐑕 𐑕𐑲𐑛𐑢𐑱𐑟 𐑕𐑢𐑳𐑙, 𐑑𐑦𐑤 𐑞 𐑛𐑮𐑳𐑙𐑒 𐑖𐑦𐑐 𐑮𐑰𐑤𐑛 𐑯 𐑖𐑫𐑒 𐑨𐑟 𐑦𐑓 𐑕𐑥𐑦𐑑𐑩𐑯 𐑚𐑲 𐑩𐑯 𐑲𐑕𐑚𐑻𐑜. 𐑞 𐑢𐑳𐑯 𐑮𐑦𐑥𐑱𐑯𐑦𐑙 𐑣𐑫𐑒, 𐑩𐑐𐑪𐑯 𐑢𐑦𐑗 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑕𐑑𐑮𐑱𐑯 𐑯𐑬 𐑛𐑦𐑐𐑧𐑯𐑛𐑩𐑛, 𐑕𐑰𐑥𐑛 𐑧𐑝𐑮𐑦 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑑 𐑚𐑰 𐑪𐑯 𐑞 𐑐𐑶𐑯𐑑 𐑝 𐑜𐑦𐑝𐑦𐑙 𐑢𐑱; 𐑩𐑯 𐑦𐑝𐑧𐑯𐑑 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑤𐑲𐑒𐑤𐑦 𐑓𐑮𐑪𐑥 𐑞 𐑝𐑲𐑩𐑤𐑩𐑯𐑑 𐑥𐑴𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑣𐑧𐑛.

“𐑒𐑳𐑥 𐑛𐑬𐑯, 𐑒𐑳𐑥 𐑛𐑬𐑯!” 𐑘𐑧𐑤𐑛 𐑞 𐑕𐑰𐑥𐑧𐑯 𐑑 ·𐑛𐑨𐑜𐑵, 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑣𐑴𐑤𐑛𐑦𐑙 𐑪𐑯 𐑑 𐑞 𐑣𐑧𐑝𐑦 𐑑𐑨𐑒𐑩𐑤𐑟, 𐑕𐑴 𐑞𐑨𐑑 𐑦𐑓 𐑞 𐑣𐑧𐑛 𐑖𐑵𐑛 𐑛𐑮𐑪𐑐, 𐑣𐑰 𐑢𐑫𐑛 𐑕𐑑𐑦𐑤 𐑮𐑦𐑥𐑱𐑯 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛; 𐑞 𐑯𐑰𐑜𐑮𐑴 𐑣𐑨𐑝𐑦𐑙 𐑒𐑤𐑽𐑛 𐑞 𐑓𐑬𐑤 𐑤𐑲𐑯, 𐑮𐑨𐑥𐑛 𐑛𐑬𐑯 𐑞 𐑚𐑳𐑒𐑩𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑯𐑬 𐑒𐑩𐑤𐑨𐑐𐑕𐑑 𐑢𐑧𐑤, 𐑥𐑰𐑯𐑦𐑙 𐑞𐑨𐑑 𐑞 𐑚𐑧𐑮𐑦𐑛 𐑣𐑸𐑐𐑵𐑯𐑽 𐑖𐑵𐑛 𐑜𐑮𐑭𐑕𐑐 𐑦𐑑, 𐑯 𐑕𐑴 𐑚𐑰 𐑣𐑶𐑕𐑑𐑩𐑛 𐑬𐑑.

“𐑦𐑯 𐑣𐑧𐑝𐑩𐑯'𐑟 𐑯𐑱𐑥, 𐑥𐑨𐑯,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, “𐑸 𐑿 𐑮𐑨𐑥𐑦𐑙 𐑣𐑴𐑥 𐑩 𐑒𐑸𐑑𐑮𐑦𐑡 𐑞𐑺?—𐑩𐑝𐑭𐑕𐑑! 𐑣𐑬 𐑢𐑦𐑤 𐑞𐑨𐑑 𐑣𐑧𐑤𐑐 𐑣𐑦𐑥; 𐑡𐑨𐑥𐑦𐑙 𐑞𐑨𐑑 𐑲𐑼𐑯-𐑚𐑬𐑯𐑛 𐑚𐑳𐑒𐑩𐑑 𐑪𐑯 𐑑𐑪𐑐 𐑝 𐑣𐑦𐑟 𐑣𐑧𐑛? 𐑩𐑝𐑭𐑕𐑑, 𐑢𐑦𐑤 𐑘𐑰!”

“𐑕𐑑𐑨𐑯𐑛 𐑒𐑤𐑽 𐑝 𐑞 𐑑𐑨𐑒𐑩𐑤!” 𐑒𐑮𐑲𐑛 𐑩 𐑝𐑶𐑕 𐑤𐑲𐑒 𐑞 𐑚𐑻𐑕𐑑𐑦𐑙 𐑝 𐑩 𐑮𐑪𐑒𐑩𐑑.

𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑢𐑦𐑞 𐑩 𐑔𐑳𐑯𐑛𐑼-𐑚𐑵𐑥, 𐑞 𐑦𐑯𐑹𐑥𐑩𐑕 𐑥𐑨𐑕 𐑛𐑮𐑪𐑐𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰, 𐑤𐑲𐑒 ·𐑯𐑲𐑨𐑜𐑼𐑩'𐑟 ·𐑑𐑱𐑚𐑩𐑤-·𐑮𐑪𐑒 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑻𐑤𐑐𐑵𐑤; 𐑞 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑮𐑦𐑤𐑰𐑝𐑛 𐑣𐑳𐑤 𐑮𐑴𐑤𐑛 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑦𐑑, 𐑑 𐑓𐑸 𐑛𐑬𐑯 𐑣𐑻 𐑜𐑤𐑦𐑑𐑼𐑦𐑙 𐑒𐑪𐑐𐑼; 𐑯 𐑷𐑤 𐑒𐑷𐑑 𐑞𐑺 𐑚𐑮𐑧𐑔, 𐑨𐑟 𐑣𐑭𐑓 𐑕𐑢𐑦𐑙𐑦𐑙—𐑯𐑬 𐑴𐑝𐑼 𐑞 𐑕𐑱𐑤𐑼𐑟' 𐑣𐑧𐑛𐑟, 𐑯 𐑯𐑬 𐑴𐑝𐑼 𐑞 𐑢𐑷𐑑𐑼—·𐑛𐑨𐑜𐑵, 𐑔𐑮𐑵 𐑩 𐑔𐑦𐑒 𐑥𐑦𐑕𐑑 𐑝 𐑕𐑐𐑮𐑱, 𐑢𐑪𐑟 𐑛𐑦𐑥𐑤𐑦 𐑚𐑦𐑣𐑧𐑤𐑛 𐑒𐑤𐑦𐑙𐑦𐑙 𐑑 𐑞 𐑐𐑧𐑯𐑡𐑩𐑤𐑩𐑕 𐑑𐑨𐑒𐑩𐑤𐑟, 𐑢𐑲𐑤 𐑐𐑫𐑼, 𐑚𐑧𐑮𐑦𐑛-𐑩𐑤𐑲𐑝 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑢𐑪𐑟 𐑕𐑦𐑙𐑒𐑦𐑙 𐑳𐑑𐑼𐑤𐑦 𐑛𐑬𐑯 𐑑 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 𐑞 𐑕𐑰! 𐑚𐑳𐑑 𐑣𐑸𐑛𐑤𐑦 𐑣𐑨𐑛 𐑞 𐑚𐑤𐑲𐑯𐑛𐑦𐑙 𐑝𐑱𐑐𐑼 𐑒𐑤𐑽𐑛 𐑩𐑢𐑱, 𐑢𐑧𐑯 𐑩 𐑯𐑱𐑒𐑩𐑛 𐑓𐑦𐑜𐑼 𐑢𐑦𐑞 𐑩 𐑚𐑹𐑛𐑦𐑙-𐑕𐑹𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛, 𐑢𐑪𐑟 𐑓 𐑢𐑳𐑯 𐑕𐑢𐑦𐑓𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑕𐑰𐑯 𐑣𐑪𐑝𐑼𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕. 𐑞 𐑯𐑧𐑒𐑕𐑑, 𐑩 𐑤𐑬𐑛 𐑕𐑐𐑤𐑨𐑖 𐑩𐑯𐑬𐑯𐑕𐑑 𐑞𐑨𐑑 𐑥𐑲 𐑚𐑮𐑱𐑝 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑣𐑨𐑛 𐑛𐑲𐑝𐑛 𐑑 𐑞 𐑮𐑧𐑕𐑒𐑿. 𐑢𐑳𐑯 𐑐𐑨𐑒𐑑 𐑮𐑳𐑖 𐑢𐑪𐑟 𐑥𐑱𐑛 𐑑 𐑞 𐑕𐑲𐑛, 𐑯 𐑧𐑝𐑮𐑦 𐑲 𐑒𐑬𐑯𐑑𐑩𐑛 𐑧𐑝𐑮𐑦 𐑮𐑦𐑐𐑩𐑤, 𐑨𐑟 𐑥𐑴𐑥𐑧𐑯𐑑 𐑓𐑪𐑤𐑴𐑛 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑯 𐑯𐑴 𐑕𐑲𐑯 𐑝 𐑲𐑞𐑼 𐑞 𐑕𐑦𐑙𐑒𐑼 𐑹 𐑞 𐑛𐑲𐑝𐑼 𐑒𐑫𐑛 𐑚𐑰 𐑕𐑰𐑯. 𐑕𐑳𐑥 𐑣𐑨𐑯𐑛𐑟 𐑯𐑬 𐑡𐑳𐑥𐑐𐑑 𐑦𐑯𐑑𐑵 𐑩 𐑚𐑴𐑑 𐑩𐑤𐑪𐑙𐑕𐑲𐑛, 𐑯 𐑐𐑫𐑖𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐.

“𐑣𐑭! 𐑣𐑭!” 𐑒𐑮𐑲𐑛 ·𐑛𐑨𐑜𐑵, 𐑷𐑤 𐑨𐑑 𐑢𐑳𐑯𐑕, 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑯𐑬 𐑒𐑢𐑲𐑩𐑑, 𐑕𐑢𐑦𐑙𐑦𐑙 𐑐𐑻𐑗 𐑴𐑝𐑼𐑣𐑧𐑛; 𐑯 𐑤𐑫𐑒𐑦𐑙 𐑓𐑻𐑞𐑼 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑲𐑛, 𐑢𐑰 𐑕𐑷 𐑩𐑯 𐑸𐑥 𐑔𐑮𐑳𐑕𐑑 𐑳𐑐𐑮𐑲𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑤𐑵 𐑢𐑱𐑝𐑟; 𐑩 𐑕𐑲𐑑 𐑕𐑑𐑮𐑱𐑯𐑡 𐑑 𐑕𐑰, 𐑨𐑟 𐑩𐑯 𐑸𐑥 𐑔𐑮𐑳𐑕𐑑 𐑓𐑹𐑔 𐑓𐑮𐑪𐑥 𐑞 𐑜𐑮𐑭𐑕 𐑴𐑝𐑼 𐑩 𐑜𐑮𐑱𐑝.

“𐑚𐑴𐑔! 𐑚𐑴𐑔!—𐑦𐑑 𐑦𐑟 𐑚𐑴𐑔!”—𐑒𐑮𐑲𐑛 ·𐑛𐑨𐑜𐑵 𐑩𐑜𐑱𐑯 𐑢𐑦𐑞 𐑩 𐑡𐑶𐑓𐑩𐑤 𐑖𐑬𐑑; 𐑯 𐑕𐑵𐑯 𐑭𐑓𐑑𐑼, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑢𐑪𐑟 𐑕𐑰𐑯 𐑚𐑴𐑤𐑛𐑤𐑦 𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑬𐑑 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛, 𐑯 𐑢𐑦𐑞 𐑞 𐑳𐑞𐑼 𐑒𐑤𐑳𐑗𐑦𐑙 𐑞 𐑤𐑪𐑙 𐑣𐑺 𐑝 𐑞 ·𐑦𐑯𐑛𐑾𐑯. 𐑛𐑮𐑷𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑱𐑑𐑦𐑙 𐑚𐑴𐑑, 𐑞𐑱 𐑢𐑻 𐑒𐑢𐑦𐑒𐑤𐑦 𐑚𐑮𐑷𐑑 𐑑 𐑞 𐑛𐑧𐑒; 𐑚𐑳𐑑 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑢𐑪𐑟 𐑤𐑪𐑙 𐑦𐑯 𐑒𐑳𐑥𐑦𐑙 𐑑, 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑤𐑫𐑒 𐑝𐑧𐑮𐑦 𐑚𐑮𐑦𐑕𐑒.
