𐑞 𐑥𐑹𐑯𐑦𐑙 𐑝 𐑞 𐑔𐑻𐑛 𐑛𐑱 𐑛𐑷𐑯𐑛 𐑓𐑺 𐑯 𐑓𐑮𐑧𐑖, 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑞 𐑕𐑪𐑤𐑦𐑑𐑼𐑦 𐑯𐑲𐑑-𐑥𐑨𐑯 𐑨𐑑 𐑞 𐑓𐑹-𐑥𐑭𐑕𐑑-𐑣𐑧𐑛 𐑢𐑪𐑟 𐑮𐑦𐑤𐑰𐑝𐑛 𐑚𐑲 𐑒𐑮𐑬𐑛𐑟 𐑝 𐑞 𐑛𐑱𐑤𐑲𐑑 𐑤𐑫𐑒-𐑬𐑑𐑕, 𐑣𐑵 𐑛𐑪𐑑𐑩𐑛 𐑧𐑝𐑮𐑦 𐑥𐑭𐑕𐑑 𐑯 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑝𐑮𐑦 𐑕𐑐𐑸.

“·𐑛'𐑘𐑰 𐑕𐑰 𐑣𐑦𐑥?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚; 𐑚𐑳𐑑 𐑞 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑘𐑧𐑑 𐑦𐑯 𐑕𐑲𐑑.

𐑦𐑯 𐑧𐑝𐑮𐑦 𐑓𐑲𐑑 𐑦𐑑 𐑣𐑨𐑟 𐑞 𐑤𐑭𐑕𐑑 𐑯 𐑚𐑦𐑑𐑼𐑩𐑕𐑑 𐑚𐑤𐑴. 𐑮𐑳𐑯 𐑑𐑦𐑤𐑑𐑦𐑙 𐑨𐑑 𐑦𐑑, 𐑯 𐑿 𐑚𐑳𐑑 𐑮𐑳𐑯 𐑔𐑮𐑵 𐑦𐑑. 𐑣𐑭! 𐑩 𐑒𐑬𐑼𐑛 𐑢𐑦𐑯𐑛 𐑞𐑨𐑑 𐑕𐑑𐑮𐑲𐑒𐑕 𐑕𐑑𐑸𐑒 𐑯𐑱𐑒𐑩𐑛 𐑥𐑧𐑯, 𐑚𐑳𐑑 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑕𐑑𐑨𐑯𐑛 𐑑 𐑮𐑦𐑕𐑰𐑝 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑚𐑤𐑴. 𐑰𐑝𐑩𐑯 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑩 𐑚𐑮𐑱𐑝𐑼 𐑔𐑦𐑙—𐑩 𐑯𐑴𐑚𐑩𐑤𐑼 𐑔𐑦𐑙 𐑞𐑨𐑯 𐑞𐑨𐑑. 𐑢𐑫𐑛 𐑯𐑬 𐑞 𐑢𐑦𐑯𐑛 𐑚𐑳𐑑 𐑣𐑨𐑛 𐑩 𐑚𐑪𐑛𐑦; 𐑚𐑳𐑑 𐑷𐑤 𐑞 𐑔𐑦𐑙𐑟 𐑞𐑨𐑑 𐑥𐑴𐑕𐑑 𐑦𐑜𐑟𐑨𐑕𐑐𐑼𐑱𐑑 𐑯 𐑬𐑑𐑮𐑱𐑡 𐑥𐑹𐑑𐑩𐑤 𐑥𐑨𐑯, 𐑷𐑤 𐑞𐑰𐑟 𐑔𐑦𐑙𐑟 𐑸 𐑚𐑪𐑛𐑦𐑤𐑩𐑕, 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑚𐑪𐑛𐑦𐑤𐑩𐑕 𐑨𐑟 𐑪𐑚𐑡𐑧𐑒𐑑𐑕, 𐑯𐑪𐑑 𐑨𐑟 𐑱𐑡𐑩𐑯𐑑𐑕. 𐑞𐑺'𐑟 𐑩 𐑥𐑴𐑕𐑑 𐑕𐑐𐑧𐑖𐑩𐑤, 𐑩 𐑥𐑴𐑕𐑑 𐑒𐑳𐑯𐑦𐑙, 𐑴, 𐑩 𐑥𐑴𐑕𐑑 𐑥𐑩𐑤𐑦𐑖𐑩𐑕 𐑛𐑦𐑓𐑼𐑩𐑯𐑕! 𐑯 𐑘𐑧𐑑, 𐑲 𐑕𐑱 𐑩𐑜𐑱𐑯, 𐑯 𐑕𐑢𐑺 𐑦𐑑 𐑯𐑬, 𐑞𐑨𐑑 𐑞𐑺'𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑷𐑤 𐑜𐑤𐑹𐑾𐑕 𐑯 𐑜𐑮𐑱𐑖𐑩𐑕 𐑦𐑯 𐑞 𐑢𐑦𐑯𐑛. 𐑞𐑰𐑟 𐑢𐑹𐑥 ·𐑑𐑮𐑱𐑛 ·𐑢𐑦𐑯𐑛𐑟, 𐑨𐑑 𐑤𐑰𐑕𐑑, 𐑞𐑨𐑑 𐑦𐑯 𐑞 𐑒𐑤𐑽 𐑣𐑧𐑝𐑩𐑯𐑟 𐑚𐑤𐑴 𐑕𐑑𐑮𐑱𐑑 𐑪𐑯, 𐑦𐑯 𐑕𐑑𐑮𐑪𐑙 𐑯 𐑕𐑑𐑧𐑛𐑓𐑭𐑕𐑑, 𐑝𐑦𐑜𐑼𐑩𐑕 𐑥𐑲𐑤𐑛𐑯𐑩𐑕; 𐑯 𐑝𐑽 𐑯𐑪𐑑 𐑓𐑮𐑪𐑥 𐑞𐑺 𐑥𐑸𐑒, 𐑣𐑬𐑧𐑝𐑼 𐑞 𐑚𐑱𐑕𐑼 𐑒𐑳𐑮𐑩𐑯𐑑𐑕 𐑝 𐑞 𐑕𐑰 𐑥𐑱 𐑑𐑻𐑯 𐑯 𐑑𐑨𐑒, 𐑯 𐑥𐑲𐑑𐑾𐑕𐑑 ·𐑥𐑦𐑕𐑦𐑕𐑦𐑐𐑦𐑟 𐑝 𐑞 𐑤𐑨𐑯𐑛 𐑕𐑢𐑦𐑓𐑑 𐑯 𐑕𐑢𐑻𐑝 𐑩𐑚𐑬𐑑, 𐑳𐑯𐑕𐑻𐑑𐑩𐑯 𐑢𐑺 𐑑 𐑜𐑴 𐑨𐑑 𐑤𐑭𐑕𐑑. 𐑯 𐑚𐑲 𐑞 𐑦𐑑𐑻𐑯𐑩𐑤 ·𐑐𐑴𐑤𐑟! 𐑞𐑰𐑟 𐑕𐑱𐑥 ·𐑑𐑮𐑱𐑛𐑟 𐑞𐑨𐑑 𐑕𐑴 𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑚𐑤𐑴 𐑥𐑲 𐑜𐑫𐑛 𐑖𐑦𐑐 𐑪𐑯; 𐑞𐑰𐑟 ·𐑑𐑮𐑱𐑛𐑟, 𐑹 𐑕𐑳𐑥𐑔𐑦𐑙 𐑤𐑲𐑒 𐑞𐑧𐑥—𐑕𐑳𐑥𐑔𐑦𐑙 𐑕𐑴 𐑳𐑯𐑗𐑱𐑯𐑡𐑩𐑚𐑩𐑤, 𐑯 𐑓𐑫𐑤 𐑨𐑟 𐑕𐑑𐑮𐑪𐑙, 𐑚𐑤𐑴 𐑥𐑲 𐑒𐑰𐑤𐑛 𐑕𐑴𐑤 𐑩𐑤𐑪𐑙! 𐑑 𐑦𐑑! 𐑩𐑤𐑪𐑓𐑑 𐑞𐑺! 𐑢𐑪𐑑 𐑛'𐑘𐑰 𐑕𐑰?”

“𐑯𐑳𐑔𐑦𐑙! 𐑯 𐑯𐑵𐑯 𐑨𐑑 𐑣𐑨𐑯𐑛! 𐑞 𐑛𐑳𐑚𐑤𐑵𐑯 𐑜𐑴𐑟 𐑩-𐑚𐑧𐑜𐑦𐑙! 𐑕𐑰 𐑞 𐑕𐑳𐑯! 𐑲, 𐑲, 𐑦𐑑 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑕𐑴. 𐑲'𐑝 𐑴𐑝𐑼𐑕𐑱𐑤𐑛 𐑣𐑦𐑥. 𐑣𐑬, 𐑜𐑪𐑑 𐑞 𐑕𐑑𐑸𐑑? 𐑲, 𐑣𐑰'𐑟 𐑗𐑱𐑕𐑦𐑙 𐑥𐑰 𐑯𐑬; 𐑯𐑪𐑑 𐑲, 𐑣𐑦𐑥—𐑞𐑨𐑑'𐑕 𐑚𐑨𐑛; 𐑲 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑯𐑴𐑯 𐑦𐑑, 𐑑𐑵. 𐑓𐑵𐑤! 𐑞 𐑤𐑲𐑯𐑟—𐑞 𐑣𐑸𐑐𐑵𐑯𐑟 𐑣𐑰'𐑟 𐑑𐑴𐑦𐑙. 𐑲, 𐑲, 𐑲 𐑣𐑨𐑝 𐑮𐑳𐑯 𐑣𐑦𐑥 𐑚𐑲 𐑤𐑭𐑕𐑑 𐑯𐑲𐑑. 𐑩𐑚𐑬𐑑! 𐑩𐑚𐑬𐑑! 𐑒𐑳𐑥 𐑛𐑬𐑯, 𐑷𐑤 𐑝 𐑘𐑰, 𐑚𐑳𐑑 𐑞 𐑮𐑧𐑜𐑿𐑤𐑼 𐑤𐑫𐑒 𐑬𐑑𐑕! 𐑥𐑨𐑯 𐑞 𐑚𐑮𐑱𐑕𐑩𐑟!”

𐑕𐑑𐑽𐑦𐑙 𐑨𐑟 𐑖𐑰 𐑣𐑨𐑛 𐑛𐑳𐑯, 𐑞 𐑢𐑦𐑯𐑛 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑳𐑥𐑢𐑪𐑑 𐑪𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑒𐑢𐑹𐑑𐑼, 𐑕𐑴 𐑞𐑨𐑑 𐑯𐑬 𐑚𐑰𐑦𐑙 𐑐𐑶𐑯𐑑𐑩𐑛 𐑦𐑯 𐑞 𐑮𐑦𐑝𐑻𐑕 𐑛𐑦𐑮𐑧𐑒𐑖𐑩𐑯, 𐑞 𐑚𐑮𐑱𐑕𐑑 𐑖𐑦𐑐 𐑕𐑱𐑤𐑛 𐑣𐑸𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑮𐑰𐑟 𐑨𐑟 𐑖𐑰 𐑮𐑰𐑗𐑻𐑯𐑛 𐑞 𐑒𐑮𐑰𐑥 𐑦𐑯 𐑣𐑻 𐑴𐑯 𐑢𐑲𐑑 𐑢𐑱𐑒.

“𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑢𐑦𐑯𐑛 𐑣𐑰 𐑯𐑬 𐑕𐑑𐑽𐑟 𐑓 𐑞 𐑴𐑐𐑩𐑯 𐑡𐑷,” 𐑥𐑻𐑥𐑼𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑨𐑟 𐑣𐑰 𐑒𐑶𐑤𐑛 𐑞 𐑯𐑿-𐑣𐑷𐑤𐑛 𐑥𐑱𐑯-𐑚𐑮𐑱𐑕 𐑩𐑐𐑪𐑯 𐑞 𐑮𐑱𐑤. “𐑜𐑪𐑛 𐑒𐑰𐑐 𐑳𐑕, 𐑚𐑳𐑑 𐑷𐑤𐑮𐑧𐑛𐑦 𐑥𐑲 𐑚𐑴𐑯𐑟 𐑓𐑰𐑤 𐑛𐑨𐑥𐑐 𐑢𐑦𐑞𐑦𐑯 𐑥𐑰, 𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑦𐑯𐑕𐑲𐑛 𐑢𐑧𐑑 𐑥𐑲 𐑓𐑤𐑧𐑖. 𐑲 𐑥𐑦𐑕𐑛𐑬𐑑 𐑥𐑰 𐑞𐑨𐑑 𐑲 𐑛𐑦𐑕𐑴𐑚𐑱 𐑥𐑲 ·𐑜𐑪𐑛 𐑦𐑯 𐑴𐑚𐑱𐑦𐑙 𐑣𐑦𐑥!”

“𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑕𐑢𐑱 𐑥𐑰 𐑳𐑐!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑑 𐑞 𐑣𐑧𐑥𐑐𐑩𐑯 𐑚𐑭𐑕𐑒𐑩𐑑. “𐑢𐑰 𐑖𐑵𐑛 𐑥𐑰𐑑 𐑣𐑦𐑥 𐑕𐑵𐑯.”

“𐑲, 𐑲, 𐑕𐑻,” 𐑯 𐑕𐑑𐑮𐑱𐑑𐑢𐑱 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑛𐑦𐑛 ·𐑱𐑣𐑨𐑚'𐑟 𐑚𐑦𐑛𐑦𐑙, 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 ·𐑱𐑣𐑨𐑚 𐑕𐑢𐑳𐑙 𐑪𐑯 𐑣𐑲.

𐑩 𐑣𐑴𐑤 𐑬𐑼 𐑯𐑬 𐑐𐑭𐑕𐑑; 𐑜𐑴𐑤𐑛-𐑚𐑰𐑑𐑩𐑯 𐑬𐑑 𐑑 𐑱𐑡𐑩𐑟. 𐑑𐑲𐑥 𐑦𐑑𐑕𐑧𐑤𐑓 𐑯𐑬 𐑣𐑧𐑤𐑛 𐑤𐑪𐑙 𐑚𐑮𐑧𐑔𐑕 𐑢𐑦𐑞 𐑒𐑰𐑯 𐑕𐑩𐑕𐑐𐑧𐑯𐑕. 𐑚𐑳𐑑 𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑕𐑳𐑥 𐑔𐑮𐑰 𐑐𐑶𐑯𐑑𐑕 𐑪𐑓 𐑞 𐑢𐑧𐑞𐑼 𐑚𐑬, ·𐑱𐑣𐑨𐑚 𐑛𐑦𐑕𐑒𐑮𐑲𐑛 𐑞 𐑕𐑐𐑬𐑑 𐑩𐑜𐑱𐑯, 𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑓𐑮𐑪𐑥 𐑞 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑔𐑮𐑰 𐑖𐑮𐑰𐑒𐑕 𐑢𐑧𐑯𐑑 𐑳𐑐 𐑨𐑟 𐑦𐑓 𐑞 𐑑𐑳𐑙𐑟 𐑝 𐑓𐑲𐑼 𐑣𐑨𐑛 𐑝𐑶𐑕𐑑 𐑦𐑑.

“𐑓𐑪𐑮𐑦𐑛 𐑑 𐑓𐑪𐑮𐑦𐑛 𐑲 𐑥𐑰𐑑 𐑞𐑰, 𐑞𐑦𐑕 𐑔𐑻𐑛 𐑑𐑲𐑥, ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒! 𐑪𐑯 𐑛𐑧𐑒 𐑞𐑺!—𐑚𐑮𐑱𐑕 𐑖𐑸𐑐𐑼 𐑳𐑐; 𐑒𐑮𐑬𐑛 𐑣𐑻 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑦𐑯𐑛'𐑟 𐑲. 𐑣𐑰'𐑟 𐑑𐑵 𐑓𐑸 𐑪𐑓 𐑑 𐑤𐑴𐑼 𐑘𐑧𐑑, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒. 𐑞 𐑕𐑱𐑤𐑟 𐑖𐑱𐑒! 𐑕𐑑𐑨𐑯𐑛 𐑴𐑝𐑼 𐑞𐑨𐑑 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯 𐑢𐑦𐑞 𐑩 𐑑𐑪𐑐-𐑥𐑷𐑤! 𐑕𐑴, 𐑕𐑴; 𐑣𐑰 𐑑𐑮𐑨𐑝𐑩𐑤𐑟 𐑓𐑭𐑕𐑑, 𐑯 𐑲 𐑥𐑳𐑕𐑑 𐑛𐑬𐑯. 𐑚𐑳𐑑 𐑤𐑧𐑑 𐑥𐑰 𐑣𐑨𐑝 𐑢𐑳𐑯 𐑥𐑹 𐑜𐑫𐑛 𐑮𐑬𐑯𐑛 𐑤𐑫𐑒 𐑩𐑤𐑪𐑓𐑑 𐑣𐑽 𐑨𐑑 𐑞 𐑕𐑰; 𐑞𐑺'𐑟 𐑑𐑲𐑥 𐑓 𐑞𐑨𐑑. 𐑩𐑯 𐑴𐑤𐑛, 𐑴𐑤𐑛 𐑕𐑲𐑑, 𐑯 𐑘𐑧𐑑 𐑕𐑳𐑥𐑣𐑬 𐑕𐑴 𐑘𐑳𐑙; 𐑲, 𐑯 𐑯𐑪𐑑 𐑗𐑱𐑯𐑡𐑛 𐑩 𐑢𐑦𐑙𐑒 𐑕𐑦𐑯𐑕 𐑲 𐑓𐑻𐑕𐑑 𐑕𐑷 𐑦𐑑, 𐑩 𐑚𐑶, 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑨𐑯𐑛-𐑣𐑦𐑤𐑟 𐑝 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑! 𐑞 𐑕𐑱𐑥!—𐑞 𐑕𐑱𐑥!—𐑞 𐑕𐑱𐑥 𐑑 ·𐑯𐑴𐑩 𐑨𐑟 𐑑 𐑥𐑰. 𐑞𐑺'𐑟 𐑩 𐑕𐑪𐑓𐑑 𐑖𐑬𐑼 𐑑 𐑤𐑰𐑢𐑼𐑛. 𐑕𐑳𐑗 𐑤𐑳𐑝𐑤𐑦 𐑤𐑰𐑢𐑼𐑛𐑦𐑙𐑟! 𐑞𐑱 𐑥𐑳𐑕𐑑 𐑤𐑰𐑛 𐑕𐑳𐑥𐑢𐑺—𐑑 𐑕𐑳𐑥𐑔𐑦𐑙 𐑧𐑤𐑕 𐑞𐑨𐑯 𐑒𐑪𐑥𐑩𐑯 𐑤𐑨𐑯𐑛, 𐑥𐑹 𐑐𐑭𐑥𐑦 𐑞𐑨𐑯 𐑞 𐑐𐑭𐑥𐑟. 𐑤𐑰𐑢𐑼𐑛! 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑜𐑴𐑟 𐑞𐑨𐑑 𐑢𐑱; 𐑤𐑫𐑒 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛, 𐑞𐑧𐑯; 𐑞 𐑚𐑧𐑑𐑼 𐑦𐑓 𐑞 𐑚𐑦𐑑𐑼𐑼 𐑒𐑢𐑹𐑑𐑼. 𐑚𐑳𐑑 𐑜𐑫𐑛 𐑚𐑲, 𐑜𐑫𐑛 𐑚𐑲, 𐑴𐑤𐑛 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛! 𐑢𐑪𐑑'𐑕 𐑞𐑦𐑕?—𐑜𐑮𐑰𐑯? 𐑲, 𐑑𐑲𐑯𐑦 𐑥𐑪𐑕𐑩𐑟 𐑦𐑯 𐑞𐑰𐑟 𐑢𐑹𐑐𐑑 𐑒𐑮𐑨𐑒𐑕. 𐑯𐑴 𐑕𐑳𐑗 𐑜𐑮𐑰𐑯 𐑢𐑧𐑞𐑼 𐑕𐑑𐑱𐑯𐑟 𐑪𐑯 ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑧𐑛! 𐑞𐑺'𐑟 𐑞 𐑛𐑦𐑓𐑼𐑩𐑯𐑕 𐑯𐑬 𐑚𐑦𐑑𐑢𐑰𐑯 𐑥𐑨𐑯'𐑟 𐑴𐑤𐑛 𐑱𐑡 𐑯 𐑥𐑨𐑑𐑼'𐑟. 𐑚𐑳𐑑 𐑲, 𐑴𐑤𐑛 𐑥𐑭𐑕𐑑, 𐑢𐑰 𐑚𐑴𐑔 𐑜𐑮𐑴 𐑴𐑤𐑛 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑕𐑬𐑯𐑛 𐑦𐑯 𐑬𐑼 𐑣𐑳𐑤𐑟, 𐑞𐑴, 𐑸 𐑢𐑰 𐑯𐑪𐑑, 𐑥𐑲 𐑖𐑦𐑐? 𐑲, 𐑥𐑲𐑯𐑩𐑕 𐑩 𐑤𐑧𐑜, 𐑞𐑨𐑑'𐑕 𐑷𐑤. 𐑚𐑲 𐑣𐑧𐑝𐑩𐑯 𐑞𐑦𐑕 𐑛𐑧𐑛 𐑢𐑫𐑛 𐑣𐑨𐑟 𐑞 𐑚𐑧𐑑𐑼 𐑝 𐑥𐑲 𐑤𐑲𐑝 𐑓𐑤𐑧𐑖 𐑧𐑝𐑮𐑦 𐑢𐑱. 𐑲 𐑒𐑭𐑯𐑑 𐑒𐑩𐑥𐑐𐑺 𐑢𐑦𐑞 𐑦𐑑; 𐑯 𐑲'𐑝 𐑯𐑴𐑯 𐑕𐑳𐑥 𐑖𐑦𐑐𐑕 𐑥𐑱𐑛 𐑝 𐑛𐑧𐑛 𐑑𐑮𐑰𐑟 𐑬𐑑𐑤𐑭𐑕𐑑 𐑞 𐑤𐑲𐑝𐑟 𐑝 𐑥𐑧𐑯 𐑥𐑱𐑛 𐑝 𐑞 𐑥𐑴𐑕𐑑 𐑝𐑲𐑑𐑩𐑤 𐑕𐑑𐑳𐑓 𐑝 𐑝𐑲𐑑𐑩𐑤 𐑓𐑭𐑞𐑼𐑟. 𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑣𐑰 𐑕𐑧𐑛? 𐑣𐑰 𐑖𐑵𐑛 𐑕𐑑𐑦𐑤 𐑜𐑴 𐑚𐑦𐑓𐑹 𐑥𐑰, 𐑥𐑲 𐑐𐑲𐑤𐑩𐑑; 𐑯 𐑘𐑧𐑑 𐑑 𐑚𐑰 𐑕𐑰𐑯 𐑩𐑜𐑱𐑯? 𐑚𐑳𐑑 𐑢𐑺? 𐑢𐑦𐑤 𐑲 𐑣𐑨𐑝 𐑲𐑟 𐑨𐑑 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 𐑞 𐑕𐑰, 𐑕𐑩𐑐𐑴𐑟𐑦𐑙 𐑲 𐑛𐑦𐑕𐑧𐑯𐑛 𐑞𐑴𐑟 𐑧𐑯𐑛𐑤𐑩𐑕 𐑕𐑑𐑺𐑟? 𐑯 𐑷𐑤 𐑯𐑲𐑑 𐑲'𐑝 𐑚𐑰𐑯 𐑕𐑱𐑤𐑦𐑙 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑢𐑺𐑧𐑝𐑼 𐑣𐑰 𐑛𐑦𐑛 𐑕𐑦𐑙𐑒 𐑑. 𐑲, 𐑲, 𐑤𐑲𐑒 𐑥𐑧𐑯𐑦 𐑥𐑹 𐑞𐑬 𐑑𐑴𐑤𐑛'𐑕𐑑 𐑛𐑲𐑼𐑓𐑩𐑤 𐑑𐑮𐑵𐑔 𐑨𐑟 𐑑𐑳𐑗𐑦𐑙 𐑞𐑲𐑕𐑧𐑤𐑓, 𐑴 ·𐑐𐑸𐑟𐑰; 𐑚𐑳𐑑, ·𐑱𐑣𐑨𐑚, 𐑞𐑺 𐑞𐑲 𐑖𐑪𐑑 𐑓𐑧𐑤 𐑖𐑹𐑑. 𐑜𐑫𐑛-𐑚𐑲, 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛—𐑒𐑰𐑐 𐑩 𐑜𐑫𐑛 𐑲 𐑩𐑐𐑪𐑯 𐑞 𐑢𐑱𐑤, 𐑞 𐑢𐑲𐑤 𐑲'𐑥 𐑜𐑪𐑯. 𐑢𐑰'𐑤 𐑑𐑷𐑒 𐑑-𐑥𐑪𐑮𐑴, 𐑯𐑱, 𐑑-𐑯𐑲𐑑, 𐑢𐑧𐑯 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑤𐑲𐑟 𐑛𐑬𐑯 𐑞𐑺, 𐑑𐑲𐑛 𐑚𐑲 𐑣𐑧𐑛 𐑯 𐑑𐑱𐑤.”

𐑣𐑰 𐑜𐑱𐑝 𐑞 𐑢𐑻𐑛; 𐑯 𐑕𐑑𐑦𐑤 𐑜𐑱𐑟𐑦𐑙 𐑮𐑬𐑯𐑛 𐑣𐑦𐑥, 𐑢𐑪𐑟 𐑕𐑑𐑧𐑛𐑦𐑤𐑦 𐑤𐑴𐑼𐑛 𐑔𐑮𐑵 𐑞 𐑒𐑤𐑴𐑝𐑩𐑯 𐑚𐑤𐑵 𐑺 𐑑 𐑞 𐑛𐑧𐑒.

𐑦𐑯 𐑛𐑿 𐑑𐑲𐑥 𐑞 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑤𐑴𐑼𐑛; 𐑚𐑳𐑑 𐑨𐑟 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑖𐑨𐑤𐑪𐑐'𐑕 𐑕𐑑𐑻𐑯, ·𐑱𐑣𐑨𐑚 𐑡𐑳𐑕𐑑 𐑣𐑪𐑝𐑼𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑐𐑶𐑯𐑑 𐑝 𐑞 𐑛𐑦𐑕𐑧𐑯𐑑, 𐑣𐑰 𐑢𐑱𐑝𐑛 𐑑 𐑞 𐑥𐑱𐑑,—𐑣𐑵 𐑣𐑧𐑤𐑛 𐑢𐑳𐑯 𐑝 𐑞 𐑑𐑨𐑒𐑩𐑤-𐑮𐑴𐑐𐑕 𐑪𐑯 𐑛𐑧𐑒—𐑯 𐑚𐑨𐑛 𐑣𐑦𐑥 𐑐𐑷𐑟.

“𐑕𐑑𐑸𐑚𐑳𐑒!”

“𐑕𐑻?”

“𐑓 𐑞 𐑔𐑻𐑛 𐑑𐑲𐑥 𐑥𐑲 𐑕𐑴𐑤'𐑟 𐑖𐑦𐑐 𐑕𐑑𐑸𐑑𐑕 𐑩𐑐𐑪𐑯 𐑞𐑦𐑕 𐑝𐑶𐑦𐑡, ·𐑕𐑑𐑸𐑚𐑳𐑒.”

“𐑲, 𐑕𐑻, 𐑞𐑬 𐑢𐑦𐑤𐑑 𐑣𐑨𐑝 𐑦𐑑 𐑕𐑴.”

“𐑕𐑳𐑥 𐑖𐑦𐑐𐑕 𐑕𐑱𐑤 𐑓𐑮𐑪𐑥 𐑞𐑺 𐑐𐑹𐑑𐑕, 𐑯 𐑧𐑝𐑼 𐑭𐑓𐑑𐑼𐑢𐑼𐑛𐑟 𐑸 𐑥𐑦𐑕𐑦𐑙, ·𐑕𐑑𐑸𐑚𐑳𐑒!”

“𐑑𐑮𐑵𐑔, 𐑕𐑻: 𐑕𐑨𐑛𐑩𐑕𐑑 𐑑𐑮𐑵𐑔.”

“𐑕𐑳𐑥 𐑥𐑧𐑯 𐑛𐑲 𐑨𐑑 𐑧𐑚 𐑑𐑲𐑛; 𐑕𐑳𐑥 𐑨𐑑 𐑤𐑴 𐑢𐑷𐑑𐑼; 𐑕𐑳𐑥 𐑨𐑑 𐑞 𐑓𐑫𐑤 𐑝 𐑞 𐑓𐑤𐑳𐑛;—𐑯 𐑲 𐑓𐑰𐑤 𐑯𐑬 𐑤𐑲𐑒 𐑩 𐑚𐑦𐑤𐑴 𐑞𐑨𐑑'𐑕 𐑷𐑤 𐑢𐑳𐑯 𐑒𐑮𐑧𐑕𐑑𐑩𐑛 𐑒𐑴𐑥, ·𐑕𐑑𐑸𐑚𐑳𐑒. 𐑲 𐑨𐑥 𐑴𐑤𐑛;—𐑖𐑱𐑒 𐑣𐑨𐑯𐑛𐑟 𐑢𐑦𐑞 𐑥𐑰, 𐑥𐑨𐑯.”

𐑞𐑺 𐑣𐑨𐑯𐑛𐑟 𐑥𐑧𐑑; 𐑞𐑺 𐑲𐑟 𐑓𐑭𐑕𐑩𐑯𐑛; ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑑𐑺𐑟 𐑞 𐑜𐑤𐑵.

“𐑴, 𐑥𐑲 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑥𐑲 𐑒𐑨𐑐𐑑𐑦𐑯!—𐑯𐑴𐑚𐑩𐑤 𐑣𐑸𐑑—𐑜𐑴 𐑯𐑪𐑑—𐑜𐑴 𐑯𐑪𐑑!—𐑕𐑰, 𐑦𐑑'𐑕 𐑩 𐑚𐑮𐑱𐑝 𐑥𐑨𐑯 𐑞𐑨𐑑 𐑢𐑰𐑐𐑕; 𐑣𐑬 𐑜𐑮𐑱𐑑 𐑞 𐑨𐑜𐑩𐑯𐑦 𐑝 𐑞 𐑐𐑼𐑕𐑢𐑱𐑠𐑩𐑯 𐑞𐑧𐑯!”

“𐑤𐑴𐑼 𐑩𐑢𐑱!”—𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑑𐑪𐑕𐑦𐑙 𐑞 𐑥𐑱𐑑'𐑕 𐑸𐑥 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥. “𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑞 𐑒𐑮𐑵!”

𐑦𐑯 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑞 𐑚𐑴𐑑 𐑢𐑪𐑟 𐑐𐑫𐑤𐑦𐑙 𐑮𐑬𐑯𐑛 𐑒𐑤𐑴𐑕 𐑳𐑯𐑛𐑼 𐑞 𐑕𐑑𐑻𐑯.

“𐑞 𐑖𐑸𐑒𐑕! 𐑞 𐑖𐑸𐑒𐑕!” 𐑒𐑮𐑲𐑛 𐑩 𐑝𐑶𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑤𐑴 𐑒𐑨𐑚𐑦𐑯-𐑢𐑦𐑯𐑛𐑴 𐑞𐑺; “𐑴 𐑥𐑭𐑕𐑑𐑼, 𐑥𐑲 𐑥𐑭𐑕𐑑𐑼, 𐑒𐑳𐑥 𐑚𐑨𐑒!”

𐑚𐑳𐑑 ·𐑱𐑣𐑨𐑚 𐑣𐑻𐑛 𐑯𐑳𐑔𐑦𐑙; 𐑓 𐑣𐑦𐑟 𐑴𐑯 𐑝𐑶𐑕 𐑢𐑪𐑟 𐑣𐑲-𐑤𐑦𐑓𐑑𐑩𐑛 𐑞𐑧𐑯; 𐑯 𐑞 𐑚𐑴𐑑 𐑤𐑰𐑐𐑑 𐑪𐑯.

𐑘𐑧𐑑 𐑞 𐑝𐑶𐑕 𐑕𐑐𐑱𐑒 𐑑𐑮𐑵; 𐑓 𐑕𐑒𐑺𐑕 𐑣𐑨𐑛 𐑣𐑰 𐑐𐑫𐑖𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐, 𐑢𐑧𐑯 𐑯𐑳𐑥𐑚𐑼𐑟 𐑝 𐑖𐑸𐑒𐑕, 𐑕𐑰𐑥𐑦𐑙𐑤𐑦 𐑮𐑲𐑟𐑦𐑙 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑞 𐑛𐑸𐑒 𐑢𐑷𐑑𐑼𐑟 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑣𐑳𐑤, 𐑥𐑩𐑤𐑦𐑖𐑩𐑕𐑤𐑦 𐑕𐑯𐑨𐑐𐑑 𐑨𐑑 𐑞 𐑚𐑤𐑱𐑛𐑟 𐑝 𐑞 𐑹𐑟, 𐑧𐑝𐑮𐑦 𐑑𐑲𐑥 𐑞𐑱 𐑛𐑦𐑐𐑑 𐑦𐑯 𐑞 𐑢𐑷𐑑𐑼; 𐑯 𐑦𐑯 𐑞𐑦𐑕 𐑢𐑱 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑛 𐑞 𐑚𐑴𐑑 𐑢𐑦𐑞 𐑞𐑺 𐑚𐑲𐑑𐑕. 𐑦𐑑 𐑦𐑟 𐑩 𐑔𐑦𐑙 𐑯𐑪𐑑 𐑳𐑯𐑒𐑪𐑥𐑩𐑯𐑤𐑦 𐑣𐑨𐑐𐑩𐑯𐑦𐑙 𐑑 𐑞 𐑢𐑱𐑤-𐑚𐑴𐑑𐑕 𐑦𐑯 𐑞𐑴𐑟 𐑕𐑢𐑹𐑥𐑦𐑙 𐑕𐑰𐑟; 𐑞 𐑖𐑸𐑒𐑕 𐑨𐑑 𐑑𐑲𐑥𐑟 𐑩𐑐𐑨𐑮𐑩𐑯𐑑𐑤𐑦 𐑓𐑪𐑤𐑴𐑦𐑙 𐑞𐑧𐑥 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑐𐑮𐑧𐑕𐑾𐑯𐑑 𐑢𐑱 𐑞𐑨𐑑 𐑝𐑳𐑤𐑗𐑼𐑟 𐑣𐑪𐑝𐑼 𐑴𐑝𐑼 𐑞 𐑚𐑨𐑯𐑼𐑟 𐑝 𐑥𐑸𐑗𐑦𐑙 𐑮𐑧𐑡𐑦𐑥𐑩𐑯𐑑𐑕 𐑦𐑯 𐑞 𐑰𐑕𐑑. 𐑚𐑳𐑑 𐑞𐑰𐑟 𐑢𐑻 𐑞 𐑓𐑻𐑕𐑑 𐑖𐑸𐑒𐑕 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑩𐑚𐑟𐑻𐑝𐑛 𐑚𐑲 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑕𐑦𐑯𐑕 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑓𐑻𐑕𐑑 𐑛𐑦𐑕𐑒𐑮𐑲𐑛; 𐑯 𐑢𐑧𐑞𐑼 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 ·𐑱𐑣𐑨𐑚'𐑟 𐑒𐑮𐑵 𐑢𐑻 𐑷𐑤 𐑕𐑳𐑗 𐑑𐑲𐑜𐑼-𐑘𐑧𐑤𐑴 𐑚𐑸𐑚𐑼𐑾𐑯𐑟, 𐑯 𐑞𐑺𐑓𐑹 𐑞𐑺 𐑓𐑤𐑧𐑖 𐑥𐑹 𐑥𐑳𐑕𐑒𐑦 𐑑 𐑞 𐑕𐑧𐑯𐑕𐑩𐑟 𐑝 𐑞 𐑖𐑸𐑒𐑕—𐑩 𐑥𐑨𐑑𐑼 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑢𐑧𐑤 𐑯𐑴𐑯 𐑑 𐑩𐑓𐑧𐑒𐑑 𐑞𐑧𐑥,—𐑣𐑬𐑧𐑝𐑼 𐑦𐑑 𐑢𐑪𐑟, 𐑞𐑱 𐑕𐑰𐑥𐑛 𐑑 𐑓𐑪𐑤𐑴 𐑞𐑨𐑑 𐑢𐑳𐑯 𐑚𐑴𐑑 𐑢𐑦𐑞𐑬𐑑 𐑥𐑩𐑤𐑧𐑕𐑑𐑦𐑙 𐑞 𐑳𐑞𐑼𐑟.

“𐑣𐑸𐑑 𐑝 𐑮𐑷𐑑 𐑕𐑑𐑰𐑤!” 𐑥𐑻𐑥𐑼𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑜𐑱𐑟𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑕𐑲𐑛, 𐑯 𐑓𐑪𐑤𐑴𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑲𐑟 𐑞 𐑮𐑦𐑕𐑰𐑛𐑦𐑙 𐑚𐑴𐑑—“𐑒𐑨𐑯𐑕𐑑 𐑞𐑬 𐑘𐑧𐑑 𐑮𐑦𐑙 𐑚𐑴𐑤𐑛𐑤𐑦 𐑑 𐑞𐑨𐑑 𐑕𐑲𐑑?—𐑤𐑴𐑼𐑦𐑙 𐑞𐑲 𐑒𐑰𐑤 𐑩𐑥𐑳𐑙 𐑮𐑨𐑝𐑩𐑯𐑦𐑙 𐑖𐑸𐑒𐑕, 𐑯 𐑓𐑪𐑤𐑴𐑛 𐑚𐑲 𐑞𐑧𐑥, 𐑴𐑐𐑩𐑯-𐑥𐑬𐑞𐑛 𐑑 𐑞 𐑗𐑱𐑕; 𐑯 𐑞𐑦𐑕 𐑞 𐑒𐑮𐑦𐑑𐑦𐑒𐑩𐑤 𐑔𐑻𐑛 𐑛𐑱?—𐑓 𐑢𐑧𐑯 𐑔𐑮𐑰 𐑛𐑱𐑟 𐑓𐑤𐑴 𐑑𐑩𐑜𐑧𐑞𐑼 𐑦𐑯 𐑢𐑳𐑯 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑕 𐑦𐑯𐑑𐑧𐑯𐑕 𐑐𐑼𐑕𐑿𐑑; 𐑚𐑰 𐑖𐑫𐑼 𐑞 𐑓𐑻𐑕𐑑 𐑦𐑟 𐑞 𐑥𐑹𐑯𐑦𐑙, 𐑞 𐑕𐑧𐑒𐑩𐑯𐑛 𐑞 𐑯𐑵𐑯, 𐑯 𐑞 𐑔𐑻𐑛 𐑞 𐑰𐑝𐑯𐑦𐑙 𐑯 𐑞 𐑧𐑯𐑛 𐑝 𐑞𐑨𐑑 𐑔𐑦𐑙—𐑚𐑰 𐑞𐑨𐑑 𐑧𐑯𐑛 𐑢𐑪𐑑 𐑦𐑑 𐑥𐑱. 𐑴! 𐑥𐑲 ·𐑜𐑪𐑛! 𐑢𐑪𐑑 𐑦𐑟 𐑞𐑦𐑕 𐑞𐑨𐑑 𐑖𐑵𐑑𐑕 𐑔𐑮𐑵 𐑥𐑰, 𐑯 𐑤𐑰𐑝𐑟 𐑥𐑰 𐑕𐑴 𐑛𐑧𐑛𐑤𐑦 𐑒𐑭𐑥, 𐑘𐑧𐑑 𐑦𐑒𐑕𐑐𐑧𐑒𐑑𐑩𐑯𐑑,—𐑓𐑦𐑒𐑕𐑑 𐑨𐑑 𐑞 𐑑𐑪𐑐 𐑝 𐑩 𐑖𐑳𐑛𐑼! 𐑓𐑿𐑗𐑼 𐑔𐑦𐑙𐑟 𐑕𐑢𐑦𐑥 𐑚𐑦𐑓𐑹 𐑥𐑰, 𐑨𐑟 𐑦𐑯 𐑧𐑥𐑐𐑑𐑦 𐑬𐑑𐑤𐑲𐑯𐑟 𐑯 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯𐑟; 𐑷𐑤 𐑞 𐑐𐑭𐑕𐑑 𐑦𐑟 𐑕𐑳𐑥𐑣𐑬 𐑜𐑮𐑴𐑯 𐑛𐑦𐑥. 𐑥𐑺𐑦, 𐑜𐑻𐑤! 𐑞𐑬 𐑓𐑱𐑛𐑩𐑕𐑑 𐑦𐑯 𐑐𐑱𐑤 𐑜𐑤𐑹𐑦𐑟 𐑚𐑦𐑣𐑲𐑯𐑛 𐑥𐑰; 𐑚𐑶! 𐑲 𐑕𐑰𐑥 𐑑 𐑕𐑰 𐑚𐑳𐑑 𐑞𐑲 𐑲𐑟 𐑜𐑮𐑴𐑯 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑚𐑤𐑵. 𐑕𐑑𐑮𐑱𐑯𐑡𐑩𐑕𐑑 𐑐𐑮𐑪𐑚𐑤𐑩𐑥𐑟 𐑝 𐑤𐑲𐑓 𐑕𐑰𐑥 𐑒𐑤𐑽𐑦𐑙; 𐑚𐑳𐑑 𐑒𐑤𐑬𐑛𐑟 𐑕𐑢𐑰𐑐 𐑚𐑦𐑑𐑢𐑰𐑯—𐑦𐑟 𐑥𐑲 𐑡𐑻𐑯𐑦'𐑟 𐑧𐑯𐑛 𐑒𐑳𐑥𐑦𐑙? 𐑥𐑲 𐑤𐑧𐑜𐑟 𐑓𐑰𐑤 𐑓𐑱𐑯𐑑; 𐑤𐑲𐑒 𐑣𐑦𐑟 𐑣𐑵 𐑣𐑨𐑟 𐑓𐑫𐑑𐑩𐑛 𐑦𐑑 𐑷𐑤 𐑛𐑱. 𐑓𐑰𐑤 𐑞𐑲 𐑣𐑸𐑑,—𐑚𐑰𐑑𐑕 𐑦𐑑 𐑘𐑧𐑑? 𐑕𐑑𐑻 𐑞𐑲𐑕𐑧𐑤𐑓, ·𐑕𐑑𐑸𐑚𐑳𐑒!—𐑕𐑑𐑱𐑝 𐑦𐑑 𐑪𐑓—𐑥𐑵𐑝, 𐑥𐑵𐑝! 𐑕𐑐𐑰𐑒 𐑩𐑤𐑬𐑛!—𐑥𐑭𐑕𐑑-𐑣𐑧𐑛 𐑞𐑺! 𐑕𐑰 𐑘𐑰 𐑥𐑲 𐑚𐑶'𐑟 𐑣𐑨𐑯𐑛 𐑪𐑯 𐑞 𐑣𐑦𐑤?—𐑒𐑮𐑱𐑟𐑛;—𐑩𐑤𐑪𐑓𐑑 𐑞𐑺!—𐑒𐑰𐑐 𐑞𐑲 𐑒𐑰𐑯𐑩𐑕𐑑 𐑲 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑴𐑑𐑕:—𐑥𐑸𐑒 𐑢𐑧𐑤 𐑞 𐑢𐑱𐑤!—𐑣𐑴! 𐑩𐑜𐑱𐑯!—𐑛𐑮𐑲𐑝 𐑪𐑓 𐑞𐑨𐑑 𐑣𐑷𐑒! 𐑕𐑰! 𐑣𐑰 𐑐𐑧𐑒𐑕—𐑣𐑰 𐑑𐑺𐑟 𐑞 𐑝𐑱𐑯”—𐑐𐑶𐑯𐑑𐑦𐑙 𐑑 𐑞 𐑮𐑧𐑛 𐑓𐑤𐑨𐑜 𐑓𐑤𐑲𐑦𐑙 𐑨𐑑 𐑞 𐑥𐑱𐑯-𐑑𐑮𐑳𐑒—“𐑣𐑭! 𐑣𐑰 𐑕𐑹𐑟 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑦𐑑!—𐑢𐑺'𐑟 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑯𐑬? 𐑕𐑰'𐑕𐑑 𐑞𐑬 𐑞𐑨𐑑 𐑕𐑲𐑑, 𐑴 ·𐑱𐑣𐑨𐑚!—𐑖𐑳𐑛𐑼, 𐑖𐑳𐑛𐑼!”

𐑞 𐑚𐑴𐑑𐑕 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑜𐑪𐑯 𐑝𐑧𐑮𐑦 𐑓𐑸, 𐑢𐑧𐑯 𐑚𐑲 𐑩 𐑕𐑦𐑜𐑯𐑩𐑤 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟—𐑩 𐑛𐑬𐑯𐑢𐑼𐑛 𐑐𐑶𐑯𐑑𐑩𐑛 𐑸𐑥, ·𐑱𐑣𐑨𐑚 𐑯𐑿 𐑞𐑨𐑑 𐑞 𐑢𐑱𐑤 𐑣𐑨𐑛 𐑕𐑬𐑯𐑛𐑩𐑛; 𐑚𐑳𐑑 𐑦𐑯𐑑𐑧𐑯𐑛𐑦𐑙 𐑑 𐑚𐑰 𐑯𐑽 𐑣𐑦𐑥 𐑨𐑑 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑮𐑲𐑟𐑦𐑙, 𐑣𐑰 𐑣𐑧𐑤𐑛 𐑪𐑯 𐑣𐑦𐑟 𐑢𐑱 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑕𐑲𐑛𐑢𐑱𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑝𐑧𐑕𐑩𐑤; 𐑞 𐑚𐑦𐑗𐑸𐑥𐑛 𐑒𐑮𐑵 𐑥𐑱𐑯𐑑𐑱𐑯𐑦𐑙 𐑞 𐑐𐑮𐑩𐑓𐑬𐑯𐑛𐑩𐑕𐑑 𐑕𐑲𐑤𐑩𐑯𐑕, 𐑨𐑟 𐑞 𐑣𐑧𐑛-𐑚𐑰𐑑 𐑢𐑱𐑝𐑟 𐑣𐑨𐑥𐑼𐑛 𐑯 𐑣𐑨𐑥𐑼𐑛 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑩𐑐𐑴𐑟𐑦𐑙 𐑚𐑬.

“𐑛𐑮𐑲𐑝, 𐑛𐑮𐑲𐑝 𐑦𐑯 𐑘𐑹 𐑯𐑱𐑤𐑟, 𐑴 𐑘𐑰 𐑢𐑱𐑝𐑟! 𐑑 𐑞𐑺 𐑳𐑑𐑼𐑥𐑴𐑕𐑑 𐑣𐑧𐑛𐑟 𐑛𐑮𐑲𐑝 𐑞𐑧𐑥 𐑦𐑯! 𐑘𐑰 𐑚𐑳𐑑 𐑕𐑑𐑮𐑲𐑒 𐑩 𐑔𐑦𐑙 𐑢𐑦𐑞𐑬𐑑 𐑩 𐑤𐑦𐑛; 𐑯 𐑯𐑴 𐑒𐑪𐑓𐑦𐑯 𐑯 𐑯𐑴 𐑣𐑻𐑕 𐑒𐑨𐑯 𐑚𐑰 𐑥𐑲𐑯:—𐑯 𐑣𐑧𐑥𐑐 𐑴𐑯𐑤𐑦 𐑒𐑨𐑯 𐑒𐑦𐑤 𐑥𐑰! 𐑣𐑭! 𐑣𐑭!”

𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑞 𐑢𐑷𐑑𐑼𐑟 𐑼𐑬𐑯𐑛 𐑞𐑧𐑥 𐑕𐑤𐑴𐑤𐑦 𐑕𐑢𐑧𐑤𐑛 𐑦𐑯 𐑚𐑮𐑷𐑛 𐑕𐑻𐑒𐑩𐑤𐑟; 𐑞𐑧𐑯 𐑒𐑢𐑦𐑒𐑤𐑦 𐑳𐑐𐑣𐑰𐑝𐑛, 𐑨𐑟 𐑦𐑓 𐑕𐑲𐑛𐑢𐑱𐑟 𐑕𐑤𐑲𐑛𐑦𐑙 𐑓𐑮𐑪𐑥 𐑩 𐑕𐑩𐑚𐑥𐑻𐑡𐑛 𐑚𐑻𐑜 𐑝 𐑲𐑕, 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑮𐑲𐑟𐑦𐑙 𐑑 𐑞 𐑕𐑻𐑓𐑦𐑕. 𐑩 𐑤𐑴 𐑮𐑳𐑥𐑚𐑩𐑤𐑦𐑙 𐑕𐑬𐑯𐑛 𐑢𐑪𐑟 𐑣𐑻𐑛; 𐑩 𐑕𐑳𐑚𐑑𐑼𐑱𐑯𐑾𐑕 𐑣𐑳𐑥; 𐑯 𐑞𐑧𐑯 𐑷𐑤 𐑣𐑧𐑤𐑛 𐑞𐑺 𐑚𐑮𐑧𐑔𐑕; 𐑨𐑟 𐑚𐑦𐑛𐑮𐑨𐑜𐑩𐑤𐑛 𐑢𐑦𐑞 𐑑𐑮𐑱𐑤𐑦𐑙 𐑮𐑴𐑐𐑕, 𐑯 𐑣𐑸𐑐𐑵𐑯𐑟, 𐑯 𐑤𐑭𐑯𐑕𐑩𐑟, 𐑩 𐑝𐑭𐑕𐑑 𐑓𐑹𐑥 𐑖𐑪𐑑 𐑤𐑧𐑙𐑔𐑢𐑲𐑟, 𐑚𐑳𐑑 𐑴𐑚𐑤𐑰𐑒𐑤𐑦 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑰. 𐑖𐑮𐑬𐑛𐑩𐑛 𐑦𐑯 𐑩 𐑔𐑦𐑯 𐑛𐑮𐑵𐑐𐑦𐑙 𐑝𐑱𐑤 𐑝 𐑥𐑦𐑕𐑑, 𐑦𐑑 𐑣𐑪𐑝𐑼𐑛 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑦𐑯 𐑞 𐑮𐑱𐑯𐑚𐑴𐑛 𐑺; 𐑯 𐑞𐑧𐑯 𐑓𐑧𐑤 𐑕𐑢𐑪𐑥𐑐𐑦𐑙 𐑚𐑨𐑒 𐑦𐑯𐑑𐑵 𐑞 𐑛𐑰𐑐. 𐑒𐑮𐑳𐑖𐑑 𐑔𐑻𐑑𐑦 𐑓𐑰𐑑 𐑳𐑐𐑢𐑼𐑛𐑟, 𐑞 𐑢𐑷𐑑𐑼𐑟 𐑓𐑤𐑨𐑖𐑑 𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑤𐑲𐑒 𐑣𐑰𐑐𐑕 𐑝 𐑓𐑬𐑯𐑑𐑩𐑯𐑟, 𐑞𐑧𐑯 𐑚𐑮𐑴𐑒𐑩𐑯𐑤𐑦 𐑕𐑨𐑙𐑒 𐑦𐑯 𐑩 𐑖𐑬𐑼 𐑝 𐑓𐑤𐑱𐑒𐑕, 𐑤𐑰𐑝𐑦𐑙 𐑞 𐑕𐑻𐑒𐑩𐑤𐑦𐑙 𐑕𐑻𐑓𐑦𐑕 𐑒𐑮𐑰𐑥𐑛 𐑤𐑲𐑒 𐑯𐑿 𐑥𐑦𐑤𐑒 𐑮𐑬𐑯𐑛 𐑞 𐑥𐑸𐑚𐑩𐑤 𐑑𐑮𐑳𐑙𐑒 𐑝 𐑞 𐑢𐑱𐑤.

“𐑜𐑦𐑝 𐑢𐑱!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚 𐑑 𐑞 𐑹𐑟𐑥𐑧𐑯, 𐑯 𐑞 𐑚𐑴𐑑𐑕 𐑛𐑸𐑑𐑩𐑛 𐑓𐑹𐑢𐑼𐑛 𐑑 𐑞 𐑩𐑑𐑨𐑒; 𐑚𐑳𐑑 𐑥𐑨𐑛𐑩𐑯𐑛 𐑚𐑲 𐑘𐑧𐑕𐑑𐑼𐑛𐑱'𐑟 𐑓𐑮𐑧𐑖 𐑲𐑼𐑯𐑟 𐑞𐑨𐑑 𐑒𐑼𐑴𐑛𐑩𐑛 𐑦𐑯 𐑣𐑦𐑥, ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑕𐑰𐑥𐑛 𐑒𐑩𐑥𐑚𐑲𐑯𐑛𐑤𐑦 𐑐𐑩𐑟𐑧𐑕𐑑 𐑚𐑲 𐑷𐑤 𐑞 𐑱𐑯𐑡𐑩𐑤𐑟 𐑞𐑨𐑑 𐑓𐑧𐑤 𐑓𐑮𐑪𐑥 𐑣𐑧𐑝𐑩𐑯. 𐑞 𐑢𐑲𐑛 𐑑𐑽𐑟 𐑝 𐑢𐑧𐑤𐑛𐑩𐑛 𐑑𐑧𐑯𐑛𐑩𐑯𐑟 𐑴𐑝𐑼𐑕𐑐𐑮𐑧𐑛𐑦𐑙 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛 𐑢𐑲𐑑 𐑓𐑪𐑮𐑦𐑛, 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑑𐑮𐑨𐑯𐑕𐑐𐑨𐑮𐑩𐑯𐑑 𐑕𐑒𐑦𐑯, 𐑤𐑫𐑒𐑑 𐑯𐑦𐑑𐑩𐑛 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑨𐑟 𐑣𐑧𐑛 𐑪𐑯, 𐑣𐑰 𐑒𐑱𐑥 𐑗𐑻𐑯𐑦𐑙 𐑣𐑦𐑟 𐑑𐑱𐑤 𐑩𐑥𐑳𐑙 𐑞 𐑚𐑴𐑑𐑕; 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑓𐑤𐑱𐑤𐑛 𐑞𐑧𐑥 𐑩𐑐𐑸𐑑; 𐑕𐑐𐑦𐑤𐑦𐑙 𐑬𐑑 𐑞 𐑲𐑼𐑯𐑟 𐑯 𐑤𐑭𐑯𐑕𐑩𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑵 𐑥𐑱𐑑𐑕' 𐑚𐑴𐑑𐑕, 𐑯 𐑛𐑨𐑖𐑦𐑙 𐑦𐑯 𐑢𐑳𐑯 𐑕𐑲𐑛 𐑝 𐑞 𐑳𐑐𐑼 𐑐𐑸𐑑 𐑝 𐑞𐑺 𐑚𐑬𐑟, 𐑚𐑳𐑑 𐑤𐑰𐑝𐑦𐑙 ·𐑱𐑣𐑨𐑚'𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑢𐑦𐑞𐑬𐑑 𐑩 𐑕𐑒𐑸.

𐑢𐑲𐑤 ·𐑛𐑨𐑜𐑵 𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑢𐑻 𐑕𐑑𐑪𐑐𐑦𐑙 𐑞 𐑕𐑑𐑮𐑱𐑯𐑛 𐑐𐑤𐑨𐑙𐑒𐑕; 𐑯 𐑨𐑟 𐑞 𐑢𐑱𐑤 𐑕𐑢𐑦𐑥𐑦𐑙 𐑬𐑑 𐑓𐑮𐑪𐑥 𐑞𐑧𐑥, 𐑑𐑻𐑯𐑛, 𐑯 𐑖𐑴𐑛 𐑢𐑳𐑯 𐑦𐑯𐑑𐑲𐑼 𐑓𐑤𐑨𐑙𐑒 𐑨𐑟 𐑣𐑰 𐑖𐑪𐑑 𐑚𐑲 𐑞𐑧𐑥 𐑩𐑜𐑱𐑯; 𐑨𐑑 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑩 𐑒𐑢𐑦𐑒 𐑒𐑮𐑲 𐑢𐑧𐑯𐑑 𐑳𐑐. 𐑤𐑨𐑖𐑑 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛 𐑑 𐑞 𐑓𐑦𐑖'𐑩𐑟 𐑚𐑨𐑒; 𐑐𐑦𐑯𐑘𐑩𐑯𐑛 𐑦𐑯 𐑞 𐑑𐑻𐑯𐑟 𐑩𐑐𐑪𐑯 𐑑𐑻𐑯𐑟 𐑦𐑯 𐑢𐑦𐑗, 𐑛𐑘𐑫𐑼𐑦𐑙 𐑞 𐑐𐑭𐑕𐑑 𐑯𐑲𐑑, 𐑞 𐑢𐑱𐑤 𐑣𐑨𐑛 𐑮𐑰𐑤𐑛 𐑞 𐑦𐑯𐑝𐑩𐑤𐑵𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑤𐑲𐑯𐑟 𐑼𐑬𐑯𐑛 𐑣𐑦𐑥, 𐑞 𐑣𐑭𐑓 𐑑𐑹𐑯 𐑚𐑪𐑛𐑦 𐑝 𐑞 ·𐑐𐑸𐑟𐑰 𐑢𐑪𐑟 𐑕𐑰𐑯; 𐑣𐑦𐑟 𐑕𐑱𐑚𐑩𐑤 𐑮𐑱𐑥𐑩𐑯𐑑 𐑓𐑮𐑱𐑛 𐑑 𐑖𐑮𐑧𐑛𐑟; 𐑣𐑦𐑟 𐑛𐑦𐑕𐑑𐑧𐑯𐑛𐑩𐑛 𐑲𐑟 𐑑𐑻𐑯𐑛 𐑓𐑫𐑤 𐑩𐑐𐑪𐑯 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚.

𐑞 𐑣𐑸𐑐𐑵𐑯 𐑛𐑮𐑪𐑐𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛.

“𐑚𐑦𐑓𐑵𐑤𐑛, 𐑚𐑦𐑓𐑵𐑤𐑛!”—𐑛𐑮𐑷𐑦𐑙 𐑦𐑯 𐑩 𐑤𐑪𐑙 𐑤𐑰𐑯 𐑚𐑮𐑧𐑔—“𐑲, ·𐑐𐑸𐑟𐑰! 𐑲 𐑕𐑰 𐑞𐑰 𐑩𐑜𐑱𐑯.—𐑲, 𐑯 𐑞𐑬 𐑜𐑴𐑩𐑕𐑑 𐑚𐑦𐑓𐑹; 𐑯 𐑞𐑦𐑕, 𐑞𐑦𐑕 𐑞𐑧𐑯 𐑦𐑟 𐑞 𐑣𐑻𐑕 𐑞𐑨𐑑 𐑞𐑬 𐑛𐑦𐑛𐑕𐑑 𐑐𐑮𐑪𐑥𐑦𐑕. 𐑚𐑳𐑑 𐑲 𐑣𐑴𐑤𐑛 𐑞𐑰 𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑤𐑧𐑑𐑼 𐑝 𐑞𐑲 𐑢𐑻𐑛. 𐑢𐑺 𐑦𐑟 𐑞 𐑕𐑧𐑒𐑩𐑯𐑛 𐑣𐑻𐑕? 𐑩𐑢𐑱, 𐑥𐑱𐑑𐑕, 𐑑 𐑞 𐑖𐑦𐑐! 𐑞𐑴𐑟 𐑚𐑴𐑑𐑕 𐑸 𐑿𐑕𐑤𐑩𐑕 𐑯𐑬; 𐑮𐑦𐑐𐑺 𐑞𐑧𐑥 𐑦𐑓 𐑘𐑰 𐑒𐑨𐑯 𐑦𐑯 𐑑𐑲𐑥, 𐑯 𐑮𐑦𐑑𐑻𐑯 𐑑 𐑥𐑰; 𐑦𐑓 𐑯𐑪𐑑, ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑦𐑯𐑳𐑓 𐑑 𐑛𐑲—·𐑛𐑬𐑯, 𐑥𐑧𐑯! 𐑞 𐑓𐑻𐑕𐑑 𐑔𐑦𐑙 𐑞𐑨𐑑 𐑚𐑳𐑑 𐑪𐑓𐑼𐑟 𐑑 𐑡𐑳𐑥𐑐 𐑓𐑮𐑪𐑥 𐑞𐑦𐑕 𐑚𐑴𐑑 𐑲 𐑕𐑑𐑨𐑯𐑛 𐑦𐑯, 𐑞𐑨𐑑 𐑔𐑦𐑙 𐑲 𐑣𐑸𐑐𐑵𐑯. 𐑘𐑰 𐑸 𐑯𐑪𐑑 𐑳𐑞𐑼 𐑥𐑧𐑯, 𐑚𐑳𐑑 𐑥𐑲 𐑸𐑥𐑟 𐑯 𐑥𐑲 𐑤𐑧𐑜𐑟; 𐑯 𐑕𐑴 𐑴𐑚𐑱 𐑥𐑰.—𐑢𐑺'𐑟 𐑞 𐑢𐑱𐑤? 𐑜𐑪𐑯 𐑛𐑬𐑯 𐑩𐑜𐑱𐑯?”

𐑚𐑳𐑑 𐑣𐑰 𐑤𐑫𐑒𐑑 𐑑𐑵 𐑯𐑲 𐑞 𐑚𐑴𐑑; 𐑓 𐑨𐑟 𐑦𐑓 𐑚𐑧𐑯𐑑 𐑩𐑐𐑪𐑯 𐑦𐑕𐑒𐑱𐑐𐑦𐑙 𐑢𐑦𐑞 𐑞 𐑒𐑹𐑐𐑕 𐑣𐑰 𐑚𐑹, 𐑯 𐑨𐑟 𐑦𐑓 𐑞 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑐𐑤𐑱𐑕 𐑝 𐑞 𐑤𐑭𐑕𐑑 𐑦𐑯𐑒𐑬𐑯𐑑𐑼 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑚𐑳𐑑 𐑩 𐑕𐑑𐑱𐑡 𐑦𐑯 𐑣𐑦𐑟 𐑤𐑰𐑢𐑼𐑛 𐑝𐑶𐑦𐑡, ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑢𐑪𐑟 𐑯𐑬 𐑩𐑜𐑱𐑯 𐑕𐑑𐑧𐑛𐑦𐑤𐑦 𐑕𐑢𐑦𐑥𐑦𐑙 𐑓𐑹𐑢𐑼𐑛; 𐑯 𐑣𐑨𐑛 𐑷𐑤𐑥𐑴𐑕𐑑 𐑐𐑭𐑕𐑑 𐑞 𐑖𐑦𐑐,—𐑢𐑦𐑗 𐑞𐑳𐑕 𐑓𐑸 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑱𐑤𐑦𐑙 𐑦𐑯 𐑞 𐑒𐑪𐑯𐑑𐑮𐑺𐑦 𐑛𐑦𐑮𐑧𐑒𐑖𐑩𐑯 𐑑 𐑣𐑦𐑥, 𐑞𐑴 𐑓 𐑞 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑣𐑻 𐑣𐑧𐑛𐑢𐑱 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑑𐑪𐑐𐑑. 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑕𐑢𐑦𐑥𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑳𐑑𐑥𐑴𐑕𐑑 𐑝𐑦𐑤𐑪𐑕𐑦𐑑𐑦, 𐑯 𐑯𐑬 𐑴𐑯𐑤𐑦 𐑦𐑯𐑑𐑧𐑯𐑑 𐑩𐑐𐑪𐑯 𐑐𐑼𐑕𐑿𐑦𐑙 𐑣𐑦𐑟 𐑴𐑯 𐑕𐑑𐑮𐑱𐑑 𐑐𐑭𐑔 𐑦𐑯 𐑞 𐑕𐑰.

“𐑴! 𐑱𐑣𐑨𐑚,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒, “𐑯𐑪𐑑 𐑑𐑵 𐑤𐑱𐑑 𐑦𐑟 𐑦𐑑, 𐑰𐑝𐑩𐑯 𐑯𐑬, 𐑞 𐑔𐑻𐑛 𐑛𐑱, 𐑑 𐑛𐑦𐑟𐑦𐑕𐑑. 𐑕𐑰! 𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑕𐑰𐑒𐑕 𐑞𐑰 𐑯𐑪𐑑. 𐑦𐑑 𐑦𐑟 𐑞𐑬, 𐑞𐑬, 𐑞𐑨𐑑 𐑥𐑨𐑛𐑤𐑦 𐑕𐑰𐑒𐑩𐑕𐑑 𐑣𐑦𐑥!”

𐑕𐑧𐑑𐑦𐑙 𐑕𐑱𐑤 𐑑 𐑞 𐑮𐑲𐑟𐑦𐑙 𐑢𐑦𐑯𐑛, 𐑞 𐑤𐑴𐑯𐑤𐑦 𐑚𐑴𐑑 𐑢𐑪𐑟 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑦𐑥𐑐𐑧𐑤𐑛 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑚𐑲 𐑚𐑴𐑔 𐑹𐑟 𐑯 𐑒𐑨𐑯𐑝𐑩𐑕. 𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑢𐑧𐑯 ·𐑱𐑣𐑨𐑚 𐑢𐑪𐑟 𐑕𐑤𐑲𐑛𐑦𐑙 𐑚𐑲 𐑞 𐑝𐑧𐑕𐑩𐑤, 𐑕𐑴 𐑯𐑽 𐑨𐑟 𐑐𐑤𐑱𐑯𐑤𐑦 𐑑 𐑛𐑦𐑕𐑑𐑦𐑙𐑜𐑢𐑦𐑖 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑓𐑱𐑕 𐑨𐑟 𐑣𐑰 𐑤𐑰𐑯𐑛 𐑴𐑝𐑼 𐑞 𐑮𐑱𐑤, 𐑣𐑰 𐑣𐑱𐑤𐑛 𐑣𐑦𐑥 𐑑 𐑑𐑻𐑯 𐑞 𐑝𐑧𐑕𐑩𐑤 𐑩𐑚𐑬𐑑, 𐑯 𐑓𐑪𐑤𐑴 𐑣𐑦𐑥, 𐑯𐑪𐑑 𐑑𐑵 𐑕𐑢𐑦𐑓𐑑𐑤𐑦, 𐑨𐑑 𐑩 𐑡𐑵𐑛𐑦𐑖𐑩𐑕 𐑦𐑯𐑑𐑼𐑝𐑩𐑤. 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑳𐑐𐑢𐑼𐑛𐑟, 𐑣𐑰 𐑕𐑷 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑯 ·𐑛𐑨𐑜𐑵, 𐑰𐑜𐑼𐑤𐑦 𐑥𐑬𐑯𐑑𐑦𐑙 𐑑 𐑞 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟; 𐑢𐑲𐑤 𐑞 𐑹𐑟𐑥𐑧𐑯 𐑢𐑻 𐑮𐑪𐑒𐑦𐑙 𐑦𐑯 𐑞 𐑑𐑵 𐑕𐑑𐑱𐑝𐑛 𐑚𐑴𐑑𐑕 𐑢𐑦𐑗 𐑣𐑨𐑛 𐑚𐑳𐑑 𐑡𐑳𐑕𐑑 𐑚𐑰𐑯 𐑣𐑶𐑕𐑑𐑩𐑛 𐑑 𐑞 𐑕𐑲𐑛, 𐑯 𐑢𐑻 𐑚𐑦𐑟𐑦𐑤𐑦 𐑨𐑑 𐑢𐑻𐑒 𐑦𐑯 𐑮𐑦𐑐𐑺𐑦𐑙 𐑞𐑧𐑥. 𐑢𐑳𐑯 𐑭𐑓𐑑𐑼 𐑞 𐑳𐑞𐑼, 𐑔𐑮𐑵 𐑞 𐑐𐑹𐑑-𐑣𐑴𐑤𐑟, 𐑨𐑟 𐑣𐑰 𐑕𐑐𐑧𐑛, 𐑣𐑰 𐑷𐑤𐑕𐑴 𐑒𐑷𐑑 𐑓𐑤𐑲𐑦𐑙 𐑜𐑤𐑦𐑥𐑐𐑕𐑩𐑟 𐑝 ·𐑕𐑑𐑳𐑚 𐑯 ·𐑓𐑤𐑭𐑕𐑒, 𐑚𐑦𐑟𐑦𐑦𐑙 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑪𐑯 𐑛𐑧𐑒 𐑩𐑥𐑳𐑙 𐑚𐑳𐑯𐑛𐑩𐑤𐑟 𐑝 𐑯𐑿 𐑲𐑼𐑯𐑟 𐑯 𐑤𐑭𐑯𐑕𐑩𐑟. 𐑨𐑟 𐑣𐑰 𐑕𐑷 𐑷𐑤 𐑞𐑦𐑕; 𐑨𐑟 𐑣𐑰 𐑣𐑻𐑛 𐑞 𐑣𐑨𐑥𐑼𐑟 𐑦𐑯 𐑞 𐑚𐑮𐑴𐑒𐑩𐑯 𐑚𐑴𐑑𐑕; 𐑓𐑸 𐑳𐑞𐑼 𐑣𐑨𐑥𐑼𐑟 𐑕𐑰𐑥𐑛 𐑛𐑮𐑲𐑝𐑦𐑙 𐑩 𐑯𐑱𐑤 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑣𐑸𐑑. 𐑚𐑳𐑑 𐑣𐑰 𐑮𐑨𐑤𐑦𐑛. 𐑯 𐑯𐑬 𐑥𐑸𐑒𐑦𐑙 𐑞𐑨𐑑 𐑞 𐑝𐑱𐑯 𐑹 𐑓𐑤𐑨𐑜 𐑢𐑪𐑟 𐑜𐑪𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑-𐑣𐑧𐑛, 𐑣𐑰 𐑖𐑬𐑑𐑩𐑛 𐑑 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, 𐑣𐑵 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑜𐑱𐑯𐑛 𐑞𐑨𐑑 𐑐𐑻𐑗, 𐑑 𐑛𐑦𐑕𐑧𐑯𐑛 𐑩𐑜𐑱𐑯 𐑓 𐑩𐑯𐑳𐑞𐑼 𐑓𐑤𐑨𐑜, 𐑯 𐑩 𐑣𐑨𐑥𐑼 𐑯 𐑯𐑱𐑤𐑟, 𐑯 𐑕𐑴 𐑯𐑱𐑤 𐑦𐑑 𐑑 𐑞 𐑥𐑭𐑕𐑑.

𐑢𐑧𐑞𐑼 𐑓𐑨𐑜𐑛 𐑚𐑲 𐑞 𐑔𐑮𐑰 𐑛𐑱𐑟' 𐑮𐑳𐑯𐑦𐑙 𐑗𐑱𐑕, 𐑯 𐑞 𐑮𐑦𐑟𐑦𐑕𐑑𐑩𐑯𐑕 𐑑 𐑣𐑦𐑟 𐑕𐑢𐑦𐑥𐑦𐑙 𐑦𐑯 𐑞 𐑯𐑪𐑑𐑩𐑛 𐑣𐑨𐑥𐑐𐑼 𐑣𐑰 𐑚𐑹; 𐑹 𐑢𐑧𐑞𐑼 𐑦𐑑 𐑢𐑪𐑟 𐑕𐑳𐑥 𐑤𐑱𐑑𐑩𐑯𐑑 𐑛𐑦𐑕𐑰𐑑𐑓𐑩𐑤𐑯𐑩𐑕 𐑯 𐑥𐑨𐑤𐑦𐑕 𐑦𐑯 𐑣𐑦𐑥: 𐑢𐑦𐑗𐑧𐑝𐑼 𐑢𐑪𐑟 𐑑𐑮𐑵, 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑢𐑱 𐑯𐑬 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑩𐑚𐑱𐑑, 𐑨𐑟 𐑦𐑑 𐑕𐑰𐑥𐑛, 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑑 𐑕𐑴 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑯𐑽𐑦𐑙 𐑣𐑦𐑥 𐑢𐑳𐑯𐑕 𐑥𐑹; 𐑞𐑴 𐑦𐑯𐑛𐑰𐑛 𐑞 𐑢𐑱𐑤'𐑟 𐑤𐑭𐑕𐑑 𐑕𐑑𐑸𐑑 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑚𐑰𐑯 𐑕𐑴 𐑤𐑪𐑙 𐑩 𐑢𐑳𐑯 𐑨𐑟 𐑚𐑦𐑓𐑹. 𐑯 𐑕𐑑𐑦𐑤 𐑨𐑟 ·𐑱𐑣𐑨𐑚 𐑜𐑤𐑲𐑛𐑩𐑛 𐑴𐑝𐑼 𐑞 𐑢𐑱𐑝𐑟 𐑞 𐑳𐑯𐑐𐑦𐑑𐑦𐑦𐑙 𐑖𐑸𐑒𐑕 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑛 𐑣𐑦𐑥; 𐑯 𐑕𐑴 𐑐𐑻𐑑𐑦𐑯𐑱𐑖𐑩𐑕𐑤𐑦 𐑕𐑑𐑳𐑒 𐑑 𐑞 𐑚𐑴𐑑; 𐑯 𐑕𐑴 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤𐑦 𐑚𐑦𐑑 𐑨𐑑 𐑞 𐑐𐑤𐑲𐑦𐑙 𐑹𐑟, 𐑞𐑨𐑑 𐑞 𐑚𐑤𐑱𐑛𐑟 𐑚𐑦𐑒𐑱𐑥 𐑡𐑨𐑜𐑩𐑛 𐑯 𐑒𐑮𐑳𐑯𐑗𐑑, 𐑯 𐑤𐑧𐑓𐑑 𐑕𐑥𐑷𐑤 𐑕𐑐𐑤𐑦𐑯𐑑𐑼𐑟 𐑦𐑯 𐑞 𐑕𐑰, 𐑨𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑝𐑮𐑦 𐑛𐑦𐑐.

𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑦𐑥𐑩𐑤𐑑𐑱𐑯𐑾𐑕𐑤𐑦, 𐑢𐑦𐑞 𐑩 𐑥𐑲𐑑𐑦 𐑝𐑴𐑤𐑦𐑖𐑩𐑯 𐑝 𐑳𐑯𐑜𐑮𐑨𐑡𐑵𐑱𐑑𐑩𐑛, 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑱𐑯𐑾𐑕 𐑕𐑢𐑦𐑓𐑑𐑯𐑩𐑕, 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑛𐑸𐑑𐑩𐑛 𐑔𐑮𐑵 𐑞 𐑢𐑧𐑤𐑑𐑼𐑦𐑙 𐑕𐑰. 𐑚𐑳𐑑 𐑢𐑧𐑯 ·𐑱𐑣𐑨𐑚 𐑒𐑮𐑲𐑛 𐑬𐑑 𐑑 𐑞 𐑕𐑑𐑽𐑟𐑥𐑩𐑯 𐑑 𐑑𐑱𐑒 𐑯𐑿 𐑑𐑻𐑯𐑟 𐑢𐑦𐑞 𐑞 𐑤𐑲𐑯, 𐑯 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑕𐑴; 𐑯 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛 𐑞 𐑒𐑮𐑵 𐑑 𐑑𐑻𐑯 𐑮𐑬𐑯𐑛 𐑪𐑯 𐑞𐑺 𐑕𐑰𐑑𐑕, 𐑯 𐑑𐑴 𐑞 𐑚𐑴𐑑 𐑳𐑐 𐑑 𐑞 𐑥𐑸𐑒; 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑 𐑞 𐑑𐑮𐑧𐑗𐑼𐑩𐑕 𐑤𐑲𐑯 𐑓𐑧𐑤𐑑 𐑞𐑨𐑑 𐑛𐑳𐑚𐑩𐑤 𐑕𐑑𐑮𐑱𐑯 𐑯 𐑑𐑳𐑜, 𐑦𐑑 𐑕𐑯𐑨𐑐𐑑 𐑦𐑯 𐑞 𐑧𐑥𐑐𐑑𐑦 𐑺!

“𐑢𐑪𐑑 𐑚𐑮𐑱𐑒𐑕 𐑦𐑯 𐑥𐑰? 𐑕𐑳𐑥 𐑕𐑦𐑯𐑿 𐑒𐑮𐑨𐑒𐑕!—'𐑑𐑦𐑟 𐑣𐑴𐑤 𐑩𐑜𐑱𐑯; 𐑹𐑟! 𐑹𐑟! 𐑚𐑻𐑕𐑑 𐑦𐑯 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥!”

𐑣𐑽𐑦𐑙 𐑞 𐑑𐑮𐑦𐑥𐑧𐑯𐑛𐑩𐑕 𐑮𐑳𐑖 𐑝 𐑞 𐑕𐑰-𐑒𐑮𐑨𐑖𐑦𐑙 𐑚𐑴𐑑, 𐑞 𐑢𐑱𐑤 𐑢𐑰𐑤𐑛 𐑮𐑬𐑯𐑛 𐑑 𐑐𐑮𐑦𐑟𐑧𐑯𐑑 𐑣𐑦𐑟 𐑚𐑤𐑨𐑙𐑒 𐑓𐑪𐑮𐑦𐑛 𐑨𐑑 𐑚𐑱; 𐑚𐑳𐑑 𐑦𐑯 𐑞𐑨𐑑 𐑰𐑝𐑩𐑤𐑵𐑖𐑩𐑯, 𐑒𐑨𐑗𐑦𐑙 𐑕𐑲𐑑 𐑝 𐑞 𐑯𐑽𐑦𐑙 𐑚𐑤𐑨𐑒 𐑣𐑳𐑤 𐑝 𐑞 𐑖𐑦𐑐; 𐑕𐑰𐑥𐑦𐑙𐑤𐑦 𐑕𐑰𐑦𐑙 𐑦𐑯 𐑦𐑑 𐑞 𐑕𐑹𐑕 𐑝 𐑷𐑤 𐑣𐑦𐑟 𐑐𐑻𐑕𐑦𐑒𐑿𐑖𐑩𐑯𐑟; 𐑚𐑦𐑔𐑦𐑙𐑒𐑦𐑙 𐑦𐑑—𐑦𐑑 𐑥𐑱 𐑚𐑰—𐑩 𐑤𐑸𐑡𐑼 𐑯 𐑯𐑴𐑚𐑩𐑤𐑼 𐑓𐑴; 𐑝 𐑩 𐑕𐑳𐑛𐑩𐑯, 𐑣𐑰 𐑚𐑹 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑦𐑑𐑕 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑐𐑮𐑬, 𐑕𐑥𐑲𐑑𐑦𐑙 𐑣𐑦𐑟 𐑡𐑷𐑟 𐑩𐑥𐑦𐑛 𐑓𐑲𐑼𐑦 𐑖𐑬𐑼𐑟 𐑝 𐑓𐑴𐑥.

𐑱𐑣𐑨𐑚 𐑕𐑑𐑨𐑜𐑼𐑛; 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑕𐑥𐑴𐑑 𐑣𐑦𐑟 𐑓𐑪𐑮𐑦𐑛. “𐑲 𐑜𐑮𐑴 𐑚𐑤𐑲𐑯𐑛; 𐑣𐑨𐑯𐑛𐑟! 𐑕𐑑𐑮𐑧𐑗 𐑬𐑑 𐑚𐑦𐑓𐑹 𐑥𐑰 𐑞𐑨𐑑 𐑲 𐑥𐑱 𐑘𐑧𐑑 𐑜𐑮𐑴𐑐 𐑥𐑲 𐑢𐑱. ·𐑦𐑕'𐑑 𐑯𐑲𐑑?”

“𐑞 𐑢𐑱𐑤! 𐑞 𐑖𐑦𐑐!” 𐑒𐑮𐑲𐑛 𐑞 𐑒𐑮𐑦𐑯𐑡𐑦𐑙 𐑹𐑟𐑥𐑧𐑯.

𐑚𐑳𐑑 𐑨𐑟 𐑞 𐑹𐑟𐑥𐑧𐑯 𐑝𐑲𐑩𐑤𐑩𐑯𐑑𐑤𐑦 𐑓𐑹𐑕𐑑 𐑞𐑺 𐑚𐑴𐑑 𐑔𐑮𐑵 𐑞 𐑕𐑤𐑧𐑡-𐑣𐑨𐑥𐑼𐑦𐑙 𐑕𐑰𐑟, 𐑞 𐑚𐑦𐑓𐑹 𐑢𐑱𐑤-𐑕𐑥𐑦𐑑𐑩𐑯 𐑚𐑬-𐑧𐑯𐑛𐑟 𐑝 𐑑𐑵 𐑐𐑤𐑨𐑙𐑒𐑕 𐑚𐑻𐑕𐑑 𐑔𐑮𐑵, 𐑯 𐑦𐑯 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑷𐑤𐑥𐑴𐑕𐑑, 𐑞 𐑑𐑧𐑥𐑐𐑼𐑺𐑦𐑤𐑦 𐑛𐑦𐑕𐑱𐑚𐑩𐑤𐑛 𐑚𐑴𐑑 𐑤𐑱 𐑯𐑽𐑤𐑦 𐑤𐑧𐑝𐑩𐑤 𐑢𐑦𐑞 𐑞 𐑢𐑱𐑝𐑟; 𐑦𐑑𐑕 𐑣𐑭𐑓-𐑢𐑱𐑛𐑦𐑙, 𐑕𐑐𐑤𐑨𐑖𐑦𐑙 𐑒𐑮𐑵, 𐑑𐑮𐑲𐑦𐑙 𐑣𐑸𐑛 𐑑 𐑕𐑑𐑪𐑐 𐑞 𐑜𐑨𐑐 𐑯 𐑚𐑱𐑤 𐑬𐑑 𐑞 𐑐𐑹𐑦𐑙 𐑢𐑷𐑑𐑼.

𐑥𐑰𐑯𐑑𐑲𐑥, 𐑓 𐑞𐑨𐑑 𐑢𐑳𐑯 𐑚𐑦𐑣𐑴𐑤𐑛𐑦𐑙 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴'𐑟 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛 𐑣𐑨𐑥𐑼 𐑮𐑦𐑥𐑱𐑯𐑛 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛; 𐑯 𐑞 𐑮𐑧𐑛 𐑓𐑤𐑨𐑜, 𐑣𐑭𐑓-𐑮𐑨𐑐𐑦𐑙 𐑣𐑦𐑥 𐑨𐑟 𐑢𐑦𐑞 𐑩 𐑐𐑤𐑨𐑛, 𐑞𐑧𐑯 𐑕𐑑𐑮𐑰𐑥𐑛 𐑦𐑑𐑕𐑧𐑤𐑓 𐑕𐑑𐑮𐑱𐑑 𐑬𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑨𐑟 𐑣𐑦𐑟 𐑴𐑯 𐑓𐑹𐑢𐑼𐑛-𐑓𐑤𐑴𐑦𐑙 𐑣𐑸𐑑; 𐑢𐑲𐑤 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑯 ·𐑕𐑑𐑳𐑚, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑴𐑕𐑐𐑮𐑦𐑑 𐑚𐑦𐑯𐑰𐑔, 𐑒𐑷𐑑 𐑕𐑲𐑑 𐑝 𐑞 𐑛𐑬𐑯-𐑒𐑳𐑥𐑦𐑙 𐑥𐑪𐑯𐑕𐑑𐑼 𐑡𐑳𐑕𐑑 𐑨𐑟 𐑕𐑵𐑯 𐑨𐑟 𐑣𐑰.

“𐑞 𐑢𐑱𐑤, 𐑞 𐑢𐑱𐑤! 𐑳𐑐 𐑣𐑧𐑤𐑥, 𐑳𐑐 𐑣𐑧𐑤𐑥! 𐑴, 𐑷𐑤 𐑘𐑰 𐑕𐑢𐑰𐑑 𐑐𐑬𐑼𐑟 𐑝 𐑺, 𐑯𐑬 𐑣𐑳𐑜 𐑥𐑰 𐑒𐑤𐑴𐑕! 𐑤𐑧𐑑 𐑯𐑪𐑑 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑛𐑲, 𐑦𐑓 𐑛𐑲 𐑣𐑰 𐑥𐑳𐑕𐑑, 𐑦𐑯 𐑩 𐑢𐑫𐑥𐑩𐑯'𐑟 𐑓𐑱𐑯𐑑𐑦𐑙 𐑓𐑦𐑑. 𐑳𐑐 𐑣𐑧𐑤𐑥, 𐑲 𐑕𐑱—𐑘𐑰 𐑓𐑵𐑤𐑟, 𐑞 𐑡𐑷! 𐑞 𐑡𐑷! 𐑦𐑟 𐑞𐑦𐑕 𐑞 𐑧𐑯𐑛 𐑝 𐑷𐑤 𐑥𐑲 𐑚𐑻𐑕𐑑𐑦𐑙 𐑐𐑮𐑱𐑼𐑟? 𐑷𐑤 𐑥𐑲 𐑤𐑲𐑓-𐑤𐑪𐑙 𐑓𐑦𐑛𐑧𐑤𐑦𐑑𐑦𐑟? 𐑴, ·𐑱𐑣𐑨𐑚, ·𐑱𐑣𐑨𐑚, 𐑤𐑴, 𐑞𐑲 𐑢𐑻𐑒. 𐑕𐑑𐑧𐑛𐑦! 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯, 𐑕𐑑𐑧𐑛𐑦. 𐑯𐑱, 𐑯𐑱! 𐑳𐑐 𐑣𐑧𐑤𐑥 𐑩𐑜𐑱𐑯! 𐑣𐑰 𐑑𐑻𐑯𐑟 𐑑 𐑥𐑰𐑑 𐑳𐑕! 𐑴, 𐑣𐑦𐑟 𐑳𐑯𐑩𐑐𐑰𐑟𐑩𐑚𐑩𐑤 𐑚𐑮𐑬 𐑛𐑮𐑲𐑝𐑟 𐑪𐑯 𐑑𐑩𐑢𐑹𐑛𐑟 𐑢𐑳𐑯, 𐑣𐑵𐑟 𐑛𐑿𐑑𐑦 𐑑𐑧𐑤𐑟 𐑣𐑦𐑥 𐑣𐑰 𐑒𐑨𐑯𐑪𐑑 𐑛𐑦𐑐𐑸𐑑. 𐑥𐑲 ·𐑜𐑪𐑛, 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑥𐑰 𐑯𐑬!”

𐑞 𐑣𐑸𐑐𐑵𐑯 𐑢𐑪𐑟 𐑛𐑸𐑑𐑩𐑛; 𐑞 𐑕𐑑𐑮𐑦𐑒𐑩𐑯 𐑢𐑱𐑤 𐑓𐑤𐑵 𐑓𐑹𐑢𐑼𐑛; 𐑢𐑦𐑞 𐑦𐑜𐑯𐑲𐑑𐑦𐑙 𐑝𐑦𐑤𐑪𐑕𐑦𐑑𐑦 𐑞 𐑤𐑲𐑯 𐑮𐑨𐑯 𐑔𐑮𐑵 𐑞 𐑜𐑮𐑵𐑝𐑟;—𐑮𐑨𐑯 𐑓𐑬𐑤. 𐑱𐑣𐑨𐑚 𐑕𐑑𐑵𐑐𐑑 𐑑 𐑒𐑤𐑽 𐑦𐑑; 𐑣𐑰 𐑛𐑦𐑛 𐑒𐑤𐑽 𐑦𐑑; 𐑚𐑳𐑑 𐑞 𐑓𐑤𐑲𐑦𐑙 𐑑𐑻𐑯 𐑒𐑷𐑑 𐑣𐑦𐑥 𐑮𐑬𐑯𐑛 𐑞 𐑯𐑧𐑒, 𐑯 𐑝𐑶𐑕𐑤𐑩𐑕𐑤𐑦 𐑨𐑟 ·𐑑𐑻𐑒𐑦𐑖 𐑥𐑿𐑑𐑕 𐑚𐑴𐑕𐑑𐑮𐑦𐑙 𐑞𐑺 𐑝𐑦𐑒𐑑𐑦𐑥, 𐑣𐑰 𐑢𐑪𐑟 𐑖𐑪𐑑 𐑬𐑑 𐑝 𐑞 𐑚𐑴𐑑, 𐑺 𐑞 𐑒𐑮𐑵 𐑯𐑿 𐑣𐑰 𐑢𐑪𐑟 𐑜𐑪𐑯. 𐑯𐑧𐑒𐑕𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑞 𐑣𐑧𐑝𐑦 𐑲-𐑕𐑐𐑤𐑲𐑕 𐑦𐑯 𐑞 𐑮𐑴𐑐'𐑕 𐑓𐑲𐑯𐑩𐑤 𐑧𐑯𐑛 𐑓𐑤𐑵 𐑬𐑑 𐑝 𐑞 𐑕𐑑𐑸𐑒-𐑧𐑥𐑐𐑑𐑦 𐑑𐑳𐑚, 𐑯𐑪𐑒𐑑 𐑛𐑬𐑯 𐑩𐑯 𐑹𐑟𐑥𐑩𐑯, 𐑯 𐑕𐑥𐑲𐑑𐑦𐑙 𐑞 𐑕𐑰, 𐑛𐑦𐑕𐑩𐑐𐑽𐑛 𐑦𐑯 𐑦𐑑𐑕 𐑛𐑧𐑐𐑔𐑕.

𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑞 𐑑𐑮𐑭𐑯𐑕𐑑 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵 𐑕𐑑𐑫𐑛 𐑕𐑑𐑦𐑤; 𐑞𐑧𐑯 𐑑𐑻𐑯𐑛. “𐑞 𐑖𐑦𐑐? 𐑜𐑮𐑱𐑑 ·𐑜𐑪𐑛, 𐑢𐑺 𐑦𐑟 𐑞 𐑖𐑦𐑐?” 𐑕𐑵𐑯 𐑞𐑱 𐑔𐑮𐑵 𐑛𐑦𐑥, 𐑚𐑦𐑢𐑦𐑤𐑛𐑼𐑦𐑙 𐑥𐑰𐑛𐑾𐑥𐑟 𐑕𐑷 𐑣𐑻 𐑕𐑲𐑛𐑤𐑪𐑙 𐑓𐑱𐑛𐑦𐑙 𐑓𐑨𐑯𐑑𐑩𐑥, 𐑨𐑟 𐑦𐑯 𐑞 𐑜𐑨𐑖𐑩𐑕 ·𐑓𐑱𐑑𐑩 ·𐑥𐑹𐑜𐑩𐑯𐑩; 𐑴𐑯𐑤𐑦 𐑞 𐑳𐑐𐑼𐑥𐑴𐑕𐑑 𐑥𐑭𐑕𐑑𐑕 𐑬𐑑 𐑝 𐑢𐑷𐑑𐑼; 𐑢𐑲𐑤 𐑓𐑦𐑒𐑕𐑑 𐑚𐑲 𐑦𐑯𐑓𐑨𐑗𐑵𐑱𐑖𐑩𐑯, 𐑹 𐑓𐑦𐑛𐑧𐑤𐑦𐑑𐑦, 𐑹 𐑓𐑱𐑑, 𐑑 𐑞𐑺 𐑢𐑳𐑯𐑕 𐑤𐑪𐑓𐑑𐑦 𐑐𐑻𐑗𐑩𐑟, 𐑞 𐑐𐑱𐑜𐑩𐑯 𐑣𐑸𐑐𐑵𐑯𐑽𐑟 𐑕𐑑𐑦𐑤 𐑥𐑱𐑯𐑑𐑱𐑯𐑛 𐑞𐑺 𐑕𐑦𐑙𐑒𐑦𐑙 𐑤𐑫𐑒𐑬𐑑𐑕 𐑪𐑯 𐑞 𐑕𐑰. 𐑯 𐑯𐑬, 𐑒𐑩𐑯𐑕𐑧𐑯𐑑𐑮𐑦𐑒 𐑕𐑻𐑒𐑩𐑤𐑟 𐑕𐑰𐑟𐑛 𐑞 𐑤𐑴𐑯 𐑚𐑴𐑑 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑯 𐑷𐑤 𐑦𐑑𐑕 𐑒𐑮𐑵, 𐑯 𐑰𐑗 𐑓𐑤𐑴𐑑𐑦𐑙 𐑹, 𐑯 𐑧𐑝𐑮𐑦 𐑤𐑭𐑯𐑕-𐑐𐑴𐑤, 𐑯 𐑕𐑐𐑦𐑯𐑦𐑙, 𐑨𐑯𐑦𐑥𐑩𐑑 𐑯 𐑦𐑯𐑨𐑯𐑦𐑥𐑩𐑑, 𐑷𐑤 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛 𐑦𐑯 𐑢𐑳𐑯 𐑝𐑹𐑑𐑧𐑒𐑕, 𐑒𐑨𐑮𐑦𐑛 𐑞 𐑕𐑥𐑷𐑤𐑩𐑕𐑑 𐑗𐑦𐑐 𐑝 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑬𐑑 𐑝 𐑕𐑲𐑑.

𐑚𐑳𐑑 𐑨𐑟 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑧𐑤𐑥𐑦𐑙𐑟 𐑦𐑯𐑑𐑼𐑥𐑦𐑒𐑕𐑦𐑙𐑤𐑦 𐑐𐑹𐑛 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑴𐑝𐑼 𐑞 𐑕𐑳𐑙𐑒𐑩𐑯 𐑣𐑧𐑛 𐑝 𐑞 ·𐑦𐑯𐑛𐑾𐑯 𐑨𐑑 𐑞 𐑥𐑱𐑯𐑥𐑭𐑕𐑑, 𐑤𐑰𐑝𐑦𐑙 𐑩 𐑓𐑿 𐑦𐑯𐑗𐑩𐑟 𐑝 𐑞 𐑦𐑮𐑧𐑒𐑑 𐑕𐑐𐑸 𐑘𐑧𐑑 𐑝𐑦𐑟𐑩𐑚𐑩𐑤, 𐑑𐑩𐑜𐑧𐑞𐑼 𐑢𐑦𐑞 𐑤𐑪𐑙 𐑕𐑑𐑮𐑰𐑥𐑦𐑙 𐑘𐑸𐑛𐑟 𐑝 𐑞 𐑓𐑤𐑨𐑜, 𐑢𐑦𐑗 𐑒𐑭𐑥𐑤𐑦 𐑳𐑯𐑡𐑩𐑤𐑱𐑑𐑩𐑛, 𐑢𐑦𐑞 𐑲𐑮𐑪𐑯𐑦𐑒𐑩𐑤 𐑒𐑴𐑦𐑯𐑕𐑲𐑛𐑦𐑙𐑟, 𐑴𐑝𐑼 𐑞 𐑛𐑦𐑕𐑑𐑮𐑶𐑦𐑙 𐑚𐑦𐑤𐑴𐑟 𐑞𐑱 𐑷𐑤𐑥𐑴𐑕𐑑 𐑑𐑳𐑗𐑑;—𐑨𐑑 𐑞𐑨𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑩 𐑮𐑧𐑛 𐑸𐑥 𐑯 𐑩 𐑣𐑨𐑥𐑼 𐑣𐑪𐑝𐑼𐑛 𐑚𐑨𐑒𐑢𐑼𐑛𐑤𐑦 𐑳𐑐𐑤𐑦𐑓𐑑𐑩𐑛 𐑦𐑯 𐑞 𐑴𐑐𐑩𐑯 𐑺, 𐑦𐑯 𐑞 𐑨𐑒𐑑 𐑝 𐑯𐑱𐑤𐑦𐑙 𐑞 𐑓𐑤𐑨𐑜 𐑓𐑭𐑕𐑑𐑼 𐑯 𐑘𐑧𐑑 𐑓𐑭𐑕𐑑𐑼 𐑑 𐑞 𐑕𐑩𐑚𐑕𐑲𐑛𐑦𐑙 𐑕𐑐𐑸. 𐑩 𐑕𐑒𐑲-𐑣𐑷𐑒 𐑞𐑨𐑑 𐑑𐑷𐑯𐑑𐑦𐑙𐑤𐑦 𐑣𐑨𐑛 𐑓𐑪𐑤𐑴𐑛 𐑞 𐑥𐑱𐑯-𐑑𐑮𐑳𐑒 𐑛𐑬𐑯𐑢𐑼𐑛𐑟 𐑓𐑮𐑪𐑥 𐑦𐑑𐑕 𐑯𐑨𐑗𐑼𐑩𐑤 𐑣𐑴𐑥 𐑩𐑥𐑳𐑙 𐑞 𐑕𐑑𐑸𐑟, 𐑐𐑧𐑒𐑦𐑙 𐑨𐑑 𐑞 𐑓𐑤𐑨𐑜, 𐑯 𐑦𐑯𐑒𐑩𐑥𐑴𐑛𐑦𐑙 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑞𐑺; 𐑞𐑦𐑕 𐑚𐑻𐑛 𐑯𐑬 𐑗𐑭𐑯𐑕𐑑 𐑑 𐑦𐑯𐑑𐑼𐑕𐑧𐑐𐑑 𐑦𐑑𐑕 𐑚𐑮𐑷𐑛 𐑓𐑤𐑳𐑑𐑼𐑦𐑙 𐑢𐑦𐑙 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑣𐑨𐑥𐑼 𐑯 𐑞 𐑢𐑫𐑛; 𐑯 𐑕𐑦𐑥𐑩𐑤𐑑𐑱𐑯𐑾𐑕𐑤𐑦 𐑓𐑰𐑤𐑦𐑙 𐑞𐑨𐑑 ·𐑧𐑔𐑽𐑾𐑤 𐑔𐑮𐑦𐑤, 𐑞 𐑕𐑩𐑚𐑥𐑻𐑡𐑛 𐑕𐑨𐑝𐑦𐑡 𐑚𐑦𐑯𐑰𐑔, 𐑦𐑯 𐑣𐑦𐑟 𐑛𐑧𐑔-𐑜𐑭𐑕𐑐, 𐑒𐑧𐑐𐑑 𐑣𐑦𐑟 𐑣𐑨𐑥𐑼 𐑓𐑮𐑴𐑟𐑩𐑯 𐑞𐑺; 𐑯 𐑕𐑴 𐑞 𐑚𐑻𐑛 𐑝 𐑣𐑧𐑝𐑩𐑯, 𐑢𐑦𐑞 𐑸𐑒𐑱𐑯𐑡𐑩𐑤𐑦𐑒 𐑖𐑮𐑰𐑒𐑕, 𐑯 𐑣𐑦𐑟 𐑦𐑥𐑐𐑽𐑾𐑤 𐑚𐑰𐑒 𐑔𐑮𐑳𐑕𐑑 𐑳𐑐𐑢𐑼𐑛𐑟, 𐑯 𐑣𐑦𐑟 𐑣𐑴𐑤 𐑒𐑨𐑐𐑑𐑦𐑝 𐑓𐑹𐑥 𐑓𐑴𐑤𐑛𐑩𐑛 𐑦𐑯 𐑞 𐑓𐑤𐑨𐑜 𐑝 ·𐑱𐑣𐑨𐑚, 𐑢𐑧𐑯𐑑 𐑛𐑬𐑯 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑖𐑦𐑐, 𐑢𐑦𐑗, 𐑤𐑲𐑒 ·𐑕𐑱𐑑𐑩𐑯, 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑕𐑦𐑙𐑒 𐑑 𐑣𐑧𐑤 𐑑𐑦𐑤 𐑖𐑰 𐑣𐑨𐑛 𐑛𐑮𐑨𐑜𐑛 𐑩 𐑤𐑦𐑝𐑦𐑙 𐑐𐑸𐑑 𐑝 𐑣𐑧𐑝𐑩𐑯 𐑩𐑤𐑪𐑙 𐑢𐑦𐑞 𐑣𐑻, 𐑯 𐑣𐑧𐑤𐑥𐑩𐑑𐑩𐑛 𐑣𐑻𐑕𐑧𐑤𐑓 𐑢𐑦𐑞 𐑦𐑑.

𐑯𐑬 𐑕𐑥𐑷𐑤 𐑓𐑬𐑤𐑟 𐑓𐑤𐑵 𐑕𐑒𐑮𐑰𐑥𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑘𐑧𐑑 𐑘𐑷𐑯𐑦𐑙 𐑜𐑳𐑤𐑓; 𐑩 𐑕𐑳𐑤𐑩𐑯 𐑢𐑲𐑑 𐑕𐑻𐑓 𐑚𐑰𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑦𐑑𐑕 𐑕𐑑𐑰𐑐 𐑕𐑲𐑛𐑟; 𐑞𐑧𐑯 𐑷𐑤 𐑒𐑩𐑤𐑨𐑐𐑕𐑑, 𐑯 𐑞 𐑜𐑮𐑱𐑑 𐑖𐑮𐑬𐑛 𐑝 𐑞 𐑕𐑰 𐑮𐑴𐑤𐑛 𐑪𐑯 𐑨𐑟 𐑦𐑑 𐑮𐑴𐑤𐑛 𐑓𐑲𐑝 𐑔𐑬𐑟𐑩𐑯𐑛 𐑘𐑽𐑟 𐑩𐑜𐑴.
