·𐑗𐑨𐑐𐑑𐑼 133. 𐑞 ·𐑗𐑱𐑕—𐑓𐑻𐑕𐑑 𐑛𐑱.

𐑞𐑨𐑑 𐑯𐑲𐑑, 𐑦𐑯 𐑞 𐑥𐑦𐑛-𐑢𐑪𐑗, 𐑢𐑧𐑯 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯—𐑨𐑟 𐑣𐑦𐑟 𐑢𐑴𐑯𐑑 𐑨𐑑 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟—𐑕𐑑𐑧𐑐𐑑 𐑓𐑹𐑔 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤 𐑦𐑯 𐑢𐑦𐑗 𐑣𐑰 𐑤𐑰𐑯𐑛, 𐑯 𐑢𐑧𐑯𐑑 𐑑 𐑣𐑦𐑟 𐑐𐑦𐑝𐑩𐑑-𐑣𐑴𐑤, 𐑣𐑰 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑔𐑮𐑳𐑕𐑑 𐑬𐑑 𐑣𐑦𐑟 𐑓𐑱𐑕 𐑓𐑽𐑕𐑤𐑦, 𐑕𐑯𐑳𐑓𐑦𐑙 𐑳𐑐 𐑞 𐑕𐑰 𐑺 𐑨𐑟 𐑩 𐑕𐑩𐑜𐑱𐑖𐑩𐑕 𐑖𐑦𐑐'𐑕 𐑛𐑪𐑜 𐑢𐑦𐑤, 𐑦𐑯 𐑛𐑮𐑷𐑦𐑙 𐑯𐑲 𐑑 𐑕𐑳𐑥 𐑚𐑸𐑚𐑼𐑩𐑕 𐑲𐑤. 𐑣𐑰 𐑛𐑦𐑒𐑤𐑺𐑛 𐑞𐑨𐑑 𐑩 𐑢𐑱𐑤 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑯𐑽. 𐑕𐑵𐑯 𐑞𐑨𐑑 𐑐𐑦𐑒𐑿𐑤𐑽 𐑴𐑛𐑼, 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑑 𐑩 𐑜𐑮𐑱𐑑 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑜𐑦𐑝𐑩𐑯 𐑓𐑹𐑔 𐑚𐑲 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤, 𐑢𐑪𐑟 𐑐𐑨𐑤𐑐𐑩𐑚𐑩𐑤 𐑑 𐑷𐑤 𐑞 𐑢𐑪𐑗; 𐑯𐑹 𐑢𐑪𐑟 𐑧𐑯𐑦 𐑥𐑨𐑮𐑦𐑯𐑼 𐑕𐑼𐑐𐑮𐑲𐑟𐑛 𐑢𐑧𐑯, 𐑭𐑓𐑑𐑼 𐑦𐑯𐑕𐑐𐑧𐑒𐑑𐑦𐑙 𐑞 𐑒𐑳𐑥𐑐𐑩𐑕, 𐑯 𐑞𐑧𐑯 𐑞 𐑛𐑪𐑜-𐑝𐑱𐑯, 𐑯 𐑞𐑧𐑯 𐑨𐑕𐑼𐑑𐑱𐑯𐑦𐑙 𐑞 𐑐𐑮𐑦𐑕𐑲𐑕 𐑚𐑺𐑦𐑙 𐑝 𐑞 𐑴𐑛𐑼 𐑨𐑟 𐑯𐑽𐑤𐑦 𐑨𐑟 𐑐𐑪𐑕𐑩𐑚𐑩𐑤, ·𐑱𐑣𐑨𐑚 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑹𐑛𐑼𐑛 𐑞 𐑖𐑦𐑐'𐑕 𐑒𐑹𐑕 𐑑 𐑚𐑰 𐑕𐑤𐑲𐑑𐑤𐑦 𐑷𐑤𐑑𐑼𐑛, 𐑯 𐑞 𐑕𐑱𐑤 𐑑 𐑚𐑰 𐑖𐑹𐑑𐑩𐑯𐑛.

𐑞 𐑩𐑒𐑿𐑑 𐑐𐑪𐑤𐑩𐑕𐑦 𐑛𐑦𐑒𐑑𐑱𐑑𐑦𐑙 𐑞𐑰𐑟 𐑥𐑵𐑝𐑥𐑩𐑯𐑑𐑕 𐑢𐑪𐑟 𐑕𐑩𐑓𐑦𐑖𐑩𐑯𐑑𐑤𐑦 𐑝𐑦𐑯𐑛𐑦𐑒𐑱𐑑𐑩𐑛 𐑨𐑑 𐑛𐑱𐑚𐑮𐑱𐑒, 𐑚𐑲 𐑞 𐑕𐑲𐑑 𐑝 𐑩 𐑤𐑪𐑙 𐑕𐑤𐑰𐑒 𐑪𐑯 𐑞 𐑕𐑰 𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑯 𐑤𐑧𐑙𐑔𐑢𐑲𐑟 𐑩𐑣𐑧𐑛, 𐑕𐑥𐑵𐑞 𐑨𐑟 𐑶𐑤, 𐑯 𐑮𐑦𐑟𐑧𐑥𐑚𐑩𐑤𐑦𐑙 𐑦𐑯 𐑞 𐑐𐑤𐑰𐑑𐑩𐑛 𐑢𐑷𐑑𐑼𐑦 𐑮𐑦𐑙𐑒𐑩𐑤𐑟 𐑚𐑹𐑛𐑼𐑦𐑙 𐑦𐑑, 𐑞 𐑐𐑪𐑤𐑦𐑖𐑑 𐑥𐑦𐑑𐑨𐑤𐑦𐑒-𐑤𐑲𐑒 𐑥𐑸𐑒𐑕 𐑝 𐑕𐑳𐑥 𐑕𐑢𐑦𐑓𐑑 𐑑𐑲𐑛-𐑮𐑦𐑐, 𐑨𐑑 𐑞 𐑥𐑬𐑔 𐑝 𐑩 𐑛𐑰𐑐, 𐑮𐑨𐑐𐑦𐑛 𐑕𐑑𐑮𐑰𐑥.

“𐑥𐑨𐑯 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟! 𐑒𐑷𐑤 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟!”

𐑔𐑳𐑯𐑛𐑼𐑦𐑙 𐑢𐑦𐑞 𐑞 𐑚𐑳𐑑𐑕 𐑝 𐑔𐑮𐑰 𐑒𐑤𐑳𐑚𐑛 𐑣𐑨𐑯𐑛𐑕𐑐𐑲𐑒𐑕 𐑪𐑯 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤 𐑛𐑧𐑒, ·𐑛𐑨𐑜𐑵 𐑮𐑬𐑟𐑛 𐑞 𐑕𐑤𐑰𐑐𐑼𐑟 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑡𐑳𐑡𐑥𐑩𐑯𐑑 𐑒𐑤𐑨𐑐𐑕 𐑞𐑨𐑑 𐑞𐑱 𐑕𐑰𐑥𐑛 𐑑 𐑧𐑒𐑕𐑣𐑱𐑤 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤, 𐑕𐑴 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑱𐑯𐑾𐑕𐑤𐑦 𐑛𐑦𐑛 𐑞𐑱 𐑩𐑐𐑽 𐑢𐑦𐑞 𐑞𐑺 𐑒𐑤𐑴𐑞𐑟 𐑦𐑯 𐑞𐑺 𐑣𐑨𐑯𐑛𐑟.

“𐑢𐑪𐑑 𐑛'𐑘𐑰 𐑕𐑰?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑓𐑤𐑨𐑑𐑩𐑯𐑦𐑙 𐑣𐑦𐑟 𐑓𐑱𐑕 𐑑 𐑞 𐑕𐑒𐑲.

“𐑯𐑳𐑔𐑦𐑙, 𐑯𐑳𐑔𐑦𐑙 𐑕𐑻!” 𐑢𐑪𐑟 𐑞 𐑕𐑬𐑯𐑛 𐑣𐑱𐑤𐑦𐑙 𐑛𐑬𐑯 𐑦𐑯 𐑮𐑦𐑐𐑤𐑲.

“·𐑑'𐑜𐑨𐑤𐑩𐑯𐑑 𐑕𐑱𐑤𐑟!—𐑕𐑑𐑳𐑯𐑕𐑱𐑤𐑟! 𐑱𐑤𐑴 𐑯 𐑩𐑤𐑪𐑓𐑑, 𐑯 𐑪𐑯 𐑚𐑴𐑔 𐑕𐑲𐑛𐑟!”

𐑷𐑤 𐑕𐑱𐑤 𐑚𐑰𐑦𐑙 𐑕𐑧𐑑, 𐑣𐑰 𐑯𐑬 𐑒𐑭𐑕𐑑 𐑤𐑵𐑕 𐑞 𐑤𐑲𐑓-𐑤𐑲𐑯, 𐑮𐑦𐑟𐑻𐑝𐑛 𐑓 𐑕𐑢𐑱𐑦𐑙 𐑣𐑦𐑥 𐑑 𐑞 𐑥𐑱𐑯 𐑮𐑶𐑩𐑤-𐑥𐑭𐑕𐑑 𐑣𐑧𐑛; 𐑯 𐑦𐑯 𐑩 𐑓𐑿 𐑥𐑴𐑥𐑧𐑯𐑑𐑕 𐑞𐑱 𐑢𐑻 𐑣𐑶𐑕𐑑𐑦𐑙 𐑣𐑦𐑥 𐑞𐑦𐑞𐑼, 𐑢𐑧𐑯, 𐑢𐑲𐑤 𐑚𐑳𐑑 𐑑𐑵 𐑔𐑻𐑛𐑟 𐑝 𐑞 𐑢𐑱 𐑩𐑤𐑪𐑓𐑑, 𐑯 𐑢𐑲𐑤 𐑐𐑽𐑦𐑙 𐑩𐑣𐑧𐑛 𐑔𐑮𐑵 𐑞 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤 𐑝𐑱𐑒𐑩𐑯𐑕𐑦 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑥𐑱𐑯-𐑑𐑪𐑐-𐑕𐑱𐑤 𐑯 𐑑𐑪𐑐-𐑜𐑨𐑤𐑩𐑯𐑑-𐑕𐑱𐑤, 𐑣𐑰 𐑮𐑱𐑟𐑛 𐑩 𐑜𐑳𐑤-𐑤𐑲𐑒 𐑒𐑮𐑲 𐑦𐑯 𐑞 𐑺. “𐑞𐑺 𐑖𐑰 𐑚𐑤𐑴𐑟!—𐑞𐑺 𐑖𐑰 𐑚𐑤𐑴𐑟! 𐑩 𐑣𐑳𐑥𐑐 𐑤𐑲𐑒 𐑩 𐑕𐑯𐑴-𐑣𐑦𐑤! 𐑦𐑑 𐑦𐑟 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒!”

𐑓𐑲𐑼𐑛 𐑚𐑲 𐑞 𐑒𐑮𐑲 𐑢𐑦𐑗 𐑕𐑰𐑥𐑛 𐑕𐑦𐑥𐑩𐑤𐑑𐑱𐑯𐑾𐑕𐑤𐑦 𐑑𐑱𐑒𐑩𐑯 𐑳𐑐 𐑚𐑲 𐑞 𐑔𐑮𐑰 𐑤𐑫𐑒-𐑬𐑑𐑕, 𐑞 𐑥𐑧𐑯 𐑪𐑯 𐑛𐑧𐑒 𐑮𐑳𐑖𐑑 𐑑 𐑞 𐑮𐑦𐑜𐑦𐑙 𐑑 𐑚𐑦𐑣𐑴𐑤𐑛 𐑞 𐑓𐑱𐑥𐑩𐑕 𐑢𐑱𐑤 𐑞𐑱 𐑣𐑨𐑛 𐑕𐑴 𐑤𐑪𐑙 𐑚𐑰𐑯 𐑐𐑼𐑕𐑿𐑦𐑙. 𐑱𐑣𐑨𐑚 𐑣𐑨𐑛 𐑯𐑬 𐑜𐑱𐑯𐑛 𐑣𐑦𐑟 𐑓𐑲𐑯𐑩𐑤 𐑐𐑻𐑗, 𐑕𐑳𐑥 𐑓𐑰𐑑 𐑩𐑚𐑳𐑝 𐑞 𐑳𐑞𐑼 𐑤𐑫𐑒-𐑬𐑑𐑕, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑡𐑳𐑕𐑑 𐑚𐑦𐑯𐑰𐑔 𐑣𐑦𐑥 𐑪𐑯 𐑞 𐑒𐑨𐑐 𐑝 𐑞 𐑑𐑪𐑐-𐑜𐑨𐑤𐑩𐑯𐑑-𐑥𐑭𐑕𐑑, 𐑕𐑴 𐑞𐑨𐑑 𐑞 ·𐑦𐑯𐑛𐑾𐑯'𐑟 𐑣𐑧𐑛 𐑢𐑪𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑪𐑯 𐑩 𐑤𐑧𐑝𐑩𐑤 𐑢𐑦𐑞 ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑰𐑤. 𐑓𐑮𐑪𐑥 𐑞𐑦𐑕 𐑣𐑲𐑑 𐑞 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑯𐑬 𐑕𐑰𐑯 𐑕𐑳𐑥 𐑥𐑲𐑤 𐑹 𐑕𐑴 𐑩𐑣𐑧𐑛, 𐑨𐑑 𐑧𐑝𐑮𐑦 𐑮𐑴𐑤 𐑝 𐑞 𐑕𐑰 𐑮𐑦𐑝𐑰𐑤𐑦𐑙 𐑣𐑦𐑟 𐑣𐑲 𐑕𐑐𐑸𐑒𐑩𐑤𐑦𐑙 𐑣𐑳𐑥𐑐, 𐑯 𐑮𐑧𐑜𐑿𐑤𐑼𐑤𐑦 𐑡𐑧𐑑𐑦𐑙 𐑣𐑦𐑟 𐑕𐑲𐑤𐑩𐑯𐑑 𐑕𐑐𐑬𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑺. 𐑑 𐑞 𐑒𐑮𐑧𐑡𐑵𐑤𐑩𐑕 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑞 𐑕𐑱𐑥 𐑕𐑲𐑤𐑩𐑯𐑑 𐑕𐑐𐑬𐑑 𐑞𐑱 𐑣𐑨𐑛 𐑕𐑴 𐑤𐑪𐑙 𐑩𐑜𐑴 𐑚𐑦𐑣𐑧𐑤𐑛 𐑦𐑯 𐑞 𐑥𐑵𐑯𐑤𐑦𐑑 ·𐑩𐑑𐑤𐑨𐑯𐑑𐑦𐑒 𐑯 ·𐑦𐑯𐑛𐑾𐑯 ·𐑴𐑖𐑩𐑯𐑟.

“𐑯 𐑛𐑦𐑛 𐑯𐑳𐑯 𐑝 𐑘𐑰 𐑕𐑰 𐑦𐑑 𐑚𐑦𐑓𐑹?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑣𐑱𐑤𐑦𐑙 𐑞 𐑐𐑻𐑗𐑑 𐑥𐑧𐑯 𐑷𐑤 𐑼𐑬𐑯𐑛 𐑣𐑦𐑥.

“𐑲 𐑕𐑷 𐑣𐑦𐑥 𐑷𐑤𐑥𐑴𐑕𐑑 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑕𐑻, 𐑞𐑨𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑛𐑦𐑛, 𐑯 𐑲 𐑒𐑮𐑲𐑛 𐑬𐑑,” 𐑕𐑧𐑛 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴.

“𐑯𐑪𐑑 𐑞 𐑕𐑱𐑥 𐑦𐑯𐑕𐑑𐑩𐑯𐑑; 𐑯𐑪𐑑 𐑞 𐑕𐑱𐑥—𐑯𐑴, 𐑞 𐑛𐑳𐑚𐑤𐑵𐑯 𐑦𐑟 𐑥𐑲𐑯, ·𐑓𐑱𐑑 𐑮𐑦𐑟𐑻𐑝𐑛 𐑞 𐑛𐑳𐑚𐑤𐑵𐑯 𐑓 𐑥𐑰. 𐑲 𐑴𐑯𐑤𐑦; 𐑯𐑳𐑯 𐑝 𐑘𐑰 𐑒𐑫𐑛 𐑣𐑨𐑝 𐑮𐑱𐑟𐑛 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑓𐑻𐑕𐑑. 𐑞𐑺 𐑖𐑰 𐑚𐑤𐑴𐑟!—𐑞𐑺 𐑖𐑰 𐑚𐑤𐑴𐑟!—𐑞𐑺 𐑖𐑰 𐑚𐑤𐑴𐑟! 𐑞𐑺 𐑩𐑜𐑱𐑯!—𐑞𐑺 𐑩𐑜𐑱𐑯!” 𐑣𐑰 𐑒𐑮𐑲𐑛, 𐑦𐑯 𐑤𐑪𐑙-𐑛𐑮𐑷𐑯, 𐑤𐑦𐑙𐑜𐑼𐑦𐑙, 𐑥𐑦𐑔𐑪𐑛𐑦𐑒 𐑑𐑴𐑯𐑟, 𐑩𐑑𐑿𐑯𐑛 𐑑 𐑞 𐑜𐑮𐑨𐑡𐑵𐑩𐑤 𐑐𐑮𐑴𐑤𐑪𐑙𐑦𐑙𐑟 𐑝 𐑞 𐑢𐑱𐑤'𐑟 𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑡𐑧𐑑𐑕. “𐑣𐑰'𐑟 𐑜𐑴𐑦𐑙 𐑑 𐑕𐑬𐑯𐑛! 𐑦𐑯 𐑕𐑑𐑳𐑯𐑕𐑱𐑤𐑟! 𐑛𐑬𐑯 𐑑𐑪𐑐-𐑜𐑨𐑤𐑩𐑯𐑑-𐑕𐑱𐑤𐑟! 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕. ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑮𐑦𐑥𐑧𐑥𐑚𐑼, 𐑕𐑑𐑱 𐑪𐑯 𐑚𐑹𐑛, 𐑯 𐑒𐑰𐑐 𐑞 𐑖𐑦𐑐. 𐑣𐑧𐑤𐑥 𐑞𐑺! 𐑤𐑳𐑓, 𐑤𐑳𐑓 𐑩 𐑐𐑶𐑯𐑑! 𐑕𐑴; 𐑕𐑑𐑧𐑛𐑦, 𐑥𐑨𐑯, 𐑕𐑑𐑧𐑛𐑦! 𐑞𐑺 𐑜𐑴 𐑓𐑤𐑵𐑒𐑕! 𐑯𐑴, 𐑯𐑴; 𐑴𐑯𐑤𐑦 𐑚𐑤𐑨𐑒 𐑢𐑷𐑑𐑼! 𐑷𐑤 𐑮𐑧𐑛𐑦 𐑞 𐑚𐑴𐑑𐑕 𐑞𐑺? 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲, 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲! 𐑤𐑴𐑼 𐑥𐑰, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑤𐑴𐑼, 𐑤𐑴𐑼,—𐑒𐑢𐑦𐑒, 𐑒𐑢𐑦𐑒𐑼!” 𐑯 𐑣𐑰 𐑕𐑤𐑦𐑛 𐑔𐑮𐑵 𐑞 𐑺 𐑑 𐑞 𐑛𐑧𐑒.

“𐑣𐑰 𐑦𐑟 𐑣𐑧𐑛𐑦𐑙 𐑕𐑑𐑮𐑱𐑑 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑕𐑻,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, “𐑮𐑲𐑑 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑳𐑕; 𐑒𐑨𐑯𐑪𐑑 𐑣𐑨𐑝 𐑕𐑰𐑯 𐑞 𐑖𐑦𐑐 𐑘𐑧𐑑.”

“𐑚𐑰 𐑛𐑳𐑥, 𐑥𐑨𐑯! 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑞 𐑚𐑮𐑱𐑕𐑩𐑟! 𐑣𐑸𐑛 𐑛𐑬𐑯 𐑞 𐑣𐑧𐑤𐑥!—𐑚𐑮𐑱𐑕 𐑳𐑐! 𐑖𐑦𐑝𐑼 𐑣𐑻!—𐑖𐑦𐑝𐑼 𐑣𐑻!—𐑕𐑴; 𐑢𐑧𐑤 𐑞𐑨𐑑! 𐑚𐑴𐑑𐑕, 𐑚𐑴𐑑𐑕!”

𐑕𐑵𐑯 𐑷𐑤 𐑞 𐑚𐑴𐑑𐑕 𐑚𐑳𐑑 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑢𐑻 𐑛𐑮𐑪𐑐𐑑; 𐑷𐑤 𐑞 𐑚𐑴𐑑-𐑕𐑱𐑤𐑟 𐑕𐑧𐑑—𐑷𐑤 𐑞 𐑐𐑨𐑛𐑩𐑤𐑟 𐑐𐑤𐑲𐑦𐑙; 𐑢𐑦𐑞 𐑮𐑦𐑐𐑩𐑤𐑦𐑙 𐑕𐑢𐑦𐑓𐑑𐑯𐑩𐑕, 𐑖𐑵𐑑𐑦𐑙 𐑑 𐑤𐑰𐑢𐑼𐑛; 𐑯 ·𐑱𐑣𐑨𐑚 𐑣𐑧𐑛𐑦𐑙 𐑞 𐑪𐑯𐑕𐑧𐑑. 𐑩 𐑐𐑱𐑤, 𐑛𐑧𐑔-𐑜𐑤𐑦𐑥𐑼 𐑤𐑦𐑑 𐑳𐑐 ·𐑓𐑧𐑛𐑩𐑤𐑩'𐑟 𐑕𐑳𐑙𐑒𐑩𐑯 𐑲𐑟; 𐑩 𐑣𐑦𐑛𐑾𐑕 𐑥𐑴𐑖𐑩𐑯 𐑯𐑷𐑛 𐑣𐑦𐑟 𐑥𐑬𐑔.

𐑤𐑲𐑒 𐑯𐑶𐑟𐑤𐑩𐑕 𐑯𐑷𐑑𐑦𐑤𐑩𐑕 𐑖𐑧𐑤𐑟, 𐑞𐑺 𐑤𐑲𐑑 𐑐𐑮𐑬𐑟 𐑕𐑐𐑧𐑛 𐑔𐑮𐑵 𐑞 𐑕𐑰; 𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑕𐑤𐑴𐑤𐑦 𐑞𐑱 𐑯𐑽𐑛 𐑞 𐑓𐑴. 𐑨𐑟 𐑞𐑱 𐑯𐑽𐑛 𐑣𐑦𐑥, 𐑞 𐑴𐑖𐑩𐑯 𐑜𐑮𐑵 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑕𐑥𐑵𐑞; 𐑕𐑰𐑥𐑛 𐑛𐑮𐑷𐑦𐑙 𐑩 𐑒𐑸𐑐𐑩𐑑 𐑴𐑝𐑼 𐑦𐑑𐑕 𐑢𐑱𐑝𐑟; 𐑕𐑰𐑥𐑛 𐑩 𐑯𐑵𐑯-𐑥𐑧𐑛𐑴, 𐑕𐑴 𐑕𐑼𐑰𐑯𐑤𐑦 𐑦𐑑 𐑕𐑐𐑮𐑧𐑛. 𐑨𐑑 𐑤𐑧𐑙𐑔 𐑞 𐑚𐑮𐑧𐑔𐑤𐑩𐑕 𐑣𐑳𐑯𐑑𐑼 𐑒𐑱𐑥 𐑕𐑴 𐑯𐑲 𐑣𐑦𐑟 𐑕𐑰𐑥𐑦𐑙𐑤𐑦 𐑳𐑯𐑕𐑩𐑕𐑐𐑧𐑒𐑑𐑦𐑙 𐑐𐑮𐑱, 𐑞𐑨𐑑 𐑣𐑦𐑟 𐑦𐑯𐑑𐑲𐑼 𐑛𐑨𐑟𐑩𐑤𐑦𐑙 𐑣𐑳𐑥𐑐 𐑢𐑪𐑟 𐑛𐑦𐑕𐑑𐑦𐑙𐑒𐑑𐑤𐑦 𐑝𐑦𐑟𐑩𐑚𐑩𐑤, 𐑕𐑤𐑲𐑛𐑦𐑙 𐑩𐑤𐑪𐑙 𐑞 𐑕𐑰 𐑨𐑟 𐑦𐑓 𐑩𐑯 𐑲𐑕𐑩𐑤𐑱𐑑𐑩𐑛 𐑔𐑦𐑙, 𐑯 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤𐑦 𐑕𐑧𐑑 𐑦𐑯 𐑩 𐑮𐑦𐑝𐑪𐑤𐑝𐑦𐑙 𐑮𐑦𐑙 𐑝 𐑓𐑲𐑯𐑩𐑕𐑑, 𐑓𐑤𐑰𐑕𐑦, 𐑜𐑮𐑰𐑯𐑦𐑖 𐑓𐑴𐑥. 𐑣𐑰 𐑕𐑷 𐑞 𐑝𐑭𐑕𐑑, 𐑦𐑯𐑝𐑪𐑤𐑝𐑛 𐑮𐑦𐑙𐑒𐑩𐑤𐑟 𐑝 𐑞 𐑕𐑤𐑲𐑑𐑤𐑦 𐑐𐑮𐑩𐑡𐑧𐑒𐑑𐑦𐑙 𐑣𐑧𐑛 𐑚𐑦𐑘𐑪𐑯𐑛. 𐑚𐑦𐑓𐑹 𐑦𐑑, 𐑓𐑸 𐑬𐑑 𐑪𐑯 𐑞 𐑕𐑪𐑓𐑑 ·𐑑𐑻𐑒𐑦𐑖-𐑮𐑳𐑜𐑩𐑛 𐑢𐑷𐑑𐑼𐑟, 𐑢𐑧𐑯𐑑 𐑞 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑢𐑲𐑑 𐑖𐑨𐑛𐑴 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛, 𐑥𐑦𐑤𐑒𐑦 𐑓𐑪𐑮𐑦𐑛, 𐑩 𐑥𐑿𐑟𐑦𐑒𐑩𐑤 𐑮𐑦𐑐𐑩𐑤𐑦𐑙 𐑐𐑤𐑱𐑓𐑩𐑤𐑦 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑦𐑙 𐑞 𐑖𐑱𐑛; 𐑯 𐑚𐑦𐑣𐑲𐑯𐑛, 𐑞 𐑚𐑤𐑵 𐑢𐑷𐑑𐑼𐑟 𐑦𐑯𐑑𐑼𐑗𐑱𐑯𐑡𐑩𐑚𐑤𐑦 𐑓𐑤𐑴𐑛 𐑴𐑝𐑼 𐑦𐑯𐑑𐑵 𐑞 𐑥𐑵𐑝𐑦𐑙 𐑝𐑨𐑤𐑦 𐑝 𐑣𐑦𐑟 𐑕𐑑𐑧𐑛𐑦 𐑢𐑱𐑒; 𐑯 𐑪𐑯 𐑲𐑞𐑼 𐑣𐑨𐑯𐑛 𐑚𐑮𐑲𐑑 𐑚𐑳𐑚𐑩𐑤𐑟 𐑼𐑴𐑟 𐑯 𐑛𐑭𐑯𐑕𐑑 𐑚𐑲 𐑣𐑦𐑟 𐑕𐑲𐑛. 𐑚𐑳𐑑 𐑞𐑰𐑟 𐑢𐑻 𐑚𐑮𐑴𐑒𐑩𐑯 𐑩𐑜𐑱𐑯 𐑚𐑲 𐑞 𐑤𐑲𐑑 𐑑𐑴𐑟 𐑝 𐑣𐑳𐑯𐑛𐑮𐑩𐑛𐑟 𐑝 𐑜𐑱 𐑓𐑬𐑤 𐑕𐑪𐑓𐑑𐑤𐑦 𐑓𐑧𐑞𐑼𐑦𐑙 𐑞 𐑕𐑰, 𐑷𐑤𐑑𐑼𐑯𐑩𐑑 𐑢𐑦𐑞 𐑞𐑺 𐑓𐑦𐑑𐑓𐑩𐑤 𐑓𐑤𐑲𐑑; 𐑯 𐑤𐑲𐑒 𐑑 𐑕𐑳𐑥 𐑓𐑤𐑨𐑜-𐑕𐑑𐑭𐑓 𐑮𐑲𐑟𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑐𐑱𐑯𐑑𐑩𐑛 𐑣𐑳𐑤 𐑝 𐑩𐑯 𐑸𐑜𐑩𐑕𐑦, 𐑞 𐑑𐑷𐑤 𐑚𐑳𐑑 𐑖𐑨𐑑𐑼𐑛 𐑐𐑴𐑤 𐑝 𐑩 𐑮𐑰𐑕𐑩𐑯𐑑 𐑤𐑭𐑯𐑕 𐑐𐑮𐑩𐑡𐑧𐑒𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤'𐑟 𐑚𐑨𐑒; 𐑯 𐑨𐑑 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑢𐑳𐑯 𐑝 𐑞 𐑒𐑤𐑬𐑛 𐑝 𐑕𐑪𐑓𐑑-𐑑𐑴𐑛 𐑓𐑬𐑤𐑟 𐑣𐑪𐑝𐑼𐑦𐑙, 𐑯 𐑑 𐑯 𐑓𐑮𐑴 𐑕𐑒𐑦𐑥𐑦𐑙 𐑤𐑲𐑒 𐑩 𐑒𐑨𐑯𐑩𐑐𐑦 𐑴𐑝𐑼 𐑞 𐑓𐑦𐑖, 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑐𐑻𐑗𐑑 𐑯 𐑮𐑪𐑒𐑑 𐑪𐑯 𐑞𐑦𐑕 𐑐𐑴𐑤, 𐑞 𐑤𐑪𐑙 𐑑𐑱𐑤 𐑓𐑧𐑞𐑼𐑟 𐑕𐑑𐑮𐑰𐑥𐑦𐑙 𐑤𐑲𐑒 𐑐𐑧𐑯𐑩𐑯𐑟.

𐑩 𐑡𐑧𐑯𐑑𐑩𐑤 𐑡𐑶𐑩𐑕𐑯𐑩𐑕—𐑩 𐑥𐑲𐑑𐑦 𐑥𐑲𐑤𐑛𐑯𐑩𐑕 𐑝 𐑮𐑦𐑐𐑴𐑟 𐑦𐑯 𐑕𐑢𐑦𐑓𐑑𐑯𐑩𐑕, 𐑦𐑯𐑝𐑧𐑕𐑑𐑩𐑛 𐑞 𐑜𐑤𐑲𐑛𐑦𐑙 𐑢𐑱𐑤. 𐑯𐑪𐑑 𐑞 𐑢𐑲𐑑 𐑚𐑫𐑤 ·𐑡𐑵𐑐𐑦𐑑𐑼 𐑕𐑢𐑦𐑥𐑦𐑙 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑮𐑨𐑝𐑦𐑖𐑑 ·𐑘𐑫𐑼𐑴𐑐𐑩 𐑒𐑤𐑦𐑙𐑦𐑙 𐑑 𐑣𐑦𐑟 𐑜𐑮𐑱𐑕𐑓𐑩𐑤 𐑣𐑹𐑯𐑟; 𐑣𐑦𐑟 𐑤𐑳𐑝𐑤𐑦, 𐑤𐑽𐑦𐑙 𐑲𐑟 𐑕𐑲𐑛𐑢𐑱𐑟 𐑦𐑯𐑑𐑧𐑯𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑥𐑱𐑛; 𐑢𐑦𐑞 𐑕𐑥𐑵𐑞 𐑚𐑦𐑢𐑦𐑗𐑦𐑙 𐑓𐑤𐑰𐑑𐑯𐑩𐑕, 𐑮𐑦𐑐𐑩𐑤𐑦𐑙 𐑕𐑑𐑮𐑱𐑑 𐑓 𐑞 𐑯𐑳𐑐𐑖𐑩𐑤 𐑚𐑬𐑼 𐑦𐑯 ·𐑒𐑮𐑰𐑑; 𐑯𐑪𐑑 ·𐑡𐑴𐑝, 𐑯𐑪𐑑 𐑞𐑨𐑑 𐑜𐑮𐑱𐑑 𐑥𐑨𐑡𐑩𐑕𐑑𐑦 ·𐑕𐑵𐑐𐑮𐑰𐑥! 𐑛𐑦𐑛 𐑕𐑼𐑐𐑭𐑕 𐑞 𐑜𐑤𐑹𐑦𐑓𐑲𐑛 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑨𐑟 𐑣𐑰 𐑕𐑴 𐑛𐑦𐑝𐑲𐑯𐑤𐑦 𐑕𐑢𐑨𐑥.

𐑪𐑯 𐑰𐑗 𐑕𐑪𐑓𐑑 𐑕𐑲𐑛—𐑒𐑴𐑦𐑯𐑕𐑦𐑛𐑩𐑯𐑑 𐑢𐑦𐑞 𐑞 𐑐𐑸𐑑𐑩𐑛 𐑕𐑢𐑧𐑤, 𐑞𐑨𐑑 𐑚𐑳𐑑 𐑢𐑳𐑯𐑕 𐑤𐑰𐑝𐑦𐑙 𐑣𐑦𐑥, 𐑞𐑧𐑯 𐑓𐑤𐑴𐑛 𐑕𐑴 𐑢𐑲𐑛 𐑩𐑢𐑱—𐑪𐑯 𐑰𐑗 𐑚𐑮𐑲𐑑 𐑕𐑲𐑛, 𐑞 𐑢𐑱𐑤 𐑖𐑧𐑛 𐑪𐑓 𐑦𐑯𐑑𐑲𐑕𐑦𐑙𐑟. 𐑯𐑴 𐑢𐑳𐑯𐑛𐑼 𐑞𐑺 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑳𐑥 𐑩𐑥𐑳𐑙 𐑞 𐑣𐑳𐑯𐑑𐑼𐑟 𐑣𐑵 𐑯𐑱𐑥𐑤𐑩𐑕𐑤𐑦 𐑑𐑮𐑨𐑯𐑕𐑐𐑹𐑑𐑩𐑛 𐑯 𐑩𐑤𐑫𐑼𐑛 𐑚𐑲 𐑷𐑤 𐑞𐑦𐑕 𐑕𐑼𐑧𐑯𐑦𐑑𐑦, 𐑣𐑨𐑛 𐑝𐑧𐑯𐑗𐑼𐑛 𐑑 𐑩𐑕𐑱𐑤 𐑦𐑑; 𐑚𐑳𐑑 𐑣𐑨𐑛 𐑓𐑱𐑑𐑩𐑤𐑦 𐑓𐑬𐑯𐑛 𐑞𐑨𐑑 𐑒𐑢𐑲𐑩𐑑𐑵𐑛 𐑚𐑳𐑑 𐑞 𐑝𐑧𐑕𐑗𐑼 𐑝 𐑑𐑹𐑯𐑱𐑛𐑴𐑟. 𐑘𐑧𐑑 𐑒𐑭𐑥, 𐑦𐑯𐑑𐑲𐑕𐑦𐑙 𐑒𐑭𐑥, 𐑴, 𐑢𐑱𐑤! 𐑞𐑬 𐑜𐑤𐑲𐑛𐑩𐑕𐑑 𐑪𐑯, 𐑑 𐑷𐑤 𐑣𐑵 𐑓 𐑞 𐑓𐑻𐑕𐑑 𐑑𐑲𐑥 𐑲 𐑞𐑰, 𐑯𐑴 𐑥𐑨𐑑𐑼 𐑣𐑬 𐑥𐑧𐑯𐑦 𐑦𐑯 𐑞𐑨𐑑 𐑕𐑱𐑥 𐑢𐑱 𐑞𐑬 𐑥𐑳'𐑕𐑑 𐑣𐑨𐑝 𐑚𐑦𐑡𐑳𐑜𐑩𐑤𐑛 𐑯 𐑛𐑦𐑕𐑑𐑮𐑶𐑛 𐑚𐑦𐑓𐑹.

𐑯 𐑞𐑳𐑕, 𐑔𐑮𐑵 𐑞 𐑕𐑼𐑰𐑯 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤𐑦𐑑𐑦𐑟 𐑝 𐑞 𐑑𐑮𐑪𐑐𐑦𐑒𐑩𐑤 𐑕𐑰, 𐑩𐑥𐑳𐑙 𐑢𐑱𐑝𐑟 𐑣𐑵𐑟 𐑣𐑨𐑯𐑛-𐑒𐑤𐑨𐑐𐑦𐑙𐑟 𐑢𐑻 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑚𐑲 𐑦𐑒𐑕𐑰𐑛𐑦𐑙 𐑮𐑨𐑐𐑗𐑼, ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑥𐑵𐑝𐑛 𐑪𐑯, 𐑕𐑑𐑦𐑤 𐑢𐑦𐑞𐑣𐑴𐑤𐑛𐑦𐑙 𐑓𐑮𐑪𐑥 𐑕𐑲𐑑 𐑞 𐑓𐑫𐑤 𐑑𐑧𐑮𐑼𐑟 𐑝 𐑣𐑦𐑟 𐑕𐑩𐑚𐑥𐑻𐑡𐑛 𐑑𐑮𐑳𐑙𐑒, 𐑦𐑯𐑑𐑲𐑼𐑤𐑦 𐑣𐑲𐑛𐑦𐑙 𐑞 𐑮𐑧𐑯𐑗𐑑 𐑣𐑦𐑛𐑾𐑕𐑯𐑩𐑕 𐑝 𐑣𐑦𐑟 𐑡𐑷. 𐑚𐑳𐑑 𐑕𐑵𐑯 𐑞 𐑓𐑹 𐑐𐑸𐑑 𐑝 𐑣𐑦𐑥 𐑕𐑤𐑴𐑤𐑦 𐑮𐑴𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑷𐑑𐑼; 𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑣𐑦𐑟 𐑣𐑴𐑤 𐑥𐑸𐑚𐑩𐑤𐑲𐑟𐑛 𐑚𐑪𐑛𐑦 𐑓𐑹𐑥𐑛 𐑩 𐑣𐑲 𐑸𐑗, 𐑤𐑲𐑒 ·𐑝𐑼𐑡𐑦𐑯𐑾'𐑟 ·𐑯𐑨𐑗𐑼𐑩𐑤 ·𐑚𐑮𐑦𐑡, 𐑯 𐑢𐑹𐑯𐑦𐑙𐑤𐑦 𐑢𐑱𐑝𐑦𐑙 𐑣𐑦𐑟 𐑚𐑨𐑯𐑼𐑛 𐑓𐑤𐑵𐑒𐑕 𐑦𐑯 𐑞 𐑺, 𐑞 𐑜𐑮𐑨𐑯𐑛 𐑜𐑪𐑛 𐑮𐑦𐑝𐑰𐑤𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑕𐑬𐑯𐑛𐑩𐑛, 𐑯 𐑢𐑧𐑯𐑑 𐑬𐑑 𐑝 𐑕𐑲𐑑. 𐑣𐑪𐑝𐑼𐑦𐑙𐑤𐑦 𐑣𐑷𐑤𐑑𐑦𐑙, 𐑯 𐑛𐑦𐑐𐑦𐑙 𐑪𐑯 𐑞 𐑢𐑦𐑙, 𐑞 𐑢𐑲𐑑 𐑕𐑰-𐑓𐑬𐑤𐑟 𐑤𐑪𐑙𐑦𐑙𐑤𐑦 𐑤𐑦𐑙𐑜𐑼𐑛 𐑴𐑝𐑼 𐑞 𐑨𐑡𐑦𐑑𐑱𐑑𐑩𐑛 𐑐𐑵𐑤 𐑞𐑨𐑑 𐑣𐑰 𐑤𐑧𐑓𐑑.

𐑢𐑦𐑞 𐑹𐑟 𐑱𐑐𐑰𐑒, 𐑯 𐑐𐑨𐑛𐑩𐑤𐑟 𐑛𐑬𐑯, 𐑞 𐑖𐑰𐑑𐑕 𐑝 𐑞𐑺 𐑕𐑱𐑤𐑟 𐑩𐑛𐑮𐑦𐑓𐑑, 𐑞 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕 𐑯𐑬 𐑕𐑑𐑦𐑤𐑦 𐑓𐑤𐑴𐑑𐑩𐑛, 𐑩𐑢𐑱𐑑𐑦𐑙 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒'𐑕 𐑮𐑰𐑩𐑐𐑽𐑩𐑯𐑕.

“𐑩𐑯 𐑬𐑼,” 𐑕𐑧𐑛 ·𐑱𐑣𐑨𐑚, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑮𐑵𐑑𐑩𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑕𐑑𐑻𐑯; 𐑯 𐑣𐑰 𐑜𐑱𐑟𐑛 𐑚𐑦𐑘𐑪𐑯𐑛 𐑞 𐑢𐑱𐑤'𐑟 𐑐𐑤𐑱𐑕, 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑛𐑦𐑥 𐑚𐑤𐑵 𐑕𐑐𐑱𐑕𐑩𐑟 𐑯 𐑢𐑲𐑛 𐑢𐑵𐑦𐑙 𐑝𐑱𐑒𐑩𐑯𐑕𐑦𐑟 𐑑 𐑤𐑰𐑢𐑼𐑛. 𐑦𐑑 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑; 𐑓 𐑩𐑜𐑱𐑯 𐑣𐑦𐑟 𐑲𐑟 𐑕𐑰𐑥𐑛 𐑢𐑻𐑤𐑦𐑙 𐑮𐑬𐑯𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑨𐑟 𐑣𐑰 𐑕𐑢𐑧𐑐𐑑 𐑞 𐑢𐑷𐑑𐑼𐑦 𐑕𐑻𐑒𐑩𐑤. 𐑞 𐑚𐑮𐑰𐑟 𐑯𐑬 𐑓𐑮𐑧𐑖𐑩𐑯𐑛; 𐑞 𐑕𐑰 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑕𐑢𐑧𐑤.

“𐑞 𐑚𐑻𐑛𐑟!—𐑞 𐑚𐑻𐑛𐑟!” 𐑒𐑮𐑲𐑛 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴.

𐑦𐑯 𐑤𐑪𐑙 ·𐑦𐑯𐑛𐑾𐑯 𐑓𐑲𐑤, 𐑨𐑟 𐑢𐑧𐑯 𐑣𐑧𐑮𐑩𐑯𐑟 𐑑𐑱𐑒 𐑢𐑦𐑙, 𐑞 𐑢𐑲𐑑 𐑚𐑻𐑛𐑟 𐑢𐑻 𐑯𐑬 𐑷𐑤 𐑓𐑤𐑲𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 ·𐑱𐑣𐑨𐑚'𐑟 𐑚𐑴𐑑; 𐑯 𐑢𐑧𐑯 𐑢𐑦𐑞𐑦𐑯 𐑩 𐑓𐑿 𐑘𐑸𐑛𐑟 𐑚𐑦𐑜𐑨𐑯 𐑓𐑤𐑳𐑑𐑼𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑢𐑷𐑑𐑼 𐑞𐑺, 𐑢𐑰𐑤𐑦𐑙 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛, 𐑢𐑦𐑞 𐑡𐑶𐑩𐑕, 𐑦𐑒𐑕𐑐𐑧𐑒𐑑𐑩𐑯𐑑 𐑒𐑮𐑲𐑟. 𐑞𐑺 𐑝𐑦𐑠𐑩𐑯 𐑢𐑪𐑟 𐑒𐑰𐑯𐑼 𐑞𐑨𐑯 𐑥𐑨𐑯'𐑟; ·𐑱𐑣𐑨𐑚 𐑒𐑫𐑛 𐑛𐑦𐑕𐑒𐑳𐑝𐑼 𐑯𐑴 𐑕𐑲𐑯 𐑦𐑯 𐑞 𐑕𐑰. 𐑚𐑳𐑑 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑨𐑟 𐑣𐑰 𐑐𐑽𐑛 𐑛𐑬𐑯 𐑯 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑦𐑑𐑕 𐑛𐑧𐑐𐑔𐑕, 𐑣𐑰 𐑐𐑮𐑩𐑓𐑬𐑯𐑛𐑤𐑦 𐑕𐑷 𐑩 𐑢𐑲𐑑 𐑤𐑦𐑝𐑦𐑙 𐑕𐑐𐑪𐑑 𐑯𐑴 𐑚𐑦𐑜𐑼 𐑞𐑨𐑯 𐑩 𐑢𐑲𐑑 𐑢𐑰𐑟𐑩𐑤, 𐑢𐑦𐑞 𐑢𐑳𐑯𐑛𐑼𐑓𐑩𐑤 𐑕𐑩𐑤𐑧𐑮𐑦𐑑𐑦 𐑳𐑐𐑮𐑲𐑟𐑦𐑙, 𐑯 𐑥𐑨𐑜𐑯𐑦𐑓𐑲𐑦𐑙 𐑨𐑟 𐑦𐑑 𐑮𐑴𐑟, 𐑑𐑦𐑤 𐑦𐑑 𐑑𐑻𐑯𐑛, 𐑯 𐑞𐑧𐑯 𐑞𐑺 𐑢𐑻 𐑐𐑤𐑱𐑯𐑤𐑦 𐑮𐑦𐑝𐑰𐑤𐑛 𐑑𐑵 𐑤𐑪𐑙 𐑒𐑮𐑫𐑒𐑑 𐑮𐑴𐑟 𐑝 𐑢𐑲𐑑, 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑑𐑰𐑔, 𐑓𐑤𐑴𐑑𐑦𐑙 𐑳𐑐 𐑓𐑮𐑪𐑥 𐑞 𐑳𐑯𐑛𐑦𐑕𐑒𐑳𐑝𐑼𐑩𐑚𐑩𐑤 𐑚𐑪𐑑𐑩𐑥. 𐑦𐑑 𐑢𐑪𐑟 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒'𐑕 𐑴𐑐𐑩𐑯 𐑥𐑬𐑔 𐑯 𐑕𐑒𐑮𐑴𐑤𐑛 𐑡𐑷; 𐑣𐑦𐑟 𐑝𐑭𐑕𐑑, 𐑖𐑨𐑛𐑴𐑛 𐑚𐑳𐑤𐑒 𐑕𐑑𐑦𐑤 𐑣𐑭𐑓 𐑚𐑤𐑧𐑯𐑛𐑦𐑙 𐑢𐑦𐑞 𐑞 𐑚𐑤𐑵 𐑝 𐑞 𐑕𐑰. 𐑞 𐑜𐑤𐑦𐑑𐑼𐑦𐑙 𐑥𐑬𐑔 𐑘𐑷𐑯𐑛 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑚𐑴𐑑 𐑤𐑲𐑒 𐑩𐑯 𐑴𐑐𐑩𐑯-𐑛𐑹𐑛 𐑥𐑸𐑚𐑩𐑤 𐑑𐑵𐑥; 𐑯 𐑜𐑦𐑝𐑦𐑙 𐑢𐑳𐑯 𐑕𐑲𐑛𐑤𐑪𐑙 𐑕𐑢𐑰𐑐 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑕𐑑𐑽𐑦𐑙 𐑹, ·𐑱𐑣𐑨𐑚 𐑢𐑻𐑤𐑛 𐑞 𐑒𐑮𐑭𐑓𐑑 𐑩𐑕𐑲𐑛 𐑓𐑮𐑪𐑥 𐑞𐑦𐑕 𐑑𐑮𐑦𐑥𐑧𐑯𐑛𐑩𐑕 𐑨𐑐𐑼𐑦𐑖𐑩𐑯. 𐑞𐑧𐑯, 𐑒𐑷𐑤𐑦𐑙 𐑩𐑐𐑪𐑯 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑑 𐑗𐑱𐑯𐑡 𐑐𐑤𐑱𐑕𐑩𐑟 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑢𐑧𐑯𐑑 𐑓𐑹𐑢𐑼𐑛 𐑑 𐑞 𐑚𐑬𐑟, 𐑯 𐑕𐑰𐑟𐑦𐑙 ·𐑐𐑻𐑔'𐑕 𐑣𐑸𐑐𐑵𐑯, 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛 𐑣𐑦𐑟 𐑒𐑮𐑵 𐑑 𐑜𐑮𐑭𐑕𐑐 𐑞𐑺 𐑹𐑟 𐑯 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑕𐑑𐑻𐑯.

𐑚𐑳𐑑 𐑕𐑵𐑯 𐑮𐑦𐑟𐑿𐑥𐑦𐑙 𐑣𐑦𐑟 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤 𐑨𐑑𐑦𐑑𐑿𐑛, ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑕𐑢𐑨𐑥 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛 𐑞 𐑮𐑧𐑒𐑑 𐑒𐑮𐑵; 𐑕𐑲𐑛𐑢𐑱𐑟 𐑗𐑻𐑯𐑦𐑙 𐑞 𐑢𐑷𐑑𐑼 𐑦𐑯 𐑣𐑦𐑟 𐑝𐑧𐑯𐑡𐑓𐑩𐑤 𐑢𐑱𐑒, 𐑨𐑟 𐑦𐑓 𐑤𐑨𐑖𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑳𐑐 𐑑 𐑕𐑑𐑦𐑤 𐑩𐑯𐑳𐑞𐑼 𐑯 𐑥𐑹 𐑛𐑧𐑛𐑤𐑦 𐑩𐑕𐑷𐑤𐑑. 𐑞 𐑕𐑲𐑑 𐑝 𐑞 𐑕𐑐𐑤𐑦𐑯𐑑𐑼𐑛 𐑚𐑴𐑑 𐑕𐑰𐑥𐑛 𐑑 𐑥𐑨𐑛𐑩𐑯 𐑣𐑦𐑥, 𐑨𐑟 𐑞 𐑚𐑤𐑳𐑛 𐑝 𐑜𐑮𐑱𐑐𐑕 𐑯 𐑥𐑳𐑤𐑚𐑼𐑦𐑟 𐑒𐑭𐑕𐑑 𐑚𐑦𐑓𐑹 ·𐑨𐑯𐑑𐑦𐑪𐑒𐑩𐑕'𐑩𐑟 𐑧𐑤𐑦𐑓𐑩𐑯𐑑𐑕 𐑦𐑯 𐑞 𐑚𐑫𐑒 𐑝 ·𐑥𐑨𐑒𐑩𐑚𐑰𐑟. 𐑥𐑰𐑯𐑢𐑲𐑤 ·𐑱𐑣𐑨𐑚 𐑣𐑭𐑓 𐑕𐑥𐑳𐑞𐑼𐑛 𐑦𐑯 𐑞 𐑓𐑴𐑥 𐑝 𐑞 𐑢𐑱𐑤'𐑟 𐑦𐑯𐑕𐑩𐑤𐑩𐑯𐑑 𐑑𐑱𐑤, 𐑯 𐑑𐑵 𐑥𐑳𐑗 𐑝 𐑩 𐑒𐑮𐑦𐑐𐑩𐑤 𐑑 𐑕𐑢𐑦𐑥,—𐑞𐑴 𐑣𐑰 𐑒𐑫𐑛 𐑕𐑑𐑦𐑤 𐑒𐑰𐑐 𐑩𐑓𐑤𐑴𐑑, 𐑰𐑝𐑩𐑯 𐑦𐑯 𐑞 𐑣𐑸𐑑 𐑝 𐑕𐑳𐑗 𐑩 𐑢𐑻𐑤𐑐𐑵𐑤 𐑨𐑟 𐑞𐑨𐑑; 𐑣𐑧𐑤𐑐𐑤𐑩𐑕 ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑧𐑛 𐑢𐑪𐑟 𐑕𐑰𐑯, 𐑤𐑲𐑒 𐑩 𐑑𐑪𐑕𐑑 𐑚𐑳𐑚𐑩𐑤 𐑢𐑦𐑗 𐑞 𐑤𐑰𐑕𐑑 𐑗𐑭𐑯𐑕 𐑖𐑪𐑒 𐑥𐑲𐑑 𐑚𐑻𐑕𐑑. 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑑'𐑕 𐑓𐑮𐑨𐑜𐑥𐑩𐑯𐑑𐑼𐑦 𐑕𐑑𐑻𐑯, ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑦𐑯𐑒𐑘𐑫𐑼𐑾𐑕𐑤𐑦 𐑯 𐑥𐑲𐑤𐑛𐑤𐑦 𐑲𐑛 𐑣𐑦𐑥; 𐑞 𐑒𐑤𐑦𐑙𐑦𐑙 𐑒𐑮𐑵, 𐑨𐑑 𐑞 𐑳𐑞𐑼 𐑛𐑮𐑦𐑓𐑑𐑦𐑙 𐑧𐑯𐑛, 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑕𐑳𐑒𐑼 𐑣𐑦𐑥; 𐑥𐑹 𐑞𐑨𐑯 𐑦𐑯𐑳𐑓 𐑢𐑪𐑟 𐑦𐑑 𐑓 𐑞𐑧𐑥 𐑑 𐑤𐑫𐑒 𐑑 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟. 𐑓 𐑕𐑴 𐑮𐑦𐑝𐑪𐑤𐑝𐑦𐑙𐑤𐑦 𐑩𐑐𐑷𐑤𐑦𐑙 𐑢𐑪𐑟 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑨𐑕𐑐𐑧𐑒𐑑, 𐑯 𐑕𐑴 𐑐𐑤𐑨𐑯𐑩𐑑𐑺𐑦𐑤𐑦 𐑕𐑢𐑦𐑓𐑑 𐑞 𐑧𐑝𐑼-𐑒𐑩𐑯𐑑𐑮𐑨𐑒𐑑𐑦𐑙 𐑕𐑻𐑒𐑩𐑤𐑟 𐑣𐑰 𐑥𐑱𐑛, 𐑞𐑨𐑑 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤𐑦 𐑕𐑢𐑵𐑐𐑦𐑙 𐑩𐑐𐑪𐑯 𐑞𐑧𐑥. 𐑯 𐑞𐑴 𐑞 𐑳𐑞𐑼 𐑚𐑴𐑑𐑕, 𐑳𐑯𐑣𐑸𐑥𐑛, 𐑕𐑑𐑦𐑤 𐑣𐑪𐑝𐑼𐑛 𐑣𐑸𐑛 𐑚𐑲; 𐑕𐑑𐑦𐑤 𐑞𐑱 𐑛𐑺𐑛 𐑯𐑪𐑑 𐑐𐑫𐑤 𐑦𐑯𐑑𐑵 𐑞 𐑧𐑛𐑦 𐑑 𐑕𐑑𐑮𐑲𐑒, 𐑤𐑧𐑕𐑑 𐑞𐑨𐑑 𐑖𐑵𐑛 𐑚𐑰 𐑞 𐑕𐑦𐑜𐑯𐑩𐑤 𐑓 𐑞 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑛𐑦𐑕𐑑𐑮𐑳𐑒𐑖𐑩𐑯 𐑝 𐑞 𐑡𐑧𐑐𐑼𐑛𐑲𐑟𐑛 𐑒𐑭𐑕𐑑𐑩𐑢𐑱𐑟, ·𐑱𐑣𐑨𐑚 𐑯 𐑷𐑤; 𐑯𐑹 𐑦𐑯 𐑞𐑨𐑑 𐑒𐑱𐑕 𐑒𐑫𐑛 𐑞𐑱 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑣𐑴𐑐 𐑑 𐑦𐑕𐑒𐑱𐑐. 𐑢𐑦𐑞 𐑕𐑑𐑮𐑱𐑯𐑦𐑙 𐑲𐑟, 𐑞𐑧𐑯, 𐑞𐑱 𐑮𐑦𐑥𐑱𐑯𐑛 𐑪𐑯 𐑞 𐑬𐑑𐑼 𐑧𐑡 𐑝 𐑞 𐑛𐑲𐑼𐑓𐑩𐑤 𐑟𐑴𐑯, 𐑣𐑵𐑟 𐑕𐑧𐑯𐑑𐑼 𐑣𐑨𐑛 𐑯𐑬 𐑚𐑦𐑒𐑳𐑥 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑣𐑧𐑛.

𐑥𐑰𐑯𐑑𐑲𐑥, 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑦𐑜𐑦𐑯𐑦𐑙 𐑷𐑤 𐑞𐑦𐑕 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑛𐑦𐑕𐑒𐑮𐑲𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐'𐑕 𐑥𐑭𐑕𐑑 𐑣𐑧𐑛𐑟; 𐑯 𐑕𐑒𐑢𐑺𐑦𐑙 𐑣𐑻 𐑘𐑸𐑛𐑟, 𐑖𐑰 𐑣𐑨𐑛 𐑚𐑹𐑯 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑰𐑯; 𐑯 𐑢𐑪𐑟 𐑯𐑬 𐑕𐑴 𐑯𐑲, 𐑞𐑨𐑑 ·𐑱𐑣𐑨𐑚 𐑦𐑯 𐑞 𐑢𐑷𐑑𐑼 𐑣𐑱𐑤𐑛 𐑣𐑻!—“𐑕𐑱𐑤 𐑪𐑯 𐑞”—𐑚𐑳𐑑 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑩 𐑚𐑮𐑱𐑒𐑦𐑙 𐑕𐑰 𐑛𐑨𐑖𐑑 𐑪𐑯 𐑣𐑦𐑥 𐑓𐑮𐑪𐑥 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒, 𐑯 𐑢𐑧𐑤𐑥𐑛 𐑣𐑦𐑥 𐑓 𐑞 𐑑𐑲𐑥. 𐑚𐑳𐑑 𐑕𐑑𐑮𐑳𐑜𐑩𐑤𐑦𐑙 𐑬𐑑 𐑝 𐑦𐑑 𐑩𐑜𐑱𐑯, 𐑯 𐑗𐑭𐑯𐑕𐑦𐑙 𐑑 𐑮𐑲𐑟 𐑪𐑯 𐑩 𐑑𐑬𐑼𐑦𐑙 𐑒𐑮𐑧𐑕𐑑, 𐑣𐑰 𐑖𐑬𐑑𐑩𐑛,—“𐑕𐑱𐑤 𐑪𐑯 𐑞 𐑢𐑱𐑤!—𐑛𐑮𐑲𐑝 𐑣𐑦𐑥 𐑪𐑓!”

𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑐𐑮𐑬𐑟 𐑢𐑻 𐑐𐑶𐑯𐑑𐑩𐑛; 𐑯 𐑚𐑮𐑱𐑒𐑦𐑙 𐑳𐑐 𐑞 𐑗𐑸𐑥𐑛 𐑕𐑻𐑒𐑩𐑤, 𐑖𐑰 𐑦𐑓𐑧𐑒𐑗𐑵𐑩𐑤𐑦 𐑐𐑸𐑑𐑩𐑛 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑝𐑦𐑒𐑑𐑦𐑥. 𐑨𐑟 𐑣𐑰 𐑕𐑳𐑤𐑩𐑯𐑤𐑦 𐑕𐑢𐑨𐑥 𐑪𐑓, 𐑞 𐑚𐑴𐑑𐑕 𐑓𐑤𐑵 𐑑 𐑞 𐑮𐑧𐑕𐑒𐑿.

𐑛𐑮𐑨𐑜𐑛 𐑦𐑯𐑑𐑵 ·𐑕𐑑𐑳𐑚'𐑟 𐑚𐑴𐑑 𐑢𐑦𐑞 𐑚𐑤𐑳𐑛-𐑖𐑪𐑑, 𐑚𐑤𐑲𐑯𐑛𐑩𐑛 𐑲𐑟, 𐑞 𐑢𐑲𐑑 𐑚𐑮𐑲𐑯 𐑒𐑱𐑒𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑮𐑦𐑙𐑒𐑩𐑤𐑟; 𐑞 𐑤𐑪𐑙 𐑑𐑧𐑯𐑖𐑩𐑯 𐑝 ·𐑱𐑣𐑨𐑚'𐑟 𐑚𐑪𐑛𐑦𐑤𐑦 𐑕𐑑𐑮𐑧𐑙𐑔 𐑛𐑦𐑛 𐑒𐑮𐑨𐑒, 𐑯 𐑣𐑧𐑤𐑐𐑤𐑩𐑕𐑤𐑦 𐑣𐑰 𐑘𐑰𐑤𐑛𐑩𐑛 𐑑 𐑣𐑦𐑟 𐑚𐑪𐑛𐑦'𐑟 𐑛𐑵𐑥: 𐑓 𐑩 𐑑𐑲𐑥, 𐑤𐑲𐑦𐑙 𐑷𐑤 𐑒𐑮𐑳𐑖𐑑 𐑦𐑯 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 ·𐑕𐑑𐑳𐑚'𐑟 𐑚𐑴𐑑, 𐑤𐑲𐑒 𐑢𐑳𐑯 𐑑𐑮𐑪𐑛𐑩𐑯 𐑳𐑯𐑛𐑼 𐑓𐑫𐑑 𐑝 𐑣𐑻𐑛𐑟 𐑝 𐑧𐑤𐑦𐑓𐑩𐑯𐑑𐑕. 𐑓𐑸 𐑦𐑯𐑤𐑩𐑯𐑛, 𐑯𐑱𐑥𐑤𐑩𐑕 𐑢𐑱𐑤𐑟 𐑒𐑱𐑥 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑨𐑟 𐑛𐑧𐑕𐑩𐑤𐑩𐑑 𐑕𐑬𐑯𐑛𐑟 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑮𐑩𐑝𐑰𐑯𐑟.

𐑚𐑳𐑑 𐑞𐑦𐑕 𐑦𐑯𐑑𐑧𐑯𐑕𐑦𐑑𐑦 𐑝 𐑣𐑦𐑟 𐑓𐑦𐑟𐑦𐑒𐑩𐑤 𐑐𐑮𐑪𐑕𐑑𐑮𐑱𐑖𐑩𐑯 𐑛𐑦𐑛 𐑚𐑳𐑑 𐑕𐑴 𐑥𐑳𐑗 𐑞 𐑥𐑹 𐑩𐑚𐑮𐑰𐑝𐑦𐑱𐑑 𐑦𐑑. 𐑦𐑯 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑'𐑕 𐑒𐑳𐑥𐑐𐑩𐑕, 𐑜𐑮𐑱𐑑 𐑣𐑸𐑑𐑕 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑒𐑩𐑯𐑛𐑧𐑯𐑕 𐑑 𐑢𐑳𐑯 𐑛𐑰𐑐 𐑐𐑨𐑙, 𐑞 𐑕𐑳𐑥 𐑑𐑴𐑑𐑩𐑤 𐑝 𐑞𐑴𐑟 𐑖𐑨𐑤𐑴 𐑐𐑱𐑯𐑟 𐑒𐑲𐑯𐑛𐑤𐑦 𐑛𐑦𐑓𐑿𐑟𐑛 𐑔𐑮𐑵 𐑓𐑰𐑚𐑩𐑤𐑼 𐑥𐑧𐑯'𐑟 𐑣𐑴𐑤 𐑤𐑲𐑝𐑟. 𐑯 𐑕𐑴, 𐑕𐑳𐑗 𐑣𐑸𐑑𐑕, 𐑞𐑴 𐑕𐑳𐑥𐑼𐑦 𐑦𐑯 𐑰𐑗 𐑢𐑳𐑯 𐑕𐑳𐑓𐑼𐑦𐑙; 𐑕𐑑𐑦𐑤, 𐑦𐑓 𐑞 𐑜𐑪𐑛𐑟 𐑛𐑦𐑒𐑮𐑰 𐑦𐑑, 𐑦𐑯 𐑞𐑺 𐑤𐑲𐑓-𐑑𐑲𐑥 𐑨𐑜𐑮𐑦𐑜𐑱𐑑 𐑩 𐑣𐑴𐑤 𐑱𐑡 𐑝 𐑢𐑴, 𐑣𐑴𐑤𐑦 𐑥𐑱𐑛 𐑳𐑐 𐑝 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑱𐑯𐑾𐑕 𐑦𐑯𐑑𐑧𐑯𐑕𐑦𐑑𐑦𐑟; 𐑓 𐑰𐑝𐑩𐑯 𐑦𐑯 𐑞𐑺 𐑐𐑶𐑯𐑑𐑤𐑩𐑕 𐑕𐑧𐑯𐑑𐑼𐑟, 𐑞𐑴𐑟 𐑯𐑴𐑚𐑩𐑤 𐑯𐑱𐑗𐑼𐑟 𐑒𐑩𐑯𐑑𐑱𐑯 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑕𐑼𐑒𐑳𐑥𐑓𐑼𐑩𐑯𐑕𐑩𐑟 𐑝 𐑦𐑯𐑓𐑽𐑽 𐑕𐑴𐑤𐑟.

“𐑞 𐑣𐑸𐑐𐑵𐑯,” 𐑕𐑧𐑛 ·𐑱𐑣𐑨𐑚, 𐑣𐑭𐑓 𐑢𐑱 𐑮𐑲𐑟𐑦𐑙, 𐑯 𐑛𐑮𐑨𐑜𐑦𐑙𐑤𐑦 𐑤𐑰𐑯𐑦𐑙 𐑪𐑯 𐑢𐑳𐑯 𐑚𐑧𐑯𐑛𐑩𐑛 𐑸𐑥—“𐑦𐑟 𐑦𐑑 𐑕𐑱𐑓?”

“𐑲, 𐑕𐑻, 𐑓 𐑦𐑑 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑛𐑸𐑑𐑩𐑛; 𐑞𐑦𐑕 𐑦𐑟 𐑦𐑑,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑖𐑴𐑦𐑙 𐑦𐑑.

“𐑤𐑱 𐑦𐑑 𐑚𐑦𐑓𐑹 𐑥𐑰;—𐑧𐑯𐑦 𐑥𐑦𐑕𐑦𐑙 𐑥𐑧𐑯?”

“𐑢𐑳𐑯, 𐑑𐑵, 𐑔𐑮𐑰, 𐑓𐑹, 𐑓𐑲𐑝;—𐑞𐑺 𐑢𐑻 𐑓𐑲𐑝 𐑹𐑟, 𐑕𐑻, 𐑯 𐑣𐑽 𐑸 𐑓𐑲𐑝 𐑥𐑧𐑯.”

“𐑞𐑨𐑑'𐑕 𐑜𐑫𐑛.—𐑣𐑧𐑤𐑐 𐑥𐑰, 𐑥𐑨𐑯; 𐑲 𐑢𐑦𐑖 𐑑 𐑕𐑑𐑨𐑯𐑛. 𐑕𐑴, 𐑕𐑴, 𐑲 𐑕𐑰 𐑣𐑦𐑥! 𐑞𐑺! 𐑞𐑺! 𐑜𐑴𐑦𐑙 𐑑 𐑤𐑰𐑢𐑼𐑛 𐑕𐑑𐑦𐑤; 𐑢𐑪𐑑 𐑩 𐑤𐑰𐑐𐑦𐑙 𐑕𐑐𐑬𐑑!—𐑣𐑨𐑯𐑛𐑟 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑥𐑰! 𐑞 𐑦𐑑𐑻𐑯𐑩𐑤 𐑕𐑨𐑐 𐑮𐑳𐑯𐑟 𐑳𐑐 𐑦𐑯 ·𐑱𐑣𐑨𐑚'𐑟 𐑚𐑴𐑯𐑟 𐑩𐑜𐑱𐑯! 𐑕𐑧𐑑 𐑞 𐑕𐑱𐑤; 𐑬𐑑 𐑹𐑟; 𐑞 𐑣𐑧𐑤𐑥!”

𐑨𐑟 𐑣𐑰 𐑢𐑪𐑟 𐑞𐑳𐑕 𐑢𐑷𐑒𐑦𐑙, 𐑳𐑑𐑼𐑦𐑙 𐑯𐑴 𐑕𐑬𐑯𐑛, 𐑦𐑒𐑕𐑧𐑐𐑑 𐑑 𐑣𐑱𐑤 𐑞 𐑥𐑧𐑯 𐑩𐑤𐑪𐑓𐑑, 𐑹 𐑑 𐑚𐑦𐑛 𐑞𐑧𐑥 𐑣𐑶𐑕𐑑 𐑩 𐑕𐑱𐑤 𐑕𐑑𐑦𐑤 𐑣𐑲𐑼, 𐑹 𐑑 𐑕𐑐𐑮𐑧𐑛 𐑢𐑳𐑯 𐑑 𐑩 𐑕𐑑𐑦𐑤 𐑜𐑮𐑱𐑑𐑼 𐑚𐑮𐑧𐑛𐑔—𐑞𐑳𐑕 𐑑 𐑯 𐑓𐑮𐑴 𐑐𐑱𐑕𐑦𐑙, 𐑚𐑦𐑯𐑰𐑔 𐑣𐑦𐑟 𐑕𐑤𐑬𐑗𐑑 𐑣𐑨𐑑, 𐑨𐑑 𐑧𐑝𐑮𐑦 𐑑𐑻𐑯 𐑣𐑰 𐑐𐑭𐑕𐑑 𐑣𐑦𐑟 𐑴𐑯 𐑮𐑧𐑒𐑑 𐑚𐑴𐑑, 𐑢𐑦𐑗 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑛𐑮𐑪𐑐𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒, 𐑯 𐑤𐑱 𐑞𐑺 𐑮𐑦𐑝𐑻𐑕𐑑; 𐑚𐑮𐑴𐑒𐑩𐑯 𐑚𐑬 𐑑 𐑖𐑨𐑑𐑼𐑛 𐑕𐑑𐑻𐑯. 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑣𐑰 𐑐𐑷𐑟𐑛 𐑚𐑦𐑓𐑹 𐑦𐑑; 𐑯 𐑨𐑟 𐑦𐑯 𐑩𐑯 𐑷𐑤𐑮𐑧𐑛𐑦 𐑴𐑝𐑼-𐑒𐑤𐑬𐑛𐑩𐑛 𐑕𐑒𐑲 𐑓𐑮𐑧𐑖 𐑑𐑮𐑵𐑐𐑕 𐑝 𐑒𐑤𐑬𐑛𐑟 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑕𐑱𐑤 𐑩𐑒𐑮𐑪𐑕, 𐑕𐑴 𐑴𐑝𐑼 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑓𐑱𐑕 𐑞𐑺 𐑯𐑬 𐑕𐑑𐑴𐑤 𐑕𐑳𐑥 𐑕𐑳𐑗 𐑨𐑛𐑩𐑛 𐑜𐑤𐑵𐑥 𐑨𐑟 𐑞𐑦𐑕.

𐑕𐑑𐑳𐑚 𐑕𐑷 𐑣𐑦𐑥 𐑐𐑷𐑟; 𐑯 𐑐𐑼𐑣𐑨𐑐𐑕 𐑦𐑯𐑑𐑧𐑯𐑛𐑦𐑙, 𐑯𐑪𐑑 𐑝𐑱𐑯𐑤𐑦, 𐑞𐑴, 𐑑 𐑦𐑝𐑦𐑯𐑕 𐑣𐑦𐑟 𐑴𐑯 𐑳𐑯𐑩𐑚𐑱𐑑𐑩𐑛 𐑓𐑹𐑑𐑦𐑑𐑿𐑛, 𐑯 𐑞𐑳𐑕 𐑒𐑰𐑐 𐑳𐑐 𐑩 𐑝𐑨𐑤𐑾𐑯𐑑 𐑐𐑤𐑱𐑕 𐑦𐑯 𐑣𐑦𐑟 ·𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑥𐑲𐑯𐑛, 𐑣𐑰 𐑩𐑛𐑝𐑭𐑯𐑕𐑑, 𐑯 𐑲𐑦𐑙 𐑞 𐑮𐑧𐑒 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛—“𐑞 𐑔𐑦𐑕𐑩𐑤 𐑞 𐑨𐑕 𐑮𐑦𐑓𐑿𐑟𐑛; 𐑦𐑑 𐑐𐑮𐑦𐑒𐑑 𐑣𐑦𐑟 𐑥𐑬𐑔 𐑑𐑵 𐑒𐑰𐑯𐑤𐑦, 𐑕𐑻; 𐑣𐑭! 𐑣𐑭!”

“𐑢𐑪𐑑 𐑕𐑴𐑤𐑤𐑩𐑕 𐑔𐑦𐑙 𐑦𐑟 𐑞𐑦𐑕 𐑞𐑨𐑑 𐑤𐑭𐑓𐑕 𐑚𐑦𐑓𐑹 𐑩 𐑮𐑧𐑒? 𐑥𐑨𐑯, 𐑥𐑨𐑯! 𐑛𐑦𐑛 𐑲 𐑯𐑪𐑑 𐑯𐑴 𐑞𐑰 𐑚𐑮𐑱𐑝 𐑨𐑟 𐑓𐑽𐑤𐑩𐑕 𐑓𐑲𐑼 (𐑯 𐑨𐑟 𐑥𐑦𐑒𐑨𐑯𐑦𐑒𐑩𐑤) 𐑲 𐑒𐑫𐑛 𐑕𐑢𐑺 𐑞𐑬 ·𐑝𐑧𐑮𐑑 𐑩 𐑐𐑪𐑤𐑑𐑮𐑵𐑯. 𐑜𐑮𐑴𐑯 𐑯𐑹 𐑤𐑭𐑓 𐑖𐑵𐑛 𐑚𐑰 𐑣𐑻𐑛 𐑚𐑦𐑓𐑹 𐑩 𐑮𐑧𐑒.”

“𐑲, 𐑕𐑻,” 𐑕𐑧𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑛𐑮𐑷𐑦𐑙 𐑯𐑽, “'𐑑𐑦𐑟 𐑩 𐑕𐑪𐑤𐑩𐑥 𐑕𐑲𐑑; 𐑩𐑯 𐑴𐑥𐑩𐑯, 𐑯 𐑩𐑯 𐑦𐑤 𐑢𐑳𐑯.”

“𐑴𐑥𐑩𐑯? 𐑴𐑥𐑩𐑯?—𐑞 𐑛𐑦𐑒𐑖𐑩𐑯𐑼𐑦! 𐑦𐑓 𐑞 𐑜𐑪𐑛𐑟 𐑔𐑦𐑙𐑒 𐑑 𐑕𐑐𐑰𐑒 𐑬𐑑𐑮𐑲𐑑 𐑑 𐑥𐑨𐑯, 𐑞𐑱 𐑢𐑦𐑤 𐑪𐑯𐑼𐑩𐑚𐑤𐑦 𐑕𐑐𐑰𐑒 𐑬𐑑𐑮𐑲𐑑; 𐑯𐑪𐑑 𐑖𐑱𐑒 𐑞𐑺 𐑣𐑧𐑛𐑟, 𐑯 𐑜𐑦𐑝 𐑩𐑯 𐑴𐑤𐑛 𐑢𐑲𐑝𐑟' 𐑛𐑸𐑒𐑤𐑦𐑙 𐑣𐑦𐑯𐑑.—𐑚𐑦𐑜𐑪𐑯! 𐑘𐑰 𐑑𐑵 𐑸 𐑞 𐑪𐑐𐑩𐑟𐑦𐑑 𐑐𐑴𐑤𐑟 𐑝 𐑢𐑳𐑯 𐑔𐑦𐑙; ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑦𐑟 ·𐑕𐑑𐑳𐑚 𐑮𐑦𐑝𐑻𐑕𐑑, 𐑯 ·𐑕𐑑𐑳𐑚 𐑦𐑟 ·𐑕𐑑𐑸𐑚𐑳𐑒; 𐑯 𐑘𐑰 𐑑𐑵 𐑸 𐑷𐑤 𐑥𐑨𐑯𐑒𐑲𐑯𐑛; 𐑯 ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑨𐑯𐑛𐑟 𐑩𐑤𐑴𐑯 𐑩𐑥𐑳𐑙 𐑞 𐑥𐑦𐑤𐑾𐑯𐑟 𐑝 𐑞 𐑐𐑰𐑐𐑩𐑤𐑛 𐑻𐑔, 𐑯𐑹 𐑜𐑪𐑛𐑟 𐑯𐑹 𐑥𐑧𐑯 𐑣𐑦𐑟 𐑯𐑱𐑚𐑼𐑟! 𐑒𐑴𐑤𐑛, 𐑒𐑴𐑤𐑛—𐑲 𐑖𐑦𐑝𐑼!—𐑣𐑬 𐑯𐑬? 𐑩𐑤𐑪𐑓𐑑 𐑞𐑺! ·𐑛'𐑘𐑰 𐑕𐑰 𐑣𐑦𐑥? 𐑕𐑦𐑙 𐑬𐑑 𐑓 𐑧𐑝𐑮𐑦 𐑕𐑐𐑬𐑑, 𐑞𐑴 𐑣𐑰 𐑕𐑐𐑬𐑑 𐑑𐑧𐑯 𐑑𐑲𐑥𐑟 𐑩 𐑕𐑧𐑒𐑩𐑯𐑛!”

𐑞 𐑛𐑱 𐑢𐑪𐑟 𐑯𐑽𐑤𐑦 𐑛𐑳𐑯; 𐑴𐑯𐑤𐑦 𐑞 𐑣𐑧𐑥 𐑝 𐑣𐑦𐑟 𐑜𐑴𐑤𐑛𐑩𐑯 𐑮𐑴𐑚 𐑢𐑪𐑟 𐑮𐑳𐑕𐑩𐑤𐑦𐑙. 𐑕𐑵𐑯, 𐑦𐑑 𐑢𐑪𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑛𐑸𐑒, 𐑚𐑳𐑑 𐑞 𐑤𐑫𐑒-𐑬𐑑 𐑥𐑧𐑯 𐑕𐑑𐑦𐑤 𐑮𐑦𐑥𐑱𐑯𐑛 𐑳𐑯𐑕𐑧𐑑.

“𐑒𐑭𐑯𐑑 𐑕𐑰 𐑞 𐑕𐑐𐑬𐑑 𐑯𐑬, 𐑕𐑻;—𐑑𐑵 𐑛𐑸𐑒”—𐑒𐑮𐑲𐑛 𐑩 𐑝𐑶𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑺.

“𐑣𐑬 𐑣𐑧𐑛𐑦𐑙 𐑢𐑧𐑯 𐑤𐑭𐑕𐑑 𐑕𐑰𐑯?”

“𐑨𐑟 𐑚𐑦𐑓𐑹, 𐑕𐑻,—𐑕𐑑𐑮𐑱𐑑 𐑑 𐑤𐑰𐑢𐑼𐑛.”

“𐑜𐑫𐑛! 𐑣𐑰 𐑢𐑦𐑤 𐑑𐑮𐑨𐑝𐑩𐑤 𐑕𐑤𐑴𐑼 𐑯𐑬 '𐑑𐑦𐑟 𐑯𐑲𐑑. 𐑛𐑬𐑯 𐑮𐑶𐑩𐑤𐑟 𐑯 𐑑𐑪𐑐-𐑜𐑨𐑤𐑩𐑯𐑑 𐑕𐑑𐑳𐑯-𐑕𐑱𐑤𐑟, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒. 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑯𐑪𐑑 𐑮𐑳𐑯 𐑴𐑝𐑼 𐑣𐑦𐑥 𐑚𐑦𐑓𐑹 𐑥𐑹𐑯𐑦𐑙; 𐑣𐑰'𐑟 𐑥𐑱𐑒𐑦𐑙 𐑩 𐑐𐑨𐑕𐑦𐑡 𐑯𐑬, 𐑯 𐑥𐑱 𐑣𐑰𐑝-𐑑 𐑩 𐑢𐑲𐑤. 𐑣𐑧𐑤𐑥 𐑞𐑺! 𐑒𐑰𐑐 𐑣𐑻 𐑓𐑫𐑤 𐑚𐑦𐑓𐑹 𐑞 𐑢𐑦𐑯𐑛!—𐑩𐑤𐑪𐑓𐑑! 𐑒𐑳𐑥 𐑛𐑬𐑯!—·𐑥𐑼 ·𐑕𐑑𐑳𐑚, 𐑕𐑧𐑯𐑛 𐑩 𐑓𐑮𐑧𐑖 𐑣𐑨𐑯𐑛 𐑑 𐑞 𐑓𐑹-𐑥𐑭𐑕𐑑 𐑣𐑧𐑛, 𐑯 𐑕𐑰 𐑦𐑑 𐑥𐑨𐑯𐑛 𐑑𐑦𐑤 𐑥𐑹𐑯𐑦𐑙.”—𐑞𐑧𐑯 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑛𐑳𐑚𐑤𐑵𐑯 𐑦𐑯 𐑞 𐑥𐑱𐑯-𐑥𐑭𐑕𐑑—“𐑥𐑧𐑯, 𐑞𐑦𐑕 𐑜𐑴𐑤𐑛 𐑦𐑟 𐑥𐑲𐑯, 𐑓 𐑲 𐑻𐑯𐑛 𐑦𐑑; 𐑚𐑳𐑑 𐑲 𐑖𐑨𐑤 𐑤𐑧𐑑 𐑦𐑑 𐑩𐑚𐑲𐑛 𐑣𐑽 𐑑𐑦𐑤 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑦𐑟 𐑛𐑧𐑛; 𐑯 𐑞𐑧𐑯, 𐑣𐑵𐑕𐑴𐑧𐑝𐑼 𐑝 𐑘𐑰 𐑓𐑻𐑕𐑑 𐑮𐑱𐑟𐑩𐑟 𐑣𐑦𐑥, 𐑩𐑐𐑪𐑯 𐑞 𐑛𐑱 𐑣𐑰 𐑖𐑨𐑤 𐑚𐑰 𐑒𐑦𐑤𐑛, 𐑞𐑦𐑕 𐑜𐑴𐑤𐑛 𐑦𐑟 𐑞𐑨𐑑 𐑥𐑨𐑯'𐑟; 𐑯 𐑦𐑓 𐑪𐑯 𐑞𐑨𐑑 𐑛𐑱 𐑲 𐑖𐑨𐑤 𐑩𐑜𐑱𐑯 𐑮𐑱𐑟 𐑣𐑦𐑥, 𐑞𐑧𐑯, 𐑑𐑧𐑯 𐑑𐑲𐑥𐑟 𐑦𐑑𐑕 𐑕𐑳𐑥 𐑖𐑨𐑤 𐑚𐑰 𐑛𐑦𐑝𐑲𐑛𐑩𐑛 𐑩𐑥𐑳𐑙 𐑷𐑤 𐑝 𐑘𐑰! 𐑩𐑢𐑱 𐑯𐑬!—𐑞 𐑛𐑧𐑒 𐑦𐑟 𐑞𐑲𐑯, 𐑕𐑻!”

𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙, 𐑣𐑰 𐑐𐑤𐑱𐑕𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑣𐑭𐑓 𐑢𐑱 𐑢𐑦𐑞𐑦𐑯 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤, 𐑯 𐑕𐑤𐑬𐑗𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑑, 𐑕𐑑𐑫𐑛 𐑞𐑺 𐑑𐑦𐑤 𐑛𐑷𐑯, 𐑦𐑒𐑕𐑧𐑐𐑑 𐑢𐑧𐑯 𐑨𐑑 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟 𐑮𐑬𐑟𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑑 𐑕𐑰 𐑣𐑬 𐑞 𐑯𐑲𐑑 𐑢𐑹 𐑪𐑯.
