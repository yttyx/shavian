·𐑗𐑨𐑐𐑑𐑼 31. 𐑒𐑢𐑰𐑯 ·𐑥𐑨𐑚.

𐑯𐑧𐑒𐑕𐑑 𐑥𐑹𐑯𐑦𐑙 ·𐑕𐑑𐑳𐑚 𐑩𐑒𐑪𐑕𐑑𐑩𐑛 ·𐑓𐑤𐑭𐑕𐑒.

“𐑕𐑳𐑗 𐑩 𐑒𐑢𐑽 𐑛𐑮𐑰𐑥, ·𐑒𐑦𐑙-·𐑐𐑴𐑕𐑑, 𐑲 𐑯𐑧𐑝𐑼 𐑣𐑨𐑛. 𐑿 𐑯𐑴 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜, 𐑢𐑧𐑤 𐑲 𐑛𐑮𐑰𐑥𐑛 𐑣𐑰 𐑒𐑦𐑒𐑑 𐑥𐑰 𐑢𐑦𐑞 𐑦𐑑; 𐑯 𐑢𐑧𐑯 𐑲 𐑑𐑮𐑲𐑛 𐑑 𐑒𐑦𐑒 𐑚𐑨𐑒, 𐑩𐑐𐑪𐑯 𐑥𐑲 𐑕𐑴𐑤, 𐑥𐑲 𐑤𐑦𐑑𐑩𐑤 𐑥𐑨𐑯, 𐑲 𐑒𐑦𐑒𐑑 𐑥𐑲 𐑤𐑧𐑜 𐑮𐑲𐑑 𐑪𐑓! 𐑯 𐑞𐑧𐑯, 𐑐𐑮𐑧𐑕𐑑𐑴! 𐑱𐑣𐑨𐑚 𐑕𐑰𐑥𐑛 𐑩 𐑐𐑦𐑮𐑩𐑥𐑦𐑛, 𐑯 𐑲, 𐑤𐑲𐑒 𐑩 𐑚𐑤𐑱𐑟𐑦𐑙 𐑓𐑵𐑤, 𐑒𐑧𐑐𐑑 𐑒𐑦𐑒𐑦𐑙 𐑨𐑑 𐑦𐑑. 𐑚𐑳𐑑 𐑢𐑪𐑑 𐑢𐑪𐑟 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑒𐑘𐑫𐑼𐑾𐑕, ·𐑓𐑤𐑭𐑕𐑒—𐑿 𐑯𐑴 𐑣𐑬 𐑒𐑘𐑫𐑼𐑾𐑕 𐑷𐑤 𐑛𐑮𐑰𐑥𐑟 𐑸—𐑔𐑮𐑵 𐑷𐑤 𐑞𐑦𐑕 𐑮𐑱𐑡 𐑞𐑨𐑑 𐑲 𐑢𐑪𐑟 𐑦𐑯, 𐑲 𐑕𐑳𐑥𐑣𐑬 𐑕𐑰𐑥𐑛 𐑑 𐑚𐑰 𐑔𐑦𐑙𐑒𐑦𐑙 𐑑 𐑥𐑲𐑕𐑧𐑤𐑓, 𐑞𐑨𐑑 𐑭𐑓𐑑𐑼 𐑷𐑤, 𐑦𐑑 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑥𐑳𐑗 𐑝 𐑩𐑯 𐑦𐑯𐑕𐑳𐑤𐑑, 𐑞𐑨𐑑 𐑒𐑦𐑒 𐑓𐑮𐑪𐑥 ·𐑱𐑣𐑨𐑚. '𐑢𐑲,' 𐑔𐑦𐑙𐑒𐑕 𐑲, '𐑢𐑪𐑑'𐑕 𐑞 𐑮𐑴? 𐑦𐑑'𐑕 𐑯𐑪𐑑 𐑩 𐑮𐑾𐑤 𐑤𐑧𐑜, 𐑴𐑯𐑤𐑦 𐑩 𐑓𐑷𐑤𐑕 𐑤𐑧𐑜.' 𐑯 𐑞𐑺'𐑟 𐑩 𐑥𐑲𐑑𐑦 𐑛𐑦𐑓𐑼𐑩𐑯𐑕 𐑚𐑦𐑑𐑢𐑰𐑯 𐑩 𐑤𐑦𐑝𐑦𐑙 𐑔𐑳𐑥𐑐 𐑯 𐑩 𐑛𐑧𐑛 𐑔𐑳𐑥𐑐. 𐑞𐑨𐑑'𐑕 𐑢𐑪𐑑 𐑥𐑱𐑒𐑕 𐑩 𐑚𐑤𐑴 𐑓𐑮𐑪𐑥 𐑞 𐑣𐑨𐑯𐑛, ·𐑓𐑤𐑭𐑕𐑒, 𐑓𐑦𐑓𐑑𐑦 𐑑𐑲𐑥𐑟 𐑥𐑹 𐑕𐑨𐑝𐑦𐑡 𐑑 𐑚𐑺 𐑞𐑨𐑯 𐑩 𐑚𐑤𐑴 𐑓𐑮𐑪𐑥 𐑩 𐑒𐑱𐑯. 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑥𐑧𐑥𐑚𐑼—𐑞𐑨𐑑 𐑥𐑱𐑒𐑕 𐑞 𐑤𐑦𐑝𐑦𐑙 𐑦𐑯𐑕𐑳𐑤𐑑, 𐑥𐑲 𐑤𐑦𐑑𐑩𐑤 𐑥𐑨𐑯. 𐑯 𐑔𐑦𐑙𐑒𐑕 𐑲 𐑑 𐑥𐑲𐑕𐑧𐑤𐑓 𐑷𐑤 𐑞 𐑢𐑲𐑤, 𐑥𐑲𐑯𐑛, 𐑢𐑲𐑤 𐑲 𐑢𐑪𐑟 𐑕𐑑𐑳𐑚𐑦𐑙 𐑥𐑲 𐑕𐑦𐑤𐑦 𐑑𐑴𐑟 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞𐑨𐑑 𐑒𐑻𐑕𐑑 𐑐𐑦𐑮𐑩𐑥𐑦𐑛—𐑕𐑴 𐑒𐑩𐑯𐑓𐑬𐑯𐑛𐑩𐑛𐑤𐑦 𐑒𐑪𐑯𐑑𐑮𐑩𐑛𐑦𐑒𐑑𐑼𐑦 𐑢𐑪𐑟 𐑦𐑑 𐑷𐑤, 𐑷𐑤 𐑞 𐑢𐑲𐑤, 𐑲 𐑕𐑱, 𐑲 𐑢𐑪𐑟 𐑔𐑦𐑙𐑒𐑦𐑙 𐑑 𐑥𐑲𐑕𐑧𐑤𐑓, '𐑢𐑪𐑑'𐑕 𐑣𐑦𐑟 𐑤𐑧𐑜 𐑯𐑬, 𐑚𐑳𐑑 𐑩 𐑒𐑱𐑯—𐑩 𐑢𐑱𐑤𐑚𐑴𐑯 𐑒𐑱𐑯. 𐑘𐑧𐑕,' 𐑔𐑦𐑙𐑒𐑕 𐑲, '𐑦𐑑 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑩 𐑐𐑤𐑱𐑓𐑩𐑤 𐑒𐑳𐑡𐑩𐑤𐑦𐑙—𐑦𐑯 𐑓𐑨𐑒𐑑, 𐑴𐑯𐑤𐑦 𐑩 𐑢𐑱𐑤𐑚𐑴𐑯𐑦𐑙 𐑞𐑨𐑑 𐑣𐑰 𐑜𐑱𐑝 𐑥𐑰—𐑯𐑪𐑑 𐑩 𐑚𐑱𐑕 𐑒𐑦𐑒. 𐑚𐑦𐑕𐑲𐑛𐑟,' 𐑔𐑦𐑙𐑒𐑕 𐑲, '𐑤𐑫𐑒 𐑨𐑑 𐑦𐑑 𐑢𐑳𐑯𐑕; 𐑢𐑲, 𐑞 𐑧𐑯𐑛 𐑝 𐑦𐑑—𐑞 𐑓𐑫𐑑 𐑐𐑸𐑑—𐑢𐑪𐑑 𐑩 𐑕𐑥𐑷𐑤 𐑕𐑹𐑑 𐑝 𐑧𐑯𐑛 𐑦𐑑 𐑦𐑟; 𐑢𐑺𐑨𐑟, 𐑦𐑓 𐑩 𐑚𐑮𐑷𐑛 𐑓𐑫𐑑𐑩𐑛 𐑓𐑸𐑥𐑼 𐑒𐑦𐑒𐑑 𐑥𐑰, 𐑞𐑺'𐑟 𐑩 𐑛𐑧𐑝𐑩𐑤𐑦𐑖 𐑚𐑮𐑷𐑛 𐑦𐑯𐑕𐑳𐑤𐑑. 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑦𐑯𐑕𐑳𐑤𐑑 𐑦𐑟 𐑢𐑦𐑑𐑩𐑤𐑛 𐑛𐑬𐑯 𐑑 𐑩 𐑐𐑶𐑯𐑑 𐑴𐑯𐑤𐑦.' 𐑚𐑳𐑑 𐑯𐑬 𐑒𐑳𐑥𐑟 𐑞 𐑜𐑮𐑱𐑑𐑩𐑕𐑑 𐑡𐑴𐑒 𐑝 𐑞 𐑛𐑮𐑰𐑥, ·𐑓𐑤𐑭𐑕𐑒. 𐑢𐑲𐑤 𐑲 𐑢𐑪𐑟 𐑚𐑨𐑑𐑼𐑦𐑙 𐑩𐑢𐑱 𐑨𐑑 𐑞 𐑐𐑦𐑮𐑩𐑥𐑦𐑛, 𐑩 𐑕𐑹𐑑 𐑝 𐑚𐑨𐑡𐑼-𐑣𐑺𐑛 𐑴𐑤𐑛 𐑥𐑻𐑥𐑩𐑯, 𐑢𐑦𐑞 𐑩 𐑣𐑳𐑥𐑐 𐑪𐑯 𐑣𐑦𐑟 𐑚𐑨𐑒, 𐑑𐑱𐑒𐑕 𐑥𐑰 𐑚𐑲 𐑞 𐑖𐑴𐑤𐑛𐑼𐑟, 𐑯 𐑕𐑤𐑵𐑟 𐑥𐑰 𐑮𐑬𐑯𐑛. '𐑢𐑪𐑑 𐑸 𐑿 '𐑚𐑬𐑑?' 𐑕𐑧𐑟 𐑣𐑰. 𐑕𐑤𐑦𐑛! 𐑥𐑨𐑯, 𐑚𐑳𐑑 𐑲 𐑢𐑪𐑟 𐑓𐑮𐑲𐑑𐑩𐑯𐑛. 𐑕𐑳𐑗 𐑩 𐑓𐑦𐑟! 𐑚𐑳𐑑, 𐑕𐑳𐑥𐑣𐑬, 𐑯𐑧𐑒𐑕𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑲 𐑢𐑪𐑟 𐑴𐑝𐑼 𐑞 𐑓𐑮𐑲𐑑. '𐑢𐑪𐑑 𐑨𐑥 𐑲 𐑩𐑚𐑬𐑑?' 𐑕𐑧𐑟 𐑲 𐑨𐑑 𐑤𐑭𐑕𐑑. '𐑯 𐑢𐑪𐑑 𐑚𐑦𐑟𐑯𐑩𐑕 𐑦𐑟 𐑞𐑨𐑑 𐑝 𐑘𐑹𐑟, 𐑲 𐑖𐑵𐑛 𐑤𐑲𐑒 𐑑 𐑯𐑴, ·𐑥𐑼 ·𐑣𐑳𐑥𐑐𐑚𐑨𐑒? 𐑛𐑵 𐑿 𐑢𐑪𐑯𐑑 𐑩 𐑒𐑦𐑒?' 𐑚𐑲 𐑞 𐑤𐑹𐑛, ·𐑓𐑤𐑭𐑕𐑒, 𐑲 𐑣𐑨𐑛 𐑯𐑴 𐑕𐑵𐑯𐑼 𐑕𐑧𐑛 𐑞𐑨𐑑, 𐑞𐑨𐑯 𐑣𐑰 𐑑𐑻𐑯𐑛 𐑮𐑬𐑯𐑛 𐑣𐑦𐑟 𐑕𐑑𐑻𐑯 𐑑 𐑥𐑰, 𐑚𐑧𐑯𐑑 𐑴𐑝𐑼, 𐑯 𐑛𐑮𐑨𐑜𐑦𐑙 𐑳𐑐 𐑩 𐑤𐑪𐑑 𐑝 𐑕𐑰𐑢𐑰𐑛 𐑣𐑰 𐑣𐑨𐑛 𐑓 𐑩 𐑒𐑤𐑬𐑑—𐑢𐑪𐑑 𐑛𐑵 𐑿 𐑔𐑦𐑙𐑒, 𐑲 𐑕𐑷?—𐑢𐑲 𐑔𐑳𐑯𐑛𐑼 𐑩𐑤𐑲𐑝, 𐑥𐑨𐑯, 𐑣𐑦𐑟 𐑕𐑑𐑻𐑯 𐑢𐑪𐑟 𐑕𐑑𐑳𐑒 𐑓𐑫𐑤 𐑝 𐑯𐑸𐑤𐑦𐑯𐑕𐑐𐑲𐑒𐑕, 𐑢𐑦𐑞 𐑞 𐑐𐑶𐑯𐑑𐑕 𐑬𐑑. 𐑕𐑧𐑟 𐑲, 𐑪𐑯 𐑕𐑧𐑒𐑩𐑯𐑛 𐑔𐑷𐑑𐑕, '𐑲 𐑜𐑧𐑕 𐑲 𐑢𐑴𐑯𐑑 𐑒𐑦𐑒 𐑿, 𐑴𐑤𐑛 𐑓𐑧𐑤𐑴.' '𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚,' 𐑕𐑧𐑛 𐑣𐑰, '𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚;' 𐑯 𐑒𐑧𐑐𐑑 𐑥𐑳𐑑𐑼𐑦𐑙 𐑦𐑑 𐑷𐑤 𐑞 𐑑𐑲𐑥, 𐑩 𐑕𐑹𐑑 𐑝 𐑰𐑑𐑦𐑙 𐑝 𐑣𐑦𐑟 𐑴𐑯 𐑜𐑳𐑥𐑟 𐑤𐑲𐑒 𐑩 𐑗𐑦𐑥𐑯𐑦 𐑣𐑨𐑜. 𐑕𐑰𐑦𐑙 𐑣𐑰 𐑢𐑪𐑟𐑩𐑯𐑑 𐑜𐑴𐑦𐑙 𐑑 𐑕𐑑𐑪𐑐 𐑕𐑱𐑦𐑙 𐑴𐑝𐑼 𐑣𐑦𐑟 '𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚, 𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚,' 𐑲 𐑔𐑷𐑑 𐑲 𐑥𐑲𐑑 𐑨𐑟 𐑢𐑧𐑤 𐑓𐑷𐑤 𐑑 𐑒𐑦𐑒𐑦𐑙 𐑞 𐑐𐑦𐑮𐑩𐑥𐑦𐑛 𐑩𐑜𐑱𐑯. 𐑚𐑳𐑑 𐑲 𐑣𐑨𐑛 𐑴𐑯𐑤𐑦 𐑡𐑳𐑕𐑑 𐑤𐑦𐑓𐑑𐑩𐑛 𐑥𐑲 𐑓𐑫𐑑 𐑓 𐑦𐑑, 𐑢𐑧𐑯 𐑣𐑰 𐑮𐑹𐑛 𐑬𐑑, '·𐑕𐑑𐑪𐑐 𐑞𐑨𐑑 𐑒𐑦𐑒𐑦𐑙!' '𐑣𐑩𐑤𐑴,' 𐑕𐑧𐑟 𐑲, '𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑯𐑬, 𐑴𐑤𐑛 𐑓𐑧𐑤𐑴?' '𐑤𐑫𐑒 𐑘𐑰 𐑣𐑽,' 𐑕𐑧𐑟 𐑣𐑰; '𐑤𐑧𐑑'𐑕 𐑸𐑜𐑿 𐑞 𐑦𐑯𐑕𐑳𐑤𐑑. 𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑒𐑦𐑒𐑑 𐑘𐑰, 𐑛𐑦𐑛𐑩𐑯𐑑 𐑣𐑰?' '𐑘𐑧𐑕, 𐑣𐑰 𐑛𐑦𐑛,' 𐑕𐑧𐑟 𐑲—'𐑮𐑲𐑑 𐑣𐑽 𐑦𐑑 𐑢𐑪𐑟.' '𐑝𐑧𐑮𐑦 𐑜𐑫𐑛,' 𐑕𐑧𐑟 𐑣𐑰—'𐑣𐑰 𐑿𐑟𐑛 𐑣𐑦𐑟 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜, 𐑛𐑦𐑛𐑩𐑯𐑑 𐑣𐑰?' '𐑘𐑧𐑕, 𐑣𐑰 𐑛𐑦𐑛,' 𐑕𐑧𐑟 𐑲. '𐑢𐑧𐑤 𐑞𐑧𐑯,' 𐑕𐑧𐑟 𐑣𐑰, '𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚, 𐑢𐑪𐑑 𐑣𐑨𐑝 𐑿 𐑑 𐑒𐑩𐑥𐑐𐑤𐑱𐑯 𐑝? 𐑛𐑦𐑛𐑩𐑯𐑑 𐑣𐑰 𐑒𐑦𐑒 𐑢𐑦𐑞 𐑮𐑲𐑑 𐑜𐑫𐑛 𐑢𐑦𐑤? 𐑦𐑑 𐑢𐑪𐑟𐑩𐑯𐑑 𐑩 𐑒𐑪𐑥𐑩𐑯 𐑐𐑦𐑗 𐑐𐑲𐑯 𐑤𐑧𐑜 𐑣𐑰 𐑒𐑦𐑒𐑑 𐑢𐑦𐑞, 𐑢𐑪𐑟 𐑦𐑑? 𐑯𐑴, 𐑿 𐑢𐑻 𐑒𐑦𐑒𐑑 𐑚𐑲 𐑩 𐑜𐑮𐑱𐑑 𐑥𐑨𐑯, 𐑯 𐑢𐑦𐑞 𐑩 𐑚𐑿𐑑𐑦𐑓𐑩𐑤 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜, ·𐑕𐑑𐑳𐑚. 𐑦𐑑'𐑕 𐑩𐑯 𐑪𐑯𐑼; 𐑲 𐑒𐑩𐑯𐑕𐑦𐑛𐑼 𐑦𐑑 𐑩𐑯 𐑪𐑯𐑼. 𐑤𐑦𐑕𐑩𐑯, 𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚. 𐑦𐑯 𐑴𐑤𐑛 ·𐑦𐑙𐑜𐑤𐑩𐑯𐑛 𐑞 𐑜𐑮𐑱𐑑𐑩𐑕𐑑 𐑤𐑹𐑛𐑟 𐑔𐑦𐑙𐑒 𐑦𐑑 𐑜𐑮𐑱𐑑 𐑜𐑤𐑹𐑦 𐑑 𐑚𐑰 𐑕𐑤𐑨𐑐𐑑 𐑚𐑲 𐑩 𐑒𐑢𐑰𐑯, 𐑯 𐑥𐑱𐑛 𐑜𐑸𐑑𐑼-𐑯𐑲𐑑𐑕 𐑝; 𐑚𐑳𐑑, 𐑚𐑰 𐑘𐑹 𐑚𐑴𐑕𐑑, ·𐑕𐑑𐑳𐑚, 𐑞𐑨𐑑 𐑘𐑰 𐑢𐑻 𐑒𐑦𐑒𐑑 𐑚𐑲 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚, 𐑯 𐑥𐑱𐑛 𐑩 𐑢𐑲𐑟 𐑥𐑨𐑯 𐑝. 𐑮𐑦𐑥𐑧𐑥𐑚𐑼 𐑢𐑪𐑑 𐑲 𐑕𐑱; 𐑚𐑰 𐑒𐑦𐑒𐑑 𐑚𐑲 𐑣𐑦𐑥; 𐑩𐑒𐑬𐑯𐑑 𐑣𐑦𐑟 𐑒𐑦𐑒𐑕 𐑪𐑯𐑼𐑟; 𐑯 𐑪𐑯 𐑯𐑴 𐑩𐑒𐑬𐑯𐑑 𐑒𐑦𐑒 𐑚𐑨𐑒; 𐑓 𐑿 𐑒𐑭𐑯𐑑 𐑣𐑧𐑤𐑐 𐑘𐑹𐑕𐑧𐑤𐑓, 𐑢𐑲𐑟 ·𐑕𐑑𐑳𐑚. 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑰 𐑞𐑨𐑑 𐑐𐑦𐑮𐑩𐑥𐑦𐑛?' 𐑢𐑦𐑞 𐑞𐑨𐑑, 𐑣𐑰 𐑷𐑤 𐑝 𐑩 𐑕𐑳𐑛𐑩𐑯 𐑕𐑰𐑥𐑛 𐑕𐑳𐑥𐑣𐑬, 𐑦𐑯 𐑕𐑳𐑥 𐑒𐑢𐑽 𐑓𐑨𐑖𐑩𐑯, 𐑑 𐑕𐑢𐑦𐑥 𐑪𐑓 𐑦𐑯𐑑𐑵 𐑞 𐑺. 𐑲 𐑕𐑯𐑹𐑛; 𐑮𐑴𐑤𐑛 𐑴𐑝𐑼; 𐑯 𐑞𐑺 𐑲 𐑢𐑪𐑟 𐑦𐑯 𐑥𐑲 𐑣𐑨𐑥𐑩𐑒! 𐑯𐑬, 𐑢𐑪𐑑 𐑛𐑵 𐑿 𐑔𐑦𐑙𐑒 𐑝 𐑞𐑨𐑑 𐑛𐑮𐑰𐑥, ·𐑓𐑤𐑭𐑕𐑒?”

“𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴; 𐑦𐑑 𐑕𐑰𐑥𐑟 𐑩 𐑕𐑹𐑑 𐑝 𐑓𐑵𐑤𐑦𐑖 𐑑 𐑥𐑰, 𐑞𐑴.'”

“𐑥𐑱 𐑚𐑰; 𐑥𐑱 𐑚𐑰. 𐑚𐑳𐑑 𐑦𐑑'𐑕 𐑥𐑱𐑛 𐑩 𐑢𐑲𐑟 𐑥𐑨𐑯 𐑝 𐑥𐑰, ·𐑓𐑤𐑭𐑕𐑒. ·𐑛'𐑘𐑰 𐑕𐑰 ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑞𐑺, 𐑕𐑲𐑛𐑢𐑱𐑟 𐑤𐑫𐑒𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑕𐑑𐑻𐑯? 𐑢𐑧𐑤, 𐑞 𐑚𐑧𐑕𐑑 𐑔𐑦𐑙 𐑿 𐑒𐑨𐑯 𐑛𐑵, ·𐑓𐑤𐑭𐑕𐑒, 𐑦𐑟 𐑑 𐑤𐑧𐑑 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑩𐑤𐑴𐑯; 𐑯𐑧𐑝𐑼 𐑕𐑐𐑰𐑒 𐑑 𐑣𐑦𐑥, 𐑢𐑪𐑑𐑧𐑝𐑼 𐑣𐑰 𐑕𐑧𐑟. 𐑣𐑩𐑤𐑴! 𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑣𐑰 𐑖𐑬𐑑𐑕? 𐑣𐑸𐑒!”

“𐑥𐑭𐑕𐑑-𐑣𐑧𐑛, 𐑞𐑺! 𐑤𐑫𐑒 𐑖𐑸𐑐, 𐑷𐑤 𐑝 𐑘𐑰! 𐑞𐑺 𐑸 𐑢𐑱𐑤𐑟 𐑣𐑽𐑩𐑚𐑬𐑑𐑕!

“𐑦𐑓 𐑘𐑰 𐑕𐑰 𐑩 𐑢𐑲𐑑 𐑢𐑳𐑯, 𐑕𐑐𐑤𐑦𐑑 𐑘𐑹 𐑤𐑳𐑙𐑟 𐑓 𐑣𐑦𐑥!

“𐑢𐑪𐑑 𐑛𐑵 𐑿 𐑔𐑦𐑙𐑒 𐑝 𐑞𐑨𐑑 𐑯𐑬, ·𐑓𐑤𐑭𐑕𐑒? 𐑱𐑯𐑑 𐑞𐑺 𐑩 𐑕𐑥𐑷𐑤 𐑛𐑮𐑪𐑐 𐑝 𐑕𐑳𐑥𐑔𐑦𐑙 𐑒𐑢𐑽 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑, 𐑱? 𐑩 𐑢𐑲𐑑 𐑢𐑱𐑤—𐑛𐑦𐑛 𐑘𐑰 𐑥𐑸𐑒 𐑞𐑨𐑑, 𐑥𐑨𐑯? 𐑤𐑫𐑒 𐑘𐑰—𐑞𐑺'𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑕𐑐𐑧𐑖𐑩𐑤 𐑦𐑯 𐑞 𐑢𐑦𐑯𐑛. 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑓 𐑦𐑑, ·𐑓𐑤𐑭𐑕𐑒. 𐑱𐑣𐑨𐑚 𐑣𐑨𐑟 𐑞𐑨𐑑 𐑞𐑨𐑑'𐑕 𐑚𐑤𐑳𐑛𐑦 𐑪𐑯 𐑣𐑦𐑟 𐑥𐑲𐑯𐑛. 𐑚𐑳𐑑, 𐑥𐑳𐑥; 𐑣𐑰 𐑒𐑳𐑥𐑟 𐑞𐑦𐑕 𐑢𐑱.”
