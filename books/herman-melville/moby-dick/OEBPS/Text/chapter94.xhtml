·𐑗𐑨𐑐𐑑𐑼 94. 𐑩 ·𐑕𐑒𐑢𐑰𐑟 𐑝 𐑞 ·𐑣𐑨𐑯𐑛.

𐑞𐑨𐑑 𐑢𐑱𐑤 𐑝 ·𐑕𐑑𐑳𐑚'𐑟, 𐑕𐑴 𐑛𐑽𐑤𐑦 𐑐𐑻𐑗𐑩𐑕𐑑, 𐑢𐑪𐑟 𐑛𐑵𐑤𐑦 𐑚𐑮𐑷𐑑 𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑕𐑲𐑛, 𐑢𐑺 𐑷𐑤 𐑞𐑴𐑟 𐑒𐑳𐑑𐑦𐑙 𐑯 𐑣𐑶𐑕𐑑𐑦𐑙 𐑪𐑐𐑼𐑱𐑖𐑩𐑯𐑟 𐑐𐑮𐑰𐑝𐑾𐑕𐑤𐑦 𐑛𐑰𐑑𐑱𐑤𐑛, 𐑢𐑻 𐑮𐑧𐑜𐑿𐑤𐑼𐑤𐑦 𐑜𐑪𐑯 𐑔𐑮𐑵, 𐑰𐑝𐑩𐑯 𐑑 𐑞 𐑚𐑱𐑤𐑦𐑙 𐑝 𐑞 ·𐑣𐑲𐑛𐑩𐑤𐑚𐑼𐑩 ·𐑑𐑳𐑯, 𐑹 ·𐑒𐑱𐑕.

𐑢𐑲𐑤 𐑕𐑳𐑥 𐑢𐑻 𐑪𐑒𐑿𐑐𐑲𐑛 𐑢𐑦𐑞 𐑞𐑦𐑕 𐑤𐑨𐑑𐑼 𐑛𐑿𐑑𐑦, 𐑳𐑞𐑼𐑟 𐑢𐑻 𐑦𐑥𐑐𐑤𐑶𐑛 𐑦𐑯 𐑛𐑮𐑨𐑜𐑦𐑙 𐑩𐑢𐑱 𐑞 𐑤𐑸𐑡𐑼 𐑑𐑳𐑚𐑟, 𐑕𐑴 𐑕𐑵𐑯 𐑨𐑟 𐑓𐑦𐑤𐑛 𐑢𐑦𐑞 𐑞 𐑕𐑐𐑻𐑥; 𐑯 𐑢𐑧𐑯 𐑞 𐑐𐑮𐑪𐑐𐑼 𐑑𐑲𐑥 𐑼𐑲𐑝𐑛, 𐑞𐑦𐑕 𐑕𐑱𐑥 𐑕𐑐𐑻𐑥 𐑢𐑪𐑟 𐑒𐑺𐑓𐑩𐑤𐑦 𐑥𐑩𐑯𐑦𐑐𐑿𐑤𐑱𐑑𐑩𐑛 𐑺 𐑜𐑴𐑦𐑙 𐑑 𐑞 𐑑𐑮𐑲-𐑢𐑻𐑒𐑕, 𐑝 𐑢𐑦𐑗 𐑩𐑯𐑪𐑯.

𐑦𐑑 𐑣𐑨𐑛 𐑒𐑵𐑤𐑛 𐑯 𐑒𐑮𐑦𐑕𐑑𐑩𐑤𐑲𐑟𐑛 𐑑 𐑕𐑳𐑗 𐑩 𐑛𐑦𐑜𐑮𐑰, 𐑞𐑨𐑑 𐑢𐑧𐑯, 𐑢𐑦𐑞 𐑕𐑧𐑝𐑼𐑩𐑤 𐑳𐑞𐑼𐑟, 𐑲 𐑕𐑨𐑑 𐑛𐑬𐑯 𐑚𐑦𐑓𐑹 𐑩 𐑤𐑸𐑡 ·𐑒𐑪𐑯𐑕𐑑𐑩𐑯𐑑𐑰𐑯'𐑟 𐑚𐑭𐑔 𐑝 𐑦𐑑, 𐑲 𐑓𐑬𐑯𐑛 𐑦𐑑 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑒𐑪𐑯𐑒𐑮𐑰𐑑𐑩𐑛 𐑦𐑯𐑑𐑵 𐑤𐑳𐑥𐑐𐑕, 𐑣𐑽 𐑯 𐑞𐑺 𐑮𐑴𐑤𐑦𐑙 𐑩𐑚𐑬𐑑 𐑦𐑯 𐑞 𐑤𐑦𐑒𐑢𐑦𐑛 𐑐𐑸𐑑. 𐑦𐑑 𐑢𐑪𐑟 𐑬𐑼 𐑚𐑦𐑟𐑯𐑩𐑕 𐑑 𐑕𐑒𐑢𐑰𐑟 𐑞𐑰𐑟 𐑤𐑳𐑥𐑐𐑕 𐑚𐑨𐑒 𐑦𐑯𐑑𐑵 𐑓𐑤𐑵𐑦𐑛. 𐑩 𐑕𐑢𐑰𐑑 𐑯 𐑳𐑙𐑒𐑗𐑵𐑩𐑕 𐑛𐑿𐑑𐑦! 𐑯𐑴 𐑢𐑳𐑯𐑛𐑼 𐑞𐑨𐑑 𐑦𐑯 𐑴𐑤𐑛 𐑑𐑲𐑥𐑟 𐑞𐑦𐑕 𐑕𐑐𐑻𐑥 𐑢𐑪𐑟 𐑕𐑳𐑗 𐑩 𐑓𐑱𐑝𐑼𐑦𐑑 𐑒𐑪𐑟𐑥𐑧𐑑𐑦𐑒. 𐑕𐑳𐑗 𐑩 𐑒𐑤𐑽𐑼! 𐑕𐑳𐑗 𐑩 𐑕𐑢𐑰𐑑𐑩𐑯𐑼! 𐑕𐑳𐑗 𐑩 𐑕𐑪𐑓𐑩𐑯𐑼! 𐑕𐑳𐑗 𐑩 𐑛𐑦𐑤𐑦𐑖𐑩𐑕 𐑥𐑴𐑤𐑦𐑓𐑲𐑼! 𐑭𐑓𐑑𐑼 𐑣𐑨𐑝𐑦𐑙 𐑥𐑲 𐑣𐑨𐑯𐑛𐑟 𐑦𐑯 𐑦𐑑 𐑓 𐑴𐑯𐑤𐑦 𐑩 𐑓𐑿 𐑥𐑦𐑯𐑦𐑑𐑕, 𐑥𐑲 𐑓𐑦𐑙𐑜𐑼𐑟 𐑓𐑧𐑤𐑑 𐑤𐑲𐑒 𐑰𐑤𐑟, 𐑯 𐑚𐑦𐑜𐑨𐑯, 𐑨𐑟 𐑦𐑑 𐑢𐑻, 𐑑 𐑕𐑻𐑐𐑩𐑯𐑑𐑲𐑯 𐑯 𐑕𐑐𐑲𐑮𐑩𐑤𐑲𐑟.

𐑨𐑟 𐑲 𐑕𐑨𐑑 𐑞𐑺 𐑨𐑑 𐑥𐑲 𐑰𐑟, 𐑒𐑮𐑪𐑕-𐑤𐑧𐑜𐑩𐑛 𐑪𐑯 𐑞 𐑛𐑧𐑒; 𐑭𐑓𐑑𐑼 𐑞 𐑚𐑦𐑑𐑼 𐑦𐑜𐑟𐑻𐑖𐑩𐑯 𐑨𐑑 𐑞 𐑢𐑦𐑯𐑛𐑤𐑩𐑕; 𐑳𐑯𐑛𐑼 𐑩 𐑚𐑤𐑵 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤 𐑕𐑒𐑲; 𐑞 𐑖𐑦𐑐 𐑳𐑯𐑛𐑼 𐑦𐑯𐑛𐑩𐑤𐑩𐑯𐑑 𐑕𐑱𐑤, 𐑯 𐑜𐑤𐑲𐑛𐑦𐑙 𐑕𐑴 𐑕𐑼𐑰𐑯𐑤𐑦 𐑩𐑤𐑪𐑙; 𐑨𐑟 𐑲 𐑚𐑱𐑞𐑛 𐑥𐑲 𐑣𐑨𐑯𐑛𐑟 𐑩𐑥𐑳𐑙 𐑞𐑴𐑟 𐑕𐑪𐑓𐑑, 𐑡𐑧𐑯𐑑𐑩𐑤 𐑜𐑤𐑪𐑚𐑿𐑤𐑟 𐑝 𐑦𐑯𐑓𐑦𐑤𐑑𐑮𐑱𐑑𐑩𐑛 𐑑𐑦𐑖𐑿𐑟, 𐑢𐑴𐑝𐑩𐑯 𐑷𐑤𐑥𐑴𐑕𐑑 𐑢𐑦𐑞𐑦𐑯 𐑞 𐑬𐑼; 𐑨𐑟 𐑞𐑱 𐑮𐑦𐑗𐑤𐑦 𐑚𐑮𐑴𐑒 𐑑 𐑥𐑲 𐑓𐑦𐑙𐑜𐑼𐑟, 𐑯 𐑛𐑦𐑕𐑗𐑸𐑡𐑛 𐑷𐑤 𐑞𐑺 𐑪𐑐𐑿𐑤𐑩𐑯𐑕, 𐑤𐑲𐑒 𐑓𐑫𐑤𐑦 𐑮𐑲𐑐 𐑜𐑮𐑱𐑐𐑕 𐑞𐑺 𐑢𐑲𐑯; 𐑨𐑟 𐑲 𐑕𐑯𐑳𐑓𐑑 𐑳𐑐 𐑞𐑨𐑑 𐑳𐑯𐑒𐑩𐑯𐑑𐑨𐑥𐑦𐑯𐑱𐑑𐑩𐑛 𐑼𐑴𐑥𐑩,—𐑤𐑦𐑑𐑼𐑩𐑤𐑦 𐑯 𐑑𐑮𐑵𐑤𐑦, 𐑤𐑲𐑒 𐑞 𐑕𐑥𐑧𐑤 𐑝 𐑕𐑐𐑮𐑦𐑙 𐑝𐑲𐑩𐑤𐑩𐑑𐑕; 𐑲 𐑛𐑦𐑒𐑤𐑺 𐑑 𐑿, 𐑞𐑨𐑑 𐑓 𐑞 𐑑𐑲𐑥 𐑲 𐑤𐑦𐑝𐑛 𐑨𐑟 𐑦𐑯 𐑩 𐑥𐑳𐑕𐑒𐑦 𐑥𐑧𐑛𐑴; 𐑲 𐑓𐑼𐑜𐑪𐑑 𐑷𐑤 𐑩𐑚𐑬𐑑 𐑬𐑼 𐑣𐑪𐑮𐑦𐑚𐑩𐑤 𐑴𐑔; 𐑦𐑯 𐑞𐑨𐑑 𐑦𐑯𐑦𐑒𐑕𐑐𐑮𐑧𐑕𐑩𐑚𐑩𐑤 𐑕𐑐𐑻𐑥, 𐑲 𐑢𐑪𐑖𐑑 𐑥𐑲 𐑣𐑨𐑯𐑛𐑟 𐑯 𐑥𐑲 𐑣𐑸𐑑 𐑝 𐑦𐑑; 𐑲 𐑷𐑤𐑥𐑴𐑕𐑑 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑒𐑮𐑧𐑛𐑦𐑑 𐑞 𐑴𐑤𐑛 ·𐑐𐑨𐑮𐑩𐑕𐑧𐑤𐑕𐑩𐑯 𐑕𐑵𐑐𐑼𐑕𐑑𐑦𐑖𐑩𐑯 𐑞𐑨𐑑 𐑕𐑐𐑻𐑥 𐑦𐑟 𐑝 𐑮𐑺 𐑝𐑻𐑗𐑵 𐑦𐑯 𐑩𐑤𐑱𐑦𐑙 𐑞 𐑣𐑰𐑑 𐑝 𐑨𐑙𐑜𐑼; 𐑢𐑲𐑤 𐑚𐑱𐑞𐑦𐑙 𐑦𐑯 𐑞𐑨𐑑 𐑚𐑭𐑔, 𐑲 𐑓𐑧𐑤𐑑 𐑛𐑦𐑝𐑲𐑯𐑤𐑦 𐑓𐑮𐑰 𐑓𐑮𐑪𐑥 𐑷𐑤 𐑦𐑤-𐑢𐑦𐑤, 𐑹 𐑐𐑧𐑑𐑿𐑤𐑩𐑯𐑕, 𐑹 𐑥𐑨𐑤𐑦𐑕, 𐑝 𐑧𐑯𐑦 𐑕𐑹𐑑 𐑢𐑪𐑑𐑕𐑴𐑧𐑝𐑼.

𐑕𐑒𐑢𐑰𐑟! 𐑕𐑒𐑢𐑰𐑟! 𐑕𐑒𐑢𐑰𐑟! 𐑷𐑤 𐑞 𐑥𐑹𐑯𐑦𐑙 𐑤𐑪𐑙; 𐑲 𐑕𐑒𐑢𐑰𐑟𐑛 𐑞𐑨𐑑 𐑕𐑐𐑻𐑥 𐑑𐑦𐑤 𐑲 𐑥𐑲𐑕𐑧𐑤𐑓 𐑷𐑤𐑥𐑴𐑕𐑑 𐑥𐑧𐑤𐑑𐑩𐑛 𐑦𐑯𐑑𐑵 𐑦𐑑; 𐑲 𐑕𐑒𐑢𐑰𐑟𐑛 𐑞𐑨𐑑 𐑕𐑐𐑻𐑥 𐑑𐑦𐑤 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡 𐑕𐑹𐑑 𐑝 𐑦𐑯𐑕𐑨𐑯𐑦𐑑𐑦 𐑒𐑱𐑥 𐑴𐑝𐑼 𐑥𐑰; 𐑯 𐑲 𐑓𐑬𐑯𐑛 𐑥𐑲𐑕𐑧𐑤𐑓 𐑳𐑯𐑢𐑦𐑑𐑦𐑙𐑤𐑦 𐑕𐑒𐑢𐑰𐑟𐑦𐑙 𐑥𐑲 𐑒𐑴-𐑤𐑱𐑚𐑼𐑼𐑟' 𐑣𐑨𐑯𐑛𐑟 𐑦𐑯 𐑦𐑑, 𐑥𐑦𐑕𐑑𐑱𐑒𐑦𐑙 𐑞𐑺 𐑣𐑨𐑯𐑛𐑟 𐑓 𐑞 𐑡𐑧𐑯𐑑𐑩𐑤 𐑜𐑤𐑪𐑚𐑿𐑤𐑟. 𐑕𐑳𐑗 𐑩𐑯 𐑩𐑚𐑬𐑯𐑛𐑦𐑙, 𐑩𐑓𐑧𐑒𐑖𐑩𐑯𐑩𐑑, 𐑓𐑮𐑧𐑯𐑛𐑤𐑦, 𐑤𐑳𐑝𐑦𐑙 𐑓𐑰𐑤𐑦𐑙 𐑛𐑦𐑛 𐑞𐑦𐑕 𐑨𐑝𐑩𐑒𐑱𐑖𐑩𐑯 𐑚𐑦𐑜𐑧𐑑; 𐑞𐑨𐑑 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑲 𐑢𐑪𐑟 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤𐑦 𐑕𐑒𐑢𐑰𐑟𐑦𐑙 𐑞𐑺 𐑣𐑨𐑯𐑛𐑟, 𐑯 𐑤𐑫𐑒𐑦𐑙 𐑳𐑐 𐑦𐑯𐑑𐑵 𐑞𐑺 𐑲𐑟 𐑕𐑧𐑯𐑑𐑦𐑥𐑧𐑯𐑑𐑩𐑤𐑦; 𐑨𐑟 𐑥𐑳𐑗 𐑨𐑟 𐑑 𐑕𐑱,—𐑴! 𐑥𐑲 𐑛𐑽 𐑓𐑧𐑤𐑴 𐑚𐑰𐑦𐑙𐑟, 𐑢𐑲 𐑖𐑵𐑛 𐑢𐑰 𐑤𐑪𐑙𐑜𐑼 𐑗𐑧𐑮𐑦𐑖 𐑧𐑯𐑦 𐑕𐑴𐑖𐑩𐑤 𐑩𐑕𐑻𐑚𐑦𐑑𐑦𐑟, 𐑹 𐑯𐑴 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑦𐑤-𐑣𐑿𐑥𐑼 𐑹 𐑧𐑯𐑝𐑦! 𐑒𐑳𐑥; 𐑤𐑧𐑑 𐑳𐑕 𐑕𐑒𐑢𐑰𐑟 𐑣𐑨𐑯𐑛𐑟 𐑷𐑤 𐑮𐑬𐑯𐑛; 𐑯𐑱, 𐑤𐑧𐑑 𐑳𐑕 𐑷𐑤 𐑕𐑒𐑢𐑰𐑟 𐑬𐑼𐑕𐑧𐑤𐑝𐑟 𐑦𐑯𐑑𐑵 𐑰𐑗 𐑳𐑞𐑼; 𐑤𐑧𐑑 𐑳𐑕 𐑕𐑒𐑢𐑰𐑟 𐑬𐑼𐑕𐑧𐑤𐑝𐑟 𐑿𐑯𐑦𐑝𐑻𐑕𐑩𐑤𐑦 𐑦𐑯𐑑𐑵 𐑞 𐑝𐑧𐑮𐑦 𐑥𐑦𐑤𐑒 𐑯 𐑕𐑐𐑻𐑥 𐑝 𐑒𐑲𐑯𐑛𐑯𐑩𐑕.

𐑯𐑬, 𐑢𐑲𐑤 𐑛𐑦𐑕𐑒𐑹𐑕𐑦𐑙 𐑝 𐑕𐑐𐑻𐑥, 𐑦𐑑 𐑚𐑦𐑣𐑵𐑝𐑟 𐑑 𐑕𐑐𐑰𐑒 𐑝 𐑳𐑞𐑼 𐑔𐑦𐑙𐑟 𐑩𐑒𐑦𐑯 𐑑 𐑦𐑑, 𐑦𐑯 𐑞 𐑚𐑦𐑟𐑯𐑩𐑕 𐑝 𐑐𐑮𐑦𐑐𐑺𐑦𐑙 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤 𐑓 𐑞 𐑑𐑮𐑲-𐑢𐑻𐑒𐑕.

𐑓𐑻𐑕𐑑 𐑒𐑳𐑥𐑟 𐑢𐑲𐑑-𐑣𐑹𐑕, 𐑕𐑴 𐑒𐑷𐑤𐑛, 𐑢𐑦𐑗 𐑦𐑟 𐑩𐑚𐑑𐑱𐑯𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑱𐑐𐑼𐑦𐑙 𐑐𐑸𐑑 𐑝 𐑞 𐑓𐑦𐑖, 𐑯 𐑷𐑤𐑕𐑴 𐑓𐑮𐑪𐑥 𐑞 𐑔𐑦𐑒𐑼 𐑐𐑹𐑖𐑩𐑯𐑟 𐑝 𐑣𐑦𐑟 𐑓𐑤𐑵𐑒𐑕. 𐑦𐑑 𐑦𐑟 𐑑𐑳𐑓 𐑢𐑦𐑞 𐑒𐑩𐑯𐑡𐑰𐑤𐑛 𐑑𐑧𐑯𐑛𐑩𐑯𐑟—𐑩 𐑢𐑪𐑛 𐑝 𐑥𐑳𐑕𐑩𐑤—𐑚𐑳𐑑 𐑕𐑑𐑦𐑤 𐑒𐑩𐑯𐑑𐑱𐑯𐑟 𐑕𐑳𐑥 𐑶𐑤. 𐑭𐑓𐑑𐑼 𐑚𐑰𐑦𐑙 𐑕𐑧𐑝𐑼𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑱𐑤, 𐑞 𐑢𐑲𐑑-𐑣𐑹𐑕 𐑦𐑟 𐑓𐑻𐑕𐑑 𐑒𐑳𐑑 𐑦𐑯𐑑𐑵 𐑐𐑹𐑑𐑩𐑚𐑩𐑤 𐑪𐑚𐑤𐑪𐑙𐑟 𐑺 𐑜𐑴𐑦𐑙 𐑑 𐑞 𐑥𐑦𐑯𐑕𐑼. 𐑞𐑱 𐑤𐑫𐑒 𐑥𐑳𐑗 𐑤𐑲𐑒 𐑚𐑤𐑪𐑒𐑕 𐑝 ·𐑚𐑸𐑒𐑖𐑼 𐑥𐑸𐑚𐑩𐑤.

𐑐𐑤𐑳𐑥-𐑐𐑫𐑛𐑦𐑙 𐑦𐑟 𐑞 𐑑𐑻𐑥 𐑚𐑦𐑕𐑑𐑴𐑛 𐑩𐑐𐑪𐑯 𐑕𐑻𐑑𐑩𐑯 𐑓𐑮𐑨𐑜𐑥𐑩𐑯𐑑𐑼𐑦 𐑐𐑸𐑑𐑕 𐑝 𐑞 𐑢𐑱𐑤'𐑟 𐑓𐑤𐑧𐑖, 𐑣𐑽 𐑯 𐑞𐑺 𐑨𐑛𐑣𐑽𐑦𐑙 𐑑 𐑞 𐑚𐑤𐑨𐑙𐑒𐑩𐑑 𐑝 𐑚𐑤𐑳𐑚𐑼, 𐑯 𐑪𐑓𐑩𐑯 𐑐𐑸𐑑𐑦𐑕𐑦𐑐𐑱𐑑𐑦𐑙 𐑑 𐑩 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑛𐑦𐑜𐑮𐑰 𐑦𐑯 𐑦𐑑𐑕 𐑳𐑙𐑒𐑗𐑵𐑩𐑕𐑯𐑩𐑕. 𐑦𐑑 𐑦𐑟 𐑩 𐑥𐑴𐑕𐑑 𐑮𐑦𐑓𐑮𐑧𐑖𐑦𐑙, 𐑒𐑩𐑯𐑝𐑦𐑝𐑾𐑤, 𐑚𐑿𐑑𐑦𐑓𐑩𐑤 𐑪𐑚𐑡𐑧𐑒𐑑 𐑑 𐑚𐑦𐑣𐑴𐑤𐑛. 𐑨𐑟 𐑦𐑑𐑕 𐑯𐑱𐑥 𐑦𐑥𐑐𐑹𐑑𐑕, 𐑦𐑑 𐑦𐑟 𐑝 𐑩𐑯 𐑦𐑒𐑕𐑰𐑛𐑦𐑙𐑤𐑦 𐑮𐑦𐑗, 𐑥𐑪𐑑𐑩𐑤𐑛 𐑑𐑦𐑯𐑑, 𐑢𐑦𐑞 𐑩 𐑚𐑧𐑕𐑑𐑮𐑰𐑒𐑑 𐑕𐑯𐑴𐑦 𐑯 𐑜𐑴𐑤𐑛𐑩𐑯 𐑜𐑮𐑬𐑯𐑛, 𐑛𐑪𐑑𐑩𐑛 𐑢𐑦𐑞 𐑕𐑐𐑪𐑑𐑕 𐑝 𐑞 𐑛𐑰𐑐𐑩𐑕𐑑 𐑒𐑮𐑦𐑥𐑟𐑩𐑯 𐑯 𐑐𐑻𐑐𐑩𐑤. 𐑦𐑑 𐑦𐑟 𐑐𐑤𐑳𐑥𐑟 𐑝 𐑮𐑵𐑚𐑦𐑟, 𐑦𐑯 𐑐𐑦𐑒𐑗𐑼𐑟 𐑝 𐑕𐑦𐑑𐑮𐑩𐑯. 𐑕𐑐𐑲𐑑 𐑝 𐑮𐑰𐑟𐑩𐑯, 𐑦𐑑 𐑦𐑟 𐑣𐑸𐑛 𐑑 𐑒𐑰𐑐 𐑘𐑹𐑕𐑧𐑤𐑓 𐑓𐑮𐑪𐑥 𐑰𐑑𐑦𐑙 𐑦𐑑. 𐑲 𐑒𐑩𐑯𐑓𐑧𐑕, 𐑞𐑨𐑑 𐑢𐑳𐑯𐑕 𐑲 𐑕𐑑𐑴𐑤 𐑚𐑦𐑣𐑲𐑯𐑛 𐑞 𐑓𐑹𐑥𐑭𐑕𐑑 𐑑 𐑑𐑮𐑲 𐑦𐑑. 𐑦𐑑 𐑑𐑱𐑕𐑑𐑩𐑛 𐑕𐑳𐑥𐑔𐑦𐑙 𐑨𐑟 𐑲 𐑖𐑵𐑛 𐑒𐑩𐑯𐑕𐑰𐑝 𐑩 𐑮𐑶𐑩𐑤 𐑒𐑳𐑑𐑤𐑩𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑔𐑲 𐑝 ·𐑤𐑵𐑦 𐑤𐑩 ·𐑜𐑮𐑴𐑟 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑑𐑱𐑕𐑑𐑩𐑛, 𐑕𐑩𐑐𐑴𐑟𐑦𐑙 𐑣𐑦𐑥 𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑒𐑦𐑤𐑛 𐑞 𐑓𐑻𐑕𐑑 𐑛𐑱 𐑭𐑓𐑑𐑼 𐑞 𐑝𐑧𐑯𐑦𐑟𐑩𐑯 𐑕𐑰𐑟𐑩𐑯, 𐑯 𐑞𐑨𐑑 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑝𐑧𐑯𐑦𐑟𐑩𐑯 𐑕𐑰𐑟𐑩𐑯 𐑒𐑩𐑯𐑑𐑧𐑥𐑐𐑼𐑼𐑦 𐑢𐑦𐑞 𐑩𐑯 𐑳𐑯𐑿𐑠𐑵𐑩𐑤𐑦 𐑓𐑲𐑯 𐑝𐑦𐑯𐑑𐑦𐑡 𐑝 𐑞 𐑝𐑦𐑯𐑘𐑼𐑛𐑟 𐑝 ·𐑖𐑨𐑥𐑐𐑱𐑯.

𐑞𐑺 𐑦𐑟 𐑩𐑯𐑳𐑞𐑼 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕, 𐑯 𐑩 𐑝𐑧𐑮𐑦 𐑕𐑦𐑙𐑜𐑿𐑤𐑼 𐑢𐑳𐑯, 𐑢𐑦𐑗 𐑑𐑻𐑯𐑟 𐑳𐑐 𐑦𐑯 𐑞 𐑒𐑹𐑕 𐑝 𐑞𐑦𐑕 𐑚𐑦𐑟𐑯𐑩𐑕, 𐑚𐑳𐑑 𐑢𐑦𐑗 𐑲 𐑓𐑰𐑤 𐑦𐑑 𐑑 𐑚𐑰 𐑝𐑧𐑮𐑦 𐑐𐑳𐑟𐑩𐑤𐑦𐑙 𐑨𐑛𐑦𐑒𐑢𐑩𐑑𐑤𐑦 𐑑 𐑛𐑦𐑕𐑒𐑮𐑲𐑚. 𐑦𐑑 𐑦𐑟 𐑒𐑷𐑤𐑛 𐑕𐑤𐑪𐑚𐑜𐑴𐑤𐑾𐑯; 𐑩𐑯 𐑨𐑐𐑩𐑤𐑱𐑖𐑩𐑯 𐑼𐑦𐑡𐑦𐑯𐑩𐑤 𐑢𐑦𐑞 𐑞 𐑢𐑱𐑤𐑥𐑧𐑯, 𐑯 𐑰𐑝𐑩𐑯 𐑕𐑴 𐑦𐑟 𐑞 𐑯𐑱𐑗𐑼 𐑝 𐑞 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕. 𐑦𐑑 𐑦𐑟 𐑩𐑯 𐑦𐑯𐑧𐑓𐑩𐑚𐑤𐑦 𐑵𐑟𐑦, 𐑕𐑑𐑮𐑦𐑙𐑦 𐑩𐑓𐑺, 𐑥𐑴𐑕𐑑 𐑓𐑮𐑰𐑒𐑢𐑩𐑯𐑑𐑤𐑦 𐑓𐑬𐑯𐑛 𐑦𐑯 𐑞 𐑑𐑳𐑚𐑟 𐑝 𐑕𐑐𐑻𐑥, 𐑭𐑓𐑑𐑼 𐑩 𐑐𐑮𐑴𐑤𐑪𐑙𐑛 𐑕𐑒𐑢𐑰𐑟𐑦𐑙, 𐑯 𐑕𐑳𐑚𐑕𐑦𐑒𐑢𐑩𐑯𐑑 𐑛𐑦𐑒𐑨𐑯𐑑𐑦𐑙. 𐑲 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑑 𐑚𐑰 𐑞 𐑢𐑳𐑯𐑛𐑮𐑩𐑕𐑤𐑦 𐑔𐑦𐑯, 𐑮𐑳𐑐𐑗𐑼𐑛 𐑥𐑧𐑥𐑚𐑮𐑱𐑯𐑟 𐑝 𐑞 𐑒𐑱𐑕, 𐑒𐑴𐑩𐑤𐑧𐑕𐑦𐑙.

𐑜𐑻𐑦, 𐑕𐑴 𐑒𐑷𐑤𐑛, 𐑦𐑟 𐑩 𐑑𐑻𐑥 𐑐𐑮𐑪𐑐𐑼𐑤𐑦 𐑚𐑦𐑤𐑪𐑙𐑦𐑙 𐑑 𐑮𐑲𐑑 𐑢𐑱𐑤𐑥𐑧𐑯, 𐑚𐑳𐑑 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑦𐑯𐑕𐑦𐑛𐑧𐑯𐑑𐑩𐑤𐑦 𐑿𐑟𐑛 𐑚𐑲 𐑞 𐑕𐑐𐑻𐑥 𐑓𐑦𐑖𐑼𐑥𐑧𐑯. 𐑦𐑑 𐑛𐑧𐑟𐑦𐑜𐑯𐑱𐑑𐑕 𐑞 𐑛𐑸𐑒, 𐑜𐑤𐑵𐑑𐑦𐑯𐑩𐑕 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕 𐑢𐑦𐑗 𐑦𐑟 𐑕𐑒𐑮𐑱𐑐𐑑 𐑪𐑓 𐑞 𐑚𐑨𐑒 𐑝 𐑞 ·𐑜𐑮𐑰𐑯𐑤𐑩𐑯𐑛 𐑹 𐑮𐑲𐑑 𐑢𐑱𐑤, 𐑯 𐑥𐑳𐑗 𐑝 𐑢𐑦𐑗 𐑒𐑳𐑝𐑼𐑟 𐑞 𐑛𐑧𐑒𐑕 𐑝 𐑞𐑴𐑟 𐑦𐑯𐑓𐑽𐑽 𐑕𐑴𐑤𐑟 𐑣𐑵 𐑣𐑳𐑯𐑑 𐑞𐑨𐑑 𐑦𐑜𐑯𐑴𐑚𐑩𐑤 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯.

𐑯𐑦𐑐𐑼𐑟. 𐑕𐑑𐑮𐑦𐑒𐑑𐑤𐑦 𐑞𐑦𐑕 𐑢𐑻𐑛 𐑦𐑟 𐑯𐑪𐑑 𐑦𐑯𐑛𐑦𐑡𐑦𐑯𐑩𐑕 𐑑 𐑞 𐑢𐑱𐑤'𐑟 𐑝𐑴𐑒𐑨𐑚𐑿𐑤𐑼𐑦. 𐑚𐑳𐑑 𐑨𐑟 𐑩𐑐𐑤𐑲𐑛 𐑚𐑲 𐑢𐑱𐑤𐑥𐑧𐑯, 𐑦𐑑 𐑚𐑦𐑒𐑳𐑥𐑟 𐑕𐑴. 𐑩 𐑢𐑱𐑤𐑥𐑩𐑯'𐑟 𐑯𐑦𐑐𐑼 𐑦𐑟 𐑩 𐑖𐑹𐑑 𐑓𐑻𐑥 𐑕𐑑𐑮𐑦𐑐 𐑝 𐑑𐑧𐑯𐑛𐑦𐑯𐑩𐑕 𐑕𐑑𐑳𐑓 𐑒𐑳𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑱𐑐𐑼𐑦𐑙 𐑐𐑸𐑑 𐑝 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯'𐑟 𐑑𐑱𐑤: 𐑦𐑑 𐑨𐑝𐑼𐑦𐑡𐑩𐑟 𐑩𐑯 𐑦𐑯𐑗 𐑦𐑯 𐑔𐑦𐑒𐑯𐑩𐑕, 𐑯 𐑓 𐑞 𐑮𐑧𐑕𐑑, 𐑦𐑟 𐑩𐑚𐑬𐑑 𐑞 𐑕𐑲𐑟 𐑝 𐑞 𐑲𐑼𐑯 𐑐𐑸𐑑 𐑝 𐑩 𐑣𐑴. 𐑧𐑡𐑢𐑲𐑟 𐑥𐑵𐑝𐑛 𐑩𐑤𐑪𐑙 𐑞 𐑶𐑤𐑦 𐑛𐑧𐑒, 𐑦𐑑 𐑪𐑐𐑼𐑱𐑑𐑕 𐑤𐑲𐑒 𐑩 𐑤𐑧𐑞𐑼𐑯 𐑕𐑒𐑢𐑦𐑤𐑡𐑰; 𐑯 𐑚𐑲 𐑯𐑱𐑥𐑤𐑩𐑕 𐑚𐑤𐑨𐑯𐑛𐑦𐑖𐑥𐑩𐑯𐑑𐑕, 𐑨𐑟 𐑝 𐑥𐑨𐑡𐑦𐑒, 𐑩𐑤𐑫𐑼𐑟 𐑩𐑤𐑪𐑙 𐑢𐑦𐑞 𐑦𐑑 𐑷𐑤 𐑦𐑥𐑐𐑘𐑫𐑼𐑦𐑑𐑦𐑟.

𐑚𐑳𐑑 𐑑 𐑤𐑻𐑯 𐑷𐑤 𐑩𐑚𐑬𐑑 𐑞𐑰𐑟 𐑮𐑧𐑒𐑩𐑯𐑛𐑲𐑑 𐑥𐑨𐑑𐑼𐑟, 𐑘𐑹 𐑚𐑧𐑕𐑑 𐑢𐑱 𐑦𐑟 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑑 𐑛𐑦𐑕𐑧𐑯𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑚𐑤𐑳𐑚𐑼-𐑮𐑵𐑥, 𐑯 𐑣𐑨𐑝 𐑩 𐑤𐑪𐑙 𐑑𐑷𐑒 𐑢𐑦𐑞 𐑦𐑑𐑕 𐑦𐑯𐑥𐑱𐑑𐑕. 𐑞𐑦𐑕 𐑐𐑤𐑱𐑕 𐑣𐑨𐑟 𐑐𐑮𐑰𐑝𐑾𐑕𐑤𐑦 𐑚𐑰𐑯 𐑥𐑧𐑯𐑖𐑩𐑯𐑛 𐑨𐑟 𐑞 𐑮𐑦𐑕𐑧𐑐𐑑𐑩𐑒𐑩𐑤 𐑓 𐑞 𐑚𐑤𐑨𐑙𐑒𐑩𐑑-𐑐𐑰𐑕𐑩𐑟, 𐑢𐑧𐑯 𐑕𐑑𐑮𐑦𐑐𐑑 𐑯 𐑣𐑶𐑕𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑱𐑤. 𐑢𐑧𐑯 𐑞 𐑐𐑮𐑪𐑐𐑼 𐑑𐑲𐑥 𐑼𐑲𐑝𐑟 𐑓 𐑒𐑳𐑑𐑦𐑙 𐑳𐑐 𐑦𐑑𐑕 𐑒𐑪𐑯𐑑𐑧𐑯𐑑𐑕, 𐑞𐑦𐑕 𐑩𐑐𐑸𐑑𐑥𐑩𐑯𐑑 𐑦𐑟 𐑩 𐑕𐑰𐑯 𐑝 𐑑𐑧𐑮𐑼 𐑑 𐑷𐑤 𐑑𐑲𐑮𐑴𐑟, 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑚𐑲 𐑯𐑲𐑑. 𐑪𐑯 𐑢𐑳𐑯 𐑕𐑲𐑛, 𐑤𐑦𐑑 𐑚𐑲 𐑩 𐑛𐑳𐑤 𐑤𐑨𐑯𐑑𐑼𐑯, 𐑩 𐑕𐑐𐑱𐑕 𐑣𐑨𐑟 𐑚𐑰𐑯 𐑤𐑧𐑓𐑑 𐑒𐑤𐑽 𐑓 𐑞 𐑢𐑻𐑒𐑥𐑧𐑯. 𐑞𐑱 𐑡𐑧𐑯𐑼𐑩𐑤𐑦 𐑜𐑴 𐑦𐑯 𐑐𐑺𐑟,—𐑩 𐑐𐑲𐑒-𐑯-𐑜𐑨𐑓𐑥𐑩𐑯 𐑯 𐑩 𐑕𐑐𐑱𐑛-𐑥𐑨𐑯. 𐑞 𐑢𐑱𐑤𐑦𐑙-𐑐𐑲𐑒 𐑦𐑟 𐑕𐑦𐑥𐑦𐑤𐑼 𐑑 𐑩 𐑓𐑮𐑦𐑜𐑩𐑑'𐑕 𐑚𐑹𐑛𐑦𐑙-𐑢𐑧𐑐𐑩𐑯 𐑝 𐑞 𐑕𐑱𐑥 𐑯𐑱𐑥. 𐑞 𐑜𐑨𐑓 𐑦𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑤𐑲𐑒 𐑩 𐑚𐑴𐑑-𐑣𐑫𐑒. 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑜𐑨𐑓, 𐑞 𐑜𐑨𐑓𐑥𐑩𐑯 𐑣𐑫𐑒𐑕 𐑪𐑯 𐑑 𐑩 𐑖𐑰𐑑 𐑝 𐑚𐑤𐑳𐑚𐑼, 𐑯 𐑕𐑑𐑮𐑲𐑝𐑟 𐑑 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑓𐑮𐑪𐑥 𐑕𐑤𐑦𐑐𐑦𐑙, 𐑨𐑟 𐑞 𐑖𐑦𐑐 𐑐𐑦𐑗𐑩𐑟 𐑯 𐑤𐑻𐑗𐑩𐑟 𐑩𐑚𐑬𐑑. 𐑥𐑰𐑯𐑢𐑲𐑤, 𐑞 𐑕𐑐𐑱𐑛-𐑥𐑨𐑯 𐑕𐑑𐑨𐑯𐑛𐑟 𐑪𐑯 𐑞 𐑖𐑰𐑑 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑐𐑻𐑐𐑩𐑯𐑛𐑦𐑒𐑿𐑤𐑼𐑤𐑦 𐑗𐑪𐑐𐑦𐑙 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑐𐑹𐑑𐑩𐑚𐑩𐑤 𐑣𐑹𐑕-𐑐𐑰𐑕𐑩𐑟. 𐑞𐑦𐑕 𐑕𐑐𐑱𐑛 𐑦𐑟 𐑖𐑸𐑐 𐑨𐑟 𐑣𐑴𐑯 𐑒𐑨𐑯 𐑥𐑱𐑒 𐑦𐑑; 𐑞 𐑕𐑐𐑱𐑛𐑥𐑩𐑯'𐑟 𐑓𐑰𐑑 𐑸 𐑖𐑵𐑤𐑩𐑕; 𐑞 𐑔𐑦𐑙 𐑣𐑰 𐑕𐑑𐑨𐑯𐑛𐑟 𐑪𐑯 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑦𐑮𐑦𐑟𐑦𐑕𐑑𐑩𐑚𐑤𐑦 𐑕𐑤𐑲𐑛 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑤𐑲𐑒 𐑩 𐑕𐑤𐑧𐑡. 𐑦𐑓 𐑣𐑰 𐑒𐑳𐑑𐑕 𐑪𐑓 𐑢𐑳𐑯 𐑝 𐑣𐑦𐑟 𐑴𐑯 𐑑𐑴𐑟, 𐑹 𐑢𐑳𐑯 𐑝 𐑣𐑦𐑟 𐑩𐑕𐑦𐑕𐑑𐑩𐑯𐑑𐑕', 𐑢𐑫𐑛 𐑿 𐑚𐑰 𐑝𐑧𐑮𐑦 𐑥𐑳𐑗 𐑩𐑕𐑑𐑪𐑯𐑦𐑖𐑑? 𐑑𐑴𐑟 𐑸 𐑕𐑒𐑺𐑕 𐑩𐑥𐑳𐑙 𐑝𐑧𐑑𐑼𐑩𐑯 𐑚𐑤𐑳𐑚𐑼-𐑮𐑵𐑥 𐑥𐑧𐑯.
