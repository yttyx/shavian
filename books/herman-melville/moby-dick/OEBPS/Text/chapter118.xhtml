·𐑗𐑨𐑐𐑑𐑼 118. 𐑞 ·𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑.

𐑞 𐑕𐑰𐑟𐑩𐑯 𐑓 𐑞 ·𐑤𐑲𐑯 𐑨𐑑 𐑤𐑧𐑙𐑔 𐑛𐑮𐑵 𐑯𐑽; 𐑯 𐑧𐑝𐑮𐑦 𐑛𐑱 𐑢𐑧𐑯 ·𐑱𐑣𐑨𐑚, 𐑒𐑳𐑥𐑦𐑙 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯, 𐑒𐑭𐑕𐑑 𐑣𐑦𐑟 𐑲𐑟 𐑩𐑤𐑪𐑓𐑑, 𐑞 𐑝𐑦𐑡𐑦𐑤𐑩𐑯𐑑 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯 𐑢𐑫𐑛 𐑪𐑕𐑑𐑧𐑯𐑑𐑱𐑖𐑩𐑕𐑤𐑦 𐑣𐑨𐑯𐑛𐑩𐑤 𐑣𐑦𐑟 𐑕𐑐𐑴𐑒𐑕, 𐑯 𐑞 𐑰𐑜𐑼 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑒𐑢𐑦𐑒𐑤𐑦 𐑮𐑳𐑯 𐑑 𐑞 𐑚𐑮𐑱𐑕𐑩𐑟, 𐑯 𐑢𐑫𐑛 𐑕𐑑𐑨𐑯𐑛 𐑞𐑺 𐑢𐑦𐑞 𐑷𐑤 𐑞𐑺 𐑲𐑟 𐑕𐑧𐑯𐑑𐑮𐑩𐑤𐑦 𐑓𐑦𐑒𐑕𐑑 𐑪𐑯 𐑞 𐑯𐑱𐑤𐑛 𐑛𐑳𐑚𐑤𐑵𐑯; 𐑦𐑥𐑐𐑱𐑖𐑩𐑯𐑑 𐑓 𐑞 𐑹𐑛𐑼 𐑑 𐑐𐑶𐑯𐑑 𐑞 𐑖𐑦𐑐'𐑕 𐑐𐑮𐑬 𐑓 𐑞 𐑦𐑒𐑢𐑱𐑑𐑼. 𐑦𐑯 𐑜𐑫𐑛 𐑑𐑲𐑥 𐑞 𐑹𐑛𐑼 𐑒𐑱𐑥. 𐑦𐑑 𐑢𐑪𐑟 𐑣𐑸𐑛 𐑩𐑐𐑪𐑯 𐑣𐑲 𐑯𐑵𐑯; 𐑯 ·𐑱𐑣𐑨𐑚, 𐑕𐑰𐑑𐑩𐑛 𐑦𐑯 𐑞 𐑚𐑬𐑟 𐑝 𐑣𐑦𐑟 𐑣𐑲-𐑣𐑶𐑕𐑑𐑩𐑛 𐑚𐑴𐑑, 𐑢𐑪𐑟 𐑩𐑚𐑬𐑑 𐑑𐑱𐑒𐑦𐑙 𐑣𐑦𐑟 𐑢𐑴𐑯𐑑𐑩𐑛 𐑛𐑱𐑤𐑦 𐑪𐑚𐑟𐑼𐑝𐑱𐑖𐑩𐑯 𐑝 𐑞 𐑕𐑳𐑯 𐑑 𐑛𐑦𐑑𐑻𐑥𐑦𐑯 𐑣𐑦𐑟 𐑤𐑨𐑑𐑦𐑑𐑿𐑛.

𐑞𐑧𐑯 𐑜𐑱𐑟𐑦𐑙 𐑨𐑑 𐑣𐑦𐑟 𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑, 𐑯 𐑣𐑨𐑯𐑛𐑩𐑤𐑦𐑙, 𐑢𐑳𐑯 𐑭𐑓𐑑𐑼 𐑞 𐑳𐑞𐑼, 𐑦𐑑𐑕 𐑯𐑿𐑥𐑼𐑩𐑕 𐑒𐑩𐑚𐑨𐑤𐑦𐑕𐑑𐑦𐑒𐑩𐑤 𐑒𐑩𐑯𐑑𐑮𐑲𐑝𐑩𐑯𐑕𐑩𐑟, 𐑣𐑰 𐑐𐑪𐑯𐑛𐑼𐑛 𐑩𐑜𐑱𐑯, 𐑯 𐑥𐑳𐑑𐑼𐑛: “𐑓𐑵𐑤𐑦𐑖 𐑑𐑶! 𐑚𐑱𐑚𐑦𐑟' 𐑐𐑤𐑱𐑔𐑦𐑙 𐑝 𐑣𐑷𐑑𐑦 ·𐑨𐑛𐑥𐑼𐑩𐑤𐑟, 𐑯 ·𐑒𐑪𐑥𐑩𐑛𐑹𐑟, 𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯𐑟; 𐑞 𐑢𐑻𐑤𐑛 𐑚𐑮𐑨𐑜𐑟 𐑝 𐑞𐑰, 𐑝 𐑞𐑲 𐑒𐑳𐑯𐑦𐑙 𐑯 𐑥𐑲𐑑; 𐑚𐑳𐑑 𐑢𐑪𐑑 𐑭𐑓𐑑𐑼 𐑷𐑤 𐑒𐑨𐑯𐑕𐑑 𐑞𐑬 𐑛𐑵, 𐑚𐑳𐑑 𐑑𐑧𐑤 𐑞 𐑐𐑫𐑼, 𐑐𐑦𐑑𐑦𐑓𐑩𐑤 𐑐𐑶𐑯𐑑, 𐑢𐑺 𐑞𐑬 𐑞𐑲𐑕𐑧𐑤𐑓 𐑣𐑨𐑐𐑩𐑯𐑩𐑕𐑑 𐑑 𐑚𐑰 𐑪𐑯 𐑞𐑦𐑕 𐑢𐑲𐑛 𐑐𐑤𐑨𐑯𐑩𐑑, 𐑯 𐑞 𐑣𐑨𐑯𐑛 𐑞𐑨𐑑 𐑣𐑴𐑤𐑛𐑟 𐑞𐑰: 𐑯𐑴! 𐑯𐑪𐑑 𐑢𐑳𐑯 𐑡𐑪𐑑 𐑥𐑹! 𐑞𐑬 𐑒𐑨𐑯𐑕𐑑 𐑯𐑪𐑑 𐑑𐑧𐑤 𐑢𐑺 𐑢𐑳𐑯 𐑛𐑮𐑪𐑐 𐑝 𐑢𐑷𐑑𐑼 𐑹 𐑢𐑳𐑯 𐑜𐑮𐑱𐑯 𐑝 𐑕𐑨𐑯𐑛 𐑢𐑦𐑤 𐑚𐑰 𐑑-𐑥𐑪𐑮𐑴 𐑯𐑵𐑯; 𐑯 𐑘𐑧𐑑 𐑢𐑦𐑞 𐑞𐑲 𐑦𐑥𐑐𐑩𐑑𐑩𐑯𐑕 𐑞𐑬 𐑦𐑯𐑕𐑳𐑤𐑑𐑩𐑕𐑑 𐑞 𐑕𐑳𐑯! 𐑕𐑲𐑩𐑯𐑕! 𐑒𐑻𐑕 𐑞𐑰, 𐑞𐑬 𐑝𐑱𐑯 𐑑𐑶; 𐑯 𐑒𐑻𐑕𐑑 𐑚𐑰 𐑷𐑤 𐑞 𐑔𐑦𐑙𐑟 𐑞𐑨𐑑 𐑒𐑭𐑕𐑑 𐑥𐑨𐑯'𐑟 𐑲𐑟 𐑩𐑤𐑪𐑓𐑑 𐑑 𐑞𐑨𐑑 𐑣𐑧𐑝𐑩𐑯, 𐑣𐑵𐑟 𐑤𐑲𐑝 𐑝𐑦𐑝𐑦𐑛𐑯𐑩𐑕 𐑚𐑳𐑑 𐑕𐑒𐑹𐑗𐑩𐑟 𐑣𐑦𐑥, 𐑨𐑟 𐑞𐑰𐑟 𐑴𐑤𐑛 𐑲𐑟 𐑸 𐑰𐑝𐑩𐑯 𐑯𐑬 𐑕𐑒𐑹𐑗𐑑 𐑢𐑦𐑞 𐑞𐑲 𐑤𐑲𐑑, 𐑴 𐑕𐑳𐑯! 𐑤𐑧𐑝𐑩𐑤 𐑚𐑲 𐑯𐑱𐑗𐑼 𐑑 𐑞𐑦𐑕 𐑻𐑔'𐑕 𐑣𐑼𐑲𐑟𐑩𐑯 𐑸 𐑞 𐑜𐑤𐑭𐑯𐑕𐑩𐑟 𐑝 𐑥𐑨𐑯'𐑟 𐑲𐑟; 𐑯𐑪𐑑 𐑖𐑪𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑮𐑬𐑯 𐑝 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑨𐑟 𐑦𐑓 ·𐑜𐑪𐑛 𐑣𐑨𐑛 𐑥𐑧𐑯𐑑 𐑣𐑦𐑥 𐑑 𐑜𐑱𐑟 𐑪𐑯 𐑣𐑦𐑟 𐑓𐑻𐑥𐑩𐑥𐑩𐑯𐑑. 𐑒𐑻𐑕 𐑞𐑰, 𐑞𐑬 𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑!” 𐑛𐑨𐑖𐑦𐑙 𐑦𐑑 𐑑 𐑞 𐑛𐑧𐑒, “𐑯𐑴 𐑤𐑪𐑙𐑜𐑼 𐑢𐑦𐑤 𐑲 𐑜𐑲𐑛 𐑥𐑲 𐑻𐑔𐑤𐑦 𐑢𐑱 𐑚𐑲 𐑞𐑰; 𐑞 𐑤𐑧𐑝𐑩𐑤 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑕, 𐑯 𐑞 𐑤𐑧𐑝𐑩𐑤 𐑛𐑧𐑛-𐑮𐑧𐑒𐑩𐑯𐑦𐑙, 𐑚𐑲 𐑤𐑪𐑜 𐑯 𐑚𐑲 𐑤𐑲𐑯; 𐑞𐑰𐑟 𐑖𐑨𐑤 𐑒𐑩𐑯𐑛𐑳𐑒𐑑 𐑥𐑰, 𐑯 𐑖𐑴 𐑥𐑰 𐑥𐑲 𐑐𐑤𐑱𐑕 𐑪𐑯 𐑞 𐑕𐑰. 𐑲,” 𐑤𐑲𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑑 𐑑 𐑞 𐑛𐑧𐑒, “𐑞𐑳𐑕 𐑲 𐑑𐑮𐑨𐑥𐑐𐑩𐑤 𐑪𐑯 𐑞𐑰, 𐑞𐑬 𐑐𐑷𐑤𐑑𐑮𐑦 𐑔𐑦𐑙 𐑞𐑨𐑑 𐑓𐑰𐑚𐑤𐑦 𐑐𐑶𐑯𐑑𐑩𐑕𐑑 𐑪𐑯 𐑣𐑲; 𐑞𐑳𐑕 𐑲 𐑕𐑐𐑤𐑦𐑑 𐑯 𐑛𐑦𐑕𐑑𐑮𐑶 𐑞𐑰!”

𐑨𐑟 𐑞 𐑓𐑮𐑨𐑯𐑑𐑦𐑒 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑞𐑳𐑕 𐑕𐑐𐑴𐑒 𐑯 𐑞𐑳𐑕 𐑑𐑮𐑨𐑥𐑐𐑩𐑤𐑛 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑤𐑲𐑝 𐑯 𐑛𐑧𐑛 𐑓𐑰𐑑, 𐑩 𐑕𐑯𐑽𐑦𐑙 𐑑𐑮𐑲𐑳𐑥𐑓 𐑞𐑨𐑑 𐑕𐑰𐑥𐑛 𐑥𐑧𐑯𐑑 𐑓 ·𐑱𐑣𐑨𐑚, 𐑯 𐑩 𐑓𐑱𐑑𐑩𐑤𐑦𐑕𐑑𐑦𐑒 𐑛𐑦𐑕𐑐𐑺 𐑞𐑨𐑑 𐑕𐑰𐑥𐑛 𐑥𐑧𐑯𐑑 𐑓 𐑣𐑦𐑥𐑕𐑧𐑤𐑓—𐑞𐑰𐑟 𐑐𐑭𐑕𐑑 𐑴𐑝𐑼 𐑞 𐑥𐑿𐑑, 𐑥𐑴𐑖𐑩𐑯𐑤𐑩𐑕 ·𐑐𐑸𐑟𐑰'𐑟 𐑓𐑱𐑕. 𐑳𐑯𐑩𐑚𐑟𐑻𐑝𐑛 𐑣𐑰 𐑮𐑴𐑟 𐑯 𐑜𐑤𐑲𐑛𐑩𐑛 𐑩𐑢𐑱; 𐑢𐑲𐑤, 𐑷𐑕𐑑𐑮𐑳𐑒 𐑚𐑲 𐑞 𐑨𐑕𐑐𐑧𐑒𐑑 𐑝 𐑞𐑺 𐑒𐑩𐑥𐑭𐑯𐑛𐑼, 𐑞 𐑕𐑰𐑥𐑧𐑯 𐑒𐑤𐑳𐑕𐑑𐑼𐑛 𐑑𐑩𐑜𐑧𐑞𐑼 𐑪𐑯 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤, 𐑑𐑦𐑤 ·𐑱𐑣𐑨𐑚, 𐑑𐑮𐑳𐑚𐑩𐑤𐑩𐑛𐑤𐑦 𐑐𐑱𐑕𐑦𐑙 𐑞 𐑛𐑧𐑒, 𐑖𐑬𐑑𐑩𐑛 𐑬𐑑—“𐑑 𐑞 𐑚𐑮𐑱𐑕𐑩𐑟! 𐑳𐑐 𐑣𐑧𐑤𐑥!—𐑕𐑒𐑢𐑺 𐑦𐑯!”

𐑦𐑯 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑞 𐑘𐑸𐑛𐑟 𐑕𐑢𐑳𐑙 𐑮𐑬𐑯𐑛; 𐑯 𐑨𐑟 𐑞 𐑖𐑦𐑐 𐑣𐑭𐑓-𐑢𐑰𐑤𐑛 𐑩𐑐𐑪𐑯 𐑣𐑻 𐑣𐑰𐑤, 𐑣𐑻 𐑔𐑮𐑰 𐑓𐑻𐑥-𐑕𐑰𐑑𐑩𐑛 𐑜𐑮𐑱𐑕𐑓𐑩𐑤 𐑥𐑭𐑕𐑑𐑕 𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑐𐑶𐑟𐑛 𐑩𐑐𐑪𐑯 𐑣𐑻 𐑤𐑪𐑙, 𐑮𐑦𐑚𐑛 𐑣𐑳𐑤, 𐑕𐑰𐑥𐑛 𐑨𐑟 𐑞 𐑔𐑮𐑰 ·𐑣𐑪𐑮𐑱𐑑𐑦 𐑐𐑦𐑮𐑵𐑧𐑑𐑦𐑙 𐑪𐑯 𐑢𐑳𐑯 𐑕𐑩𐑓𐑦𐑖𐑩𐑯𐑑 𐑕𐑑𐑰𐑛.

𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑯𐑲𐑑-𐑣𐑧𐑛𐑟, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑢𐑪𐑗𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑑𐑿𐑥𐑳𐑤𐑗𐑵𐑩𐑕 𐑢𐑱, 𐑯 ·𐑱𐑣𐑨𐑚'𐑟 𐑷𐑤𐑕𐑴, 𐑨𐑟 𐑣𐑰 𐑢𐑧𐑯𐑑 𐑤𐑻𐑗𐑦𐑙 𐑩𐑤𐑪𐑙 𐑞 𐑛𐑧𐑒.

“𐑲 𐑣𐑨𐑝 𐑕𐑨𐑑 𐑚𐑦𐑓𐑹 𐑞 𐑛𐑧𐑯𐑕 𐑒𐑴𐑤 𐑓𐑲𐑼 𐑯 𐑢𐑪𐑗𐑑 𐑦𐑑 𐑷𐑤 𐑩𐑜𐑤𐑴, 𐑓𐑫𐑤 𐑝 𐑦𐑑𐑕 𐑑𐑹𐑥𐑧𐑯𐑑𐑩𐑛 𐑓𐑤𐑱𐑥𐑦𐑙 𐑤𐑲𐑓; 𐑯 𐑲 𐑣𐑨𐑝 𐑕𐑰𐑯 𐑦𐑑 𐑢𐑱𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑛𐑬𐑯, 𐑛𐑬𐑯, 𐑑 𐑛𐑳𐑥𐑩𐑕𐑑 𐑛𐑳𐑕𐑑. 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑝 𐑴𐑖𐑩𐑯𐑟! 𐑝 𐑷𐑤 𐑞𐑦𐑕 𐑓𐑲𐑼𐑦 𐑤𐑲𐑓 𐑝 𐑞𐑲𐑯, 𐑢𐑪𐑑 𐑢𐑦𐑤 𐑨𐑑 𐑤𐑧𐑙𐑔 𐑮𐑦𐑥𐑱𐑯 𐑚𐑳𐑑 𐑢𐑳𐑯 𐑤𐑦𐑑𐑩𐑤 𐑣𐑰𐑐 𐑝 𐑨𐑖𐑩𐑟!”

“𐑲,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, “𐑚𐑳𐑑 𐑕𐑰-𐑒𐑴𐑤 𐑨𐑖𐑩𐑟—𐑥𐑲𐑯𐑛 𐑘𐑰 𐑞𐑨𐑑, ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒—𐑕𐑰-𐑒𐑴𐑤, 𐑯𐑪𐑑 𐑘𐑹 𐑒𐑪𐑥𐑩𐑯 𐑗𐑸𐑒𐑴𐑤. 𐑢𐑧𐑤, 𐑢𐑧𐑤; 𐑲 𐑣𐑻𐑛 ·𐑱𐑣𐑨𐑚 𐑥𐑳𐑑𐑼, '·𐑣𐑽 𐑕𐑳𐑥 𐑢𐑳𐑯 𐑔𐑮𐑳𐑕𐑑𐑕 𐑞𐑰𐑟 𐑒𐑸𐑛𐑟 𐑦𐑯𐑑𐑵 𐑞𐑰𐑟 𐑴𐑤𐑛 𐑣𐑨𐑯𐑛𐑟 𐑝 𐑥𐑲𐑯; 𐑕𐑢𐑺𐑟 𐑞𐑨𐑑 𐑲 𐑥𐑳𐑕𐑑 𐑐𐑤𐑱 𐑞𐑧𐑥, 𐑯 𐑯𐑴 𐑳𐑞𐑼𐑟.' 𐑯 𐑛𐑨𐑥 𐑥𐑰, ·𐑱𐑣𐑨𐑚, 𐑚𐑳𐑑 𐑞𐑬 𐑨𐑒𐑑𐑩𐑕𐑑 𐑮𐑲𐑑; 𐑤𐑦𐑝 𐑦𐑯 𐑞 𐑜𐑱𐑥, 𐑯 𐑛𐑲 𐑦𐑯 𐑦𐑑!”
