·𐑗𐑨𐑐𐑑𐑼 32. ·𐑕𐑧𐑑𐑪𐑤𐑩𐑡𐑦.

𐑷𐑤𐑮𐑧𐑛𐑦 𐑢𐑰 𐑸 𐑚𐑴𐑤𐑛𐑤𐑦 𐑤𐑷𐑯𐑗𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑛𐑰𐑐; 𐑚𐑳𐑑 𐑕𐑵𐑯 𐑢𐑰 𐑖𐑨𐑤 𐑚𐑰 𐑤𐑪𐑕𐑑 𐑦𐑯 𐑦𐑑𐑕 𐑳𐑯𐑖𐑹𐑛, 𐑣𐑸𐑚𐑼𐑤𐑩𐑕 𐑦𐑥𐑧𐑯𐑕𐑦𐑑𐑦𐑟. 𐑺 𐑞𐑨𐑑 𐑒𐑳𐑥 𐑑 𐑐𐑭𐑕; 𐑺 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑢𐑰𐑛𐑦 𐑣𐑳𐑤 𐑮𐑴𐑤𐑟 𐑕𐑲𐑛 𐑚𐑲 𐑕𐑲𐑛 𐑢𐑦𐑞 𐑞 𐑚𐑸𐑯𐑩𐑒𐑩𐑤𐑛 𐑣𐑳𐑤𐑟 𐑝 𐑞 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯; 𐑨𐑑 𐑞 𐑬𐑑𐑕𐑧𐑑 𐑦𐑑 𐑦𐑟 𐑚𐑳𐑑 𐑢𐑧𐑤 𐑑 𐑩𐑑𐑧𐑯𐑛 𐑑 𐑩 𐑥𐑨𐑑𐑼 𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑯𐑛𐑦𐑕𐑐𐑧𐑯𐑕𐑩𐑚𐑩𐑤 𐑑 𐑩 𐑔𐑳𐑮𐑴 𐑩𐑐𐑮𐑰𐑖𐑩𐑑𐑦𐑝 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑝 𐑞 𐑥𐑹 𐑕𐑐𐑧𐑖𐑩𐑤 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯𐑦𐑒 𐑮𐑧𐑝𐑩𐑤𐑱𐑖𐑩𐑯𐑟 𐑯 𐑩𐑤𐑵𐑠𐑩𐑯𐑟 𐑝 𐑷𐑤 𐑕𐑹𐑑𐑕 𐑢𐑦𐑗 𐑸 𐑑 𐑓𐑪𐑤𐑴.

𐑦𐑑 𐑦𐑟 𐑕𐑳𐑥 𐑕𐑦𐑕𐑑𐑩𐑥𐑩𐑑𐑲𐑟𐑛 𐑧𐑒𐑕𐑦𐑚𐑦𐑖𐑩𐑯 𐑝 𐑞 𐑢𐑱𐑤 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛 𐑡𐑧𐑯𐑼𐑩, 𐑞𐑨𐑑 𐑲 𐑢𐑫𐑛 𐑯𐑬 𐑓𐑱𐑯 𐑐𐑫𐑑 𐑚𐑦𐑓𐑹 𐑿. 𐑘𐑧𐑑 𐑦𐑟 𐑦𐑑 𐑯𐑴 𐑰𐑟𐑦 𐑑𐑭𐑕𐑒. 𐑞 𐑒𐑤𐑨𐑕𐑦𐑓𐑦𐑒𐑱𐑖𐑩𐑯 𐑝 𐑞 𐑒𐑩𐑯𐑕𐑑𐑦𐑗𐑵𐑩𐑯𐑑𐑕 𐑝 𐑩 𐑒𐑱𐑪𐑕, 𐑯𐑳𐑔𐑦𐑙 𐑤𐑧𐑕 𐑦𐑟 𐑣𐑽 𐑧𐑕𐑱𐑛. 𐑤𐑦𐑕𐑩𐑯 𐑑 𐑢𐑪𐑑 𐑞 𐑚𐑧𐑕𐑑 𐑯 𐑤𐑱𐑑𐑩𐑕𐑑 𐑷𐑔𐑪𐑮𐑦𐑑𐑦𐑟 𐑣𐑨𐑝 𐑤𐑱𐑛 𐑛𐑬𐑯.

“𐑯𐑴 𐑚𐑮𐑭𐑯𐑗 𐑝 ·𐑟𐑴𐑪𐑤𐑩𐑡𐑦 𐑦𐑟 𐑕𐑴 𐑥𐑳𐑗 𐑦𐑯𐑝𐑪𐑤𐑝𐑛 𐑨𐑟 𐑞𐑨𐑑 𐑢𐑦𐑗 𐑦𐑟 𐑦𐑯𐑑𐑲𐑑𐑩𐑤𐑛 ·𐑕𐑧𐑑𐑪𐑤𐑩𐑡𐑦,” 𐑕𐑧𐑟 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑕𐑒𐑹𐑟𐑚𐑦, ⸰𐑱𐑛 1820.

“𐑦𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑥𐑲 𐑦𐑯𐑑𐑧𐑯𐑖𐑩𐑯, 𐑢𐑻 𐑦𐑑 𐑦𐑯 𐑥𐑲 𐑐𐑬𐑼, 𐑑 𐑧𐑯𐑑𐑼 𐑦𐑯𐑑𐑵 𐑞 𐑦𐑯𐑒𐑢𐑲𐑼𐑦 𐑨𐑟 𐑑 𐑞 𐑑𐑮𐑵 𐑥𐑧𐑔𐑩𐑛 𐑝 𐑛𐑦𐑝𐑲𐑛𐑦𐑙 𐑞 ·𐑕𐑧𐑑𐑱𐑖𐑩 𐑦𐑯𐑑𐑵 𐑜𐑮𐑵𐑐𐑕 𐑯 𐑓𐑨𐑥𐑤𐑦𐑟. * * * 𐑳𐑑𐑼 𐑒𐑩𐑯𐑓𐑿𐑠𐑩𐑯 𐑦𐑜𐑟𐑦𐑕𐑑𐑕 𐑩𐑥𐑳𐑙 𐑞 𐑣𐑦𐑕𐑑𐑹𐑾𐑯𐑟 𐑝 𐑞𐑦𐑕 𐑨𐑯𐑦𐑥𐑩𐑤” (𐑕𐑐𐑻𐑥 𐑢𐑱𐑤), 𐑕𐑧𐑟 ·𐑕𐑻𐑡𐑩𐑯 ·𐑚𐑦𐑱𐑤, ⸰𐑱𐑛 1839.

“𐑳𐑯𐑓𐑦𐑑𐑯𐑩𐑕 𐑑 𐑐𐑼𐑕𐑿 𐑬𐑼 𐑮𐑰𐑕𐑻𐑗 𐑦𐑯 𐑞 𐑳𐑯𐑓𐑨𐑞𐑩𐑥𐑩𐑚𐑩𐑤 𐑢𐑷𐑑𐑼𐑟.” “𐑦𐑥𐑐𐑧𐑯𐑦𐑑𐑮𐑩𐑚𐑩𐑤 𐑝𐑱𐑤 𐑒𐑳𐑝𐑼𐑦𐑙 𐑬𐑼 𐑯𐑪𐑤𐑦𐑡 𐑝 𐑞 ·𐑕𐑧𐑑𐑱𐑖𐑩.” “𐑩 𐑓𐑰𐑤𐑛 𐑕𐑑𐑮𐑵𐑯 𐑢𐑦𐑞 𐑔𐑹𐑯𐑟.” “𐑷𐑤 𐑞𐑰𐑟 𐑦𐑯𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑦𐑯𐑛𐑦𐑒𐑱𐑖𐑩𐑯𐑟 𐑚𐑳𐑑 𐑕𐑻𐑝 𐑑 𐑑𐑹𐑗𐑼 𐑳𐑕 𐑯𐑨𐑗𐑼𐑩𐑤𐑦𐑕𐑑𐑕.”

𐑞𐑳𐑕 𐑕𐑐𐑰𐑒 𐑝 𐑞 𐑢𐑱𐑤, 𐑞 𐑜𐑮𐑱𐑑 ·𐑒𐑿𐑝𐑽, 𐑯 ·𐑡𐑪𐑯 ·𐑣𐑳𐑯𐑑𐑼, 𐑯 ·𐑤𐑧𐑕𐑩𐑯, 𐑞𐑴𐑟 𐑤𐑲𐑑𐑕 𐑝 𐑟𐑴𐑪𐑤𐑩𐑡𐑦 𐑯 𐑩𐑯𐑨𐑑𐑩𐑥𐑦. 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕, 𐑞𐑴 𐑝 𐑮𐑾𐑤 𐑯𐑪𐑤𐑦𐑡 𐑞𐑺 𐑚𐑰 𐑤𐑦𐑑𐑩𐑤, 𐑘𐑧𐑑 𐑝 𐑚𐑫𐑒𐑕 𐑞𐑺 𐑸 𐑩 𐑐𐑤𐑧𐑯𐑑𐑦; 𐑯 𐑕𐑴 𐑦𐑯 𐑕𐑳𐑥 𐑕𐑥𐑷𐑤 𐑛𐑦𐑜𐑮𐑰, 𐑢𐑦𐑞 ·𐑕𐑧𐑑𐑪𐑤𐑩𐑡𐑦, 𐑹 𐑞 𐑕𐑲𐑩𐑯𐑕 𐑝 𐑢𐑱𐑤𐑟. 𐑥𐑧𐑯𐑦 𐑸 𐑞 𐑥𐑧𐑯, 𐑕𐑥𐑷𐑤 𐑯 𐑜𐑮𐑱𐑑, 𐑴𐑤𐑛 𐑯 𐑯𐑿, 𐑤𐑨𐑯𐑛𐑟𐑥𐑧𐑯 𐑯 𐑕𐑰𐑥𐑧𐑯, 𐑣𐑵 𐑣𐑨𐑝 𐑨𐑑 𐑤𐑸𐑡 𐑹 𐑦𐑯 𐑤𐑦𐑑𐑩𐑤, 𐑮𐑦𐑑𐑩𐑯 𐑝 𐑞 𐑢𐑱𐑤. 𐑮𐑳𐑯 𐑴𐑝𐑼 𐑩 𐑓𐑿:—𐑞 ·𐑷𐑔𐑼𐑟 𐑝 𐑞 ·𐑚𐑲𐑚𐑩𐑤; ·𐑨𐑮𐑦𐑕𐑑𐑪𐑑𐑩𐑤; ·𐑐𐑤𐑲𐑯𐑦; ·𐑷𐑤𐑛𐑮𐑴𐑝𐑨𐑯𐑛𐑦; ·𐑕𐑻 ·𐑑𐑪𐑥𐑩𐑕 ·𐑚𐑮𐑬𐑯; ·𐑡𐑦𐑟𐑯𐑼; ·𐑮𐑱; ·𐑤𐑦𐑯𐑾𐑕; ·𐑮𐑪𐑯𐑛𐑩𐑤𐑩𐑖𐑩𐑕; ·𐑢𐑦𐑤𐑴𐑚𐑦; ·𐑜𐑮𐑰𐑯; ·𐑸𐑑𐑩𐑛𐑦; ·𐑕𐑦𐑚𐑨𐑤𐑑; ·𐑚𐑮𐑦𐑕𐑩𐑯; ·𐑥𐑸𐑑𐑦𐑯; ·𐑤𐑱𐑕𐑐𐑰𐑛; Bonneterre; ·𐑛𐑧𐑥𐑸𐑱; ·𐑚𐑨𐑮𐑩𐑯 ·𐑒𐑿𐑝𐑽; ·𐑓𐑮𐑧𐑛𐑼𐑦𐑒 ·𐑒𐑿𐑝𐑽; ·𐑡𐑪𐑯 ·𐑣𐑳𐑯𐑑𐑼; ·𐑴𐑢𐑧𐑯; ·𐑕𐑒𐑹𐑟𐑚𐑦; ·𐑚𐑦𐑱𐑤; ·𐑚𐑧𐑯𐑩𐑑; J. ·𐑮𐑪𐑕 ·𐑚𐑮𐑬𐑯; 𐑞 ·𐑷𐑔𐑼 𐑝 ·𐑥𐑦𐑮𐑾𐑥 ·𐑒𐑪𐑓𐑦𐑯; ·𐑴𐑤𐑥𐑕𐑑𐑧𐑛; 𐑯 𐑞 ·𐑮𐑧𐑝 T. ·𐑗𐑰𐑝𐑼. 𐑚𐑳𐑑 𐑑 𐑢𐑪𐑑 𐑳𐑤𐑑𐑦𐑥𐑩𐑑 𐑡𐑧𐑯𐑼𐑩𐑤𐑲𐑟𐑦𐑙 𐑐𐑻𐑐𐑩𐑕 𐑷𐑤 𐑞𐑰𐑟 𐑣𐑨𐑝 𐑮𐑦𐑑𐑩𐑯, 𐑞 𐑩𐑚𐑳𐑝 𐑕𐑲𐑑𐑩𐑛 𐑧𐑒𐑕𐑑𐑮𐑨𐑒𐑑𐑕 𐑢𐑦𐑤 𐑖𐑴.

𐑓𐑻𐑕𐑑: 𐑞 𐑳𐑯𐑕𐑻𐑑𐑩𐑯, 𐑳𐑯𐑕𐑧𐑑𐑩𐑤𐑛 𐑒𐑩𐑯𐑛𐑦𐑖𐑩𐑯 𐑝 𐑞𐑦𐑕 𐑕𐑲𐑩𐑯𐑕 𐑝 ·𐑕𐑧𐑑𐑪𐑤𐑩𐑡𐑦 𐑦𐑟 𐑦𐑯 𐑞 𐑝𐑧𐑮𐑦 𐑝𐑧𐑕𐑑𐑦𐑚𐑿𐑤 𐑩𐑑𐑧𐑕𐑑𐑩𐑛 𐑚𐑲 𐑞 𐑓𐑨𐑒𐑑, 𐑞𐑨𐑑 𐑦𐑯 𐑕𐑳𐑥 𐑒𐑢𐑹𐑑𐑼𐑟 𐑦𐑑 𐑕𐑑𐑦𐑤 𐑮𐑦𐑥𐑱𐑯𐑟 𐑩 𐑥𐑵𐑑 𐑐𐑶𐑯𐑑 𐑢𐑧𐑞𐑼 𐑩 𐑢𐑱𐑤 𐑚𐑰 𐑩 𐑓𐑦𐑖. 𐑦𐑯 𐑣𐑦𐑟 ·𐑕𐑦𐑕𐑑𐑩𐑥 𐑝 ·𐑯𐑱𐑗𐑼, ⸰𐑱𐑛 1776, ·𐑤𐑦𐑯𐑾𐑕 𐑛𐑦𐑒𐑤𐑺𐑟, “𐑲 𐑣𐑽𐑚𐑲 𐑕𐑧𐑐𐑼𐑩𐑑 𐑞 𐑢𐑱𐑤𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑦𐑖.” 𐑚𐑳𐑑 𐑝 𐑥𐑲 𐑴𐑯 𐑯𐑪𐑤𐑦𐑡, 𐑲 𐑯𐑴 𐑞𐑨𐑑 𐑛𐑬𐑯 𐑑 𐑞 𐑘𐑽 1850, 𐑖𐑸𐑒𐑕 𐑯 𐑖𐑨𐑛, ·𐑱𐑤𐑵𐑦𐑝𐑟 𐑯 𐑣𐑧𐑮𐑦𐑙, 𐑩𐑜𐑱𐑯𐑕𐑑 ·𐑤𐑦𐑯𐑾𐑕'𐑩𐑟 𐑦𐑒𐑕𐑐𐑮𐑧𐑕 𐑰𐑛𐑦𐑒𐑑, 𐑢𐑻 𐑕𐑑𐑦𐑤 𐑓𐑬𐑯𐑛 𐑛𐑦𐑝𐑲𐑛𐑦𐑙 𐑞 𐑐𐑩𐑟𐑧𐑖𐑩𐑯 𐑝 𐑞 𐑕𐑱𐑥 𐑕𐑰𐑟 𐑢𐑦𐑞 𐑞 ·𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯.

𐑚𐑰 𐑦𐑑 𐑯𐑴𐑯 𐑞𐑨𐑑, 𐑢𐑱𐑝𐑦𐑙 𐑷𐑤 𐑸𐑜𐑿𐑥𐑩𐑯𐑑, 𐑲 𐑑𐑱𐑒 𐑞 𐑜𐑫𐑛 𐑴𐑤𐑛 𐑓𐑨𐑖𐑩𐑯𐑛 𐑜𐑮𐑬𐑯𐑛 𐑞𐑨𐑑 𐑞 𐑢𐑱𐑤 𐑦𐑟 𐑩 𐑓𐑦𐑖, 𐑯 𐑒𐑷𐑤 𐑩𐑐𐑪𐑯 𐑣𐑴𐑤𐑦 ·𐑡𐑴𐑯𐑩 𐑑 𐑚𐑨𐑒 𐑥𐑰. 𐑞𐑦𐑕 𐑓𐑳𐑯𐑛𐑩𐑥𐑧𐑯𐑑𐑩𐑤 𐑔𐑦𐑙 𐑕𐑧𐑑𐑩𐑤𐑛, 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑐𐑶𐑯𐑑 𐑦𐑟, 𐑦𐑯 𐑢𐑪𐑑 𐑦𐑯𐑑𐑻𐑯𐑩𐑤 𐑮𐑦𐑕𐑐𐑧𐑒𐑑 𐑛𐑳𐑟 𐑞 𐑢𐑱𐑤 𐑛𐑦𐑓𐑼 𐑓𐑮𐑪𐑥 𐑳𐑞𐑼 𐑓𐑦𐑖. 𐑩𐑚𐑳𐑝, ·𐑤𐑦𐑯𐑾𐑕 𐑣𐑨𐑟 𐑜𐑦𐑝𐑩𐑯 𐑿 𐑞𐑴𐑟 𐑲𐑑𐑩𐑥𐑟. 𐑚𐑳𐑑 𐑦𐑯 𐑚𐑮𐑰𐑓, 𐑞𐑱 𐑸 𐑞𐑰𐑟: 𐑤𐑳𐑙𐑟 𐑯 𐑢𐑹𐑥 𐑚𐑤𐑳𐑛; 𐑢𐑺𐑨𐑟, 𐑷𐑤 𐑳𐑞𐑼 𐑓𐑦𐑖 𐑸 𐑤𐑳𐑙𐑤𐑩𐑕 𐑯 𐑒𐑴𐑤𐑛 𐑚𐑤𐑳𐑛𐑩𐑛.

𐑯𐑧𐑒𐑕𐑑: 𐑣𐑬 𐑖𐑨𐑤 𐑢𐑰 𐑛𐑦𐑓𐑲𐑯 𐑞 𐑢𐑱𐑤, 𐑚𐑲 𐑣𐑦𐑟 𐑪𐑚𐑝𐑾𐑕 𐑧𐑒𐑕𐑑𐑼𐑯𐑩𐑤𐑟, 𐑕𐑴 𐑨𐑟 𐑒𐑩𐑯𐑕𐑐𐑦𐑒𐑿𐑩𐑕𐑤𐑦 𐑑 𐑤𐑱𐑚𐑩𐑤 𐑣𐑦𐑥 𐑓 𐑷𐑤 𐑑𐑲𐑥 𐑑 𐑒𐑳𐑥? 𐑑 𐑚𐑰 𐑖𐑹𐑑, 𐑞𐑧𐑯, 𐑩 𐑢𐑱𐑤 𐑦𐑟 𐑩 𐑕𐑐𐑬𐑑𐑦𐑙 𐑓𐑦𐑖 𐑢𐑦𐑞 𐑩 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤 𐑑𐑱𐑤. 𐑞𐑺 𐑿 𐑣𐑨𐑝 𐑣𐑦𐑥. 𐑣𐑬𐑧𐑝𐑼 𐑒𐑩𐑯𐑑𐑮𐑨𐑒𐑑𐑩𐑛, 𐑞𐑨𐑑 𐑛𐑧𐑓𐑦𐑯𐑦𐑖𐑩𐑯 𐑦𐑟 𐑞 𐑮𐑦𐑟𐑳𐑤𐑑 𐑝 𐑦𐑒𐑕𐑐𐑨𐑯𐑛𐑩𐑛 𐑥𐑧𐑛𐑦𐑑𐑱𐑖𐑩𐑯. 𐑩 𐑢𐑷𐑤𐑮𐑩𐑕 𐑕𐑐𐑬𐑑𐑕 𐑥𐑳𐑗 𐑤𐑲𐑒 𐑩 𐑢𐑱𐑤, 𐑚𐑳𐑑 𐑞 𐑢𐑷𐑤𐑮𐑩𐑕 𐑦𐑟 𐑯𐑪𐑑 𐑩 𐑓𐑦𐑖, 𐑚𐑦𐑒𐑪𐑟 𐑣𐑰 𐑦𐑟 𐑨𐑥𐑓𐑦𐑚𐑾𐑕. 𐑚𐑳𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑑𐑻𐑥 𐑝 𐑞 𐑛𐑧𐑓𐑦𐑯𐑦𐑖𐑩𐑯 𐑦𐑟 𐑕𐑑𐑦𐑤 𐑥𐑹 𐑒𐑴𐑡𐑩𐑯𐑑, 𐑨𐑟 𐑒𐑳𐑐𐑩𐑤𐑛 𐑢𐑦𐑞 𐑞 𐑓𐑻𐑕𐑑. 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑯𐑦 𐑢𐑳𐑯 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑯𐑴𐑑𐑦𐑕𐑑 𐑞𐑨𐑑 𐑷𐑤 𐑞 𐑓𐑦𐑖 𐑓𐑩𐑥𐑦𐑤𐑽 𐑑 𐑤𐑨𐑯𐑛𐑟𐑥𐑧𐑯 𐑣𐑨𐑝 𐑯𐑪𐑑 𐑩 𐑓𐑤𐑨𐑑, 𐑚𐑳𐑑 𐑩 𐑝𐑻𐑑𐑦𐑒𐑩𐑤, 𐑹 𐑳𐑐-𐑯-𐑛𐑬𐑯 𐑑𐑱𐑤. 𐑢𐑺𐑨𐑟, 𐑩𐑥𐑳𐑙 𐑕𐑐𐑬𐑑𐑦𐑙 𐑓𐑦𐑖 𐑞 𐑑𐑱𐑤, 𐑞𐑴 𐑦𐑑 𐑥𐑱 𐑚𐑰 𐑕𐑦𐑥𐑦𐑤𐑼𐑤𐑦 𐑖𐑱𐑐𐑑, 𐑦𐑯𐑝𐑺𐑾𐑚𐑤𐑦 𐑩𐑕𐑿𐑥𐑟 𐑩 𐑣𐑪𐑮𐑦𐑟𐑪𐑯𐑑𐑩𐑤 𐑐𐑩𐑟𐑦𐑖𐑩𐑯.

·𐑚𐑫𐑒 𐑲. (𐑓𐑴𐑤𐑦𐑴), ·𐑗𐑨𐑐𐑑𐑼 VI. (𐑣𐑩𐑟𐑭 ·𐑐𐑹𐑐𐑩𐑕).—𐑞𐑦𐑕 𐑦𐑟 𐑞 𐑒𐑪𐑥𐑩𐑯 𐑐𐑹𐑐𐑩𐑕 𐑓𐑬𐑯𐑛 𐑷𐑤𐑥𐑴𐑕𐑑 𐑷𐑤 𐑴𐑝𐑼 𐑞 𐑜𐑤𐑴𐑚. 𐑞 𐑯𐑱𐑥 𐑦𐑟 𐑝 𐑥𐑲 𐑴𐑯 𐑚𐑦𐑕𐑑𐑴𐑩𐑤; 𐑓 𐑞𐑺 𐑸 𐑥𐑹 𐑞𐑨𐑯 𐑢𐑳𐑯 𐑕𐑹𐑑 𐑝 𐑐𐑹𐑐𐑩𐑕𐑩𐑟, 𐑯 𐑕𐑳𐑥𐑔𐑦𐑙 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑛𐑳𐑯 𐑑 𐑛𐑦𐑕𐑑𐑦𐑙𐑜𐑢𐑦𐑖 𐑞𐑧𐑥. 𐑲 𐑒𐑷𐑤 𐑣𐑦𐑥 𐑞𐑳𐑕, 𐑚𐑦𐑒𐑪𐑟 𐑣𐑰 𐑷𐑤𐑢𐑱𐑟 𐑕𐑢𐑦𐑥𐑟 𐑦𐑯 𐑣𐑦𐑤𐑺𐑾𐑕 𐑖𐑴𐑤𐑟, 𐑢𐑦𐑗 𐑩𐑐𐑪𐑯 𐑞 𐑚𐑮𐑷𐑛 𐑕𐑰 𐑒𐑰𐑐 𐑑𐑪𐑕𐑦𐑙 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑑 𐑣𐑧𐑝𐑩𐑯 𐑤𐑲𐑒 𐑒𐑨𐑐𐑕 𐑦𐑯 𐑩 ·𐑓𐑹𐑔-𐑝-𐑡𐑩𐑤𐑲 𐑒𐑮𐑬𐑛. 𐑞𐑺 𐑩𐑐𐑽𐑩𐑯𐑕 𐑦𐑟 𐑡𐑧𐑯𐑼𐑩𐑤𐑦 𐑣𐑱𐑤𐑛 𐑢𐑦𐑞 𐑛𐑦𐑤𐑲𐑑 𐑚𐑲 𐑞 𐑥𐑨𐑮𐑦𐑯𐑼. 𐑓𐑫𐑤 𐑝 𐑓𐑲𐑯 𐑕𐑐𐑦𐑮𐑦𐑑𐑕, 𐑞𐑱 𐑦𐑯𐑝𐑺𐑾𐑚𐑤𐑦 𐑒𐑳𐑥 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑮𐑰𐑟𐑦 𐑚𐑦𐑤𐑴𐑟 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛. 𐑞𐑱 𐑸 𐑞 𐑤𐑨𐑛𐑟 𐑞𐑨𐑑 𐑷𐑤𐑢𐑱𐑟 𐑤𐑦𐑝 𐑚𐑦𐑓𐑹 𐑞 𐑢𐑦𐑯𐑛. 𐑞𐑱 𐑸 𐑩𐑒𐑬𐑯𐑑𐑩𐑛 𐑩 𐑤𐑳𐑒𐑦 𐑴𐑥𐑩𐑯. 𐑦𐑓 𐑿 𐑘𐑹𐑕𐑧𐑤𐑓 𐑒𐑨𐑯 𐑢𐑦𐑞𐑕𐑑𐑨𐑯𐑛 𐑔𐑮𐑰 𐑗𐑽𐑟 𐑨𐑑 𐑚𐑦𐑣𐑴𐑤𐑛𐑦𐑙 𐑞𐑰𐑟 𐑝𐑦𐑝𐑱𐑖𐑩𐑕 𐑓𐑦𐑖, 𐑞𐑧𐑯 𐑣𐑧𐑝𐑩𐑯 𐑣𐑧𐑤𐑐 𐑘𐑰; 𐑞 𐑕𐑐𐑦𐑮𐑦𐑑 𐑝 𐑜𐑪𐑛𐑤𐑦 𐑜𐑱𐑥𐑕𐑩𐑥𐑯𐑩𐑕 𐑦𐑟 𐑯𐑪𐑑 𐑦𐑯 𐑘𐑰. 𐑩 𐑢𐑧𐑤-𐑓𐑧𐑛, 𐑐𐑤𐑳𐑥𐑐 ·𐑣𐑩𐑟𐑭 ·𐑐𐑹𐑐𐑩𐑕 𐑢𐑦𐑤 𐑘𐑰𐑤𐑛 𐑿 𐑢𐑳𐑯 𐑜𐑫𐑛 𐑜𐑨𐑤𐑩𐑯 𐑝 𐑜𐑫𐑛 𐑶𐑤. 𐑚𐑳𐑑 𐑞 𐑓𐑲𐑯 𐑯 𐑛𐑧𐑤𐑦𐑒𐑩𐑑 𐑓𐑤𐑵𐑦𐑛 𐑧𐑒𐑕𐑑𐑮𐑨𐑒𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑡𐑷𐑟 𐑦𐑟 𐑦𐑒𐑕𐑰𐑛𐑦𐑙𐑤𐑦 𐑝𐑨𐑤𐑿𐑩𐑚𐑩𐑤. 𐑦𐑑 𐑦𐑟 𐑦𐑯 𐑮𐑦𐑒𐑢𐑧𐑕𐑑 𐑩𐑥𐑳𐑙 𐑡𐑵𐑩𐑤𐑼𐑟 𐑯 𐑢𐑪𐑗𐑥𐑱𐑒𐑼𐑟. 𐑕𐑱𐑤𐑼𐑟 𐑐𐑫𐑑 𐑦𐑑 𐑪𐑯 𐑞𐑺 𐑣𐑴𐑯𐑟. 𐑐𐑹𐑐𐑩𐑕 𐑥𐑰𐑑 𐑦𐑟 𐑜𐑫𐑛 𐑰𐑑𐑦𐑙, 𐑿 𐑯𐑴. 𐑦𐑑 𐑥𐑱 𐑯𐑧𐑝𐑼 𐑣𐑨𐑝 𐑩𐑒𐑻𐑛 𐑑 𐑿 𐑞𐑨𐑑 𐑩 𐑐𐑹𐑐𐑩𐑕 𐑕𐑐𐑬𐑑𐑕. 𐑦𐑯𐑛𐑰𐑛, 𐑣𐑦𐑟 𐑕𐑐𐑬𐑑 𐑦𐑟 𐑕𐑴 𐑕𐑥𐑷𐑤 𐑞𐑨𐑑 𐑦𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑝𐑧𐑮𐑦 𐑮𐑧𐑛𐑦𐑤𐑦 𐑛𐑦𐑕𐑻𐑯𐑩𐑚𐑩𐑤. 𐑚𐑳𐑑 𐑞 𐑯𐑧𐑒𐑕𐑑 𐑑𐑲𐑥 𐑿 𐑣𐑨𐑝 𐑩 𐑗𐑭𐑯𐑕, 𐑢𐑪𐑗 𐑣𐑦𐑥; 𐑯 𐑿 𐑢𐑦𐑤 𐑞𐑧𐑯 𐑕𐑰 𐑞 𐑜𐑮𐑱𐑑 ·𐑕𐑐𐑻𐑥 𐑢𐑱𐑤 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑦𐑯 𐑥𐑦𐑯𐑦𐑗𐑼.
