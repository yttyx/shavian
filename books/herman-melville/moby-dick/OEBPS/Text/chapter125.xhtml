·𐑗𐑨𐑐𐑑𐑼 125. 𐑞 ·𐑤𐑪𐑜 𐑯 ·𐑤𐑲𐑯.

𐑢𐑲𐑤 𐑯𐑬 𐑞 𐑓𐑱𐑑𐑩𐑛 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑴 𐑤𐑪𐑙 𐑩𐑓𐑤𐑴𐑑 𐑞𐑦𐑕 𐑝𐑶𐑦𐑡, 𐑞 𐑤𐑪𐑜 𐑯 𐑤𐑲𐑯 𐑣𐑨𐑛 𐑚𐑳𐑑 𐑝𐑧𐑮𐑦 𐑕𐑧𐑤𐑛𐑩𐑥 𐑚𐑰𐑯 𐑦𐑯 𐑿𐑕. 𐑴𐑦𐑙 𐑑 𐑩 𐑒𐑪𐑯𐑓𐑦𐑛𐑩𐑯𐑑 𐑮𐑦𐑤𐑲𐑩𐑯𐑕 𐑩𐑐𐑪𐑯 𐑳𐑞𐑼 𐑥𐑰𐑯𐑟 𐑝 𐑛𐑦𐑑𐑻𐑥𐑦𐑯𐑦𐑙 𐑞 𐑝𐑧𐑕𐑩𐑤'𐑟 𐑐𐑤𐑱𐑕, 𐑕𐑳𐑥 𐑥𐑻𐑗𐑩𐑯𐑑𐑥𐑧𐑯, 𐑯 𐑥𐑧𐑯𐑦 𐑢𐑱𐑤𐑥𐑧𐑯, 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑢𐑧𐑯 𐑒𐑮𐑵𐑟𐑦𐑙, 𐑣𐑴𐑤𐑦 𐑯𐑦𐑜𐑤𐑧𐑒𐑑 𐑑 𐑣𐑰𐑝 𐑞 𐑤𐑪𐑜; 𐑞𐑴 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥, 𐑯 𐑓𐑮𐑰𐑒𐑢𐑩𐑯𐑑𐑤𐑦 𐑥𐑹 𐑓 𐑓𐑹𐑥'𐑟 𐑕𐑱𐑒 𐑞𐑨𐑯 𐑧𐑯𐑦𐑔𐑦𐑙 𐑧𐑤𐑕, 𐑮𐑧𐑜𐑿𐑤𐑼𐑤𐑦 𐑐𐑫𐑑𐑦𐑙 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑞 𐑒𐑳𐑕𐑑𐑩𐑥𐑼𐑦 𐑕𐑤𐑱𐑑 𐑞 𐑒𐑹𐑕 𐑕𐑑𐑽𐑛 𐑚𐑲 𐑞 𐑖𐑦𐑐, 𐑨𐑟 𐑢𐑧𐑤 𐑨𐑟 𐑞 𐑐𐑮𐑦𐑟𐑿𐑥𐑛 𐑨𐑝𐑼𐑦𐑡 𐑮𐑱𐑑 𐑝 𐑐𐑮𐑴𐑜𐑮𐑧𐑖𐑩𐑯 𐑧𐑝𐑮𐑦 𐑬𐑼. 𐑦𐑑 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑞𐑳𐑕 𐑢𐑦𐑞 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛. 𐑞 𐑢𐑫𐑛𐑩𐑯 𐑮𐑰𐑤 𐑯 𐑨𐑙𐑜𐑿𐑤𐑼 𐑤𐑪𐑜 𐑩𐑑𐑨𐑗𐑑 𐑣𐑳𐑙, 𐑤𐑪𐑙 𐑳𐑯𐑑𐑳𐑗𐑑, 𐑡𐑳𐑕𐑑 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑮𐑱𐑤𐑦𐑙 𐑝 𐑞 𐑭𐑓𐑑𐑼 𐑚𐑫𐑤𐑢𐑼𐑒𐑕. 𐑮𐑱𐑯𐑟 𐑯 𐑕𐑐𐑮𐑱 𐑣𐑨𐑛 𐑛𐑨𐑥𐑐𐑑 𐑦𐑑; 𐑕𐑳𐑯 𐑯 𐑢𐑦𐑯𐑛 𐑣𐑨𐑛 𐑢𐑹𐑐𐑑 𐑦𐑑; 𐑷𐑤 𐑞 𐑧𐑤𐑦𐑥𐑩𐑯𐑑𐑕 𐑣𐑨𐑛 𐑒𐑩𐑥𐑚𐑲𐑯𐑛 𐑑 𐑮𐑪𐑑 𐑩 𐑔𐑦𐑙 𐑞𐑨𐑑 𐑣𐑳𐑙 𐑕𐑴 𐑲𐑛𐑤𐑦. 𐑚𐑳𐑑 𐑣𐑰𐑛𐑤𐑩𐑕 𐑝 𐑷𐑤 𐑞𐑦𐑕, 𐑣𐑦𐑟 𐑥𐑵𐑛 𐑕𐑰𐑟𐑛 ·𐑱𐑣𐑨𐑚, 𐑨𐑟 𐑣𐑰 𐑣𐑨𐑐𐑩𐑯𐑛 𐑑 𐑜𐑤𐑭𐑯𐑕 𐑩𐑐𐑪𐑯 𐑞 𐑮𐑰𐑤, 𐑯𐑪𐑑 𐑥𐑧𐑯𐑦 𐑬𐑼𐑟 𐑭𐑓𐑑𐑼 𐑞 𐑥𐑨𐑜𐑯𐑩𐑑 𐑕𐑰𐑯, 𐑯 𐑣𐑰 𐑮𐑦𐑥𐑧𐑥𐑚𐑼𐑛 𐑣𐑬 𐑣𐑦𐑟 𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑 𐑢𐑪𐑟 𐑯𐑴 𐑥𐑹, 𐑯 𐑮𐑦𐑒𐑷𐑤𐑛 𐑣𐑦𐑟 𐑓𐑮𐑨𐑯𐑑𐑦𐑒 𐑴𐑔 𐑩𐑚𐑬𐑑 𐑞 𐑤𐑧𐑝𐑩𐑤 𐑤𐑪𐑜 𐑯 𐑤𐑲𐑯. 𐑞 𐑖𐑦𐑐 𐑢𐑪𐑟 𐑕𐑱𐑤𐑦𐑙 𐑐𐑤𐑳𐑯𐑡𐑦𐑙𐑤𐑦; 𐑩𐑕𐑑𐑻𐑯 𐑞 𐑚𐑦𐑤𐑴𐑟 𐑮𐑴𐑤𐑛 𐑦𐑯 𐑮𐑲𐑩𐑑𐑕.

“𐑓𐑹𐑢𐑼𐑛, 𐑞𐑺! 𐑣𐑰𐑝 𐑞 𐑤𐑪𐑜!”

𐑑𐑵 𐑕𐑰𐑥𐑧𐑯 𐑒𐑱𐑥. 𐑞 𐑜𐑴𐑤𐑛𐑩𐑯-𐑣𐑿𐑛 ·𐑑𐑭𐑣𐑰𐑖𐑩𐑯 𐑯 𐑞 𐑜𐑮𐑦𐑟𐑤𐑦 ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯. “𐑑𐑱𐑒 𐑞 𐑮𐑰𐑤, 𐑢𐑳𐑯 𐑝 𐑘𐑰, 𐑲'𐑤 𐑣𐑰𐑝.”

𐑞𐑱 𐑢𐑧𐑯𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑧𐑒𐑕𐑑𐑮𐑰𐑥 𐑕𐑑𐑻𐑯, 𐑪𐑯 𐑞 𐑖𐑦𐑐'𐑕 𐑤𐑰 𐑕𐑲𐑛, 𐑢𐑺 𐑞 𐑛𐑧𐑒, 𐑢𐑦𐑞 𐑞 𐑴𐑚𐑤𐑰𐑒 𐑧𐑯𐑼𐑡𐑦 𐑝 𐑞 𐑢𐑦𐑯𐑛, 𐑢𐑪𐑟 𐑯𐑬 𐑷𐑤𐑥𐑴𐑕𐑑 𐑛𐑦𐑐𐑦𐑙 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑮𐑰𐑥𐑦, 𐑕𐑲𐑛𐑤𐑪𐑙-𐑮𐑳𐑖𐑦𐑙 𐑕𐑰.

𐑞 ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯 𐑑𐑫𐑒 𐑞 𐑮𐑰𐑤, 𐑯 𐑣𐑴𐑤𐑛𐑦𐑙 𐑦𐑑 𐑣𐑲 𐑳𐑐, 𐑚𐑲 𐑞 𐑐𐑮𐑩𐑡𐑧𐑒𐑑𐑦𐑙 𐑣𐑨𐑯𐑛𐑩𐑤-𐑧𐑯𐑛𐑟 𐑝 𐑞 𐑕𐑐𐑦𐑯𐑛𐑩𐑤, 𐑮𐑬𐑯𐑛 𐑢𐑦𐑗 𐑞 𐑕𐑐𐑵𐑤 𐑝 𐑤𐑲𐑯 𐑮𐑦𐑝𐑪𐑤𐑝𐑛, 𐑕𐑴 𐑕𐑑𐑫𐑛 𐑢𐑦𐑞 𐑞 𐑨𐑙𐑜𐑿𐑤𐑼 𐑤𐑪𐑜 𐑣𐑨𐑙𐑦𐑙 𐑛𐑬𐑯𐑢𐑼𐑛𐑟, 𐑑𐑦𐑤 ·𐑱𐑣𐑨𐑚 𐑩𐑛𐑝𐑭𐑯𐑕𐑑 𐑑 𐑣𐑦𐑥.

𐑱𐑣𐑨𐑚 𐑕𐑑𐑫𐑛 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥, 𐑯 𐑢𐑪𐑟 𐑤𐑲𐑑𐑤𐑦 𐑳𐑯𐑢𐑲𐑯𐑛𐑦𐑙 𐑕𐑳𐑥 𐑔𐑻𐑑𐑦 𐑹 𐑓𐑹𐑑𐑦 𐑑𐑻𐑯𐑟 𐑑 𐑓𐑹𐑥 𐑩 𐑐𐑮𐑦𐑤𐑦𐑥𐑦𐑯𐑼𐑦 𐑣𐑨𐑯𐑛-𐑒𐑶𐑤 𐑑 𐑑𐑪𐑕 𐑴𐑝𐑼𐑚𐑹𐑛, 𐑢𐑧𐑯 𐑞 𐑴𐑤𐑛 ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯, 𐑣𐑵 𐑢𐑪𐑟 𐑦𐑯𐑑𐑧𐑯𐑑𐑤𐑦 𐑲𐑦𐑙 𐑚𐑴𐑔 𐑣𐑦𐑥 𐑯 𐑞 𐑤𐑲𐑯, 𐑥𐑱𐑛 𐑚𐑴𐑤𐑛 𐑑 𐑕𐑐𐑰𐑒.

“𐑕𐑻, 𐑲 𐑥𐑦𐑕𐑑𐑮𐑳𐑕𐑑 𐑦𐑑; 𐑞𐑦𐑕 𐑤𐑲𐑯 𐑤𐑫𐑒𐑕 𐑓𐑸 𐑜𐑪𐑯, 𐑤𐑪𐑙 𐑣𐑰𐑑 𐑯 𐑢𐑧𐑑 𐑣𐑨𐑝 𐑕𐑐𐑶𐑤𐑛 𐑦𐑑.”

“'𐑑𐑢𐑦𐑤 𐑣𐑴𐑤𐑛, 𐑴𐑤𐑛 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯. 𐑤𐑪𐑙 𐑣𐑰𐑑 𐑯 𐑢𐑧𐑑, 𐑣𐑨𐑝 𐑞𐑱 𐑕𐑐𐑶𐑤𐑛 𐑞𐑰? 𐑞𐑬 𐑕𐑰𐑥'𐑕𐑑 𐑑 𐑣𐑴𐑤𐑛. 𐑹, 𐑑𐑮𐑵𐑼 𐑐𐑼𐑣𐑨𐑐𐑕, 𐑤𐑲𐑓 𐑣𐑴𐑤𐑛𐑟 𐑞𐑰; 𐑯𐑪𐑑 𐑞𐑬 𐑦𐑑.”

“𐑲 𐑣𐑴𐑤𐑛 𐑞 𐑕𐑐𐑵𐑤, 𐑕𐑻. 𐑚𐑳𐑑 𐑡𐑳𐑕𐑑 𐑨𐑟 𐑥𐑲 𐑒𐑨𐑐𐑑𐑦𐑯 𐑕𐑧𐑟. 𐑢𐑦𐑞 𐑞𐑰𐑟 𐑜𐑮𐑱 𐑣𐑺𐑟 𐑝 𐑥𐑲𐑯 '𐑑𐑦𐑟 𐑯𐑪𐑑 𐑢𐑻𐑔 𐑢𐑲𐑤 𐑛𐑦𐑕𐑐𐑿𐑑𐑦𐑙, '𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑢𐑦𐑞 𐑩 𐑕𐑵𐑐𐑽𐑽, 𐑣𐑵'𐑤 𐑯𐑺 𐑒𐑩𐑯𐑓𐑧𐑕.”

“𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑? 𐑞𐑺 𐑯𐑬'𐑟 𐑩 𐑐𐑨𐑗𐑑 𐑐𐑮𐑩𐑓𐑧𐑕𐑼 𐑦𐑯 ·𐑒𐑢𐑰𐑯 ·𐑯𐑱𐑗𐑼'𐑟 𐑜𐑮𐑨𐑯𐑦𐑑-𐑓𐑬𐑯𐑛𐑩𐑛 ·𐑒𐑪𐑤𐑦𐑡; 𐑚𐑳𐑑 𐑥𐑦𐑔𐑦𐑙𐑒𐑕 𐑣𐑰'𐑟 𐑑𐑵 𐑕𐑩𐑚𐑕𐑻𐑝𐑾𐑯𐑑. 𐑢𐑺 ·𐑝𐑧𐑮𐑑 𐑞𐑬 𐑚𐑹𐑯?”

“𐑦𐑯 𐑞 𐑤𐑦𐑑𐑩𐑤 𐑮𐑪𐑒𐑦 ·𐑲𐑤 𐑝 ·𐑥𐑨𐑯, 𐑕𐑻.”

“𐑧𐑒𐑕𐑩𐑤𐑩𐑯𐑑! 𐑞𐑬'𐑕𐑑 𐑣𐑦𐑑 𐑞 𐑢𐑻𐑤𐑛 𐑚𐑲 𐑞𐑨𐑑.”

“𐑲 𐑯𐑴 𐑯𐑪𐑑, 𐑕𐑻, 𐑚𐑳𐑑 𐑲 𐑢𐑪𐑟 𐑚𐑹𐑯 𐑞𐑺.”

“𐑦𐑯 𐑞 ·𐑲𐑤 𐑝 ·𐑥𐑨𐑯, 𐑣𐑱? 𐑢𐑧𐑤, 𐑞 𐑳𐑞𐑼 𐑢𐑱, 𐑦𐑑'𐑕 𐑜𐑫𐑛. 𐑣𐑽'𐑟 𐑩 𐑥𐑨𐑯 𐑓𐑮𐑪𐑥 ·𐑥𐑨𐑯; 𐑩 𐑥𐑨𐑯 𐑚𐑹𐑯 𐑦𐑯 𐑢𐑳𐑯𐑕 𐑦𐑯𐑛𐑦𐑐𐑧𐑯𐑛𐑩𐑯𐑑 ·𐑥𐑨𐑯, 𐑯 𐑯𐑬 𐑳𐑯𐑥𐑨𐑯𐑛 𐑝 ·𐑥𐑨𐑯; 𐑢𐑦𐑗 𐑦𐑟 𐑕𐑳𐑒𐑑 𐑦𐑯—𐑚𐑲 𐑢𐑪𐑑? 𐑳𐑐 𐑢𐑦𐑞 𐑞 𐑮𐑰𐑤! 𐑞 𐑛𐑧𐑛, 𐑚𐑤𐑲𐑯𐑛 𐑢𐑷𐑤 𐑚𐑳𐑑𐑕 𐑷𐑤 𐑦𐑯𐑒𐑢𐑲𐑼𐑦𐑙 𐑣𐑧𐑛𐑟 𐑨𐑑 𐑤𐑭𐑕𐑑. 𐑳𐑐 𐑢𐑦𐑞 𐑦𐑑! 𐑕𐑴.”

𐑞 𐑤𐑪𐑜 𐑢𐑪𐑟 𐑣𐑰𐑝𐑛. 𐑞 𐑤𐑵𐑕 𐑒𐑶𐑤𐑟 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑕𐑑𐑮𐑱𐑑𐑩𐑯𐑛 𐑬𐑑 𐑦𐑯 𐑩 𐑤𐑪𐑙 𐑛𐑮𐑨𐑜𐑦𐑙 𐑤𐑲𐑯 𐑩𐑕𐑑𐑻𐑯, 𐑯 𐑞𐑧𐑯, 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦, 𐑞 𐑮𐑰𐑤 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑢𐑻𐑤. 𐑦𐑯 𐑑𐑻𐑯, 𐑡𐑻𐑒𐑦𐑙𐑤𐑦 𐑮𐑱𐑟𐑛 𐑯 𐑤𐑴𐑼𐑛 𐑚𐑲 𐑞 𐑮𐑴𐑤𐑦𐑙 𐑚𐑦𐑤𐑴𐑟, 𐑞 𐑑𐑴𐑦𐑙 𐑮𐑦𐑟𐑦𐑕𐑑𐑩𐑯𐑕 𐑝 𐑞 𐑤𐑪𐑜 𐑒𐑷𐑟𐑛 𐑞 𐑴𐑤𐑛 𐑮𐑰𐑤𐑥𐑩𐑯 𐑑 𐑕𐑑𐑨𐑜𐑼 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦.

“𐑣𐑴𐑤𐑛 𐑣𐑸𐑛!”

𐑕𐑯𐑨𐑐! 𐑞 𐑴𐑝𐑼𐑕𐑑𐑮𐑱𐑯𐑛 𐑤𐑲𐑯 𐑕𐑨𐑜𐑛 𐑛𐑬𐑯 𐑦𐑯 𐑢𐑳𐑯 𐑤𐑪𐑙 𐑓𐑧𐑕𐑑𐑵𐑯; 𐑞 𐑑𐑳𐑜𐑦𐑙 𐑤𐑪𐑜 𐑢𐑪𐑟 𐑜𐑪𐑯.

“𐑲 𐑒𐑮𐑳𐑖 𐑞 𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑, 𐑞 𐑔𐑳𐑯𐑛𐑼 𐑑𐑻𐑯𐑟 𐑞 𐑯𐑰𐑛𐑩𐑤𐑟, 𐑯 𐑯𐑬 𐑞 𐑥𐑨𐑛 𐑕𐑰 𐑐𐑸𐑑𐑕 𐑞 𐑤𐑪𐑜-𐑤𐑲𐑯. 𐑚𐑳𐑑 ·𐑱𐑣𐑨𐑚 𐑒𐑨𐑯 𐑥𐑧𐑯𐑛 𐑷𐑤. 𐑣𐑷𐑤 𐑦𐑯 𐑣𐑽, ·𐑑𐑭𐑣𐑰𐑖𐑩𐑯; 𐑮𐑰𐑤 𐑳𐑐, ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯. 𐑯 𐑤𐑫𐑒 𐑘𐑰, 𐑤𐑧𐑑 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑥𐑱𐑒 𐑩𐑯𐑳𐑞𐑼 𐑤𐑪𐑜, 𐑯 𐑥𐑧𐑯𐑛 𐑞𐑬 𐑞 𐑤𐑲𐑯. 𐑕𐑰 𐑑 𐑦𐑑.”

“𐑞𐑺 𐑣𐑰 𐑜𐑴𐑟 𐑯𐑬; 𐑑 𐑣𐑦𐑥 𐑯𐑳𐑔𐑦𐑙'𐑟 𐑣𐑨𐑐𐑩𐑯𐑛; 𐑚𐑳𐑑 𐑑 𐑥𐑰, 𐑞 𐑕𐑒𐑿𐑼 𐑕𐑰𐑥𐑟 𐑤𐑵𐑕𐑩𐑯𐑦𐑙 𐑬𐑑 𐑝 𐑞 𐑥𐑦𐑛𐑩𐑤 𐑝 𐑞 𐑢𐑻𐑤𐑛. 𐑣𐑷𐑤 𐑦𐑯, 𐑣𐑷𐑤 𐑦𐑯, ·𐑑𐑭𐑣𐑰𐑖𐑩𐑯! 𐑞𐑰𐑟 𐑤𐑲𐑯𐑟 𐑮𐑳𐑯 𐑣𐑴𐑤, 𐑯 𐑢𐑻𐑤𐑦𐑙 𐑬𐑑: 𐑒𐑳𐑥 𐑦𐑯 𐑚𐑮𐑴𐑒𐑩𐑯, 𐑯 𐑛𐑮𐑨𐑜𐑦𐑙 𐑕𐑤𐑴. 𐑣𐑭, ·𐑐𐑦𐑐? 𐑒𐑳𐑥 𐑑 𐑣𐑧𐑤𐑐; 𐑱, ·𐑐𐑦𐑐?”

“𐑐𐑦𐑐? 𐑣𐑵𐑥 𐑒𐑷𐑤 𐑘𐑰 ·𐑐𐑦𐑐? 𐑐𐑦𐑐 𐑡𐑳𐑥𐑐𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑱𐑤-𐑚𐑴𐑑. 𐑐𐑦𐑐'𐑕 𐑥𐑦𐑕𐑦𐑙. 𐑤𐑧𐑑'𐑕 𐑕𐑰 𐑯𐑬 𐑦𐑓 𐑘𐑰 𐑣𐑨𐑝𐑩𐑯𐑑 𐑓𐑦𐑖𐑑 𐑣𐑦𐑥 𐑳𐑐 𐑣𐑽, 𐑓𐑦𐑖𐑼𐑥𐑩𐑯. 𐑦𐑑 𐑛𐑮𐑨𐑜𐑟 𐑣𐑸𐑛; 𐑲 𐑜𐑧𐑕 𐑣𐑰'𐑟 𐑣𐑴𐑤𐑛𐑦𐑙 𐑪𐑯. 𐑡𐑻𐑒 𐑣𐑦𐑥, ·𐑑𐑭𐑣𐑰𐑑𐑦! 𐑡𐑻𐑒 𐑣𐑦𐑥 𐑪𐑓; 𐑢𐑰 𐑣𐑷𐑤 𐑦𐑯 𐑯𐑴 𐑒𐑬𐑼𐑛𐑟 𐑣𐑽. 𐑣𐑴! 𐑞𐑺'𐑟 𐑣𐑦𐑟 𐑸𐑥 𐑡𐑳𐑕𐑑 𐑚𐑮𐑱𐑒𐑦𐑙 𐑢𐑷𐑑𐑼. 𐑩 𐑣𐑨𐑗𐑩𐑑! 𐑩 𐑣𐑨𐑗𐑩𐑑! 𐑒𐑳𐑑 𐑦𐑑 𐑪𐑓—𐑢𐑰 𐑣𐑷𐑤 𐑦𐑯 𐑯𐑴 𐑒𐑬𐑼𐑛𐑟 𐑣𐑽. 𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚! 𐑕𐑻, 𐑕𐑻! 𐑣𐑽'𐑟 ·𐑐𐑦𐑐, 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑜𐑧𐑑 𐑪𐑯 𐑚𐑹𐑛 𐑩𐑜𐑱𐑯.”

“𐑐𐑰𐑕, 𐑞𐑬 𐑒𐑮𐑱𐑟𐑦 𐑤𐑵𐑯,” 𐑒𐑮𐑲𐑛 𐑞 ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯, 𐑕𐑰𐑟𐑦𐑙 𐑣𐑦𐑥 𐑚𐑲 𐑞 𐑸𐑥. “𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒!”

“𐑞 𐑜𐑮𐑱𐑑𐑼 𐑦𐑛𐑾𐑑 𐑧𐑝𐑼 𐑕𐑒𐑴𐑤𐑛𐑟 𐑞 𐑤𐑧𐑕𐑼,” 𐑥𐑳𐑑𐑼𐑛 ·𐑱𐑣𐑨𐑚, 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙. “𐑣𐑨𐑯𐑛𐑟 𐑪𐑓 𐑓𐑮𐑪𐑥 𐑞𐑨𐑑 𐑣𐑴𐑤𐑦𐑯𐑩𐑕! 𐑢𐑺 𐑕𐑱𐑩𐑕𐑑 𐑞𐑬 ·𐑐𐑦𐑐 𐑢𐑪𐑟, 𐑚𐑶?

“𐑩𐑕𐑑𐑻𐑯 𐑞𐑺, 𐑕𐑻, 𐑩𐑕𐑑𐑻𐑯! 𐑤𐑴! 𐑤𐑴!”

“𐑯 𐑣𐑵 𐑸𐑑 𐑞𐑬, 𐑚𐑶? 𐑲 𐑕𐑰 𐑯𐑪𐑑 𐑥𐑲 𐑮𐑦𐑓𐑤𐑧𐑒𐑖𐑩𐑯 𐑦𐑯 𐑞 𐑝𐑱𐑒𐑩𐑯𐑑 𐑐𐑿𐑐𐑩𐑤𐑟 𐑝 𐑞𐑲 𐑲𐑟. 𐑴 ·𐑜𐑪𐑛! 𐑞𐑨𐑑 𐑥𐑨𐑯 𐑖𐑵𐑛 𐑚𐑰 𐑩 𐑔𐑦𐑙 𐑓 𐑦𐑥𐑹𐑑𐑩𐑤 𐑕𐑴𐑤𐑟 𐑑 𐑕𐑦𐑝 𐑔𐑮𐑵! 𐑣𐑵 𐑸𐑑 𐑞𐑬, 𐑚𐑶?”

“𐑚𐑧𐑤-𐑚𐑶, 𐑕𐑻; 𐑖𐑦𐑐'𐑕-𐑒𐑮𐑲𐑼; 𐑛𐑦𐑙, 𐑛𐑪𐑙, 𐑛𐑦𐑙! 𐑐𐑦𐑐! 𐑐𐑦𐑐! 𐑐𐑦𐑐! 𐑢𐑳𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑐𐑬𐑯𐑛𐑟 𐑝 𐑒𐑤𐑱 𐑮𐑦𐑢𐑹𐑛 𐑓 ·𐑐𐑦𐑐; 𐑓𐑲𐑝 𐑓𐑰𐑑 𐑣𐑲—𐑤𐑫𐑒𐑕 𐑒𐑬𐑼𐑛𐑤𐑦—𐑒𐑢𐑦𐑒𐑩𐑕𐑑 𐑯𐑴𐑯 𐑚𐑲 𐑞𐑨𐑑! 𐑛𐑦𐑙, 𐑛𐑪𐑙, 𐑛𐑦𐑙! 𐑣𐑵'𐑟 𐑕𐑰𐑯 ·𐑐𐑦𐑐 𐑞 𐑒𐑬𐑼𐑛?”

“𐑞𐑺 𐑒𐑨𐑯 𐑚𐑰 𐑯𐑴 𐑣𐑸𐑑𐑕 𐑩𐑚𐑳𐑝 𐑞 𐑕𐑯𐑴-𐑤𐑲𐑯. 𐑴, 𐑘𐑰 𐑓𐑮𐑴𐑟𐑩𐑯 𐑣𐑧𐑝𐑩𐑯𐑟! 𐑤𐑫𐑒 𐑛𐑬𐑯 𐑣𐑽. 𐑘𐑰 𐑛𐑦𐑛 𐑚𐑦𐑜𐑧𐑑 𐑞𐑦𐑕 𐑤𐑳𐑒𐑤𐑩𐑕 𐑗𐑲𐑤𐑛, 𐑯 𐑣𐑨𐑝 𐑩𐑚𐑨𐑯𐑛𐑩𐑯𐑛 𐑣𐑦𐑥, 𐑘𐑰 𐑒𐑮𐑦𐑱𐑑𐑦𐑝 𐑤𐑦𐑚𐑼𐑑𐑰𐑯𐑟. 𐑣𐑽, 𐑚𐑶; ·𐑱𐑣𐑨𐑚'𐑟 𐑒𐑨𐑚𐑦𐑯 𐑖𐑨𐑤 𐑚𐑰 ·𐑐𐑦𐑐'𐑕 𐑣𐑴𐑥 𐑣𐑧𐑯𐑕𐑓𐑹𐑔, 𐑢𐑲𐑤 ·𐑱𐑣𐑨𐑚 𐑤𐑦𐑝𐑟. 𐑞𐑬 ·𐑑𐑬𐑗𐑧𐑕𐑑 𐑥𐑲 𐑦𐑯𐑥𐑴𐑕𐑑 𐑕𐑧𐑯𐑑𐑼, 𐑚𐑶; 𐑞𐑬 𐑸𐑑 𐑑𐑲𐑛 𐑑 𐑥𐑰 𐑚𐑲 𐑒𐑹𐑛𐑟 𐑢𐑴𐑝𐑩𐑯 𐑝 𐑥𐑲 𐑣𐑸𐑑-𐑕𐑑𐑮𐑦𐑙𐑟. 𐑒𐑳𐑥, 𐑤𐑧𐑑'𐑕 𐑛𐑬𐑯.”

“𐑢𐑪𐑑'𐑕 𐑞𐑦𐑕? 𐑣𐑽'𐑟 𐑝𐑧𐑤𐑝𐑩𐑑 𐑖𐑸𐑒-𐑕𐑒𐑦𐑯,” 𐑦𐑯𐑑𐑧𐑯𐑑𐑤𐑦 𐑜𐑱𐑟𐑦𐑙 𐑨𐑑 ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑨𐑯𐑛, 𐑯 𐑓𐑰𐑤𐑦𐑙 𐑦𐑑. “𐑭, 𐑯𐑬, 𐑣𐑨𐑛 𐑐𐑫𐑼 ·𐑐𐑦𐑐 𐑚𐑳𐑑 𐑓𐑧𐑤𐑑 𐑕𐑴 𐑒𐑲𐑯𐑛 𐑩 𐑔𐑦𐑙 𐑨𐑟 𐑞𐑦𐑕, 𐑐𐑼𐑣𐑨𐑐𐑕 𐑣𐑰 𐑣𐑨𐑛 𐑯𐑺 𐑚𐑰𐑯 𐑤𐑪𐑕𐑑! 𐑞𐑦𐑕 𐑕𐑰𐑥𐑟 𐑑 𐑥𐑰, 𐑕𐑻, 𐑨𐑟 𐑩 𐑥𐑨𐑯-𐑮𐑴𐑐; 𐑕𐑳𐑥𐑔𐑦𐑙 𐑞𐑨𐑑 𐑢𐑰𐑒 𐑕𐑴𐑤𐑟 𐑥𐑱 𐑣𐑴𐑤𐑛 𐑚𐑲. 𐑴, 𐑕𐑻, 𐑤𐑧𐑑 𐑴𐑤𐑛 ·𐑐𐑻𐑔 𐑯𐑬 𐑒𐑳𐑥 𐑯 𐑮𐑦𐑝𐑩𐑑 𐑞𐑰𐑟 𐑑𐑵 𐑣𐑨𐑯𐑛𐑟 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑞 𐑚𐑤𐑨𐑒 𐑢𐑳𐑯 𐑢𐑦𐑞 𐑞 𐑢𐑲𐑑, 𐑓 𐑲 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑤𐑧𐑑 𐑞𐑦𐑕 𐑜𐑴.”

“𐑴, 𐑚𐑶, 𐑯𐑹 𐑢𐑦𐑤 𐑲 𐑞𐑰, 𐑩𐑯𐑤𐑧𐑕 𐑲 𐑖𐑵𐑛 𐑞𐑺𐑚𐑲 𐑛𐑮𐑨𐑜 𐑞𐑰 𐑑 𐑢𐑻𐑕 𐑣𐑪𐑮𐑼𐑟 𐑞𐑨𐑯 𐑸 𐑣𐑽. 𐑒𐑳𐑥, 𐑞𐑧𐑯, 𐑑 𐑥𐑲 𐑒𐑨𐑚𐑦𐑯. 𐑤𐑴! 𐑘𐑰 𐑚𐑦𐑤𐑰𐑝𐑼𐑟 𐑦𐑯 𐑜𐑪𐑛𐑟 𐑷𐑤 𐑜𐑫𐑛𐑯𐑩𐑕, 𐑯 𐑦𐑯 𐑥𐑨𐑯 𐑷𐑤 𐑦𐑤, 𐑤𐑴 𐑿! 𐑕𐑰 𐑞 𐑪𐑥𐑯𐑦𐑕𐑾𐑯𐑑 𐑜𐑪𐑛𐑟 𐑩𐑚𐑤𐑦𐑝𐑾𐑕 𐑝 𐑕𐑳𐑓𐑼𐑦𐑙 𐑥𐑨𐑯; 𐑯 𐑥𐑨𐑯, 𐑞𐑴 𐑦𐑛𐑦𐑪𐑑𐑦𐑒, 𐑯 𐑯𐑴𐑦𐑙 𐑯𐑪𐑑 𐑢𐑪𐑑 𐑣𐑰 𐑛𐑳𐑟, 𐑘𐑧𐑑 𐑓𐑫𐑤 𐑝 𐑞 𐑕𐑢𐑰𐑑 𐑔𐑦𐑙𐑟 𐑝 𐑤𐑳𐑝 𐑯 𐑜𐑮𐑨𐑑𐑦𐑑𐑿𐑛. 𐑒𐑳𐑥! 𐑲 𐑓𐑰𐑤 𐑐𐑮𐑬𐑛𐑼 𐑤𐑰𐑛𐑦𐑙 𐑞𐑰 𐑚𐑲 𐑞𐑲 𐑚𐑤𐑨𐑒 𐑣𐑨𐑯𐑛, 𐑞𐑨𐑯 𐑞𐑴 𐑲 𐑜𐑮𐑭𐑕𐑐𐑑 𐑩𐑯 ·𐑧𐑥𐑐𐑼𐑼'𐑟!”

“𐑞𐑺 𐑜𐑴 𐑑𐑵 𐑛𐑭𐑓𐑑 𐑢𐑳𐑯𐑟 𐑯𐑬,” 𐑥𐑳𐑑𐑼𐑛 𐑞 𐑴𐑤𐑛 ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯. “𐑢𐑳𐑯 𐑛𐑭𐑓𐑑 𐑢𐑦𐑞 𐑕𐑑𐑮𐑧𐑙𐑔, 𐑞 𐑳𐑞𐑼 𐑛𐑭𐑓𐑑 𐑢𐑦𐑞 𐑢𐑰𐑒𐑯𐑩𐑕. 𐑚𐑳𐑑 𐑣𐑽'𐑟 𐑞 𐑧𐑯𐑛 𐑝 𐑞 𐑮𐑪𐑑𐑩𐑯 𐑤𐑲𐑯—𐑷𐑤 𐑛𐑮𐑦𐑐𐑦𐑙, 𐑑𐑵. 𐑥𐑧𐑯𐑛 𐑦𐑑, 𐑱? 𐑲 𐑔𐑦𐑙𐑒 𐑢𐑰 𐑣𐑨𐑛 𐑚𐑧𐑕𐑑 𐑣𐑨𐑝 𐑩 𐑯𐑿 𐑤𐑲𐑯 𐑷𐑤𐑑𐑩𐑜𐑧𐑞𐑼. 𐑲'𐑤 𐑕𐑰 ·𐑥𐑼 ·𐑕𐑑𐑳𐑚 𐑩𐑚𐑬𐑑 𐑦𐑑.”
