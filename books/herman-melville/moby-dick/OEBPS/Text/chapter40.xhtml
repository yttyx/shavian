·𐑥𐑱𐑑'𐑕 ·𐑝𐑶𐑕 𐑓𐑮𐑪𐑥 𐑞 ·𐑒𐑢𐑹𐑑𐑼-·𐑛𐑧𐑒. 𐑱𐑑 𐑚𐑧𐑤𐑟 𐑞𐑺, 𐑓𐑹𐑢𐑼𐑛!

2·𐑛 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 ·𐑕𐑱𐑤𐑼. 𐑩𐑝𐑭𐑕𐑑 𐑞 𐑒𐑹𐑩𐑕! 𐑱𐑑 𐑚𐑧𐑤𐑟 𐑞𐑺! 𐑛'𐑘𐑰 𐑣𐑽, 𐑚𐑧𐑤-𐑚𐑶? 𐑕𐑑𐑮𐑲𐑒 𐑞 𐑚𐑧𐑤 𐑱𐑑, 𐑞𐑬 ·𐑐𐑦𐑐! 𐑞𐑬 𐑚𐑤𐑨𐑒𐑤𐑦𐑙! 𐑯 𐑤𐑧𐑑 𐑥𐑰 𐑒𐑷𐑤 𐑞 𐑢𐑪𐑗. 𐑲'𐑝 𐑞 𐑕𐑹𐑑 𐑝 𐑥𐑬𐑔 𐑓 𐑞𐑨𐑑—𐑞 𐑣𐑪𐑜𐑟𐑣𐑧𐑛 𐑥𐑬𐑔. 𐑕𐑴, 𐑕𐑴, (𐑔𐑮𐑳𐑕𐑑𐑕 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑛𐑬𐑯 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤,) ·𐑕𐑑𐑸-𐑚𐑴-l-𐑰-𐑰-n-𐑟, 𐑩-h-𐑴-𐑦! 𐑱𐑑 𐑚𐑧𐑤𐑟 𐑞𐑺 𐑚𐑦𐑤𐑴! 𐑑𐑳𐑥𐑚𐑩𐑤 𐑳𐑐!

·𐑛𐑳𐑗 ·𐑕𐑱𐑤𐑼. 𐑜𐑮𐑨𐑯𐑛 𐑕𐑯𐑵𐑟𐑦𐑙 𐑑-𐑯𐑲𐑑, 𐑥𐑱𐑑𐑦; 𐑓𐑨𐑑 𐑯𐑲𐑑 𐑓 𐑞𐑨𐑑. 𐑲 𐑥𐑸𐑒 𐑞𐑦𐑕 𐑦𐑯 𐑬𐑼 𐑴𐑤𐑛 ·𐑥𐑴𐑜𐑩𐑤'𐑟 𐑢𐑲𐑯; 𐑦𐑑'𐑕 𐑒𐑢𐑲𐑑 𐑨𐑟 𐑛𐑧𐑛𐑩𐑯𐑦𐑙 𐑑 𐑕𐑳𐑥 𐑨𐑟 𐑓𐑦𐑤𐑦𐑐𐑦𐑙 𐑑 𐑳𐑞𐑼𐑟. 𐑢𐑰 𐑕𐑦𐑙; 𐑞𐑱 𐑕𐑤𐑰𐑐—𐑲, 𐑤𐑲 𐑛𐑬𐑯 𐑞𐑺, 𐑤𐑲𐑒 𐑜𐑮𐑬𐑯𐑛-𐑑𐑽 𐑚𐑳𐑑𐑕. 𐑨𐑑 '𐑩𐑥 𐑩𐑜𐑱𐑯! 𐑞𐑺, 𐑑𐑱𐑒 𐑞𐑦𐑕 𐑒𐑪𐑐𐑼-𐑐𐑳𐑥𐑐, 𐑯 𐑣𐑱𐑤 '𐑩𐑥 𐑔𐑮𐑵 𐑦𐑑. 𐑑𐑧𐑤 '𐑩𐑥 𐑑 𐑩𐑝𐑭𐑕𐑑 𐑛𐑮𐑰𐑥𐑦𐑙 𐑝 𐑞𐑺 𐑤𐑨𐑕𐑩𐑟. 𐑑𐑧𐑤 '𐑩𐑥 𐑦𐑑'𐑕 𐑞 𐑮𐑧𐑟𐑼𐑧𐑒𐑖𐑩𐑯; 𐑞𐑱 𐑥𐑳𐑕𐑑 𐑒𐑦𐑕 𐑞𐑺 𐑤𐑭𐑕𐑑, 𐑯 𐑒𐑳𐑥 𐑑 𐑡𐑳𐑡𐑥𐑩𐑯𐑑. 𐑞𐑨𐑑'𐑕 𐑞 𐑢𐑱—𐑞𐑨𐑑'𐑕 𐑦𐑑; 𐑞𐑲 𐑔𐑮𐑴𐑑 𐑱𐑯𐑑 𐑕𐑐𐑶𐑤𐑛 𐑢𐑦𐑞 𐑰𐑑𐑦𐑙 ·𐑨𐑥𐑕𐑑𐑼𐑛𐑨𐑥 𐑚𐑳𐑑𐑼.

·𐑓𐑮𐑧𐑯𐑗 ·𐑕𐑱𐑤𐑼. 𐑣𐑦𐑕𐑑, 𐑚𐑶𐑟! 𐑤𐑧𐑑'𐑕 𐑣𐑨𐑝 𐑩 𐑡𐑦𐑜 𐑹 𐑑𐑵 𐑚𐑦𐑓𐑹 𐑢𐑰 𐑮𐑲𐑛 𐑑 𐑨𐑙𐑒𐑼 𐑦𐑯 ·𐑚𐑤𐑨𐑙𐑒𐑩𐑑 ·𐑚𐑱. 𐑢𐑪𐑑 𐑕𐑱 𐑘𐑰? 𐑞𐑺 𐑒𐑳𐑥𐑟 𐑞 𐑳𐑞𐑼 𐑢𐑪𐑗. 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑷𐑤 𐑤𐑧𐑜𐑟! 𐑐𐑦𐑐! 𐑤𐑦𐑑𐑩𐑤 ·𐑐𐑦𐑐! 𐑣𐑼𐑭 𐑢𐑦𐑞 𐑘𐑹 𐑑𐑨𐑥𐑚𐑼𐑰𐑯!

·𐑐𐑦𐑐. (𐑕𐑳𐑤𐑒𐑦 𐑯 𐑕𐑤𐑰𐑐𐑦.) 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑢𐑺 𐑦𐑑 𐑦𐑟.

·𐑓𐑮𐑧𐑯𐑗 ·𐑕𐑱𐑤𐑼. 𐑚𐑰𐑑 𐑞𐑲 𐑚𐑧𐑤𐑦, 𐑞𐑧𐑯, 𐑯 𐑢𐑨𐑜 𐑞𐑲 𐑽𐑟. 𐑡𐑦𐑜 𐑦𐑑, 𐑥𐑧𐑯, 𐑲 𐑕𐑱; 𐑥𐑧𐑮𐑦'𐑟 𐑞 𐑢𐑻𐑛; 𐑣𐑼𐑭! 𐑛𐑨𐑥 𐑥𐑰, 𐑢𐑴𐑯𐑑 𐑿 𐑛𐑭𐑯𐑕? 𐑓𐑹𐑥, 𐑯𐑬, ·𐑦𐑯𐑛𐑾𐑯-𐑓𐑲𐑤, 𐑯 𐑜𐑨𐑤𐑩𐑐 𐑦𐑯𐑑𐑵 𐑞 𐑛𐑳𐑚𐑩𐑤-𐑖𐑳𐑓𐑩𐑤? 𐑔𐑮𐑴 𐑘𐑹𐑕𐑧𐑤𐑝𐑟! 𐑤𐑧𐑜𐑟! 𐑤𐑧𐑜𐑟!

·𐑲𐑕𐑤𐑩𐑯𐑛 ·𐑕𐑱𐑤𐑼. 𐑲 𐑛𐑴𐑯𐑑 𐑤𐑲𐑒 𐑘𐑹 𐑓𐑤𐑹, 𐑥𐑱𐑑𐑦; 𐑦𐑑'𐑕 𐑑𐑵 𐑕𐑐𐑮𐑦𐑙𐑦 𐑑 𐑥𐑲 𐑑𐑱𐑕𐑑. 𐑲'𐑥 𐑿𐑕𐑑 𐑑 𐑲𐑕-𐑓𐑤𐑹𐑟. 𐑲'𐑥 𐑕𐑪𐑮𐑦 𐑑 𐑔𐑮𐑴 𐑒𐑴𐑤𐑛 𐑢𐑷𐑑𐑼 𐑪𐑯 𐑞 𐑕𐑳𐑚𐑡𐑧𐑒𐑑; 𐑚𐑳𐑑 𐑦𐑒𐑕𐑒𐑿𐑕 𐑥𐑰.

·𐑥𐑷𐑤𐑑𐑰𐑟 ·𐑕𐑱𐑤𐑼. 𐑥𐑰 𐑑𐑵; 𐑢𐑺'𐑟 𐑘𐑹 𐑜𐑻𐑤𐑟? 𐑣𐑵 𐑚𐑳𐑑 𐑩 𐑓𐑵𐑤 𐑢𐑫𐑛 𐑑𐑱𐑒 𐑣𐑦𐑟 𐑤𐑧𐑓𐑑 𐑣𐑨𐑯𐑛 𐑚𐑲 𐑣𐑦𐑟 𐑮𐑲𐑑, 𐑯 𐑕𐑱 𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑣𐑬 𐑛'𐑘𐑰 𐑛𐑵? 𐑐𐑸𐑑𐑯𐑼𐑟! 𐑲 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑐𐑸𐑑𐑯𐑼𐑟!

·𐑕𐑦𐑕𐑦𐑤𐑾𐑯 ·𐑕𐑱𐑤𐑼. 𐑲; 𐑜𐑻𐑤𐑟 𐑯 𐑩 𐑜𐑮𐑰𐑯!—𐑞𐑧𐑯 𐑲'𐑤 𐑣𐑪𐑐 𐑢𐑦𐑞 𐑘𐑰; 𐑘𐑱, 𐑑𐑻𐑯 𐑜𐑮𐑭𐑕𐑣𐑪𐑐𐑼!

·𐑤𐑪𐑙-·𐑲𐑤𐑩𐑯𐑛 ·𐑕𐑱𐑤𐑼. 𐑢𐑧𐑤, 𐑢𐑧𐑤, 𐑘𐑰 𐑕𐑳𐑤𐑒𐑦𐑟, 𐑞𐑺'𐑟 𐑐𐑤𐑧𐑯𐑑𐑦 𐑥𐑹 𐑝 𐑳𐑕. 𐑣𐑴 𐑒𐑹𐑯 𐑢𐑧𐑯 𐑿 𐑥𐑱, 𐑕𐑱 𐑲. 𐑷𐑤 𐑤𐑧𐑜𐑟 𐑜𐑴 𐑑 𐑣𐑸𐑝𐑦𐑕𐑑 𐑕𐑵𐑯. 𐑭! 𐑣𐑽 𐑒𐑳𐑥𐑟 𐑞 𐑥𐑿𐑟𐑦𐑒; 𐑯𐑬 𐑓 𐑦𐑑!

·𐑩𐑟𐑹 ·𐑕𐑱𐑤𐑼. (𐑩𐑕𐑧𐑯𐑛𐑦𐑙, 𐑯 𐑐𐑦𐑗𐑦𐑙 𐑞 𐑑𐑨𐑥𐑚𐑼𐑰𐑯 𐑳𐑐 𐑞 𐑕𐑒𐑳𐑑𐑩𐑤.) 𐑣𐑽 𐑿 𐑸, ·𐑐𐑦𐑐; 𐑯 𐑞𐑺'𐑟 𐑞 𐑢𐑦𐑯𐑛𐑤𐑩𐑕-𐑚𐑦𐑑𐑕; 𐑳𐑐 𐑿 𐑥𐑬𐑯𐑑! 𐑯𐑬, 𐑚𐑶𐑟! (𐑞 𐑣𐑭𐑓 𐑝 𐑞𐑧𐑥 𐑛𐑭𐑯𐑕 𐑑 𐑞 𐑑𐑨𐑥𐑚𐑼𐑰𐑯; 𐑕𐑳𐑥 𐑜𐑴 𐑚𐑦𐑤𐑴; 𐑕𐑳𐑥 𐑕𐑤𐑰𐑐 𐑹 𐑤𐑲 𐑩𐑥𐑳𐑙 𐑞 𐑒𐑶𐑤𐑟 𐑝 𐑮𐑦𐑜𐑦𐑙. 𐑴𐑞𐑟 𐑩-𐑐𐑤𐑧𐑯𐑑𐑦.)

·𐑩𐑟𐑹 ·𐑕𐑱𐑤𐑼. (𐑛𐑭𐑯𐑕𐑦𐑙) 𐑜𐑴 𐑦𐑑, ·𐑐𐑦𐑐! 𐑚𐑨𐑙 𐑦𐑑, 𐑚𐑧𐑤-𐑚𐑶! 𐑮𐑦𐑜 𐑦𐑑, 𐑛𐑦𐑜 𐑦𐑑, 𐑕𐑑𐑦𐑜 𐑦𐑑, 𐑒𐑢𐑦𐑜 𐑦𐑑, 𐑚𐑧𐑤-𐑚𐑶! 𐑥𐑱𐑒 𐑓𐑲𐑼-𐑓𐑤𐑲𐑟; 𐑚𐑮𐑱𐑒 𐑞 𐑡𐑦𐑙𐑜𐑩𐑤𐑼𐑟!

·𐑐𐑦𐑐. 𐑡𐑦𐑙𐑜𐑩𐑤𐑼𐑟, 𐑿 𐑕𐑱?—𐑞𐑺 𐑜𐑴𐑟 𐑩𐑯𐑳𐑞𐑼, 𐑛𐑮𐑪𐑐𐑑 𐑪𐑓; 𐑲 𐑐𐑬𐑯𐑛 𐑦𐑑 𐑕𐑴.

·𐑗𐑲𐑯𐑩 ·𐑕𐑱𐑤𐑼. 𐑮𐑨𐑑𐑩𐑤 𐑞𐑲 𐑑𐑰𐑔, 𐑞𐑧𐑯, 𐑯 𐑐𐑬𐑯𐑛 𐑩𐑢𐑱; 𐑥𐑱𐑒 𐑩 𐑐𐑩𐑜𐑴𐑛𐑩 𐑝 𐑞𐑲𐑕𐑧𐑤𐑓.
