·𐑗𐑨𐑐𐑑𐑼 29. 𐑧𐑯𐑑𐑼 ·𐑱𐑣𐑨𐑚; 𐑑 𐑣𐑦𐑥, ·𐑕𐑑𐑳𐑚.

𐑕𐑳𐑥 𐑛𐑱𐑟 𐑦𐑤𐑨𐑐𐑕𐑑, 𐑯 𐑲𐑕 𐑯 𐑲𐑕𐑚𐑻𐑜𐑟 𐑷𐑤 𐑩𐑕𐑑𐑻𐑯, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑯𐑬 𐑢𐑧𐑯𐑑 𐑮𐑴𐑤𐑦𐑙 𐑔𐑮𐑵 𐑞 𐑚𐑮𐑲𐑑 ·𐑒𐑰𐑑𐑴 𐑕𐑐𐑮𐑦𐑙, 𐑢𐑦𐑗, 𐑨𐑑 𐑕𐑰, 𐑷𐑤𐑥𐑴𐑕𐑑 𐑐𐑼𐑐𐑧𐑗𐑵𐑩𐑤𐑦 𐑮𐑱𐑯𐑟 𐑪𐑯 𐑞 𐑔𐑮𐑧𐑖𐑣𐑴𐑤𐑛 𐑝 𐑞 𐑦𐑑𐑻𐑯𐑩𐑤 𐑷𐑜𐑩𐑕𐑑 𐑝 𐑞 ·𐑑𐑮𐑪𐑐𐑦𐑒. 𐑞 𐑢𐑹𐑥𐑤𐑦 𐑒𐑵𐑤, 𐑒𐑤𐑽, 𐑮𐑦𐑙𐑦𐑙, 𐑐𐑼𐑓𐑿𐑥𐑛, 𐑴𐑝𐑼𐑓𐑤𐑴𐑦𐑙, 𐑮𐑦𐑛𐑳𐑯𐑛𐑩𐑯𐑑 𐑛𐑱𐑟, 𐑢𐑻 𐑨𐑟 𐑒𐑮𐑦𐑕𐑑𐑩𐑤 𐑜𐑪𐑚𐑤𐑩𐑑𐑕 𐑝 ·𐑐𐑻𐑖𐑩𐑯 𐑖𐑻𐑚𐑩𐑑, 𐑣𐑰𐑐𐑑 𐑳𐑐—𐑓𐑤𐑱𐑒𐑑 𐑳𐑐, 𐑢𐑦𐑞 𐑮𐑴𐑟-𐑢𐑷𐑑𐑼 𐑕𐑯𐑴. 𐑞 𐑕𐑑𐑸𐑛 𐑯 𐑕𐑑𐑱𐑑𐑤𐑦 𐑯𐑲𐑑𐑕 𐑕𐑰𐑥𐑛 𐑣𐑷𐑑𐑦 𐑛𐑱𐑥𐑟 𐑦𐑯 𐑡𐑵𐑩𐑤𐑛 𐑝𐑧𐑤𐑝𐑩𐑑𐑕, 𐑯𐑻𐑕𐑦𐑙 𐑨𐑑 𐑣𐑴𐑥 𐑦𐑯 𐑤𐑴𐑯𐑤𐑦 𐑐𐑮𐑲𐑛, 𐑞 𐑥𐑧𐑥𐑼𐑦 𐑝 𐑞𐑺 𐑨𐑚𐑕𐑩𐑯𐑑 𐑒𐑪𐑙𐑒𐑼𐑦𐑙 ·𐑻𐑤𐑟, 𐑞 𐑜𐑴𐑤𐑛𐑩𐑯 𐑣𐑧𐑤𐑥𐑩𐑑𐑩𐑛 𐑕𐑳𐑯𐑟! 𐑓 𐑕𐑤𐑰𐑐𐑦𐑙 𐑥𐑨𐑯, '𐑑𐑢𐑪𐑟 𐑣𐑸𐑛 𐑑 𐑗𐑵𐑟 𐑚𐑦𐑑𐑢𐑰𐑯 𐑕𐑳𐑗 𐑢𐑦𐑯𐑕𐑩𐑥 𐑛𐑱𐑟 𐑯 𐑕𐑳𐑗 𐑕𐑦𐑛𐑿𐑕𐑦𐑙 𐑯𐑲𐑑𐑕. 𐑚𐑳𐑑 𐑷𐑤 𐑞 𐑢𐑦𐑗𐑼𐑦𐑟 𐑝 𐑞𐑨𐑑 𐑳𐑯𐑢𐑱𐑯𐑦𐑙 𐑢𐑧𐑞𐑼 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑥𐑽𐑤𐑦 𐑤𐑧𐑯𐑛 𐑯𐑿 𐑕𐑐𐑧𐑤𐑟 𐑯 𐑐𐑴𐑑𐑩𐑯𐑕𐑦𐑟 𐑑 𐑞 𐑬𐑑𐑢𐑼𐑛 𐑢𐑻𐑤𐑛. 𐑦𐑯𐑢𐑼𐑛 𐑞𐑱 𐑑𐑻𐑯𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑴𐑤, 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑢𐑧𐑯 𐑞 𐑕𐑑𐑦𐑤 𐑥𐑲𐑤𐑛 𐑬𐑼𐑟 𐑝 𐑰𐑝 𐑒𐑱𐑥 𐑪𐑯; 𐑞𐑧𐑯, 𐑥𐑧𐑥𐑼𐑦 𐑖𐑪𐑑 𐑣𐑻 𐑒𐑮𐑦𐑕𐑑𐑩𐑤𐑟 𐑨𐑟 𐑞 𐑒𐑤𐑽 𐑲𐑕 𐑥𐑴𐑕𐑑 𐑓𐑹𐑥𐑟 𐑝 𐑯𐑶𐑟𐑤𐑩𐑕 𐑑𐑢𐑲𐑤𐑲𐑑𐑕. 𐑯 𐑷𐑤 𐑞𐑰𐑟 𐑕𐑳𐑑𐑩𐑤 𐑱𐑡𐑩𐑯𐑕𐑦𐑟, 𐑥𐑹 𐑯 𐑥𐑹 𐑞𐑱 𐑮𐑷𐑑 𐑪𐑯 ·𐑱𐑣𐑨𐑚'𐑟 𐑑𐑧𐑒𐑕𐑗𐑼.

𐑴𐑤𐑛 𐑱𐑡 𐑦𐑟 𐑷𐑤𐑢𐑱𐑟 𐑢𐑱𐑒𐑓𐑩𐑤; 𐑨𐑟 𐑦𐑓, 𐑞 𐑤𐑪𐑙𐑜𐑼 𐑤𐑦𐑙𐑒𐑑 𐑢𐑦𐑞 𐑤𐑲𐑓, 𐑞 𐑤𐑧𐑕 𐑥𐑨𐑯 𐑣𐑨𐑕 𐑑 𐑛𐑵 𐑢𐑦𐑞 𐑷𐑑 𐑞𐑨𐑑 𐑤𐑫𐑒𐑕 𐑤𐑲𐑒 𐑛𐑧𐑔. 𐑩𐑥𐑳𐑙 𐑕𐑰-𐑒𐑩𐑥𐑭𐑯𐑛𐑼𐑟, 𐑞 𐑴𐑤𐑛 𐑜𐑮𐑱𐑚𐑽𐑛𐑟 𐑢𐑦𐑤 𐑪𐑓𐑩𐑯𐑩𐑕𐑑 𐑤𐑰𐑝 𐑞𐑺 𐑚𐑻𐑔𐑕 𐑑 𐑝𐑦𐑟𐑦𐑑 𐑞 𐑯𐑲𐑑-𐑒𐑤𐑴𐑒𐑑 𐑛𐑧𐑒. 𐑦𐑑 𐑢𐑪𐑟 𐑕𐑴 𐑢𐑦𐑞 ·𐑱𐑣𐑨𐑚; 𐑴𐑯𐑤𐑦 𐑞𐑨𐑑 𐑯𐑬, 𐑝 𐑤𐑱𐑑, 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑕𐑴 𐑥𐑳𐑗 𐑑 𐑤𐑦𐑝 𐑦𐑯 𐑞 𐑴𐑐𐑩𐑯 𐑺, 𐑞𐑨𐑑 𐑑𐑮𐑵𐑤𐑦 𐑕𐑐𐑰𐑒𐑦𐑙, 𐑣𐑦𐑟 𐑝𐑦𐑟𐑦𐑑𐑕 𐑢𐑻 𐑥𐑹 𐑑 𐑞 𐑒𐑨𐑚𐑦𐑯, 𐑞𐑨𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑑 𐑞 𐑐𐑤𐑨𐑙𐑒𐑕. “𐑦𐑑 𐑓𐑰𐑤𐑟 𐑤𐑲𐑒 𐑜𐑴𐑦𐑙 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑢𐑳𐑯'𐑟 𐑑𐑵𐑥,”—𐑣𐑰 𐑢𐑫𐑛 𐑥𐑳𐑑𐑼 𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓—“𐑓 𐑩𐑯 𐑴𐑤𐑛 𐑒𐑨𐑐𐑑𐑦𐑯 𐑤𐑲𐑒 𐑥𐑰 𐑑 𐑚𐑰 𐑛𐑦𐑕𐑧𐑯𐑛𐑦𐑙 𐑞𐑦𐑕 𐑯𐑼𐑴 𐑕𐑒𐑳𐑑𐑩𐑤, 𐑑 𐑜𐑴 𐑑 𐑥𐑲 𐑜𐑮𐑱𐑝-𐑛𐑳𐑜 𐑚𐑻𐑔.”

𐑕𐑴, 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑝𐑮𐑦 𐑑𐑢𐑧𐑯𐑑𐑦-𐑓𐑹 𐑬𐑼𐑟, 𐑢𐑧𐑯 𐑞 𐑢𐑪𐑗𐑩𐑟 𐑝 𐑞 𐑯𐑲𐑑 𐑢𐑻 𐑕𐑧𐑑, 𐑯 𐑞 𐑚𐑨𐑯𐑛 𐑪𐑯 𐑛𐑧𐑒 𐑕𐑧𐑯𐑑𐑦𐑯𐑩𐑤𐑛 𐑞 𐑕𐑤𐑳𐑥𐑚𐑼𐑟 𐑝 𐑞 𐑚𐑨𐑯𐑛 𐑚𐑦𐑤𐑴; 𐑯 𐑢𐑧𐑯 𐑦𐑓 𐑩 𐑮𐑴𐑐 𐑢𐑪𐑟 𐑑 𐑚𐑰 𐑣𐑷𐑤𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤, 𐑞 𐑕𐑱𐑤𐑼𐑟 𐑓𐑤𐑳𐑙 𐑦𐑑 𐑯𐑪𐑑 𐑮𐑵𐑛𐑤𐑦 𐑛𐑬𐑯, 𐑨𐑟 𐑚𐑲 𐑛𐑱, 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑒𐑷𐑖𐑩𐑕𐑯𐑩𐑕 𐑛𐑮𐑪𐑐𐑑 𐑦𐑑 𐑑 𐑦𐑑𐑕 𐑐𐑤𐑱𐑕 𐑓 𐑓𐑽 𐑝 𐑛𐑦𐑕𐑑𐑻𐑚𐑦𐑙 𐑞𐑺 𐑕𐑤𐑳𐑥𐑚𐑼𐑦𐑙 𐑖𐑦𐑐𐑥𐑱𐑑𐑕; 𐑢𐑧𐑯 𐑞𐑦𐑕 𐑕𐑹𐑑 𐑝 𐑕𐑑𐑧𐑛𐑦 𐑒𐑢𐑲𐑩𐑑𐑵𐑛 𐑢𐑫𐑛 𐑚𐑦𐑜𐑦𐑯 𐑑 𐑐𐑮𐑦𐑝𐑱𐑤, 𐑣𐑩𐑚𐑦𐑗𐑵𐑩𐑤𐑦, 𐑞 𐑕𐑲𐑤𐑩𐑯𐑑 𐑕𐑑𐑽𐑟𐑥𐑩𐑯 𐑢𐑫𐑛 𐑢𐑪𐑗 𐑞 𐑒𐑨𐑚𐑦𐑯-𐑕𐑒𐑳𐑑𐑩𐑤; 𐑯 𐑺 𐑤𐑪𐑙 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑢𐑫𐑛 𐑦𐑥𐑻𐑡, 𐑜𐑮𐑦𐑐𐑦𐑙 𐑨𐑑 𐑞 𐑲𐑼𐑯 𐑚𐑨𐑯𐑦𐑕𐑑𐑼, 𐑑 𐑣𐑧𐑤𐑐 𐑣𐑦𐑟 𐑒𐑮𐑦𐑐𐑩𐑤𐑛 𐑢𐑱. 𐑕𐑳𐑥 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑦𐑙 𐑑𐑳𐑗 𐑝 𐑣𐑿𐑥𐑨𐑯𐑦𐑑𐑦 𐑢𐑪𐑟 𐑦𐑯 𐑣𐑦𐑥; 𐑓 𐑨𐑑 𐑑𐑲𐑥𐑟 𐑤𐑲𐑒 𐑞𐑰𐑟, 𐑣𐑰 𐑿𐑠𐑵𐑩𐑤𐑦 𐑩𐑚𐑕𐑑𐑱𐑯𐑛 𐑓𐑮𐑪𐑥 𐑐𐑩𐑑𐑮𐑴𐑤𐑦𐑙 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒; 𐑚𐑦𐑒𐑪𐑟 𐑑 𐑣𐑦𐑟 𐑢𐑽𐑦𐑛 𐑥𐑱𐑑𐑕, 𐑕𐑰𐑒𐑦𐑙 𐑮𐑦𐑐𐑴𐑟 𐑢𐑦𐑞𐑦𐑯 𐑕𐑦𐑒𐑕 𐑦𐑯𐑗𐑩𐑟 𐑝 𐑣𐑦𐑟 𐑲𐑝𐑼𐑦 𐑣𐑰𐑤, 𐑕𐑳𐑗 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑞 𐑮𐑦𐑝𐑻𐑚𐑼𐑱𐑑𐑦𐑙 𐑒𐑮𐑨𐑒 𐑯 𐑛𐑦𐑯 𐑝 𐑞𐑨𐑑 𐑚𐑴𐑯𐑦 𐑕𐑑𐑧𐑐, 𐑞𐑨𐑑 𐑞𐑺 𐑛𐑮𐑰𐑥𐑟 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑪𐑯 𐑞 𐑒𐑮𐑳𐑯𐑗𐑦𐑙 𐑑𐑰𐑔 𐑝 𐑖𐑸𐑒𐑕. 𐑚𐑳𐑑 𐑢𐑳𐑯𐑕, 𐑞 𐑥𐑵𐑛 𐑢𐑪𐑟 𐑪𐑯 𐑣𐑦𐑥 𐑑𐑵 𐑛𐑰𐑐 𐑓 𐑒𐑪𐑥𐑩𐑯 𐑮𐑦𐑜𐑸𐑛𐑦𐑙𐑟; 𐑯 𐑨𐑟 𐑢𐑦𐑞 𐑣𐑧𐑝𐑦, 𐑤𐑳𐑥𐑚𐑼-𐑤𐑲𐑒 𐑐𐑱𐑕 𐑣𐑰 𐑢𐑪𐑟 𐑥𐑧𐑠𐑼𐑦𐑙 𐑞 𐑖𐑦𐑐 𐑓𐑮𐑪𐑥 𐑑𐑨𐑓𐑮𐑱𐑤 𐑑 𐑥𐑱𐑯𐑥𐑭𐑕𐑑, ·𐑕𐑑𐑳𐑚, 𐑞 𐑴𐑤𐑛 𐑕𐑧𐑒𐑩𐑯𐑛 𐑥𐑱𐑑, 𐑒𐑱𐑥 𐑳𐑐 𐑓𐑮𐑪𐑥 𐑚𐑦𐑤𐑴, 𐑢𐑦𐑞 𐑩 𐑕𐑻𐑑𐑩𐑯 𐑳𐑯𐑩𐑖𐑫𐑼𐑛, 𐑛𐑧𐑐𐑮𐑩𐑒𐑱𐑑𐑦𐑙 𐑣𐑿𐑥𐑼𐑩𐑕𐑯𐑩𐑕, 𐑣𐑦𐑯𐑑𐑩𐑛 𐑞𐑨𐑑 𐑦𐑓 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑢𐑪𐑟 𐑐𐑤𐑰𐑟𐑛 𐑑 𐑢𐑷𐑒 𐑞 𐑐𐑤𐑨𐑙𐑒𐑕, 𐑞𐑧𐑯, 𐑯𐑴 𐑢𐑳𐑯 𐑒𐑫𐑛 𐑕𐑱 𐑯𐑱; 𐑚𐑳𐑑 𐑞𐑺 𐑥𐑲𐑑 𐑚𐑰 𐑕𐑳𐑥 𐑢𐑱 𐑝 𐑥𐑳𐑓𐑩𐑤𐑦𐑙 𐑞 𐑯𐑶𐑟; 𐑣𐑦𐑯𐑑𐑦𐑙 𐑕𐑳𐑥𐑔𐑦𐑙 𐑦𐑯𐑛𐑦𐑕𐑑𐑦𐑙𐑒𐑑𐑤𐑦 𐑯 𐑣𐑧𐑟𐑦𐑑𐑱𐑑𐑦𐑙𐑤𐑦 𐑩𐑚𐑬𐑑 𐑩 𐑜𐑤𐑴𐑚 𐑝 𐑑𐑴, 𐑯 𐑞 𐑦𐑯𐑕𐑻𐑖𐑩𐑯 𐑦𐑯𐑑𐑵 𐑦𐑑, 𐑝 𐑞 𐑲𐑝𐑼𐑦 𐑣𐑰𐑤. 𐑭! 𐑕𐑑𐑳𐑚, 𐑞𐑬 𐑛𐑦𐑛𐑕𐑑 𐑯𐑪𐑑 𐑯𐑴 ·𐑱𐑣𐑨𐑚 𐑞𐑧𐑯.

“𐑨𐑥 𐑲 𐑩 𐑒𐑨𐑯𐑩𐑯-𐑚𐑷𐑤, ·𐑕𐑑𐑳𐑚,” 𐑕𐑧𐑛 ·𐑱𐑣𐑨𐑚, “𐑞𐑨𐑑 𐑞𐑬 𐑢𐑫𐑛𐑕𐑑 𐑢𐑪𐑛 𐑥𐑰 𐑞𐑨𐑑 𐑓𐑨𐑖𐑩𐑯? 𐑚𐑳𐑑 𐑜𐑴 𐑞𐑲 𐑢𐑱𐑟; 𐑲 𐑣𐑨𐑛 𐑓𐑼𐑜𐑪𐑑. 𐑚𐑦𐑤𐑴 𐑑 𐑞𐑲 𐑯𐑲𐑑𐑤𐑦 𐑜𐑮𐑱𐑝; 𐑢𐑺 𐑕𐑳𐑗 𐑨𐑟 𐑘𐑰 𐑕𐑤𐑰𐑐 𐑚𐑦𐑑𐑢𐑰𐑯 𐑖𐑮𐑬𐑛𐑟, 𐑑 𐑿𐑟 𐑘𐑰 𐑑 𐑞 𐑓𐑦𐑤𐑦𐑙 𐑢𐑳𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑.—𐑛𐑬𐑯, 𐑛𐑪𐑜, 𐑯 𐑒𐑧𐑯𐑩𐑤!”

𐑕𐑑𐑸𐑑𐑦𐑙 𐑨𐑑 𐑞 𐑳𐑯𐑓𐑹𐑕𐑰𐑯 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑦𐑙 𐑧𐑒𐑕𐑒𐑤𐑩𐑥𐑱𐑖𐑩𐑯 𐑝 𐑞 𐑕𐑴 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑕𐑒𐑹𐑯𐑓𐑩𐑤 𐑴𐑤𐑛 𐑥𐑨𐑯, ·𐑕𐑑𐑳𐑚 𐑢𐑪𐑟 𐑕𐑐𐑰𐑗𐑤𐑩𐑕 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑; 𐑞𐑧𐑯 𐑕𐑧𐑛 𐑦𐑒𐑕𐑲𐑑𐑩𐑛𐑤𐑦, “𐑲 𐑨𐑥 𐑯𐑪𐑑 𐑿𐑟𐑛 𐑑 𐑚𐑰 𐑕𐑐𐑴𐑒𐑩𐑯 𐑑 𐑞𐑨𐑑 𐑢𐑱, 𐑕𐑻; 𐑲 𐑛𐑵 𐑚𐑳𐑑 𐑤𐑧𐑕 𐑞𐑨𐑯 𐑣𐑭𐑓 𐑤𐑲𐑒 𐑦𐑑, 𐑕𐑻.”

“𐑩𐑝𐑭𐑕𐑑! 𐑜𐑮𐑦𐑑𐑩𐑛 ·𐑱𐑣𐑨𐑚 𐑚𐑦𐑑𐑢𐑰𐑯 𐑣𐑦𐑟 𐑕𐑧𐑑 𐑑𐑰𐑔, 𐑯 𐑝𐑲𐑩𐑤𐑩𐑯𐑑𐑤𐑦 𐑥𐑵𐑝𐑦𐑙 𐑩𐑢𐑱, 𐑨𐑟 𐑦𐑓 𐑑 𐑩𐑝𐑶𐑛 𐑕𐑳𐑥 𐑐𐑨𐑖𐑩𐑯𐑩𐑑 𐑑𐑧𐑥𐑐𐑑𐑱𐑖𐑩𐑯.

“𐑯𐑴, 𐑕𐑻; 𐑯𐑪𐑑 𐑘𐑧𐑑,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑦𐑥𐑚𐑴𐑤𐑛𐑩𐑯𐑛, “𐑲 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑑𐑱𐑥𐑤𐑦 𐑚𐑰 𐑒𐑷𐑤𐑛 𐑩 𐑛𐑪𐑜, 𐑕𐑻.”

“𐑞𐑧𐑯 𐑚𐑰 𐑒𐑷𐑤𐑛 𐑑𐑧𐑯 𐑑𐑲𐑥𐑟 𐑩 𐑛𐑪𐑙𐑒𐑦, 𐑯 𐑩 𐑥𐑿𐑤, 𐑯 𐑩𐑯 𐑨𐑕, 𐑯 𐑚𐑦𐑜𐑪𐑯, 𐑹 𐑲'𐑤 𐑒𐑤𐑽 𐑞 𐑢𐑻𐑤𐑛 𐑝 𐑞𐑰!”

𐑨𐑟 𐑣𐑰 𐑕𐑧𐑛 𐑞𐑦𐑕, ·𐑱𐑣𐑨𐑚 𐑩𐑛𐑝𐑭𐑯𐑕𐑑 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑴𐑝𐑼𐑚𐑺𐑦𐑙 𐑑𐑧𐑮𐑼𐑟 𐑦𐑯 𐑣𐑦𐑟 𐑨𐑕𐑐𐑧𐑒𐑑, 𐑞𐑨𐑑 ·𐑕𐑑𐑳𐑚 𐑦𐑯𐑝𐑪𐑤𐑩𐑯𐑑𐑺𐑦𐑤𐑦 𐑮𐑦𐑑𐑮𐑰𐑑𐑩𐑛.

“𐑲 𐑢𐑪𐑟 𐑯𐑧𐑝𐑼 𐑕𐑻𐑝𐑛 𐑕𐑴 𐑚𐑦𐑓𐑹 𐑢𐑦𐑞𐑬𐑑 𐑜𐑦𐑝𐑦𐑙 𐑩 𐑣𐑸𐑛 𐑚𐑤𐑴 𐑓 𐑦𐑑,” 𐑥𐑳𐑑𐑼𐑛 ·𐑕𐑑𐑳𐑚, 𐑨𐑟 𐑣𐑰 𐑓𐑬𐑯𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑛𐑦𐑕𐑧𐑯𐑛𐑦𐑙 𐑞 𐑒𐑨𐑚𐑦𐑯-𐑕𐑒𐑳𐑑𐑩𐑤. “𐑦𐑑'𐑕 𐑝𐑧𐑮𐑦 𐑒𐑢𐑽. 𐑕𐑑𐑪𐑐, ·𐑕𐑑𐑳𐑚; 𐑕𐑳𐑥𐑣𐑬, 𐑯𐑬, 𐑲 𐑛𐑴𐑯𐑑 𐑢𐑧𐑤 𐑯𐑴 𐑢𐑧𐑞𐑼 𐑑 𐑜𐑴 𐑚𐑨𐑒 𐑯 𐑕𐑑𐑮𐑲𐑒 𐑣𐑦𐑥, 𐑹—𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑?—𐑛𐑬𐑯 𐑣𐑽 𐑪𐑯 𐑥𐑲 𐑯𐑰𐑟 𐑯 𐑐𐑮𐑱 𐑓 𐑣𐑦𐑥? 𐑘𐑧𐑕, 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑞 𐑔𐑷𐑑 𐑒𐑳𐑥𐑦𐑙 𐑳𐑐 𐑦𐑯 𐑥𐑰; 𐑚𐑳𐑑 𐑦𐑑 𐑢𐑫𐑛 𐑚𐑰 𐑞 𐑓𐑻𐑕𐑑 𐑑𐑲𐑥 𐑲 𐑧𐑝𐑼 𐑛𐑦𐑛 𐑐𐑮𐑱. 𐑦𐑑'𐑕 𐑒𐑢𐑽; 𐑝𐑧𐑮𐑦 𐑒𐑢𐑽; 𐑯 𐑣𐑰'𐑟 𐑒𐑢𐑽 𐑑𐑵; 𐑲, 𐑑𐑱𐑒 𐑣𐑦𐑥 𐑓𐑹 𐑯 𐑭𐑓𐑑, 𐑣𐑰'𐑟 𐑩𐑚𐑬𐑑 𐑞 𐑒𐑢𐑽𐑩𐑕𐑑 𐑴𐑤𐑛 𐑥𐑨𐑯 ·𐑕𐑑𐑳𐑚 𐑧𐑝𐑼 𐑕𐑱𐑤𐑛 𐑢𐑦𐑞. 𐑣𐑬 𐑣𐑰 𐑓𐑤𐑨𐑖𐑑 𐑨𐑑 𐑥𐑰!—𐑣𐑦𐑟 𐑲𐑟 𐑤𐑲𐑒 𐑐𐑬𐑛𐑼-𐑐𐑨𐑯𐑟! 𐑦𐑟 𐑣𐑰 𐑥𐑨𐑛? 𐑧𐑯𐑦𐑢𐑱 𐑞𐑺'𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑪𐑯 𐑣𐑦𐑟 𐑥𐑲𐑯𐑛, 𐑨𐑟 𐑖𐑫𐑼 𐑨𐑟 𐑞𐑺 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑕𐑳𐑥𐑔𐑦𐑙 𐑪𐑯 𐑩 𐑛𐑧𐑒 𐑢𐑧𐑯 𐑦𐑑 𐑒𐑮𐑨𐑒𐑕. 𐑣𐑰 𐑱𐑯𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑧𐑛 𐑯𐑬, 𐑲𐑞𐑼, 𐑥𐑹 𐑞𐑨𐑯 𐑔𐑮𐑰 𐑬𐑼𐑟 𐑬𐑑 𐑝 𐑞 𐑑𐑢𐑧𐑯𐑑𐑦-𐑓𐑹; 𐑯 𐑣𐑰 𐑛𐑴𐑯𐑑 𐑕𐑤𐑰𐑐 𐑞𐑧𐑯. 𐑛𐑦𐑛𐑩𐑯𐑑 𐑞𐑨𐑑 ·𐑛𐑴-·𐑚𐑶, 𐑞 𐑕𐑑𐑿𐑼𐑛, 𐑑𐑧𐑤 𐑥𐑰 𐑞𐑨𐑑 𐑝 𐑩 𐑥𐑹𐑯𐑦𐑙 𐑣𐑰 𐑷𐑤𐑢𐑱𐑟 𐑓𐑲𐑯𐑛𐑟 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑣𐑨𐑥𐑩𐑒 𐑒𐑤𐑴𐑞𐑟 𐑷𐑤 𐑮𐑳𐑥𐑐𐑩𐑤𐑛 𐑯 𐑑𐑳𐑥𐑚𐑩𐑤𐑛, 𐑯 𐑞 𐑖𐑰𐑑𐑕 𐑛𐑬𐑯 𐑨𐑑 𐑞 𐑓𐑫𐑑, 𐑯 𐑞 𐑒𐑳𐑝𐑼𐑤𐑦𐑛 𐑷𐑤𐑥𐑴𐑕𐑑 𐑑𐑲𐑛 𐑦𐑯𐑑𐑵 𐑯𐑪𐑑𐑕, 𐑯 𐑞 𐑐𐑦𐑤𐑴 𐑩 𐑕𐑹𐑑 𐑝 𐑓𐑮𐑲𐑑𐑓𐑩𐑤 𐑣𐑪𐑑, 𐑨𐑟 𐑞𐑴 𐑩 𐑚𐑱𐑒𐑑 𐑚𐑮𐑦𐑒 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑪𐑯 𐑦𐑑? 𐑩 𐑣𐑪𐑑 𐑴𐑤𐑛 𐑥𐑨𐑯! 𐑲 𐑜𐑧𐑕 𐑣𐑰'𐑟 𐑜𐑪𐑑 𐑢𐑪𐑑 𐑕𐑳𐑥 𐑓𐑴𐑒𐑕 𐑩𐑖𐑹 𐑒𐑷𐑤 𐑩 𐑒𐑪𐑯𐑖𐑩𐑯𐑕; 𐑦𐑑'𐑕 𐑩 𐑒𐑲𐑯𐑛 𐑝 ·𐑑𐑦𐑒-·𐑛𐑪𐑤𐑦-𐑮𐑴 𐑞𐑱 𐑕𐑱—𐑢𐑻𐑕 𐑯𐑹 𐑩 𐑑𐑵𐑔𐑱𐑒. 𐑢𐑧𐑤, 𐑢𐑧𐑤; 𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑢𐑪𐑑 𐑦𐑑 𐑦𐑟, 𐑚𐑳𐑑 𐑞 ·𐑤𐑹𐑛 𐑒𐑰𐑐 𐑥𐑰 𐑓𐑮𐑪𐑥 𐑒𐑨𐑗𐑦𐑙 𐑦𐑑. 𐑣𐑰'𐑟 𐑓𐑫𐑤 𐑝 𐑮𐑦𐑛𐑩𐑤𐑟; 𐑲 𐑢𐑳𐑯𐑛𐑼 𐑢𐑪𐑑 𐑣𐑰 𐑜𐑴𐑟 𐑦𐑯𐑑𐑵 𐑞 𐑭𐑓𐑑𐑼 𐑣𐑴𐑤𐑛 𐑓, 𐑧𐑝𐑮𐑦 𐑯𐑲𐑑, 𐑨𐑟 ·𐑛𐑴-·𐑚𐑶 𐑑𐑧𐑤𐑟 𐑥𐑰 𐑣𐑰 𐑕𐑩𐑕𐑐𐑧𐑒𐑑𐑕; 𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑓, 𐑲 𐑖𐑵𐑛 𐑤𐑲𐑒 𐑑 𐑯𐑴? 𐑣𐑵'𐑟 𐑥𐑱𐑛 𐑩𐑐𐑶𐑯𐑑𐑥𐑩𐑯𐑑𐑕 𐑢𐑦𐑞 𐑣𐑦𐑥 𐑦𐑯 𐑞 𐑣𐑴𐑤𐑛? 𐑱𐑯𐑑 𐑞𐑨𐑑 𐑒𐑢𐑽, 𐑯𐑬? 𐑚𐑳𐑑 𐑞𐑺'𐑟 𐑯𐑴 𐑑𐑧𐑤𐑦𐑙, 𐑦𐑑'𐑕 𐑞 𐑴𐑤𐑛 𐑜𐑱𐑥—·𐑣𐑽 𐑜𐑴𐑟 𐑓 𐑩 𐑕𐑯𐑵𐑟. 𐑛𐑨𐑥 𐑥𐑰, 𐑦𐑑'𐑕 𐑢𐑻𐑔 𐑩 𐑓𐑧𐑤𐑴'𐑟 𐑢𐑲𐑤 𐑑 𐑚𐑰 𐑚𐑹𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑻𐑤𐑛, 𐑦𐑓 𐑴𐑯𐑤𐑦 𐑑 𐑓𐑷𐑤 𐑮𐑲𐑑 𐑩𐑕𐑤𐑰𐑐. 𐑯 𐑯𐑬 𐑞𐑨𐑑 𐑲 𐑔𐑦𐑙𐑒 𐑝 𐑦𐑑, 𐑞𐑨𐑑'𐑕 𐑩𐑚𐑬𐑑 𐑞 𐑓𐑻𐑕𐑑 𐑔𐑦𐑙 𐑚𐑱𐑚𐑦𐑟 𐑛𐑵, 𐑯 𐑞𐑨𐑑'𐑕 𐑩 𐑕𐑹𐑑 𐑝 𐑒𐑢𐑽, 𐑑𐑵. 𐑛𐑨𐑥 𐑥𐑰, 𐑚𐑳𐑑 𐑷𐑤 𐑔𐑦𐑙𐑟 𐑸 𐑒𐑢𐑽, 𐑒𐑳𐑥 𐑑 𐑔𐑦𐑙𐑒 𐑝 '𐑩𐑥. 𐑚𐑳𐑑 𐑞𐑨𐑑'𐑕 𐑩𐑜𐑱𐑯𐑕𐑑 𐑥𐑲 𐑐𐑮𐑦𐑯𐑕𐑩𐑐𐑩𐑤𐑟. 𐑔𐑦𐑙𐑒 𐑯𐑪𐑑, 𐑦𐑟 𐑥𐑲 𐑦𐑤𐑧𐑝𐑩𐑯𐑔 𐑒𐑩𐑥𐑭𐑯𐑛𐑥𐑩𐑯𐑑; 𐑯 𐑕𐑤𐑰𐑐 𐑢𐑧𐑯 𐑿 𐑒𐑨𐑯, 𐑦𐑟 𐑥𐑲 𐑑𐑢𐑧𐑤𐑓𐑔—𐑕𐑴 𐑣𐑽 𐑜𐑴𐑟 𐑩𐑜𐑱𐑯. 𐑚𐑳𐑑 𐑣𐑬'𐑟 𐑞𐑨𐑑? 𐑛𐑦𐑛𐑩𐑯𐑑 𐑣𐑰 𐑒𐑷𐑤 𐑥𐑰 𐑩 𐑛𐑪𐑜? 𐑚𐑤𐑱𐑟𐑩𐑟! 𐑣𐑰 𐑒𐑷𐑤𐑛 𐑥𐑰 𐑑𐑧𐑯 𐑑𐑲𐑥𐑟 𐑩 𐑛𐑪𐑙𐑒𐑦, 𐑯 𐑐𐑲𐑤𐑛 𐑩 𐑤𐑪𐑑 𐑝 𐑡𐑨𐑒𐑨𐑕𐑩𐑟 𐑪𐑯 𐑑𐑪𐑐 𐑝 𐑞𐑨𐑑! 𐑣𐑰 𐑥𐑲𐑑 𐑨𐑟 𐑢𐑧𐑤 𐑣𐑨𐑝 𐑒𐑦𐑒𐑑 𐑥𐑰, 𐑯 𐑛𐑳𐑯 𐑢𐑦𐑞 𐑦𐑑. 𐑥𐑱𐑚𐑰 𐑣𐑰 𐑛𐑦𐑛 𐑒𐑦𐑒 𐑥𐑰, 𐑯 𐑲 𐑛𐑦𐑛𐑩𐑯𐑑 𐑩𐑚𐑟𐑻𐑝 𐑦𐑑, 𐑲 𐑢𐑪𐑟 𐑕𐑴 𐑑𐑱𐑒𐑩𐑯 𐑷𐑤 𐑩𐑚𐑨𐑒 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑚𐑮𐑬, 𐑕𐑳𐑥𐑣𐑬. 𐑦𐑑 𐑓𐑤𐑨𐑖𐑑 𐑤𐑲𐑒 𐑩 𐑚𐑤𐑰𐑗𐑑 𐑚𐑴𐑯. 𐑢𐑪𐑑 𐑞 𐑛𐑧𐑝𐑩𐑤'𐑟 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑥𐑰? 𐑲 𐑛𐑴𐑯𐑑 𐑕𐑑𐑨𐑯𐑛 𐑮𐑲𐑑 𐑪𐑯 𐑥𐑲 𐑤𐑧𐑜𐑟. 𐑒𐑳𐑥𐑦𐑙 𐑩𐑓𐑬𐑤 𐑝 𐑞𐑨𐑑 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑣𐑨𐑟 𐑩 𐑕𐑹𐑑 𐑝 𐑑𐑻𐑯𐑛 𐑥𐑰 𐑮𐑪𐑙 𐑕𐑲𐑛 𐑬𐑑. 𐑚𐑲 𐑞 ·𐑤𐑹𐑛, 𐑲 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑛𐑮𐑰𐑥𐑦𐑙, 𐑞𐑴—𐑣𐑬? 𐑣𐑬? 𐑣𐑬?—𐑚𐑳𐑑 𐑞 𐑴𐑯𐑤𐑦 𐑢𐑱'𐑟 𐑑 𐑕𐑑𐑨𐑖 𐑦𐑑; 𐑕𐑴 𐑣𐑽 𐑜𐑴𐑟 𐑑 𐑣𐑨𐑥𐑩𐑒 𐑩𐑜𐑱𐑯; 𐑯 𐑦𐑯 𐑞 𐑥𐑹𐑯𐑦𐑙, 𐑲'𐑤 𐑕𐑰 𐑣𐑬 𐑞𐑦𐑕 𐑐𐑤𐑱𐑜𐑦 𐑡𐑳𐑜𐑩𐑤𐑦𐑙 𐑔𐑦𐑙𐑒𐑕 𐑴𐑝𐑼 𐑚𐑲 𐑛𐑱𐑤𐑲𐑑.”
