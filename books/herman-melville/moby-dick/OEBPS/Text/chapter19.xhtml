·𐑗𐑨𐑐𐑑𐑼 19. 𐑞 ·𐑐𐑮𐑪𐑓𐑩𐑑.

“𐑖𐑦𐑐𐑥𐑱𐑑𐑕, 𐑣𐑨𐑝 𐑘𐑰 𐑖𐑦𐑐𐑑 𐑦𐑯 𐑞𐑨𐑑 𐑖𐑦𐑐?”

·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑯 𐑲 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑤𐑧𐑓𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑯 𐑢𐑻 𐑕𐑷𐑯𐑑𐑼𐑦𐑙 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑷𐑑𐑼, 𐑓 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑 𐑰𐑗 𐑪𐑒𐑿𐑐𐑲𐑛 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑴𐑯 𐑔𐑷𐑑𐑕, 𐑢𐑧𐑯 𐑞 𐑩𐑚𐑳𐑝 𐑢𐑻𐑛𐑟 𐑢𐑻 𐑐𐑫𐑑 𐑑 𐑳𐑕 𐑚𐑲 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, 𐑣𐑵, 𐑐𐑷𐑟𐑦𐑙 𐑚𐑦𐑓𐑹 𐑳𐑕, 𐑤𐑧𐑝𐑩𐑤𐑛 𐑣𐑦𐑟 𐑥𐑨𐑕𐑦𐑝 𐑓𐑹𐑓𐑦𐑙𐑜𐑼 𐑨𐑑 𐑞 𐑝𐑧𐑕𐑩𐑤 𐑦𐑯 𐑒𐑢𐑧𐑕𐑗𐑩𐑯. 𐑣𐑰 𐑢𐑪𐑟 𐑚𐑳𐑑 𐑖𐑨𐑚𐑦𐑤𐑦 𐑩𐑐𐑨𐑮𐑩𐑤𐑛 𐑦𐑯 𐑓𐑱𐑛𐑩𐑛 𐑡𐑨𐑒𐑩𐑑 𐑯 𐑐𐑨𐑗𐑑 𐑑𐑮𐑴𐑟𐑼𐑟; 𐑩 𐑮𐑨𐑜 𐑝 𐑩 𐑚𐑤𐑨𐑒 𐑣𐑨𐑙𐑒𐑼𐑗𐑦𐑓 𐑦𐑯𐑝𐑧𐑕𐑑𐑦𐑙 𐑣𐑦𐑟 𐑯𐑧𐑒. 𐑩 𐑒𐑪𐑯𐑓𐑤𐑵𐑩𐑯𐑑 𐑕𐑥𐑷𐑤-𐑐𐑪𐑒𐑕 𐑣𐑨𐑛 𐑦𐑯 𐑷𐑤 𐑛𐑦𐑮𐑧𐑒𐑖𐑩𐑯𐑟 𐑓𐑤𐑴𐑛 𐑴𐑝𐑼 𐑣𐑦𐑟 𐑓𐑱𐑕, 𐑯 𐑤𐑧𐑓𐑑 𐑦𐑑 𐑤𐑲𐑒 𐑞 𐑒𐑪𐑥𐑐𐑤𐑦𐑒𐑱𐑑𐑩𐑛 𐑮𐑦𐑚𐑛 𐑚𐑧𐑛 𐑝 𐑩 𐑑𐑪𐑮𐑩𐑯𐑑, 𐑢𐑧𐑯 𐑞 𐑮𐑳𐑖𐑦𐑙 𐑢𐑷𐑑𐑼𐑟 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑛𐑮𐑲𐑛 𐑳𐑐.

“𐑣𐑨𐑝 𐑘𐑰 𐑖𐑦𐑐𐑑 𐑦𐑯 𐑣𐑻?” 𐑣𐑰 𐑮𐑦𐑐𐑰𐑑𐑩𐑛.

“𐑿 𐑥𐑰𐑯 𐑞 𐑖𐑦𐑐 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑲 𐑕𐑩𐑐𐑴𐑟,” 𐑕𐑧𐑛 𐑲, 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑜𐑱𐑯 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑥𐑹 𐑑𐑲𐑥 𐑓 𐑩𐑯 𐑳𐑯𐑦𐑯𐑑𐑼𐑳𐑐𐑑𐑩𐑛 𐑤𐑫𐑒 𐑨𐑑 𐑣𐑦𐑥.

“𐑲, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛—𐑞𐑨𐑑 𐑖𐑦𐑐 𐑞𐑺,” 𐑣𐑰 𐑕𐑧𐑛, 𐑛𐑮𐑷𐑦𐑙 𐑚𐑨𐑒 𐑣𐑦𐑟 𐑣𐑴𐑤 𐑸𐑥, 𐑯 𐑞𐑧𐑯 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑖𐑳𐑝𐑦𐑙 𐑦𐑑 𐑕𐑑𐑮𐑱𐑑 𐑬𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑢𐑦𐑞 𐑞 𐑓𐑦𐑒𐑕𐑑 𐑚𐑱𐑩𐑯𐑧𐑑 𐑝 𐑣𐑦𐑟 𐑐𐑶𐑯𐑑𐑩𐑛 𐑓𐑦𐑙𐑜𐑼 𐑛𐑸𐑑𐑩𐑛 𐑓𐑫𐑤 𐑨𐑑 𐑞 𐑪𐑚𐑡𐑧𐑒𐑑.

“𐑘𐑧𐑕,” 𐑕𐑧𐑛 𐑲, “𐑢𐑰 𐑣𐑨𐑝 𐑡𐑳𐑕𐑑 𐑕𐑲𐑯𐑛 𐑞 𐑸𐑑𐑦𐑒𐑩𐑤𐑟.”

“𐑧𐑯𐑦𐑔𐑦𐑙 𐑛𐑬𐑯 𐑞𐑺 𐑩𐑚𐑬𐑑 𐑘𐑹 𐑕𐑴𐑤𐑟?”

“𐑩𐑚𐑬𐑑 𐑢𐑪𐑑?”

“𐑴, 𐑐𐑼𐑣𐑨𐑐𐑕 𐑿 𐑙𐑨𐑝'𐑩𐑯𐑑 𐑜𐑪𐑑 𐑧𐑯𐑦,” 𐑣𐑰 𐑕𐑧𐑛 𐑒𐑢𐑦𐑒𐑤𐑦. “𐑯𐑴 𐑥𐑨𐑑𐑼 𐑞𐑴, 𐑲 𐑯𐑴 𐑥𐑧𐑯𐑦 𐑗𐑨𐑐𐑕 𐑞𐑨𐑑 𐑙𐑨𐑝'𐑩𐑯𐑑 𐑜𐑪𐑑 𐑧𐑯𐑦,—𐑜𐑫𐑛 𐑤𐑳𐑒 𐑑 '𐑩𐑥; 𐑯 𐑞𐑱 𐑸 𐑷𐑤 𐑞 𐑚𐑧𐑑𐑼 𐑪𐑓 𐑓 𐑦𐑑. 𐑩 𐑕𐑴𐑤'𐑟 𐑩 𐑕𐑹𐑑 𐑝 𐑩 𐑓𐑦𐑓𐑔 𐑢𐑰𐑤 𐑑 𐑩 𐑢𐑨𐑜𐑩𐑯.”

“𐑢𐑪𐑑 𐑸 𐑿 𐑡𐑨𐑚𐑼𐑦𐑙 𐑩𐑚𐑬𐑑, 𐑖𐑦𐑐𐑥𐑱𐑑?” 𐑕𐑧𐑛 𐑲.

“𐑣𐑰'𐑟 𐑜𐑪𐑑 𐑦𐑯𐑳𐑓, 𐑞𐑴, 𐑑 𐑥𐑱𐑒 𐑳𐑐 𐑓 𐑷𐑤 𐑛𐑦𐑓𐑦𐑖𐑩𐑯𐑕𐑦𐑟 𐑝 𐑞𐑨𐑑 𐑕𐑹𐑑 𐑦𐑯 𐑳𐑞𐑼 𐑗𐑨𐑐𐑕,” 𐑩𐑚𐑮𐑳𐑐𐑑𐑤𐑦 𐑕𐑧𐑛 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, 𐑐𐑤𐑱𐑕𐑦𐑙 𐑩 𐑯𐑻𐑝𐑩𐑕 𐑧𐑥𐑓𐑩𐑕𐑦𐑕 𐑩𐑐𐑪𐑯 𐑞 𐑢𐑻𐑛 𐑣𐑰.

“·𐑒𐑢𐑰𐑒𐑢𐑧𐑜,” 𐑕𐑧𐑛 𐑲, “𐑤𐑧𐑑'𐑕 𐑜𐑴; 𐑞𐑦𐑕 𐑓𐑧𐑤𐑴 𐑣𐑨𐑟 𐑚𐑮𐑴𐑒𐑩𐑯 𐑤𐑵𐑕 𐑓𐑮𐑪𐑥 𐑕𐑳𐑥𐑢𐑺; 𐑣𐑰'𐑟 𐑑𐑷𐑒𐑦𐑙 𐑩𐑚𐑬𐑑 𐑕𐑳𐑥𐑔𐑦𐑙 𐑯 𐑕𐑳𐑥𐑚𐑪𐑛𐑦 𐑢𐑰 𐑛𐑴𐑯𐑑 𐑯𐑴.”

“𐑕𐑑𐑪𐑐!” 𐑒𐑮𐑲𐑛 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼. “𐑘𐑰 𐑕𐑧𐑛 𐑑𐑮𐑵—𐑘𐑰 𐑙𐑨𐑝'𐑩𐑯𐑑 𐑕𐑰𐑯 ·𐑴𐑤𐑛 ·𐑔𐑳𐑯𐑛𐑼 𐑘𐑧𐑑, 𐑣𐑨𐑝 𐑘𐑰?”

“𐑣𐑵'𐑟 ·𐑴𐑤𐑛 ·𐑔𐑳𐑯𐑛𐑼?” 𐑕𐑧𐑛 𐑲, 𐑩𐑜𐑱𐑯 𐑮𐑦𐑝𐑩𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞 𐑦𐑯𐑕𐑱𐑯 𐑻𐑯𐑩𐑕𐑑𐑯𐑩𐑕 𐑝 𐑣𐑦𐑟 𐑥𐑨𐑯𐑼.

“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚.”

“𐑢𐑪𐑑! 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑝 𐑬𐑼 𐑖𐑦𐑐, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛?”

“𐑲, 𐑩𐑥𐑳𐑙 𐑕𐑳𐑥 𐑝 𐑳𐑕 𐑴𐑤𐑛 𐑕𐑱𐑤𐑼 𐑗𐑨𐑐𐑕, 𐑣𐑰 𐑜𐑴𐑟 𐑚𐑲 𐑞𐑨𐑑 𐑯𐑱𐑥. 𐑘𐑰 𐑙𐑨𐑝'𐑩𐑯𐑑 𐑕𐑰𐑯 𐑣𐑦𐑥 𐑘𐑧𐑑, 𐑣𐑨𐑝 𐑘𐑰?”

“𐑯𐑴, 𐑢𐑰 𐑙𐑨𐑝'𐑩𐑯𐑑. 𐑣𐑰'𐑟 𐑕𐑦𐑒 𐑞𐑱 𐑕𐑱, 𐑚𐑳𐑑 𐑦𐑟 𐑜𐑧𐑑𐑦𐑙 𐑚𐑧𐑑𐑼, 𐑯 𐑢𐑦𐑤 𐑚𐑰 𐑷𐑤 𐑮𐑲𐑑 𐑩𐑜𐑱𐑯 𐑚𐑦𐑓𐑹 𐑤𐑪𐑙.”

“𐑷𐑤 𐑮𐑲𐑑 𐑩𐑜𐑱𐑯 𐑚𐑦𐑓𐑹 𐑤𐑪𐑙!” 𐑤𐑭𐑓𐑑 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, 𐑢𐑦𐑞 𐑩 𐑕𐑪𐑤𐑩𐑥𐑤𐑦 𐑛𐑦𐑮𐑲𐑕𐑦𐑝 𐑕𐑹𐑑 𐑝 𐑤𐑭𐑓. “𐑤𐑫𐑒 𐑘𐑰; 𐑢𐑧𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑷𐑤 𐑮𐑲𐑑, 𐑞𐑧𐑯 𐑞𐑦𐑕 𐑤𐑧𐑓𐑑 𐑸𐑥 𐑝 𐑥𐑲𐑯 𐑢𐑦𐑤 𐑚𐑰 𐑷𐑤 𐑮𐑲𐑑; 𐑯𐑪𐑑 𐑚𐑦𐑓𐑹.”

“𐑢𐑪𐑑 𐑛𐑵 𐑿 𐑯𐑴 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥?”

“𐑢𐑪𐑑 𐑛𐑦𐑛 𐑞𐑱 𐑑𐑧𐑤 𐑿 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥? 𐑕𐑱 𐑞𐑨𐑑!”

“𐑞𐑱 𐑛𐑦𐑛𐑩𐑯𐑑 𐑑𐑧𐑤 𐑥𐑳𐑗 𐑝 𐑧𐑯𐑦𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥; 𐑴𐑯𐑤𐑦 𐑲'𐑝 𐑣𐑻𐑛 𐑞𐑨𐑑 𐑣𐑰'𐑟 𐑩 𐑜𐑫𐑛 𐑢𐑱𐑤-𐑣𐑳𐑯𐑑𐑼, 𐑯 𐑩 𐑜𐑫𐑛 𐑒𐑨𐑐𐑑𐑦𐑯 𐑑 𐑣𐑦𐑟 𐑒𐑮𐑵.”

“𐑞𐑨𐑑'𐑕 𐑑𐑮𐑵, 𐑞𐑨𐑑'𐑕 𐑑𐑮𐑵—𐑘𐑧𐑕, 𐑚𐑴𐑔 𐑑𐑮𐑵 𐑦𐑯𐑳𐑓. 𐑚𐑳𐑑 𐑿 𐑥𐑳𐑕𐑑 𐑡𐑳𐑥𐑐 𐑢𐑧𐑯 𐑣𐑰 𐑜𐑦𐑝𐑟 𐑩𐑯 𐑹𐑛𐑼. 𐑕𐑑𐑧𐑐 𐑯 𐑜𐑮𐑬𐑤; 𐑜𐑮𐑬𐑤 𐑯 𐑜𐑴—𐑞𐑨𐑑'𐑕 𐑞 𐑢𐑻𐑛 𐑢𐑦𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚. 𐑚𐑳𐑑 𐑯𐑳𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑 𐑔𐑦𐑙 𐑞𐑨𐑑 𐑣𐑨𐑐𐑩𐑯𐑛 𐑑 𐑣𐑦𐑥 𐑪𐑓 ·𐑒𐑱𐑐 ·𐑣𐑹𐑯, 𐑤𐑪𐑙 𐑩𐑜𐑴, 𐑢𐑧𐑯 𐑣𐑰 𐑤𐑱 𐑤𐑲𐑒 𐑛𐑧𐑛 𐑓 𐑔𐑮𐑰 𐑛𐑱𐑟 𐑯 𐑯𐑲𐑑𐑕; 𐑯𐑳𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑 𐑛𐑧𐑛𐑤𐑦 𐑕𐑒𐑮𐑦𐑥𐑦𐑡 𐑢𐑦𐑞 𐑞 ·𐑕𐑐𐑨𐑯𐑘𐑼𐑛 𐑩𐑓𐑹 𐑞 𐑷𐑤𐑑𐑼 𐑦𐑯 ·𐑕𐑨𐑯𐑑𐑩?—𐑣𐑻𐑛 𐑯𐑳𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑, 𐑱? 𐑯𐑳𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞 𐑕𐑦𐑤𐑝𐑼 𐑒𐑨𐑤𐑩𐑚𐑨𐑖 𐑣𐑰 𐑕𐑐𐑨𐑑 𐑦𐑯𐑑𐑵? 𐑯 𐑯𐑳𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑣𐑦𐑟 𐑤𐑵𐑟𐑦𐑙 𐑣𐑦𐑟 𐑤𐑧𐑜 𐑤𐑭𐑕𐑑 𐑝𐑶𐑦𐑡, 𐑩𐑒𐑹𐑛𐑦𐑙 𐑑 𐑞 𐑐𐑮𐑪𐑓𐑩𐑕𐑦. 𐑛𐑦𐑛𐑩𐑯𐑑 𐑘𐑰 𐑣𐑽 𐑩 𐑢𐑻𐑛 𐑩𐑚𐑬𐑑 𐑞𐑧𐑥 𐑥𐑨𐑑𐑼𐑟 𐑯 𐑕𐑳𐑥𐑔𐑦𐑙 𐑥𐑹, 𐑱? 𐑯𐑴, 𐑲 𐑛𐑴𐑯𐑑 𐑔𐑦𐑙𐑒 𐑘𐑰 𐑛𐑦𐑛; 𐑣𐑬 𐑒𐑫𐑛 𐑘𐑰? 𐑣𐑵 𐑯𐑴𐑟 𐑦𐑑? 𐑯𐑪𐑑 𐑷𐑤 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑, 𐑲 𐑜𐑧𐑕. 𐑚𐑳𐑑 𐑣𐑬𐑟'𐑧𐑝𐑼, 𐑥𐑱𐑣𐑨𐑐, 𐑘𐑰'𐑝 𐑣𐑻𐑛 𐑑𐑧𐑤 𐑩𐑚𐑬𐑑 𐑞 𐑤𐑧𐑜, 𐑯 𐑣𐑬 𐑣𐑰 𐑤𐑪𐑕𐑑 𐑦𐑑; 𐑲, 𐑘𐑰 𐑣𐑨𐑝 𐑣𐑻𐑛 𐑝 𐑞𐑨𐑑, 𐑲 𐑛𐑺 𐑕𐑱. 𐑴 𐑘𐑧𐑕, 𐑞𐑨𐑑 𐑧𐑝𐑮𐑦 𐑢𐑳𐑯 𐑯𐑴𐑟 𐑩'𐑯𐑴𐑕𐑑—𐑲 𐑥𐑰𐑯 𐑞𐑱 𐑯𐑴 𐑣𐑰'𐑟 𐑴𐑯𐑤𐑦 𐑢𐑳𐑯 𐑤𐑧𐑜; 𐑯 𐑞𐑨𐑑 𐑩 ·𐑐𐑸𐑥𐑩𐑕𐑧𐑑𐑦 𐑑𐑫𐑒 𐑞 𐑳𐑞𐑼 𐑪𐑓.”

“𐑥𐑲 𐑓𐑮𐑧𐑯𐑛,” 𐑕𐑧𐑛 𐑲, “𐑢𐑪𐑑 𐑷𐑤 𐑞𐑦𐑕 𐑡𐑦𐑚𐑼𐑦𐑖 𐑝 𐑘𐑹𐑟 𐑦𐑟 𐑩𐑚𐑬𐑑, 𐑲 𐑛𐑴𐑯𐑑 𐑯𐑴, 𐑯 𐑲 𐑛𐑴𐑯𐑑 𐑥𐑳𐑗 𐑒𐑺; 𐑓 𐑦𐑑 𐑕𐑰𐑥𐑟 𐑑 𐑥𐑰 𐑞𐑨𐑑 𐑿 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑛𐑨𐑥𐑦𐑡𐑛 𐑦𐑯 𐑞 𐑣𐑧𐑛. 𐑚𐑳𐑑 𐑦𐑓 𐑿 𐑸 𐑕𐑐𐑰𐑒𐑦𐑙 𐑝 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑝 𐑞𐑨𐑑 𐑖𐑦𐑐 𐑞𐑺, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑞𐑧𐑯 𐑤𐑧𐑑 𐑥𐑰 𐑑𐑧𐑤 𐑿, 𐑞𐑨𐑑 𐑲 𐑯𐑴 𐑷𐑤 𐑩𐑚𐑬𐑑 𐑞 𐑤𐑪𐑕 𐑝 𐑣𐑦𐑟 𐑤𐑧𐑜.”

“𐑷𐑤 𐑩𐑚𐑬𐑑 𐑦𐑑, 𐑱—𐑖𐑫𐑼 𐑿 𐑛𐑵?—𐑷𐑤?”

“𐑐𐑮𐑦𐑑𐑦 𐑖𐑫𐑼.”

𐑢𐑦𐑞 𐑓𐑦𐑙𐑜𐑼 𐑐𐑶𐑯𐑑𐑩𐑛 𐑯 𐑲 𐑤𐑧𐑝𐑩𐑤𐑛 𐑨𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑞 𐑚𐑧𐑜𐑼-𐑤𐑲𐑒 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑕𐑑𐑫𐑛 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑨𐑟 𐑦𐑓 𐑦𐑯 𐑩 𐑑𐑮𐑳𐑚𐑩𐑤𐑛 𐑮𐑧𐑝𐑼𐑦; 𐑞𐑧𐑯 𐑕𐑑𐑸𐑑𐑦𐑙 𐑩 𐑤𐑦𐑑𐑩𐑤, 𐑑𐑻𐑯𐑛 𐑯 𐑕𐑧𐑛:—“𐑘𐑰'𐑝 𐑖𐑦𐑐𐑑, 𐑣𐑨𐑝 𐑘𐑰? 𐑯𐑱𐑥𐑟 𐑛𐑬𐑯 𐑪𐑯 𐑞 𐑐𐑱𐑐𐑼𐑟? 𐑢𐑧𐑤, 𐑢𐑧𐑤, 𐑢𐑪𐑑'𐑕 𐑕𐑲𐑯𐑛, 𐑦𐑟 𐑕𐑲𐑯𐑛; 𐑯 𐑢𐑪𐑑'𐑕 𐑑 𐑚𐑰, 𐑢𐑦𐑤 𐑚𐑰; 𐑯 𐑞𐑧𐑯 𐑩𐑜𐑱𐑯, 𐑐𐑼𐑣𐑨𐑐𐑕 𐑦𐑑 𐑢𐑴𐑯𐑑 𐑚𐑰, 𐑭𐑓𐑑𐑼 𐑷𐑤. 𐑧𐑯𐑦𐑣𐑬, 𐑦𐑑'𐑕 𐑷𐑤 𐑓𐑦𐑒𐑕𐑑 𐑯 𐑼𐑱𐑯𐑡𐑛 𐑩'𐑮𐑧𐑛𐑦; 𐑯 𐑕𐑳𐑥 𐑕𐑱𐑤𐑼𐑟 𐑹 𐑳𐑞𐑼 𐑥𐑳𐑕𐑑 𐑜𐑴 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑲 𐑕𐑩𐑐𐑴𐑟; 𐑨𐑟 𐑢𐑧𐑤 𐑞𐑰𐑟 𐑨𐑟 𐑧𐑯𐑦 𐑳𐑞𐑼 𐑥𐑧𐑯, ·𐑜𐑪𐑛 𐑐𐑦𐑑𐑦 '𐑩𐑥! 𐑥𐑹𐑯𐑦𐑙 𐑑 𐑘𐑰, 𐑖𐑦𐑐𐑥𐑱𐑑𐑕, 𐑥𐑹𐑯𐑦𐑙; 𐑞 𐑦𐑯𐑧𐑓𐑩𐑚𐑩𐑤 𐑣𐑧𐑝𐑩𐑯𐑟 𐑚𐑤𐑧𐑕 𐑘𐑰; 𐑲'𐑥 𐑕𐑪𐑮𐑦 𐑲 𐑕𐑑𐑪𐑐𐑑 𐑘𐑰.”

“𐑤𐑫𐑒 𐑣𐑽, 𐑓𐑮𐑧𐑯𐑛,” 𐑕𐑧𐑛 𐑲, “𐑦𐑓 𐑿 𐑣𐑨𐑝 𐑧𐑯𐑦𐑔𐑦𐑙 𐑦𐑥𐑐𐑹𐑑𐑩𐑯𐑑 𐑑 𐑑𐑧𐑤 𐑳𐑕, 𐑬𐑑 𐑢𐑦𐑞 𐑦𐑑; 𐑚𐑳𐑑 𐑦𐑓 𐑿 𐑸 𐑴𐑯𐑤𐑦 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑚𐑨𐑥𐑚𐑵𐑟𐑩𐑤 𐑳𐑕, 𐑿 𐑸 𐑥𐑦𐑕𐑑𐑱𐑒𐑩𐑯 𐑦𐑯 𐑘𐑹 𐑜𐑱𐑥; 𐑞𐑨𐑑'𐑕 𐑷𐑤 𐑲 𐑣𐑨𐑓 𐑑 𐑕𐑱.”

“𐑯 𐑦𐑑'𐑕 𐑕𐑧𐑛 𐑝𐑧𐑮𐑦 𐑢𐑧𐑤, 𐑯 𐑲 𐑤𐑲𐑒 𐑑 𐑣𐑽 𐑩 𐑗𐑨𐑐 𐑑𐑷𐑒 𐑳𐑐 𐑞𐑨𐑑 𐑢𐑱; 𐑿 𐑸 𐑡𐑳𐑕𐑑 𐑞 𐑥𐑨𐑯 𐑓 𐑣𐑦𐑥—𐑞 𐑤𐑲𐑒𐑕 𐑝 𐑘𐑰. 𐑥𐑹𐑯𐑦𐑙 𐑑 𐑘𐑰, 𐑖𐑦𐑐𐑥𐑱𐑑𐑕, 𐑥𐑹𐑯𐑦𐑙! 𐑴! 𐑢𐑧𐑯 𐑘𐑰 𐑜𐑧𐑑 𐑞𐑺, 𐑑𐑧𐑤 '𐑩𐑥 𐑲'𐑝 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑯𐑪𐑑 𐑑 𐑥𐑱𐑒 𐑢𐑳𐑯 𐑝 '𐑩𐑥.”

“𐑭, 𐑥𐑲 𐑛𐑽 𐑓𐑧𐑤𐑴, 𐑿 𐑒𐑭𐑯𐑑 𐑓𐑵𐑤 𐑳𐑕 𐑞𐑨𐑑 𐑢𐑱—𐑿 𐑒𐑭𐑯𐑑 𐑓𐑵𐑤 𐑳𐑕. 𐑦𐑑 𐑦𐑟 𐑞 𐑰𐑟𐑾𐑕𐑑 𐑔𐑦𐑙 𐑦𐑯 𐑞 𐑢𐑻𐑤𐑛 𐑓 𐑩 𐑥𐑨𐑯 𐑑 𐑤𐑫𐑒 𐑨𐑟 𐑦𐑓 𐑣𐑰 𐑣𐑨𐑛 𐑩 𐑜𐑮𐑱𐑑 𐑕𐑰𐑒𐑮𐑩𐑑 𐑦𐑯 𐑣𐑦𐑥.”

“𐑥𐑹𐑯𐑦𐑙 𐑑 𐑘𐑰, 𐑖𐑦𐑐𐑥𐑱𐑑𐑕, 𐑥𐑹𐑯𐑦𐑙.”

“𐑥𐑹𐑯𐑦𐑙 𐑦𐑑 𐑦𐑟,” 𐑕𐑧𐑛 𐑲. “𐑒𐑳𐑥 𐑩𐑤𐑪𐑙, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑤𐑧𐑑'𐑕 𐑤𐑰𐑝 𐑞𐑦𐑕 𐑒𐑮𐑱𐑟𐑦 𐑥𐑨𐑯. 𐑚𐑳𐑑 𐑕𐑑𐑪𐑐, 𐑑𐑧𐑤 𐑥𐑰 𐑘𐑹 𐑯𐑱𐑥, 𐑢𐑦𐑤 𐑿?”

“·𐑦𐑤𐑲𐑡𐑩.”

·𐑦𐑤𐑲𐑡𐑩! 𐑔𐑷𐑑 𐑲, 𐑯 𐑢𐑰 𐑢𐑷𐑒𐑑 𐑩𐑢𐑱, 𐑚𐑴𐑔 𐑒𐑪𐑥𐑧𐑯𐑑𐑦𐑙, 𐑭𐑓𐑑𐑼 𐑰𐑗 𐑳𐑞𐑼'𐑟 𐑓𐑨𐑖𐑩𐑯, 𐑩𐑐𐑪𐑯 𐑞𐑦𐑕 𐑮𐑨𐑜𐑩𐑛 𐑴𐑤𐑛 𐑕𐑱𐑤𐑼; 𐑯 𐑩𐑜𐑮𐑰𐑛 𐑞𐑨𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑯𐑳𐑔𐑦𐑙 𐑚𐑳𐑑 𐑩 𐑣𐑳𐑥𐑚𐑳𐑜, 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑚𐑰 𐑩 𐑚𐑳𐑜𐑚𐑺. 𐑚𐑳𐑑 𐑢𐑰 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑜𐑪𐑯 𐑐𐑼𐑣𐑨𐑐𐑕 𐑩𐑚𐑳𐑝 𐑩 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑘𐑸𐑛𐑟, 𐑢𐑧𐑯 𐑗𐑭𐑯𐑕𐑦𐑙 𐑑 𐑑𐑻𐑯 𐑩 𐑒𐑹𐑯𐑼, 𐑯 𐑤𐑫𐑒𐑦𐑙 𐑚𐑨𐑒 𐑨𐑟 𐑲 𐑛𐑦𐑛 𐑕𐑴, 𐑣𐑵 𐑖𐑵𐑛 𐑚𐑰 𐑕𐑰𐑯 𐑚𐑳𐑑 ·𐑦𐑤𐑲𐑡𐑩 𐑓𐑪𐑤𐑴𐑦𐑙 𐑳𐑕, 𐑞𐑴 𐑨𐑑 𐑩 𐑛𐑦𐑕𐑑𐑩𐑯𐑕. 𐑕𐑳𐑥𐑣𐑬, 𐑞 𐑕𐑲𐑑 𐑝 𐑣𐑦𐑥 𐑕𐑑𐑮𐑳𐑒 𐑥𐑰 𐑕𐑴, 𐑞𐑨𐑑 𐑲 𐑕𐑧𐑛 𐑯𐑳𐑔𐑦𐑙 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑝 𐑣𐑦𐑟 𐑚𐑰𐑦𐑙 𐑚𐑦𐑣𐑲𐑯𐑛, 𐑚𐑳𐑑 𐑐𐑭𐑕𐑑 𐑪𐑯 𐑢𐑦𐑞 𐑥𐑲 𐑒𐑪𐑥𐑮𐑱𐑛, 𐑨𐑙𐑒𐑖𐑩𐑕 𐑑 𐑕𐑰 𐑢𐑧𐑞𐑼 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑢𐑫𐑛 𐑑𐑻𐑯 𐑞 𐑕𐑱𐑥 𐑒𐑹𐑯𐑼 𐑞𐑨𐑑 𐑢𐑰 𐑛𐑦𐑛. 𐑣𐑰 𐑛𐑦𐑛; 𐑯 𐑞𐑧𐑯 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑑 𐑥𐑰 𐑞𐑨𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑛𐑪𐑜𐑦𐑙 𐑳𐑕, 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑢𐑪𐑑 𐑦𐑯𐑑𐑧𐑯𐑑 𐑲 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑓 𐑞 𐑤𐑲𐑓 𐑝 𐑥𐑰 𐑦𐑥𐑨𐑡𐑦𐑯. 𐑞𐑦𐑕 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕, 𐑒𐑳𐑐𐑩𐑤𐑛 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑨𐑥𐑚𐑦𐑜𐑿𐑩𐑕, 𐑣𐑭𐑓-𐑣𐑦𐑯𐑑𐑦𐑙, 𐑣𐑭𐑓-𐑮𐑦𐑝𐑰𐑤𐑦𐑙, 𐑖𐑮𐑬𐑛𐑩𐑛 𐑕𐑹𐑑 𐑝 𐑑𐑷𐑒, 𐑯𐑬 𐑚𐑦𐑜𐑨𐑑 𐑦𐑯 𐑥𐑰 𐑷𐑤 𐑒𐑲𐑯𐑛𐑟 𐑝 𐑝𐑱𐑜 𐑢𐑳𐑯𐑛𐑼𐑥𐑩𐑯𐑑𐑕 𐑯 𐑣𐑭𐑓-𐑨𐑐𐑮𐑦𐑣𐑧𐑯𐑖𐑩𐑯𐑟, 𐑯 𐑷𐑤 𐑒𐑩𐑯𐑧𐑒𐑑𐑩𐑛 𐑢𐑦𐑞 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛; 𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚; 𐑯 𐑞 𐑤𐑧𐑜 𐑣𐑰 𐑣𐑨𐑛 𐑤𐑪𐑕𐑑; 𐑯 𐑞 ·𐑒𐑱𐑐 ·𐑣𐑹𐑯 𐑓𐑦𐑑; 𐑯 𐑞 𐑕𐑦𐑤𐑝𐑼 𐑒𐑨𐑤𐑩𐑚𐑨𐑖; 𐑯 𐑢𐑪𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑐𐑰𐑤𐑧𐑜 𐑣𐑨𐑛 𐑕𐑧𐑛 𐑝 𐑣𐑦𐑥, 𐑢𐑧𐑯 𐑲 𐑤𐑧𐑓𐑑 𐑞 𐑖𐑦𐑐 𐑞 𐑛𐑱 𐑐𐑮𐑰𐑝𐑾𐑕; 𐑯 𐑞 𐑐𐑮𐑦𐑛𐑦𐑒𐑖𐑩𐑯 𐑝 𐑞 𐑕𐑒𐑢𐑷 ·𐑑𐑦𐑕𐑑𐑨𐑜; 𐑯 𐑞 𐑝𐑶𐑦𐑡 𐑢𐑰 𐑣𐑨𐑛 𐑚𐑬𐑯𐑛 𐑬𐑼𐑕𐑧𐑤𐑝𐑟 𐑑 𐑕𐑱𐑤; 𐑯 𐑩 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑳𐑞𐑼 𐑖𐑨𐑛𐑴𐑦 𐑔𐑦𐑙𐑟.

𐑲 𐑢𐑪𐑟 𐑮𐑦𐑟𐑪𐑤𐑝𐑛 𐑑 𐑕𐑨𐑑𐑦𐑕𐑓𐑲 𐑥𐑲𐑕𐑧𐑤𐑓 𐑢𐑧𐑞𐑼 𐑞𐑦𐑕 𐑮𐑨𐑜𐑩𐑛 ·𐑦𐑤𐑲𐑡𐑩 𐑢𐑪𐑟 𐑮𐑾𐑤𐑦 𐑛𐑪𐑜𐑦𐑙 𐑳𐑕 𐑹 𐑯𐑪𐑑, 𐑯 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑦𐑯𐑑𐑧𐑯𐑑 𐑒𐑮𐑪𐑕𐑑 𐑞 𐑢𐑱 𐑢𐑦𐑞 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑯 𐑪𐑯 𐑞𐑨𐑑 𐑕𐑲𐑛 𐑝 𐑦𐑑 𐑮𐑰𐑑𐑮𐑱𐑕𐑑 𐑬𐑼 𐑕𐑑𐑧𐑐𐑕. 𐑚𐑳𐑑 ·𐑦𐑤𐑲𐑡𐑩 𐑐𐑭𐑕𐑑 𐑪𐑯, 𐑢𐑦𐑞𐑬𐑑 𐑕𐑰𐑥𐑦𐑙 𐑑 𐑯𐑴𐑑𐑦𐑕 𐑳𐑕. 𐑞𐑦𐑕 𐑮𐑦𐑤𐑰𐑝𐑛 𐑥𐑰; 𐑯 𐑢𐑳𐑯𐑕 𐑥𐑹, 𐑯 𐑓𐑲𐑯𐑩𐑤𐑦 𐑨𐑟 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑑 𐑥𐑰, 𐑲 𐑐𐑮𐑩𐑯𐑬𐑯𐑕𐑑 𐑣𐑦𐑥 𐑦𐑯 𐑥𐑲 𐑣𐑸𐑑, 𐑩 𐑣𐑳𐑥𐑚𐑳𐑜.
