·𐑗𐑨𐑐𐑑𐑼 58. ·𐑚𐑮𐑦𐑑.

𐑕𐑑𐑽𐑦𐑙 𐑯𐑹𐑔-𐑰𐑕𐑑𐑢𐑼𐑛 𐑓𐑮𐑪𐑥 𐑞 ·𐑒𐑮𐑴𐑟𐑧𐑑𐑕, 𐑢𐑰 𐑓𐑧𐑤 𐑦𐑯 𐑢𐑦𐑞 𐑝𐑭𐑕𐑑 𐑥𐑧𐑛𐑴𐑟 𐑝 ·𐑚𐑮𐑦𐑑, 𐑞 𐑥𐑦𐑯𐑦𐑑, 𐑘𐑧𐑤𐑴 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕, 𐑩𐑐𐑪𐑯 𐑢𐑦𐑗 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑤𐑸𐑡𐑤𐑦 𐑓𐑰𐑛𐑟. 𐑓 𐑤𐑰𐑜𐑟 𐑯 𐑤𐑰𐑜𐑟 𐑦𐑑 𐑳𐑯𐑡𐑩𐑤𐑱𐑑𐑩𐑛 𐑮𐑬𐑯𐑛 𐑳𐑕, 𐑕𐑴 𐑞𐑨𐑑 𐑢𐑰 𐑕𐑰𐑥𐑛 𐑑 𐑚𐑰 𐑕𐑱𐑤𐑦𐑙 𐑔𐑮𐑵 𐑚𐑬𐑯𐑛𐑤𐑩𐑕 𐑓𐑰𐑤𐑛𐑟 𐑝 𐑮𐑲𐑐 𐑯 𐑜𐑴𐑤𐑛𐑩𐑯 𐑢𐑰𐑑.

𐑪𐑯 𐑞 𐑕𐑧𐑒𐑩𐑯𐑛 𐑛𐑱, 𐑯𐑳𐑥𐑚𐑼𐑟 𐑝 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤𐑟 𐑢𐑻 𐑕𐑰𐑯, 𐑣𐑵, 𐑕𐑦𐑒𐑘𐑫𐑼 𐑓𐑮𐑪𐑥 𐑞 𐑩𐑑𐑨𐑒 𐑝 𐑩 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤𐑼 𐑤𐑲𐑒 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑢𐑦𐑞 𐑴𐑐𐑩𐑯 𐑡𐑷𐑟 𐑕𐑤𐑳𐑜𐑦𐑖𐑤𐑦 𐑕𐑢𐑨𐑥 𐑔𐑮𐑵 𐑞 ·𐑚𐑮𐑦𐑑, 𐑢𐑦𐑗, 𐑨𐑛𐑣𐑽𐑦𐑙 𐑑 𐑞 𐑓𐑮𐑦𐑯𐑡𐑦𐑙 𐑓𐑲𐑚𐑼𐑟 𐑝 𐑞𐑨𐑑 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 ·𐑝𐑩𐑯𐑰𐑖𐑩𐑯 𐑚𐑤𐑲𐑯𐑛 𐑦𐑯 𐑞𐑺 𐑥𐑬𐑞𐑟, 𐑢𐑪𐑟 𐑦𐑯 𐑞𐑨𐑑 𐑥𐑨𐑯𐑼 𐑕𐑧𐑐𐑼𐑱𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑢𐑷𐑑𐑼 𐑞𐑨𐑑 𐑦𐑕𐑒𐑱𐑐𐑑 𐑨𐑑 𐑞 𐑤𐑦𐑐.

𐑨𐑟 𐑥𐑹𐑯𐑦𐑙 𐑥𐑴𐑼𐑟, 𐑣𐑵 𐑕𐑲𐑛 𐑚𐑲 𐑕𐑲𐑛 𐑕𐑤𐑴𐑤𐑦 𐑯 𐑕𐑰𐑞𐑦𐑙𐑤𐑦 𐑩𐑛𐑝𐑭𐑯𐑕 𐑞𐑺 𐑕𐑲𐑞𐑟 𐑔𐑮𐑵 𐑞 𐑤𐑪𐑙 𐑢𐑧𐑑 𐑜𐑮𐑭𐑕 𐑝 𐑥𐑸𐑖𐑦 𐑥𐑰𐑛𐑟; 𐑰𐑝𐑩𐑯 𐑕𐑴 𐑞𐑰𐑟 𐑥𐑪𐑯𐑕𐑑𐑼𐑟 𐑕𐑢𐑨𐑥, 𐑥𐑱𐑒𐑦𐑙 𐑩 𐑕𐑑𐑮𐑱𐑯𐑡, 𐑜𐑮𐑭𐑕𐑦, 𐑒𐑳𐑑𐑦𐑙 𐑕𐑬𐑯𐑛; 𐑯 𐑤𐑰𐑝𐑦𐑙 𐑚𐑦𐑣𐑲𐑯𐑛 𐑞𐑧𐑥 𐑧𐑯𐑛𐑤𐑩𐑕 𐑕𐑢𐑭𐑔𐑕 𐑝 𐑚𐑤𐑵 𐑩𐑐𐑪𐑯 𐑞 𐑘𐑧𐑤𐑴 𐑕𐑰.*

*𐑞𐑨𐑑 𐑐𐑸𐑑 𐑝 𐑞 𐑕𐑰 𐑯𐑴𐑯 𐑩𐑥𐑳𐑙 𐑢𐑱𐑤𐑥𐑧𐑯 𐑨𐑟 𐑞 “·𐑚𐑮𐑩𐑟𐑦𐑤 ·𐑚𐑨𐑙𐑒𐑕” 𐑛𐑳𐑟 𐑯𐑪𐑑 𐑚𐑺 𐑞𐑨𐑑 𐑯𐑱𐑥 𐑨𐑟 𐑞 ·𐑚𐑨𐑙𐑒𐑕 𐑝 ·𐑯𐑿𐑓𐑩𐑯𐑤𐑩𐑯𐑛 𐑛𐑵, 𐑚𐑦𐑒𐑪𐑟 𐑝 𐑞𐑺 𐑚𐑰𐑦𐑙 𐑖𐑨𐑤𐑴𐑟 𐑯 𐑕𐑬𐑯𐑛𐑦𐑙𐑟 𐑞𐑺, 𐑚𐑳𐑑 𐑚𐑦𐑒𐑪𐑟 𐑝 𐑞𐑦𐑕 𐑮𐑦𐑥𐑸𐑒𐑩𐑚𐑩𐑤 𐑥𐑧𐑛𐑴-𐑤𐑲𐑒 𐑩𐑐𐑽𐑩𐑯𐑕, 𐑒𐑷𐑟𐑛 𐑚𐑲 𐑞 𐑝𐑭𐑕𐑑 𐑛𐑮𐑦𐑓𐑑𐑕 𐑝 ·𐑚𐑮𐑦𐑑 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤𐑦 𐑓𐑤𐑴𐑑𐑦𐑙 𐑦𐑯 𐑞𐑴𐑟 𐑤𐑨𐑑𐑦𐑑𐑿𐑛𐑟, 𐑢𐑺 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑦𐑟 𐑪𐑓𐑩𐑯 𐑗𐑱𐑕𐑑.

𐑚𐑳𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑞 𐑕𐑬𐑯𐑛 𐑞𐑱 𐑥𐑱𐑛 𐑨𐑟 𐑞𐑱 𐑐𐑸𐑑𐑩𐑛 𐑞 ·𐑚𐑮𐑦𐑑 𐑢𐑦𐑗 𐑨𐑑 𐑷𐑤 𐑮𐑦𐑥𐑲𐑯𐑛𐑩𐑛 𐑢𐑳𐑯 𐑝 𐑥𐑴𐑼𐑟. 𐑕𐑰𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟, 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑢𐑧𐑯 𐑞𐑱 𐑐𐑷𐑟𐑛 𐑯 𐑢𐑻 𐑕𐑑𐑱𐑖𐑩𐑯𐑼𐑦 𐑓 𐑩 𐑢𐑲𐑤, 𐑞𐑺 𐑝𐑭𐑕𐑑 𐑚𐑤𐑨𐑒 𐑓𐑹𐑥𐑟 𐑤𐑫𐑒𐑑 𐑥𐑹 𐑤𐑲𐑒 𐑤𐑲𐑓𐑤𐑩𐑕 𐑥𐑨𐑕𐑩𐑟 𐑝 𐑮𐑪𐑒 𐑞𐑨𐑯 𐑧𐑯𐑦𐑔𐑦𐑙 𐑧𐑤𐑕. 𐑯 𐑨𐑟 𐑦𐑯 𐑞 𐑜𐑮𐑱𐑑 𐑣𐑳𐑯𐑑𐑦𐑙 𐑒𐑳𐑯𐑑𐑮𐑦𐑟 𐑝 ·𐑦𐑯𐑛𐑾, 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑨𐑑 𐑩 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑢𐑦𐑤 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑐𐑭𐑕 𐑪𐑯 𐑞 𐑐𐑤𐑱𐑯𐑟 𐑮𐑦𐑒𐑳𐑥𐑚𐑩𐑯𐑑 𐑧𐑤𐑦𐑓𐑩𐑯𐑑𐑕 𐑢𐑦𐑞𐑬𐑑 𐑯𐑴𐑦𐑙 𐑞𐑧𐑥 𐑑 𐑚𐑰 𐑕𐑳𐑗, 𐑑𐑱𐑒𐑦𐑙 𐑞𐑧𐑥 𐑓 𐑚𐑺, 𐑚𐑤𐑨𐑒𐑩𐑯𐑛 𐑧𐑤𐑦𐑝𐑱𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑕𐑶𐑤; 𐑰𐑝𐑩𐑯 𐑕𐑴, 𐑪𐑓𐑩𐑯, 𐑢𐑦𐑞 𐑣𐑦𐑥, 𐑣𐑵 𐑓 𐑞 𐑓𐑻𐑕𐑑 𐑑𐑲𐑥 𐑚𐑦𐑣𐑴𐑤𐑛𐑟 𐑞𐑦𐑕 𐑕𐑐𐑰𐑖𐑰𐑟 𐑝 𐑞 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯𐑟 𐑝 𐑞 𐑕𐑰. 𐑯 𐑰𐑝𐑩𐑯 𐑢𐑧𐑯 𐑮𐑧𐑒𐑩𐑜𐑯𐑲𐑟𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑, 𐑞𐑺 𐑦𐑥𐑧𐑯𐑕 𐑥𐑨𐑜𐑯𐑦𐑑𐑿𐑛 𐑮𐑧𐑯𐑛𐑼𐑟 𐑦𐑑 𐑝𐑧𐑮𐑦 𐑣𐑸𐑛 𐑮𐑾𐑤𐑦 𐑑 𐑚𐑦𐑤𐑰𐑝 𐑞𐑨𐑑 𐑕𐑳𐑗 𐑚𐑳𐑤𐑒𐑦 𐑥𐑨𐑕𐑩𐑟 𐑝 𐑴𐑝𐑼𐑜𐑮𐑴𐑔 𐑒𐑨𐑯 𐑐𐑪𐑕𐑩𐑚𐑤𐑦 𐑚𐑰 𐑦𐑯𐑕𐑑𐑦𐑙𐑒𐑑, 𐑦𐑯 𐑷𐑤 𐑐𐑸𐑑𐑕, 𐑢𐑦𐑞 𐑞 𐑕𐑱𐑥 𐑕𐑹𐑑 𐑝 𐑤𐑲𐑓 𐑞𐑨𐑑 𐑤𐑦𐑝𐑟 𐑦𐑯 𐑩 𐑛𐑪𐑜 𐑹 𐑩 𐑣𐑹𐑕.

𐑦𐑯𐑛𐑰𐑛, 𐑦𐑯 𐑳𐑞𐑼 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑕, 𐑿 𐑒𐑨𐑯 𐑣𐑸𐑛𐑤𐑦 𐑮𐑦𐑜𐑸𐑛 𐑧𐑯𐑦 𐑒𐑮𐑰𐑗𐑼𐑟 𐑝 𐑞 𐑛𐑰𐑐 𐑢𐑦𐑞 𐑞 𐑕𐑱𐑥 𐑓𐑰𐑤𐑦𐑙𐑟 𐑞𐑨𐑑 𐑿 𐑛𐑵 𐑞𐑴𐑟 𐑝 𐑞 𐑖𐑹. 𐑓 𐑞𐑴 𐑕𐑳𐑥 𐑴𐑤𐑛 𐑯𐑨𐑗𐑼𐑩𐑤𐑦𐑕𐑑𐑕 𐑣𐑨𐑝 𐑥𐑱𐑯𐑑𐑱𐑯𐑛 𐑞𐑨𐑑 𐑷𐑤 𐑒𐑮𐑰𐑗𐑼𐑟 𐑝 𐑞 𐑤𐑨𐑯𐑛 𐑸 𐑝 𐑞𐑺 𐑒𐑲𐑯𐑛 𐑦𐑯 𐑞 𐑕𐑰; 𐑯 𐑞𐑴 𐑑𐑱𐑒𐑦𐑙 𐑩 𐑚𐑮𐑷𐑛 𐑡𐑧𐑯𐑼𐑩𐑤 𐑝𐑿 𐑝 𐑞 𐑔𐑦𐑙, 𐑞𐑦𐑕 𐑥𐑱 𐑝𐑧𐑮𐑦 𐑢𐑧𐑤 𐑚𐑰; 𐑘𐑧𐑑 𐑒𐑳𐑥𐑦𐑙 𐑑 𐑕𐑐𐑧𐑖𐑩𐑤𐑑𐑦𐑟, 𐑢𐑺, 𐑓 𐑦𐑜𐑟𐑭𐑥𐑐𐑩𐑤, 𐑛𐑳𐑟 𐑞 𐑴𐑖𐑩𐑯 𐑓𐑻𐑯𐑦𐑖 𐑧𐑯𐑦 𐑓𐑦𐑖 𐑞𐑨𐑑 𐑦𐑯 𐑛𐑦𐑕𐑐𐑩𐑟𐑦𐑖𐑩𐑯 𐑭𐑯𐑕𐑼𐑟 𐑑 𐑞 𐑕𐑩𐑜𐑱𐑖𐑩𐑕 𐑒𐑲𐑯𐑛𐑯𐑩𐑕 𐑝 𐑞 𐑛𐑪𐑜? 𐑞 𐑩𐑒𐑻𐑕𐑩𐑛 𐑖𐑸𐑒 𐑩𐑤𐑴𐑯 𐑒𐑨𐑯 𐑦𐑯 𐑧𐑯𐑦 𐑡𐑦𐑯𐑧𐑮𐑦𐑒 𐑮𐑦𐑕𐑐𐑧𐑒𐑑 𐑚𐑰 𐑕𐑧𐑛 𐑑 𐑚𐑺 𐑒𐑩𐑥𐑐𐑨𐑮𐑩𐑑𐑦𐑝 𐑩𐑯𐑨𐑤𐑩𐑡𐑦 𐑑 𐑣𐑦𐑥.

𐑚𐑳𐑑 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑦𐑟 𐑞 𐑕𐑰 𐑕𐑳𐑗 𐑩 𐑓𐑴 𐑑 𐑥𐑨𐑯 𐑣𐑵 𐑦𐑟 𐑩𐑯 𐑱𐑤𐑾𐑯 𐑑 𐑦𐑑, 𐑚𐑳𐑑 𐑦𐑑 𐑦𐑟 𐑷𐑤𐑕𐑴 𐑩 𐑓𐑰𐑯𐑛 𐑑 𐑦𐑑𐑕 𐑴𐑯 𐑪𐑓-𐑕𐑐𐑮𐑦𐑙; 𐑢𐑻𐑕 𐑞𐑨𐑯 𐑞 ·𐑐𐑻𐑖𐑩𐑯 𐑣𐑴𐑕𐑑 𐑣𐑵 𐑥𐑻𐑛𐑼𐑛 𐑣𐑦𐑟 𐑴𐑯 𐑜𐑧𐑕𐑑𐑕; 𐑕𐑐𐑺𐑦𐑙 𐑯𐑪𐑑 𐑞 𐑒𐑮𐑰𐑗𐑼𐑟 𐑢𐑦𐑗 𐑦𐑑𐑕𐑧𐑤𐑓 𐑣𐑨𐑔 𐑕𐑐𐑷𐑯𐑛. 𐑤𐑲𐑒 𐑩 𐑕𐑨𐑝𐑦𐑡 𐑑𐑲𐑜𐑮𐑩𐑕 𐑞𐑨𐑑 𐑑𐑪𐑕𐑦𐑙 𐑦𐑯 𐑞 𐑡𐑳𐑙𐑜𐑩𐑤 𐑴𐑝𐑼𐑤𐑱𐑟 𐑣𐑻 𐑴𐑯 𐑒𐑳𐑚𐑟, 𐑕𐑴 𐑞 𐑕𐑰 𐑛𐑨𐑖𐑩𐑟 𐑰𐑝𐑩𐑯 𐑞 𐑥𐑲𐑑𐑾𐑕𐑑 𐑢𐑱𐑤𐑟 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑮𐑪𐑒𐑕, 𐑯 𐑤𐑰𐑝𐑟 𐑞𐑧𐑥 𐑞𐑺 𐑕𐑲𐑛 𐑚𐑲 𐑕𐑲𐑛 𐑢𐑦𐑞 𐑞 𐑕𐑐𐑤𐑦𐑑 𐑮𐑧𐑒𐑕 𐑝 𐑖𐑦𐑐𐑕. 𐑯𐑴 𐑥𐑻𐑕𐑦, 𐑯𐑴 𐑐𐑬𐑼 𐑚𐑳𐑑 𐑦𐑑𐑕 𐑴𐑯 𐑒𐑩𐑯𐑑𐑮𐑴𐑤𐑟 𐑦𐑑. 𐑐𐑨𐑯𐑑𐑦𐑙 𐑯 𐑕𐑯𐑹𐑑𐑦𐑙 𐑤𐑲𐑒 𐑩 𐑥𐑨𐑛 𐑚𐑨𐑑𐑩𐑤 𐑕𐑑𐑰𐑛 𐑞𐑨𐑑 𐑣𐑨𐑟 𐑤𐑪𐑕𐑑 𐑦𐑑𐑕 𐑮𐑲𐑛𐑼, 𐑞 𐑥𐑭𐑕𐑑𐑼𐑤𐑩𐑕 𐑴𐑖𐑩𐑯 𐑴𐑝𐑼𐑮𐑳𐑯𐑟 𐑞 𐑜𐑤𐑴𐑚.
