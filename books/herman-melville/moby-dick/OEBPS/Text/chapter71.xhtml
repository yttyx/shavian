·𐑗𐑨𐑐𐑑𐑼 71. 𐑞 ·𐑡𐑧𐑮𐑩𐑚𐑴𐑩𐑥'𐑟 ·𐑕𐑑𐑹𐑦.

𐑣𐑨𐑯𐑛 𐑦𐑯 𐑣𐑨𐑯𐑛, 𐑖𐑦𐑐 𐑯 𐑚𐑮𐑰𐑟 𐑚𐑤𐑵 𐑪𐑯; 𐑚𐑳𐑑 𐑞 𐑚𐑮𐑰𐑟 𐑒𐑱𐑥 𐑓𐑭𐑕𐑑𐑼 𐑞𐑨𐑯 𐑞 𐑖𐑦𐑐, 𐑯 𐑕𐑵𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑮𐑪𐑒.

𐑚𐑲 𐑯 𐑚𐑲, 𐑔𐑮𐑵 𐑞 𐑜𐑤𐑭𐑕 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼'𐑟 𐑚𐑴𐑑𐑕 𐑯 𐑥𐑨𐑯𐑛 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑐𐑮𐑵𐑝𐑛 𐑣𐑻 𐑩 𐑢𐑱𐑤-𐑖𐑦𐑐. 𐑚𐑳𐑑 𐑨𐑟 𐑖𐑰 𐑢𐑪𐑟 𐑕𐑴 𐑓𐑸 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛, 𐑯 𐑖𐑵𐑑𐑦𐑙 𐑚𐑲, 𐑩𐑐𐑨𐑮𐑩𐑯𐑑𐑤𐑦 𐑥𐑱𐑒𐑦𐑙 𐑩 𐑐𐑨𐑕𐑦𐑡 𐑑 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑜𐑮𐑬𐑯𐑛, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑣𐑴𐑐 𐑑 𐑮𐑰𐑗 𐑣𐑻. 𐑕𐑴 𐑞 𐑕𐑦𐑜𐑯𐑩𐑤 𐑢𐑪𐑟 𐑕𐑧𐑑 𐑑 𐑕𐑰 𐑢𐑪𐑑 𐑮𐑦𐑕𐑐𐑪𐑯𐑕 𐑢𐑫𐑛 𐑚𐑰 𐑥𐑱𐑛.

𐑣𐑽 𐑚𐑰 𐑦𐑑 𐑕𐑧𐑛, 𐑞𐑨𐑑 𐑤𐑲𐑒 𐑞 𐑝𐑧𐑕𐑩𐑤𐑟 𐑝 𐑥𐑦𐑤𐑦𐑑𐑺𐑦 𐑥𐑼𐑰𐑯𐑟, 𐑞 𐑖𐑦𐑐𐑕 𐑝 𐑞 ·𐑩𐑥𐑧𐑮𐑦𐑒𐑩𐑯 ·𐑢𐑱𐑤 ·𐑓𐑤𐑰𐑑 𐑣𐑨𐑝 𐑰𐑗 𐑩 𐑐𐑮𐑲𐑝𐑩𐑑 𐑕𐑦𐑜𐑯𐑩𐑤; 𐑷𐑤 𐑢𐑦𐑗 𐑕𐑦𐑜𐑯𐑩𐑤𐑟 𐑚𐑰𐑦𐑙 𐑒𐑩𐑤𐑧𐑒𐑑𐑩𐑛 𐑦𐑯 𐑩 𐑚𐑫𐑒 𐑢𐑦𐑞 𐑞 𐑯𐑱𐑥𐑟 𐑝 𐑞 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑦𐑝 𐑝𐑧𐑕𐑩𐑤𐑟 𐑩𐑑𐑨𐑗𐑑, 𐑧𐑝𐑮𐑦 𐑒𐑨𐑐𐑑𐑦𐑯 𐑦𐑟 𐑐𐑮𐑩𐑝𐑲𐑛𐑩𐑛 𐑢𐑦𐑞 𐑦𐑑. 𐑞𐑺𐑚𐑲, 𐑞 𐑢𐑱𐑤 𐑒𐑩𐑥𐑭𐑯𐑛𐑼𐑟 𐑸 𐑦𐑯𐑱𐑚𐑩𐑤𐑛 𐑑 𐑮𐑧𐑒𐑩𐑜𐑯𐑲𐑟 𐑰𐑗 𐑳𐑞𐑼 𐑩𐑐𐑪𐑯 𐑞 𐑴𐑖𐑩𐑯, 𐑰𐑝𐑩𐑯 𐑨𐑑 𐑒𐑩𐑯𐑕𐑦𐑛𐑼𐑩𐑚𐑩𐑤 𐑛𐑦𐑕𐑑𐑩𐑯𐑕𐑩𐑟 𐑯 𐑢𐑦𐑞 𐑯𐑴 𐑕𐑥𐑷𐑤 𐑓𐑩𐑕𐑦𐑤𐑦𐑑𐑦.

𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑕𐑦𐑜𐑯𐑩𐑤 𐑢𐑪𐑟 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑮𐑦𐑕𐑐𐑪𐑯𐑛𐑩𐑛 𐑑 𐑚𐑲 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼'𐑟 𐑕𐑧𐑑𐑦𐑙 𐑣𐑻 𐑴𐑯; 𐑢𐑦𐑗 𐑐𐑮𐑵𐑝𐑛 𐑞 𐑖𐑦𐑐 𐑑 𐑚𐑰 𐑞 ·𐑡𐑧𐑮𐑩𐑚𐑴𐑩𐑥 𐑝 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑. 𐑕𐑒𐑢𐑺𐑦𐑙 𐑣𐑻 𐑘𐑸𐑛𐑟, 𐑖𐑰 𐑚𐑹 𐑛𐑬𐑯, 𐑮𐑱𐑯𐑡𐑛 𐑩𐑚𐑰𐑥 𐑳𐑯𐑛𐑼 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑤𐑰, 𐑯 𐑤𐑴𐑼𐑛 𐑩 𐑚𐑴𐑑; 𐑦𐑑 𐑕𐑵𐑯 𐑛𐑮𐑵 𐑯𐑲; 𐑚𐑳𐑑, 𐑨𐑟 𐑞 𐑕𐑲𐑛-𐑤𐑨𐑛𐑼 𐑢𐑪𐑟 𐑚𐑰𐑦𐑙 𐑮𐑦𐑜𐑛 𐑚𐑲 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑹𐑛𐑼 𐑑 𐑩𐑒𐑪𐑥𐑩𐑛𐑱𐑑 𐑞 𐑝𐑦𐑟𐑦𐑑𐑦𐑙 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑦𐑯 𐑒𐑢𐑧𐑕𐑗𐑩𐑯 𐑢𐑱𐑝𐑛 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑕𐑑𐑻𐑯 𐑦𐑯 𐑑𐑴𐑒𐑩𐑯 𐑝 𐑞𐑨𐑑 𐑐𐑮𐑴𐑕𐑰𐑛𐑦𐑙 𐑚𐑰𐑦𐑙 𐑦𐑯𐑑𐑲𐑼𐑤𐑦 𐑳𐑯𐑯𐑧𐑕𐑦𐑕𐑺𐑦. 𐑦𐑑 𐑑𐑻𐑯𐑛 𐑬𐑑 𐑞𐑨𐑑 𐑞 ·𐑡𐑧𐑮𐑩𐑚𐑴𐑩𐑥 𐑣𐑨𐑛 𐑩 𐑥𐑩𐑤𐑦𐑜𐑯𐑩𐑯𐑑 𐑧𐑐𐑦𐑛𐑧𐑥𐑦𐑒 𐑪𐑯 𐑚𐑹𐑛, 𐑯 𐑞𐑨𐑑 ·𐑥𐑱𐑣𐑿, 𐑣𐑻 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑢𐑪𐑟 𐑓𐑽𐑓𐑩𐑤 𐑝 𐑦𐑯𐑓𐑧𐑒𐑑𐑦𐑙 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑒𐑳𐑥𐑐𐑩𐑯𐑦. 𐑓, 𐑞𐑴 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑯 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵 𐑮𐑦𐑥𐑱𐑯𐑛 𐑳𐑯𐑑𐑱𐑯𐑑𐑩𐑛, 𐑯 𐑞𐑴 𐑣𐑦𐑟 𐑖𐑦𐑐 𐑢𐑪𐑟 𐑣𐑭𐑓 𐑩 𐑮𐑲𐑓𐑩𐑤-𐑖𐑪𐑑 𐑪𐑓, 𐑯 𐑩𐑯 𐑦𐑯𐑒𐑼𐑳𐑐𐑑𐑩𐑚𐑩𐑤 𐑕𐑰 𐑯 𐑺 𐑮𐑴𐑤𐑦𐑙 𐑯 𐑓𐑤𐑴𐑦𐑙 𐑚𐑦𐑑𐑢𐑰𐑯; 𐑘𐑧𐑑 𐑒𐑪𐑯𐑖𐑦𐑧𐑯𐑖𐑩𐑕𐑤𐑦 𐑨𐑛𐑣𐑽𐑦𐑙 𐑑 𐑞 𐑑𐑦𐑥𐑦𐑛 𐑒𐑢𐑪𐑮𐑩𐑯𐑑𐑰𐑯 𐑝 𐑞 𐑤𐑨𐑯𐑛, 𐑣𐑰 𐑐𐑼𐑧𐑥𐑐𐑑𐑼𐑦𐑤𐑦 𐑮𐑦𐑓𐑿𐑟𐑛 𐑑 𐑒𐑳𐑥 𐑦𐑯𐑑𐑵 𐑛𐑦𐑮𐑧𐑒𐑑 𐑒𐑪𐑯𐑑𐑨𐑒𐑑 𐑢𐑦𐑞 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛.

“𐑲 𐑓𐑽 𐑯𐑪𐑑 𐑞𐑲 𐑧𐑐𐑦𐑛𐑧𐑥𐑦𐑒, 𐑥𐑨𐑯,” 𐑕𐑧𐑛 ·𐑱𐑣𐑨𐑚 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑥𐑱𐑣𐑿, 𐑣𐑵 𐑕𐑑𐑫𐑛 𐑦𐑯 𐑞 𐑚𐑴𐑑'𐑕 𐑕𐑑𐑻𐑯; “𐑒𐑳𐑥 𐑪𐑯 𐑚𐑹𐑛.”

𐑚𐑳𐑑 𐑯𐑬 ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤 𐑕𐑑𐑸𐑑𐑩𐑛 𐑑 𐑣𐑦𐑟 𐑓𐑰𐑑.

“𐑔𐑦𐑙𐑒, 𐑔𐑦𐑙𐑒 𐑝 𐑞 𐑓𐑰𐑝𐑼𐑟, 𐑘𐑧𐑤𐑴 𐑯 𐑚𐑦𐑤𐑾𐑕! 𐑚𐑦𐑢𐑺 𐑝 𐑞 𐑣𐑪𐑮𐑦𐑚𐑩𐑤 𐑐𐑤𐑱𐑜!”

“·𐑜𐑭𐑚𐑮𐑦𐑧𐑤! ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤!” 𐑒𐑮𐑲𐑛 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑥𐑱𐑣𐑿; “𐑞𐑬 𐑥𐑳𐑕𐑑 𐑲𐑞𐑼—” 𐑚𐑳𐑑 𐑞𐑨𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑩 𐑣𐑧𐑛𐑤𐑪𐑙 𐑢𐑱𐑝 𐑖𐑪𐑑 𐑞 𐑚𐑴𐑑 𐑓𐑸 𐑩𐑣𐑧𐑛, 𐑯 𐑦𐑑𐑕 𐑕𐑰𐑞𐑦𐑙𐑟 𐑛𐑮𐑬𐑯𐑛 𐑷𐑤 𐑕𐑐𐑰𐑗.

“𐑣𐑨𐑕𐑑 𐑞𐑬 𐑕𐑰𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?” 𐑛𐑦𐑥𐑭𐑯𐑛𐑩𐑛 ·𐑱𐑣𐑨𐑚, 𐑢𐑧𐑯 𐑞 𐑚𐑴𐑑 𐑛𐑮𐑦𐑓𐑑𐑩𐑛 𐑚𐑨𐑒.

“𐑔𐑦𐑙𐑒, 𐑔𐑦𐑙𐑒 𐑝 𐑞𐑲 𐑢𐑱𐑤-𐑚𐑴𐑑, 𐑕𐑑𐑴𐑝𐑩𐑯 𐑯 𐑕𐑳𐑙𐑒! 𐑚𐑦𐑢𐑺 𐑝 𐑞 𐑣𐑪𐑮𐑦𐑚𐑩𐑤 𐑑𐑱𐑤!”

“𐑲 𐑑𐑧𐑤 𐑞𐑰 𐑩𐑜𐑱𐑯, ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤, 𐑞𐑨𐑑—” 𐑚𐑳𐑑 𐑩𐑜𐑱𐑯 𐑞 𐑚𐑴𐑑 𐑑𐑹 𐑩𐑣𐑧𐑛 𐑨𐑟 𐑦𐑓 𐑛𐑮𐑨𐑜𐑛 𐑚𐑲 𐑓𐑰𐑯𐑛𐑟. 𐑯𐑳𐑔𐑦𐑙 𐑢𐑪𐑟 𐑕𐑧𐑛 𐑓 𐑕𐑳𐑥 𐑥𐑴𐑥𐑧𐑯𐑑𐑕, 𐑢𐑲𐑤 𐑩 𐑕𐑩𐑒𐑕𐑧𐑖𐑩𐑯 𐑝 𐑮𐑲𐑩𐑑𐑩𐑕 𐑢𐑱𐑝𐑟 𐑮𐑴𐑤𐑛 𐑚𐑲, 𐑢𐑦𐑗 𐑚𐑲 𐑢𐑳𐑯 𐑝 𐑞𐑴𐑟 𐑩𐑒𐑱𐑠𐑩𐑯𐑩𐑤 𐑒𐑩𐑐𐑮𐑰𐑕𐑩𐑟 𐑝 𐑞 𐑕𐑰𐑟 𐑢𐑻 𐑑𐑳𐑥𐑚𐑩𐑤𐑦𐑙, 𐑯𐑪𐑑 𐑣𐑰𐑝𐑦𐑙 𐑦𐑑. 𐑥𐑰𐑯𐑑𐑲𐑥, 𐑞 𐑣𐑶𐑕𐑑𐑩𐑛 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛 𐑡𐑪𐑜𐑛 𐑩𐑚𐑬𐑑 𐑝𐑧𐑮𐑦 𐑝𐑲𐑩𐑤𐑩𐑯𐑑𐑤𐑦, 𐑯 ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤 𐑢𐑪𐑟 𐑕𐑰𐑯 𐑲𐑦𐑙 𐑦𐑑 𐑢𐑦𐑞 𐑮𐑭𐑞𐑼 𐑥𐑹 𐑨𐑐𐑮𐑦𐑣𐑧𐑯𐑕𐑦𐑝𐑯𐑩𐑕 𐑞𐑨𐑯 𐑣𐑦𐑟 𐑸𐑒𐑱𐑯𐑡𐑩𐑤 𐑯𐑱𐑗𐑼 𐑕𐑰𐑥𐑛 𐑑 𐑢𐑪𐑮𐑩𐑯𐑑.

·𐑥𐑱𐑣𐑿 𐑣𐑨𐑝𐑦𐑙 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑣𐑦𐑟 𐑯𐑼𐑱𐑖𐑩𐑯, ·𐑱𐑣𐑨𐑚 𐑐𐑫𐑑 𐑕𐑳𐑗 𐑒𐑢𐑧𐑕𐑗𐑩𐑯𐑟 𐑑 𐑣𐑦𐑥, 𐑞𐑨𐑑 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑒𐑨𐑐𐑑𐑦𐑯 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑓𐑹𐑚𐑺 𐑦𐑯𐑒𐑢𐑲𐑼𐑦𐑙 𐑢𐑧𐑞𐑼 𐑣𐑰 𐑦𐑯𐑑𐑧𐑯𐑛𐑩𐑛 𐑑 𐑣𐑳𐑯𐑑 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤, 𐑦𐑓 𐑪𐑐𐑼𐑑𐑿𐑯𐑦𐑑𐑦 𐑖𐑵𐑛 𐑪𐑓𐑼. 𐑑 𐑢𐑦𐑗 ·𐑱𐑣𐑨𐑚 𐑭𐑯𐑕𐑼𐑛—“𐑲.” 𐑕𐑑𐑮𐑱𐑑𐑢𐑱, 𐑞𐑧𐑯, ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑕𐑑𐑸𐑑𐑩𐑛 𐑑 𐑣𐑦𐑟 𐑓𐑰𐑑, 𐑜𐑤𐑺𐑦𐑙 𐑩𐑐𐑪𐑯 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑯 𐑝𐑾𐑥𐑩𐑯𐑑𐑤𐑦 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛, 𐑢𐑦𐑞 𐑛𐑬𐑯𐑢𐑼𐑛 𐑐𐑶𐑯𐑑𐑩𐑛 𐑓𐑦𐑙𐑜𐑼—“𐑔𐑦𐑙𐑒, 𐑔𐑦𐑙𐑒 𐑝 𐑞 𐑚𐑤𐑨𐑕𐑓𐑰𐑥𐑼—𐑛𐑧𐑛, 𐑯 𐑛𐑬𐑯 𐑞𐑺!—𐑚𐑦𐑢𐑺 𐑝 𐑞 𐑚𐑤𐑨𐑕𐑓𐑰𐑥𐑼'𐑟 𐑧𐑯𐑛!”

𐑱𐑣𐑨𐑚 𐑕𐑑𐑪𐑤𐑦𐑛𐑤𐑦 𐑑𐑻𐑯𐑛 𐑩𐑕𐑲𐑛; 𐑞𐑧𐑯 𐑕𐑧𐑛 𐑑 ·𐑥𐑱𐑣𐑿, “𐑒𐑨𐑐𐑑𐑦𐑯, 𐑲 𐑣𐑨𐑝 𐑡𐑳𐑕𐑑 𐑚𐑦𐑔𐑷𐑑 𐑥𐑰 𐑝 𐑥𐑲 𐑤𐑧𐑑𐑼-𐑚𐑨𐑜; 𐑞𐑺 𐑦𐑟 𐑩 𐑤𐑧𐑑𐑼 𐑓 𐑢𐑳𐑯 𐑝 𐑞𐑲 𐑪𐑓𐑦𐑕𐑼𐑟, 𐑦𐑓 𐑲 𐑥𐑦𐑕𐑑𐑱𐑒 𐑯𐑪𐑑. 𐑕𐑑𐑸𐑚𐑳𐑒, 𐑤𐑫𐑒 𐑴𐑝𐑼 𐑞 𐑚𐑨𐑜.”

𐑧𐑝𐑮𐑦 𐑢𐑱𐑤-𐑖𐑦𐑐 𐑑𐑱𐑒𐑕 𐑬𐑑 𐑩 𐑜𐑫𐑛𐑤𐑦 𐑯𐑳𐑥𐑚𐑼 𐑝 𐑤𐑧𐑑𐑼𐑟 𐑓 𐑝𐑺𐑾𐑕 𐑖𐑦𐑐𐑕, 𐑣𐑵𐑟 𐑛𐑦𐑤𐑦𐑝𐑼𐑦 𐑑 𐑞 𐑐𐑻𐑕𐑩𐑯𐑟 𐑑 𐑣𐑵𐑥 𐑞𐑱 𐑥𐑱 𐑚𐑰 𐑩𐑛𐑮𐑧𐑕𐑑, 𐑛𐑦𐑐𐑧𐑯𐑛𐑟 𐑩𐑐𐑪𐑯 𐑞 𐑥𐑽 𐑗𐑭𐑯𐑕 𐑝 𐑦𐑯𐑒𐑬𐑯𐑑𐑼𐑦𐑙 𐑞𐑧𐑥 𐑦𐑯 𐑞 𐑓𐑹 𐑴𐑖𐑩𐑯𐑟. 𐑞𐑳𐑕, 𐑥𐑴𐑕𐑑 𐑤𐑧𐑑𐑼𐑟 𐑯𐑧𐑝𐑼 𐑮𐑰𐑗 𐑞𐑺 𐑥𐑸𐑒; 𐑯 𐑥𐑧𐑯𐑦 𐑸 𐑴𐑯𐑤𐑦 𐑮𐑦𐑕𐑰𐑝𐑛 𐑭𐑓𐑑𐑼 𐑩𐑑𐑱𐑯𐑦𐑙 𐑩𐑯 𐑱𐑡 𐑝 𐑑𐑵 𐑹 𐑔𐑮𐑰 𐑘𐑽𐑟 𐑹 𐑥𐑹.

𐑕𐑵𐑯 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑮𐑩𐑑𐑻𐑯𐑛 𐑢𐑦𐑞 𐑩 𐑤𐑧𐑑𐑼 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛. 𐑦𐑑 𐑢𐑪𐑟 𐑕𐑹𐑤𐑦 𐑑𐑳𐑥𐑚𐑩𐑤𐑛, 𐑛𐑨𐑥𐑐, 𐑯 𐑒𐑳𐑝𐑼𐑛 𐑢𐑦𐑞 𐑩 𐑛𐑳𐑤, 𐑕𐑐𐑪𐑑𐑩𐑛, 𐑜𐑮𐑰𐑯 𐑥𐑴𐑤𐑛, 𐑦𐑯 𐑒𐑪𐑯𐑕𐑦𐑒𐑢𐑩𐑯𐑕 𐑝 𐑚𐑰𐑦𐑙 𐑒𐑧𐑐𐑑 𐑦𐑯 𐑩 𐑛𐑸𐑒 𐑤𐑪𐑒𐑼 𐑝 𐑞 𐑒𐑨𐑚𐑦𐑯. 𐑝 𐑕𐑳𐑗 𐑩 𐑤𐑧𐑑𐑼, ·𐑛𐑧𐑔 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑥𐑲𐑑 𐑢𐑧𐑤 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑞 𐑐𐑴𐑕𐑑-𐑚𐑶.

“𐑒𐑨𐑯'𐑕𐑑 𐑯𐑪𐑑 𐑮𐑰𐑛 𐑦𐑑?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚. “𐑜𐑦𐑝 𐑦𐑑 𐑥𐑰, 𐑥𐑨𐑯. 𐑲, 𐑲, 𐑦𐑑'𐑕 𐑚𐑳𐑑 𐑩 𐑛𐑦𐑥 𐑕𐑒𐑮𐑷𐑤;—𐑢𐑪𐑑'𐑕 𐑞𐑦𐑕?” 𐑨𐑟 𐑣𐑰 𐑢𐑪𐑟 𐑕𐑑𐑳𐑛𐑦𐑦𐑙 𐑦𐑑 𐑬𐑑, ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑑𐑫𐑒 𐑩 𐑤𐑪𐑙 𐑒𐑳𐑑𐑦𐑙-𐑕𐑐𐑱𐑛 𐑐𐑴𐑤, 𐑯 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑯𐑲𐑓 𐑕𐑤𐑲𐑑𐑤𐑦 𐑕𐑐𐑤𐑦𐑑 𐑞 𐑧𐑯𐑛, 𐑑 𐑦𐑯𐑕𐑻𐑑 𐑞 𐑤𐑧𐑑𐑼 𐑞𐑺, 𐑯 𐑦𐑯 𐑞𐑨𐑑 𐑢𐑱, 𐑣𐑨𐑯𐑛 𐑦𐑑 𐑑 𐑞 𐑚𐑴𐑑, 𐑢𐑦𐑞𐑬𐑑 𐑦𐑑𐑕 𐑒𐑳𐑥𐑦𐑙 𐑧𐑯𐑦 𐑒𐑤𐑴𐑟𐑼 𐑑 𐑞 𐑖𐑦𐑐.

𐑥𐑰𐑯𐑑𐑲𐑥, ·𐑱𐑣𐑨𐑚 𐑣𐑴𐑤𐑛𐑦𐑙 𐑞 𐑤𐑧𐑑𐑼, 𐑥𐑳𐑑𐑼𐑛, “·𐑥𐑼 ·𐑣𐑸—𐑘𐑧𐑕, ·𐑥𐑼 ·𐑣𐑨𐑮𐑦—(𐑩 𐑢𐑫𐑥𐑩𐑯'𐑟 𐑐𐑦𐑯𐑦 𐑣𐑨𐑯𐑛,—𐑞 𐑥𐑨𐑯'𐑟 𐑢𐑲𐑓, 𐑲'𐑤 𐑢𐑱𐑡𐑼)—·𐑲—·𐑥𐑼 ·𐑣𐑨𐑮𐑦 ·𐑥𐑱𐑕𐑦, ·𐑖𐑦𐑐 ·𐑡𐑧𐑮𐑩𐑚𐑴𐑩𐑥;—𐑢𐑲 𐑦𐑑'𐑕 ·𐑥𐑱𐑕𐑦, 𐑯 𐑣𐑰'𐑟 𐑛𐑧𐑛!”

“𐑐𐑫𐑼 𐑓𐑧𐑤𐑴! 𐑐𐑫𐑼 𐑓𐑧𐑤𐑴! 𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑢𐑲𐑓,” 𐑕𐑲𐑛 ·𐑥𐑱𐑣𐑿; “𐑚𐑳𐑑 𐑤𐑧𐑑 𐑥𐑰 𐑣𐑨𐑝 𐑦𐑑.”

“𐑯𐑱, 𐑒𐑰𐑐 𐑦𐑑 𐑞𐑲𐑕𐑧𐑤𐑓,” 𐑒𐑮𐑲𐑛 ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤 𐑑 ·𐑱𐑣𐑨𐑚; “𐑞𐑬 𐑸𐑑 𐑕𐑵𐑯 𐑜𐑴𐑦𐑙 𐑞𐑨𐑑 𐑢𐑱.”

“𐑒𐑻𐑕𐑩𐑟 𐑔𐑮𐑪𐑑𐑩𐑤 𐑞𐑰!” 𐑘𐑧𐑤𐑛 ·𐑱𐑣𐑨𐑚. “𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑥𐑱𐑣𐑿, 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑯𐑬 𐑑 𐑮𐑦𐑕𐑰𐑝 𐑦𐑑”; 𐑯 𐑑𐑱𐑒𐑦𐑙 𐑞 𐑓𐑱𐑑𐑩𐑤 𐑥𐑦𐑕𐑦𐑝 𐑓𐑮𐑪𐑥 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑣𐑨𐑯𐑛𐑟, 𐑣𐑰 𐑒𐑷𐑑 𐑦𐑑 𐑦𐑯 𐑞 𐑕𐑤𐑦𐑑 𐑝 𐑞 𐑐𐑴𐑤, 𐑯 𐑮𐑰𐑗𐑑 𐑦𐑑 𐑴𐑝𐑼 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑚𐑴𐑑. 𐑚𐑳𐑑 𐑨𐑟 𐑣𐑰 𐑛𐑦𐑛 𐑕𐑴, 𐑞 𐑹𐑟𐑥𐑧𐑯 𐑦𐑒𐑕𐑐𐑧𐑒𐑑𐑩𐑯𐑑𐑤𐑦 𐑛𐑦𐑟𐑦𐑕𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑮𐑴𐑦𐑙; 𐑞 𐑚𐑴𐑑 𐑛𐑮𐑦𐑓𐑑𐑩𐑛 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑖𐑦𐑐'𐑕 𐑕𐑑𐑻𐑯; 𐑕𐑴 𐑞𐑨𐑑, 𐑨𐑟 𐑦𐑓 𐑚𐑲 𐑥𐑨𐑡𐑦𐑒, 𐑞 𐑤𐑧𐑑𐑼 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑮𐑱𐑯𐑡𐑛 𐑩𐑤𐑪𐑙 𐑢𐑦𐑞 ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤'𐑟 𐑰𐑜𐑼 𐑣𐑨𐑯𐑛. 𐑣𐑰 𐑒𐑤𐑳𐑗𐑑 𐑦𐑑 𐑦𐑯 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑, 𐑕𐑰𐑟𐑛 𐑞 𐑚𐑴𐑑-𐑯𐑲𐑓, 𐑯 𐑦𐑥𐑐𐑱𐑤𐑦𐑙 𐑞 𐑤𐑧𐑑𐑼 𐑪𐑯 𐑦𐑑, 𐑕𐑧𐑯𐑑 𐑦𐑑 𐑞𐑳𐑕 𐑤𐑴𐑛𐑩𐑛 𐑚𐑨𐑒 𐑦𐑯𐑑𐑵 𐑞 𐑖𐑦𐑐. 𐑦𐑑 𐑓𐑧𐑤 𐑨𐑑 ·𐑱𐑣𐑨𐑚'𐑟 𐑓𐑰𐑑. 𐑞𐑧𐑯 ·𐑜𐑭𐑚𐑮𐑦𐑧𐑤 𐑖𐑮𐑰𐑒𐑑 𐑬𐑑 𐑑 𐑣𐑦𐑟 𐑒𐑪𐑥𐑮𐑱𐑛𐑟 𐑑 𐑜𐑦𐑝 𐑢𐑱 𐑢𐑦𐑞 𐑞𐑺 𐑹𐑟, 𐑯 𐑦𐑯 𐑞𐑨𐑑 𐑥𐑨𐑯𐑼 𐑞 𐑥𐑿𐑑𐑦𐑯𐑩𐑕 𐑚𐑴𐑑 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑖𐑪𐑑 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛.

𐑨𐑟, 𐑭𐑓𐑑𐑼 𐑞𐑦𐑕 𐑦𐑯𐑑𐑼𐑤𐑵𐑛, 𐑞 𐑕𐑰𐑥𐑧𐑯 𐑮𐑦𐑟𐑿𐑥𐑛 𐑞𐑺 𐑢𐑻𐑒 𐑩𐑐𐑪𐑯 𐑞 𐑡𐑨𐑒𐑩𐑑 𐑝 𐑞 𐑢𐑱𐑤, 𐑥𐑧𐑯𐑦 𐑕𐑑𐑮𐑱𐑯𐑡 𐑔𐑦𐑙𐑟 𐑢𐑻 𐑣𐑦𐑯𐑑𐑩𐑛 𐑦𐑯 𐑮𐑧𐑓𐑼𐑩𐑯𐑕 𐑑 𐑞𐑦𐑕 𐑢𐑲𐑤𐑛 𐑩𐑓𐑺.
