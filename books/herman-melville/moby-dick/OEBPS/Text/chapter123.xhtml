·𐑗𐑨𐑐𐑑𐑼 123. 𐑞 ·𐑥𐑳𐑕𐑒𐑩𐑑.

𐑛𐑘𐑫𐑼𐑦𐑙 𐑞 𐑥𐑴𐑕𐑑 𐑝𐑲𐑩𐑤𐑩𐑯𐑑 𐑖𐑪𐑒𐑕 𐑝 𐑞 ·𐑑𐑲𐑓𐑵𐑯, 𐑞 𐑥𐑨𐑯 𐑨𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑡𐑷-𐑚𐑴𐑯 𐑑𐑦𐑤𐑼 𐑣𐑨𐑛 𐑕𐑧𐑝𐑼𐑩𐑤 𐑑𐑲𐑥𐑟 𐑚𐑰𐑯 𐑮𐑰𐑤𐑦𐑙𐑤𐑦 𐑣𐑻𐑤𐑛 𐑑 𐑞 𐑛𐑧𐑒 𐑚𐑲 𐑦𐑑𐑕 𐑕𐑐𐑨𐑟𐑥𐑪𐑛𐑦𐑒 𐑥𐑴𐑖𐑩𐑯𐑟, 𐑰𐑝𐑩𐑯 𐑞𐑴 𐑐𐑮𐑦𐑝𐑧𐑯𐑑𐑼 𐑑𐑨𐑒𐑩𐑤𐑟 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑩𐑑𐑨𐑗𐑑 𐑑 𐑦𐑑—𐑓 𐑞𐑱 𐑢𐑻 𐑕𐑤𐑨𐑒—𐑚𐑦𐑒𐑪𐑟 𐑕𐑳𐑥 𐑐𐑤𐑱 𐑑 𐑞 𐑑𐑦𐑤𐑼 𐑢𐑪𐑟 𐑦𐑯𐑛𐑦𐑕𐑐𐑧𐑯𐑕𐑩𐑚𐑩𐑤.

𐑦𐑯 𐑩 𐑕𐑦𐑝𐑽 𐑜𐑱𐑤 𐑤𐑲𐑒 𐑞𐑦𐑕, 𐑢𐑲𐑤 𐑞 𐑖𐑦𐑐 𐑦𐑟 𐑚𐑳𐑑 𐑩 𐑑𐑪𐑕𐑑 𐑖𐑳𐑑𐑩𐑤𐑒𐑪𐑒 𐑑 𐑞 𐑚𐑤𐑭𐑕𐑑, 𐑦𐑑 𐑦𐑟 𐑚𐑲 𐑯𐑴 𐑥𐑰𐑯𐑟 𐑳𐑯𐑒𐑪𐑥𐑩𐑯 𐑑 𐑕𐑰 𐑞 𐑯𐑰𐑛𐑩𐑤𐑟 𐑦𐑯 𐑞 𐑒𐑳𐑥𐑐𐑩𐑕𐑩𐑟, 𐑨𐑑 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟, 𐑜𐑴 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛. 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑳𐑕 𐑢𐑦𐑞 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟; 𐑨𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑝𐑮𐑦 𐑖𐑪𐑒 𐑞 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑓𐑱𐑤𐑛 𐑑 𐑯𐑴𐑑𐑦𐑕 𐑞 𐑢𐑻𐑤𐑦𐑙 𐑝𐑦𐑤𐑪𐑕𐑦𐑑𐑦 𐑢𐑦𐑞 𐑢𐑦𐑗 𐑞𐑱 𐑮𐑦𐑝𐑪𐑤𐑝𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑒𐑸𐑛𐑟; 𐑦𐑑 𐑦𐑟 𐑩 𐑕𐑲𐑑 𐑞𐑨𐑑 𐑣𐑸𐑛𐑤𐑦 𐑧𐑯𐑦𐑢𐑳𐑯 𐑒𐑨𐑯 𐑚𐑦𐑣𐑴𐑤𐑛 𐑢𐑦𐑞𐑬𐑑 𐑕𐑳𐑥 𐑕𐑹𐑑 𐑝 𐑳𐑯𐑢𐑴𐑯𐑑𐑩𐑛 𐑦𐑥𐑴𐑖𐑩𐑯.

𐑕𐑳𐑥 𐑬𐑼𐑟 𐑭𐑓𐑑𐑼 𐑥𐑦𐑛𐑯𐑲𐑑, 𐑞 ·𐑑𐑲𐑓𐑵𐑯 𐑩𐑚𐑱𐑑𐑩𐑛 𐑕𐑴 𐑥𐑳𐑗, 𐑞𐑨𐑑 𐑔𐑮𐑵 𐑞 𐑕𐑑𐑮𐑧𐑯𐑿𐑩𐑕 𐑦𐑜𐑟𐑻𐑖𐑩𐑯𐑟 𐑝 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑯 ·𐑕𐑑𐑳𐑚—𐑢𐑳𐑯 𐑦𐑯𐑜𐑱𐑡𐑛 𐑓𐑹𐑢𐑼𐑛 𐑯 𐑞 𐑳𐑞𐑼 𐑭𐑓𐑑—𐑞 𐑖𐑦𐑝𐑼𐑛 𐑮𐑧𐑥𐑯𐑩𐑯𐑑𐑕 𐑝 𐑞 𐑡𐑦𐑚 𐑯 𐑓𐑹 𐑯 𐑥𐑱𐑯-𐑑𐑪𐑐-𐑕𐑱𐑤𐑟 𐑢𐑻 𐑒𐑳𐑑 𐑩𐑛𐑮𐑦𐑓𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑐𐑸𐑟, 𐑯 𐑢𐑧𐑯𐑑 𐑧𐑛𐑦𐑦𐑙 𐑩𐑢𐑱 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑤𐑲𐑒 𐑞 𐑓𐑧𐑞𐑼𐑟 𐑝 𐑩𐑯 𐑨𐑤𐑚𐑩𐑑𐑮𐑪𐑕, 𐑢𐑦𐑗 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑸 𐑒𐑭𐑕𐑑 𐑑 𐑞 𐑢𐑦𐑯𐑛𐑟 𐑢𐑧𐑯 𐑞𐑨𐑑 𐑕𐑑𐑹𐑥-𐑑𐑪𐑕𐑑 𐑚𐑻𐑛 𐑦𐑟 𐑪𐑯 𐑞 𐑢𐑦𐑙.

𐑞 𐑔𐑮𐑰 𐑒𐑪𐑮𐑦𐑕𐑐𐑪𐑯𐑛𐑦𐑙 𐑯𐑿 𐑕𐑱𐑤𐑟 𐑢𐑻 𐑯𐑬 𐑚𐑧𐑯𐑑 𐑯 𐑮𐑰𐑓𐑑, 𐑯 𐑩 𐑕𐑑𐑹𐑥-𐑑𐑮𐑲𐑕𐑱𐑤 𐑢𐑪𐑟 𐑕𐑧𐑑 𐑓𐑻𐑞𐑼 𐑭𐑓𐑑; 𐑕𐑴 𐑞𐑨𐑑 𐑞 𐑖𐑦𐑐 𐑕𐑵𐑯 𐑢𐑧𐑯𐑑 𐑔𐑮𐑵 𐑞 𐑢𐑷𐑑𐑼 𐑢𐑦𐑞 𐑕𐑳𐑥 𐑐𐑮𐑦𐑕𐑦𐑠𐑩𐑯 𐑩𐑜𐑱𐑯; 𐑯 𐑞 𐑒𐑹𐑕—𐑓 𐑞 𐑐𐑮𐑧𐑟𐑩𐑯𐑑, ·𐑰𐑕𐑑-𐑕𐑬𐑔-𐑰𐑕𐑑—𐑢𐑦𐑗 𐑣𐑰 𐑢𐑪𐑟 𐑑 𐑕𐑑𐑽, 𐑦𐑓 𐑐𐑮𐑨𐑒𐑑𐑦𐑒𐑩𐑚𐑩𐑤, 𐑢𐑪𐑟 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑜𐑦𐑝𐑩𐑯 𐑑 𐑞 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯. 𐑓 𐑛𐑘𐑫𐑼𐑦𐑙 𐑞 𐑝𐑲𐑩𐑤𐑩𐑯𐑕 𐑝 𐑞 𐑜𐑱𐑤, 𐑣𐑰 𐑣𐑨𐑛 𐑴𐑯𐑤𐑦 𐑕𐑑𐑽𐑛 𐑩𐑒𐑹𐑛𐑦𐑙 𐑑 𐑦𐑑𐑕 𐑝𐑦𐑕𐑦𐑕𐑦𐑑𐑿𐑛𐑟. 𐑚𐑳𐑑 𐑨𐑟 𐑣𐑰 𐑢𐑪𐑟 𐑯𐑬 𐑚𐑮𐑦𐑙𐑦𐑙 𐑞 𐑖𐑦𐑐 𐑨𐑟 𐑯𐑽 𐑣𐑻 𐑒𐑹𐑕 𐑨𐑟 𐑐𐑪𐑕𐑩𐑚𐑩𐑤, 𐑢𐑪𐑗𐑦𐑙 𐑞 𐑒𐑳𐑥𐑐𐑩𐑕 𐑥𐑰𐑯𐑢𐑲𐑤, 𐑤𐑴! 𐑩 𐑜𐑫𐑛 𐑕𐑲𐑯! 𐑞 𐑢𐑦𐑯𐑛 𐑕𐑰𐑥𐑛 𐑒𐑳𐑥𐑦𐑙 𐑮𐑬𐑯𐑛 𐑩𐑕𐑑𐑻𐑯; 𐑲, 𐑞 𐑓𐑬𐑤 𐑚𐑮𐑰𐑟 𐑚𐑦𐑒𐑱𐑥 𐑓𐑺!

𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑞 𐑘𐑸𐑛𐑟 𐑢𐑻 𐑕𐑒𐑢𐑺𐑛, 𐑑 𐑞 𐑤𐑲𐑝𐑤𐑦 𐑕𐑪𐑙 𐑝 “𐑣𐑴! 𐑞 𐑓𐑺 𐑢𐑦𐑯𐑛! 𐑴-𐑘𐑰-𐑣𐑴, 𐑗𐑽𐑤𐑦 𐑥𐑧𐑯!” 𐑞 𐑒𐑮𐑵 𐑕𐑦𐑙𐑦𐑙 𐑓 𐑡𐑶, 𐑞𐑨𐑑 𐑕𐑴 𐑐𐑮𐑪𐑥𐑦𐑕𐑦𐑙 𐑩𐑯 𐑦𐑝𐑧𐑯𐑑 𐑖𐑵𐑛 𐑕𐑴 𐑕𐑵𐑯 𐑣𐑨𐑝 𐑓𐑷𐑤𐑕𐑦𐑓𐑲𐑛 𐑞 𐑰𐑝𐑩𐑤 𐑐𐑹𐑑𐑧𐑯𐑑𐑕 𐑐𐑮𐑦𐑕𐑰𐑛𐑦𐑙 𐑦𐑑.

𐑦𐑯 𐑒𐑩𐑥𐑐𐑤𐑲𐑩𐑯𐑕 𐑢𐑦𐑞 𐑞 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑹𐑛𐑼 𐑝 𐑣𐑦𐑟 𐑒𐑩𐑥𐑭𐑯𐑛𐑼—𐑑 𐑮𐑦𐑐𐑹𐑑 𐑦𐑥𐑰𐑛𐑾𐑑𐑤𐑦, 𐑯 𐑨𐑑 𐑧𐑯𐑦 𐑢𐑳𐑯 𐑝 𐑞 𐑑𐑢𐑧𐑯𐑑𐑦-𐑓𐑹 𐑬𐑼𐑟, 𐑧𐑯𐑦 𐑛𐑦𐑕𐑲𐑛𐑩𐑛 𐑗𐑱𐑯𐑡 𐑦𐑯 𐑞 𐑩𐑓𐑺𐑟 𐑝 𐑞 𐑛𐑧𐑒,—·𐑕𐑑𐑸𐑚𐑳𐑒 𐑣𐑨𐑛 𐑯𐑴 𐑕𐑵𐑯𐑼 𐑑𐑮𐑦𐑥𐑛 𐑞 𐑘𐑸𐑛𐑟 𐑑 𐑞 𐑚𐑮𐑰𐑟—𐑣𐑬𐑧𐑝𐑼 𐑮𐑦𐑤𐑳𐑒𐑑𐑩𐑯𐑑𐑤𐑦 𐑯 𐑜𐑤𐑵𐑥𐑦𐑤𐑦,—𐑞𐑨𐑯 𐑣𐑰 𐑥𐑦𐑒𐑨𐑯𐑦𐑒𐑤𐑦 𐑢𐑧𐑯𐑑 𐑚𐑦𐑤𐑴 𐑑 𐑩𐑐𐑮𐑲𐑟 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑝 𐑞 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕.

𐑺 𐑯𐑪𐑒𐑦𐑙 𐑨𐑑 𐑣𐑦𐑟 𐑕𐑑𐑱𐑑-𐑮𐑵𐑥, 𐑣𐑰 𐑦𐑯𐑝𐑪𐑤𐑩𐑯𐑑𐑺𐑦𐑤𐑦 𐑐𐑷𐑟𐑛 𐑚𐑦𐑓𐑹 𐑦𐑑 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑. 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑤𐑨𐑥𐑐—𐑑𐑱𐑒𐑦𐑙 𐑤𐑪𐑙 𐑕𐑢𐑦𐑙𐑟 𐑞𐑦𐑕 𐑢𐑱 𐑯 𐑞𐑨𐑑—𐑢𐑪𐑟 𐑚𐑻𐑯𐑦𐑙 𐑓𐑦𐑑𐑓𐑩𐑤𐑦, 𐑯 𐑒𐑭𐑕𐑑𐑦𐑙 𐑓𐑦𐑑𐑓𐑩𐑤 𐑖𐑨𐑛𐑴𐑟 𐑩𐑐𐑪𐑯 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑚𐑴𐑤𐑑𐑩𐑛 𐑛𐑹,—𐑩 𐑔𐑦𐑯 𐑢𐑳𐑯, 𐑢𐑦𐑞 𐑓𐑦𐑒𐑕𐑑 𐑚𐑤𐑲𐑯𐑛𐑟 𐑦𐑯𐑕𐑻𐑑𐑩𐑛, 𐑦𐑯 𐑐𐑤𐑱𐑕 𐑝 𐑳𐑐𐑼 𐑐𐑨𐑯𐑩𐑤𐑟. 𐑞 𐑲𐑕𐑩𐑤𐑱𐑑𐑩𐑛 𐑕𐑳𐑚𐑑𐑼𐑱𐑯𐑾𐑕𐑯𐑩𐑕 𐑝 𐑞 𐑒𐑨𐑚𐑦𐑯 𐑥𐑱𐑛 𐑩 𐑕𐑻𐑑𐑩𐑯 𐑣𐑳𐑥𐑦𐑙 𐑕𐑲𐑤𐑩𐑯𐑕 𐑑 𐑮𐑱𐑯 𐑞𐑺, 𐑞𐑴 𐑦𐑑 𐑢𐑪𐑟 𐑣𐑵𐑐𐑑 𐑮𐑬𐑯𐑛 𐑚𐑲 𐑷𐑤 𐑞 𐑮𐑹 𐑝 𐑞 𐑧𐑤𐑦𐑥𐑩𐑯𐑑𐑕. 𐑞 𐑤𐑴𐑛𐑩𐑛 𐑥𐑳𐑕𐑒𐑩𐑑𐑕 𐑦𐑯 𐑞 𐑮𐑨𐑒 𐑢𐑻 𐑖𐑲𐑯𐑦𐑙𐑤𐑦 𐑮𐑦𐑝𐑰𐑤𐑛, 𐑨𐑟 𐑞𐑱 𐑕𐑑𐑫𐑛 𐑳𐑐𐑮𐑲𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑓𐑹𐑢𐑼𐑛 𐑚𐑳𐑤𐑒𐑣𐑧𐑛. 𐑕𐑑𐑸𐑚𐑳𐑒 𐑢𐑪𐑟 𐑩𐑯 𐑪𐑯𐑩𐑕𐑑, 𐑳𐑐𐑮𐑲𐑑 𐑥𐑨𐑯; 𐑚𐑳𐑑 𐑬𐑑 𐑝 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑣𐑸𐑑, 𐑨𐑑 𐑞𐑨𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑢𐑧𐑯 𐑣𐑰 𐑕𐑷 𐑞 𐑥𐑳𐑕𐑒𐑩𐑑𐑕, 𐑞𐑺 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑦𐑝𐑪𐑤𐑝𐑛 𐑩𐑯 𐑰𐑝𐑩𐑤 𐑔𐑷𐑑; 𐑚𐑳𐑑 𐑕𐑴 𐑚𐑤𐑧𐑯𐑑 𐑢𐑦𐑞 𐑦𐑑𐑕 𐑯𐑿𐑑𐑮𐑩𐑤 𐑹 𐑜𐑫𐑛 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑥𐑩𐑯𐑑𐑕 𐑞𐑨𐑑 𐑓 𐑞 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑣𐑰 𐑣𐑸𐑛𐑤𐑦 𐑯𐑿 𐑦𐑑 𐑓 𐑦𐑑𐑕𐑧𐑤𐑓.

“𐑣𐑰 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑖𐑪𐑑 𐑥𐑰 𐑢𐑳𐑯𐑕,” 𐑣𐑰 𐑥𐑻𐑥𐑼𐑛, “𐑘𐑧𐑕, 𐑞𐑺'𐑟 𐑞 𐑝𐑧𐑮𐑦 𐑥𐑳𐑕𐑒𐑩𐑑 𐑞𐑨𐑑 𐑣𐑰 𐑐𐑶𐑯𐑑𐑩𐑛 𐑨𐑑 𐑥𐑰;—𐑞𐑨𐑑 𐑢𐑳𐑯 𐑢𐑦𐑞 𐑞 𐑕𐑑𐑳𐑛𐑩𐑛 𐑕𐑑𐑪𐑒; 𐑤𐑧𐑑 𐑥𐑰 𐑑𐑳𐑗 𐑦𐑑—𐑤𐑦𐑓𐑑 𐑦𐑑. 𐑕𐑑𐑮𐑱𐑯𐑡, 𐑞𐑨𐑑 𐑲, 𐑣𐑵 𐑣𐑨𐑝 𐑣𐑨𐑯𐑛𐑩𐑤𐑛 𐑕𐑴 𐑥𐑧𐑯𐑦 𐑛𐑧𐑛𐑤𐑦 𐑤𐑭𐑯𐑕𐑩𐑟, 𐑕𐑑𐑮𐑱𐑯𐑡, 𐑞𐑨𐑑 𐑲 𐑖𐑵𐑛 𐑖𐑱𐑒 𐑕𐑴 𐑯𐑬. 𐑤𐑴𐑛𐑩𐑛? 𐑲 𐑥𐑳𐑕𐑑 𐑕𐑰. 𐑲, 𐑲; 𐑯 𐑐𐑬𐑛𐑼 𐑦𐑯 𐑞 𐑐𐑨𐑯;—𐑞𐑨𐑑'𐑕 𐑯𐑪𐑑 𐑜𐑫𐑛. 𐑚𐑧𐑕𐑑 𐑕𐑐𐑦𐑤 𐑦𐑑?—𐑢𐑱𐑑. 𐑲'𐑤 𐑒𐑘𐑫𐑼 𐑥𐑲𐑕𐑧𐑤𐑓 𐑝 𐑞𐑦𐑕. 𐑲'𐑤 𐑣𐑴𐑤𐑛 𐑞 𐑥𐑳𐑕𐑒𐑩𐑑 𐑚𐑴𐑤𐑛𐑤𐑦 𐑢𐑲𐑤 𐑲 𐑔𐑦𐑙𐑒.—𐑲 𐑒𐑳𐑥 𐑑 𐑮𐑦𐑐𐑹𐑑 𐑩 𐑓𐑺 𐑢𐑦𐑯𐑛 𐑑 𐑣𐑦𐑥. 𐑚𐑳𐑑 𐑣𐑬 𐑓𐑺? 𐑓𐑺 𐑓 𐑛𐑧𐑔 𐑯 𐑛𐑵𐑥,—𐑞𐑨𐑑'𐑕 𐑓𐑺 𐑓 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒. 𐑦𐑑'𐑕 𐑩 𐑓𐑺 𐑢𐑦𐑯𐑛 𐑞𐑨𐑑'𐑕 𐑴𐑯𐑤𐑦 𐑓𐑺 𐑓 𐑞𐑨𐑑 𐑩𐑒𐑻𐑕𐑩𐑛 𐑓𐑦𐑖.—𐑞 𐑝𐑧𐑮𐑦 𐑑𐑿𐑚 𐑣𐑰 𐑐𐑶𐑯𐑑𐑩𐑛 𐑨𐑑 𐑥𐑰!—𐑞 𐑝𐑧𐑮𐑦 𐑢𐑳𐑯; 𐑞𐑦𐑕 𐑢𐑳𐑯—𐑲 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑣𐑽; 𐑣𐑰 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑒𐑦𐑤𐑛 𐑥𐑰 𐑢𐑦𐑞 𐑞 𐑝𐑧𐑮𐑦 𐑔𐑦𐑙 𐑲 𐑣𐑨𐑯𐑛𐑩𐑤 𐑯𐑬.—𐑲 𐑯 𐑣𐑰 𐑢𐑫𐑛 𐑓𐑱𐑯 𐑒𐑦𐑤 𐑷𐑤 𐑣𐑦𐑟 𐑒𐑮𐑵. 𐑛𐑳𐑟 𐑣𐑰 𐑯𐑪𐑑 𐑕𐑱 𐑣𐑰 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑕𐑑𐑮𐑲𐑒 𐑣𐑦𐑟 𐑕𐑐𐑸𐑟 𐑑 𐑧𐑯𐑦 𐑜𐑱𐑤? 𐑣𐑨𐑟 𐑣𐑰 𐑯𐑪𐑑 𐑛𐑨𐑖𐑑 𐑣𐑦𐑟 𐑣𐑧𐑝𐑩𐑯𐑤𐑦 𐑒𐑢𐑪𐑛𐑮𐑩𐑯𐑑? 𐑯 𐑦𐑯 𐑞𐑰𐑟 𐑕𐑱𐑥 𐑐𐑧𐑮𐑦𐑤𐑩𐑕 𐑕𐑰𐑟, 𐑜𐑮𐑴𐑐𐑕 𐑣𐑰 𐑯𐑪𐑑 𐑣𐑦𐑟 𐑢𐑱 𐑚𐑲 𐑥𐑽 𐑛𐑧𐑛 𐑮𐑧𐑒𐑩𐑯𐑦𐑙 𐑝 𐑞 𐑻𐑼-𐑩𐑚𐑬𐑯𐑛𐑦𐑙 𐑤𐑪𐑜? 𐑯 𐑦𐑯 𐑞𐑦𐑕 𐑝𐑧𐑮𐑦 ·𐑑𐑲𐑓𐑵𐑯, 𐑛𐑦𐑛 𐑣𐑰 𐑯𐑪𐑑 𐑕𐑢𐑺 𐑞𐑨𐑑 𐑣𐑰 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑯𐑴 𐑤𐑲𐑑𐑯𐑦𐑙-𐑮𐑪𐑛𐑟? 𐑚𐑳𐑑 𐑖𐑨𐑤 𐑞𐑦𐑕 𐑒𐑮𐑱𐑟𐑛 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑚𐑰 𐑑𐑱𐑥𐑤𐑦 𐑕𐑳𐑓𐑼𐑛 𐑑 𐑛𐑮𐑨𐑜 𐑩 𐑣𐑴𐑤 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦 𐑛𐑬𐑯 𐑑 𐑛𐑵𐑥 𐑢𐑦𐑞 𐑣𐑦𐑥?—𐑘𐑧𐑕, 𐑦𐑑 𐑢𐑫𐑛 𐑥𐑱𐑒 𐑣𐑦𐑥 𐑞 𐑢𐑦𐑤𐑓𐑩𐑤 𐑥𐑻𐑛𐑼𐑼 𐑝 𐑔𐑻𐑑𐑦 𐑥𐑧𐑯 𐑯 𐑥𐑹, 𐑦𐑓 𐑞𐑦𐑕 𐑖𐑦𐑐 𐑒𐑳𐑥 𐑑 𐑧𐑯𐑦 𐑛𐑧𐑛𐑤𐑦 𐑣𐑸𐑥; 𐑯 𐑒𐑳𐑥 𐑑 𐑛𐑧𐑛𐑤𐑦 𐑣𐑸𐑥, 𐑥𐑲 𐑕𐑴𐑤 𐑕𐑢𐑺𐑟 𐑞𐑦𐑕 𐑖𐑦𐑐 𐑢𐑦𐑤, 𐑦𐑓 ·𐑱𐑣𐑨𐑚 𐑣𐑨𐑝 𐑣𐑦𐑟 𐑢𐑱. 𐑦𐑓, 𐑞𐑧𐑯, 𐑣𐑰 𐑢𐑻 𐑞𐑦𐑕 𐑦𐑯𐑕𐑑𐑩𐑯𐑑—𐑐𐑫𐑑 𐑩𐑕𐑲𐑛, 𐑞𐑨𐑑 𐑒𐑮𐑲𐑥 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑚𐑰 𐑣𐑦𐑟. 𐑣𐑭! 𐑦𐑟 𐑣𐑰 𐑥𐑳𐑑𐑼𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑕𐑤𐑰𐑐? 𐑘𐑧𐑕, 𐑡𐑳𐑕𐑑 𐑞𐑺,—𐑦𐑯 𐑞𐑺, 𐑣𐑰'𐑟 𐑕𐑤𐑰𐑐𐑦𐑙. 𐑕𐑤𐑰𐑐𐑦𐑙? 𐑲, 𐑚𐑳𐑑 𐑕𐑑𐑦𐑤 𐑩𐑤𐑲𐑝, 𐑯 𐑕𐑵𐑯 𐑩𐑢𐑱𐑒 𐑩𐑜𐑱𐑯. 𐑲 𐑒𐑭𐑯𐑑 𐑢𐑦𐑞𐑕𐑑𐑨𐑯𐑛 𐑞𐑰, 𐑞𐑧𐑯, 𐑴𐑤𐑛 𐑥𐑨𐑯. 𐑯𐑪𐑑 𐑮𐑰𐑟𐑩𐑯𐑦𐑙; 𐑯𐑪𐑑 𐑮𐑦𐑥𐑪𐑯𐑕𐑑𐑮𐑩𐑯𐑕; 𐑯𐑪𐑑 𐑦𐑯𐑑𐑮𐑰𐑑𐑦 𐑢𐑦𐑤𐑑 𐑞𐑬 𐑣𐑸𐑒𐑩𐑯 𐑑; 𐑷𐑤 𐑞𐑦𐑕 𐑞𐑬 𐑕𐑒𐑹𐑯𐑩𐑕𐑑. 𐑓𐑤𐑨𐑑 𐑴𐑚𐑰𐑛𐑾𐑯𐑕 𐑑 𐑞𐑲 𐑴𐑯 𐑓𐑤𐑨𐑑 𐑒𐑩𐑥𐑭𐑯𐑛𐑟, 𐑞𐑦𐑕 𐑦𐑟 𐑷𐑤 𐑞𐑬 𐑚𐑮𐑰𐑞𐑩𐑕𐑑. 𐑲, 𐑯 𐑕𐑱'𐑕𐑑 𐑞 𐑥𐑧𐑯 𐑣𐑨𐑝 𐑝𐑬'𐑛 𐑞𐑲 𐑝𐑬; 𐑕𐑱'𐑕𐑑 𐑷𐑤 𐑝 𐑳𐑕 𐑸 ·𐑱𐑣𐑨𐑚𐑟. 𐑜𐑮𐑱𐑑 ·𐑜𐑪𐑛 𐑓𐑼𐑚𐑦𐑛!—𐑚𐑳𐑑 𐑦𐑟 𐑞𐑺 𐑯𐑴 𐑳𐑞𐑼 𐑢𐑱? 𐑯𐑴 𐑤𐑷𐑓𐑩𐑤 𐑢𐑱?—𐑥𐑱𐑒 𐑣𐑦𐑥 𐑩 𐑐𐑮𐑦𐑟𐑩𐑯𐑼 𐑑 𐑚𐑰 𐑑𐑱𐑒𐑩𐑯 𐑣𐑴𐑥? 𐑢𐑪𐑑! 𐑣𐑴𐑐 𐑑 𐑮𐑧𐑕𐑑 𐑞𐑦𐑕 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑤𐑦𐑝𐑦𐑙 𐑐𐑬𐑼 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑴𐑯 𐑤𐑦𐑝𐑦𐑙 𐑣𐑨𐑯𐑛𐑟? 𐑴𐑯𐑤𐑦 𐑩 𐑓𐑵𐑤 𐑢𐑫𐑛 𐑑𐑮𐑲 𐑦𐑑. 𐑕𐑱 𐑣𐑰 𐑢𐑻 𐑐𐑦𐑯𐑘𐑩𐑯𐑛 𐑰𐑝𐑩𐑯; 𐑯𐑪𐑑𐑩𐑛 𐑷𐑤 𐑴𐑝𐑼 𐑢𐑦𐑞 𐑮𐑴𐑐𐑕 𐑯 𐑣𐑷𐑟𐑼𐑟; 𐑗𐑱𐑯𐑛 𐑛𐑬𐑯 𐑑 𐑮𐑦𐑙-𐑚𐑴𐑤𐑑𐑕 𐑪𐑯 𐑞𐑦𐑕 𐑒𐑨𐑚𐑦𐑯 𐑓𐑤𐑹; 𐑣𐑰 𐑢𐑫𐑛 𐑚𐑰 𐑥𐑹 𐑣𐑦𐑛𐑾𐑕 𐑞𐑨𐑯 𐑩 𐑒𐑱𐑡𐑛 𐑑𐑲𐑜𐑼, 𐑞𐑧𐑯. 𐑲 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑦𐑯𐑛𐑻 𐑞 𐑕𐑲𐑑; 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑐𐑪𐑕𐑩𐑚𐑤𐑦 𐑓𐑤𐑲 𐑣𐑦𐑟 𐑣𐑬𐑤𐑦𐑙𐑟; 𐑷𐑤 𐑒𐑳𐑥𐑓𐑼𐑑, 𐑕𐑤𐑰𐑐 𐑦𐑑𐑕𐑧𐑤𐑓, 𐑦𐑯𐑧𐑕𐑑𐑦𐑥𐑩𐑚𐑩𐑤 𐑮𐑰𐑟𐑩𐑯 𐑢𐑫𐑛 𐑤𐑰𐑝 𐑥𐑰 𐑪𐑯 𐑞 𐑤𐑪𐑙 𐑦𐑯𐑑𐑪𐑤𐑼𐑩𐑚𐑩𐑤 𐑝𐑶𐑦𐑡. 𐑢𐑪𐑑, 𐑞𐑧𐑯, 𐑮𐑦𐑥𐑱𐑯𐑟? 𐑞 𐑤𐑨𐑯𐑛 𐑦𐑟 𐑣𐑳𐑯𐑛𐑮𐑩𐑛𐑟 𐑝 𐑤𐑰𐑜𐑟 𐑩𐑢𐑱, 𐑯 𐑤𐑪𐑒𐑑 ·𐑡𐑩𐑐𐑨𐑯 𐑞 𐑯𐑽𐑩𐑕𐑑. 𐑲 𐑕𐑑𐑨𐑯𐑛 𐑩𐑤𐑴𐑯 𐑣𐑽 𐑩𐑐𐑪𐑯 𐑩𐑯 𐑴𐑐𐑩𐑯 𐑕𐑰, 𐑢𐑦𐑞 𐑑𐑵 𐑴𐑖𐑩𐑯𐑟 𐑯 𐑩 𐑣𐑴𐑤 𐑒𐑪𐑯𐑑𐑦𐑯𐑩𐑯𐑑 𐑚𐑦𐑑𐑢𐑰𐑯 𐑥𐑰 𐑯 𐑤𐑷.—𐑲, 𐑲, '𐑑𐑦𐑟 𐑕𐑴.—𐑦𐑟 𐑣𐑧𐑝𐑩𐑯 𐑩 𐑥𐑻𐑛𐑼𐑼 𐑢𐑧𐑯 𐑦𐑑𐑕 𐑤𐑲𐑑𐑯𐑦𐑙 𐑕𐑑𐑮𐑲𐑒𐑕 𐑩 𐑢𐑫𐑛-𐑚𐑰 𐑥𐑻𐑛𐑼𐑼 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑧𐑛, 𐑑𐑦𐑯𐑛𐑼𐑦𐑙 𐑖𐑰𐑑𐑕 𐑯 𐑕𐑒𐑦𐑯 𐑑𐑩𐑜𐑧𐑞𐑼?—𐑯 𐑢𐑫𐑛 𐑲 𐑚𐑰 𐑩 𐑥𐑻𐑛𐑼𐑼, 𐑞𐑧𐑯, 𐑦𐑓”—𐑯 𐑕𐑤𐑴𐑤𐑦, 𐑕𐑑𐑧𐑤𐑔𐑦𐑤𐑦, 𐑯 𐑣𐑭𐑓 𐑕𐑲𐑛𐑢𐑱𐑟 𐑤𐑫𐑒𐑦𐑙, 𐑣𐑰 𐑐𐑤𐑱𐑕𐑑 𐑞 𐑤𐑴𐑛𐑩𐑛 𐑥𐑳𐑕𐑒𐑩𐑑'𐑕 𐑧𐑯𐑛 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑛𐑹.

“𐑪𐑯 𐑞𐑦𐑕 𐑤𐑧𐑝𐑩𐑤, ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑨𐑥𐑩𐑒 𐑕𐑢𐑦𐑙𐑟 𐑢𐑦𐑞𐑦𐑯; 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑞𐑦𐑕 𐑢𐑱. 𐑩 𐑑𐑳𐑗, 𐑯 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑥𐑱 𐑕𐑼𐑝𐑲𐑝 𐑑 𐑣𐑳𐑜 𐑣𐑦𐑟 𐑢𐑲𐑓 𐑯 𐑗𐑲𐑤𐑛 𐑩𐑜𐑱𐑯.—𐑴 ·𐑥𐑺𐑦! 𐑥𐑺𐑦!—𐑚𐑶! 𐑚𐑶! 𐑚𐑶!—𐑚𐑳𐑑 𐑦𐑓 𐑲 𐑢𐑱𐑒 𐑞𐑰 𐑯𐑪𐑑 𐑑 𐑛𐑧𐑔, 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑣𐑵 𐑒𐑨𐑯 𐑑𐑧𐑤 𐑑 𐑢𐑪𐑑 𐑳𐑯𐑕𐑬𐑯𐑛𐑩𐑛 𐑛𐑰𐑐𐑕 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑚𐑪𐑛𐑦 𐑞𐑦𐑕 𐑛𐑱 𐑢𐑰𐑒 𐑥𐑱 𐑕𐑦𐑙𐑒, 𐑢𐑦𐑞 𐑷𐑤 𐑞 𐑒𐑮𐑵! 𐑜𐑮𐑱𐑑 ·𐑜𐑪𐑛, 𐑢𐑺 𐑸𐑑 ·𐑞𐑬? 𐑖𐑨𐑤 𐑲? 𐑖𐑨𐑤 𐑲?—𐑞 𐑢𐑦𐑯𐑛 𐑣𐑨𐑟 𐑜𐑪𐑯 𐑛𐑬𐑯 𐑯 𐑖𐑦𐑓𐑑𐑩𐑛, 𐑕𐑻; 𐑞 𐑓𐑹 𐑯 𐑥𐑱𐑯 𐑑𐑪𐑐𐑕𐑱𐑤𐑟 𐑸 𐑮𐑰𐑓𐑑 𐑯 𐑕𐑧𐑑; 𐑖𐑰 𐑣𐑧𐑛𐑟 𐑣𐑻 𐑒𐑹𐑕.”

“𐑕𐑑𐑻𐑯 𐑷𐑤! 𐑴 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒, 𐑲 𐑒𐑤𐑳𐑗 𐑞𐑲 𐑣𐑸𐑑 𐑨𐑑 𐑤𐑭𐑕𐑑!”

𐑕𐑳𐑗 𐑢𐑻 𐑞 𐑕𐑬𐑯𐑛𐑟 𐑞𐑨𐑑 𐑯𐑬 𐑒𐑱𐑥 𐑣𐑻𐑑𐑩𐑤𐑦𐑙 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑑𐑹𐑥𐑧𐑯𐑑𐑩𐑛 𐑕𐑤𐑰𐑐, 𐑨𐑟 𐑦𐑓 ·𐑕𐑑𐑸𐑚𐑳𐑒'𐑕 𐑝𐑶𐑕 𐑣𐑨𐑛 𐑒𐑷𐑟𐑛 𐑞 𐑤𐑪𐑙 𐑛𐑳𐑥 𐑛𐑮𐑰𐑥 𐑑 𐑕𐑐𐑰𐑒.

𐑞 𐑘𐑧𐑑 𐑤𐑧𐑝𐑩𐑤𐑛 𐑥𐑳𐑕𐑒𐑩𐑑 𐑖𐑫𐑒 𐑤𐑲𐑒 𐑩 𐑛𐑮𐑳𐑯𐑒𐑼𐑛'𐑟 𐑸𐑥 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑐𐑨𐑯𐑩𐑤; ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑕𐑰𐑥𐑛 𐑮𐑧𐑕𐑩𐑤𐑦𐑙 𐑢𐑦𐑞 𐑩𐑯 𐑱𐑯𐑡𐑩𐑤; 𐑚𐑳𐑑 𐑑𐑻𐑯𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑛𐑹, 𐑣𐑰 𐑐𐑤𐑱𐑕𐑑 𐑞 𐑛𐑧𐑔-𐑑𐑿𐑚 𐑦𐑯 𐑦𐑑𐑕 𐑮𐑨𐑒, 𐑯 𐑤𐑧𐑓𐑑 𐑞 𐑐𐑤𐑱𐑕.

“𐑣𐑰'𐑟 𐑑𐑵 𐑕𐑬𐑯𐑛 𐑩𐑕𐑤𐑰𐑐, ·𐑥𐑼 ·𐑕𐑑𐑳𐑚; 𐑜𐑴 𐑞𐑬 𐑛𐑬𐑯, 𐑯 𐑢𐑱𐑒 𐑣𐑦𐑥, 𐑯 𐑑𐑧𐑤 𐑣𐑦𐑥. 𐑲 𐑥𐑳𐑕𐑑 𐑕𐑰 𐑑 𐑞 𐑛𐑧𐑒 𐑣𐑽. 𐑞𐑬 𐑯𐑴'𐑕𐑑 𐑢𐑪𐑑 𐑑 𐑕𐑱.”
