·𐑗𐑨𐑐𐑑𐑼 61. 𐑕𐑑𐑳𐑚 ·𐑒𐑦𐑤𐑟 𐑩 ·𐑢𐑱𐑤.

𐑦𐑓 𐑑 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑞 𐑨𐑐𐑼𐑦𐑖𐑩𐑯 𐑝 𐑞 ·𐑕𐑒𐑢𐑦𐑛 𐑢𐑪𐑟 𐑩 𐑔𐑦𐑙 𐑝 𐑐𐑹𐑑𐑧𐑯𐑑𐑕, 𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑦𐑑 𐑢𐑪𐑟 𐑒𐑢𐑲𐑑 𐑩 𐑛𐑦𐑓𐑼𐑩𐑯𐑑 𐑪𐑚𐑡𐑧𐑒𐑑.

“𐑢𐑧𐑯 𐑿 𐑕𐑰 𐑣𐑦𐑥 '𐑒𐑢𐑦𐑛,” 𐑕𐑧𐑛 𐑞 𐑕𐑨𐑝𐑦𐑡, 𐑣𐑴𐑯𐑦𐑙 𐑣𐑦𐑟 𐑣𐑸𐑐𐑵𐑯 𐑦𐑯 𐑞 𐑚𐑬 𐑝 𐑣𐑦𐑟 𐑣𐑶𐑕𐑑𐑩𐑛 𐑚𐑴𐑑, “𐑞𐑧𐑯 𐑿 𐑒𐑢𐑦𐑒 𐑕𐑰 𐑣𐑦𐑥 '·𐑐𐑸𐑥 𐑢𐑱𐑤.”

𐑞 𐑯𐑧𐑒𐑕𐑑 𐑛𐑱 𐑢𐑪𐑟 𐑦𐑒𐑕𐑰𐑛𐑦𐑙𐑤𐑦 𐑕𐑑𐑦𐑤 𐑯 𐑕𐑳𐑤𐑑𐑮𐑦, 𐑯 𐑢𐑦𐑞 𐑯𐑳𐑔𐑦𐑙 𐑕𐑐𐑧𐑖𐑩𐑤 𐑑 𐑦𐑯𐑜𐑱𐑡 𐑞𐑧𐑥, 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑒𐑮𐑵 𐑒𐑫𐑛 𐑣𐑸𐑛𐑤𐑦 𐑮𐑦𐑟𐑦𐑕𐑑 𐑞 𐑕𐑐𐑧𐑤 𐑝 𐑕𐑤𐑰𐑐 𐑦𐑯𐑛𐑿𐑕𐑑 𐑚𐑲 𐑕𐑳𐑗 𐑩 𐑝𐑱𐑒𐑩𐑯𐑑 𐑕𐑰. 𐑓 𐑞𐑦𐑕 𐑐𐑸𐑑 𐑝 𐑞 ·𐑦𐑯𐑛𐑾𐑯 ·𐑴𐑖𐑩𐑯 𐑔𐑮𐑵 𐑢𐑦𐑗 𐑢𐑰 𐑞𐑧𐑯 𐑢𐑻 𐑝𐑶𐑦𐑡𐑦𐑙 𐑦𐑟 𐑯𐑪𐑑 𐑢𐑪𐑑 𐑢𐑱𐑤𐑥𐑧𐑯 𐑒𐑷𐑤 𐑩 𐑤𐑲𐑝𐑤𐑦 𐑜𐑮𐑬𐑯𐑛; 𐑞𐑨𐑑 𐑦𐑟, 𐑦𐑑 𐑩𐑓𐑹𐑛𐑟 𐑓𐑿𐑼 𐑜𐑤𐑦𐑥𐑐𐑕𐑩𐑟 𐑝 𐑐𐑹𐑐𐑩𐑕𐑩𐑟, 𐑛𐑪𐑤𐑓𐑦𐑯𐑟, 𐑓𐑤𐑲𐑦𐑙-𐑓𐑦𐑖, 𐑯 𐑳𐑞𐑼 𐑝𐑦𐑝𐑱𐑖𐑩𐑕 𐑛𐑧𐑯𐑦𐑟𐑩𐑯𐑟 𐑝 𐑥𐑹 𐑕𐑑𐑻𐑦𐑙 𐑢𐑷𐑑𐑼𐑟, 𐑞𐑨𐑯 𐑞𐑴𐑟 𐑪𐑓 𐑞 ·𐑮𐑰𐑴 𐑛𐑦 𐑤𐑭 ·𐑐𐑤𐑭𐑑𐑩, 𐑹 𐑞 𐑦𐑯-𐑖𐑹 𐑜𐑮𐑬𐑯𐑛 𐑪𐑓 ·𐑐𐑼𐑵.

𐑦𐑑 𐑢𐑪𐑟 𐑥𐑲 𐑑𐑻𐑯 𐑑 𐑕𐑑𐑨𐑯𐑛 𐑨𐑑 𐑞 𐑓𐑹𐑥𐑭𐑕𐑑-𐑣𐑧𐑛; 𐑯 𐑢𐑦𐑞 𐑥𐑲 𐑖𐑴𐑤𐑛𐑼𐑟 𐑤𐑰𐑯𐑦𐑙 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑕𐑤𐑨𐑒𐑩𐑯𐑛 𐑮𐑶𐑩𐑤 𐑖𐑮𐑬𐑛𐑟, 𐑑 𐑯 𐑓𐑮𐑴 𐑲 𐑲𐑛𐑤𐑦 𐑕𐑢𐑱𐑛 𐑦𐑯 𐑢𐑪𐑑 𐑕𐑰𐑥𐑛 𐑩𐑯 𐑦𐑯𐑗𐑭𐑯𐑑𐑩𐑛 𐑺. 𐑯𐑴 𐑮𐑧𐑟𐑩𐑤𐑵𐑖𐑩𐑯 𐑒𐑫𐑛 𐑢𐑦𐑞𐑕𐑑𐑨𐑯𐑛 𐑦𐑑; 𐑦𐑯 𐑞𐑨𐑑 𐑛𐑮𐑰𐑥𐑦 𐑥𐑵𐑛 𐑤𐑵𐑟𐑦𐑙 𐑷𐑤 𐑒𐑪𐑯𐑖𐑩𐑕𐑯𐑩𐑕, 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑥𐑲 𐑕𐑴𐑤 𐑢𐑧𐑯𐑑 𐑬𐑑 𐑝 𐑥𐑲 𐑚𐑪𐑛𐑦; 𐑞𐑴 𐑥𐑲 𐑚𐑪𐑛𐑦 𐑕𐑑𐑦𐑤 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑛 𐑑 𐑕𐑢𐑱 𐑨𐑟 𐑩 𐑐𐑧𐑯𐑡𐑩𐑤𐑩𐑥 𐑢𐑦𐑤, 𐑤𐑪𐑙 𐑭𐑓𐑑𐑼 𐑞 𐑐𐑬𐑼 𐑢𐑦𐑗 𐑓𐑻𐑕𐑑 𐑥𐑵𐑝𐑛 𐑦𐑑 𐑦𐑟 𐑢𐑦𐑞𐑛𐑮𐑷𐑯.

𐑺 𐑓𐑼𐑜𐑧𐑑𐑓𐑩𐑤𐑯𐑩𐑕 𐑷𐑤𐑑𐑩𐑜𐑧𐑞𐑼 𐑒𐑱𐑥 𐑴𐑝𐑼 𐑥𐑰, 𐑲 𐑣𐑨𐑛 𐑯𐑴𐑑𐑦𐑕𐑑 𐑞𐑨𐑑 𐑞 𐑕𐑰𐑥𐑧𐑯 𐑨𐑑 𐑞 𐑥𐑱𐑯 𐑯 𐑥𐑦𐑟𐑩𐑯-𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑢𐑻 𐑷𐑤𐑮𐑧𐑛𐑦 𐑛𐑮𐑬𐑟𐑦. 𐑕𐑴 𐑞𐑨𐑑 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑷𐑤 𐑔𐑮𐑰 𐑝 𐑳𐑕 𐑤𐑲𐑓𐑤𐑩𐑕𐑤𐑦 𐑕𐑢𐑳𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑐𐑸𐑟, 𐑯 𐑓 𐑧𐑝𐑮𐑦 𐑕𐑢𐑦𐑙 𐑞𐑨𐑑 𐑢𐑰 𐑥𐑱𐑛 𐑞𐑺 𐑢𐑪𐑟 𐑩 𐑯𐑪𐑛 𐑓𐑮𐑪𐑥 𐑚𐑦𐑤𐑴 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑤𐑳𐑥𐑚𐑼𐑦𐑙 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯. 𐑞 𐑢𐑱𐑝𐑟, 𐑑𐑵, 𐑯𐑪𐑛𐑩𐑛 𐑞𐑺 𐑦𐑯𐑛𐑩𐑤𐑩𐑯𐑑 𐑒𐑮𐑧𐑕𐑑𐑕; 𐑯 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑢𐑲𐑛 𐑑𐑮𐑭𐑯𐑕 𐑝 𐑞 𐑕𐑰, 𐑰𐑕𐑑 𐑯𐑪𐑛𐑩𐑛 𐑑 𐑢𐑧𐑕𐑑, 𐑯 𐑞 𐑕𐑳𐑯 𐑴𐑝𐑼 𐑷𐑤.

𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑚𐑳𐑚𐑩𐑤𐑟 𐑕𐑰𐑥𐑛 𐑚𐑻𐑕𐑑𐑦𐑙 𐑚𐑦𐑯𐑰𐑔 𐑥𐑲 𐑒𐑤𐑴𐑟𐑛 𐑲𐑟; 𐑤𐑲𐑒 𐑝𐑲𐑕𐑩𐑟 𐑥𐑲 𐑣𐑨𐑯𐑛𐑟 𐑜𐑮𐑭𐑕𐑐𐑑 𐑞 𐑖𐑮𐑬𐑛𐑟; 𐑕𐑳𐑥 𐑦𐑯𐑝𐑦𐑟𐑩𐑚𐑩𐑤, 𐑜𐑮𐑱𐑖𐑩𐑕 𐑱𐑡𐑩𐑯𐑕𐑦 𐑐𐑮𐑦𐑟𐑻𐑝𐑛 𐑥𐑰; 𐑢𐑦𐑞 𐑩 𐑖𐑪𐑒 𐑲 𐑒𐑱𐑥 𐑚𐑨𐑒 𐑑 𐑤𐑲𐑓. 𐑯 𐑤𐑴! 𐑒𐑤𐑴𐑕 𐑳𐑯𐑛𐑼 𐑬𐑼 𐑤𐑰, 𐑯𐑪𐑑 𐑓𐑹𐑑𐑦 𐑓𐑨𐑞𐑩𐑥𐑟 𐑪𐑓, 𐑩 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑤𐑱 𐑮𐑴𐑤𐑦𐑙 𐑦𐑯 𐑞 𐑢𐑷𐑑𐑼 𐑤𐑲𐑒 𐑞 𐑒𐑨𐑐𐑕𐑲𐑟𐑛 𐑣𐑳𐑤 𐑝 𐑩 𐑓𐑮𐑦𐑜𐑩𐑑, 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛, 𐑜𐑤𐑪𐑕𐑦 𐑚𐑨𐑒, 𐑝 𐑩𐑯 ·𐑰𐑔𐑦𐑴𐑐𐑾𐑯 𐑣𐑿, 𐑜𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑦𐑯 𐑞 𐑕𐑳𐑯'𐑟 𐑮𐑱𐑟 𐑤𐑲𐑒 𐑩 𐑥𐑦𐑮𐑼. 𐑚𐑳𐑑 𐑤𐑱𐑟𐑦𐑤𐑦 𐑳𐑯𐑡𐑩𐑤𐑱𐑑𐑦𐑙 𐑦𐑯 𐑞 𐑑𐑮𐑪𐑓 𐑝 𐑞 𐑕𐑰, 𐑯 𐑧𐑝𐑼 𐑯 𐑩𐑯𐑪𐑯 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤𐑦 𐑕𐑐𐑬𐑑𐑦𐑙 𐑣𐑦𐑟 𐑝𐑱𐑐𐑼𐑦 𐑡𐑧𐑑, 𐑞 𐑢𐑱𐑤 𐑤𐑫𐑒𐑑 𐑤𐑲𐑒 𐑩 𐑐𐑹𐑑𐑤𐑦 𐑚𐑻𐑜𐑼 𐑕𐑥𐑴𐑒𐑦𐑙 𐑣𐑦𐑟 𐑐𐑲𐑐 𐑝 𐑩 𐑢𐑹𐑥 𐑭𐑓𐑑𐑼𐑯𐑵𐑯. 𐑚𐑳𐑑 𐑞𐑨𐑑 𐑐𐑲𐑐, 𐑐𐑫𐑼 𐑢𐑱𐑤, 𐑢𐑪𐑟 𐑞𐑲 𐑤𐑭𐑕𐑑. 𐑨𐑟 𐑦𐑓 𐑕𐑑𐑮𐑳𐑒 𐑚𐑲 𐑕𐑳𐑥 𐑦𐑯𐑗𐑭𐑯𐑑𐑼'𐑟 𐑢𐑪𐑯𐑛, 𐑞 𐑕𐑤𐑰𐑐𐑦 𐑖𐑦𐑐 𐑯 𐑧𐑝𐑮𐑦 𐑕𐑤𐑰𐑐𐑼 𐑦𐑯 𐑦𐑑 𐑷𐑤 𐑨𐑑 𐑢𐑳𐑯𐑕 𐑕𐑑𐑸𐑑𐑩𐑛 𐑦𐑯𐑑𐑵 𐑢𐑱𐑒𐑓𐑩𐑤𐑯𐑩𐑕; 𐑯 𐑥𐑹 𐑞𐑨𐑯 𐑩 𐑕𐑒𐑹 𐑝 𐑝𐑶𐑕𐑩𐑟 𐑓𐑮𐑪𐑥 𐑷𐑤 𐑐𐑸𐑑𐑕 𐑝 𐑞 𐑝𐑧𐑕𐑩𐑤, 𐑕𐑦𐑥𐑩𐑤𐑑𐑱𐑯𐑾𐑕𐑤𐑦 𐑢𐑦𐑞 𐑞 𐑔𐑮𐑰 𐑯𐑴𐑑𐑕 𐑓𐑮𐑪𐑥 𐑩𐑤𐑪𐑓𐑑, 𐑖𐑬𐑑𐑩𐑛 𐑓𐑹𐑔 𐑞 𐑩𐑒𐑳𐑕𐑑𐑩𐑥𐑛 𐑒𐑮𐑲, 𐑨𐑟 𐑞 𐑜𐑮𐑱𐑑 𐑓𐑦𐑖 𐑕𐑤𐑴𐑤𐑦 𐑯 𐑮𐑧𐑜𐑿𐑤𐑼𐑤𐑦 𐑕𐑐𐑬𐑑𐑩𐑛 𐑞 𐑕𐑐𐑸𐑒𐑩𐑤𐑦𐑙 𐑚𐑮𐑲𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑺.

“𐑒𐑤𐑽 𐑩𐑢𐑱 𐑞 𐑚𐑴𐑑𐑕! 𐑤𐑳𐑓!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚. 𐑯 𐑴𐑚𐑱𐑦𐑙 𐑣𐑦𐑟 𐑴𐑯 𐑹𐑛𐑼, 𐑣𐑰 𐑛𐑨𐑖𐑑 𐑞 𐑣𐑧𐑤𐑥 𐑛𐑬𐑯 𐑚𐑦𐑓𐑹 𐑞 𐑣𐑧𐑤𐑥𐑟𐑥𐑩𐑯 𐑒𐑫𐑛 𐑣𐑨𐑯𐑛𐑩𐑤 𐑞 𐑕𐑐𐑴𐑒𐑕.

𐑞 𐑕𐑳𐑛𐑩𐑯 𐑧𐑒𐑕𐑒𐑤𐑩𐑥𐑱𐑖𐑩𐑯𐑟 𐑝 𐑞 𐑒𐑮𐑵 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑩𐑤𐑸𐑥𐑛 𐑞 𐑢𐑱𐑤; 𐑯 𐑺 𐑞 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑛𐑬𐑯, 𐑥𐑩𐑡𐑧𐑕𐑑𐑦𐑒𐑤𐑦 𐑑𐑻𐑯𐑦𐑙, 𐑣𐑰 𐑕𐑢𐑨𐑥 𐑩𐑢𐑱 𐑑 𐑞 𐑤𐑰𐑢𐑼𐑛, 𐑚𐑳𐑑 𐑢𐑦𐑞 𐑕𐑳𐑗 𐑩 𐑕𐑑𐑧𐑛𐑦 𐑑𐑮𐑨𐑙𐑒𐑢𐑦𐑤𐑦𐑑𐑦, 𐑯 𐑥𐑱𐑒𐑦𐑙 𐑕𐑴 𐑓𐑿 𐑮𐑦𐑐𐑩𐑤𐑟 𐑨𐑟 𐑣𐑰 𐑕𐑢𐑨𐑥, 𐑞𐑨𐑑 𐑔𐑦𐑙𐑒𐑦𐑙 𐑭𐑓𐑑𐑼 𐑷𐑤 𐑣𐑰 𐑥𐑲𐑑 𐑯𐑪𐑑 𐑨𐑟 𐑘𐑧𐑑 𐑚𐑰 𐑩𐑤𐑸𐑥𐑛, ·𐑱𐑣𐑨𐑚 𐑜𐑱𐑝 𐑹𐑛𐑼𐑟 𐑞𐑨𐑑 𐑯𐑪𐑑 𐑩𐑯 𐑹 𐑖𐑵𐑛 𐑚𐑰 𐑿𐑟𐑛, 𐑯 𐑯𐑴 𐑥𐑨𐑯 𐑥𐑳𐑕𐑑 𐑕𐑐𐑰𐑒 𐑚𐑳𐑑 𐑦𐑯 𐑢𐑦𐑕𐑐𐑼𐑟. 𐑕𐑴 𐑕𐑰𐑑𐑩𐑛 𐑤𐑲𐑒 ·𐑪𐑯𐑑𐑺𐑦𐑴 ·𐑦𐑯𐑛𐑾𐑯𐑟 𐑪𐑯 𐑞 𐑜𐑳𐑯𐑩𐑤𐑟 𐑝 𐑞 𐑚𐑴𐑑𐑕, 𐑢𐑰 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑚𐑳𐑑 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑐𐑨𐑛𐑩𐑤𐑛 𐑩𐑤𐑪𐑙; 𐑞 𐑒𐑭𐑥 𐑯𐑪𐑑 𐑩𐑛𐑥𐑦𐑑𐑦𐑙 𐑝 𐑞 𐑯𐑶𐑟𐑤𐑩𐑕 𐑕𐑱𐑤𐑟 𐑚𐑰𐑦𐑙 𐑕𐑧𐑑. 𐑐𐑮𐑧𐑟𐑩𐑯𐑑𐑤𐑦, 𐑨𐑟 𐑢𐑰 𐑞𐑳𐑕 𐑜𐑤𐑲𐑛𐑩𐑛 𐑦𐑯 𐑗𐑱𐑕, 𐑞 𐑥𐑪𐑯𐑕𐑑𐑼 𐑐𐑻𐑐𐑩𐑯𐑛𐑦𐑒𐑿𐑤𐑼𐑤𐑦 𐑓𐑤𐑦𐑑𐑩𐑛 𐑣𐑦𐑟 𐑑𐑱𐑤 𐑓𐑹𐑑𐑦 𐑓𐑰𐑑 𐑦𐑯𐑑𐑵 𐑞 𐑺, 𐑯 𐑞𐑧𐑯 𐑕𐑨𐑙𐑒 𐑬𐑑 𐑝 𐑕𐑲𐑑 𐑤𐑲𐑒 𐑩 𐑑𐑬𐑼 𐑕𐑢𐑪𐑤𐑴𐑛 𐑳𐑐.

“𐑞𐑺 𐑜𐑴 𐑓𐑤𐑵𐑒𐑕!” 𐑢𐑪𐑟 𐑞 𐑒𐑮𐑲, 𐑩𐑯 𐑩𐑯𐑬𐑯𐑕𐑥𐑩𐑯𐑑 𐑦𐑥𐑰𐑛𐑾𐑑𐑤𐑦 𐑓𐑪𐑤𐑴𐑛 𐑚𐑲 ·𐑕𐑑𐑳𐑚'𐑟 𐑐𐑮𐑩𐑛𐑿𐑕𐑦𐑙 𐑣𐑦𐑟 𐑥𐑨𐑗 𐑯 𐑦𐑜𐑯𐑲𐑑𐑦𐑙 𐑣𐑦𐑟 𐑐𐑲𐑐, 𐑓 𐑯𐑬 𐑩 𐑮𐑧𐑕𐑐𐑲𐑑 𐑢𐑪𐑟 𐑜𐑮𐑭𐑯𐑑𐑩𐑛. 𐑭𐑓𐑑𐑼 𐑞 𐑓𐑫𐑤 𐑦𐑯𐑑𐑼𐑝𐑩𐑤 𐑝 𐑣𐑦𐑟 𐑕𐑬𐑯𐑛𐑦𐑙 𐑣𐑨𐑛 𐑦𐑤𐑨𐑐𐑕𐑑, 𐑞 𐑢𐑱𐑤 𐑮𐑴𐑟 𐑩𐑜𐑱𐑯, 𐑯 𐑚𐑰𐑦𐑙 𐑯𐑬 𐑦𐑯 𐑩𐑛𐑝𐑭𐑯𐑕 𐑝 𐑞 𐑕𐑥𐑴𐑒𐑼'𐑟 𐑚𐑴𐑑, 𐑯 𐑥𐑳𐑗 𐑯𐑽𐑼 𐑑 𐑦𐑑 𐑞𐑨𐑯 𐑑 𐑧𐑯𐑦 𐑝 𐑞 𐑳𐑞𐑼𐑟, ·𐑕𐑑𐑳𐑚 𐑒𐑬𐑯𐑑𐑩𐑛 𐑩𐑐𐑪𐑯 𐑞 𐑪𐑯𐑼 𐑝 𐑞 𐑒𐑨𐑐𐑗𐑼. 𐑦𐑑 𐑢𐑪𐑟 𐑪𐑚𐑝𐑾𐑕, 𐑯𐑬, 𐑞𐑨𐑑 𐑞 𐑢𐑱𐑤 𐑣𐑨𐑛 𐑨𐑑 𐑤𐑧𐑙𐑔 𐑚𐑦𐑒𐑳𐑥 𐑩𐑢𐑺 𐑝 𐑣𐑦𐑟 𐑐𐑼𐑕𐑿𐑼𐑟. 𐑷𐑤 𐑕𐑲𐑤𐑩𐑯𐑕 𐑝 𐑒𐑷𐑖𐑩𐑕𐑯𐑩𐑕 𐑢𐑪𐑟 𐑞𐑺𐑓𐑹 𐑯𐑴 𐑤𐑪𐑙𐑜𐑼 𐑝 𐑿𐑕. 𐑐𐑨𐑛𐑩𐑤𐑟 𐑢𐑻 𐑛𐑮𐑪𐑐𐑑, 𐑯 𐑹𐑟 𐑒𐑱𐑥 𐑤𐑬𐑛𐑤𐑦 𐑦𐑯𐑑𐑵 𐑐𐑤𐑱. 𐑯 𐑕𐑑𐑦𐑤 𐑐𐑳𐑓𐑦𐑙 𐑨𐑑 𐑣𐑦𐑟 𐑐𐑲𐑐, ·𐑕𐑑𐑳𐑚 𐑗𐑽𐑛 𐑪𐑯 𐑣𐑦𐑟 𐑒𐑮𐑵 𐑑 𐑞 𐑩𐑕𐑷𐑤𐑑.

𐑘𐑧𐑕, 𐑩 𐑥𐑲𐑑𐑦 𐑗𐑱𐑯𐑡 𐑣𐑨𐑛 𐑒𐑳𐑥 𐑴𐑝𐑼 𐑞 𐑓𐑦𐑖. 𐑷𐑤 𐑩𐑤𐑲𐑝 𐑑 𐑣𐑦𐑟 𐑡𐑧𐑐𐑼𐑛𐑦, 𐑣𐑰 𐑢𐑪𐑟 𐑜𐑴𐑦𐑙 “𐑣𐑧𐑛 𐑬𐑑”; 𐑞𐑨𐑑 𐑐𐑸𐑑 𐑴𐑚𐑤𐑰𐑒𐑤𐑦 𐑐𐑮𐑩𐑡𐑧𐑒𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑨𐑛 𐑘𐑰𐑕𐑑 𐑢𐑦𐑗 𐑣𐑰 𐑚𐑮𐑵𐑛.*

*𐑦𐑑 𐑢𐑦𐑤 𐑚𐑰 𐑕𐑰𐑯 𐑦𐑯 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑐𐑤𐑱𐑕 𐑝 𐑢𐑪𐑑 𐑩 𐑝𐑧𐑮𐑦 𐑤𐑲𐑑 𐑕𐑳𐑚𐑕𐑑𐑩𐑯𐑕 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑦𐑯𐑑𐑽𐑽 𐑝 𐑞 𐑕𐑐𐑻𐑥 𐑢𐑱𐑤'𐑟 𐑦𐑯𐑹𐑥𐑩𐑕 𐑣𐑧𐑛 𐑒𐑩𐑯𐑕𐑦𐑕𐑑𐑕. 𐑞𐑴 𐑩𐑐𐑨𐑮𐑩𐑯𐑑𐑤𐑦 𐑞 𐑥𐑴𐑕𐑑 𐑥𐑨𐑕𐑦𐑝, 𐑦𐑑 𐑦𐑟 𐑚𐑲 𐑓𐑸 𐑞 𐑥𐑴𐑕𐑑 𐑚𐑶𐑩𐑯𐑑 𐑐𐑸𐑑 𐑩𐑚𐑬𐑑 𐑣𐑦𐑥. 𐑕𐑴 𐑞𐑨𐑑 𐑢𐑦𐑞 𐑰𐑟 𐑣𐑰 𐑧𐑤𐑦𐑝𐑱𐑑𐑕 𐑦𐑑 𐑦𐑯 𐑞 𐑺, 𐑯 𐑦𐑯𐑝𐑺𐑾𐑚𐑤𐑦 𐑛𐑳𐑟 𐑕𐑴 𐑢𐑧𐑯 𐑜𐑴𐑦𐑙 𐑨𐑑 𐑣𐑦𐑟 𐑳𐑑𐑥𐑴𐑕𐑑 𐑕𐑐𐑰𐑛. 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑕𐑳𐑗 𐑦𐑟 𐑞 𐑚𐑮𐑧𐑛𐑔 𐑝 𐑞 𐑳𐑐𐑼 𐑐𐑸𐑑 𐑝 𐑞 𐑓𐑮𐑳𐑯𐑑 𐑝 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑯 𐑕𐑳𐑗 𐑞 𐑑𐑱𐑐𐑼𐑦𐑙 𐑒𐑳𐑑-𐑢𐑷𐑑𐑼 𐑓𐑹𐑥𐑱𐑖𐑩𐑯 𐑝 𐑞 𐑤𐑴𐑼 𐑐𐑸𐑑, 𐑞𐑨𐑑 𐑚𐑲 𐑴𐑚𐑤𐑰𐑒𐑤𐑦 𐑧𐑤𐑦𐑝𐑱𐑑𐑦𐑙 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑣𐑰 𐑞𐑺𐑚𐑲 𐑥𐑱 𐑚𐑰 𐑕𐑧𐑛 𐑑 𐑑𐑮𐑨𐑯𐑕𐑓𐑹𐑥 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑓𐑮𐑪𐑥 𐑩 𐑚𐑤𐑳𐑓-𐑚𐑬𐑛 𐑕𐑤𐑳𐑜𐑦𐑖 𐑜𐑨𐑤𐑦𐑪𐑑 𐑦𐑯𐑑𐑵 𐑩 𐑖𐑸𐑐𐑐𐑶𐑯𐑑𐑩𐑛 ·𐑯𐑿 ·𐑘𐑹𐑒 𐑐𐑲𐑤𐑩𐑑-𐑚𐑴𐑑.

“𐑕𐑑𐑸𐑑 𐑣𐑻, 𐑕𐑑𐑸𐑑 𐑣𐑻, 𐑥𐑲 𐑥𐑧𐑯! 𐑛𐑴𐑯𐑑 𐑣𐑳𐑮𐑦 𐑘𐑹𐑕𐑧𐑤𐑝𐑟; 𐑑𐑱𐑒 𐑐𐑤𐑧𐑯𐑑𐑦 𐑝 𐑑𐑲𐑥—𐑚𐑳𐑑 𐑕𐑑𐑸𐑑 𐑣𐑻; 𐑕𐑑𐑸𐑑 𐑣𐑻 𐑤𐑲𐑒 𐑔𐑳𐑯𐑛𐑼-𐑒𐑤𐑨𐑐𐑕, 𐑞𐑨𐑑'𐑕 𐑷𐑤,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, 𐑕𐑐𐑤𐑳𐑑𐑼𐑦𐑙 𐑬𐑑 𐑞 𐑕𐑥𐑴𐑒 𐑨𐑟 𐑣𐑰 𐑕𐑐𐑴𐑒. “𐑕𐑑𐑸𐑑 𐑣𐑻, 𐑯𐑬; 𐑜𐑦𐑝 '𐑩𐑥 𐑞 𐑤𐑪𐑙 𐑯 𐑕𐑑𐑮𐑪𐑙 𐑕𐑑𐑮𐑴𐑒, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴. 𐑕𐑑𐑸𐑑 𐑣𐑻, ·𐑑𐑨𐑖, 𐑥𐑲 𐑚𐑶—𐑕𐑑𐑸𐑑 𐑣𐑻, 𐑷𐑤; 𐑚𐑳𐑑 𐑒𐑰𐑐 𐑒𐑵𐑤, 𐑒𐑰𐑐 𐑒𐑵𐑤—𐑒𐑿𐑒𐑳𐑥𐑚𐑼𐑟 𐑦𐑟 𐑞 𐑢𐑻𐑛—𐑰𐑟𐑦, 𐑰𐑟𐑦—𐑴𐑯𐑤𐑦 𐑕𐑑𐑸𐑑 𐑣𐑻 𐑤𐑲𐑒 𐑜𐑮𐑦𐑥 𐑛𐑧𐑔 𐑯 𐑜𐑮𐑦𐑯𐑦𐑙 𐑛𐑧𐑝𐑩𐑤𐑟, 𐑯 𐑮𐑱𐑟 𐑞 𐑚𐑧𐑮𐑦𐑛 𐑛𐑧𐑛 𐑐𐑻𐑐𐑩𐑯𐑛𐑦𐑒𐑿𐑤𐑼 𐑬𐑑 𐑝 𐑞𐑺 𐑜𐑮𐑱𐑝𐑟, 𐑚𐑶𐑟—𐑞𐑨𐑑'𐑕 𐑷𐑤. 𐑕𐑑𐑸𐑑 𐑣𐑻!”

“𐑢𐑵-𐑣𐑵! ·𐑢𐑭-𐑣𐑰!” 𐑕𐑒𐑮𐑰𐑥𐑛 𐑞 ·𐑜𐑱-·𐑣𐑧𐑛𐑼 𐑦𐑯 𐑮𐑦𐑐𐑤𐑲, 𐑮𐑱𐑟𐑦𐑙 𐑕𐑳𐑥 𐑴𐑤𐑛 𐑢𐑹-𐑢𐑵𐑐 𐑑 𐑞 𐑕𐑒𐑲𐑟; 𐑨𐑟 𐑧𐑝𐑮𐑦 𐑹𐑟𐑥𐑩𐑯 𐑦𐑯 𐑞 𐑕𐑑𐑮𐑱𐑯𐑛 𐑚𐑴𐑑 𐑦𐑯𐑝𐑪𐑤𐑩𐑯𐑑𐑺𐑦𐑤𐑦 𐑚𐑬𐑯𐑕𐑑 𐑓𐑹𐑢𐑼𐑛 𐑢𐑦𐑞 𐑞 𐑢𐑳𐑯 𐑑𐑮𐑦𐑥𐑧𐑯𐑛𐑩𐑕 𐑤𐑰𐑛𐑦𐑙 𐑕𐑑𐑮𐑴𐑒 𐑢𐑦𐑗 𐑞 𐑰𐑜𐑼 ·𐑦𐑯𐑛𐑾𐑯 𐑜𐑱𐑝.

𐑚𐑳𐑑 𐑣𐑦𐑟 𐑢𐑲𐑤𐑛 𐑕𐑒𐑮𐑰𐑥𐑟 𐑢𐑻 𐑭𐑯𐑕𐑼𐑛 𐑚𐑲 𐑳𐑞𐑼𐑟 𐑒𐑢𐑲𐑑 𐑨𐑟 𐑢𐑲𐑤𐑛. “𐑒𐑰-𐑣𐑰! 𐑒𐑰-𐑣𐑰!” 𐑘𐑧𐑤𐑛 ·𐑛𐑨𐑜𐑵, 𐑕𐑑𐑮𐑱𐑯𐑦𐑙 𐑓𐑹𐑢𐑼𐑛𐑟 𐑯 𐑚𐑨𐑒𐑢𐑼𐑛𐑟 𐑪𐑯 𐑣𐑦𐑟 𐑕𐑰𐑑, 𐑤𐑲𐑒 𐑩 𐑐𐑱𐑕𐑦𐑙 𐑑𐑲𐑜𐑼 𐑦𐑯 𐑣𐑦𐑟 𐑒𐑱𐑡.

“𐑒𐑭-𐑤𐑭! ·𐑜𐑵-𐑤𐑵!” 𐑣𐑬𐑤𐑛 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑨𐑟 𐑦𐑓 𐑕𐑥𐑨𐑒𐑦𐑙 𐑣𐑦𐑟 𐑤𐑦𐑐𐑕 𐑴𐑝𐑼 𐑩 𐑥𐑬𐑔𐑓𐑫𐑤 𐑝 ·𐑜𐑮𐑧𐑯𐑩𐑛𐑽'𐑟 𐑕𐑑𐑱𐑒. 𐑯 𐑞𐑳𐑕 𐑢𐑦𐑞 𐑹𐑟 𐑯 𐑘𐑧𐑤𐑟 𐑞 𐑒𐑰𐑤𐑟 𐑒𐑳𐑑 𐑞 𐑕𐑰. 𐑥𐑰𐑯𐑢𐑲𐑤, ·𐑕𐑑𐑳𐑚 𐑮𐑦𐑑𐑱𐑯𐑦𐑙 𐑣𐑦𐑟 𐑐𐑤𐑱𐑕 𐑦𐑯 𐑞 𐑝𐑨𐑯, 𐑕𐑑𐑦𐑤 𐑦𐑯𐑒𐑳𐑮𐑦𐑡𐑛 𐑣𐑦𐑟 𐑥𐑧𐑯 𐑑 𐑞 𐑪𐑯𐑕𐑧𐑑, 𐑷𐑤 𐑞 𐑢𐑲𐑤 𐑐𐑳𐑓𐑦𐑙 𐑞 𐑕𐑥𐑴𐑒 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑥𐑬𐑔. 𐑤𐑲𐑒 𐑛𐑧𐑕𐑐𐑼𐑭𐑛𐑴𐑟 𐑞𐑱 𐑑𐑳𐑜𐑛 𐑯 𐑞𐑱 𐑕𐑑𐑮𐑱𐑯𐑛, 𐑑𐑦𐑤 𐑞 𐑢𐑧𐑤𐑒𐑩𐑥 𐑒𐑮𐑲 𐑢𐑪𐑟 𐑣𐑻𐑛—“𐑕𐑑𐑨𐑯𐑛 𐑳𐑐, ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴!—𐑜𐑦𐑝 𐑦𐑑 𐑑 𐑣𐑦𐑥!” 𐑞 𐑣𐑸𐑐𐑵𐑯 𐑢𐑪𐑟 𐑣𐑻𐑤𐑛. “𐑕𐑑𐑻𐑯 𐑷𐑤!” 𐑞 𐑹𐑟𐑥𐑧𐑯 𐑚𐑨𐑒𐑑 𐑢𐑷𐑑𐑼; 𐑞 𐑕𐑱𐑥 𐑥𐑴𐑥𐑧𐑯𐑑 𐑕𐑳𐑥𐑔𐑦𐑙 𐑢𐑧𐑯𐑑 𐑣𐑪𐑑 𐑯 𐑣𐑦𐑕𐑦𐑙 𐑩𐑤𐑪𐑙 𐑧𐑝𐑮𐑦 𐑢𐑳𐑯 𐑝 𐑞𐑺 𐑮𐑦𐑕𐑑𐑕. 𐑦𐑑 𐑢𐑪𐑟 𐑞 𐑥𐑨𐑡𐑦𐑒𐑩𐑤 𐑤𐑲𐑯. 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑚𐑦𐑓𐑹, ·𐑕𐑑𐑳𐑚 𐑣𐑨𐑛 𐑕𐑢𐑦𐑓𐑑𐑤𐑦 𐑒𐑷𐑑 𐑑𐑵 𐑩𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑑𐑻𐑯𐑟 𐑢𐑦𐑞 𐑦𐑑 𐑮𐑬𐑯𐑛 𐑞 𐑤𐑪𐑜𐑼𐑣𐑧𐑛, 𐑢𐑧𐑯𐑕, 𐑚𐑲 𐑮𐑰𐑟𐑩𐑯 𐑝 𐑦𐑑𐑕 𐑦𐑯𐑒𐑮𐑰𐑕𐑑 𐑮𐑨𐑐𐑦𐑛 𐑕𐑻𐑒𐑩𐑤𐑦𐑙𐑟, 𐑩 𐑣𐑧𐑥𐑐𐑩𐑯 𐑚𐑤𐑵 𐑕𐑥𐑴𐑒 𐑯𐑬 𐑡𐑧𐑑𐑩𐑛 𐑳𐑐 𐑯 𐑥𐑦𐑙𐑜𐑩𐑤𐑛 𐑢𐑦𐑞 𐑞 𐑕𐑑𐑧𐑛𐑦 𐑓𐑿𐑥𐑟 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑐𐑲𐑐. 𐑨𐑟 𐑞 𐑤𐑲𐑯 𐑐𐑭𐑕𐑑 𐑮𐑬𐑯𐑛 𐑯 𐑮𐑬𐑯𐑛 𐑞 𐑤𐑪𐑜𐑼𐑣𐑧𐑛; 𐑕𐑴 𐑷𐑤𐑕𐑴, 𐑡𐑳𐑕𐑑 𐑚𐑦𐑓𐑹 𐑮𐑰𐑗𐑦𐑙 𐑞𐑨𐑑 𐑐𐑶𐑯𐑑, 𐑦𐑑 𐑚𐑤𐑦𐑕𐑑𐑼𐑦𐑙𐑤𐑦 𐑐𐑭𐑕𐑑 𐑔𐑮𐑵 𐑯 𐑔𐑮𐑵 𐑚𐑴𐑔 𐑝 ·𐑕𐑑𐑳𐑚'𐑟 𐑣𐑨𐑯𐑛𐑟, 𐑓𐑮𐑪𐑥 𐑢𐑦𐑗 𐑞 𐑣𐑨𐑯𐑛-𐑒𐑤𐑪𐑔𐑕, 𐑹 𐑕𐑒𐑢𐑺𐑟 𐑝 𐑒𐑢𐑦𐑤𐑑𐑩𐑛 𐑒𐑨𐑯𐑝𐑩𐑕 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑢𐑹𐑯 𐑨𐑑 𐑞𐑰𐑟 𐑑𐑲𐑥𐑟, 𐑣𐑨𐑛 𐑨𐑒𐑕𐑦𐑛𐑧𐑯𐑑𐑩𐑤𐑦 𐑛𐑮𐑪𐑐𐑑. 𐑦𐑑 𐑢𐑪𐑟 𐑤𐑲𐑒 𐑣𐑴𐑤𐑛𐑦𐑙 𐑩𐑯 𐑧𐑯𐑩𐑥𐑦'𐑟 𐑖𐑸𐑐 𐑑𐑵-𐑧𐑡𐑛 𐑕𐑹𐑛 𐑚𐑲 𐑞 𐑚𐑤𐑱𐑛, 𐑯 𐑞𐑨𐑑 𐑧𐑯𐑩𐑥𐑦 𐑷𐑤 𐑞 𐑑𐑲𐑥 𐑕𐑑𐑮𐑲𐑝𐑦𐑙 𐑑 𐑮𐑧𐑕𐑑 𐑦𐑑 𐑬𐑑 𐑝 𐑘𐑹 𐑒𐑤𐑳𐑗.

“𐑢𐑧𐑑 𐑞 𐑤𐑲𐑯! 𐑢𐑧𐑑 𐑞 𐑤𐑲𐑯!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚 𐑑 𐑞 𐑑𐑳𐑚 𐑹𐑟𐑥𐑩𐑯 (𐑣𐑦𐑥 𐑕𐑰𐑑𐑩𐑛 𐑚𐑲 𐑞 𐑑𐑳𐑚) 𐑣𐑵, 𐑕𐑯𐑨𐑗𐑦𐑙 𐑪𐑓 𐑣𐑦𐑟 𐑣𐑨𐑑, 𐑛𐑨𐑖𐑑 𐑕𐑰-𐑢𐑷𐑑𐑼 𐑦𐑯𐑑𐑵 𐑦𐑑.* 𐑥𐑹 𐑑𐑻𐑯𐑟 𐑢𐑻 𐑑𐑱𐑒𐑩𐑯, 𐑕𐑴 𐑞𐑨𐑑 𐑞 𐑤𐑲𐑯 𐑚𐑦𐑜𐑨𐑯 𐑣𐑴𐑤𐑛𐑦𐑙 𐑦𐑑𐑕 𐑐𐑤𐑱𐑕. 𐑞 𐑚𐑴𐑑 𐑯𐑬 𐑓𐑤𐑵 𐑔𐑮𐑵 𐑞 𐑚𐑶𐑤𐑦𐑙 𐑢𐑷𐑑𐑼 𐑤𐑲𐑒 𐑩 𐑖𐑸𐑒 𐑷𐑤 𐑓𐑦𐑯𐑟. 𐑕𐑑𐑳𐑚 𐑯 ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴 𐑣𐑽 𐑗𐑱𐑯𐑡𐑛 𐑐𐑤𐑱𐑕𐑩𐑟—𐑕𐑑𐑧𐑥 𐑓 𐑕𐑑𐑻𐑯—𐑩 𐑕𐑑𐑨𐑜𐑼𐑦𐑙 𐑚𐑦𐑟𐑯𐑩𐑕 𐑑𐑮𐑵𐑤𐑦 𐑦𐑯 𐑞𐑨𐑑 𐑮𐑪𐑒𐑦𐑙 𐑒𐑩𐑥𐑴𐑖𐑩𐑯.
