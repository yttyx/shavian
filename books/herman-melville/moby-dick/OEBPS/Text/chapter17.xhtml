·𐑗𐑨𐑐𐑑𐑼 17. 𐑞 ·𐑮𐑭𐑥𐑩𐑛𐑭𐑯.

𐑨𐑟 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 ·𐑮𐑭𐑥𐑩𐑛𐑭𐑯, 𐑹 ·𐑓𐑭𐑕𐑑𐑦𐑙 𐑯 ·𐑣𐑿𐑥𐑦𐑤𐑦𐑱𐑖𐑩𐑯, 𐑢𐑪𐑟 𐑑 𐑒𐑩𐑯𐑑𐑦𐑯𐑿 𐑷𐑤 𐑛𐑱, 𐑲 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑗𐑵𐑟 𐑑 𐑛𐑦𐑕𐑑𐑻𐑚 𐑣𐑦𐑥 𐑑𐑦𐑤 𐑑𐑩𐑢𐑹𐑛𐑟 𐑯𐑲𐑑-𐑓𐑷𐑤; 𐑓 𐑲 𐑗𐑧𐑮𐑦𐑖 𐑞 𐑜𐑮𐑱𐑑𐑩𐑕𐑑 𐑮𐑦𐑕𐑐𐑧𐑒𐑑 𐑑𐑩𐑢𐑹𐑛𐑟 𐑧𐑝𐑮𐑦𐑚𐑪𐑛𐑦'𐑟 𐑮𐑦𐑤𐑦𐑡𐑩𐑕 𐑪𐑚𐑤𐑦𐑜𐑱𐑖𐑩𐑯𐑟, 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛 𐑣𐑬 𐑒𐑪𐑥𐑦𐑒𐑩𐑤, 𐑯 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑓𐑲𐑯𐑛 𐑦𐑑 𐑦𐑯 𐑥𐑲 𐑣𐑸𐑑 𐑑 𐑳𐑯𐑛𐑼𐑝𐑨𐑤𐑿 𐑰𐑝𐑩𐑯 𐑩 𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑖𐑩𐑯 𐑝 𐑨𐑯𐑑𐑕 𐑢𐑻𐑖𐑦𐑐𐑦𐑙 𐑩 𐑑𐑴𐑛-𐑕𐑑𐑵𐑤; 𐑹 𐑞𐑴𐑟 𐑳𐑞𐑼 𐑒𐑮𐑰𐑗𐑼𐑟 𐑦𐑯 𐑕𐑻𐑑𐑩𐑯 𐑐𐑸𐑑𐑕 𐑝 𐑬𐑼 𐑻𐑔, 𐑣𐑵 𐑢𐑦𐑞 𐑩 𐑛𐑦𐑜𐑮𐑰 𐑝 𐑓𐑫𐑑𐑥𐑩𐑯𐑦𐑟𐑩𐑥 𐑒𐑢𐑲𐑑 𐑳𐑯𐑐𐑮𐑧𐑕𐑦𐑛𐑧𐑯𐑑𐑩𐑛 𐑦𐑯 𐑳𐑞𐑼 𐑐𐑤𐑨𐑯𐑩𐑑𐑕, 𐑚𐑬 𐑛𐑬𐑯 𐑚𐑦𐑓𐑹 𐑞 𐑑𐑹𐑕𐑴 𐑝 𐑩 𐑛𐑦𐑕𐑰𐑕𐑑 𐑤𐑨𐑯𐑛𐑩𐑛 𐑐𐑮𐑩𐑐𐑮𐑲𐑩𐑑𐑼 𐑥𐑽𐑤𐑦 𐑪𐑯 𐑩𐑒𐑬𐑯𐑑 𐑝 𐑞 𐑦𐑯𐑹𐑛𐑦𐑯𐑩𐑑 𐑐𐑩𐑟𐑧𐑖𐑩𐑯𐑟 𐑘𐑧𐑑 𐑴𐑯𐑛 𐑯 𐑮𐑧𐑯𐑑𐑩𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑯𐑱𐑥.

𐑲 𐑕𐑱, 𐑢𐑰 𐑜𐑫𐑛 ·𐑐𐑮𐑧𐑟𐑚𐑦𐑑𐑽𐑾𐑯 ·𐑒𐑮𐑦𐑕𐑗𐑩𐑯𐑟 𐑖𐑵𐑛 𐑚𐑰 𐑗𐑨𐑮𐑦𐑑𐑩𐑚𐑩𐑤 𐑦𐑯 𐑞𐑰𐑟 𐑔𐑦𐑙𐑟, 𐑯 𐑯𐑪𐑑 𐑓𐑨𐑯𐑕𐑦 𐑬𐑼𐑕𐑧𐑤𐑝𐑟 𐑕𐑴 𐑝𐑭𐑕𐑑𐑤𐑦 𐑕𐑵𐑐𐑽𐑽 𐑑 𐑳𐑞𐑼 𐑥𐑹𐑑𐑩𐑤𐑟, 𐑐𐑱𐑜𐑩𐑯𐑟 𐑯 𐑢𐑪𐑑 𐑯𐑪𐑑, 𐑚𐑦𐑒𐑪𐑟 𐑝 𐑞𐑺 𐑣𐑭𐑓-𐑒𐑮𐑱𐑟𐑦 𐑒𐑩𐑯𐑕𐑰𐑑𐑕 𐑪𐑯 𐑞𐑰𐑟 𐑕𐑳𐑚𐑡𐑧𐑒𐑑𐑕. 𐑞𐑺 𐑢𐑪𐑟 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑯𐑬, 𐑕𐑻𐑑𐑩𐑯𐑤𐑦 𐑧𐑯𐑑𐑼𐑑𐑱𐑯𐑦𐑙 𐑞 𐑥𐑴𐑕𐑑 𐑩𐑚𐑕𐑻𐑛 𐑯𐑴𐑖𐑩𐑯𐑟 𐑩𐑚𐑬𐑑 ·𐑘𐑴𐑡𐑴 𐑯 𐑣𐑦𐑟 ·𐑮𐑭𐑥𐑩𐑛𐑭𐑯;—𐑚𐑳𐑑 𐑢𐑪𐑑 𐑝 𐑞𐑨𐑑? ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑔𐑷𐑑 𐑣𐑰 𐑯𐑿 𐑢𐑪𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑩𐑚𐑬𐑑, 𐑲 𐑕𐑩𐑐𐑴𐑟; 𐑣𐑰 𐑕𐑰𐑥𐑛 𐑑 𐑚𐑰 𐑒𐑩𐑯𐑑𐑧𐑯𐑑; 𐑯 𐑞𐑺 𐑤𐑧𐑑 𐑣𐑦𐑥 𐑮𐑧𐑕𐑑. 𐑷𐑤 𐑬𐑼 𐑸𐑜𐑿𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑥 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑩𐑝𐑱𐑤; 𐑤𐑧𐑑 𐑣𐑦𐑥 𐑚𐑰, 𐑲 𐑕𐑱: 𐑯 ·𐑣𐑧𐑝𐑩𐑯 𐑣𐑨𐑝 𐑥𐑻𐑕𐑦 𐑪𐑯 𐑳𐑕 𐑷𐑤—·𐑐𐑮𐑧𐑟𐑚𐑦𐑑𐑽𐑾𐑯𐑟 𐑯 ·𐑐𐑱𐑜𐑩𐑯𐑟 𐑩𐑤𐑲𐑒—𐑓 𐑢𐑰 𐑸 𐑷𐑤 𐑕𐑳𐑥𐑣𐑬 𐑛𐑮𐑧𐑛𐑓𐑩𐑤𐑦 𐑒𐑮𐑨𐑒𐑑 𐑩𐑚𐑬𐑑 𐑞 𐑣𐑧𐑛, 𐑯 𐑕𐑨𐑛𐑤𐑦 𐑯𐑰𐑛 𐑥𐑧𐑯𐑛𐑦𐑙.

𐑑𐑩𐑢𐑹𐑛𐑟 𐑰𐑝𐑯𐑦𐑙, 𐑢𐑧𐑯 𐑲 𐑓𐑧𐑤𐑑 𐑩𐑖𐑫𐑼𐑛 𐑞𐑨𐑑 𐑷𐑤 𐑣𐑦𐑟 𐑐𐑼𐑓𐑹𐑥𐑩𐑯𐑕𐑩𐑟 𐑯 𐑮𐑦𐑗𐑵𐑩𐑤𐑟 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑴𐑝𐑼, 𐑲 𐑢𐑧𐑯𐑑 𐑳𐑐 𐑑 𐑣𐑦𐑟 𐑮𐑵𐑥 𐑯 𐑯𐑪𐑒𐑑 𐑨𐑑 𐑞 𐑛𐑹; 𐑚𐑳𐑑 𐑯𐑴 𐑭𐑯𐑕𐑼. 𐑲 𐑑𐑮𐑲𐑛 𐑑 𐑴𐑐𐑩𐑯 𐑦𐑑, 𐑚𐑳𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑓𐑭𐑕𐑩𐑯𐑛 𐑦𐑯𐑕𐑲𐑛. “·𐑒𐑢𐑰𐑒𐑢𐑧𐑜,” 𐑕𐑧𐑛 𐑲 𐑕𐑪𐑓𐑑𐑤𐑦 𐑔𐑮𐑵 𐑞 𐑒𐑰-𐑣𐑴𐑤:—𐑷𐑤 𐑕𐑲𐑤𐑩𐑯𐑑. “𐑲 𐑕𐑱, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜! 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑐𐑰𐑒? 𐑦𐑑'𐑕 𐑲—·𐑦𐑖𐑥𐑱𐑩𐑤.” 𐑚𐑳𐑑 𐑷𐑤 𐑮𐑦𐑥𐑱𐑯𐑛 𐑕𐑑𐑦𐑤 𐑨𐑟 𐑚𐑦𐑓𐑹. 𐑲 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑜𐑮𐑴 𐑩𐑤𐑸𐑥𐑛. 𐑲 𐑣𐑨𐑛 𐑩𐑤𐑬𐑛 𐑣𐑦𐑥 𐑕𐑳𐑗 𐑩𐑚𐑳𐑯𐑛𐑩𐑯𐑑 𐑑𐑲𐑥; 𐑲 𐑔𐑷𐑑 𐑣𐑰 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑣𐑨𐑛 𐑩𐑯 𐑨𐑐𐑩𐑐𐑤𐑧𐑒𐑑𐑦𐑒 𐑓𐑦𐑑. 𐑲 𐑤𐑫𐑒𐑑 𐑔𐑮𐑵 𐑞 𐑒𐑰-𐑣𐑴𐑤; 𐑚𐑳𐑑 𐑞 𐑛𐑹 𐑴𐑐𐑩𐑯𐑦𐑙 𐑦𐑯𐑑𐑵 𐑩𐑯 𐑪𐑛 𐑒𐑹𐑯𐑼 𐑝 𐑞 𐑮𐑵𐑥, 𐑞 𐑒𐑰-𐑣𐑴𐑤 𐑐𐑮𐑪𐑕𐑐𐑧𐑒𐑑 𐑢𐑪𐑟 𐑚𐑳𐑑 𐑩 𐑒𐑮𐑫𐑒𐑩𐑛 𐑯 𐑕𐑦𐑯𐑦𐑕𐑑𐑼 𐑢𐑳𐑯. 𐑲 𐑒𐑫𐑛 𐑴𐑯𐑤𐑦 𐑕𐑰 𐑐𐑸𐑑 𐑝 𐑞 𐑓𐑫𐑑-𐑚𐑹𐑛 𐑝 𐑞 𐑚𐑧𐑛 𐑯 𐑩 𐑤𐑲𐑯 𐑝 𐑞 𐑢𐑷𐑤, 𐑚𐑳𐑑 𐑯𐑳𐑔𐑦𐑙 𐑥𐑹. 𐑲 𐑢𐑪𐑟 𐑕𐑼𐑐𐑮𐑲𐑟𐑛 𐑑 𐑚𐑦𐑣𐑴𐑤𐑛 𐑮𐑧𐑕𐑑𐑦𐑙 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑢𐑷𐑤 𐑞 𐑢𐑫𐑛𐑩𐑯 𐑖𐑭𐑓𐑑 𐑝 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑣𐑸𐑐𐑵𐑯, 𐑢𐑦𐑗 𐑞 𐑤𐑨𐑯𐑛𐑤𐑱𐑛𐑦 𐑞 𐑰𐑝𐑯𐑦𐑙 𐑐𐑮𐑰𐑝𐑾𐑕 𐑣𐑨𐑛 𐑑𐑱𐑒𐑩𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑚𐑦𐑓𐑹 𐑬𐑼 𐑥𐑬𐑯𐑑𐑦𐑙 𐑑 𐑞 𐑗𐑱𐑥𐑚𐑼. 𐑞𐑨𐑑'𐑕 𐑕𐑑𐑮𐑱𐑯𐑡, 𐑔𐑷𐑑 𐑲; 𐑚𐑳𐑑 𐑨𐑑 𐑧𐑯𐑦 𐑮𐑱𐑑, 𐑕𐑦𐑯𐑕 𐑞 𐑣𐑸𐑐𐑵𐑯 𐑕𐑑𐑨𐑯𐑛𐑟 𐑘𐑪𐑯𐑛𐑼, 𐑯 𐑣𐑰 𐑕𐑧𐑤𐑛𐑩𐑥 𐑹 𐑯𐑧𐑝𐑼 𐑜𐑴𐑟 𐑩𐑚𐑮𐑷𐑛 𐑢𐑦𐑞𐑬𐑑 𐑦𐑑, 𐑞𐑺𐑓𐑹 𐑣𐑰 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑦𐑯𐑕𐑲𐑛 𐑣𐑽, 𐑯 𐑯𐑴 𐑐𐑪𐑕𐑩𐑚𐑩𐑤 𐑥𐑦𐑕𐑑𐑱𐑒.

“·𐑒𐑢𐑰𐑒𐑢𐑧𐑜!—·𐑒𐑢𐑰𐑒𐑢𐑧𐑜!”—𐑷𐑤 𐑕𐑑𐑦𐑤. 𐑕𐑳𐑥𐑔𐑦𐑙 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑣𐑨𐑐𐑩𐑯𐑛. 𐑨𐑐𐑩𐑐𐑤𐑧𐑒𐑕𐑦! 𐑲 𐑑𐑮𐑲𐑛 𐑑 𐑚𐑻𐑕𐑑 𐑴𐑐𐑩𐑯 𐑞 𐑛𐑹; 𐑚𐑳𐑑 𐑦𐑑 𐑕𐑑𐑳𐑚𐑼𐑯𐑤𐑦 𐑮𐑦𐑟𐑦𐑕𐑑𐑩𐑛. 𐑮𐑳𐑯𐑦𐑙 𐑛𐑬𐑯 𐑕𐑑𐑺𐑟, 𐑲 𐑒𐑢𐑦𐑒𐑤𐑦 𐑕𐑑𐑱𐑑𐑩𐑛 𐑥𐑲 𐑕𐑩𐑕𐑐𐑦𐑖𐑩𐑯𐑟 𐑑 𐑞 𐑓𐑻𐑕𐑑 𐑐𐑻𐑕𐑩𐑯 𐑲 𐑥𐑧𐑑—𐑞 𐑗𐑱𐑥𐑚𐑼-𐑥𐑱𐑛. “𐑤𐑭! 𐑤𐑭!” 𐑖𐑰 𐑒𐑮𐑲𐑛, “𐑲 𐑔𐑷𐑑 𐑕𐑳𐑥𐑔𐑦𐑙 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑞 𐑥𐑨𐑑𐑼. 𐑲 𐑢𐑧𐑯𐑑 𐑑 𐑥𐑱𐑒 𐑞 𐑚𐑧𐑛 𐑭𐑓𐑑𐑼 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑, 𐑯 𐑞 𐑛𐑹 𐑢𐑪𐑟 𐑤𐑪𐑒𐑑; 𐑯 𐑯𐑪𐑑 𐑩 𐑥𐑬𐑕 𐑑 𐑚𐑰 𐑣𐑻𐑛; 𐑯 𐑦𐑑'𐑕 𐑚𐑰𐑯 𐑡𐑳𐑕𐑑 𐑕𐑴 𐑕𐑲𐑤𐑩𐑯𐑑 𐑧𐑝𐑼 𐑕𐑦𐑯𐑕. 𐑚𐑳𐑑 𐑲 𐑔𐑷𐑑, 𐑥𐑱 𐑚𐑰, 𐑿 𐑣𐑨𐑛 𐑚𐑴𐑔 𐑜𐑪𐑯 𐑪𐑓 𐑯 𐑤𐑪𐑒𐑑 𐑘𐑹 𐑚𐑨𐑜𐑦𐑡 𐑦𐑯 𐑓 𐑕𐑱𐑓 𐑒𐑰𐑐𐑦𐑙. 𐑤𐑭! 𐑤𐑭, 𐑥𐑨𐑥!—𐑥𐑦𐑕𐑑𐑮𐑩𐑕! 𐑥𐑻𐑛𐑼! ·𐑥𐑕𐑟 ·𐑣𐑳𐑕𐑦! 𐑨𐑐𐑩𐑐𐑤𐑧𐑒𐑕𐑦!”—𐑯 𐑢𐑦𐑞 𐑞𐑰𐑟 𐑒𐑮𐑲𐑟, 𐑖𐑰 𐑮𐑨𐑯 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑒𐑦𐑗𐑦𐑯, 𐑲 𐑓𐑪𐑤𐑴𐑦𐑙.

·𐑥𐑕𐑟 ·𐑣𐑳𐑕𐑦 𐑕𐑵𐑯 𐑩𐑐𐑽𐑛, 𐑢𐑦𐑞 𐑩 𐑥𐑳𐑕𐑑𐑼𐑛-𐑐𐑪𐑑 𐑦𐑯 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑯 𐑩 𐑝𐑦𐑯𐑦𐑜𐑼-𐑒𐑮𐑵𐑩𐑑 𐑦𐑯 𐑞 𐑳𐑞𐑼, 𐑣𐑨𐑝𐑦𐑙 𐑡𐑳𐑕𐑑 𐑚𐑮𐑴𐑒𐑩𐑯 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞 𐑪𐑒𐑿𐑐𐑱𐑖𐑩𐑯 𐑝 𐑩𐑑𐑧𐑯𐑛𐑦𐑙 𐑑 𐑞 𐑒𐑭𐑕𐑑𐑼𐑟, 𐑯 𐑕𐑒𐑴𐑤𐑛𐑦𐑙 𐑣𐑻 𐑤𐑦𐑑𐑩𐑤 𐑚𐑤𐑨𐑒 𐑚𐑶 𐑥𐑰𐑯𐑑𐑲𐑥.

“𐑢𐑫𐑛-𐑣𐑬𐑕!” 𐑒𐑮𐑲𐑛 𐑲, “𐑢𐑦𐑗 𐑢𐑱 𐑑 𐑦𐑑? 𐑮𐑳𐑯 𐑓 ·𐑜𐑪𐑛'𐑟 𐑕𐑱𐑒, 𐑯 𐑓𐑧𐑗 𐑕𐑳𐑥𐑔𐑦𐑙 𐑑 𐑐𐑮𐑲 𐑴𐑐𐑩𐑯 𐑞 𐑛𐑹—𐑞 𐑨𐑒𐑕!—𐑞 𐑨𐑒𐑕! 𐑣𐑰'𐑟 𐑣𐑨𐑛 𐑩 𐑕𐑑𐑮𐑴𐑒; 𐑛𐑦𐑐𐑧𐑯𐑛 𐑩𐑐𐑪𐑯 𐑦𐑑!”—𐑯 𐑕𐑴 𐑕𐑱𐑦𐑙 𐑲 𐑢𐑪𐑟 𐑳𐑯𐑥𐑦𐑔𐑪𐑛𐑦𐑒𐑤𐑦 𐑮𐑳𐑖𐑦𐑙 𐑳𐑐 𐑕𐑑𐑺𐑟 𐑩𐑜𐑱𐑯 𐑧𐑥𐑐𐑑𐑦-𐑣𐑨𐑯𐑛𐑩𐑛, 𐑢𐑧𐑯 ·𐑥𐑕𐑟 ·𐑣𐑳𐑕𐑦 𐑦𐑯𐑑𐑼𐑐𐑴𐑟𐑛 𐑞 𐑥𐑳𐑕𐑑𐑼𐑛-𐑐𐑪𐑑 𐑯 𐑝𐑦𐑯𐑦𐑜𐑼-𐑒𐑮𐑵𐑩𐑑, 𐑯 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑒𐑭𐑕𐑑𐑼 𐑝 𐑣𐑻 𐑒𐑬𐑯𐑑𐑩𐑯𐑩𐑯𐑕.

“𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑿, 𐑘𐑳𐑙 𐑥𐑨𐑯?”

“𐑜𐑧𐑑 𐑞 𐑨𐑒𐑕! 𐑓 ·𐑜𐑪𐑛'𐑟 𐑕𐑱𐑒, 𐑮𐑳𐑯 𐑓 𐑞 𐑛𐑪𐑒𐑑𐑼, 𐑕𐑳𐑥 𐑢𐑳𐑯, 𐑢𐑲𐑤 𐑲 𐑐𐑮𐑲 𐑦𐑑 𐑴𐑐𐑩𐑯!”

“𐑤𐑫𐑒 𐑣𐑽,” 𐑕𐑧𐑛 𐑞 𐑤𐑨𐑯𐑛𐑤𐑱𐑛𐑦, 𐑒𐑢𐑦𐑒𐑤𐑦 𐑐𐑫𐑑𐑦𐑙 𐑛𐑬𐑯 𐑞 𐑝𐑦𐑯𐑦𐑜𐑼-𐑒𐑮𐑵𐑩𐑑, 𐑕𐑴 𐑨𐑟 𐑑 𐑣𐑨𐑝 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑓𐑮𐑰; “𐑤𐑫𐑒 𐑣𐑽; 𐑸 𐑿 𐑑𐑷𐑒𐑦𐑙 𐑩𐑚𐑬𐑑 𐑐𐑮𐑲𐑦𐑙 𐑴𐑐𐑩𐑯 𐑧𐑯𐑦 𐑝 𐑥𐑲 𐑛𐑹𐑟?”—𐑯 𐑢𐑦𐑞 𐑞𐑨𐑑 𐑖𐑰 𐑕𐑰𐑟𐑛 𐑥𐑲 𐑸𐑥. “𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑿? 𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑿, 𐑖𐑦𐑐𐑥𐑱𐑑?”

𐑦𐑯 𐑨𐑟 𐑒𐑭𐑥, 𐑚𐑳𐑑 𐑮𐑨𐑐𐑦𐑛 𐑩 𐑥𐑨𐑯𐑼 𐑨𐑟 𐑐𐑪𐑕𐑩𐑚𐑩𐑤, 𐑲 𐑜𐑱𐑝 𐑣𐑻 𐑑 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑞 𐑣𐑴𐑤 𐑒𐑱𐑕. 𐑳𐑯𐑒𐑪𐑯𐑖𐑩𐑕𐑤𐑦 𐑒𐑤𐑨𐑐𐑦𐑙 𐑞 𐑝𐑦𐑯𐑦𐑜𐑼-𐑒𐑮𐑵𐑩𐑑 𐑑 𐑢𐑳𐑯 𐑕𐑲𐑛 𐑝 𐑣𐑻 𐑯𐑴𐑟, 𐑖𐑰 𐑮𐑵𐑥𐑦𐑯𐑱𐑑𐑩𐑛 𐑓 𐑩𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑; 𐑞𐑧𐑯 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛—“𐑯𐑴! 𐑲 𐑣𐑨𐑝𐑩𐑯𐑑 𐑕𐑰𐑯 𐑦𐑑 𐑕𐑦𐑯𐑕 𐑲 𐑐𐑫𐑑 𐑦𐑑 𐑞𐑺.” 𐑮𐑳𐑯𐑦𐑙 𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑒𐑤𐑪𐑟𐑩𐑑 𐑳𐑯𐑛𐑼 𐑞 𐑤𐑨𐑯𐑛𐑦𐑙 𐑝 𐑞 𐑕𐑑𐑺𐑟, 𐑖𐑰 𐑜𐑤𐑭𐑯𐑕𐑑 𐑦𐑯, 𐑯 𐑮𐑦𐑑𐑻𐑯𐑦𐑙, 𐑑𐑴𐑤𐑛 𐑥𐑰 𐑞𐑨𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑣𐑸𐑐𐑵𐑯 𐑢𐑪𐑟 𐑥𐑦𐑕𐑦𐑙. “𐑣𐑰'𐑟 𐑒𐑦𐑤𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓,” 𐑖𐑰 𐑒𐑮𐑲𐑛. “𐑦𐑑'𐑕 𐑳𐑯𐑓𐑹𐑑'𐑯𐑱𐑑 ·𐑕𐑑𐑦𐑜𐑟 𐑛𐑳𐑯 𐑴𐑝𐑼 𐑩𐑜𐑱𐑯—𐑞𐑺 𐑜𐑴𐑟 𐑩𐑯𐑳𐑞𐑼 𐑒𐑬𐑯𐑑𐑼𐑐𐑱𐑯—·𐑜𐑪𐑛 𐑐𐑦𐑑𐑦 𐑣𐑦𐑟 𐑐𐑫𐑼 𐑥𐑳𐑞𐑼!—𐑦𐑑 𐑢𐑦𐑤 𐑚𐑰 𐑞 𐑮𐑵𐑦𐑯 𐑝 𐑥𐑲 𐑣𐑬𐑕. 𐑣𐑨𐑟 𐑞 𐑐𐑫𐑼 𐑤𐑨𐑛 𐑩 𐑕𐑦𐑕𐑑𐑼? 𐑢𐑺'𐑟 𐑞𐑨𐑑 𐑜𐑻𐑤?—𐑞𐑺, ·𐑚𐑧𐑑𐑦, 𐑜𐑴 𐑑 ·𐑕𐑥𐑸𐑤𐑕 𐑞 ·𐑐𐑱𐑯𐑑𐑼, 𐑯 𐑑𐑧𐑤 𐑣𐑦𐑥 𐑑 𐑐𐑱𐑯𐑑 𐑥𐑰 𐑩 𐑕𐑲𐑯, 𐑢𐑦𐑞—“𐑯𐑴 𐑕𐑵𐑦𐑕𐑲𐑛𐑟 𐑐𐑼𐑥𐑦𐑑𐑩𐑛 𐑣𐑽, 𐑯 𐑯𐑴 𐑕𐑥𐑴𐑒𐑦𐑙 𐑦𐑯 𐑞 𐑐𐑸𐑤𐑼;”—𐑥𐑲𐑑 𐑨𐑟 𐑢𐑧𐑤 𐑒𐑦𐑤 𐑚𐑴𐑔 𐑚𐑻𐑛𐑟 𐑨𐑑 𐑢𐑳𐑯𐑕. 𐑒𐑦𐑤? 𐑞 ·𐑤𐑹𐑛 𐑚𐑰 𐑥𐑻𐑕𐑦𐑓𐑩𐑤 𐑑 𐑣𐑦𐑟 𐑜𐑴𐑕𐑑! 𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑯𐑶𐑟 𐑞𐑺? 𐑿, 𐑘𐑳𐑙 𐑥𐑨𐑯, 𐑩𐑝𐑭𐑕𐑑 𐑞𐑺!”

𐑯 𐑮𐑳𐑯𐑦𐑙 𐑳𐑐 𐑭𐑓𐑑𐑼 𐑥𐑰, 𐑖𐑰 𐑒𐑷𐑑 𐑥𐑰 𐑨𐑟 𐑲 𐑢𐑪𐑟 𐑩𐑜𐑱𐑯 𐑑𐑮𐑲𐑦𐑙 𐑑 𐑓𐑹𐑕 𐑴𐑐𐑩𐑯 𐑞 𐑛𐑹.

“𐑲 𐑛𐑴𐑯𐑑 𐑩𐑤𐑬 𐑦𐑑; 𐑲 𐑢𐑴𐑯𐑑 𐑣𐑨𐑝 𐑥𐑲 𐑐𐑮𐑧𐑥𐑦𐑕𐑩𐑟 𐑕𐑐𐑶𐑤𐑛. 𐑜𐑴 𐑓 𐑞 𐑤𐑪𐑒𐑕𐑥𐑦𐑔, 𐑞𐑺'𐑟 𐑢𐑳𐑯 𐑩𐑚𐑬𐑑 𐑩 𐑥𐑲𐑤 𐑓𐑮𐑪𐑥 𐑣𐑽. 𐑚𐑳𐑑 𐑩𐑝𐑭𐑕𐑑!” 𐑐𐑫𐑑𐑦𐑙 𐑣𐑻 𐑣𐑨𐑯𐑛 𐑦𐑯 𐑣𐑻 𐑕𐑲𐑛-𐑐𐑪𐑒𐑩𐑑, “𐑣𐑽'𐑟 𐑩 𐑒𐑰 𐑞𐑨𐑑'𐑤 𐑓𐑦𐑑, 𐑲 𐑜𐑧𐑕; 𐑤𐑧𐑑'𐑕 𐑕𐑰.” 𐑯 𐑢𐑦𐑞 𐑞𐑨𐑑, 𐑖𐑰 𐑑𐑻𐑯𐑛 𐑦𐑑 𐑦𐑯 𐑞 𐑤𐑪𐑒; 𐑚𐑳𐑑, 𐑩𐑤𐑨𐑕! ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜'𐑟 𐑕𐑳𐑐𐑤𐑦𐑥𐑩𐑯𐑑𐑩𐑤 𐑚𐑴𐑤𐑑 𐑮𐑦𐑥𐑱𐑯𐑛 𐑳𐑯𐑢𐑦𐑞𐑛𐑮𐑷𐑯 𐑢𐑦𐑞𐑦𐑯.

“𐑣𐑨𐑓 𐑑 𐑚𐑻𐑕𐑑 𐑦𐑑 𐑴𐑐𐑩𐑯,” 𐑕𐑧𐑛 𐑲, 𐑯 𐑢𐑪𐑟 𐑮𐑳𐑯𐑦𐑙 𐑛𐑬𐑯 𐑞 𐑧𐑯𐑑𐑮𐑦 𐑩 𐑤𐑦𐑑𐑩𐑤, 𐑓 𐑩 𐑜𐑫𐑛 𐑕𐑑𐑸𐑑, 𐑢𐑧𐑯 𐑞 𐑤𐑨𐑯𐑛𐑤𐑱𐑛𐑦 𐑒𐑷𐑑 𐑨𐑑 𐑥𐑰, 𐑩𐑜𐑱𐑯 𐑝𐑬𐑦𐑙 𐑲 𐑖𐑵𐑛 𐑯𐑪𐑑 𐑚𐑮𐑱𐑒 𐑛𐑬𐑯 𐑣𐑻 𐑐𐑮𐑧𐑥𐑦𐑕𐑩𐑟; 𐑚𐑳𐑑 𐑲 𐑑𐑹 𐑓𐑮𐑪𐑥 𐑣𐑻, 𐑯 𐑢𐑦𐑞 𐑩 𐑕𐑳𐑛𐑩𐑯 𐑚𐑪𐑛𐑦𐑤𐑦 𐑮𐑳𐑖 𐑛𐑨𐑖𐑑 𐑥𐑲𐑕𐑧𐑤𐑓 𐑓𐑫𐑤 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑥𐑸𐑒.

𐑢𐑦𐑞 𐑩 𐑐𐑮𐑩𐑛𐑦𐑡𐑩𐑕 𐑯𐑶𐑟 𐑞 𐑛𐑹 𐑓𐑤𐑵 𐑴𐑐𐑩𐑯, 𐑯 𐑞 𐑯𐑪𐑚 𐑕𐑤𐑨𐑥𐑦𐑙 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑢𐑷𐑤, 𐑕𐑧𐑯𐑑 𐑞 𐑐𐑤𐑭𐑕𐑑𐑼 𐑑 𐑞 𐑕𐑰𐑤𐑦𐑙; 𐑯 𐑞𐑺, 𐑜𐑫𐑛 𐑣𐑧𐑝𐑩𐑯𐑟! 𐑞𐑺 𐑕𐑨𐑑 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑷𐑤𐑑𐑩𐑜𐑧𐑞𐑼 𐑒𐑵𐑤 𐑯 𐑕𐑧𐑤𐑓-𐑒𐑩𐑤𐑧𐑒𐑑𐑩𐑛; 𐑮𐑲𐑑 𐑦𐑯 𐑞 𐑥𐑦𐑛𐑩𐑤 𐑝 𐑞 𐑮𐑵𐑥; 𐑕𐑒𐑢𐑪𐑑𐑦𐑙 𐑪𐑯 𐑣𐑦𐑟 𐑣𐑨𐑥𐑟, 𐑯 𐑣𐑴𐑤𐑛𐑦𐑙 ·𐑘𐑴𐑡𐑴 𐑪𐑯 𐑑𐑪𐑐 𐑝 𐑣𐑦𐑟 𐑣𐑧𐑛. 𐑣𐑰 𐑤𐑫𐑒𐑑 𐑯𐑲𐑞𐑼 𐑢𐑳𐑯 𐑢𐑱 𐑯𐑹 𐑞 𐑳𐑞𐑼 𐑢𐑱, 𐑚𐑳𐑑 𐑕𐑨𐑑 𐑤𐑲𐑒 𐑩 𐑒𐑸𐑝𐑛 𐑦𐑥𐑦𐑡 𐑢𐑦𐑞 𐑕𐑒𐑺𐑕 𐑩 𐑕𐑲𐑯 𐑝 𐑨𐑒𐑑𐑦𐑝 𐑤𐑲𐑓.

“·𐑒𐑢𐑰𐑒𐑢𐑧𐑜,” 𐑕𐑧𐑛 𐑲, 𐑜𐑴𐑦𐑙 𐑳𐑐 𐑑 𐑣𐑦𐑥, “·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑿?”

“𐑣𐑰 𐑣𐑱𐑩𐑯𐑑 𐑚𐑰𐑯 𐑩 𐑕𐑦𐑑𐑦𐑯' 𐑕𐑴 𐑷𐑤 𐑛𐑱, 𐑣𐑨𐑟 𐑣𐑰?” 𐑕𐑧𐑛 𐑞 𐑤𐑨𐑯𐑛𐑤𐑱𐑛𐑦.

𐑚𐑳𐑑 𐑷𐑤 𐑢𐑰 𐑕𐑧𐑛, 𐑯𐑪𐑑 𐑩 𐑢𐑻𐑛 𐑒𐑫𐑛 𐑢𐑰 𐑛𐑮𐑨𐑜 𐑬𐑑 𐑝 𐑣𐑦𐑥; 𐑲 𐑷𐑤𐑥𐑴𐑕𐑑 𐑓𐑧𐑤𐑑 𐑤𐑲𐑒 𐑐𐑫𐑖𐑦𐑙 𐑣𐑦𐑥 𐑴𐑝𐑼, 𐑕𐑴 𐑨𐑟 𐑑 𐑗𐑱𐑯𐑡 𐑣𐑦𐑟 𐑐𐑩𐑟𐑦𐑖𐑩𐑯, 𐑓 𐑦𐑑 𐑢𐑪𐑟 𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑯𐑑𐑪𐑤𐑼𐑩𐑚𐑩𐑤, 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑕𐑴 𐑐𐑱𐑯𐑓𐑩𐑤𐑦 𐑯 𐑳𐑯𐑯𐑨𐑗𐑼𐑩𐑤𐑦 𐑒𐑩𐑯𐑕𐑑𐑮𐑱𐑯𐑛; 𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦, 𐑨𐑟 𐑦𐑯 𐑷𐑤 𐑐𐑮𐑪𐑚𐑩𐑚𐑦𐑤𐑦𐑑𐑦 𐑣𐑰 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑦𐑑𐑦𐑙 𐑕𐑴 𐑓 𐑳𐑐𐑢𐑼𐑛𐑟 𐑝 𐑱𐑑 𐑹 𐑑𐑧𐑯 𐑬𐑼𐑟, 𐑜𐑴𐑦𐑙 𐑑𐑵 𐑢𐑦𐑞𐑬𐑑 𐑣𐑦𐑟 𐑮𐑧𐑜𐑿𐑤𐑼 𐑥𐑰𐑤𐑟.

“·𐑥𐑕𐑟 ·𐑣𐑳𐑕𐑦,” 𐑕𐑧𐑛 𐑲, “𐑣𐑰'𐑟 𐑩𐑤𐑲𐑝 𐑨𐑑 𐑷𐑤 𐑦𐑝𐑧𐑯𐑑𐑕; 𐑕𐑴 𐑤𐑰𐑝 𐑳𐑕, 𐑦𐑓 𐑿 𐑐𐑤𐑰𐑟, 𐑯 𐑲 𐑢𐑦𐑤 𐑕𐑰 𐑑 𐑞𐑦𐑕 𐑕𐑑𐑮𐑱𐑯𐑡 𐑩𐑓𐑺 𐑥𐑲𐑕𐑧𐑤𐑓.”

𐑒𐑤𐑴𐑟𐑦𐑙 𐑞 𐑛𐑹 𐑩𐑐𐑪𐑯 𐑞 𐑤𐑨𐑯𐑛𐑤𐑱𐑛𐑦, 𐑲 𐑦𐑯𐑛𐑧𐑝𐑼𐑛 𐑑 𐑐𐑮𐑦𐑝𐑱𐑤 𐑩𐑐𐑪𐑯 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑑 𐑑𐑱𐑒 𐑩 𐑗𐑺; 𐑚𐑳𐑑 𐑦𐑯 𐑝𐑱𐑯. 𐑞𐑺 𐑣𐑰 𐑕𐑨𐑑; 𐑯 𐑷𐑤 𐑣𐑰 𐑒𐑫𐑛 𐑛𐑵—𐑓 𐑷𐑤 𐑥𐑲 𐑐𐑩𐑤𐑲𐑑 𐑸𐑑𐑕 𐑯 𐑚𐑤𐑨𐑯𐑛𐑦𐑖𐑥𐑩𐑯𐑑𐑕—𐑣𐑰 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑥𐑵𐑝 𐑩 𐑐𐑧𐑜, 𐑯𐑹 𐑕𐑱 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑢𐑻𐑛, 𐑯𐑹 𐑰𐑝𐑩𐑯 𐑤𐑫𐑒 𐑨𐑑 𐑥𐑰, 𐑯𐑹 𐑯𐑴𐑑𐑦𐑕 𐑥𐑲 𐑐𐑮𐑧𐑟𐑩𐑯𐑕 𐑦𐑯 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑢𐑱.

𐑲 𐑢𐑳𐑯𐑛𐑼, 𐑔𐑷𐑑 𐑲, 𐑦𐑓 𐑞𐑦𐑕 𐑒𐑨𐑯 𐑐𐑪𐑕𐑩𐑚𐑤𐑦 𐑚𐑰 𐑩 𐑐𐑸𐑑 𐑝 𐑣𐑦𐑟 ·𐑮𐑭𐑥𐑩𐑛𐑭𐑯; 𐑛𐑵 𐑞𐑱 𐑓𐑭𐑕𐑑 𐑪𐑯 𐑞𐑺 𐑣𐑨𐑥𐑟 𐑞𐑨𐑑 𐑢𐑱 𐑦𐑯 𐑣𐑦𐑟 𐑯𐑱𐑑𐑦𐑝 𐑲𐑤𐑩𐑯𐑛. 𐑦𐑑 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑕𐑴; 𐑘𐑧𐑕, 𐑦𐑑'𐑕 𐑐𐑸𐑑 𐑝 𐑣𐑦𐑟 𐑒𐑮𐑰𐑛, 𐑲 𐑕𐑩𐑐𐑴𐑟; 𐑢𐑧𐑤, 𐑞𐑧𐑯, 𐑤𐑧𐑑 𐑣𐑦𐑥 𐑮𐑧𐑕𐑑; 𐑣𐑰'𐑤 𐑜𐑧𐑑 𐑳𐑐 𐑕𐑵𐑯𐑼 𐑹 𐑤𐑱𐑑𐑼, 𐑯𐑴 𐑛𐑬𐑑. 𐑦𐑑 𐑒𐑭𐑯𐑑 𐑤𐑭𐑕𐑑 𐑓 𐑧𐑝𐑼, 𐑔𐑨𐑙𐑒 ·𐑜𐑪𐑛, 𐑯 𐑣𐑦𐑟 ·𐑮𐑭𐑥𐑩𐑛𐑭𐑯 𐑴𐑯𐑤𐑦 𐑒𐑳𐑥𐑟 𐑢𐑳𐑯𐑕 𐑩 𐑘𐑽; 𐑯 𐑲 𐑛𐑴𐑯𐑑 𐑚𐑦𐑤𐑰𐑝 𐑦𐑑'𐑕 𐑝𐑧𐑮𐑦 𐑐𐑳𐑙𐑒𐑗𐑵𐑩𐑤 𐑞𐑧𐑯.

𐑲 𐑢𐑧𐑯𐑑 𐑛𐑬𐑯 𐑑 𐑕𐑳𐑐𐑼. 𐑭𐑓𐑑𐑼 𐑕𐑦𐑑𐑦𐑙 𐑩 𐑤𐑪𐑙 𐑑𐑲𐑥 𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑑 𐑞 𐑤𐑪𐑙 𐑕𐑑𐑹𐑦𐑟 𐑝 𐑕𐑳𐑥 𐑕𐑱𐑤𐑼𐑟 𐑣𐑵 𐑣𐑨𐑛 𐑡𐑳𐑕𐑑 𐑒𐑳𐑥 𐑓𐑮𐑪𐑥 𐑩 𐑐𐑤𐑳𐑥-𐑐𐑫𐑛𐑦𐑙 𐑝𐑶𐑦𐑡, 𐑨𐑟 𐑞𐑱 𐑒𐑷𐑤𐑛 𐑦𐑑 (𐑞𐑨𐑑 𐑦𐑟, 𐑩 𐑖𐑹𐑑 𐑢𐑱𐑤𐑦𐑙-𐑝𐑶𐑦𐑡 𐑦𐑯 𐑩 𐑕𐑒𐑵𐑯𐑼 𐑹 𐑚𐑮𐑦𐑜, 𐑒𐑩𐑯𐑓𐑲𐑯𐑛 𐑑 𐑞 𐑯𐑹𐑔 𐑝 𐑞 𐑤𐑲𐑯, 𐑦𐑯 𐑞 ·𐑩𐑑𐑤𐑨𐑯𐑑𐑦𐑒 ·𐑴𐑖𐑩𐑯 𐑴𐑯𐑤𐑦); 𐑭𐑓𐑑𐑼 𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑑 𐑞𐑰𐑟 𐑐𐑤𐑳𐑥-𐑐𐑫𐑛𐑦𐑙𐑼𐑟 𐑑𐑦𐑤 𐑯𐑽𐑤𐑦 𐑦𐑤𐑧𐑝𐑩𐑯 𐑴'𐑒𐑤𐑪𐑒, 𐑲 𐑢𐑧𐑯𐑑 𐑳𐑐 𐑕𐑑𐑺𐑟 𐑑 𐑜𐑴 𐑑 𐑚𐑧𐑛, 𐑓𐑰𐑤𐑦𐑙 𐑒𐑢𐑲𐑑 𐑖𐑫𐑼 𐑚𐑲 𐑞𐑦𐑕 𐑑𐑲𐑥 ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜 𐑥𐑳𐑕𐑑 𐑕𐑻𐑑𐑩𐑯𐑤𐑦 𐑣𐑨𐑝 𐑚𐑮𐑷𐑑 𐑣𐑦𐑟 ·𐑮𐑭𐑥𐑩𐑛𐑭𐑯 𐑑 𐑩 𐑑𐑻𐑥𐑦𐑯𐑱𐑖𐑩𐑯. 𐑚𐑳𐑑 𐑯𐑴; 𐑞𐑺 𐑣𐑰 𐑢𐑪𐑟 𐑡𐑳𐑕𐑑 𐑢𐑺 𐑲 𐑣𐑨𐑛 𐑤𐑧𐑓𐑑 𐑣𐑦𐑥; 𐑣𐑰 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑕𐑑𐑻𐑛 𐑩𐑯 𐑦𐑯𐑗. 𐑲 𐑚𐑦𐑜𐑨𐑯 𐑑 𐑜𐑮𐑴 𐑝𐑧𐑒𐑕𐑑 𐑢𐑦𐑞 𐑣𐑦𐑥; 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑕𐑴 𐑛𐑬𐑯𐑮𐑲𐑑 𐑕𐑧𐑯𐑕𐑤𐑩𐑕 𐑯 𐑦𐑯𐑕𐑱𐑯 𐑑 𐑚𐑰 𐑕𐑦𐑑𐑦𐑙 𐑞𐑺 𐑷𐑤 𐑛𐑱 𐑯 𐑣𐑭𐑓 𐑞 𐑯𐑲𐑑 𐑪𐑯 𐑣𐑦𐑟 𐑣𐑨𐑥𐑟 𐑦𐑯 𐑩 𐑒𐑴𐑤𐑛 𐑮𐑵𐑥, 𐑣𐑴𐑤𐑛𐑦𐑙 𐑩 𐑐𐑰𐑕 𐑝 𐑢𐑫𐑛 𐑪𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛.

“𐑓 𐑣𐑧𐑝𐑩𐑯'𐑟 𐑕𐑱𐑒, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, 𐑜𐑧𐑑 𐑳𐑐 𐑯 𐑖𐑱𐑒 𐑘𐑹𐑕𐑧𐑤𐑓; 𐑜𐑧𐑑 𐑳𐑐 𐑯 𐑣𐑨𐑝 𐑕𐑳𐑥 𐑕𐑳𐑐𐑼. 𐑿'𐑤 𐑕𐑑𐑸𐑝; 𐑿'𐑤 𐑒𐑦𐑤 𐑘𐑹𐑕𐑧𐑤𐑓, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜.” 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑩 𐑢𐑻𐑛 𐑛𐑦𐑛 𐑣𐑰 𐑮𐑦𐑐𐑤𐑲.
