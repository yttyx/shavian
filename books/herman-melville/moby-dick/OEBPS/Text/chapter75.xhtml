·𐑗𐑨𐑐𐑑𐑼 75. 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 ·𐑣𐑧𐑛—·𐑒𐑩𐑯𐑑𐑮𐑭𐑕𐑑𐑩𐑛 ·𐑝𐑿.

𐑒𐑮𐑪𐑕𐑦𐑙 𐑞 𐑛𐑧𐑒, 𐑤𐑧𐑑 𐑳𐑕 𐑯𐑬 𐑣𐑨𐑝 𐑩 𐑜𐑫𐑛 𐑤𐑪𐑙 𐑤𐑫𐑒 𐑨𐑑 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛.

𐑨𐑟 𐑦𐑯 𐑡𐑧𐑯𐑼𐑩𐑤 𐑖𐑱𐑐 𐑞 𐑯𐑴𐑚𐑩𐑤 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛 𐑥𐑱 𐑚𐑰 𐑒𐑩𐑥𐑐𐑺𐑛 𐑑 𐑩 ·𐑮𐑴𐑥𐑩𐑯 𐑢𐑹-𐑗𐑨𐑮𐑾𐑑 (𐑦𐑕𐑐𐑧𐑖𐑩𐑤𐑦 𐑦𐑯 𐑓𐑮𐑳𐑯𐑑, 𐑢𐑺 𐑦𐑑 𐑦𐑟 𐑕𐑴 𐑚𐑮𐑷𐑛𐑤𐑦 𐑮𐑬𐑯𐑛𐑩𐑛); 𐑕𐑴, 𐑨𐑑 𐑩 𐑚𐑮𐑷𐑛 𐑝𐑿, 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛 𐑚𐑺𐑟 𐑩 𐑮𐑭𐑞𐑼 𐑦𐑯𐑧𐑤𐑦𐑜𐑩𐑯𐑑 𐑮𐑦𐑟𐑧𐑥𐑚𐑩𐑤𐑩𐑯𐑕 𐑑 𐑩 𐑡𐑲𐑜𐑨𐑯𐑑𐑦𐑒 𐑜𐑨𐑤𐑦𐑪𐑑-𐑑𐑴𐑛 𐑖𐑵. 𐑑𐑵 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑘𐑽𐑟 𐑩𐑜𐑴 𐑩𐑯 𐑴𐑤𐑛 ·𐑛𐑳𐑗 𐑝𐑶𐑦𐑡𐑼 𐑤𐑲𐑒𐑩𐑯𐑛 𐑦𐑑𐑕 𐑖𐑱𐑐 𐑑 𐑞𐑨𐑑 𐑝 𐑩 𐑖𐑵𐑥𐑱𐑒𐑼'𐑟 𐑤𐑭𐑕𐑑. 𐑯 𐑦𐑯 𐑞𐑦𐑕 𐑕𐑱𐑥 𐑤𐑭𐑕𐑑 𐑹 𐑖𐑵, 𐑞𐑨𐑑 𐑴𐑤𐑛 𐑢𐑫𐑥𐑩𐑯 𐑝 𐑞 𐑯𐑻𐑕𐑼𐑦 𐑑𐑱𐑤, 𐑢𐑦𐑞 𐑞 𐑕𐑢𐑹𐑥𐑦𐑙 𐑚𐑮𐑵𐑛, 𐑥𐑲𐑑 𐑝𐑧𐑮𐑦 𐑒𐑳𐑥𐑓𐑼𐑑𐑩𐑚𐑤𐑦 𐑚𐑰 𐑤𐑪𐑡𐑛, 𐑖𐑰 𐑯 𐑷𐑤 𐑣𐑻 𐑐𐑮𐑪𐑡𐑩𐑯𐑦.

*𐑞𐑦𐑕 𐑮𐑦𐑥𐑲𐑯𐑛𐑟 𐑳𐑕 𐑞𐑨𐑑 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑮𐑾𐑤𐑦 𐑣𐑨𐑟 𐑩 𐑕𐑹𐑑 𐑝 𐑢𐑦𐑕𐑒𐑼, 𐑹 𐑮𐑭𐑞𐑼 𐑩 𐑥𐑳𐑕𐑑𐑭𐑖, 𐑒𐑩𐑯𐑕𐑦𐑕𐑑𐑦𐑙 𐑝 𐑩 𐑓𐑿 𐑕𐑒𐑨𐑑𐑼𐑛 𐑢𐑲𐑑 𐑣𐑺𐑟 𐑪𐑯 𐑞 𐑳𐑐𐑼 𐑐𐑸𐑑 𐑝 𐑞 𐑬𐑑𐑼 𐑧𐑯𐑛 𐑝 𐑞 𐑤𐑴𐑼 𐑡𐑷. 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑞𐑰𐑟 𐑑𐑳𐑓𐑑𐑕 𐑦𐑥𐑐𐑸𐑑 𐑩 𐑮𐑭𐑞𐑼 𐑚𐑮𐑦𐑜𐑩𐑯𐑛𐑦𐑖 𐑦𐑒𐑕𐑐𐑮𐑧𐑖𐑩𐑯 𐑑 𐑣𐑦𐑟 𐑳𐑞𐑼𐑢𐑲𐑟 𐑕𐑪𐑤𐑩𐑥 𐑒𐑬𐑯𐑑𐑩𐑯𐑩𐑯𐑕.

𐑨𐑟 𐑧𐑝𐑮𐑦 𐑢𐑳𐑯 𐑯𐑴𐑟, 𐑞𐑰𐑟 𐑕𐑱𐑥 “𐑣𐑪𐑜𐑟' 𐑚𐑮𐑦𐑕𐑩𐑤𐑟,” “𐑓𐑦𐑯𐑟,” “𐑢𐑦𐑕𐑒𐑼𐑟,” “𐑚𐑤𐑲𐑯𐑛𐑟,” 𐑹 𐑢𐑪𐑑𐑧𐑝𐑼 𐑿 𐑐𐑤𐑰𐑟, 𐑓𐑻𐑯𐑦𐑖 𐑑 𐑞 𐑤𐑱𐑛𐑦𐑟 𐑞𐑺 𐑚𐑳𐑕𐑒𐑕 𐑯 𐑳𐑞𐑼 𐑕𐑑𐑦𐑓𐑩𐑯𐑦𐑙 𐑒𐑩𐑯𐑑𐑮𐑲𐑝𐑩𐑯𐑕𐑩𐑟. 𐑚𐑳𐑑 𐑦𐑯 𐑞𐑦𐑕 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼, 𐑞 𐑛𐑦𐑥𐑭𐑯𐑛 𐑣𐑨𐑟 𐑤𐑪𐑙 𐑚𐑰𐑯 𐑪𐑯 𐑞 𐑛𐑦𐑒𐑤𐑲𐑯. 𐑦𐑑 𐑢𐑪𐑟 𐑦𐑯 ·𐑒𐑢𐑰𐑯 ·𐑨𐑯'𐑟 𐑑𐑲𐑥 𐑞𐑨𐑑 𐑞 𐑚𐑴𐑯 𐑢𐑪𐑟 𐑦𐑯 𐑦𐑑𐑕 𐑜𐑤𐑹𐑦, 𐑞 𐑓𐑸𐑞𐑦𐑙𐑜𐑱𐑤 𐑚𐑰𐑦𐑙 𐑞𐑧𐑯 𐑷𐑤 𐑞 𐑓𐑨𐑖𐑩𐑯. 𐑯 𐑨𐑟 𐑞𐑴𐑟 𐑱𐑯𐑗𐑩𐑯𐑑 𐑛𐑱𐑥𐑟 𐑥𐑵𐑝𐑛 𐑩𐑚𐑬𐑑 𐑜𐑱𐑤𐑦, 𐑞𐑴 𐑦𐑯 𐑞 𐑡𐑷𐑟 𐑝 𐑞 𐑢𐑱𐑤, 𐑨𐑟 𐑿 𐑥𐑱 𐑕𐑱; 𐑰𐑝𐑩𐑯 𐑕𐑴, 𐑦𐑯 𐑩 𐑖𐑬𐑼, 𐑢𐑦𐑞 𐑞 𐑤𐑲𐑒 𐑔𐑷𐑑𐑤𐑩𐑕𐑯𐑩𐑕, 𐑛𐑵 𐑢𐑰 𐑯𐑬𐑩𐑛𐑱𐑟 𐑓𐑤𐑲 𐑳𐑯𐑛𐑼 𐑞 𐑕𐑱𐑥 𐑡𐑷𐑟 𐑓 𐑐𐑮𐑩𐑑𐑧𐑒𐑖𐑩𐑯; 𐑞 𐑳𐑥𐑚𐑮𐑧𐑤𐑩 𐑚𐑰𐑦𐑙 𐑩 𐑑𐑧𐑯𐑑 𐑕𐑐𐑮𐑧𐑛 𐑴𐑝𐑼 𐑞 𐑕𐑱𐑥 𐑚𐑴𐑯.

𐑚𐑳𐑑 𐑯𐑬 𐑓𐑼𐑜𐑧𐑑 𐑷𐑤 𐑩𐑚𐑬𐑑 𐑚𐑤𐑲𐑯𐑛𐑟 𐑯 𐑢𐑦𐑕𐑒𐑼𐑟 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑯, 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑥𐑬𐑔, 𐑤𐑫𐑒 𐑼𐑬𐑯𐑛 𐑿 𐑩𐑓𐑮𐑧𐑖. 𐑕𐑰𐑦𐑙 𐑷𐑤 𐑞𐑰𐑟 𐑒𐑪𐑤𐑩𐑯𐑱𐑛𐑟 𐑝 𐑚𐑴𐑯 𐑕𐑴 𐑥𐑦𐑔𐑪𐑛𐑦𐑒𐑤𐑦 𐑮𐑱𐑯𐑡𐑛 𐑩𐑚𐑬𐑑, 𐑢𐑫𐑛 𐑿 𐑯𐑪𐑑 𐑔𐑦𐑙𐑒 𐑿 𐑢𐑻 𐑦𐑯𐑕𐑲𐑛 𐑝 𐑞 𐑜𐑮𐑱𐑑 ·𐑣𐑸𐑤𐑩𐑥 𐑹𐑜𐑩𐑯, 𐑯 𐑜𐑱𐑟𐑦𐑙 𐑩𐑐𐑪𐑯 𐑦𐑑𐑕 𐑔𐑬𐑟𐑩𐑯𐑛 𐑐𐑲𐑐𐑕? 𐑓 𐑩 𐑒𐑸𐑐𐑩𐑑 𐑑 𐑞 𐑹𐑜𐑩𐑯 𐑢𐑰 𐑣𐑨𐑝 𐑩 𐑮𐑳𐑜 𐑝 𐑞 𐑕𐑪𐑓𐑑𐑩𐑕𐑑 ·𐑑𐑻𐑒𐑦—𐑞 𐑑𐑳𐑙, 𐑢𐑦𐑗 𐑦𐑟 𐑜𐑤𐑵𐑛, 𐑨𐑟 𐑦𐑑 𐑢𐑻, 𐑑 𐑞 𐑓𐑤𐑹 𐑝 𐑞 𐑥𐑬𐑔. 𐑦𐑑 𐑦𐑟 𐑝𐑧𐑮𐑦 𐑓𐑨𐑑 𐑯 𐑑𐑧𐑯𐑛𐑼, 𐑯 𐑨𐑐𐑑 𐑑 𐑑𐑺 𐑦𐑯 𐑐𐑰𐑕𐑩𐑟 𐑦𐑯 𐑣𐑶𐑕𐑑𐑦𐑙 𐑦𐑑 𐑪𐑯 𐑛𐑧𐑒. 𐑞𐑦𐑕 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑑𐑳𐑙 𐑯𐑬 𐑚𐑦𐑓𐑹 𐑳𐑕; 𐑨𐑑 𐑩 𐑐𐑭𐑕𐑦𐑙 𐑜𐑤𐑭𐑯𐑕 𐑲 𐑖𐑵𐑛 𐑕𐑱 𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑕𐑦𐑒𐑕-𐑚𐑨𐑮𐑩𐑤𐑼; 𐑞𐑨𐑑 𐑦𐑟, 𐑦𐑑 𐑢𐑦𐑤 𐑘𐑰𐑤𐑛 𐑿 𐑩𐑚𐑬𐑑 𐑞𐑨𐑑 𐑩𐑥𐑬𐑯𐑑 𐑝 𐑶𐑤.

𐑺 𐑞𐑦𐑕, 𐑿 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑐𐑤𐑱𐑯𐑤𐑦 𐑕𐑰𐑯 𐑞 𐑑𐑮𐑵𐑔 𐑝 𐑢𐑪𐑑 𐑲 𐑕𐑑𐑸𐑑𐑩𐑛 𐑢𐑦𐑞—𐑞𐑨𐑑 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑯 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑣𐑨𐑝 𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑯𐑑𐑲𐑼𐑤𐑦 𐑛𐑦𐑓𐑼𐑩𐑯𐑑 𐑣𐑧𐑛𐑟. 𐑑 𐑕𐑳𐑥 𐑳𐑐, 𐑞𐑧𐑯: 𐑦𐑯 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑞𐑺 𐑦𐑟 𐑯𐑴 𐑜𐑮𐑱𐑑 𐑢𐑧𐑤 𐑝 𐑕𐑐𐑻𐑥; 𐑯𐑴 𐑲𐑝𐑼𐑦 𐑑𐑰𐑔 𐑨𐑑 𐑷𐑤; 𐑯𐑴 𐑤𐑪𐑙, 𐑕𐑤𐑧𐑯𐑛𐑼 𐑥𐑨𐑯𐑛𐑦𐑚𐑩𐑤 𐑝 𐑩 𐑤𐑴𐑼 𐑡𐑷, 𐑤𐑲𐑒 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟. 𐑯𐑹 𐑦𐑯 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑸 𐑞𐑺 𐑧𐑯𐑦 𐑝 𐑞𐑴𐑟 𐑚𐑤𐑲𐑯𐑛𐑟 𐑝 𐑚𐑴𐑯; 𐑯𐑴 𐑣𐑿𐑡 𐑤𐑴𐑼 𐑤𐑦𐑐; 𐑯 𐑕𐑒𐑺𐑕𐑤𐑦 𐑧𐑯𐑦𐑔𐑦𐑙 𐑝 𐑩 𐑑𐑳𐑙. 𐑩𐑜𐑱𐑯, 𐑞 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑣𐑨𐑟 𐑑𐑵 𐑧𐑒𐑕𐑑𐑼𐑯𐑩𐑤 𐑕𐑐𐑬𐑑-𐑣𐑴𐑤𐑟, 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑴𐑯𐑤𐑦 𐑢𐑳𐑯.

𐑤𐑫𐑒 𐑘𐑹 𐑤𐑭𐑕𐑑, 𐑯𐑬, 𐑪𐑯 𐑞𐑰𐑟 𐑝𐑧𐑯𐑼𐑩𐑚𐑩𐑤 𐑣𐑫𐑛𐑩𐑛 𐑣𐑧𐑛𐑟, 𐑢𐑲𐑤 𐑞𐑱 𐑘𐑧𐑑 𐑤𐑲 𐑑𐑩𐑜𐑧𐑞𐑼; 𐑓 𐑢𐑳𐑯 𐑢𐑦𐑤 𐑕𐑵𐑯 𐑕𐑦𐑙𐑒, 𐑳𐑯𐑮𐑦𐑒𐑹𐑛𐑩𐑛, 𐑦𐑯 𐑞 𐑕𐑰; 𐑞 𐑳𐑞𐑼 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑚𐑰 𐑝𐑧𐑮𐑦 𐑤𐑪𐑙 𐑦𐑯 𐑓𐑪𐑤𐑴𐑦𐑙.

𐑒𐑨𐑯 𐑿 𐑒𐑨𐑗 𐑞 𐑦𐑒𐑕𐑐𐑮𐑧𐑖𐑩𐑯 𐑝 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤'𐑟 𐑞𐑺? 𐑦𐑑 𐑦𐑟 𐑞 𐑕𐑱𐑥 𐑣𐑰 𐑛𐑲𐑛 𐑢𐑦𐑞, 𐑴𐑯𐑤𐑦 𐑕𐑳𐑥 𐑝 𐑞 𐑤𐑪𐑙𐑜𐑼 𐑮𐑦𐑙𐑒𐑩𐑤𐑟 𐑦𐑯 𐑞 𐑓𐑪𐑮𐑦𐑛 𐑕𐑰𐑥 𐑯𐑬 𐑓𐑱𐑛𐑩𐑛 𐑩𐑢𐑱. 𐑲 𐑔𐑦𐑙𐑒 𐑣𐑦𐑟 𐑚𐑮𐑷𐑛 𐑚𐑮𐑬 𐑑 𐑚𐑰 𐑓𐑫𐑤 𐑝 𐑩 𐑐𐑮𐑺𐑦-𐑤𐑲𐑒 𐑐𐑤𐑩𐑕𐑦𐑛𐑦𐑑𐑦, 𐑚𐑹𐑯 𐑝 𐑩 𐑕𐑐𐑧𐑒𐑿𐑤𐑩𐑑𐑦𐑝 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑕 𐑨𐑟 𐑑 𐑛𐑧𐑔. 𐑚𐑳𐑑 𐑥𐑸𐑒 𐑞 𐑳𐑞𐑼 𐑣𐑧𐑛'𐑟 𐑦𐑒𐑕𐑐𐑮𐑧𐑖𐑩𐑯. 𐑕𐑰 𐑞𐑨𐑑 𐑩𐑥𐑱𐑟𐑦𐑙 𐑤𐑴𐑼 𐑤𐑦𐑐, 𐑐𐑮𐑧𐑕𐑑 𐑚𐑲 𐑨𐑒𐑕𐑦𐑛𐑩𐑯𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑝𐑧𐑕𐑩𐑤'𐑟 𐑕𐑲𐑛, 𐑕𐑴 𐑨𐑟 𐑓𐑻𐑥𐑤𐑦 𐑑 𐑦𐑥𐑚𐑮𐑱𐑕 𐑞 𐑡𐑷. 𐑛𐑳𐑟 𐑯𐑪𐑑 𐑞𐑦𐑕 𐑣𐑴𐑤 𐑣𐑧𐑛 𐑕𐑰𐑥 𐑑 𐑕𐑐𐑰𐑒 𐑝 𐑩𐑯 𐑦𐑯𐑹𐑥𐑩𐑕 𐑐𐑮𐑨𐑒𐑑𐑦𐑒𐑩𐑤 𐑮𐑧𐑟𐑩𐑤𐑵𐑖𐑩𐑯 𐑦𐑯 𐑓𐑱𐑕𐑦𐑙 𐑛𐑧𐑔? 𐑞𐑦𐑕 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤 𐑲 𐑑𐑱𐑒 𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑩 ·𐑕𐑑𐑴𐑦𐑒; 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤, 𐑩 ·𐑐𐑤𐑱𐑑𐑴𐑯𐑾𐑯, 𐑣𐑵 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑑𐑱𐑒𐑩𐑯 𐑳𐑐 ·𐑕𐑐𐑦𐑯𐑴𐑟𐑩 𐑦𐑯 𐑣𐑦𐑟 𐑤𐑨𐑑𐑼 𐑘𐑽𐑟.
