·𐑗𐑨𐑐𐑑𐑼 131. 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 ·𐑥𐑰𐑑𐑕 𐑞 ·𐑛𐑦𐑤𐑲𐑑.

𐑞 𐑦𐑯𐑑𐑧𐑯𐑕 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑕𐑱𐑤𐑛 𐑪𐑯; 𐑞 𐑮𐑴𐑤𐑦𐑙 𐑢𐑱𐑝𐑟 𐑯 𐑛𐑱𐑟 𐑢𐑧𐑯𐑑 𐑚𐑲; 𐑞 𐑤𐑲𐑓-𐑚𐑶-𐑒𐑪𐑓𐑦𐑯 𐑕𐑑𐑦𐑤 𐑤𐑲𐑑𐑤𐑦 𐑕𐑢𐑳𐑙; 𐑯 𐑩𐑯𐑳𐑞𐑼 𐑖𐑦𐑐, 𐑥𐑴𐑕𐑑 𐑥𐑦𐑟𐑼𐑩𐑚𐑤𐑦 𐑥𐑦𐑕𐑯𐑱𐑥𐑛 𐑞 ·𐑛𐑦𐑤𐑲𐑑, 𐑢𐑪𐑟 𐑛𐑦𐑕𐑒𐑮𐑲𐑛. 𐑨𐑟 𐑖𐑰 𐑛𐑮𐑵 𐑯𐑲, 𐑷𐑤 𐑲𐑟 𐑢𐑻 𐑓𐑦𐑒𐑕𐑑 𐑩𐑐𐑪𐑯 𐑣𐑻 𐑚𐑮𐑷𐑛 𐑚𐑰𐑥𐑟, 𐑒𐑷𐑤𐑛 𐑖𐑽𐑟, 𐑢𐑦𐑗, 𐑦𐑯 𐑕𐑳𐑥 𐑢𐑱𐑤𐑦𐑙-𐑖𐑦𐑐𐑕, 𐑒𐑮𐑪𐑕 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒 𐑨𐑑 𐑞 𐑣𐑲𐑑 𐑝 𐑱𐑑 𐑹 𐑯𐑲𐑯 𐑓𐑰𐑑; 𐑕𐑻𐑝𐑦𐑙 𐑑 𐑒𐑨𐑮𐑦 𐑞 𐑕𐑐𐑺, 𐑳𐑯𐑮𐑦𐑜𐑛, 𐑹 𐑛𐑦𐑕𐑱𐑚𐑩𐑤𐑛 𐑚𐑴𐑑𐑕.

𐑩𐑐𐑪𐑯 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼'𐑟 𐑖𐑽𐑟 𐑢𐑻 𐑚𐑦𐑣𐑧𐑤𐑛 𐑞 𐑖𐑨𐑑𐑼𐑛, 𐑢𐑲𐑑 𐑮𐑦𐑚𐑟, 𐑯 𐑕𐑳𐑥 𐑓𐑿 𐑕𐑐𐑤𐑦𐑯𐑑𐑼𐑛 𐑐𐑤𐑨𐑙𐑒𐑕, 𐑝 𐑢𐑪𐑑 𐑣𐑨𐑛 𐑢𐑳𐑯𐑕 𐑚𐑰𐑯 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑; 𐑚𐑳𐑑 𐑿 𐑯𐑬 𐑕𐑷 𐑔𐑮𐑵 𐑞𐑦𐑕 𐑮𐑧𐑒, 𐑨𐑟 𐑐𐑤𐑱𐑯𐑤𐑦 𐑨𐑟 𐑿 𐑕𐑰 𐑔𐑮𐑵 𐑞 𐑐𐑰𐑤𐑛, 𐑣𐑭𐑓-𐑳𐑯𐑣𐑦𐑯𐑡𐑛, 𐑯 𐑚𐑤𐑰𐑗𐑦𐑙 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯 𐑝 𐑩 𐑣𐑹𐑕.

“𐑣𐑨𐑕𐑑 𐑕𐑰𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?”

“𐑤𐑫𐑒!” 𐑮𐑦𐑐𐑤𐑲𐑛 𐑞 𐑣𐑪𐑤𐑴-𐑗𐑰𐑒𐑑 𐑒𐑨𐑐𐑑𐑦𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑑𐑨𐑓𐑮𐑱𐑤; 𐑯 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑑𐑮𐑳𐑥𐑐𐑩𐑑 𐑣𐑰 𐑐𐑶𐑯𐑑𐑩𐑛 𐑑 𐑞 𐑮𐑧𐑒.

“𐑣𐑨𐑕𐑑 𐑒𐑦𐑤𐑛 𐑣𐑦𐑥?”

“𐑞 𐑣𐑸𐑐𐑵𐑯 𐑦𐑟 𐑯𐑪𐑑 𐑘𐑧𐑑 𐑓𐑹𐑡𐑛 𐑞𐑨𐑑 𐑧𐑝𐑼 𐑢𐑦𐑤 𐑛𐑵 𐑞𐑨𐑑,” 𐑭𐑯𐑕𐑼𐑛 𐑞 𐑳𐑞𐑼, 𐑕𐑨𐑛𐑤𐑦 𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑩𐑐𐑪𐑯 𐑩 𐑮𐑬𐑯𐑛𐑩𐑛 𐑣𐑨𐑥𐑩𐑒 𐑪𐑯 𐑞 𐑛𐑧𐑒, 𐑣𐑵𐑟 𐑜𐑨𐑞𐑼𐑛 𐑕𐑲𐑛𐑟 𐑕𐑳𐑥 𐑯𐑶𐑟𐑤𐑩𐑕 𐑕𐑱𐑤𐑼𐑟 𐑢𐑻 𐑚𐑦𐑟𐑦 𐑦𐑯 𐑕𐑴𐑦𐑙 𐑑𐑩𐑜𐑧𐑞𐑼.

“𐑯𐑪𐑑 𐑓𐑹𐑡𐑛!” 𐑯 𐑕𐑯𐑨𐑗𐑦𐑙 ·𐑐𐑻𐑔'𐑕 𐑤𐑧𐑝𐑩𐑤𐑛 𐑲𐑼𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑮𐑪𐑗, ·𐑱𐑣𐑨𐑚 𐑣𐑧𐑤𐑛 𐑦𐑑 𐑬𐑑, 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑦𐑙—“𐑤𐑫𐑒 𐑘𐑰, ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼; 𐑣𐑽 𐑦𐑯 𐑞𐑦𐑕 𐑣𐑨𐑯𐑛 𐑲 𐑣𐑴𐑤𐑛 𐑣𐑦𐑟 𐑛𐑧𐑔! 𐑑𐑧𐑥𐑐𐑼𐑛 𐑦𐑯 𐑚𐑤𐑳𐑛, 𐑯 𐑑𐑧𐑥𐑐𐑼𐑛 𐑚𐑲 𐑤𐑲𐑑𐑯𐑦𐑙 𐑸 𐑞𐑰𐑟 𐑚𐑸𐑚𐑟; 𐑯 𐑲 𐑕𐑢𐑺 𐑑 𐑑𐑧𐑥𐑐𐑼 𐑞𐑧𐑥 𐑑𐑮𐑦𐑐𐑤𐑦 𐑦𐑯 𐑞𐑨𐑑 𐑣𐑪𐑑 𐑐𐑤𐑱𐑕 𐑚𐑦𐑣𐑲𐑯𐑛 𐑞 𐑓𐑦𐑯, 𐑢𐑺 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑥𐑴𐑕𐑑 𐑓𐑰𐑤𐑟 𐑣𐑦𐑟 𐑩𐑒𐑻𐑕𐑩𐑛 𐑤𐑲𐑓!”

“𐑞𐑧𐑯 ·𐑜𐑪𐑛 𐑒𐑰𐑐 𐑞𐑰, 𐑴𐑤𐑛 𐑥𐑨𐑯—𐑕𐑰'𐑕𐑑 𐑞𐑬 𐑞𐑨𐑑”—𐑐𐑶𐑯𐑑𐑦𐑙 𐑑 𐑞 𐑣𐑨𐑥𐑩𐑒—“𐑲 𐑚𐑧𐑮𐑦 𐑚𐑳𐑑 𐑢𐑳𐑯 𐑝 𐑓𐑲𐑝 𐑕𐑑𐑬𐑑 𐑥𐑧𐑯, 𐑣𐑵 𐑢𐑻 𐑩𐑤𐑲𐑝 𐑴𐑯𐑤𐑦 𐑘𐑧𐑕𐑑𐑼𐑛𐑱; 𐑚𐑳𐑑 𐑢𐑻 𐑛𐑧𐑛 𐑺 𐑯𐑲𐑑. 𐑴𐑯𐑤𐑦 𐑞𐑨𐑑 𐑢𐑳𐑯 𐑲 𐑚𐑧𐑮𐑦; 𐑞 𐑮𐑧𐑕𐑑 𐑢𐑻 𐑚𐑧𐑮𐑦𐑛 𐑚𐑦𐑓𐑹 𐑞𐑱 𐑛𐑲𐑛; 𐑿 𐑕𐑱𐑤 𐑩𐑐𐑪𐑯 𐑞𐑺 𐑑𐑵𐑥.” 𐑞𐑧𐑯 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑣𐑦𐑟 𐑒𐑮𐑵—“𐑸 𐑘𐑰 𐑮𐑧𐑛𐑦 𐑞𐑺? 𐑐𐑤𐑱𐑕 𐑞 𐑐𐑤𐑨𐑙𐑒 𐑞𐑧𐑯 𐑪𐑯 𐑞 𐑮𐑱𐑤, 𐑯 𐑤𐑦𐑓𐑑 𐑞 𐑚𐑪𐑛𐑦; 𐑕𐑴, 𐑞𐑧𐑯—𐑴! 𐑜𐑪𐑛”—𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑣𐑨𐑥𐑩𐑒 𐑢𐑦𐑞 𐑳𐑐𐑤𐑦𐑓𐑑𐑩𐑛 𐑣𐑨𐑯𐑛𐑟—“𐑥𐑱 𐑞 𐑮𐑧𐑟𐑼𐑧𐑒𐑖𐑩𐑯 𐑯 𐑞 𐑤𐑲𐑓——”

“𐑚𐑮𐑱𐑕 𐑓𐑹𐑢𐑼𐑛! 𐑳𐑐 𐑣𐑧𐑤𐑥!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚 𐑤𐑲𐑒 𐑤𐑲𐑑𐑯𐑦𐑙 𐑑 𐑣𐑦𐑟 𐑥𐑧𐑯.

𐑚𐑳𐑑 𐑞 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑕𐑑𐑸𐑑𐑩𐑛 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑒𐑢𐑦𐑒 𐑦𐑯𐑳𐑓 𐑑 𐑦𐑕𐑒𐑱𐑐 𐑞 𐑕𐑬𐑯𐑛 𐑝 𐑞 𐑕𐑐𐑤𐑨𐑖 𐑞𐑨𐑑 𐑞 𐑒𐑹𐑐𐑕 𐑕𐑵𐑯 𐑥𐑱𐑛 𐑨𐑟 𐑦𐑑 𐑕𐑑𐑮𐑳𐑒 𐑞 𐑕𐑰; 𐑯𐑪𐑑 𐑕𐑴 𐑒𐑢𐑦𐑒, 𐑦𐑯𐑛𐑰𐑛, 𐑚𐑳𐑑 𐑞𐑨𐑑 𐑕𐑳𐑥 𐑝 𐑞 𐑓𐑤𐑲𐑦𐑙 𐑚𐑳𐑚𐑩𐑤𐑟 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑕𐑐𐑮𐑦𐑙𐑒𐑩𐑤𐑛 𐑣𐑻 𐑣𐑳𐑤 𐑢𐑦𐑞 𐑞𐑺 𐑜𐑴𐑕𐑑𐑤𐑦 𐑚𐑨𐑐𐑑𐑦𐑟𐑩𐑥.

𐑨𐑟 ·𐑱𐑣𐑨𐑚 𐑯𐑬 𐑜𐑤𐑲𐑛𐑩𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑛𐑦𐑡𐑧𐑒𐑑𐑩𐑛 ·𐑛𐑦𐑤𐑲𐑑, 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡 𐑤𐑲𐑓-𐑚𐑶 𐑣𐑨𐑙𐑦𐑙 𐑨𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑕𐑑𐑻𐑯 𐑒𐑱𐑥 𐑦𐑯𐑑𐑵 𐑒𐑩𐑯𐑕𐑐𐑦𐑒𐑿𐑩𐑕 𐑮𐑦𐑤𐑰𐑓.

“𐑣𐑭! 𐑘𐑪𐑯𐑛𐑼! 𐑤𐑫𐑒 𐑘𐑪𐑯𐑛𐑼, 𐑥𐑧𐑯!” 𐑒𐑮𐑲𐑛 𐑩 𐑓𐑹𐑚𐑴𐑛𐑦𐑙 𐑝𐑶𐑕 𐑦𐑯 𐑣𐑻 𐑢𐑱𐑒. “𐑦𐑯 𐑝𐑱𐑯, 𐑴, 𐑘𐑰 𐑕𐑑𐑮𐑱𐑯𐑡𐑼𐑟, 𐑘𐑰 𐑓𐑤𐑲 𐑬𐑼 𐑕𐑨𐑛 𐑚𐑧𐑮𐑾𐑤; 𐑘𐑰 𐑚𐑳𐑑 𐑑𐑻𐑯 𐑳𐑕 𐑘𐑹 𐑑𐑨𐑓𐑮𐑱𐑤 𐑑 𐑖𐑴 𐑳𐑕 𐑘𐑹 𐑒𐑪𐑓𐑦𐑯!”
