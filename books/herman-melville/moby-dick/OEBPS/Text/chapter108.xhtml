𐑣𐑴𐑤𐑛; 𐑢𐑲𐑤 ·𐑐𐑮𐑴𐑥𐑰𐑔𐑾𐑕 𐑦𐑟 𐑩𐑚𐑬𐑑 𐑦𐑑, 𐑲'𐑤 𐑹𐑛𐑼 𐑩 𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑥𐑨𐑯 𐑭𐑓𐑑𐑼 𐑩 𐑛𐑦𐑟𐑲𐑼𐑩𐑚𐑩𐑤 𐑐𐑨𐑑𐑼𐑯. 𐑦𐑥𐑐𐑮𐑲𐑥𐑦𐑕, 𐑓𐑦𐑓𐑑𐑦 𐑓𐑰𐑑 𐑣𐑲 𐑦𐑯 𐑣𐑦𐑟 𐑕𐑪𐑒𐑕; 𐑞𐑧𐑯, 𐑗𐑧𐑕𐑑 𐑥𐑪𐑛𐑩𐑤𐑛 𐑭𐑓𐑑𐑼 𐑞 ·𐑑𐑧𐑥𐑟 ·𐑑𐑳𐑯𐑩𐑤; 𐑞𐑧𐑯, 𐑤𐑧𐑜𐑟 𐑢𐑦𐑞 𐑮𐑵𐑑𐑕 𐑑 '𐑩𐑥, 𐑑 𐑕𐑑𐑱 𐑦𐑯 𐑢𐑳𐑯 𐑐𐑤𐑱𐑕; 𐑞𐑧𐑯, 𐑸𐑥𐑟 𐑔𐑮𐑰 𐑓𐑰𐑑 𐑔𐑮𐑵 𐑞 𐑮𐑦𐑕𐑑; 𐑯𐑴 𐑣𐑸𐑑 𐑨𐑑 𐑷𐑤, 𐑚𐑮𐑭𐑕 𐑓𐑪𐑮𐑦𐑛, 𐑯 𐑩𐑚𐑬𐑑 𐑩 𐑒𐑢𐑹𐑑𐑼 𐑝 𐑩𐑯 𐑱𐑒𐑼 𐑝 𐑓𐑲𐑯 𐑚𐑮𐑱𐑯𐑟; 𐑯 𐑤𐑧𐑑 𐑥𐑰 𐑕𐑰—𐑖𐑨𐑤 𐑲 𐑹𐑛𐑼 𐑲𐑟 𐑑 𐑕𐑰 𐑬𐑑𐑢𐑼𐑛𐑟? 𐑯𐑴, 𐑚𐑳𐑑 𐑐𐑫𐑑 𐑩 𐑕𐑒𐑲-𐑤𐑲𐑑 𐑪𐑯 𐑑𐑪𐑐 𐑝 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑑 𐑦𐑤𐑵𐑥𐑦𐑯𐑱𐑑 𐑦𐑯𐑢𐑼𐑛𐑟. 𐑞𐑺, 𐑑𐑱𐑒 𐑞 𐑹𐑛𐑼, 𐑯 𐑩𐑢𐑱.

𐑯𐑬, 𐑢𐑪𐑑'𐑕 𐑣𐑰 𐑕𐑐𐑰𐑒𐑦𐑙 𐑩𐑚𐑬𐑑, 𐑯 𐑣𐑵'𐑟 𐑣𐑰 𐑕𐑐𐑰𐑒𐑦𐑙 𐑑, 𐑲 𐑖𐑵𐑛 𐑤𐑲𐑒 𐑑 𐑯𐑴? 𐑖𐑨𐑤 𐑲 𐑒𐑰𐑐 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑣𐑽? (𐑩𐑕𐑲𐑛).

'𐑑𐑦𐑟 𐑚𐑳𐑑 𐑦𐑯𐑛𐑦𐑓𐑼𐑩𐑯𐑑 𐑸𐑒𐑦𐑑𐑧𐑒𐑗𐑼 𐑑 𐑥𐑱𐑒 𐑩 𐑚𐑤𐑲𐑯𐑛 𐑛𐑴𐑥; 𐑣𐑽'𐑟 𐑢𐑳𐑯. 𐑯𐑴, 𐑯𐑴, 𐑯𐑴; 𐑲 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑩 𐑤𐑨𐑯𐑑𐑼𐑯.

𐑣𐑴, 𐑣𐑴! 𐑞𐑨𐑑'𐑕 𐑦𐑑, 𐑣𐑱? 𐑣𐑽 𐑸 𐑑𐑵, 𐑕𐑻; 𐑢𐑳𐑯 𐑢𐑦𐑤 𐑕𐑻𐑝 𐑥𐑲 𐑑𐑻𐑯.

𐑢𐑪𐑑 𐑸𐑑 𐑞𐑬 𐑔𐑮𐑳𐑕𐑑𐑦𐑙 𐑞𐑨𐑑 𐑔𐑰𐑓-𐑒𐑨𐑗𐑼 𐑦𐑯𐑑𐑵 𐑥𐑲 𐑓𐑱𐑕 𐑓, 𐑥𐑨𐑯? 𐑔𐑮𐑳𐑕𐑑𐑩𐑛 𐑤𐑲𐑑 𐑦𐑟 𐑢𐑻𐑕 𐑞𐑨𐑯 𐑐𐑮𐑦𐑟𐑧𐑯𐑑𐑩𐑛 𐑐𐑦𐑕𐑑𐑩𐑤𐑟.

𐑲 𐑔𐑷𐑑, 𐑕𐑻, 𐑞𐑨𐑑 𐑿 𐑕𐑐𐑴𐑒 𐑑 𐑒𐑸𐑐𐑩𐑯𐑑𐑼.

𐑒𐑸𐑐𐑩𐑯𐑑𐑼? 𐑢𐑲 𐑞𐑨𐑑'𐑕—𐑚𐑳𐑑 𐑯𐑴;—𐑩 𐑝𐑧𐑮𐑦 𐑑𐑲𐑛𐑦, 𐑯, 𐑲 𐑥𐑱 𐑕𐑱, 𐑩𐑯 𐑧𐑒𐑕𐑑𐑮𐑰𐑥𐑤𐑦 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯𐑤𐑲𐑒 𐑕𐑹𐑑 𐑝 𐑚𐑦𐑟𐑯𐑩𐑕 𐑞𐑬 𐑸𐑑 𐑦𐑯 𐑣𐑽, 𐑒𐑸𐑐𐑩𐑯𐑑𐑼;—𐑹 𐑢𐑫𐑛'𐑕𐑑 𐑞𐑬 𐑮𐑭𐑞𐑼 𐑢𐑻𐑒 𐑦𐑯 𐑒𐑤𐑱?

𐑕𐑻?—𐑒𐑤𐑱? 𐑒𐑤𐑱, 𐑕𐑻? 𐑞𐑨𐑑'𐑕 𐑥𐑳𐑛; 𐑢𐑰 𐑤𐑰𐑝 𐑒𐑤𐑱 𐑑 𐑛𐑦𐑗𐑼𐑟, 𐑕𐑻.

𐑞 𐑓𐑧𐑤𐑴'𐑟 𐑦𐑥𐑐𐑲𐑩𐑕! 𐑢𐑪𐑑 𐑸𐑑 𐑞𐑬 𐑕𐑯𐑰𐑟𐑦𐑙 𐑩𐑚𐑬𐑑?

𐑚𐑴𐑯 𐑦𐑟 𐑮𐑭𐑞𐑼 𐑛𐑳𐑕𐑑𐑦, 𐑕𐑻.

𐑑𐑱𐑒 𐑞 𐑣𐑦𐑯𐑑, 𐑞𐑧𐑯; 𐑯 𐑢𐑧𐑯 𐑞𐑬 𐑸𐑑 𐑛𐑧𐑛, 𐑯𐑧𐑝𐑼 𐑚𐑧𐑮𐑦 𐑞𐑲𐑕𐑧𐑤𐑓 𐑳𐑯𐑛𐑼 𐑤𐑦𐑝𐑦𐑙 𐑐𐑰𐑐𐑩𐑤'𐑟 𐑯𐑴𐑟𐑩𐑟.

𐑕𐑻?—𐑴! 𐑭!—𐑲 𐑜𐑧𐑕 𐑕𐑴;—𐑘𐑧𐑕—𐑴, 𐑛𐑽!

𐑤𐑫𐑒 𐑘𐑰, 𐑒𐑸𐑐𐑩𐑯𐑑𐑼, 𐑲 𐑛𐑺 𐑕𐑱 𐑞𐑬 𐑒𐑷𐑤𐑩𐑕𐑑 𐑞𐑲𐑕𐑧𐑤𐑓 𐑩 𐑮𐑲𐑑 𐑜𐑫𐑛 𐑢𐑻𐑒𐑥𐑩𐑯𐑤𐑲𐑒 𐑢𐑻𐑒𐑥𐑩𐑯, 𐑱? 𐑢𐑧𐑤, 𐑞𐑧𐑯, 𐑢𐑦𐑤 𐑦𐑑 𐑕𐑐𐑰𐑒 𐑔𐑳𐑮𐑴𐑤𐑦 𐑢𐑧𐑤 𐑓 𐑞𐑲 𐑢𐑻𐑒, 𐑦𐑓, 𐑢𐑧𐑯 𐑲 𐑒𐑳𐑥 𐑑 𐑥𐑬𐑯𐑑 𐑞𐑦𐑕 𐑤𐑧𐑜 𐑞𐑬 𐑥𐑱𐑒𐑩𐑕𐑑, 𐑲 𐑖𐑨𐑤 𐑯𐑧𐑝𐑼𐑞𐑩𐑤𐑧𐑕 𐑓𐑰𐑤 𐑩𐑯𐑳𐑞𐑼 𐑤𐑧𐑜 𐑦𐑯 𐑞 𐑕𐑱𐑥 𐑲𐑛𐑧𐑯𐑑𐑦𐑒𐑩𐑤 𐑐𐑤𐑱𐑕 𐑢𐑦𐑞 𐑦𐑑; 𐑞𐑨𐑑 𐑦𐑟, 𐑒𐑸𐑐𐑩𐑯𐑑𐑼, 𐑥𐑲 𐑴𐑤𐑛 𐑤𐑪𐑕𐑑 𐑤𐑧𐑜; 𐑞 𐑓𐑤𐑧𐑖 𐑯 𐑚𐑤𐑳𐑛 𐑢𐑳𐑯, 𐑲 𐑥𐑰𐑯. 𐑒𐑨𐑯𐑕𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑛𐑮𐑲𐑝 𐑞𐑨𐑑 𐑴𐑤𐑛 ·𐑨𐑛𐑩𐑥 𐑩𐑢𐑱?

𐑑𐑮𐑵𐑤𐑦, 𐑕𐑻, 𐑲 𐑚𐑦𐑜𐑦𐑯 𐑑 𐑳𐑯𐑛𐑼𐑕𐑑𐑨𐑯𐑛 𐑕𐑳𐑥𐑢𐑪𐑑 𐑯𐑬. 𐑘𐑧𐑕, 𐑲 𐑣𐑨𐑝 𐑣𐑻𐑛 𐑕𐑳𐑥𐑔𐑦𐑙 𐑒𐑘𐑫𐑼𐑾𐑕 𐑪𐑯 𐑞𐑨𐑑 𐑕𐑒𐑹, 𐑕𐑻; 𐑣𐑬 𐑞𐑨𐑑 𐑩 𐑛𐑦𐑕𐑥𐑭𐑕𐑑𐑩𐑛 𐑥𐑨𐑯 𐑯𐑧𐑝𐑼 𐑦𐑯𐑑𐑲𐑼𐑤𐑦 𐑤𐑵𐑟𐑩𐑟 𐑞 𐑓𐑰𐑤𐑦𐑙 𐑝 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑕𐑐𐑸, 𐑚𐑳𐑑 𐑦𐑑 𐑢𐑦𐑤 𐑚𐑰 𐑕𐑑𐑦𐑤 𐑐𐑮𐑦𐑒𐑦𐑙 𐑣𐑦𐑥 𐑨𐑑 𐑑𐑲𐑥𐑟. 𐑥𐑱 𐑲 𐑣𐑳𐑥𐑚𐑤𐑦 𐑭𐑕𐑒 𐑦𐑓 𐑦𐑑 𐑚𐑰 𐑮𐑾𐑤𐑦 𐑕𐑴, 𐑕𐑻?
