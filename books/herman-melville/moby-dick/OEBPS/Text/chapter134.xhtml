·𐑗𐑨𐑐𐑑𐑼 134. 𐑞 ·𐑗𐑱𐑕—·𐑕𐑧𐑒𐑩𐑯𐑛 𐑛𐑱.

𐑨𐑑 𐑛𐑱-𐑚𐑮𐑱𐑒, 𐑞 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑢𐑻 𐑐𐑳𐑙𐑒𐑗𐑵𐑩𐑤𐑦 𐑥𐑨𐑯𐑛 𐑩𐑓𐑮𐑧𐑖.

“·𐑛'𐑘𐑰 𐑕𐑰 𐑣𐑦𐑥?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚 𐑭𐑓𐑑𐑼 𐑩𐑤𐑬𐑦𐑙 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑕𐑐𐑱𐑕 𐑓 𐑞 𐑤𐑲𐑑 𐑑 𐑕𐑐𐑮𐑧𐑛.

“𐑕𐑰 𐑯𐑳𐑔𐑦𐑙, 𐑕𐑻.”

“𐑑𐑻𐑯 𐑳𐑐 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟 𐑯 𐑥𐑱𐑒 𐑕𐑱𐑤! 𐑣𐑰 𐑑𐑮𐑨𐑝𐑩𐑤𐑟 𐑓𐑭𐑕𐑑𐑼 𐑞𐑨𐑯 𐑲 𐑔𐑷𐑑 𐑓;—𐑞 𐑑𐑪𐑐-𐑜𐑨𐑤𐑩𐑯𐑑 𐑕𐑱𐑤𐑟!—𐑲, 𐑞𐑱 𐑖𐑵𐑛 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑒𐑧𐑐𐑑 𐑪𐑯 𐑣𐑻 𐑷𐑤 𐑯𐑲𐑑. 𐑚𐑳𐑑 𐑯𐑴 𐑥𐑨𐑑𐑼—'𐑑𐑦𐑟 𐑚𐑳𐑑 𐑮𐑧𐑕𐑑𐑦𐑙 𐑓 𐑞 𐑮𐑳𐑖.”

“𐑲, 𐑲!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, “𐑲 𐑯𐑿 𐑦𐑑—𐑘𐑰 𐑒𐑭𐑯𐑑 𐑦𐑕𐑒𐑱𐑐—𐑚𐑤𐑴 𐑪𐑯 𐑯 𐑕𐑐𐑤𐑦𐑑 𐑘𐑹 𐑕𐑐𐑬𐑑, 𐑴 𐑢𐑱𐑤! 𐑞 𐑥𐑨𐑛 𐑓𐑰𐑯𐑛 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑦𐑟 𐑭𐑓𐑑𐑼 𐑘𐑰! 𐑚𐑤𐑴 𐑘𐑹 𐑑𐑮𐑳𐑥𐑐—𐑚𐑤𐑦𐑕𐑑𐑼 𐑘𐑹 𐑤𐑳𐑙𐑟!—𐑱𐑣𐑨𐑚 𐑢𐑦𐑤 𐑛𐑨𐑥 𐑪𐑓 𐑘𐑹 𐑚𐑤𐑳𐑛, 𐑨𐑟 𐑩 𐑥𐑦𐑤𐑼 𐑖𐑳𐑑𐑕 𐑣𐑦𐑟 𐑢𐑷𐑑𐑼𐑜𐑱𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑕𐑑𐑮𐑰𐑥!”

𐑯 ·𐑕𐑑𐑳𐑚 𐑛𐑦𐑛 𐑚𐑳𐑑 𐑕𐑐𐑰𐑒 𐑬𐑑 𐑓 𐑢𐑧𐑤 𐑯𐑲 𐑷𐑤 𐑞𐑨𐑑 𐑒𐑮𐑵. 𐑞 𐑓𐑮𐑧𐑯𐑟𐑦𐑟 𐑝 𐑞 𐑗𐑱𐑕 𐑣𐑨𐑛 𐑚𐑲 𐑞𐑦𐑕 𐑑𐑲𐑥 𐑢𐑻𐑒𐑑 𐑞𐑧𐑥 𐑚𐑳𐑚𐑩𐑤𐑦𐑙𐑤𐑦 𐑳𐑐, 𐑤𐑲𐑒 𐑴𐑤𐑛 𐑢𐑲𐑯 𐑢𐑻𐑒𐑑 𐑩𐑯𐑿. 𐑢𐑪𐑑𐑧𐑝𐑼 𐑐𐑱𐑤 𐑓𐑽𐑟 𐑯 𐑓𐑹𐑚𐑴𐑛𐑦𐑙𐑟 𐑕𐑳𐑥 𐑝 𐑞𐑧𐑥 𐑥𐑲𐑑 𐑣𐑨𐑝 𐑓𐑧𐑤𐑑 𐑚𐑦𐑓𐑹; 𐑞𐑰𐑟 𐑢𐑻 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑯𐑬 𐑒𐑧𐑐𐑑 𐑬𐑑 𐑝 𐑕𐑲𐑑 𐑔𐑮𐑵 𐑞 𐑜𐑮𐑴𐑦𐑙 𐑷 𐑝 ·𐑱𐑣𐑨𐑚, 𐑚𐑳𐑑 𐑞𐑱 𐑢𐑻 𐑚𐑮𐑴𐑒𐑩𐑯 𐑳𐑐, 𐑯 𐑪𐑯 𐑷𐑤 𐑕𐑲𐑛𐑟 𐑮𐑬𐑑𐑩𐑛, 𐑨𐑟 𐑑𐑦𐑥𐑦𐑛 𐑐𐑮𐑺𐑦 𐑣𐑺𐑟 𐑞𐑨𐑑 𐑕𐑒𐑨𐑑𐑼 𐑚𐑦𐑓𐑹 𐑞 𐑚𐑬𐑯𐑛𐑦𐑙 𐑚𐑲𐑕𐑩𐑯. 𐑞 𐑣𐑨𐑯𐑛 𐑝 ·𐑓𐑱𐑑 𐑣𐑨𐑛 𐑕𐑯𐑨𐑗𐑑 𐑷𐑤 𐑞𐑺 𐑕𐑴𐑤𐑟; 𐑯 𐑚𐑲 𐑞 𐑕𐑑𐑻𐑦𐑙 𐑐𐑧𐑮𐑦𐑤𐑟 𐑝 𐑞 𐑐𐑮𐑰𐑝𐑾𐑕 𐑛𐑱; 𐑞 𐑮𐑨𐑒 𐑝 𐑞 𐑐𐑭𐑕𐑑 𐑯𐑲𐑑'𐑕 𐑕𐑩𐑕𐑐𐑧𐑯𐑕; 𐑞 𐑓𐑦𐑒𐑕𐑑, 𐑳𐑯𐑓𐑽𐑦𐑙, 𐑚𐑤𐑲𐑯𐑛, 𐑮𐑧𐑒𐑤𐑩𐑕 𐑢𐑱 𐑦𐑯 𐑢𐑦𐑗 𐑞𐑺 𐑢𐑲𐑤𐑛 𐑒𐑮𐑭𐑓𐑑 𐑢𐑧𐑯𐑑 𐑐𐑤𐑳𐑯𐑡𐑦𐑙 𐑑𐑩𐑢𐑹𐑛𐑟 𐑦𐑑𐑕 𐑓𐑤𐑲𐑦𐑙 𐑥𐑸𐑒; 𐑚𐑲 𐑷𐑤 𐑞𐑰𐑟 𐑔𐑦𐑙𐑟, 𐑞𐑺 𐑣𐑸𐑑𐑕 𐑢𐑻 𐑚𐑴𐑤𐑛 𐑩𐑤𐑪𐑙. 𐑞 𐑢𐑦𐑯𐑛 𐑞𐑨𐑑 𐑥𐑱𐑛 𐑜𐑮𐑱𐑑 𐑚𐑧𐑤𐑦𐑟 𐑝 𐑞𐑺 𐑕𐑱𐑤𐑟, 𐑯 𐑮𐑳𐑖𐑑 𐑞 𐑝𐑧𐑕𐑩𐑤 𐑪𐑯 𐑚𐑲 𐑸𐑥𐑟 𐑦𐑯𐑝𐑦𐑟𐑩𐑚𐑩𐑤 𐑨𐑟 𐑦𐑮𐑦𐑟𐑦𐑕𐑑𐑩𐑚𐑩𐑤; 𐑞𐑦𐑕 𐑕𐑰𐑥𐑛 𐑞 𐑕𐑦𐑥𐑚𐑩𐑤 𐑝 𐑞𐑨𐑑 𐑳𐑯𐑕𐑰𐑯 𐑱𐑡𐑩𐑯𐑕𐑦 𐑢𐑦𐑗 𐑕𐑴 𐑦𐑯𐑕𐑤𐑱𐑝𐑛 𐑞𐑧𐑥 𐑑 𐑞 𐑮𐑱𐑕.

𐑞𐑱 𐑢𐑻 𐑢𐑳𐑯 𐑥𐑨𐑯, 𐑯𐑪𐑑 𐑔𐑻𐑑𐑦. 𐑓 𐑨𐑟 𐑞 𐑢𐑳𐑯 𐑖𐑦𐑐 𐑞𐑨𐑑 𐑣𐑧𐑤𐑛 𐑞𐑧𐑥 𐑷𐑤; 𐑞𐑴 𐑦𐑑 𐑢𐑪𐑟 𐑐𐑫𐑑 𐑑𐑩𐑜𐑧𐑞𐑼 𐑝 𐑷𐑤 𐑒𐑩𐑯𐑑𐑮𐑭𐑕𐑑𐑦𐑙 𐑔𐑦𐑙𐑟—𐑴𐑒, 𐑯 𐑥𐑱𐑐𐑩𐑤, 𐑯 𐑐𐑲𐑯 𐑢𐑫𐑛; 𐑲𐑼𐑯, 𐑯 𐑐𐑦𐑗, 𐑯 𐑣𐑧𐑥𐑐—𐑘𐑧𐑑 𐑷𐑤 𐑞𐑰𐑟 𐑮𐑨𐑯 𐑦𐑯𐑑𐑵 𐑰𐑗 𐑳𐑞𐑼 𐑦𐑯 𐑞 𐑢𐑳𐑯 𐑒𐑪𐑯𐑒𐑮𐑰𐑑 𐑣𐑳𐑤, 𐑢𐑦𐑗 𐑖𐑪𐑑 𐑪𐑯 𐑦𐑑𐑕 𐑢𐑱, 𐑚𐑴𐑔 𐑚𐑨𐑤𐑩𐑯𐑕𐑑 𐑯 𐑛𐑦𐑮𐑧𐑒𐑑𐑩𐑛 𐑚𐑲 𐑞 𐑤𐑪𐑙 𐑕𐑧𐑯𐑑𐑮𐑩𐑤 𐑒𐑰𐑤; 𐑰𐑝𐑩𐑯 𐑕𐑴, 𐑷𐑤 𐑞 𐑦𐑯𐑛𐑦𐑝𐑦𐑡𐑵𐑨𐑤𐑦𐑑𐑦𐑟 𐑝 𐑞 𐑒𐑮𐑵, 𐑞𐑦𐑕 𐑥𐑨𐑯'𐑟 𐑝𐑨𐑤𐑼, 𐑞𐑨𐑑 𐑥𐑨𐑯'𐑟 𐑓𐑽; 𐑜𐑦𐑤𐑑 𐑯 𐑜𐑦𐑤𐑑𐑦𐑯𐑩𐑕, 𐑷𐑤 𐑝𐑼𐑲𐑩𐑑𐑦𐑟 𐑢𐑻 𐑢𐑧𐑤𐑛𐑩𐑛 𐑦𐑯𐑑𐑵 𐑢𐑳𐑯𐑯𐑩𐑕, 𐑯 𐑢𐑻 𐑷𐑤 𐑛𐑦𐑮𐑧𐑒𐑑𐑩𐑛 𐑑 𐑞𐑨𐑑 𐑓𐑱𐑑𐑩𐑤 𐑜𐑴𐑤 𐑢𐑦𐑗 ·𐑱𐑣𐑨𐑚 𐑞𐑺 𐑢𐑳𐑯 𐑤𐑹𐑛 𐑯 𐑒𐑰𐑤 𐑛𐑦𐑛 𐑐𐑶𐑯𐑑 𐑑.

𐑞 𐑮𐑦𐑜𐑦𐑙 𐑤𐑦𐑝𐑛. 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟, 𐑤𐑲𐑒 𐑞 𐑑𐑪𐑐𐑕 𐑝 𐑑𐑷𐑤 𐑐𐑭𐑥𐑟, 𐑢𐑻 𐑬𐑑𐑕𐑐𐑮𐑧𐑛𐑦𐑙𐑤𐑦 𐑑𐑳𐑓𐑑𐑩𐑛 𐑢𐑦𐑞 𐑸𐑥𐑟 𐑯 𐑤𐑧𐑜𐑟. 𐑒𐑤𐑦𐑙𐑦𐑙 𐑑 𐑩 𐑕𐑐𐑸 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛, 𐑕𐑳𐑥 𐑮𐑰𐑗𐑑 𐑓𐑹𐑔 𐑞 𐑳𐑞𐑼 𐑢𐑦𐑞 𐑦𐑥𐑐𐑱𐑖𐑩𐑯𐑑 𐑢𐑱𐑝𐑦𐑙𐑟; 𐑳𐑞𐑼𐑟, 𐑖𐑱𐑛𐑦𐑙 𐑞𐑺 𐑲𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑝𐑦𐑝𐑦𐑛 𐑕𐑳𐑯𐑤𐑲𐑑, 𐑕𐑨𐑑 𐑓𐑸 𐑬𐑑 𐑪𐑯 𐑞 𐑮𐑪𐑒𐑦𐑙 𐑘𐑸𐑛𐑟; 𐑷𐑤 𐑞 𐑕𐑐𐑸𐑟 𐑦𐑯 𐑓𐑫𐑤 𐑚𐑺𐑦𐑙 𐑝 𐑥𐑹𐑑𐑩𐑤𐑟, 𐑮𐑧𐑛𐑦 𐑯 𐑮𐑲𐑐 𐑓 𐑞𐑺 𐑓𐑱𐑑. 𐑭! 𐑣𐑬 𐑞𐑱 𐑕𐑑𐑦𐑤 𐑕𐑑𐑮𐑴𐑝 𐑔𐑮𐑵 𐑞𐑨𐑑 𐑦𐑯𐑓𐑦𐑯𐑦𐑑 𐑚𐑤𐑵𐑯𐑩𐑕 𐑑 𐑕𐑰𐑒 𐑬𐑑 𐑞 𐑔𐑦𐑙 𐑞𐑨𐑑 𐑥𐑲𐑑 𐑛𐑦𐑕𐑑𐑮𐑶 𐑞𐑧𐑥!

“𐑢𐑲 𐑕𐑦𐑙 𐑘𐑰 𐑯𐑪𐑑 𐑬𐑑 𐑓 𐑣𐑦𐑥, 𐑦𐑓 𐑘𐑰 𐑕𐑰 𐑣𐑦𐑥?” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑢𐑧𐑯, 𐑭𐑓𐑑𐑼 𐑞 𐑤𐑨𐑐𐑕 𐑝 𐑕𐑳𐑥 𐑥𐑦𐑯𐑦𐑑𐑕 𐑕𐑦𐑯𐑕 𐑞 𐑓𐑻𐑕𐑑 𐑒𐑮𐑲, 𐑯𐑴 𐑥𐑹 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑣𐑻𐑛. “𐑕𐑢𐑱 𐑥𐑰 𐑳𐑐, 𐑥𐑧𐑯; 𐑘𐑰 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑛𐑦𐑕𐑰𐑝𐑛; 𐑯𐑪𐑑 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑒𐑭𐑕𐑑𐑕 𐑢𐑳𐑯 𐑪𐑛 𐑡𐑧𐑑 𐑞𐑨𐑑 𐑢𐑱, 𐑯 𐑞𐑧𐑯 𐑛𐑦𐑕𐑩𐑐𐑽𐑟.”

𐑦𐑑 𐑢𐑪𐑟 𐑰𐑝𐑩𐑯 𐑕𐑴; 𐑦𐑯 𐑞𐑺 𐑣𐑧𐑛𐑤𐑪𐑙 𐑰𐑜𐑼𐑯𐑩𐑕, 𐑞 𐑥𐑧𐑯 𐑣𐑨𐑛 𐑥𐑦𐑕𐑑𐑱𐑒𐑩𐑯 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑔𐑦𐑙 𐑓 𐑞 𐑢𐑱𐑤-𐑕𐑐𐑬𐑑, 𐑨𐑟 𐑞 𐑦𐑝𐑧𐑯𐑑 𐑦𐑑𐑕𐑧𐑤𐑓 𐑕𐑵𐑯 𐑐𐑮𐑵𐑝𐑛; 𐑓 𐑣𐑸𐑛𐑤𐑦 𐑣𐑨𐑛 ·𐑱𐑣𐑨𐑚 𐑮𐑰𐑗𐑑 𐑣𐑦𐑟 𐑐𐑻𐑗; 𐑣𐑸𐑛𐑤𐑦 𐑢𐑪𐑟 𐑞 𐑮𐑴𐑐 𐑚𐑦𐑤𐑱𐑛 𐑑 𐑦𐑑𐑕 𐑐𐑦𐑯 𐑪𐑯 𐑛𐑧𐑒, 𐑢𐑧𐑯 𐑣𐑰 𐑕𐑑𐑮𐑳𐑒 𐑞 𐑒𐑰-𐑯𐑴𐑑 𐑑 𐑩𐑯 𐑹𐑒𐑦𐑕𐑑𐑮𐑩, 𐑞𐑨𐑑 𐑥𐑱𐑛 𐑞 𐑺 𐑝𐑲𐑚𐑮𐑱𐑑 𐑨𐑟 𐑢𐑦𐑞 𐑞 𐑒𐑩𐑥𐑚𐑲𐑯𐑛 𐑛𐑦𐑕𐑗𐑸𐑡𐑩𐑟 𐑝 𐑮𐑲𐑓𐑩𐑤𐑟. 𐑞 𐑑𐑮𐑲𐑳𐑥𐑓𐑩𐑯𐑑 𐑣𐑩𐑤𐑵 𐑝 𐑔𐑻𐑑𐑦 𐑚𐑳𐑒𐑕𐑒𐑦𐑯 𐑤𐑳𐑙𐑟 𐑢𐑪𐑟 𐑣𐑻𐑛, 𐑨𐑟—𐑥𐑳𐑗 𐑯𐑽𐑼 𐑑 𐑞 𐑖𐑦𐑐 𐑞𐑨𐑯 𐑞 𐑐𐑤𐑱𐑕 𐑝 𐑞 𐑦𐑥𐑨𐑡𐑦𐑯𐑼𐑦 𐑡𐑧𐑑, 𐑤𐑧𐑕 𐑞𐑨𐑯 𐑩 𐑥𐑲𐑤 𐑩𐑣𐑧𐑛—·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑚𐑪𐑛𐑦𐑤𐑦 𐑚𐑻𐑕𐑑 𐑦𐑯𐑑𐑵 𐑝𐑿! 𐑓 𐑯𐑪𐑑 𐑚𐑲 𐑧𐑯𐑦 𐑒𐑭𐑥 𐑯 𐑦𐑯𐑛𐑩𐑤𐑩𐑯𐑑 𐑕𐑐𐑬𐑑𐑦𐑙𐑟; 𐑯𐑪𐑑 𐑚𐑲 𐑞 𐑐𐑰𐑕𐑩𐑚𐑩𐑤 𐑜𐑳𐑖 𐑝 𐑞𐑨𐑑 𐑥𐑦𐑕𐑑𐑦𐑒 𐑓𐑬𐑯𐑑𐑩𐑯 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑛𐑦𐑛 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑯𐑬 𐑮𐑦𐑝𐑰𐑤 𐑣𐑦𐑟 𐑝𐑦𐑕𐑦𐑯𐑦𐑑𐑦; 𐑚𐑳𐑑 𐑚𐑲 𐑞 𐑓𐑸 𐑥𐑹 𐑢𐑳𐑯𐑛𐑮𐑩𐑕 𐑓𐑦𐑯𐑪𐑥𐑦𐑯𐑪𐑯 𐑝 𐑚𐑮𐑰𐑗𐑦𐑙. 𐑮𐑲𐑟𐑦𐑙 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑳𐑑𐑥𐑴𐑕𐑑 𐑝𐑦𐑤𐑪𐑕𐑦𐑑𐑦 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑻𐑞𐑩𐑕𐑑 𐑛𐑧𐑐𐑔𐑕, 𐑞 ·𐑕𐑐𐑻𐑥 ·𐑢𐑱𐑤 𐑞𐑳𐑕 𐑚𐑵𐑥𐑟 𐑣𐑦𐑟 𐑦𐑯𐑑𐑲𐑼 𐑚𐑳𐑤𐑒 𐑦𐑯𐑑𐑵 𐑞 𐑐𐑘𐑫𐑼 𐑧𐑤𐑦𐑥𐑩𐑯𐑑 𐑝 𐑺, 𐑯 𐑐𐑲𐑤𐑦𐑙 𐑳𐑐 𐑩 𐑥𐑬𐑯𐑑𐑩𐑯 𐑝 𐑛𐑨𐑟𐑩𐑤𐑦𐑙 𐑓𐑴𐑥, 𐑖𐑴𐑟 𐑣𐑦𐑟 𐑐𐑤𐑱𐑕 𐑑 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑝 𐑕𐑧𐑝𐑩𐑯 𐑥𐑲𐑤𐑟 𐑯 𐑥𐑹. 𐑦𐑯 𐑞𐑴𐑟 𐑥𐑴𐑥𐑧𐑯𐑑𐑕, 𐑞 𐑑𐑹𐑯, 𐑦𐑯𐑮𐑱𐑡𐑛 𐑢𐑱𐑝𐑟 𐑣𐑰 𐑖𐑱𐑒𐑕 𐑪𐑓, 𐑕𐑰𐑥 𐑣𐑦𐑟 𐑥𐑱𐑯; 𐑦𐑯 𐑕𐑳𐑥 𐑒𐑱𐑕𐑩𐑟, 𐑞𐑦𐑕 𐑚𐑮𐑰𐑗𐑦𐑙 𐑦𐑟 𐑣𐑦𐑟 𐑨𐑒𐑑 𐑝 𐑛𐑦𐑓𐑲𐑩𐑯𐑕.

“𐑞𐑺 𐑖𐑰 𐑚𐑮𐑰𐑗𐑩𐑟! 𐑞𐑺 𐑖𐑰 𐑚𐑮𐑰𐑗𐑩𐑟!” 𐑢𐑪𐑟 𐑞 𐑒𐑮𐑲, 𐑨𐑟 𐑦𐑯 𐑣𐑦𐑟 𐑦𐑥𐑧𐑠𐑼𐑩𐑚𐑩𐑤 𐑚𐑮𐑩𐑝𐑭𐑛𐑴𐑟 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑑𐑪𐑕𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑕𐑨𐑥𐑩𐑯-𐑤𐑲𐑒 𐑑 ·𐑣𐑧𐑝𐑩𐑯. 𐑕𐑴 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑕𐑰𐑯 𐑦𐑯 𐑞 𐑚𐑤𐑵 𐑐𐑤𐑱𐑯 𐑝 𐑞 𐑕𐑰, 𐑯 𐑮𐑦𐑤𐑰𐑝𐑛 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑕𐑑𐑦𐑤 𐑚𐑤𐑵𐑼 𐑥𐑸𐑡𐑦𐑯 𐑝 𐑞 𐑕𐑒𐑲, 𐑞 𐑕𐑐𐑮𐑱 𐑞𐑨𐑑 𐑣𐑰 𐑮𐑱𐑟𐑛, 𐑓 𐑞 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑦𐑯𐑑𐑪𐑤𐑼𐑩𐑚𐑤𐑦 𐑜𐑤𐑦𐑑𐑼𐑛 𐑯 𐑜𐑤𐑺𐑛 𐑤𐑲𐑒 𐑩 𐑜𐑤𐑱𐑕𐑽; 𐑯 𐑕𐑑𐑫𐑛 𐑞𐑺 𐑜𐑮𐑨𐑡𐑵𐑩𐑤𐑦 𐑓𐑱𐑛𐑦𐑙 𐑯 𐑓𐑱𐑛𐑦𐑙 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑦𐑑𐑕 𐑓𐑻𐑕𐑑 𐑕𐑐𐑸𐑒𐑩𐑤𐑦𐑙 𐑦𐑯𐑑𐑧𐑯𐑕𐑦𐑑𐑦, 𐑑 𐑞 𐑛𐑦𐑥 𐑥𐑦𐑕𐑑𐑦𐑯𐑩𐑕 𐑝 𐑩𐑯 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙 𐑖𐑬𐑼 𐑦𐑯 𐑩 𐑝𐑱𐑤.

“𐑲, 𐑚𐑮𐑰𐑗 𐑘𐑹 𐑤𐑭𐑕𐑑 𐑑 𐑞 𐑕𐑳𐑯, ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, “𐑞𐑲 𐑬𐑼 𐑯 𐑞𐑲 𐑣𐑸𐑐𐑵𐑯 𐑸 𐑨𐑑 𐑣𐑨𐑯𐑛!—𐑛𐑬𐑯! 𐑛𐑬𐑯 𐑷𐑤 𐑝 𐑘𐑰, 𐑚𐑳𐑑 𐑢𐑳𐑯 𐑥𐑨𐑯 𐑨𐑑 𐑞 𐑓𐑹. 𐑞 𐑚𐑴𐑑𐑕!—𐑕𐑑𐑨𐑯𐑛 𐑚𐑲!”

𐑳𐑯𐑥𐑲𐑯𐑛𐑓𐑩𐑤 𐑝 𐑞 𐑑𐑰𐑛𐑾𐑕 𐑮𐑴𐑐-𐑤𐑨𐑛𐑼𐑟 𐑝 𐑞 𐑖𐑮𐑬𐑛𐑟, 𐑞 𐑥𐑧𐑯, 𐑤𐑲𐑒 𐑖𐑵𐑑𐑦𐑙 𐑕𐑑𐑸𐑟, 𐑕𐑤𐑦𐑛 𐑑 𐑞 𐑛𐑧𐑒, 𐑚𐑲 𐑞 𐑲𐑕𐑩𐑤𐑱𐑑𐑩𐑛 𐑚𐑨𐑒𐑕𐑑𐑱𐑟 𐑯 𐑣𐑨𐑤𐑘𐑼𐑛𐑟; 𐑢𐑲𐑤 ·𐑱𐑣𐑨𐑚, 𐑤𐑧𐑕 𐑛𐑸𐑑𐑦𐑙𐑤𐑦, 𐑚𐑳𐑑 𐑕𐑑𐑦𐑤 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑢𐑪𐑟 𐑛𐑮𐑪𐑐𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑐𐑻𐑗.

“𐑤𐑴𐑼 𐑩𐑢𐑱,” 𐑣𐑰 𐑒𐑮𐑲𐑛, 𐑕𐑴 𐑕𐑵𐑯 𐑨𐑟 𐑣𐑰 𐑣𐑨𐑛 𐑮𐑰𐑗𐑑 𐑣𐑦𐑟 𐑚𐑴𐑑—𐑩 𐑕𐑐𐑺 𐑢𐑳𐑯, 𐑮𐑦𐑜𐑛 𐑞 𐑭𐑓𐑑𐑼𐑯𐑵𐑯 𐑐𐑮𐑰𐑝𐑾𐑕. “·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑞 𐑖𐑦𐑐 𐑦𐑟 𐑞𐑲𐑯—𐑒𐑰𐑐 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑴𐑑𐑕, 𐑚𐑳𐑑 𐑒𐑰𐑐 𐑯𐑽 𐑞𐑧𐑥. 𐑤𐑴𐑼, 𐑷𐑤!”

𐑨𐑟 𐑚𐑦𐑓𐑹, 𐑞 𐑩𐑑𐑧𐑯𐑑𐑦𐑝 𐑖𐑦𐑐 𐑣𐑨𐑝𐑦𐑙 𐑛𐑦𐑕𐑒𐑮𐑲𐑛 𐑞 𐑣𐑴𐑤 𐑓𐑲𐑑, 𐑩𐑜𐑱𐑯 𐑒𐑱𐑥 𐑚𐑺𐑦𐑙 𐑛𐑬𐑯 𐑑 𐑞 𐑮𐑧𐑕𐑒𐑿, 𐑯 𐑛𐑮𐑪𐑐𐑦𐑙 𐑩 𐑚𐑴𐑑, 𐑐𐑦𐑒𐑑 𐑳𐑐 𐑞 𐑓𐑤𐑴𐑑𐑦𐑙 𐑥𐑨𐑮𐑦𐑯𐑼𐑟, 𐑑𐑳𐑚𐑟, 𐑹𐑟, 𐑯 𐑢𐑪𐑑𐑧𐑝𐑼 𐑧𐑤𐑕 𐑒𐑫𐑛 𐑚𐑰 𐑒𐑷𐑑 𐑨𐑑, 𐑯 𐑕𐑱𐑓𐑤𐑦 𐑤𐑨𐑯𐑛𐑩𐑛 𐑞𐑧𐑥 𐑪𐑯 𐑣𐑻 𐑛𐑧𐑒𐑕. 𐑕𐑳𐑥 𐑕𐑐𐑮𐑱𐑯𐑛 𐑖𐑴𐑤𐑛𐑼𐑟, 𐑮𐑦𐑕𐑑𐑕, 𐑯 𐑨𐑙𐑒𐑩𐑤𐑟; 𐑤𐑦𐑝𐑦𐑛 𐑒𐑩𐑯𐑑𐑿𐑠𐑩𐑯𐑟; 𐑮𐑧𐑯𐑗𐑑 𐑣𐑸𐑐𐑵𐑯𐑟 𐑯 𐑤𐑭𐑯𐑕𐑩𐑟; 𐑦𐑯𐑧𐑒𐑕𐑑𐑮𐑦𐑒𐑩𐑚𐑩𐑤 𐑦𐑯𐑑𐑮𐑦𐑒𐑩𐑕𐑦𐑟 𐑝 𐑮𐑴𐑐; 𐑖𐑨𐑑𐑼𐑛 𐑹𐑟 𐑯 𐑐𐑤𐑨𐑙𐑒𐑕; 𐑷𐑤 𐑞𐑰𐑟 𐑢𐑻 𐑞𐑺; 𐑚𐑳𐑑 𐑯𐑴 𐑓𐑱𐑑𐑩𐑤 𐑹 𐑰𐑝𐑩𐑯 𐑕𐑽𐑾𐑕 𐑦𐑤 𐑕𐑰𐑥𐑛 𐑑 𐑣𐑨𐑝 𐑚𐑦𐑓𐑷𐑤𐑩𐑯 𐑧𐑯𐑦 𐑢𐑳𐑯. 𐑨𐑟 𐑢𐑦𐑞 ·𐑓𐑧𐑛𐑩𐑤𐑩 𐑞 𐑛𐑱 𐑚𐑦𐑓𐑹, 𐑕𐑴 ·𐑱𐑣𐑨𐑚 𐑢𐑪𐑟 𐑯𐑬 𐑓𐑬𐑯𐑛 𐑜𐑮𐑦𐑥𐑤𐑦 𐑒𐑤𐑦𐑙𐑦𐑙 𐑑 𐑣𐑦𐑟 𐑚𐑴𐑑'𐑕 𐑚𐑮𐑴𐑒𐑩𐑯 𐑣𐑭𐑓, 𐑢𐑦𐑗 𐑩𐑓𐑹𐑛𐑩𐑛 𐑩 𐑒𐑩𐑥𐑐𐑨𐑮𐑩𐑑𐑦𐑝𐑤𐑦 𐑰𐑟𐑦 𐑓𐑤𐑴𐑑; 𐑯𐑹 𐑛𐑦𐑛 𐑦𐑑 𐑕𐑴 𐑦𐑜𐑟𐑷𐑕𐑑 𐑣𐑦𐑥 𐑨𐑟 𐑞 𐑐𐑮𐑰𐑝𐑾𐑕 𐑛𐑱'𐑟 𐑥𐑦𐑕𐑣𐑨𐑐.

𐑚𐑳𐑑 𐑢𐑧𐑯 𐑣𐑰 𐑢𐑪𐑟 𐑣𐑧𐑤𐑐𐑑 𐑑 𐑞 𐑛𐑧𐑒, 𐑷𐑤 𐑲𐑟 𐑢𐑻 𐑓𐑭𐑕𐑩𐑯𐑛 𐑩𐑐𐑪𐑯 𐑣𐑦𐑥; 𐑨𐑟 𐑦𐑯𐑕𐑑𐑧𐑛 𐑝 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑲 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑣𐑰 𐑕𐑑𐑦𐑤 𐑣𐑭𐑓-𐑣𐑳𐑙 𐑩𐑐𐑪𐑯 𐑞 𐑖𐑴𐑤𐑛𐑼 𐑝 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑣𐑵 𐑣𐑨𐑛 𐑞𐑳𐑕 𐑓𐑸 𐑚𐑰𐑯 𐑞 𐑓𐑹𐑥𐑴𐑕𐑑 𐑑 𐑩𐑕𐑦𐑕𐑑 𐑣𐑦𐑥. 𐑣𐑦𐑟 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑯𐑨𐑐𐑑 𐑪𐑓, 𐑤𐑰𐑝𐑦𐑙 𐑚𐑳𐑑 𐑢𐑳𐑯 𐑖𐑹𐑑 𐑖𐑸𐑐 𐑕𐑐𐑤𐑦𐑯𐑑𐑼.

“𐑲, 𐑲, ·𐑕𐑑𐑸𐑚𐑳𐑒, '𐑑𐑦𐑟 𐑕𐑢𐑰𐑑 𐑑 𐑤𐑰𐑯 𐑕𐑳𐑥𐑑𐑲𐑥𐑟, 𐑚𐑰 𐑞 𐑤𐑰𐑯𐑼 𐑣𐑵 𐑣𐑰 𐑢𐑦𐑤; 𐑯 𐑢𐑫𐑛 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚 𐑣𐑨𐑛 𐑤𐑰𐑯𐑛 𐑪𐑓𐑩𐑯𐑼 𐑞𐑨𐑯 𐑣𐑰 𐑣𐑨𐑟.”

“𐑞 𐑓𐑧𐑮𐑵𐑤 𐑣𐑨𐑟 𐑯𐑪𐑑 𐑕𐑑𐑫𐑛, 𐑕𐑻,” 𐑕𐑧𐑛 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼, 𐑯𐑬 𐑒𐑳𐑥𐑦𐑙 𐑳𐑐; “𐑲 𐑐𐑫𐑑 𐑜𐑫𐑛 𐑢𐑻𐑒 𐑦𐑯𐑑𐑵 𐑞𐑨𐑑 𐑤𐑧𐑜.”

“𐑚𐑳𐑑 𐑯𐑴 𐑚𐑴𐑯𐑟 𐑚𐑮𐑴𐑒𐑩𐑯, 𐑕𐑻, 𐑲 𐑣𐑴𐑐,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚 𐑢𐑦𐑞 𐑑𐑮𐑵 𐑒𐑩𐑯𐑕𐑻𐑯.

“𐑲! 𐑯 𐑷𐑤 𐑕𐑐𐑤𐑦𐑯𐑑𐑼𐑛 𐑑 𐑐𐑰𐑕𐑩𐑟, ·𐑕𐑑𐑳𐑚!—𐑛'𐑘𐑰 𐑕𐑰 𐑦𐑑.—𐑚𐑳𐑑 𐑰𐑝𐑩𐑯 𐑢𐑦𐑞 𐑩 𐑚𐑮𐑴𐑒𐑩𐑯 𐑚𐑴𐑯, 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚 𐑦𐑟 𐑳𐑯𐑑𐑳𐑗𐑑; 𐑯 𐑲 𐑩𐑒𐑬𐑯𐑑 𐑯𐑴 𐑤𐑦𐑝𐑦𐑙 𐑚𐑴𐑯 𐑝 𐑥𐑲𐑯 𐑢𐑳𐑯 𐑡𐑪𐑑 𐑥𐑹 𐑥𐑰, 𐑞𐑨𐑯 𐑞𐑦𐑕 𐑛𐑧𐑛 𐑢𐑳𐑯 𐑞𐑨𐑑'𐑕 𐑤𐑪𐑕𐑑. 𐑯𐑹 𐑢𐑲𐑑 𐑢𐑱𐑤, 𐑯𐑹 𐑥𐑨𐑯, 𐑯𐑹 𐑓𐑰𐑯𐑛, 𐑒𐑨𐑯 𐑕𐑴 𐑥𐑳𐑗 𐑨𐑟 𐑜𐑮𐑱𐑟 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚 𐑦𐑯 𐑣𐑦𐑟 𐑴𐑯 𐑐𐑮𐑪𐑐𐑼 𐑯 𐑦𐑯𐑩𐑒𐑕𐑧𐑕𐑩𐑚𐑩𐑤 𐑚𐑰𐑦𐑙. 𐑒𐑨𐑯 𐑧𐑯𐑦 𐑤𐑰𐑛 𐑑𐑳𐑗 𐑘𐑪𐑯𐑛𐑼 𐑓𐑤𐑹, 𐑧𐑯𐑦 𐑥𐑭𐑕𐑑 𐑕𐑒𐑮𐑱𐑐 𐑘𐑪𐑯𐑛𐑼 𐑮𐑵𐑓?—𐑩𐑤𐑪𐑓𐑑 𐑞𐑺! 𐑢𐑦𐑗 𐑢𐑱?”

“𐑛𐑧𐑛 𐑑 𐑤𐑰𐑢𐑼𐑛, 𐑕𐑻.”

“𐑳𐑐 𐑣𐑧𐑤𐑥, 𐑞𐑧𐑯; 𐑐𐑲𐑤 𐑪𐑯 𐑞 𐑕𐑱𐑤 𐑩𐑜𐑱𐑯, 𐑖𐑦𐑐 𐑒𐑰𐑐𐑼𐑟! 𐑛𐑬𐑯 𐑞 𐑮𐑧𐑕𐑑 𐑝 𐑞 𐑕𐑐𐑺 𐑚𐑴𐑑𐑕 𐑯 𐑮𐑦𐑜 𐑞𐑧𐑥—·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑩𐑢𐑱, 𐑯 𐑥𐑳𐑕𐑑𐑼 𐑞 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵𐑟.”

“𐑤𐑧𐑑 𐑥𐑰 𐑓𐑻𐑕𐑑 𐑣𐑧𐑤𐑐 𐑞𐑰 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑕𐑻.”

“𐑴, 𐑴, 𐑴! 𐑣𐑬 𐑞𐑦𐑕 𐑕𐑐𐑤𐑦𐑯𐑑𐑼 𐑜𐑹𐑟 𐑥𐑰 𐑯𐑬! 𐑩𐑒𐑻𐑕𐑩𐑛 𐑓𐑱𐑑! 𐑞𐑨𐑑 𐑞 𐑳𐑯𐑒𐑪𐑙𐑒𐑼𐑩𐑚𐑩𐑤 𐑒𐑨𐑐𐑑𐑦𐑯 𐑦𐑯 𐑞 𐑕𐑴𐑤 𐑖𐑵𐑛 𐑣𐑨𐑝 𐑕𐑳𐑗 𐑩 𐑒𐑮𐑱𐑝𐑩𐑯 𐑥𐑱𐑑!”

“𐑕𐑻?”

“𐑥𐑲 𐑚𐑪𐑛𐑦, 𐑥𐑨𐑯, 𐑯𐑪𐑑 𐑞𐑰. 𐑜𐑦𐑝 𐑥𐑰 𐑕𐑳𐑥𐑔𐑦𐑙 𐑓 𐑩 𐑒𐑱𐑯—𐑞𐑺, 𐑞𐑨𐑑 𐑖𐑦𐑝𐑼𐑛 𐑤𐑭𐑯𐑕 𐑢𐑦𐑤 𐑛𐑵. 𐑥𐑳𐑕𐑑𐑼 𐑞 𐑥𐑧𐑯. 𐑖𐑫𐑼𐑤𐑦 𐑲 𐑣𐑨𐑝 𐑯𐑪𐑑 𐑕𐑰𐑯 𐑣𐑦𐑥 𐑘𐑧𐑑. 𐑚𐑲 𐑣𐑧𐑝𐑩𐑯 𐑦𐑑 𐑒𐑨𐑯𐑪𐑑 𐑚𐑰!—𐑥𐑦𐑕𐑦𐑙?—𐑒𐑢𐑦𐑒! 𐑒𐑷𐑤 𐑞𐑧𐑥 𐑷𐑤.”

𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯'𐑟 𐑣𐑦𐑯𐑑𐑩𐑛 𐑔𐑷𐑑 𐑢𐑪𐑟 𐑑𐑮𐑵. 𐑩𐑐𐑪𐑯 𐑥𐑳𐑕𐑑𐑼𐑦𐑙 𐑞 𐑒𐑳𐑥𐑐𐑩𐑯𐑦, 𐑞 ·𐑐𐑸𐑟𐑰 𐑢𐑪𐑟 𐑯𐑪𐑑 𐑞𐑺.

“𐑞 ·𐑐𐑸𐑟𐑰!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚—“𐑣𐑰 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑒𐑷𐑑 𐑦𐑯——”

“𐑞 𐑚𐑤𐑨𐑒 𐑝𐑪𐑥𐑦𐑑 𐑮𐑧𐑯𐑗 𐑞𐑰!—𐑮𐑳𐑯 𐑷𐑤 𐑝 𐑘𐑰 𐑩𐑚𐑳𐑝, 𐑱𐑤𐑴, 𐑒𐑨𐑚𐑦𐑯, 𐑓𐑴𐑒𐑕𐑩𐑤—𐑓𐑲𐑯𐑛 𐑣𐑦𐑥—𐑯𐑪𐑑 𐑜𐑪𐑯—𐑯𐑪𐑑 𐑜𐑪𐑯!”

𐑚𐑳𐑑 𐑒𐑢𐑦𐑒𐑤𐑦 𐑞𐑱 𐑮𐑩𐑑𐑻𐑯𐑛 𐑑 𐑣𐑦𐑥 𐑢𐑦𐑞 𐑞 𐑑𐑲𐑛𐑦𐑙𐑟 𐑞𐑨𐑑 𐑞 ·𐑐𐑸𐑟𐑰 𐑢𐑪𐑟 𐑯𐑴𐑢𐑺 𐑑 𐑚𐑰 𐑓𐑬𐑯𐑛.

“𐑲, 𐑕𐑻,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚—“𐑒𐑷𐑑 𐑩𐑥𐑳𐑙 𐑞 𐑑𐑨𐑙𐑜𐑩𐑤𐑟 𐑝 𐑘𐑹 𐑤𐑲𐑯—𐑲 𐑔𐑷𐑑 𐑲 𐑕𐑷 𐑣𐑦𐑥 𐑛𐑮𐑨𐑜𐑦𐑙 𐑳𐑯𐑛𐑼.”

“𐑥𐑲 𐑤𐑲𐑯! 𐑥𐑲 𐑤𐑲𐑯? 𐑜𐑪𐑯?—𐑜𐑪𐑯? 𐑢𐑪𐑑 𐑥𐑰𐑯𐑟 𐑞𐑨𐑑 𐑤𐑦𐑑𐑩𐑤 𐑢𐑻𐑛?—𐑢𐑪𐑑 𐑛𐑧𐑔-𐑯𐑧𐑤 𐑮𐑦𐑙𐑟 𐑦𐑯 𐑦𐑑, 𐑞𐑨𐑑 𐑴𐑤𐑛 ·𐑱𐑣𐑨𐑚 𐑖𐑱𐑒𐑕 𐑨𐑟 𐑦𐑓 𐑣𐑰 𐑢𐑻 𐑞 𐑚𐑧𐑤𐑓𐑮𐑦. 𐑞 𐑣𐑸𐑐𐑵𐑯, 𐑑𐑵!—𐑑𐑪𐑕 𐑴𐑝𐑼 𐑞 𐑤𐑦𐑑𐑼 𐑞𐑺,—𐑛'𐑘𐑰 𐑕𐑰 𐑦𐑑?—𐑞 𐑓𐑹𐑡𐑛 𐑲𐑼𐑯, 𐑥𐑧𐑯, 𐑞 𐑢𐑲𐑑 𐑢𐑱𐑤'𐑟—𐑯𐑴, 𐑯𐑴, 𐑯𐑴,—𐑚𐑤𐑦𐑕𐑑𐑼𐑛 𐑓𐑵𐑤! 𐑞𐑦𐑕 𐑣𐑨𐑯𐑛 𐑛𐑦𐑛 𐑛𐑸𐑑 𐑦𐑑!—'𐑑𐑦𐑟 𐑦𐑯 𐑞 𐑓𐑦𐑖!—𐑩𐑤𐑪𐑓𐑑 𐑞𐑺! 𐑒𐑰𐑐 𐑣𐑦𐑥 𐑯𐑱𐑤𐑛—·𐑒𐑢𐑦𐑒!—𐑷𐑤 𐑣𐑨𐑯𐑛𐑟 𐑑 𐑞 𐑮𐑦𐑜𐑦𐑙 𐑝 𐑞 𐑚𐑴𐑑𐑕—𐑒𐑩𐑤𐑧𐑒𐑑 𐑞 𐑹𐑟—𐑣𐑸𐑐𐑵𐑯𐑽𐑟! 𐑞 𐑲𐑼𐑯𐑟, 𐑞 𐑲𐑼𐑯𐑟!—𐑣𐑶𐑕𐑑 𐑞 𐑮𐑶𐑩𐑤𐑟 𐑣𐑲𐑼—𐑩 𐑐𐑫𐑤 𐑪𐑯 𐑷𐑤 𐑞 𐑖𐑰𐑑𐑕!—𐑣𐑧𐑤𐑥 𐑞𐑺! 𐑕𐑑𐑧𐑛𐑦, 𐑕𐑑𐑧𐑛𐑦 𐑓 𐑘𐑹 𐑤𐑲𐑓! 𐑲'𐑤 𐑑𐑧𐑯 𐑑𐑲𐑥𐑟 𐑜𐑻𐑛𐑩𐑤 𐑞 𐑳𐑯𐑥𐑧𐑠𐑼𐑛 𐑜𐑤𐑴𐑚; 𐑘𐑱 𐑯 𐑛𐑲𐑝 𐑕𐑑𐑮𐑱𐑑 𐑔𐑮𐑵 𐑦𐑑, 𐑚𐑳𐑑 𐑲'𐑤 𐑕𐑤𐑱 𐑣𐑦𐑥 𐑘𐑧𐑑!”

“𐑜𐑮𐑱𐑑 ·𐑜𐑪𐑛! 𐑚𐑳𐑑 𐑓 𐑢𐑳𐑯 𐑕𐑦𐑙𐑜𐑩𐑤 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑖𐑴 𐑞𐑲𐑕𐑧𐑤𐑓,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑸𐑚𐑳𐑒; “𐑯𐑧𐑝𐑼, 𐑯𐑧𐑝𐑼 𐑢𐑦𐑤𐑑 𐑞𐑬 𐑒𐑨𐑐𐑗𐑼 𐑣𐑦𐑥, 𐑴𐑤𐑛 𐑥𐑨𐑯—𐑦𐑯 ·𐑡𐑰𐑟𐑩𐑕' 𐑯𐑱𐑥 𐑯𐑴 𐑥𐑹 𐑝 𐑞𐑦𐑕, 𐑞𐑨𐑑'𐑕 𐑢𐑻𐑕 𐑞𐑨𐑯 𐑛𐑧𐑝𐑩𐑤'𐑟 𐑥𐑨𐑛𐑯𐑩𐑕. 𐑑𐑵 𐑛𐑱𐑟 𐑗𐑱𐑕𐑑; 𐑑𐑢𐑲𐑕 𐑕𐑑𐑴𐑝 𐑑 𐑕𐑐𐑤𐑦𐑯𐑑𐑼𐑟; 𐑞𐑲 𐑝𐑧𐑮𐑦 𐑤𐑧𐑜 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑕𐑯𐑨𐑗𐑑 𐑓𐑮𐑪𐑥 𐑳𐑯𐑛𐑼 𐑞𐑰; 𐑞𐑲 𐑰𐑝𐑩𐑤 𐑖𐑨𐑛𐑴 𐑜𐑪𐑯—𐑷𐑤 𐑜𐑫𐑛 𐑱𐑯𐑡𐑩𐑤𐑟 𐑥𐑪𐑚𐑦𐑙 𐑞𐑰 𐑢𐑦𐑞 𐑢𐑹𐑯𐑦𐑙𐑟:—𐑢𐑪𐑑 𐑥𐑹 𐑢𐑫𐑛𐑕𐑑 𐑞𐑬 𐑣𐑨𐑝?—𐑖𐑨𐑤 𐑢𐑰 𐑒𐑰𐑐 𐑗𐑱𐑕𐑦𐑙 𐑞𐑦𐑕 𐑥𐑻𐑛𐑼𐑩𐑕 𐑓𐑦𐑖 𐑑𐑦𐑤 𐑣𐑰 𐑕𐑢𐑪𐑥𐑐𐑕 𐑞 𐑤𐑭𐑕𐑑 𐑥𐑨𐑯? 𐑖𐑨𐑤 𐑢𐑰 𐑚𐑰 𐑛𐑮𐑨𐑜𐑛 𐑚𐑲 𐑣𐑦𐑥 𐑑 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 𐑞 𐑕𐑰? 𐑖𐑨𐑤 𐑢𐑰 𐑚𐑰 𐑑𐑴𐑛 𐑚𐑲 𐑣𐑦𐑥 𐑑 𐑞 𐑦𐑯𐑓𐑻𐑯𐑩𐑤 𐑢𐑻𐑤𐑛? 𐑴, 𐑴,—·𐑦𐑥𐑐𐑲𐑩𐑑𐑦 𐑯 𐑚𐑤𐑨𐑕𐑓𐑩𐑥𐑦 𐑑 𐑣𐑳𐑯𐑑 𐑣𐑦𐑥 𐑥𐑹!”

“𐑕𐑑𐑸𐑚𐑳𐑒, 𐑝 𐑤𐑱𐑑 𐑲'𐑝 𐑓𐑧𐑤𐑑 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑥𐑵𐑝𐑛 𐑑 𐑞𐑰; 𐑧𐑝𐑼 𐑕𐑦𐑯𐑕 𐑞𐑨𐑑 𐑬𐑼 𐑢𐑰 𐑚𐑴𐑔 𐑕𐑷—𐑞𐑬 𐑯𐑴'𐑕𐑑 𐑢𐑪𐑑, 𐑦𐑯 𐑢𐑳𐑯 𐑩𐑯𐑳𐑞𐑼'𐑟 𐑲𐑟. 𐑚𐑳𐑑 𐑦𐑯 𐑞𐑦𐑕 𐑥𐑨𐑑𐑼 𐑝 𐑞 𐑢𐑱𐑤, 𐑚𐑰 𐑞 𐑓𐑮𐑳𐑯𐑑 𐑝 𐑞𐑲 𐑓𐑱𐑕 𐑑 𐑥𐑰 𐑨𐑟 𐑞 𐑐𐑭𐑥 𐑝 𐑞𐑦𐑕 𐑣𐑨𐑯𐑛—𐑩 𐑤𐑦𐑐𐑤𐑩𐑕, 𐑳𐑯𐑓𐑰𐑗𐑼𐑛 𐑚𐑤𐑨𐑙𐑒. 𐑱𐑣𐑨𐑚 𐑦𐑟 𐑓 𐑧𐑝𐑼 ·𐑱𐑣𐑨𐑚, 𐑥𐑨𐑯. 𐑞𐑦𐑕 𐑣𐑴𐑤 𐑨𐑒𐑑'𐑕 𐑦𐑥𐑿𐑑𐑩𐑚𐑤𐑦 𐑛𐑦𐑒𐑮𐑰𐑛. '𐑑𐑢𐑪𐑟 𐑮𐑦𐑣𐑻𐑕𐑑 𐑚𐑲 𐑞𐑰 𐑯 𐑥𐑰 𐑩 𐑚𐑦𐑤𐑾𐑯 𐑘𐑽𐑟 𐑚𐑦𐑓𐑹 𐑞𐑦𐑕 𐑴𐑖𐑩𐑯 𐑮𐑴𐑤𐑛. 𐑓𐑵𐑤! 𐑲 𐑨𐑥 𐑞 ·𐑓𐑱𐑑𐑕' 𐑤𐑧𐑓𐑑𐑧𐑯𐑩𐑯𐑑; 𐑲 𐑨𐑒𐑑 𐑳𐑯𐑛𐑼 𐑹𐑛𐑼𐑟. 𐑤𐑫𐑒 𐑞𐑬, 𐑳𐑯𐑛𐑼𐑤𐑦𐑙! 𐑞𐑨𐑑 𐑞𐑬 𐑴𐑚𐑱𐑩𐑕𐑑 𐑥𐑲𐑯.—𐑕𐑑𐑨𐑯𐑛 𐑮𐑬𐑯𐑛 𐑥𐑰, 𐑥𐑧𐑯. 𐑘𐑰 𐑕𐑰 𐑩𐑯 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑒𐑳𐑑 𐑛𐑬𐑯 𐑑 𐑞 𐑕𐑑𐑳𐑥𐑐; 𐑤𐑰𐑯𐑦𐑙 𐑪𐑯 𐑩 𐑖𐑦𐑝𐑼𐑛 𐑤𐑭𐑯𐑕; 𐑐𐑮𐑪𐑐𐑑 𐑳𐑐 𐑪𐑯 𐑩 𐑤𐑴𐑯𐑤𐑦 𐑓𐑫𐑑. '𐑑𐑦𐑟 ·𐑱𐑣𐑨𐑚—𐑣𐑦𐑟 𐑚𐑪𐑛𐑦'𐑟 𐑐𐑸𐑑; 𐑚𐑳𐑑 ·𐑱𐑣𐑨𐑚'𐑟 𐑕𐑴𐑤'𐑟 𐑩 𐑕𐑧𐑯𐑑𐑦𐑐𐑰𐑛, 𐑞𐑨𐑑 𐑥𐑵𐑝𐑟 𐑩𐑐𐑪𐑯 𐑩 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑤𐑧𐑜𐑟. 𐑲 𐑓𐑰𐑤 𐑕𐑑𐑮𐑱𐑯𐑛, 𐑣𐑭𐑓 𐑕𐑑𐑮𐑨𐑯𐑛𐑩𐑛, 𐑨𐑟 𐑮𐑴𐑐𐑕 𐑞𐑨𐑑 𐑑𐑴 𐑛𐑦𐑕𐑥𐑭𐑕𐑑𐑩𐑛 𐑓𐑮𐑦𐑜𐑩𐑑𐑕 𐑦𐑯 𐑩 𐑜𐑱𐑤; 𐑯 𐑲 𐑥𐑱 𐑤𐑫𐑒 𐑕𐑴. 𐑚𐑳𐑑 𐑺 𐑲 𐑚𐑮𐑱𐑒, 𐑘𐑰'𐑤 𐑣𐑽 𐑥𐑰 𐑒𐑮𐑨𐑒; 𐑯 𐑑𐑦𐑤 𐑘𐑰 𐑣𐑽 𐑞𐑨𐑑, 𐑯𐑴 𐑞𐑨𐑑 ·𐑱𐑣𐑨𐑚'𐑟 𐑣𐑷𐑟𐑼 𐑑𐑴𐑟 𐑣𐑦𐑟 𐑐𐑻𐑐𐑩𐑕 𐑘𐑧𐑑. 𐑚𐑦𐑤𐑰𐑝 𐑘𐑰, 𐑥𐑧𐑯, 𐑦𐑯 𐑞 𐑔𐑦𐑙𐑟 𐑒𐑷𐑤𐑛 𐑴𐑥𐑩𐑯𐑟? 𐑞𐑧𐑯 𐑤𐑭𐑓 𐑩𐑤𐑬𐑛, 𐑯 𐑒𐑮𐑲 𐑪𐑙𐑒𐑹! 𐑓 𐑺 𐑞𐑱 𐑛𐑮𐑬𐑯, 𐑛𐑮𐑬𐑯𐑦𐑙 𐑔𐑦𐑙𐑟 𐑢𐑦𐑤 𐑑𐑢𐑲𐑕 𐑮𐑲𐑟 𐑑 𐑞 𐑕𐑻𐑓𐑦𐑕; 𐑞𐑧𐑯 𐑮𐑲𐑟 𐑩𐑜𐑱𐑯, 𐑑 𐑕𐑦𐑙𐑒 𐑓 𐑧𐑝𐑼𐑥𐑹. 𐑕𐑴 𐑢𐑦𐑞 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒—𐑑𐑵 𐑛𐑱𐑟 𐑣𐑰'𐑟 𐑓𐑤𐑴𐑑𐑩𐑛—𐑑𐑩𐑥𐑪𐑮𐑴 𐑢𐑦𐑤 𐑚𐑰 𐑞 𐑔𐑻𐑛. 𐑲, 𐑥𐑧𐑯, 𐑣𐑰'𐑤 𐑮𐑲𐑟 𐑢𐑳𐑯𐑕 𐑥𐑹,—𐑚𐑳𐑑 𐑴𐑯𐑤𐑦 𐑑 𐑕𐑐𐑬𐑑 𐑣𐑦𐑟 𐑤𐑭𐑕𐑑! ·𐑛'𐑘𐑰 𐑓𐑰𐑤 𐑚𐑮𐑱𐑝 𐑥𐑧𐑯, 𐑚𐑮𐑱𐑝?”

“𐑨𐑟 𐑓𐑽𐑤𐑩𐑕 𐑓𐑲𐑼,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚.

“𐑯 𐑨𐑟 𐑥𐑦𐑒𐑨𐑯𐑦𐑒𐑩𐑤,” 𐑥𐑳𐑑𐑼𐑛 ·𐑱𐑣𐑨𐑚. 𐑞𐑧𐑯 𐑨𐑟 𐑞 𐑥𐑧𐑯 𐑢𐑧𐑯𐑑 𐑓𐑹𐑢𐑼𐑛, 𐑣𐑰 𐑥𐑳𐑑𐑼𐑛 𐑪𐑯: “𐑞 𐑔𐑦𐑙𐑟 𐑒𐑷𐑤𐑛 𐑴𐑥𐑩𐑯𐑟! 𐑯 𐑘𐑧𐑕𐑑𐑼𐑛𐑱 𐑲 𐑑𐑷𐑒𐑑 𐑞 𐑕𐑱𐑥 𐑑 ·𐑕𐑑𐑸𐑚𐑳𐑒 𐑞𐑺, 𐑒𐑩𐑯𐑕𐑻𐑯𐑦𐑙 𐑥𐑲 𐑚𐑮𐑴𐑒𐑩𐑯 𐑚𐑴𐑑. 𐑴! 𐑣𐑬 𐑝𐑨𐑤𐑾𐑯𐑑𐑤𐑦 𐑲 𐑕𐑰𐑒 𐑑 𐑛𐑮𐑲𐑝 𐑬𐑑 𐑝 𐑳𐑞𐑼𐑟' 𐑣𐑸𐑑𐑕 𐑢𐑪𐑑'𐑕 𐑒𐑤𐑦𐑯𐑗𐑑 𐑕𐑴 𐑓𐑭𐑕𐑑 𐑦𐑯 𐑥𐑲𐑯!—𐑞 ·𐑐𐑸𐑟𐑰—𐑞 ·𐑐𐑸𐑟𐑰!—𐑜𐑪𐑯, 𐑜𐑪𐑯? 𐑯 𐑣𐑰 𐑢𐑪𐑟 𐑑 𐑜𐑴 𐑚𐑦𐑓𐑹:—𐑚𐑳𐑑 𐑕𐑑𐑦𐑤 𐑢𐑪𐑟 𐑑 𐑚𐑰 𐑕𐑰𐑯 𐑩𐑜𐑱𐑯 𐑺 𐑲 𐑒𐑫𐑛 𐑐𐑧𐑮𐑦𐑖—𐑣𐑬'𐑟 𐑞𐑨𐑑?—𐑞𐑺'𐑟 𐑩 𐑮𐑦𐑛𐑩𐑤 𐑯𐑬 𐑥𐑲𐑑 𐑚𐑨𐑓𐑩𐑤 𐑷𐑤 𐑞 𐑤𐑷𐑘𐑼𐑟 𐑚𐑨𐑒𐑑 𐑚𐑲 𐑞 𐑜𐑴𐑕𐑑𐑕 𐑝 𐑞 𐑣𐑴𐑤 𐑤𐑲𐑯 𐑝 𐑡𐑳𐑡𐑩𐑟:—𐑤𐑲𐑒 𐑩 𐑣𐑷𐑒'𐑕 𐑚𐑰𐑒 𐑦𐑑 𐑐𐑧𐑒𐑕 𐑥𐑲 𐑚𐑮𐑱𐑯. 𐑲'𐑤, 𐑲'𐑤 𐑕𐑪𐑤𐑝 𐑦𐑑, 𐑞𐑴!”

𐑢𐑧𐑯 𐑛𐑳𐑕𐑒 𐑛𐑦𐑕𐑧𐑯𐑛𐑩𐑛, 𐑞 𐑢𐑱𐑤 𐑢𐑪𐑟 𐑕𐑑𐑦𐑤 𐑦𐑯 𐑕𐑲𐑑 𐑑 𐑤𐑰𐑢𐑼𐑛.

𐑕𐑴 𐑢𐑳𐑯𐑕 𐑥𐑹 𐑞 𐑕𐑱𐑤 𐑢𐑪𐑟 𐑖𐑹𐑑𐑩𐑯𐑛, 𐑯 𐑧𐑝𐑮𐑦𐑔𐑦𐑙 𐑐𐑭𐑕𐑑 𐑯𐑽𐑤𐑦 𐑨𐑟 𐑪𐑯 𐑞 𐑐𐑮𐑰𐑝𐑾𐑕 𐑯𐑲𐑑; 𐑴𐑯𐑤𐑦, 𐑞 𐑕𐑬𐑯𐑛 𐑝 𐑣𐑨𐑥𐑼𐑟, 𐑯 𐑞 𐑣𐑳𐑥 𐑝 𐑞 𐑜𐑮𐑲𐑯𐑛𐑕𐑑𐑴𐑯 𐑢𐑪𐑟 𐑣𐑻𐑛 𐑑𐑦𐑤 𐑯𐑽𐑤𐑦 𐑛𐑱𐑤𐑲𐑑, 𐑨𐑟 𐑞 𐑥𐑧𐑯 𐑑𐑶𐑤𐑛 𐑚𐑲 𐑤𐑨𐑯𐑑𐑼𐑯𐑟 𐑦𐑯 𐑞 𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑯 𐑒𐑺𐑓𐑩𐑤 𐑮𐑦𐑜𐑦𐑙 𐑝 𐑞 𐑕𐑐𐑺 𐑚𐑴𐑑𐑕 𐑯 𐑖𐑸𐑐𐑩𐑯𐑦𐑙 𐑞𐑺 𐑓𐑮𐑧𐑖 𐑢𐑧𐑐𐑩𐑯𐑟 𐑓 𐑞 𐑥𐑪𐑮𐑴. 𐑥𐑰𐑯𐑑𐑲𐑥, 𐑝 𐑞 𐑚𐑮𐑴𐑒𐑩𐑯 𐑒𐑰𐑤 𐑝 ·𐑱𐑣𐑨𐑚'𐑟 𐑮𐑧𐑒𐑑 𐑒𐑮𐑭𐑓𐑑 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑥𐑱𐑛 𐑣𐑦𐑥 𐑩𐑯𐑳𐑞𐑼 𐑤𐑧𐑜; 𐑢𐑲𐑤 𐑕𐑑𐑦𐑤 𐑨𐑟 𐑪𐑯 𐑞 𐑯𐑲𐑑 𐑚𐑦𐑓𐑹, 𐑕𐑤𐑬𐑗𐑑 ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑫𐑛 𐑓𐑦𐑒𐑕𐑑 𐑢𐑦𐑞𐑦𐑯 𐑣𐑦𐑟 𐑕𐑒𐑳𐑑𐑩𐑤; 𐑣𐑦𐑟 𐑣𐑦𐑛, 𐑣𐑰𐑤𐑾𐑑𐑮𐑴𐑐 𐑜𐑤𐑭𐑯𐑕 𐑨𐑯𐑑𐑦𐑕𐑦𐑐𐑱𐑑𐑦𐑙𐑤𐑦 𐑜𐑪𐑯 𐑚𐑨𐑒𐑢𐑼𐑛 𐑪𐑯 𐑦𐑑𐑕 𐑛𐑲𐑩𐑤; 𐑕𐑨𐑑 𐑛𐑿 𐑰𐑕𐑑𐑢𐑼𐑛 𐑓 𐑞 𐑻𐑤𐑾𐑕𐑑 𐑕𐑳𐑯.
