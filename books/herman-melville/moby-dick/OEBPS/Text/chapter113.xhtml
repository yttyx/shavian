·𐑗𐑨𐑐𐑑𐑼 113. 𐑞 ·𐑓𐑹𐑡.

𐑢𐑦𐑞 𐑥𐑨𐑑𐑩𐑛 𐑚𐑽𐑛, 𐑯 𐑕𐑢𐑱𐑞𐑛 𐑦𐑯 𐑩 𐑚𐑮𐑦𐑕𐑩𐑤𐑦𐑙 𐑖𐑸𐑒-𐑕𐑒𐑦𐑯 𐑱𐑐𐑮𐑩𐑯, 𐑩𐑚𐑬𐑑 𐑥𐑦𐑛-𐑛𐑱, ·𐑐𐑻𐑔 𐑢𐑪𐑟 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑦𐑑𐑢𐑰𐑯 𐑣𐑦𐑟 𐑓𐑹𐑡 𐑯 𐑨𐑯𐑝𐑦𐑤, 𐑞 𐑤𐑨𐑑𐑼 𐑐𐑤𐑱𐑕𐑑 𐑩𐑐𐑪𐑯 𐑩𐑯 𐑲𐑼𐑯-𐑢𐑫𐑛 𐑤𐑪𐑜, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑣𐑴𐑤𐑛𐑦𐑙 𐑩 𐑐𐑲𐑒-𐑣𐑧𐑛 𐑦𐑯 𐑞 𐑒𐑴𐑤𐑟, 𐑯 𐑢𐑦𐑞 𐑞 𐑳𐑞𐑼 𐑨𐑑 𐑣𐑦𐑟 𐑓𐑹𐑡'𐑩𐑟 𐑤𐑳𐑙𐑟, 𐑢𐑧𐑯 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚 𐑒𐑱𐑥 𐑩𐑤𐑪𐑙, 𐑒𐑨𐑮𐑦𐑦𐑙 𐑦𐑯 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑩 𐑕𐑥𐑷𐑤 𐑮𐑳𐑕𐑑𐑦-𐑤𐑫𐑒𐑦𐑙 𐑤𐑧𐑞𐑼𐑯 𐑚𐑨𐑜. 𐑢𐑲𐑤 𐑘𐑧𐑑 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑹𐑡, 𐑥𐑵𐑛𐑦 ·𐑱𐑣𐑨𐑚 𐑐𐑷𐑟𐑛; 𐑑𐑦𐑤 𐑨𐑑 𐑤𐑭𐑕𐑑, ·𐑐𐑻𐑔, 𐑢𐑦𐑞𐑛𐑮𐑷𐑦𐑙 𐑣𐑦𐑟 𐑲𐑼𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑲𐑼, 𐑚𐑦𐑜𐑨𐑯 𐑣𐑨𐑥𐑼𐑦𐑙 𐑦𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑨𐑯𐑝𐑦𐑤—𐑞 𐑮𐑧𐑛 𐑥𐑨𐑕 𐑕𐑧𐑯𐑛𐑦𐑙 𐑪𐑓 𐑞 𐑕𐑐𐑸𐑒𐑕 𐑦𐑯 𐑔𐑦𐑒 𐑣𐑪𐑝𐑼𐑦𐑙 𐑓𐑤𐑲𐑑𐑕, 𐑕𐑳𐑥 𐑝 𐑢𐑦𐑗 𐑓𐑤𐑵 𐑒𐑤𐑴𐑕 𐑑 ·𐑱𐑣𐑨𐑚.

“𐑸 𐑞𐑰𐑟 𐑞𐑲 ·𐑥𐑳𐑞𐑼 ·𐑒𐑺𐑦'𐑟 𐑗𐑦𐑒𐑦𐑯𐑟, ·𐑐𐑻𐑔? 𐑞𐑱 𐑸 𐑷𐑤𐑢𐑱𐑟 𐑓𐑤𐑲𐑦𐑙 𐑦𐑯 𐑞𐑲 𐑢𐑱𐑒; 𐑚𐑻𐑛𐑟 𐑝 𐑜𐑫𐑛 𐑴𐑥𐑩𐑯, 𐑑𐑵, 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑑 𐑷𐑤;—𐑤𐑫𐑒 𐑣𐑽, 𐑞𐑱 𐑚𐑻𐑯; 𐑚𐑳𐑑 𐑞𐑬—𐑞𐑬 𐑤𐑦𐑝'𐑕𐑑 𐑩𐑥𐑳𐑙 𐑞𐑧𐑥 𐑢𐑦𐑞𐑬𐑑 𐑩 𐑕𐑒𐑹𐑗.”

“𐑚𐑦𐑒𐑪𐑟 𐑲 𐑨𐑥 𐑕𐑒𐑹𐑗𐑑 𐑷𐑤 𐑴𐑝𐑼, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚,” 𐑭𐑯𐑕𐑼𐑛 ·𐑐𐑻𐑔, 𐑮𐑧𐑕𐑑𐑦𐑙 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑪𐑯 𐑣𐑦𐑟 𐑣𐑨𐑥𐑼; “𐑲 𐑨𐑥 𐑐𐑭𐑕𐑑 𐑕𐑒𐑹𐑗𐑦𐑙; 𐑯𐑪𐑑 𐑰𐑟𐑦𐑤𐑦 𐑒𐑨𐑯'𐑕𐑑 𐑞𐑬 𐑕𐑒𐑹𐑗 𐑩 𐑕𐑒𐑸.”

“𐑢𐑧𐑤, 𐑢𐑧𐑤; 𐑯𐑴 𐑥𐑹. 𐑞𐑲 𐑖𐑮𐑳𐑙𐑒 𐑝𐑶𐑕 𐑕𐑬𐑯𐑛𐑟 𐑑𐑵 𐑒𐑭𐑥𐑤𐑦, 𐑕𐑱𐑯𐑤𐑦 𐑢𐑴𐑓𐑩𐑤 𐑑 𐑥𐑰. 𐑦𐑯 𐑯𐑴 ·𐑐𐑨𐑮𐑩𐑛𐑲𐑕 𐑥𐑲𐑕𐑧𐑤𐑓, 𐑲 𐑨𐑥 𐑦𐑥𐑐𐑱𐑖𐑩𐑯𐑑 𐑝 𐑷𐑤 𐑥𐑦𐑟𐑼𐑦 𐑦𐑯 𐑳𐑞𐑼𐑟 𐑞𐑨𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑥𐑨𐑛. 𐑞𐑬 𐑖𐑵𐑛'𐑕𐑑 𐑜𐑴 𐑥𐑨𐑛, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔; 𐑕𐑱, 𐑢𐑲 𐑛𐑳𐑕𐑑 𐑞𐑬 𐑯𐑪𐑑 𐑜𐑴 𐑥𐑨𐑛? 𐑣𐑬 𐑒𐑨𐑯'𐑕𐑑 𐑞𐑬 𐑦𐑯𐑛𐑻 𐑢𐑦𐑞𐑬𐑑 𐑚𐑰𐑦𐑙 𐑥𐑨𐑛? 𐑛𐑵 𐑞 𐑣𐑧𐑝𐑩𐑯𐑟 𐑘𐑧𐑑 𐑣𐑱𐑑 𐑞𐑰, 𐑞𐑨𐑑 𐑞𐑬 𐑒𐑨𐑯'𐑕𐑑 𐑯𐑪𐑑 𐑜𐑴 𐑥𐑨𐑛?—𐑢𐑪𐑑 ·𐑝𐑧𐑮𐑑 𐑞𐑬 𐑥𐑱𐑒𐑦𐑙 𐑞𐑺?”

“𐑢𐑧𐑤𐑛𐑦𐑙 𐑩𐑯 𐑴𐑤𐑛 𐑐𐑲𐑒-𐑣𐑧𐑛, 𐑕𐑻; 𐑞𐑺 𐑢𐑻 𐑕𐑰𐑥𐑟 𐑯 𐑛𐑧𐑯𐑑𐑕 𐑦𐑯 𐑦𐑑.”

“𐑯 𐑒𐑨𐑯'𐑕𐑑 𐑞𐑬 𐑥𐑱𐑒 𐑦𐑑 𐑷𐑤 𐑕𐑥𐑵𐑞 𐑩𐑜𐑱𐑯, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔, 𐑭𐑓𐑑𐑼 𐑕𐑳𐑗 𐑣𐑸𐑛 𐑿𐑕𐑦𐑡 𐑨𐑟 𐑦𐑑 𐑣𐑨𐑛?”

“𐑲 𐑔𐑦𐑙𐑒 𐑕𐑴, 𐑕𐑻.”

“𐑯 𐑲 𐑕𐑩𐑐𐑴𐑟 𐑞𐑬 𐑒𐑨𐑯'𐑕𐑑 𐑕𐑥𐑵𐑞 𐑷𐑤𐑥𐑴𐑕𐑑 𐑧𐑯𐑦 𐑕𐑰𐑥𐑟 𐑯 𐑛𐑧𐑯𐑑𐑕; 𐑯𐑧𐑝𐑼 𐑥𐑲𐑯𐑛 𐑣𐑬 𐑣𐑸𐑛 𐑞 𐑥𐑧𐑑𐑩𐑤, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔?”

“𐑲, 𐑕𐑻, 𐑲 𐑔𐑦𐑙𐑒 𐑲 𐑒𐑨𐑯; 𐑷𐑤 𐑕𐑰𐑥𐑟 𐑯 𐑛𐑧𐑯𐑑𐑕 𐑚𐑳𐑑 𐑢𐑳𐑯.”

“𐑤𐑫𐑒 𐑘𐑰 𐑣𐑽, 𐑞𐑧𐑯,” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑐𐑨𐑖𐑩𐑯𐑩𐑑𐑤𐑦 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙, 𐑯 𐑤𐑰𐑯𐑦𐑙 𐑢𐑦𐑞 𐑚𐑴𐑔 𐑣𐑨𐑯𐑛𐑟 𐑪𐑯 ·𐑐𐑻𐑔'𐑕 𐑖𐑴𐑤𐑛𐑼𐑟; “𐑤𐑫𐑒 𐑘𐑰 𐑣𐑽—𐑣𐑽—𐑒𐑨𐑯 𐑘𐑰 𐑕𐑥𐑵𐑞 𐑬𐑑 𐑩 𐑕𐑰𐑥 𐑤𐑲𐑒 𐑞𐑦𐑕, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔,” 𐑕𐑢𐑰𐑐𐑦𐑙 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑩𐑒𐑮𐑪𐑕 𐑣𐑦𐑟 𐑮𐑦𐑚𐑛 𐑚𐑮𐑬; “𐑦𐑓 𐑞𐑬 𐑒𐑫𐑛'𐑕𐑑, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔, 𐑜𐑤𐑨𐑛 𐑦𐑯𐑳𐑓 𐑢𐑫𐑛 𐑲 𐑤𐑱 𐑥𐑲 𐑣𐑧𐑛 𐑩𐑐𐑪𐑯 𐑞𐑲 𐑨𐑯𐑝𐑦𐑤, 𐑯 𐑓𐑰𐑤 𐑞𐑲 𐑣𐑧𐑝𐑾𐑕𐑑 𐑣𐑨𐑥𐑼 𐑚𐑦𐑑𐑢𐑰𐑯 𐑥𐑲 𐑲𐑟. 𐑭𐑯𐑕𐑼! 𐑒𐑨𐑯'𐑕𐑑 𐑞𐑬 𐑕𐑥𐑵𐑞 𐑞𐑦𐑕 𐑕𐑰𐑥?”

“𐑴! 𐑞𐑨𐑑 𐑦𐑟 𐑞 𐑢𐑳𐑯, 𐑕𐑻! 𐑕𐑧𐑛 𐑲 𐑯𐑪𐑑 𐑷𐑤 𐑕𐑰𐑥𐑟 𐑯 𐑛𐑧𐑯𐑑𐑕 𐑚𐑳𐑑 𐑢𐑳𐑯?”

“𐑲, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔, 𐑦𐑑 𐑦𐑟 𐑞 𐑢𐑳𐑯; 𐑲, 𐑥𐑨𐑯, 𐑦𐑑 𐑦𐑟 𐑳𐑯𐑕𐑥𐑵𐑞𐑩𐑚𐑩𐑤; 𐑓 𐑞𐑴 𐑞𐑬 𐑴𐑯𐑤𐑦 𐑕𐑰'𐑕𐑑 𐑦𐑑 𐑣𐑽 𐑦𐑯 𐑥𐑲 𐑓𐑤𐑧𐑖, 𐑦𐑑 𐑣𐑨𐑟 𐑢𐑻𐑒𐑑 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑚𐑴𐑯 𐑝 𐑥𐑲 𐑕𐑒𐑳𐑤—𐑞𐑨𐑑 𐑦𐑟 𐑷𐑤 𐑮𐑦𐑙𐑒𐑩𐑤𐑟! 𐑚𐑳𐑑, 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑗𐑲𐑤𐑛'𐑟 𐑐𐑤𐑱; 𐑯𐑴 𐑥𐑹 𐑜𐑨𐑓𐑕 𐑯 𐑐𐑲𐑒𐑕 𐑑-𐑛𐑱. 𐑤𐑫𐑒 𐑘𐑰 𐑣𐑽!” 𐑡𐑦𐑙𐑜𐑩𐑤𐑦𐑙 𐑞 𐑤𐑧𐑞𐑼𐑯 𐑚𐑨𐑜, 𐑨𐑟 𐑦𐑓 𐑦𐑑 𐑢𐑻 𐑓𐑫𐑤 𐑝 𐑜𐑴𐑤𐑛 𐑒𐑶𐑯𐑟. “𐑲, 𐑑𐑵, 𐑢𐑪𐑯𐑑 𐑩 𐑣𐑸𐑐𐑵𐑯 𐑥𐑱𐑛; 𐑢𐑳𐑯 𐑞𐑨𐑑 𐑩 𐑔𐑬𐑟𐑩𐑯𐑛 𐑘𐑴𐑒 𐑝 𐑓𐑰𐑯𐑛𐑟 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑐𐑸𐑑, ·𐑐𐑻𐑔; 𐑕𐑳𐑥𐑔𐑦𐑙 𐑞𐑨𐑑 𐑢𐑦𐑤 𐑕𐑑𐑦𐑒 𐑦𐑯 𐑩 𐑢𐑱𐑤 𐑤𐑲𐑒 𐑣𐑦𐑟 𐑴𐑯 𐑓𐑦𐑯-𐑚𐑴𐑯. 𐑞𐑺'𐑟 𐑞 𐑕𐑑𐑳𐑓,” 𐑓𐑤𐑦𐑙𐑦𐑙 𐑞 𐑐𐑬𐑗 𐑩𐑐𐑪𐑯 𐑞 𐑨𐑯𐑝𐑦𐑤. “𐑤𐑫𐑒 𐑘𐑰, 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔, 𐑞𐑰𐑟 𐑸 𐑞 𐑜𐑨𐑞𐑼𐑛 𐑯𐑱𐑤-𐑕𐑑𐑳𐑚𐑟 𐑝 𐑞 𐑕𐑑𐑰𐑤 𐑖𐑵𐑟 𐑝 𐑮𐑱𐑕𐑦𐑙 𐑣𐑹𐑕𐑩𐑟.”

“𐑣𐑹𐑕-𐑖𐑵 𐑕𐑑𐑳𐑚𐑟, 𐑕𐑻? 𐑢𐑲, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚, 𐑞𐑬 𐑣𐑨𐑕𐑑 𐑣𐑽, 𐑞𐑧𐑯, 𐑞 𐑚𐑧𐑕𐑑 𐑯 𐑕𐑑𐑳𐑚𐑼𐑯𐑩𐑕𐑑 𐑕𐑑𐑳𐑓 𐑢𐑰 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔𐑕 𐑧𐑝𐑼 𐑢𐑻𐑒.”

“𐑲 𐑯𐑴 𐑦𐑑, 𐑴𐑤𐑛 𐑥𐑨𐑯; 𐑞𐑰𐑟 𐑕𐑑𐑳𐑚𐑟 𐑢𐑦𐑤 𐑢𐑧𐑤𐑛 𐑑𐑩𐑜𐑧𐑞𐑼 𐑤𐑲𐑒 𐑜𐑤𐑵 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑧𐑤𐑑𐑩𐑛 𐑚𐑴𐑯𐑟 𐑝 𐑥𐑻𐑛𐑼𐑼𐑟. 𐑒𐑢𐑦𐑒! 𐑓𐑹𐑡 𐑥𐑰 𐑞 𐑣𐑸𐑐𐑵𐑯. 𐑯 𐑓𐑹𐑡 𐑥𐑰 𐑓𐑻𐑕𐑑, 𐑑𐑢𐑧𐑤𐑝 𐑮𐑪𐑛𐑟 𐑓 𐑦𐑑𐑕 𐑖𐑨𐑙𐑒; 𐑞𐑧𐑯 𐑢𐑦𐑯𐑛, 𐑯 𐑑𐑢𐑦𐑕𐑑, 𐑯 𐑣𐑨𐑥𐑼 𐑞𐑰𐑟 𐑑𐑢𐑧𐑤𐑝 𐑑𐑩𐑜𐑧𐑞𐑼 𐑤𐑲𐑒 𐑞 𐑘𐑸𐑯𐑟 𐑯 𐑕𐑑𐑮𐑨𐑯𐑛𐑟 𐑝 𐑩 𐑑𐑴-𐑤𐑲𐑯. 𐑒𐑢𐑦𐑒! 𐑲'𐑤 𐑚𐑤𐑴 𐑞 𐑓𐑲𐑼.”

𐑢𐑧𐑯 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑞 𐑑𐑢𐑧𐑤𐑝 𐑮𐑪𐑛𐑟 𐑢𐑻 𐑥𐑱𐑛, ·𐑱𐑣𐑨𐑚 𐑑𐑮𐑲𐑛 𐑞𐑧𐑥, 𐑢𐑳𐑯 𐑚𐑲 𐑢𐑳𐑯, 𐑚𐑲 𐑕𐑐𐑲𐑮𐑩𐑤𐑦𐑙 𐑞𐑧𐑥, 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑴𐑯 𐑣𐑨𐑯𐑛, 𐑮𐑬𐑯𐑛 𐑩 𐑤𐑪𐑙, 𐑣𐑧𐑝𐑦 𐑲𐑼𐑯 𐑚𐑴𐑤𐑑. “𐑩 𐑓𐑤𐑷!” 𐑮𐑦𐑡𐑧𐑒𐑑𐑦𐑙 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑳𐑯. “𐑢𐑻𐑒 𐑞𐑨𐑑 𐑴𐑝𐑼 𐑩𐑜𐑱𐑯, ·𐑐𐑻𐑔.”

𐑞𐑦𐑕 𐑛𐑳𐑯, ·𐑐𐑻𐑔 𐑢𐑪𐑟 𐑩𐑚𐑬𐑑 𐑑 𐑚𐑦𐑜𐑦𐑯 𐑢𐑧𐑤𐑛𐑦𐑙 𐑞 𐑑𐑢𐑧𐑤𐑝 𐑦𐑯𐑑𐑵 𐑢𐑳𐑯, 𐑢𐑧𐑯 ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑱𐑛 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛, 𐑯 𐑕𐑧𐑛 𐑣𐑰 𐑢𐑫𐑛 𐑢𐑧𐑤𐑛 𐑣𐑦𐑟 𐑴𐑯 𐑲𐑼𐑯. 𐑨𐑟, 𐑞𐑧𐑯, 𐑢𐑦𐑞 𐑮𐑧𐑜𐑿𐑤𐑼, 𐑜𐑭𐑕𐑐𐑦𐑙 𐑣𐑧𐑥𐑟, 𐑣𐑰 𐑣𐑨𐑥𐑼𐑛 𐑪𐑯 𐑞 𐑨𐑯𐑝𐑦𐑤, ·𐑐𐑻𐑔 𐑐𐑭𐑕𐑦𐑙 𐑑 𐑣𐑦𐑥 𐑞 𐑜𐑤𐑴𐑦𐑙 𐑮𐑪𐑛𐑟, 𐑢𐑳𐑯 𐑭𐑓𐑑𐑼 𐑞 𐑳𐑞𐑼, 𐑯 𐑞 𐑣𐑸𐑛 𐑐𐑮𐑧𐑕𐑑 𐑓𐑹𐑡 𐑖𐑵𐑑𐑦𐑙 𐑳𐑐 𐑦𐑑𐑕 𐑦𐑯𐑑𐑧𐑯𐑕 𐑕𐑑𐑮𐑱𐑑 𐑓𐑤𐑱𐑥, 𐑞 ·𐑐𐑸𐑟𐑰 𐑐𐑭𐑕𐑑 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦, 𐑯 𐑚𐑬𐑦𐑙 𐑴𐑝𐑼 𐑣𐑦𐑟 𐑣𐑧𐑛 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 𐑓𐑲𐑼, 𐑕𐑰𐑥𐑛 𐑦𐑯𐑝𐑴𐑒𐑦𐑙 𐑕𐑳𐑥 𐑒𐑻𐑕 𐑹 𐑕𐑳𐑥 𐑚𐑤𐑧𐑕𐑦𐑙 𐑪𐑯 𐑞 𐑑𐑶𐑤. 𐑚𐑳𐑑, 𐑨𐑟 ·𐑱𐑣𐑨𐑚 𐑤𐑫𐑒𐑑 𐑳𐑐, 𐑣𐑰 𐑕𐑤𐑦𐑛 𐑩𐑕𐑲𐑛.

“𐑢𐑪𐑑'𐑕 𐑞𐑨𐑑 𐑚𐑳𐑯𐑗 𐑝 𐑤𐑵𐑕𐑦𐑓𐑼𐑟 𐑛𐑪𐑡𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞𐑺 𐑓?” 𐑥𐑳𐑑𐑼𐑛 ·𐑕𐑑𐑳𐑚, 𐑤𐑫𐑒𐑦𐑙 𐑪𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑓𐑴𐑒𐑕𐑩𐑤. “𐑞𐑨𐑑 ·𐑐𐑸𐑟𐑰 𐑕𐑥𐑧𐑤𐑟 𐑓𐑲𐑼 𐑤𐑲𐑒 𐑩 𐑓𐑿𐑟𐑰; 𐑯 𐑕𐑥𐑧𐑤𐑟 𐑝 𐑦𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑤𐑲𐑒 𐑩 𐑣𐑪𐑑 𐑥𐑳𐑕𐑒𐑩𐑑'𐑕 𐑐𐑬𐑛𐑼-𐑐𐑨𐑯.”

𐑨𐑑 𐑤𐑭𐑕𐑑 𐑞 𐑖𐑨𐑙𐑒, 𐑦𐑯 𐑢𐑳𐑯 𐑒𐑩𐑥𐑐𐑤𐑰𐑑 𐑮𐑪𐑛, 𐑮𐑦𐑕𐑰𐑝𐑛 𐑦𐑑𐑕 𐑓𐑲𐑯𐑩𐑤 𐑣𐑰𐑑; 𐑯 𐑨𐑟 ·𐑐𐑻𐑔, 𐑑 𐑑𐑧𐑥𐑐𐑼 𐑦𐑑, 𐑐𐑤𐑳𐑯𐑡𐑛 𐑦𐑑 𐑷𐑤 𐑣𐑦𐑕𐑦𐑙 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑭𐑕𐑒 𐑝 𐑢𐑷𐑑𐑼 𐑯𐑽 𐑚𐑲, 𐑞 𐑕𐑒𐑷𐑤𐑛𐑦𐑙 𐑕𐑑𐑰𐑥 𐑖𐑪𐑑 𐑳𐑐 𐑦𐑯𐑑𐑵 ·𐑱𐑣𐑨𐑚'𐑟 𐑚𐑧𐑯𐑑 𐑓𐑱𐑕.

“𐑢𐑫𐑛'𐑕𐑑 𐑞𐑬 𐑚𐑮𐑨𐑯𐑛 𐑥𐑰, ·𐑐𐑻𐑔?” 𐑢𐑦𐑯𐑕𐑦𐑙 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑢𐑦𐑞 𐑞 𐑐𐑱𐑯; “𐑣𐑨𐑝 𐑲 𐑚𐑰𐑯 𐑚𐑳𐑑 𐑓𐑹𐑡𐑦𐑙 𐑥𐑲 𐑴𐑯 𐑚𐑮𐑨𐑯𐑛𐑦𐑙-𐑲𐑼𐑯, 𐑞𐑧𐑯?”

“𐑐𐑮𐑱 ·𐑜𐑪𐑛, 𐑯𐑪𐑑 𐑞𐑨𐑑; 𐑘𐑧𐑑 𐑲 𐑓𐑽 𐑕𐑳𐑥𐑔𐑦𐑙, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚. 𐑦𐑟 𐑯𐑪𐑑 𐑞𐑦𐑕 𐑣𐑸𐑐𐑵𐑯 𐑓 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?”

“𐑓 𐑞 𐑢𐑲𐑑 𐑓𐑰𐑯𐑛! 𐑚𐑳𐑑 𐑯𐑬 𐑓 𐑞 𐑚𐑸𐑚𐑟; 𐑞𐑬 𐑥𐑳𐑕𐑑 𐑥𐑱𐑒 𐑞𐑧𐑥 𐑞𐑲𐑕𐑧𐑤𐑓, 𐑥𐑨𐑯. 𐑣𐑽 𐑸 𐑥𐑲 𐑮𐑱𐑟𐑼𐑟—𐑞 𐑚𐑧𐑕𐑑 𐑝 𐑕𐑑𐑰𐑤; 𐑣𐑽, 𐑯 𐑥𐑱𐑒 𐑞 𐑚𐑸𐑚𐑟 𐑖𐑸𐑐 𐑨𐑟 𐑞 𐑯𐑰𐑛𐑩𐑤-𐑕𐑤𐑰𐑑 𐑝 𐑞 ·𐑲𐑕𐑦 ·𐑕𐑰.”

𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑞 𐑴𐑤𐑛 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔 𐑲𐑛 𐑞 𐑮𐑱𐑟𐑼𐑟 𐑨𐑟 𐑞𐑴 𐑣𐑰 𐑢𐑫𐑛 𐑓𐑱𐑯 𐑯𐑪𐑑 𐑿𐑟 𐑞𐑧𐑥.

“𐑑𐑱𐑒 𐑞𐑧𐑥, 𐑥𐑨𐑯, 𐑲 𐑣𐑨𐑝 𐑯𐑴 𐑯𐑰𐑛 𐑓 𐑞𐑧𐑥; 𐑓 𐑲 𐑯𐑬 𐑯𐑲𐑞𐑼 𐑖𐑱𐑝, 𐑕𐑳𐑐, 𐑯𐑹 𐑐𐑮𐑱 𐑑𐑦𐑤—𐑚𐑳𐑑 𐑣𐑽—𐑑 𐑢𐑻𐑒!”

𐑓𐑨𐑖𐑩𐑯𐑛 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑦𐑯𐑑𐑵 𐑩𐑯 𐑨𐑮𐑴𐑦 𐑖𐑱𐑐, 𐑯 𐑢𐑧𐑤𐑛𐑩𐑛 𐑚𐑲 ·𐑐𐑻𐑔 𐑑 𐑞 𐑖𐑨𐑙𐑒, 𐑞 𐑕𐑑𐑰𐑤 𐑕𐑵𐑯 𐑐𐑶𐑯𐑑𐑩𐑛 𐑞 𐑧𐑯𐑛 𐑝 𐑞 𐑲𐑼𐑯; 𐑯 𐑨𐑟 𐑞 𐑚𐑤𐑨𐑒𐑕𐑥𐑦𐑔 𐑢𐑪𐑟 𐑩𐑚𐑬𐑑 𐑜𐑦𐑝𐑦𐑙 𐑞 𐑚𐑸𐑚𐑟 𐑞𐑺 𐑓𐑲𐑯𐑩𐑤 𐑣𐑰𐑑, 𐑐𐑮𐑲𐑼 𐑑 𐑑𐑧𐑥𐑐𐑼𐑦𐑙 𐑞𐑧𐑥, 𐑣𐑰 𐑒𐑮𐑲𐑛 𐑑 ·𐑱𐑣𐑨𐑚 𐑑 𐑐𐑤𐑱𐑕 𐑞 𐑢𐑷𐑑𐑼-𐑒𐑭𐑕𐑒 𐑯𐑽.

“𐑯𐑴, 𐑯𐑴—𐑯𐑴 𐑢𐑷𐑑𐑼 𐑓 𐑞𐑨𐑑; 𐑲 𐑢𐑪𐑯𐑑 𐑦𐑑 𐑝 𐑞 𐑑𐑮𐑵 𐑛𐑧𐑔-𐑑𐑧𐑥𐑐𐑼. 𐑩𐑣𐑶, 𐑞𐑺! ·𐑑𐑨𐑖𐑑𐑱𐑜𐑴, ·𐑒𐑢𐑰𐑒𐑢𐑧𐑜, ·𐑛𐑨𐑜𐑵! 𐑢𐑪𐑑 𐑕𐑱 𐑘𐑰, 𐑐𐑱𐑜𐑩𐑯𐑟! 𐑢𐑦𐑤 𐑘𐑰 𐑜𐑦𐑝 𐑥𐑰 𐑨𐑟 𐑥𐑳𐑗 𐑚𐑤𐑳𐑛 𐑨𐑟 𐑢𐑦𐑤 𐑒𐑳𐑝𐑼 𐑞𐑦𐑕 𐑚𐑸𐑚?” 𐑣𐑴𐑤𐑛𐑦𐑙 𐑦𐑑 𐑣𐑲 𐑳𐑐. 𐑩 𐑒𐑤𐑳𐑕𐑑𐑼 𐑝 𐑛𐑸𐑒 𐑯𐑪𐑛𐑟 𐑮𐑦𐑐𐑤𐑲𐑛, ·𐑘𐑧𐑕. 𐑔𐑮𐑰 𐑐𐑳𐑙𐑒𐑗𐑼𐑟 𐑢𐑻 𐑥𐑱𐑛 𐑦𐑯 𐑞 𐑣𐑰𐑞𐑩𐑯 𐑓𐑤𐑧𐑖, 𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑚𐑸𐑚𐑟 𐑢𐑻 𐑞𐑧𐑯 𐑑𐑧𐑥𐑐𐑼𐑛.

“𐑰𐑜𐑴 𐑯𐑪𐑯 𐑚𐑨𐑐𐑑𐑲𐑟𐑴 𐑑𐑱 𐑦𐑯 𐑯𐑴𐑥𐑲𐑯 𐑐𐑨𐑑𐑮𐑦𐑕, 𐑕𐑦𐑛 𐑦𐑯 𐑯𐑴𐑥𐑲𐑯 𐑛𐑲𐑩𐑚𐑪𐑤𐑦!” 𐑛𐑦𐑤𐑽𐑾𐑕𐑤𐑦 𐑣𐑬𐑤𐑛 ·𐑱𐑣𐑨𐑚, 𐑨𐑟 𐑞 𐑥𐑩𐑤𐑦𐑜𐑯𐑩𐑯𐑑 𐑲𐑼𐑯 𐑕𐑒𐑹𐑗𐑦𐑙𐑤𐑦 𐑛𐑦𐑝𐑬𐑼𐑛 𐑞 𐑚𐑨𐑐𐑑𐑦𐑟𐑥𐑩𐑤 𐑚𐑤𐑳𐑛.

𐑯𐑬, 𐑥𐑳𐑕𐑑𐑼𐑦𐑙 𐑞 𐑕𐑐𐑺 𐑐𐑴𐑤𐑟 𐑓𐑮𐑪𐑥 𐑚𐑦𐑤𐑴, 𐑯 𐑕𐑦𐑤𐑧𐑒𐑑𐑦𐑙 𐑢𐑳𐑯 𐑝 𐑣𐑦𐑒𐑼𐑦, 𐑢𐑦𐑞 𐑞 𐑚𐑸𐑒 𐑕𐑑𐑦𐑤 𐑦𐑯𐑝𐑧𐑕𐑑𐑦𐑙 𐑦𐑑, ·𐑱𐑣𐑨𐑚 𐑓𐑦𐑑𐑩𐑛 𐑞 𐑧𐑯𐑛 𐑑 𐑞 𐑕𐑪𐑒𐑩𐑑 𐑝 𐑞 𐑲𐑼𐑯. 𐑩 𐑒𐑶𐑤 𐑝 𐑯𐑿 𐑑𐑴-𐑤𐑲𐑯 𐑢𐑪𐑟 𐑞𐑧𐑯 𐑳𐑯𐑢𐑬𐑯𐑛, 𐑯 𐑕𐑳𐑥 𐑓𐑨𐑞𐑩𐑥𐑟 𐑝 𐑦𐑑 𐑑𐑱𐑒𐑩𐑯 𐑑 𐑞 𐑢𐑦𐑯𐑛𐑤𐑩𐑕, 𐑯 𐑕𐑑𐑮𐑧𐑗𐑑 𐑑 𐑩 𐑜𐑮𐑱𐑑 𐑑𐑧𐑯𐑖𐑩𐑯. 𐑐𐑮𐑧𐑕𐑦𐑙 𐑣𐑦𐑟 𐑓𐑫𐑑 𐑩𐑐𐑪𐑯 𐑦𐑑, 𐑑𐑦𐑤 𐑞 𐑮𐑴𐑐 𐑣𐑳𐑥𐑛 𐑤𐑲𐑒 𐑩 𐑣𐑸𐑐-𐑕𐑑𐑮𐑦𐑙, 𐑞𐑧𐑯 𐑰𐑜𐑼𐑤𐑦 𐑚𐑧𐑯𐑛𐑦𐑙 𐑴𐑝𐑼 𐑦𐑑, 𐑯 𐑕𐑰𐑦𐑙 𐑯𐑴 𐑕𐑑𐑮𐑨𐑯𐑛𐑦𐑙𐑟, ·𐑱𐑣𐑨𐑚 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛, “𐑜𐑫𐑛! 𐑯 𐑯𐑬 𐑓 𐑞 𐑕𐑰𐑟𐑦𐑙𐑟.”

𐑨𐑑 𐑢𐑳𐑯 𐑦𐑒𐑕𐑑𐑮𐑧𐑥𐑦𐑑𐑦 𐑞 𐑮𐑴𐑐 𐑢𐑪𐑟 𐑳𐑯𐑕𐑑𐑮𐑨𐑯𐑛𐑩𐑛, 𐑯 𐑞 𐑕𐑧𐑐𐑼𐑩𐑑 𐑕𐑐𐑮𐑧𐑛 𐑘𐑸𐑯𐑟 𐑢𐑻 𐑷𐑤 𐑚𐑮𐑱𐑛𐑩𐑛 𐑯 𐑢𐑴𐑝𐑩𐑯 𐑮𐑬𐑯𐑛 𐑞 𐑕𐑪𐑒𐑩𐑑 𐑝 𐑞 𐑣𐑸𐑐𐑵𐑯; 𐑞 𐑐𐑴𐑤 𐑢𐑪𐑟 𐑞𐑧𐑯 𐑛𐑮𐑦𐑝𐑩𐑯 𐑣𐑸𐑛 𐑳𐑐 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑪𐑒𐑩𐑑; 𐑓𐑮𐑪𐑥 𐑞 𐑤𐑴𐑼 𐑧𐑯𐑛 𐑞 𐑮𐑴𐑐 𐑢𐑪𐑟 𐑑𐑮𐑱𐑕𐑑 𐑣𐑭𐑓-𐑢𐑱 𐑩𐑤𐑪𐑙 𐑞 𐑐𐑴𐑤'𐑟 𐑤𐑧𐑙𐑔, 𐑯 𐑓𐑻𐑥𐑤𐑦 𐑕𐑦𐑒𐑘𐑫𐑼𐑛 𐑕𐑴, 𐑢𐑦𐑞 𐑦𐑯𐑑𐑼𐑑𐑢𐑦𐑕𐑑𐑦𐑙𐑟 𐑝 𐑑𐑢𐑲𐑯. 𐑞𐑦𐑕 𐑛𐑳𐑯, 𐑐𐑴𐑤, 𐑲𐑼𐑯, 𐑯 𐑮𐑴𐑐—𐑤𐑲𐑒 𐑞 ·𐑔𐑮𐑰 ·𐑓𐑱𐑑𐑕—𐑮𐑦𐑥𐑱𐑯𐑛 𐑦𐑯𐑕𐑧𐑐𐑼𐑩𐑚𐑩𐑤, 𐑯 ·𐑱𐑣𐑨𐑚 𐑥𐑵𐑛𐑦𐑤𐑦 𐑕𐑑𐑷𐑒𐑑 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑞 𐑢𐑧𐑐𐑩𐑯; 𐑞 𐑕𐑬𐑯𐑛 𐑝 𐑣𐑦𐑟 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜, 𐑯 𐑞 𐑕𐑬𐑯𐑛 𐑝 𐑞 𐑣𐑦𐑒𐑼𐑦 𐑐𐑴𐑤, 𐑚𐑴𐑔 𐑣𐑪𐑤𐑴𐑤𐑦 𐑮𐑦𐑙𐑦𐑙 𐑩𐑤𐑪𐑙 𐑧𐑝𐑮𐑦 𐑐𐑤𐑨𐑙𐑒. 𐑚𐑳𐑑 𐑺 𐑣𐑰 𐑧𐑯𐑑𐑼𐑛 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯, 𐑤𐑲𐑑, 𐑳𐑯𐑯𐑨𐑗𐑼𐑩𐑤, 𐑣𐑭𐑓-𐑚𐑨𐑯𐑑𐑼𐑦𐑙, 𐑘𐑧𐑑 𐑥𐑴𐑕𐑑 𐑐𐑦𐑑𐑾𐑕 𐑕𐑬𐑯𐑛 𐑢𐑪𐑟 𐑣𐑻𐑛. 𐑴, ·𐑐𐑦𐑐! 𐑞𐑲 𐑮𐑧𐑗𐑩𐑛 𐑤𐑭𐑓, 𐑞𐑲 𐑲𐑛𐑩𐑤 𐑚𐑳𐑑 𐑳𐑯𐑮𐑧𐑕𐑑𐑦𐑙 𐑲; 𐑷𐑤 𐑞𐑲 𐑕𐑑𐑮𐑱𐑯𐑡 𐑥𐑳𐑥𐑼𐑦𐑟 𐑯𐑪𐑑 𐑳𐑯𐑥𐑰𐑯𐑦𐑙𐑤𐑦 𐑚𐑤𐑧𐑯𐑛𐑩𐑛 𐑢𐑦𐑞 𐑞 𐑚𐑤𐑨𐑒 𐑑𐑮𐑨𐑡𐑩𐑛𐑦 𐑝 𐑞 𐑥𐑧𐑤𐑩𐑯𐑒𐑪𐑤𐑦 𐑖𐑦𐑐, 𐑯 𐑥𐑪𐑒𐑑 𐑦𐑑!
