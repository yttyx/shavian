·𐑗𐑨𐑐𐑑𐑼 128. 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 ·𐑥𐑰𐑑𐑕 𐑞 ·𐑮𐑱𐑗𐑩𐑤.

𐑯𐑧𐑒𐑕𐑑 𐑛𐑱, 𐑩 𐑤𐑸𐑡 𐑖𐑦𐑐, 𐑞 ·𐑮𐑱𐑗𐑩𐑤, 𐑢𐑪𐑟 𐑛𐑦𐑕𐑒𐑮𐑲𐑛, 𐑚𐑺𐑦𐑙 𐑛𐑦𐑮𐑧𐑒𐑑𐑤𐑦 𐑛𐑬𐑯 𐑩𐑐𐑪𐑯 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑷𐑤 𐑣𐑻 𐑕𐑐𐑸𐑟 𐑔𐑦𐑒𐑤𐑦 𐑒𐑤𐑳𐑕𐑑𐑼𐑦𐑙 𐑢𐑦𐑞 𐑥𐑧𐑯. 𐑨𐑑 𐑞 𐑑𐑲𐑥 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑪𐑟 𐑥𐑱𐑒𐑦𐑙 𐑜𐑫𐑛 𐑕𐑐𐑰𐑛 𐑔𐑮𐑵 𐑞 𐑢𐑷𐑑𐑼; 𐑚𐑳𐑑 𐑨𐑟 𐑞 𐑚𐑮𐑷𐑛-𐑢𐑦𐑙𐑛 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑖𐑪𐑑 𐑯𐑲 𐑑 𐑣𐑻, 𐑞 𐑚𐑴𐑕𐑑𐑓𐑩𐑤 𐑕𐑱𐑤𐑟 𐑷𐑤 𐑓𐑧𐑤 𐑑𐑩𐑜𐑧𐑞𐑼 𐑨𐑟 𐑚𐑤𐑨𐑙𐑒 𐑚𐑤𐑨𐑛𐑼𐑟 𐑞𐑨𐑑 𐑸 𐑚𐑻𐑕𐑑, 𐑯 𐑷𐑤 𐑤𐑲𐑓 𐑓𐑤𐑧𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑕𐑥𐑦𐑑𐑩𐑯 𐑣𐑳𐑤.

“𐑚𐑨𐑛 𐑯𐑿𐑟; 𐑖𐑰 𐑚𐑮𐑦𐑙𐑟 𐑚𐑨𐑛 𐑯𐑿𐑟,” 𐑥𐑳𐑑𐑼𐑛 𐑞 𐑴𐑤𐑛 ·𐑥𐑨𐑙𐑒𐑕𐑥𐑩𐑯. 𐑚𐑳𐑑 𐑺 𐑣𐑻 𐑒𐑩𐑥𐑭𐑯𐑛𐑼, 𐑣𐑵, 𐑢𐑦𐑞 𐑑𐑮𐑳𐑥𐑐𐑩𐑑 𐑑 𐑥𐑬𐑞, 𐑕𐑑𐑫𐑛 𐑳𐑐 𐑦𐑯 𐑣𐑦𐑟 𐑚𐑴𐑑; 𐑺 𐑣𐑰 𐑒𐑫𐑛 𐑣𐑴𐑐𐑓𐑩𐑤𐑦 𐑣𐑱𐑤, ·𐑱𐑣𐑨𐑚'𐑟 𐑝𐑶𐑕 𐑢𐑪𐑟 𐑣𐑻𐑛.

“𐑣𐑨𐑕𐑑 𐑕𐑰𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?”

“𐑲, 𐑘𐑧𐑕𐑑𐑼𐑛𐑱. 𐑣𐑨𐑝 𐑘𐑰 𐑕𐑰𐑯 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑 𐑩𐑛𐑮𐑦𐑓𐑑?”

𐑔𐑮𐑪𐑑𐑩𐑤𐑦𐑙 𐑣𐑦𐑟 𐑡𐑶, ·𐑱𐑣𐑨𐑚 𐑯𐑧𐑜𐑩𐑑𐑦𐑝𐑤𐑦 𐑭𐑯𐑕𐑼𐑛 𐑞𐑦𐑕 𐑳𐑯𐑦𐑒𐑕𐑐𐑧𐑒𐑑𐑩𐑛 𐑒𐑢𐑧𐑕𐑗𐑩𐑯; 𐑯 𐑢𐑫𐑛 𐑞𐑧𐑯 𐑣𐑨𐑝 𐑓𐑱𐑯 𐑚𐑹𐑛𐑩𐑛 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, 𐑢𐑧𐑯 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑒𐑨𐑐𐑑𐑦𐑯 𐑣𐑦𐑥𐑕𐑧𐑤𐑓, 𐑣𐑨𐑝𐑦𐑙 𐑕𐑑𐑪𐑐𐑑 𐑣𐑦𐑟 𐑝𐑧𐑕𐑩𐑤'𐑟 𐑢𐑱, 𐑢𐑪𐑟 𐑕𐑰𐑯 𐑛𐑦𐑕𐑧𐑯𐑛𐑦𐑙 𐑣𐑻 𐑕𐑲𐑛. 𐑩 𐑓𐑿 𐑒𐑰𐑯 𐑐𐑫𐑤𐑟, 𐑯 𐑣𐑦𐑟 𐑚𐑴𐑑-𐑣𐑫𐑒 𐑕𐑵𐑯 𐑒𐑤𐑦𐑯𐑗𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛'𐑟 𐑥𐑱𐑯-𐑗𐑱𐑯𐑟, 𐑯 𐑣𐑰 𐑕𐑐𐑮𐑨𐑙 𐑑 𐑞 𐑛𐑧𐑒. 𐑦𐑥𐑰𐑛𐑾𐑑𐑤𐑦 𐑣𐑰 𐑢𐑪𐑟 𐑮𐑧𐑒𐑩𐑜𐑯𐑲𐑟𐑛 𐑚𐑲 ·𐑱𐑣𐑨𐑚 𐑓 𐑩 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼 𐑣𐑰 𐑯𐑿. 𐑚𐑳𐑑 𐑯𐑴 𐑓𐑹𐑥𐑩𐑤 𐑕𐑨𐑤𐑿𐑑𐑱𐑖𐑩𐑯 𐑢𐑪𐑟 𐑦𐑒𐑕𐑗𐑱𐑯𐑡𐑛.

“𐑢𐑺 𐑢𐑪𐑟 𐑣𐑰?—𐑯𐑪𐑑 𐑒𐑦𐑤𐑛!—𐑯𐑪𐑑 𐑒𐑦𐑤𐑛!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑒𐑤𐑴𐑕𐑤𐑦 𐑩𐑛𐑝𐑭𐑯𐑕𐑦𐑙. “𐑣𐑬 𐑢𐑪𐑟 𐑦𐑑?”

𐑦𐑑 𐑕𐑰𐑥𐑛 𐑞𐑨𐑑 𐑕𐑳𐑥𐑢𐑪𐑑 𐑤𐑱𐑑 𐑪𐑯 𐑞 𐑭𐑓𐑑𐑼𐑯𐑵𐑯 𐑝 𐑞 𐑛𐑱 𐑐𐑮𐑰𐑝𐑾𐑕, 𐑢𐑲𐑤 𐑔𐑮𐑰 𐑝 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼'𐑟 𐑚𐑴𐑑𐑕 𐑢𐑻 𐑦𐑯𐑜𐑱𐑡𐑛 𐑢𐑦𐑞 𐑩 𐑖𐑴𐑤 𐑝 𐑢𐑱𐑤𐑟, 𐑢𐑦𐑗 𐑣𐑨𐑛 𐑤𐑧𐑛 𐑞𐑧𐑥 𐑕𐑳𐑥 𐑓𐑹 𐑹 𐑓𐑲𐑝 𐑥𐑲𐑤𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐; 𐑯 𐑢𐑲𐑤 𐑞𐑱 𐑢𐑻 𐑘𐑧𐑑 𐑦𐑯 𐑕𐑢𐑦𐑓𐑑 𐑗𐑱𐑕 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛, 𐑞 𐑢𐑲𐑑 𐑣𐑳𐑥𐑐 𐑯 𐑣𐑧𐑛 𐑝 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑣𐑨𐑛 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑤𐑵𐑥𐑛 𐑳𐑐 𐑬𐑑 𐑝 𐑞 𐑢𐑷𐑑𐑼, 𐑯𐑪𐑑 𐑝𐑧𐑮𐑦 𐑓𐑸 𐑑 𐑤𐑰𐑢𐑼𐑛; 𐑢𐑺𐑩𐑐𐑪𐑯, 𐑞 𐑓𐑹𐑔 𐑮𐑦𐑜𐑛 𐑚𐑴𐑑—𐑩 𐑮𐑦𐑟𐑻𐑝𐑛 𐑢𐑳𐑯—𐑣𐑨𐑛 𐑚𐑰𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦 𐑤𐑴𐑼𐑛 𐑦𐑯 𐑗𐑱𐑕. 𐑭𐑓𐑑𐑼 𐑩 𐑒𐑰𐑯 𐑕𐑱𐑤 𐑚𐑦𐑓𐑹 𐑞 𐑢𐑦𐑯𐑛, 𐑞𐑦𐑕 𐑓𐑹𐑔 𐑚𐑴𐑑—𐑞 𐑕𐑢𐑦𐑓𐑑𐑩𐑕𐑑 𐑒𐑰𐑤𐑛 𐑝 𐑷𐑤—𐑕𐑰𐑥𐑛 𐑑 𐑣𐑨𐑝 𐑕𐑩𐑒𐑕𐑰𐑛𐑩𐑛 𐑦𐑯 𐑓𐑭𐑕𐑩𐑯𐑦𐑙—𐑨𐑑 𐑤𐑰𐑕𐑑, 𐑨𐑟 𐑢𐑧𐑤 𐑨𐑟 𐑞 𐑥𐑨𐑯 𐑨𐑑 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛 𐑒𐑫𐑛 𐑑𐑧𐑤 𐑧𐑯𐑦𐑔𐑦𐑙 𐑩𐑚𐑬𐑑 𐑦𐑑. 𐑦𐑯 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑣𐑰 𐑕𐑷 𐑞 𐑛𐑦𐑥𐑦𐑯𐑦𐑖𐑑 𐑛𐑪𐑑𐑩𐑛 𐑚𐑴𐑑; 𐑯 𐑞𐑧𐑯 𐑩 𐑕𐑢𐑦𐑓𐑑 𐑜𐑤𐑰𐑥 𐑝 𐑚𐑳𐑚𐑩𐑤𐑦𐑙 𐑢𐑲𐑑 𐑢𐑷𐑑𐑼; 𐑯 𐑭𐑓𐑑𐑼 𐑞𐑨𐑑 𐑯𐑳𐑔𐑦𐑙 𐑥𐑹; 𐑢𐑧𐑯𐑕 𐑦𐑑 𐑢𐑪𐑟 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛 𐑞𐑨𐑑 𐑞 𐑕𐑑𐑮𐑦𐑒𐑩𐑯 𐑢𐑱𐑤 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑦𐑯𐑛𐑧𐑓𐑦𐑯𐑦𐑑𐑤𐑦 𐑮𐑳𐑯 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑐𐑼𐑕𐑿𐑼𐑟, 𐑨𐑟 𐑪𐑓𐑩𐑯 𐑣𐑨𐑐𐑩𐑯𐑟. 𐑞𐑺 𐑢𐑪𐑟 𐑕𐑳𐑥 𐑨𐑐𐑮𐑦𐑣𐑧𐑯𐑖𐑩𐑯, 𐑚𐑳𐑑 𐑯𐑴 𐑐𐑪𐑟𐑦𐑑𐑦𐑝 𐑩𐑤𐑸𐑥, 𐑨𐑟 𐑘𐑧𐑑. 𐑞 𐑮𐑦𐑒𐑷𐑤 𐑕𐑦𐑜𐑯𐑩𐑤𐑟 𐑢𐑻 𐑐𐑤𐑱𐑕𐑑 𐑦𐑯 𐑞 𐑮𐑦𐑜𐑦𐑙; 𐑛𐑸𐑒𐑯𐑩𐑕 𐑒𐑱𐑥 𐑪𐑯; 𐑯 𐑓𐑹𐑕𐑑 𐑑 𐑐𐑦𐑒 𐑳𐑐 𐑣𐑻 𐑔𐑮𐑰 𐑓𐑸 𐑑 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑚𐑴𐑑𐑕—𐑺 𐑜𐑴𐑦𐑙 𐑦𐑯 𐑒𐑢𐑧𐑕𐑑 𐑝 𐑞 𐑓𐑹𐑔 𐑢𐑳𐑯 𐑦𐑯 𐑞 𐑐𐑮𐑦𐑕𐑲𐑕𐑤𐑦 𐑪𐑐𐑩𐑟𐑦𐑑 𐑛𐑦𐑮𐑧𐑒𐑖𐑩𐑯—𐑞 𐑖𐑦𐑐 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑚𐑰𐑯 𐑯𐑦𐑕𐑧𐑕𐑦𐑑𐑱𐑑𐑩𐑛 𐑑 𐑤𐑰𐑝 𐑞𐑨𐑑 𐑚𐑴𐑑 𐑑 𐑦𐑑𐑕 𐑓𐑱𐑑 𐑑𐑦𐑤 𐑯𐑽 𐑥𐑦𐑛𐑯𐑲𐑑, 𐑚𐑳𐑑, 𐑓 𐑞 𐑑𐑲𐑥, 𐑑 𐑦𐑯𐑒𐑮𐑰𐑕 𐑣𐑻 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑓𐑮𐑪𐑥 𐑦𐑑. 𐑚𐑳𐑑 𐑞 𐑮𐑧𐑕𐑑 𐑝 𐑣𐑻 𐑒𐑮𐑵 𐑚𐑰𐑦𐑙 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑕𐑱𐑓 𐑩𐑚𐑹𐑛, 𐑖𐑰 𐑒𐑮𐑬𐑛𐑩𐑛 𐑷𐑤 𐑕𐑱𐑤—𐑕𐑑𐑳𐑯𐑕𐑱𐑤 𐑪𐑯 𐑕𐑑𐑳𐑯𐑕𐑱𐑤—𐑭𐑓𐑑𐑼 𐑞 𐑥𐑦𐑕𐑦𐑙 𐑚𐑴𐑑; 𐑒𐑦𐑯𐑛𐑩𐑤𐑦𐑙 𐑩 𐑓𐑲𐑼 𐑦𐑯 𐑣𐑻 𐑑𐑮𐑲-𐑐𐑪𐑑𐑕 𐑓 𐑩 𐑚𐑰𐑒𐑩𐑯; 𐑯 𐑧𐑝𐑮𐑦 𐑳𐑞𐑼 𐑥𐑨𐑯 𐑩𐑤𐑪𐑓𐑑 𐑪𐑯 𐑞 𐑤𐑫𐑒-𐑬𐑑. 𐑚𐑳𐑑 𐑞𐑴 𐑢𐑧𐑯 𐑖𐑰 𐑣𐑨𐑛 𐑞𐑳𐑕 𐑕𐑱𐑤𐑛 𐑩 𐑕𐑩𐑓𐑦𐑖𐑩𐑯𐑑 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑑 𐑜𐑱𐑯 𐑞 𐑐𐑮𐑦𐑟𐑿𐑥𐑛 𐑐𐑤𐑱𐑕 𐑝 𐑞 𐑨𐑚𐑕𐑩𐑯𐑑 𐑢𐑳𐑯𐑟 𐑢𐑧𐑯 𐑤𐑭𐑕𐑑 𐑕𐑰𐑯; 𐑞𐑴 𐑖𐑰 𐑞𐑧𐑯 𐑐𐑷𐑟𐑛 𐑑 𐑤𐑴𐑼 𐑣𐑻 𐑕𐑐𐑺 𐑚𐑴𐑑𐑕 𐑑 𐑐𐑫𐑤 𐑷𐑤 𐑼𐑬𐑯𐑛 𐑣𐑻; 𐑯 𐑯𐑪𐑑 𐑓𐑲𐑯𐑛𐑦𐑙 𐑧𐑯𐑦𐑔𐑦𐑙, 𐑣𐑨𐑛 𐑩𐑜𐑱𐑯 𐑛𐑨𐑖𐑑 𐑪𐑯; 𐑩𐑜𐑱𐑯 𐑐𐑷𐑟𐑛, 𐑯 𐑤𐑴𐑼𐑛 𐑣𐑻 𐑚𐑴𐑑𐑕; 𐑯 𐑞𐑴 𐑖𐑰 𐑣𐑨𐑛 𐑞𐑳𐑕 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑛 𐑛𐑵𐑦𐑙 𐑑𐑦𐑤 𐑛𐑱𐑤𐑲𐑑; 𐑘𐑧𐑑 𐑯𐑪𐑑 𐑞 𐑤𐑰𐑕𐑑 𐑜𐑤𐑦𐑥𐑐𐑕 𐑝 𐑞 𐑥𐑦𐑕𐑦𐑙 𐑒𐑰𐑤 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑰𐑯.

𐑞 𐑕𐑑𐑹𐑦 𐑑𐑴𐑤𐑛, 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑦𐑥𐑰𐑛𐑾𐑑𐑤𐑦 𐑢𐑧𐑯𐑑 𐑪𐑯 𐑑 𐑮𐑦𐑝𐑰𐑤 𐑣𐑦𐑟 𐑪𐑚𐑡𐑧𐑒𐑑 𐑦𐑯 𐑚𐑹𐑛𐑦𐑙 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛. 𐑣𐑰 𐑛𐑦𐑟𐑲𐑼𐑛 𐑞𐑨𐑑 𐑖𐑦𐑐 𐑑 𐑿𐑯𐑲𐑑 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑴𐑯 𐑦𐑯 𐑞 𐑕𐑻𐑗; 𐑚𐑲 𐑕𐑱𐑤𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑕𐑰 𐑕𐑳𐑥 𐑓𐑹 𐑹 𐑓𐑲𐑝 𐑥𐑲𐑤𐑟 𐑩𐑐𐑸𐑑, 𐑪𐑯 𐑐𐑨𐑮𐑩𐑤𐑧𐑤 𐑤𐑲𐑯𐑟, 𐑯 𐑕𐑴 𐑕𐑢𐑰𐑐𐑦𐑙 𐑩 𐑛𐑳𐑚𐑩𐑤 𐑣𐑼𐑲𐑟𐑩𐑯, 𐑨𐑟 𐑦𐑑 𐑢𐑻.

“𐑲 𐑢𐑦𐑤 𐑢𐑱𐑡𐑼 𐑕𐑳𐑥𐑔𐑦𐑙 𐑯𐑬,” 𐑢𐑦𐑕𐑐𐑼𐑛 ·𐑕𐑑𐑳𐑚 𐑑 ·𐑓𐑤𐑭𐑕𐑒, “𐑞𐑨𐑑 𐑕𐑳𐑥 𐑢𐑳𐑯 𐑦𐑯 𐑞𐑨𐑑 𐑥𐑦𐑕𐑦𐑙 𐑚𐑴𐑑 𐑢𐑹 𐑪𐑓 𐑞𐑨𐑑 ·𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑚𐑧𐑕𐑑 𐑒𐑴𐑑; 𐑥𐑱𐑣𐑨𐑐, 𐑣𐑦𐑟 𐑢𐑪𐑗—𐑣𐑰'𐑟 𐑕𐑴 𐑒𐑻𐑕𐑑 𐑨𐑙𐑒𐑖𐑩𐑕 𐑑 𐑜𐑧𐑑 𐑦𐑑 𐑚𐑨𐑒. 𐑣𐑵 𐑧𐑝𐑼 𐑣𐑻𐑛 𐑝 𐑑𐑵 𐑐𐑲𐑩𐑕 𐑢𐑱𐑤-𐑖𐑦𐑐𐑕 𐑒𐑮𐑵𐑟𐑦𐑙 𐑭𐑓𐑑𐑼 𐑢𐑳𐑯 𐑥𐑦𐑕𐑦𐑙 𐑢𐑱𐑤-𐑚𐑴𐑑 𐑦𐑯 𐑞 𐑣𐑲𐑑 𐑝 𐑞 𐑢𐑱𐑤𐑦𐑙 𐑕𐑰𐑟𐑩𐑯? 𐑕𐑰, ·𐑓𐑤𐑭𐑕𐑒, 𐑴𐑯𐑤𐑦 𐑕𐑰 𐑣𐑬 𐑐𐑱𐑤 𐑣𐑰 𐑤𐑫𐑒𐑕—𐑐𐑱𐑤 𐑦𐑯 𐑞 𐑝𐑧𐑮𐑦 𐑚𐑳𐑑𐑩𐑯𐑟 𐑝 𐑣𐑦𐑟 𐑲𐑟—𐑤𐑫𐑒—𐑦𐑑 𐑢𐑪𐑟𐑩𐑯𐑑 𐑞 𐑒𐑴𐑑—𐑦𐑑 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑞—”

“𐑥𐑲 𐑚𐑶, 𐑥𐑲 𐑴𐑯 𐑚𐑶 𐑦𐑟 𐑩𐑥𐑳𐑙 𐑞𐑧𐑥. 𐑓 ·𐑜𐑪𐑛'𐑟 𐑕𐑱𐑒—𐑲 𐑚𐑧𐑜, 𐑲 𐑒𐑳𐑯𐑡𐑼”—𐑣𐑽 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑑 ·𐑱𐑣𐑨𐑚, 𐑣𐑵 𐑞𐑳𐑕 𐑓𐑸 𐑣𐑨𐑛 𐑚𐑳𐑑 𐑲𐑕𐑦𐑤𐑦 𐑮𐑦𐑕𐑰𐑝𐑛 𐑣𐑦𐑟 𐑐𐑩𐑑𐑦𐑖𐑩𐑯. “𐑓 𐑱𐑑-𐑯-𐑓𐑹𐑑𐑦 𐑬𐑼𐑟 𐑤𐑧𐑑 𐑥𐑰 𐑗𐑸𐑑𐑼 𐑘𐑹 𐑖𐑦𐑐—𐑲 𐑢𐑦𐑤 𐑜𐑤𐑨𐑛𐑤𐑦 𐑐𐑱 𐑓 𐑦𐑑, 𐑯 𐑮𐑬𐑯𐑛𐑤𐑦 𐑐𐑱 𐑓 𐑦𐑑—𐑦𐑓 𐑞𐑺 𐑚𐑰 𐑯𐑴 𐑳𐑞𐑼 𐑢𐑱—𐑓 𐑱𐑑-𐑯-𐑓𐑹𐑑𐑦 𐑬𐑼𐑟 𐑴𐑯𐑤𐑦—𐑴𐑯𐑤𐑦 𐑞𐑨𐑑—𐑿 𐑥𐑳𐑕𐑑, 𐑴, 𐑿 𐑥𐑳𐑕𐑑, 𐑯 𐑿 𐑖𐑨𐑤 𐑛𐑵 𐑞𐑦𐑕 𐑔𐑦𐑙.”

“𐑣𐑦𐑟 𐑕𐑳𐑯!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, “𐑴, 𐑦𐑑'𐑕 𐑣𐑦𐑟 𐑕𐑳𐑯 𐑣𐑰'𐑟 𐑤𐑪𐑕𐑑! 𐑲 𐑑𐑱𐑒 𐑚𐑨𐑒 𐑞 𐑒𐑴𐑑 𐑯 𐑢𐑪𐑗—𐑢𐑪𐑑 𐑕𐑧𐑟 ·𐑱𐑣𐑨𐑚? 𐑢𐑰 𐑥𐑳𐑕𐑑 𐑕𐑱𐑝 𐑞𐑨𐑑 𐑚𐑶.”

“𐑣𐑰'𐑟 𐑛𐑮𐑬𐑯𐑛 𐑢𐑦𐑞 𐑞 𐑮𐑧𐑕𐑑 𐑪𐑯 '𐑩𐑥, 𐑤𐑭𐑕𐑑 𐑯𐑲𐑑,” 𐑕𐑧𐑛 𐑞 𐑴𐑤𐑛 ·𐑥𐑨𐑙𐑒𐑕 𐑕𐑱𐑤𐑼 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑦𐑣𐑲𐑯𐑛 𐑞𐑧𐑥; “𐑲 𐑣𐑻𐑛; 𐑷𐑤 𐑝 𐑘𐑰 𐑣𐑻𐑛 𐑞𐑺 𐑕𐑐𐑦𐑮𐑦𐑑𐑕.”

𐑯𐑬, 𐑨𐑟 𐑦𐑑 𐑖𐑹𐑑𐑤𐑦 𐑑𐑻𐑯𐑛 𐑬𐑑, 𐑢𐑪𐑑 𐑥𐑱𐑛 𐑞𐑦𐑕 𐑦𐑯𐑕𐑦𐑛𐑩𐑯𐑑 𐑝 𐑞 ·𐑮𐑱𐑗𐑩𐑤'𐑟 𐑞 𐑥𐑹 𐑥𐑧𐑤𐑩𐑯𐑒𐑪𐑤𐑦, 𐑢𐑪𐑟 𐑞 𐑕𐑻𐑒𐑩𐑥𐑕𐑑𐑨𐑯𐑕, 𐑞𐑨𐑑 𐑯𐑪𐑑 𐑴𐑯𐑤𐑦 𐑢𐑪𐑟 𐑢𐑳𐑯 𐑝 𐑞 ·𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑕𐑳𐑯𐑟 𐑩𐑥𐑳𐑙 𐑞 𐑯𐑳𐑥𐑚𐑼 𐑝 𐑞 𐑥𐑦𐑕𐑦𐑙 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵; 𐑚𐑳𐑑 𐑩𐑥𐑳𐑙 𐑞 𐑯𐑳𐑥𐑚𐑼 𐑝 𐑞 𐑳𐑞𐑼 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵𐑟, 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥, 𐑚𐑳𐑑 𐑪𐑯 𐑞 𐑳𐑞𐑼 𐑣𐑨𐑯𐑛, 𐑕𐑧𐑐𐑼𐑱𐑑𐑩𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐 𐑛𐑘𐑫𐑼𐑦𐑙 𐑞 𐑛𐑸𐑒 𐑝𐑦𐑕𐑦𐑕𐑦𐑑𐑿𐑛𐑟 𐑝 𐑞 𐑗𐑱𐑕, 𐑞𐑺 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑕𐑑𐑦𐑤 𐑩𐑯𐑳𐑞𐑼 𐑕𐑳𐑯; 𐑨𐑟 𐑞𐑨𐑑 𐑓 𐑩 𐑑𐑲𐑥, 𐑞 𐑮𐑧𐑗𐑩𐑛 𐑓𐑭𐑞𐑼 𐑢𐑪𐑟 𐑐𐑤𐑳𐑯𐑡𐑛 𐑑 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 𐑞 𐑒𐑮𐑵𐑩𐑤𐑩𐑕𐑑 𐑐𐑼𐑐𐑤𐑧𐑒𐑕𐑦𐑑𐑦; 𐑢𐑦𐑗 𐑢𐑪𐑟 𐑴𐑯𐑤𐑦 𐑕𐑪𐑤𐑝𐑛 𐑓 𐑣𐑦𐑥 𐑚𐑲 𐑣𐑦𐑟 𐑗𐑰𐑓 𐑥𐑱𐑑'𐑕 𐑦𐑯𐑕𐑑𐑦𐑙𐑒𐑑𐑦𐑝𐑤𐑦 𐑩𐑛𐑪𐑐𐑑𐑦𐑙 𐑞 𐑹𐑛𐑦𐑯𐑼𐑦 𐑐𐑮𐑴𐑕𐑰𐑡𐑼 𐑝 𐑩 𐑢𐑱𐑤-𐑖𐑦𐑐 𐑦𐑯 𐑕𐑳𐑗 𐑦𐑥𐑻𐑡𐑩𐑯𐑕𐑦𐑟, 𐑞𐑨𐑑 𐑦𐑟, 𐑢𐑧𐑯 𐑐𐑤𐑱𐑕𐑑 𐑚𐑦𐑑𐑢𐑰𐑯 𐑡𐑧𐑐𐑼𐑛𐑲𐑟𐑛 𐑚𐑳𐑑 𐑛𐑦𐑝𐑲𐑛𐑩𐑛 𐑚𐑴𐑑𐑕, 𐑷𐑤𐑢𐑱𐑟 𐑑 𐑐𐑦𐑒 𐑳𐑐 𐑞 𐑥𐑩𐑡𐑪𐑮𐑦𐑑𐑦 𐑓𐑻𐑕𐑑. 𐑚𐑳𐑑 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑓 𐑕𐑳𐑥 𐑳𐑯𐑯𐑴𐑯 𐑒𐑪𐑯𐑕𐑑𐑦𐑑𐑿𐑖𐑩𐑯𐑩𐑤 𐑮𐑰𐑟𐑩𐑯, 𐑣𐑨𐑛 𐑮𐑦𐑓𐑮𐑱𐑯𐑛 𐑓𐑮𐑪𐑥 𐑥𐑧𐑯𐑖𐑩𐑯𐑦𐑙 𐑷𐑤 𐑞𐑦𐑕, 𐑯 𐑯𐑪𐑑 𐑑𐑦𐑤 𐑓𐑹𐑕𐑑 𐑑 𐑦𐑑 𐑚𐑲 ·𐑱𐑣𐑨𐑚'𐑟 𐑲𐑕𐑦𐑯𐑩𐑕 𐑛𐑦𐑛 𐑣𐑰 𐑩𐑤𐑵𐑛 𐑑 𐑣𐑦𐑟 𐑢𐑳𐑯 𐑘𐑧𐑑 𐑥𐑦𐑕𐑦𐑙 𐑚𐑶; 𐑩 𐑤𐑦𐑑𐑩𐑤 𐑤𐑨𐑛, 𐑚𐑳𐑑 𐑑𐑢𐑧𐑤𐑝 𐑘𐑽𐑟 𐑴𐑤𐑛, 𐑣𐑵𐑟 𐑓𐑭𐑞𐑼 𐑢𐑦𐑞 𐑞 𐑻𐑯𐑩𐑕𐑑 𐑚𐑳𐑑 𐑳𐑯𐑥𐑦𐑕𐑜𐑦𐑝𐑦𐑙 𐑣𐑸𐑛𐑦𐑣𐑫𐑛 𐑝 𐑩 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼'𐑟 𐑐𐑩𐑑𐑻𐑯𐑩𐑤 𐑤𐑳𐑝, 𐑣𐑨𐑛 𐑞𐑳𐑕 𐑻𐑤𐑦 𐑕𐑷𐑑 𐑑 𐑦𐑯𐑦𐑖𐑦𐑱𐑑 𐑣𐑦𐑥 𐑦𐑯 𐑞 𐑐𐑧𐑮𐑦𐑤𐑟 𐑯 𐑢𐑳𐑯𐑛𐑼𐑟 𐑝 𐑩 𐑝𐑴𐑒𐑱𐑖𐑩𐑯 𐑷𐑤𐑥𐑴𐑕𐑑 𐑦𐑥𐑩𐑥𐑹𐑾𐑤𐑦 𐑞 𐑛𐑧𐑕𐑑𐑦𐑯𐑦 𐑝 𐑷𐑤 𐑣𐑦𐑟 𐑮𐑱𐑕. 𐑯𐑹 𐑛𐑳𐑟 𐑦𐑑 𐑳𐑯𐑓𐑮𐑰𐑒𐑢𐑩𐑯𐑑𐑤𐑦 𐑩𐑒𐑻, 𐑞𐑨𐑑 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑 𐑒𐑨𐑐𐑑𐑦𐑯𐑟 𐑢𐑦𐑤 𐑕𐑧𐑯𐑛 𐑩 𐑕𐑳𐑯 𐑝 𐑕𐑳𐑗 𐑑𐑧𐑯𐑛𐑼 𐑱𐑡 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑞𐑧𐑥, 𐑓 𐑩 𐑐𐑮𐑴𐑑𐑮𐑨𐑒𐑑𐑩𐑛 𐑔𐑮𐑰 𐑹 𐑓𐑹 𐑘𐑽𐑟' 𐑝𐑶𐑦𐑡 𐑦𐑯 𐑕𐑳𐑥 𐑳𐑞𐑼 𐑖𐑦𐑐 𐑞𐑨𐑯 𐑞𐑺 𐑴𐑯; 𐑕𐑴 𐑞𐑨𐑑 𐑞𐑺 𐑓𐑻𐑕𐑑 𐑯𐑪𐑤𐑦𐑡 𐑝 𐑩 𐑢𐑱𐑤𐑥𐑩𐑯'𐑟 𐑒𐑼𐑽 𐑖𐑨𐑤 𐑚𐑰 𐑳𐑯𐑧𐑯𐑼𐑝𐑱𐑑𐑩𐑛 𐑚𐑲 𐑧𐑯𐑦 𐑗𐑭𐑯𐑕 𐑛𐑦𐑕𐑐𐑤𐑱 𐑝 𐑩 𐑓𐑭𐑞𐑼'𐑟 𐑯𐑨𐑗𐑼𐑩𐑤 𐑚𐑳𐑑 𐑳𐑯𐑑𐑲𐑥𐑤𐑦 𐑐𐑸𐑖𐑦𐑨𐑤𐑦𐑑𐑦, 𐑹 𐑳𐑯𐑛𐑿 𐑨𐑐𐑮𐑦𐑣𐑧𐑯𐑕𐑦𐑝𐑯𐑩𐑕 𐑯 𐑒𐑩𐑯𐑕𐑻𐑯.

𐑥𐑰𐑯𐑑𐑲𐑥, 𐑯𐑬 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑢𐑪𐑟 𐑕𐑑𐑦𐑤 𐑚𐑦𐑕𐑰𐑗𐑦𐑙 𐑣𐑦𐑟 𐑐𐑫𐑼 𐑚𐑵𐑯 𐑝 ·𐑱𐑣𐑨𐑚; 𐑯 ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑦𐑤 𐑕𐑑𐑫𐑛 𐑤𐑲𐑒 𐑩𐑯 𐑨𐑯𐑝𐑦𐑤, 𐑮𐑦𐑕𐑰𐑝𐑦𐑙 𐑧𐑝𐑮𐑦 𐑖𐑪𐑒, 𐑚𐑳𐑑 𐑢𐑦𐑞𐑬𐑑 𐑞 𐑤𐑰𐑕𐑑 𐑒𐑢𐑦𐑝𐑼𐑦𐑙 𐑝 𐑣𐑦𐑟 𐑴𐑯.

“𐑲 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑜𐑴,” 𐑕𐑧𐑛 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, “𐑑𐑦𐑤 𐑿 𐑕𐑱 𐑲 𐑑 𐑥𐑰. 𐑛𐑵 𐑑 𐑥𐑰 𐑨𐑟 𐑿 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑥𐑰 𐑛𐑵 𐑑 𐑿 𐑦𐑯 𐑞 𐑤𐑲𐑒 𐑒𐑱𐑕. 𐑓 𐑿 𐑑𐑵 𐑣𐑨𐑝 𐑩 𐑚𐑶, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑱𐑣𐑨𐑚—𐑞𐑴 𐑚𐑳𐑑 𐑩 𐑗𐑲𐑤𐑛, 𐑯 𐑯𐑧𐑕𐑤𐑦𐑙 𐑕𐑱𐑓𐑤𐑦 𐑨𐑑 𐑣𐑴𐑥 𐑯𐑬—𐑩 𐑗𐑲𐑤𐑛 𐑝 𐑘𐑹 𐑴𐑤𐑛 𐑱𐑡 𐑑𐑵—·𐑘𐑧𐑕, 𐑘𐑧𐑕, 𐑿 𐑮𐑦𐑤𐑧𐑯𐑑; 𐑲 𐑕𐑰 𐑦𐑑—𐑮𐑳𐑯, 𐑮𐑳𐑯, 𐑥𐑧𐑯, 𐑯𐑬, 𐑯 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑕𐑒𐑢𐑺 𐑦𐑯 𐑞 𐑘𐑸𐑛𐑟.”

“𐑩𐑝𐑭𐑕𐑑,” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚—“𐑑𐑳𐑗 𐑯𐑪𐑑 𐑩 𐑮𐑴𐑐-𐑘𐑸𐑯”; 𐑞𐑧𐑯 𐑦𐑯 𐑩 𐑝𐑶𐑕 𐑞𐑨𐑑 𐑐𐑮𐑴𐑤𐑪𐑙𐑦𐑙𐑤𐑦 𐑥𐑴𐑤𐑛𐑩𐑛 𐑧𐑝𐑮𐑦 𐑢𐑻𐑛—“𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑜𐑸𐑛𐑯𐑼, 𐑲 𐑢𐑦𐑤 𐑯𐑪𐑑 𐑛𐑵 𐑦𐑑. 𐑰𐑝𐑩𐑯 𐑯𐑬 𐑲 𐑤𐑵𐑟 𐑑𐑲𐑥. 𐑜𐑫𐑛-𐑚𐑲, 𐑜𐑫𐑛-𐑚𐑲. 𐑜𐑪𐑛 𐑚𐑤𐑧𐑕 𐑘𐑰, 𐑥𐑨𐑯, 𐑯 𐑥𐑱 𐑲 𐑓𐑼𐑜𐑦𐑝 𐑥𐑲𐑕𐑧𐑤𐑓, 𐑚𐑳𐑑 𐑲 𐑥𐑳𐑕𐑑 𐑜𐑴. ·𐑥𐑼 ·𐑕𐑑𐑸𐑚𐑳𐑒, 𐑤𐑫𐑒 𐑨𐑑 𐑞 𐑚𐑦𐑯𐑩𐑒𐑩𐑤 𐑢𐑪𐑗, 𐑯 𐑦𐑯 𐑔𐑮𐑰 𐑥𐑦𐑯𐑦𐑑𐑕 𐑓𐑮𐑪𐑥 𐑞𐑦𐑕 𐑐𐑮𐑧𐑟𐑩𐑯𐑑 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑢𐑹𐑯 𐑪𐑓 𐑷𐑤 𐑕𐑑𐑮𐑱𐑯𐑡𐑼𐑟: 𐑞𐑧𐑯 𐑚𐑮𐑱𐑕 𐑓𐑹𐑢𐑼𐑛 𐑩𐑜𐑱𐑯, 𐑯 𐑤𐑧𐑑 𐑞 𐑖𐑦𐑐 𐑕𐑱𐑤 𐑨𐑟 𐑚𐑦𐑓𐑹.”

𐑣𐑳𐑮𐑦𐑛𐑤𐑦 𐑑𐑻𐑯𐑦𐑙, 𐑢𐑦𐑞 𐑩𐑝𐑻𐑑𐑩𐑛 𐑓𐑱𐑕, 𐑣𐑰 𐑛𐑦𐑕𐑧𐑯𐑛𐑩𐑛 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑒𐑨𐑚𐑦𐑯, 𐑤𐑰𐑝𐑦𐑙 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡 𐑒𐑨𐑐𐑑𐑦𐑯 𐑑𐑮𐑨𐑯𐑕𐑓𐑦𐑒𐑕𐑑 𐑨𐑑 𐑞𐑦𐑕 𐑳𐑯𐑒𐑩𐑯𐑛𐑦𐑖𐑩𐑯𐑩𐑤 𐑯 𐑳𐑑𐑼 𐑮𐑦𐑡𐑧𐑒𐑖𐑩𐑯 𐑝 𐑣𐑦𐑟 𐑕𐑴 𐑻𐑯𐑩𐑕𐑑 𐑕𐑵𐑑. 𐑚𐑳𐑑 𐑕𐑑𐑸𐑑𐑦𐑙 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑦𐑯𐑗𐑭𐑯𐑑𐑥𐑩𐑯𐑑, ·𐑜𐑸𐑛𐑯𐑼 𐑕𐑲𐑤𐑩𐑯𐑑𐑤𐑦 𐑣𐑳𐑮𐑦𐑛 𐑑 𐑞 𐑕𐑲𐑛; 𐑥𐑹 𐑓𐑧𐑤 𐑞𐑨𐑯 𐑕𐑑𐑧𐑐𐑑 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑚𐑴𐑑, 𐑯 𐑮𐑩𐑑𐑻𐑯𐑛 𐑑 𐑣𐑦𐑟 𐑖𐑦𐑐.

𐑕𐑵𐑯 𐑞 𐑑𐑵 𐑖𐑦𐑐𐑕 𐑛𐑲𐑝𐑻𐑡𐑛 𐑞𐑺 𐑢𐑱𐑒𐑕; 𐑯 𐑤𐑪𐑙 𐑨𐑟 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡 𐑝𐑧𐑕𐑩𐑤 𐑢𐑪𐑟 𐑦𐑯 𐑝𐑿, 𐑖𐑰 𐑢𐑪𐑟 𐑕𐑰𐑯 𐑑 𐑘𐑷 𐑣𐑦𐑞𐑼 𐑯 𐑞𐑦𐑞𐑼 𐑨𐑑 𐑧𐑝𐑮𐑦 𐑛𐑸𐑒 𐑕𐑐𐑪𐑑, 𐑣𐑬𐑧𐑝𐑼 𐑕𐑥𐑷𐑤, 𐑪𐑯 𐑞 𐑕𐑰. 𐑞𐑦𐑕 𐑢𐑱 𐑯 𐑞𐑨𐑑 𐑣𐑻 𐑘𐑸𐑛𐑟 𐑢𐑻 𐑕𐑢𐑳𐑙 𐑮𐑬𐑯𐑛; 𐑕𐑑𐑸𐑚𐑼𐑛 𐑯 𐑤𐑸𐑚𐑼𐑛, 𐑖𐑰 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑛 𐑑 𐑑𐑨𐑒; 𐑯𐑬 𐑖𐑰 𐑚𐑰𐑑 𐑩𐑜𐑱𐑯𐑕𐑑 𐑩 𐑣𐑧𐑛 𐑕𐑰; 𐑯 𐑩𐑜𐑱𐑯 𐑦𐑑 𐑐𐑫𐑖𐑑 𐑣𐑻 𐑚𐑦𐑓𐑹 𐑦𐑑; 𐑢𐑲𐑤 𐑷𐑤 𐑞 𐑢𐑲𐑤, 𐑣𐑻 𐑥𐑭𐑕𐑑𐑕 𐑯 𐑘𐑸𐑛𐑟 𐑢𐑻 𐑔𐑦𐑒𐑤𐑦 𐑒𐑤𐑳𐑕𐑑𐑼𐑛 𐑢𐑦𐑞 𐑥𐑧𐑯, 𐑨𐑟 𐑔𐑮𐑰 𐑑𐑷𐑤 𐑗𐑧𐑮𐑦 𐑑𐑮𐑰𐑟, 𐑢𐑧𐑯 𐑞 𐑚𐑶𐑟 𐑸 𐑗𐑧𐑮𐑦𐑦𐑙 𐑩𐑥𐑳𐑙 𐑞 𐑚𐑬𐑟.

𐑚𐑳𐑑 𐑚𐑲 𐑣𐑻 𐑕𐑑𐑦𐑤 𐑣𐑷𐑤𐑑𐑦𐑙 𐑒𐑹𐑕 𐑯 𐑢𐑲𐑯𐑛𐑦𐑙, 𐑢𐑴𐑓𐑩𐑤 𐑢𐑱, 𐑿 𐑐𐑤𐑱𐑯𐑤𐑦 𐑕𐑷 𐑞𐑨𐑑 𐑞𐑦𐑕 𐑖𐑦𐑐 𐑞𐑨𐑑 𐑕𐑴 𐑢𐑧𐑐𐑑 𐑢𐑦𐑞 𐑕𐑐𐑮𐑱, 𐑕𐑑𐑦𐑤 𐑮𐑦𐑥𐑱𐑯𐑛 𐑢𐑦𐑞𐑬𐑑 𐑒𐑳𐑥𐑓𐑼𐑑. 𐑖𐑰 𐑢𐑪𐑟 ·𐑮𐑱𐑗𐑩𐑤, 𐑢𐑰𐑐𐑦𐑙 𐑓 𐑣𐑻 𐑗𐑦𐑤𐑛𐑮𐑩𐑯, 𐑚𐑦𐑒𐑪𐑟 𐑞𐑱 𐑢𐑻 𐑯𐑪𐑑.
