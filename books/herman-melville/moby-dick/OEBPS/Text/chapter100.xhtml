“𐑥𐑨𐑯 𐑥𐑲 𐑚𐑴𐑑!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑦𐑥𐑐𐑧𐑗𐑵𐑩𐑕𐑤𐑦, 𐑯 𐑑𐑪𐑕𐑦𐑙 𐑩𐑚𐑬𐑑 𐑞 𐑹𐑟 𐑯𐑽 𐑣𐑦𐑥—“𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑤𐑴𐑼!”

“𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤,” 𐑕𐑧𐑛 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖𐑥𐑩𐑯, 𐑐𐑶𐑯𐑑𐑦𐑙 𐑣𐑦𐑟 𐑲𐑝𐑼𐑦 𐑸𐑥 𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 ·𐑰𐑕𐑑, 𐑯 𐑑𐑱𐑒𐑦𐑙 𐑩 𐑮𐑵𐑓𐑩𐑤 𐑕𐑲𐑑 𐑩𐑤𐑪𐑙 𐑦𐑑, 𐑨𐑟 𐑦𐑓 𐑦𐑑 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑩 𐑑𐑧𐑤𐑦𐑕𐑒𐑴𐑐; “𐑞𐑺 𐑲 𐑕𐑷 𐑣𐑦𐑥, 𐑪𐑯 𐑞 ·𐑤𐑲𐑯, 𐑤𐑭𐑕𐑑 𐑕𐑰𐑟𐑩𐑯.”

“𐑯 𐑣𐑰 𐑑𐑫𐑒 𐑞𐑨𐑑 𐑸𐑥 𐑪𐑓, 𐑛𐑦𐑛 𐑣𐑰?” 𐑭𐑕𐑒𐑑 ·𐑱𐑣𐑨𐑚, 𐑯𐑬 𐑕𐑤𐑲𐑛𐑦𐑙 𐑛𐑬𐑯 𐑓𐑮𐑪𐑥 𐑞 𐑒𐑨𐑐𐑕𐑑𐑩𐑯, 𐑯 𐑮𐑧𐑕𐑑𐑦𐑙 𐑪𐑯 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖𐑥𐑩𐑯'𐑟 𐑖𐑴𐑤𐑛𐑼, 𐑨𐑟 𐑣𐑰 𐑛𐑦𐑛 𐑕𐑴.

“𐑲, 𐑣𐑰 𐑢𐑪𐑟 𐑞 𐑒𐑷𐑟 𐑝 𐑦𐑑, 𐑨𐑑 𐑤𐑰𐑕𐑑; 𐑯 𐑞𐑨𐑑 𐑤𐑧𐑜, 𐑑𐑵?”

“𐑕𐑐𐑦𐑯 𐑥𐑰 𐑞 𐑘𐑸𐑯,” 𐑕𐑧𐑛 ·𐑱𐑣𐑨𐑚; “𐑣𐑬 𐑢𐑪𐑟 𐑦𐑑?”

“𐑦𐑑 𐑢𐑪𐑟 𐑞 𐑓𐑻𐑕𐑑 𐑑𐑲𐑥 𐑦𐑯 𐑥𐑲 𐑤𐑲𐑓 𐑞𐑨𐑑 𐑲 𐑧𐑝𐑼 𐑒𐑮𐑵𐑟𐑛 𐑪𐑯 𐑞 ·𐑤𐑲𐑯,” 𐑚𐑦𐑜𐑨𐑯 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖𐑥𐑩𐑯. “𐑲 𐑢𐑪𐑟 𐑦𐑜𐑯𐑼𐑩𐑯𐑑 𐑝 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤 𐑨𐑑 𐑞𐑨𐑑 𐑑𐑲𐑥. 𐑢𐑧𐑤, 𐑢𐑳𐑯 𐑛𐑱 𐑢𐑰 𐑤𐑴𐑼𐑛 𐑓 𐑩 𐑐𐑪𐑛 𐑝 𐑓𐑹 𐑹 𐑓𐑲𐑝 𐑢𐑱𐑤𐑟, 𐑯 𐑥𐑲 𐑚𐑴𐑑 𐑓𐑭𐑕𐑩𐑯𐑛 𐑑 𐑢𐑳𐑯 𐑝 𐑞𐑧𐑥; 𐑩 𐑮𐑧𐑜𐑿𐑤𐑼 𐑕𐑻𐑒𐑩𐑕 𐑣𐑹𐑕 𐑣𐑰 𐑢𐑪𐑟, 𐑑𐑵, 𐑞𐑨𐑑 𐑢𐑧𐑯𐑑 𐑥𐑦𐑤𐑦𐑙 𐑯 𐑥𐑦𐑤𐑦𐑙 𐑮𐑬𐑯𐑛 𐑕𐑴, 𐑞𐑨𐑑 𐑥𐑲 𐑚𐑴𐑑'𐑕 𐑒𐑮𐑵 𐑒𐑫𐑛 𐑴𐑯𐑤𐑦 𐑑𐑮𐑦𐑥 𐑛𐑦𐑖, 𐑚𐑲 𐑕𐑦𐑑𐑦𐑙 𐑷𐑤 𐑞𐑺 𐑕𐑑𐑻𐑯𐑟 𐑪𐑯 𐑞 𐑬𐑑𐑼 𐑜𐑳𐑯𐑩𐑤. 𐑐𐑮𐑧𐑟𐑩𐑯𐑑𐑤𐑦 𐑳𐑐 𐑚𐑮𐑰𐑗𐑩𐑟 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑪𐑑𐑩𐑥 𐑝 𐑞 𐑕𐑰 𐑩 𐑚𐑬𐑯𐑕𐑦𐑙 𐑜𐑮𐑱𐑑 𐑢𐑱𐑤, 𐑢𐑦𐑞 𐑩 𐑥𐑦𐑤𐑒𐑦-𐑢𐑲𐑑 𐑣𐑧𐑛 𐑯 𐑣𐑳𐑥𐑐, 𐑷𐑤 𐑒𐑮𐑴𐑟' 𐑓𐑰𐑑 𐑯 𐑮𐑦𐑙𐑒𐑩𐑤𐑟.”

“𐑦𐑑 𐑢𐑪𐑟 𐑣𐑰, 𐑦𐑑 𐑢𐑪𐑟 𐑣𐑰!” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑤𐑧𐑑𐑦𐑙 𐑬𐑑 𐑣𐑦𐑟 𐑕𐑩𐑕𐑐𐑧𐑯𐑛𐑩𐑛 𐑚𐑮𐑧𐑔.

“𐑯 𐑣𐑸𐑐𐑵𐑯𐑟 𐑕𐑑𐑦𐑒𐑦𐑙 𐑦𐑯 𐑯𐑽 𐑣𐑦𐑟 𐑕𐑑𐑸𐑚𐑼𐑛 𐑓𐑦𐑯.”

“𐑲, 𐑲—𐑞𐑱 𐑢𐑻 𐑥𐑲𐑯—𐑥𐑲 𐑲𐑼𐑯𐑟,” 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑦𐑜𐑟𐑳𐑤𐑑𐑦𐑙𐑤𐑦—“𐑚𐑳𐑑 𐑪𐑯!”

“𐑲, 𐑲 𐑕𐑰!—𐑢𐑪𐑯𐑑𐑩𐑛 𐑑 𐑐𐑸𐑑 𐑦𐑑; 𐑓𐑮𐑰 𐑞 𐑓𐑭𐑕𐑑-𐑓𐑦𐑖—𐑩𐑯 𐑴𐑤𐑛 𐑑𐑮𐑦𐑒—𐑲 𐑯𐑴 𐑣𐑦𐑥.”

“𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑖𐑪𐑒𐑦𐑙 𐑚𐑨𐑛 𐑢𐑵𐑯𐑛,” 𐑚𐑦𐑜𐑨𐑯 𐑞 𐑢𐑱𐑤-𐑕𐑻𐑡𐑩𐑯; “𐑯, 𐑑𐑱𐑒𐑦𐑙 𐑥𐑲 𐑩𐑛𐑝𐑲𐑕, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑵𐑥𐑼 𐑣𐑽, 𐑕𐑑𐑫𐑛 𐑬𐑼 𐑴𐑤𐑛 ·𐑕𐑨𐑥𐑦—”

“·𐑕𐑨𐑥𐑿𐑩𐑤 ·𐑦𐑯𐑛𐑸𐑚𐑦 𐑦𐑟 𐑞 𐑯𐑱𐑥 𐑝 𐑥𐑲 𐑖𐑦𐑐,” 𐑦𐑯𐑑𐑼𐑳𐑐𐑑𐑩𐑛 𐑞 𐑢𐑳𐑯-𐑸𐑥𐑛 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑩𐑛𐑮𐑧𐑕𐑦𐑙 ·𐑱𐑣𐑨𐑚; “𐑜𐑴 𐑪𐑯, 𐑚𐑶.”

“𐑕𐑑𐑫𐑛 𐑬𐑼 𐑴𐑤𐑛 ·𐑕𐑨𐑥𐑦 𐑪𐑓 𐑑 𐑞 𐑯𐑹𐑔𐑢𐑼𐑛, 𐑑 𐑜𐑧𐑑 𐑬𐑑 𐑝 𐑞 𐑚𐑤𐑱𐑟𐑦𐑙 𐑣𐑪𐑑 𐑢𐑧𐑞𐑼 𐑞𐑺 𐑪𐑯 𐑞 ·𐑤𐑲𐑯. 𐑚𐑳𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑯𐑴 𐑿𐑕—𐑲 𐑛𐑦𐑛 𐑷𐑤 𐑲 𐑒𐑫𐑛; 𐑕𐑨𐑑 𐑳𐑐 𐑢𐑦𐑞 𐑣𐑦𐑥 𐑯𐑲𐑑𐑕; 𐑢𐑪𐑟 𐑝𐑧𐑮𐑦 𐑕𐑦𐑝𐑽 𐑢𐑦𐑞 𐑣𐑦𐑥 𐑦𐑯 𐑞 𐑥𐑨𐑑𐑼 𐑝 𐑛𐑲𐑩𐑑—”

“𐑴, 𐑝𐑧𐑮𐑦 𐑕𐑦𐑝𐑽!” 𐑗𐑲𐑥𐑛 𐑦𐑯 𐑞 𐑐𐑱𐑖𐑩𐑯𐑑 𐑣𐑦𐑥𐑕𐑧𐑤𐑓; 𐑞𐑧𐑯 𐑕𐑳𐑛𐑩𐑯𐑤𐑦 𐑷𐑤𐑑𐑼𐑦𐑙 𐑣𐑦𐑟 𐑝𐑶𐑕, “𐑛𐑮𐑦𐑙𐑒𐑦𐑙 𐑣𐑪𐑑 𐑮𐑳𐑥 𐑑𐑪𐑛𐑦𐑟 𐑢𐑦𐑞 𐑥𐑰 𐑧𐑝𐑮𐑦 𐑯𐑲𐑑, 𐑑𐑦𐑤 𐑣𐑰 𐑒𐑫𐑛𐑩𐑯𐑑 𐑕𐑰 𐑑 𐑐𐑫𐑑 𐑪𐑯 𐑞 𐑚𐑨𐑯𐑛𐑦𐑡𐑩𐑟; 𐑯 𐑕𐑧𐑯𐑛𐑦𐑙 𐑥𐑰 𐑑 𐑚𐑧𐑛, 𐑣𐑭𐑓 𐑕𐑰𐑟 𐑴𐑝𐑼, 𐑩𐑚𐑬𐑑 𐑔𐑮𐑰 𐑴'𐑒𐑤𐑪𐑒 𐑦𐑯 𐑞 𐑥𐑹𐑯𐑦𐑙. 𐑴, 𐑘𐑰 𐑕𐑑𐑸𐑟! 𐑣𐑰 𐑕𐑨𐑑 𐑳𐑐 𐑢𐑦𐑞 𐑥𐑰 𐑦𐑯𐑛𐑰𐑛, 𐑯 𐑢𐑪𐑟 𐑝𐑧𐑮𐑦 𐑕𐑦𐑝𐑽 𐑦𐑯 𐑥𐑲 𐑛𐑲𐑩𐑑. 𐑴! 𐑩 𐑜𐑮𐑱𐑑 𐑢𐑪𐑗𐑼, 𐑯 𐑝𐑧𐑮𐑦 𐑛𐑲𐑩𐑑𐑧𐑑𐑦𐑒𐑤𐑦 𐑕𐑦𐑝𐑽, 𐑦𐑟 ·𐑛𐑼 ·𐑚𐑳𐑙𐑼. (𐑚𐑳𐑙𐑼, 𐑿 𐑛𐑪𐑜, 𐑤𐑭𐑓 𐑬𐑑! 𐑢𐑲 𐑛𐑴𐑯𐑑 𐑘𐑰? 𐑿 𐑯𐑴 𐑿'𐑼 𐑩 𐑐𐑮𐑧𐑖𐑩𐑕 𐑡𐑪𐑤𐑦 𐑮𐑭𐑕𐑒𐑩𐑤.) 𐑚𐑳𐑑, 𐑣𐑰𐑝 𐑩𐑣𐑧𐑛, 𐑚𐑶, 𐑲'𐑛 𐑮𐑭𐑞𐑼 𐑚𐑰 𐑒𐑦𐑤𐑛 𐑚𐑲 𐑿 𐑞𐑨𐑯 𐑒𐑧𐑐𐑑 𐑩𐑤𐑲𐑝 𐑚𐑲 𐑧𐑯𐑦 𐑳𐑞𐑼 𐑥𐑨𐑯.”

“𐑥𐑲 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑿 𐑥𐑳𐑕𐑑 𐑣𐑨𐑝 𐑺 𐑞𐑦𐑕 𐑐𐑼𐑕𐑰𐑝𐑛, 𐑮𐑦𐑕𐑐𐑧𐑒𐑑𐑩𐑛 𐑕𐑻”—𐑕𐑧𐑛 𐑞 𐑦𐑥𐑐𐑼𐑑𐑻𐑚𐑩𐑚𐑩𐑤 𐑜𐑪𐑛𐑤𐑦-𐑤𐑫𐑒𐑦𐑙 ·𐑚𐑳𐑙𐑼, 𐑕𐑤𐑲𐑑𐑤𐑦 𐑚𐑬𐑦𐑙 𐑑 ·𐑱𐑣𐑨𐑚—“𐑦𐑟 𐑨𐑐𐑑 𐑑 𐑚𐑰 𐑓𐑩𐑕𐑰𐑖𐑩𐑕 𐑨𐑑 𐑑𐑲𐑥𐑟; 𐑣𐑰 𐑕𐑐𐑦𐑯𐑟 𐑳𐑕 𐑥𐑧𐑯𐑦 𐑒𐑤𐑧𐑝𐑼 𐑔𐑦𐑙𐑟 𐑝 𐑞𐑨𐑑 𐑕𐑹𐑑. 𐑚𐑳𐑑 𐑲 𐑥𐑱 𐑨𐑟 𐑢𐑧𐑤 𐑕𐑱—𐑧𐑯 𐑐𐑭𐑕𐑩𐑯𐑑, 𐑨𐑟 𐑞 ·𐑓𐑮𐑧𐑯𐑗 𐑮𐑦𐑥𐑸𐑒—𐑞𐑨𐑑 𐑲 𐑥𐑲𐑕𐑧𐑤𐑓—𐑞𐑨𐑑 𐑦𐑟 𐑑 𐑕𐑱, ·𐑡𐑨𐑒 ·𐑚𐑳𐑙𐑼, 𐑤𐑱𐑑 𐑝 𐑞 𐑮𐑧𐑝𐑼𐑩𐑯𐑛 𐑒𐑤𐑻𐑡𐑦—𐑨𐑥 𐑩 𐑕𐑑𐑮𐑦𐑒𐑑 𐑑𐑴𐑑𐑩𐑤 𐑨𐑚𐑕𐑑𐑦𐑯𐑩𐑯𐑕 𐑥𐑨𐑯; 𐑲 𐑯𐑧𐑝𐑼 𐑛𐑮𐑦𐑙𐑒—”

“𐑢𐑷𐑑𐑼!” 𐑒𐑮𐑲𐑛 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯; “𐑣𐑰 𐑯𐑧𐑝𐑼 𐑛𐑮𐑦𐑙𐑒𐑕 𐑦𐑑; 𐑦𐑑'𐑕 𐑩 𐑕𐑹𐑑 𐑝 𐑓𐑦𐑑𐑕 𐑑 𐑣𐑦𐑥; 𐑓𐑮𐑧𐑖 𐑢𐑷𐑑𐑼 𐑔𐑮𐑴𐑟 𐑣𐑦𐑥 𐑦𐑯𐑑𐑵 𐑞 𐑣𐑲𐑛𐑮𐑴𐑓𐑴𐑚𐑾; 𐑚𐑳𐑑 𐑜𐑴 𐑪𐑯—𐑜𐑴 𐑪𐑯 𐑢𐑦𐑞 𐑞 𐑸𐑥 𐑕𐑑𐑹𐑦.”

“𐑘𐑧𐑕, 𐑲 𐑥𐑱 𐑨𐑟 𐑢𐑧𐑤,” 𐑕𐑧𐑛 𐑞 𐑕𐑻𐑡𐑩𐑯, 𐑒𐑵𐑤𐑦. “𐑲 𐑢𐑪𐑟 𐑩𐑚𐑬𐑑 𐑩𐑚𐑟𐑻𐑝𐑦𐑙, 𐑕𐑻, 𐑚𐑦𐑓𐑹 ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑵𐑥𐑼'𐑟 𐑓𐑩𐑕𐑰𐑖𐑩𐑕 𐑦𐑯𐑑𐑼𐑳𐑐𐑖𐑩𐑯, 𐑞𐑨𐑑 𐑕𐑐𐑲𐑑 𐑝 𐑥𐑲 𐑚𐑧𐑕𐑑 𐑯 𐑕𐑦𐑝𐑽𐑩𐑕𐑑 𐑦𐑯𐑛𐑧𐑝𐑼𐑟, 𐑞 𐑢𐑵𐑯𐑛 𐑒𐑧𐑐𐑑 𐑜𐑧𐑑𐑦𐑙 𐑢𐑻𐑕 𐑯 𐑢𐑻𐑕; 𐑞 𐑑𐑮𐑵𐑔 𐑢𐑪𐑟, 𐑕𐑻, 𐑦𐑑 𐑢𐑪𐑟 𐑨𐑟 𐑳𐑜𐑤𐑦 𐑜𐑱𐑐𐑦𐑙 𐑢𐑵𐑯𐑛 𐑨𐑟 𐑕𐑻𐑡𐑩𐑯 𐑧𐑝𐑼 𐑕𐑷; 𐑥𐑹 𐑞𐑨𐑯 𐑑𐑵 𐑓𐑰𐑑 𐑯 𐑕𐑧𐑝𐑼𐑩𐑤 𐑦𐑯𐑗𐑩𐑟 𐑤𐑪𐑙. 𐑲 𐑥𐑧𐑠𐑼𐑛 𐑦𐑑 𐑢𐑦𐑞 𐑞 𐑤𐑧𐑛 𐑤𐑲𐑯. 𐑦𐑯 𐑖𐑹𐑑, 𐑦𐑑 𐑜𐑮𐑵 𐑚𐑤𐑨𐑒; 𐑲 𐑯𐑿 𐑢𐑪𐑑 𐑢𐑪𐑟 𐑔𐑮𐑧𐑑𐑩𐑯𐑛, 𐑯 𐑪𐑓 𐑦𐑑 𐑒𐑱𐑥. 𐑚𐑳𐑑 𐑲 𐑣𐑨𐑛 𐑯𐑴 𐑣𐑨𐑯𐑛 𐑦𐑯 𐑖𐑦𐑐𐑦𐑙 𐑞𐑨𐑑 𐑲𐑝𐑼𐑦 𐑸𐑥 𐑞𐑺; 𐑞𐑨𐑑 𐑔𐑦𐑙 𐑦𐑟 𐑩𐑜𐑱𐑯𐑕𐑑 𐑷𐑤 𐑮𐑵𐑤”—𐑐𐑶𐑯𐑑𐑦𐑙 𐑨𐑑 𐑦𐑑 𐑢𐑦𐑞 𐑞 𐑯𐑸𐑤𐑦𐑣𐑕𐑐𐑲𐑒—“𐑞𐑨𐑑 𐑦𐑟 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯'𐑟 𐑢𐑻𐑒, 𐑯𐑪𐑑 𐑥𐑲𐑯; 𐑣𐑰 𐑹𐑛𐑼𐑛 𐑞 𐑒𐑸𐑐𐑩𐑯𐑑𐑼 𐑑 𐑥𐑱𐑒 𐑦𐑑; 𐑣𐑰 𐑣𐑨𐑛 𐑞𐑨𐑑 𐑒𐑤𐑳𐑚-𐑣𐑨𐑥𐑼 𐑞𐑺 𐑐𐑫𐑑 𐑑 𐑞 𐑧𐑯𐑛, 𐑑 𐑯𐑪𐑒 𐑕𐑳𐑥 𐑢𐑳𐑯'𐑟 𐑚𐑮𐑱𐑯𐑟 𐑬𐑑 𐑢𐑦𐑞, 𐑲 𐑕𐑩𐑐𐑴𐑟, 𐑨𐑟 𐑣𐑰 𐑑𐑮𐑲𐑛 𐑥𐑲𐑯 𐑢𐑳𐑯𐑕. 𐑣𐑰 𐑓𐑤𐑲𐑟 𐑦𐑯𐑑𐑵 𐑛𐑲𐑩𐑚𐑪𐑤𐑦𐑒𐑩𐑤 𐑐𐑭𐑖𐑩𐑯𐑟 𐑕𐑳𐑥𐑑𐑲𐑥𐑟. 𐑛𐑵 𐑘𐑰 𐑕𐑰 𐑞𐑦𐑕 𐑛𐑧𐑯𐑑, 𐑕𐑻”—𐑮𐑦𐑥𐑵𐑝𐑦𐑙 𐑣𐑦𐑟 𐑣𐑨𐑑, 𐑯 𐑚𐑮𐑳𐑖𐑦𐑙 𐑩𐑕𐑲𐑛 𐑣𐑦𐑟 𐑣𐑺, 𐑯 𐑦𐑒𐑕𐑐𐑴𐑟𐑦𐑙 𐑩 𐑚𐑴𐑤-𐑤𐑲𐑒 𐑒𐑨𐑝𐑦𐑑𐑦 𐑦𐑯 𐑣𐑦𐑟 𐑕𐑒𐑳𐑤, 𐑚𐑳𐑑 𐑢𐑦𐑗 𐑚𐑹 𐑯𐑪𐑑 𐑞 𐑕𐑤𐑲𐑑𐑩𐑕𐑑 𐑕𐑒𐑸𐑦 𐑑𐑮𐑱𐑕, 𐑹 𐑧𐑯𐑦 𐑑𐑴𐑒𐑩𐑯 𐑝 𐑧𐑝𐑼 𐑣𐑨𐑝𐑦𐑙 𐑚𐑰𐑯 𐑩 𐑢𐑵𐑯𐑛—“𐑢𐑧𐑤, 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯 𐑞𐑺 𐑢𐑦𐑤 𐑑𐑧𐑤 𐑿 𐑣𐑬 𐑞𐑨𐑑 𐑒𐑱𐑥 𐑣𐑽; 𐑣𐑰 𐑯𐑴𐑟.”

“𐑯𐑴, 𐑲 𐑛𐑴𐑯𐑑,” 𐑕𐑧𐑛 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯, “𐑚𐑳𐑑 𐑣𐑦𐑟 𐑥𐑳𐑞𐑼 𐑛𐑦𐑛; 𐑣𐑰 𐑢𐑪𐑟 𐑚𐑹𐑯 𐑢𐑦𐑞 𐑦𐑑. 𐑴, 𐑿 𐑕𐑪𐑤𐑩𐑥 𐑮𐑴𐑜, 𐑿—𐑿 ·𐑚𐑳𐑙𐑼! 𐑢𐑪𐑟 𐑞𐑺 𐑧𐑝𐑼 𐑕𐑳𐑗 𐑩𐑯𐑳𐑞𐑼 ·𐑚𐑳𐑙𐑼 𐑦𐑯 𐑞 𐑢𐑷𐑑𐑼𐑦 𐑢𐑻𐑤𐑛? 𐑚𐑳𐑙𐑼, 𐑢𐑧𐑯 𐑿 𐑛𐑲, 𐑿 𐑷𐑑 𐑑 𐑛𐑲 𐑦𐑯 𐑐𐑦𐑒𐑩𐑤, 𐑿 𐑛𐑪𐑜; 𐑿 𐑖𐑵𐑛 𐑚𐑰 𐑐𐑮𐑦𐑟𐑻𐑝𐑛 𐑑 𐑓𐑿𐑗𐑼 𐑱𐑡𐑩𐑟, 𐑿 𐑮𐑭𐑕𐑒𐑩𐑤.”

“𐑢𐑪𐑑 𐑚𐑦𐑒𐑱𐑥 𐑝 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?” 𐑯𐑬 𐑒𐑮𐑲𐑛 ·𐑱𐑣𐑨𐑚, 𐑣𐑵 𐑞𐑳𐑕 𐑓𐑸 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑦𐑥𐑐𐑱𐑖𐑩𐑯𐑑𐑤𐑦 𐑤𐑦𐑕𐑩𐑯𐑦𐑙 𐑑 𐑞𐑦𐑕 𐑚𐑲-𐑐𐑤𐑱 𐑚𐑦𐑑𐑢𐑰𐑯 𐑞 𐑑𐑵 ·𐑦𐑙𐑜𐑤𐑦𐑖𐑥𐑧𐑯.

“𐑴!” 𐑒𐑮𐑲𐑛 𐑞 𐑢𐑳𐑯-𐑸𐑥𐑛 𐑒𐑨𐑐𐑑𐑦𐑯, “𐑴, 𐑘𐑧𐑕! 𐑢𐑧𐑤; 𐑭𐑓𐑑𐑼 𐑣𐑰 𐑕𐑬𐑯𐑛𐑩𐑛, 𐑢𐑰 𐑛𐑦𐑛𐑩𐑯𐑑 𐑕𐑰 𐑣𐑦𐑥 𐑩𐑜𐑱𐑯 𐑓 𐑕𐑳𐑥 𐑑𐑲𐑥; 𐑦𐑯 𐑓𐑨𐑒𐑑, 𐑨𐑟 𐑲 𐑚𐑦𐑓𐑹 𐑣𐑦𐑯𐑑𐑩𐑛, 𐑲 𐑛𐑦𐑛𐑩𐑯𐑑 𐑞𐑧𐑯 𐑯𐑴 𐑢𐑪𐑑 𐑢𐑱𐑤 𐑦𐑑 𐑢𐑪𐑟 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑕𐑻𐑝𐑛 𐑥𐑰 𐑕𐑳𐑗 𐑩 𐑑𐑮𐑦𐑒, 𐑑𐑦𐑤 𐑕𐑳𐑥 𐑑𐑲𐑥 𐑭𐑓𐑑𐑼𐑢𐑼𐑛𐑟, 𐑢𐑧𐑯 𐑒𐑳𐑥𐑦𐑙 𐑚𐑨𐑒 𐑑 𐑞 ·𐑤𐑲𐑯, 𐑢𐑰 𐑣𐑻𐑛 𐑩𐑚𐑬𐑑 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒—𐑨𐑟 𐑕𐑳𐑥 𐑒𐑷𐑤 𐑣𐑦𐑥—𐑯 𐑞𐑧𐑯 𐑲 𐑯𐑿 𐑦𐑑 𐑢𐑪𐑟 𐑣𐑰.”

“𐑛𐑲𐑛'𐑕𐑑 𐑞𐑬 𐑒𐑮𐑪𐑕 𐑣𐑦𐑟 𐑢𐑱𐑒 𐑩𐑜𐑱𐑯?”

“𐑑𐑢𐑲𐑕.”

“𐑚𐑳𐑑 𐑒𐑫𐑛 𐑯𐑪𐑑 𐑓𐑭𐑕𐑩𐑯?”

“𐑛𐑦𐑛𐑩𐑯𐑑 𐑢𐑪𐑯𐑑 𐑑 𐑑𐑮𐑲 𐑑: 𐑱𐑯𐑑 𐑢𐑳𐑯 𐑤𐑦𐑥 𐑦𐑯𐑳𐑓? 𐑢𐑪𐑑 𐑖𐑵𐑛 𐑲 𐑛𐑵 𐑢𐑦𐑞𐑬𐑑 𐑞𐑦𐑕 𐑳𐑞𐑼 𐑸𐑥? 𐑯 𐑲'𐑥 𐑔𐑦𐑙𐑒𐑦𐑙 ·𐑥𐑪𐑚𐑦 ·𐑛𐑦𐑒 𐑛𐑳𐑟𐑩𐑯𐑑 𐑚𐑲𐑑 𐑕𐑴 𐑥𐑳𐑗 𐑨𐑟 𐑣𐑰 𐑕𐑢𐑪𐑤𐑴𐑟.”

“𐑢𐑧𐑤, 𐑞𐑧𐑯,” 𐑦𐑯𐑑𐑼𐑳𐑐𐑑𐑩𐑛 ·𐑚𐑳𐑙𐑼, “𐑜𐑦𐑝 𐑣𐑦𐑥 𐑘𐑹 𐑤𐑧𐑓𐑑 𐑸𐑥 𐑓 𐑚𐑱𐑑 𐑑 𐑜𐑧𐑑 𐑞 𐑮𐑲𐑑. 𐑛𐑵 𐑿 𐑯𐑴, 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑧𐑯”—𐑝𐑧𐑮𐑦 𐑜𐑮𐑱𐑝𐑤𐑦 𐑯 𐑥𐑨𐑔𐑩𐑥𐑨𐑑𐑦𐑒𐑤𐑦 𐑚𐑬𐑦𐑙 𐑑 𐑰𐑗 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑦𐑯 𐑕𐑩𐑒𐑕𐑧𐑖𐑩𐑯—“𐑛𐑵 𐑿 𐑯𐑴, 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑧𐑯, 𐑞𐑨𐑑 𐑞 𐑛𐑲𐑡𐑧𐑕𐑑𐑦𐑝 𐑹𐑜𐑩𐑯𐑟 𐑝 𐑞 𐑢𐑱𐑤 𐑸 𐑕𐑴 𐑦𐑯𐑕𐑒𐑮𐑵𐑑𐑩𐑚𐑤𐑦 𐑒𐑩𐑯𐑕𐑑𐑮𐑳𐑒𐑑𐑩𐑛 𐑚𐑲 ·𐑛𐑦𐑝𐑲𐑯 ·𐑐𐑮𐑪𐑝𐑦𐑛𐑩𐑯𐑕, 𐑞𐑨𐑑 𐑦𐑑 𐑦𐑟 𐑒𐑢𐑲𐑑 𐑦𐑥𐑐𐑪𐑕𐑩𐑚𐑩𐑤 𐑓 𐑣𐑦𐑥 𐑑 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑛𐑲𐑡𐑧𐑕𐑑 𐑰𐑝𐑩𐑯 𐑩 𐑥𐑨𐑯'𐑟 𐑸𐑥? 𐑯 𐑣𐑰 𐑯𐑴𐑟 𐑦𐑑 𐑑𐑵. 𐑕𐑴 𐑞𐑨𐑑 𐑢𐑪𐑑 𐑿 𐑑𐑱𐑒 𐑓 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑥𐑨𐑤𐑦𐑕 𐑦𐑟 𐑴𐑯𐑤𐑦 𐑣𐑦𐑟 𐑷𐑒𐑢𐑼𐑛𐑯𐑩𐑕. 𐑓 𐑣𐑰 𐑯𐑧𐑝𐑼 𐑥𐑰𐑯𐑟 𐑑 𐑕𐑢𐑪𐑤𐑴 𐑩 𐑕𐑦𐑙𐑜𐑩𐑤 𐑤𐑦𐑥; 𐑣𐑰 𐑴𐑯𐑤𐑦 𐑔𐑦𐑙𐑒𐑕 𐑑 𐑑𐑧𐑮𐑦𐑓𐑲 𐑚𐑲 𐑓𐑱𐑯𐑑𐑕. 𐑚𐑳𐑑 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑣𐑰 𐑦𐑟 𐑤𐑲𐑒 𐑞 𐑴𐑤𐑛 𐑡𐑳𐑜𐑩𐑤𐑦𐑙 𐑓𐑧𐑤𐑴, 𐑓𐑹𐑥𐑼𐑤𐑦 𐑩 𐑐𐑱𐑖𐑩𐑯𐑑 𐑝 𐑥𐑲𐑯 𐑦𐑯 ·𐑕𐑱𐑤𐑪𐑯, 𐑞𐑨𐑑 𐑥𐑱𐑒𐑦𐑙 𐑚𐑦𐑤𐑰𐑝 𐑕𐑢𐑪𐑤𐑴 𐑡𐑨𐑒-𐑯𐑲𐑝𐑟, 𐑢𐑳𐑯𐑕 𐑩𐑐𐑪𐑯 𐑩 𐑑𐑲𐑥 𐑤𐑧𐑑 𐑢𐑳𐑯 𐑛𐑮𐑪𐑐 𐑦𐑯𐑑𐑵 𐑣𐑦𐑥 𐑦𐑯 𐑜𐑫𐑛 𐑻𐑯𐑩𐑕𐑑, 𐑯 𐑞𐑺 𐑦𐑑 𐑕𐑑𐑱𐑛 𐑓 𐑩 𐑑𐑢𐑧𐑤𐑝𐑥𐑳𐑯𐑔 𐑹 𐑥𐑹; 𐑢𐑧𐑯 𐑲 𐑜𐑱𐑝 𐑣𐑦𐑥 𐑩𐑯 𐑦𐑥𐑧𐑑𐑦𐑒, 𐑯 𐑣𐑰 𐑣𐑰𐑝𐑛 𐑦𐑑 𐑳𐑐 𐑦𐑯 𐑕𐑥𐑷𐑤 𐑑𐑨𐑒𐑕, 𐑛'𐑘𐑰 𐑕𐑰. 𐑯𐑴 𐑐𐑪𐑕𐑩𐑚𐑩𐑤 𐑢𐑱 𐑓 𐑣𐑦𐑥 𐑑 𐑛𐑲𐑡𐑧𐑕𐑑 𐑞𐑨𐑑 𐑡𐑨𐑒-𐑯𐑲𐑓, 𐑯 𐑓𐑫𐑤𐑦 𐑦𐑯𐑒𐑹𐑐𐑼𐑱𐑑 𐑦𐑑 𐑦𐑯𐑑𐑵 𐑣𐑦𐑟 𐑡𐑧𐑯𐑼𐑩𐑤 𐑚𐑪𐑛𐑦𐑤𐑦 𐑕𐑦𐑕𐑑𐑩𐑥. 𐑘𐑧𐑕, ·𐑒𐑨𐑐𐑑𐑦𐑯 ·𐑚𐑵𐑥𐑼, 𐑦𐑓 𐑿 𐑸 𐑒𐑢𐑦𐑒 𐑦𐑯𐑳𐑓 𐑩𐑚𐑬𐑑 𐑦𐑑, 𐑯 𐑣𐑨𐑝 𐑩 𐑥𐑲𐑯𐑛 𐑑 𐑐𐑷𐑯 𐑢𐑳𐑯 𐑸𐑥 𐑓 𐑞 𐑕𐑱𐑒 𐑝 𐑞 𐑐𐑮𐑦𐑝𐑩𐑤𐑦𐑡 𐑝 𐑜𐑦𐑝𐑦𐑙 𐑛𐑰𐑕𐑩𐑯𐑑 𐑚𐑧𐑮𐑾𐑤 𐑑 𐑞 𐑳𐑞𐑼, 𐑢𐑲 𐑦𐑯 𐑞𐑨𐑑 𐑒𐑱𐑕 𐑞 𐑸𐑥 𐑦𐑟 𐑘𐑹𐑟; 𐑴𐑯𐑤𐑦 𐑤𐑧𐑑 𐑞 𐑢𐑱𐑤 𐑣𐑨𐑝 𐑩𐑯𐑳𐑞𐑼 𐑗𐑭𐑯𐑕 𐑨𐑑 𐑿 𐑖𐑹𐑑𐑤𐑦, 𐑞𐑨𐑑'𐑕 𐑷𐑤.”

“𐑯𐑴, 𐑔𐑨𐑙𐑒 𐑘𐑰, ·𐑚𐑳𐑙𐑼,” 𐑕𐑧𐑛 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖 ·𐑒𐑨𐑐𐑑𐑦𐑯, “𐑣𐑰'𐑟 𐑢𐑧𐑤𐑒𐑩𐑥 𐑑 𐑞 𐑸𐑥 𐑣𐑰 𐑣𐑨𐑟, 𐑕𐑦𐑯𐑕 𐑲 𐑒𐑭𐑯𐑑 𐑣𐑧𐑤𐑐 𐑦𐑑, 𐑯 𐑛𐑦𐑛𐑩𐑯𐑑 𐑯𐑴 𐑣𐑦𐑥 𐑞𐑧𐑯; 𐑚𐑳𐑑 𐑯𐑪𐑑 𐑑 𐑩𐑯𐑳𐑞𐑼 𐑢𐑳𐑯. 𐑯𐑴 𐑥𐑹 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤𐑟 𐑓 𐑥𐑰; 𐑲'𐑝 𐑤𐑴𐑼𐑛 𐑓 𐑣𐑦𐑥 𐑢𐑳𐑯𐑕, 𐑯 𐑞𐑨𐑑 𐑣𐑨𐑟 𐑕𐑨𐑑𐑦𐑕𐑓𐑲𐑛 𐑥𐑰. 𐑞𐑺 𐑢𐑫𐑛 𐑚𐑰 𐑜𐑮𐑱𐑑 𐑜𐑤𐑹𐑦 𐑦𐑯 𐑒𐑦𐑤𐑦𐑙 𐑣𐑦𐑥, 𐑲 𐑯𐑴 𐑞𐑨𐑑; 𐑯 𐑞𐑺 𐑦𐑟 𐑩 𐑖𐑦𐑐-𐑤𐑴𐑛 𐑝 𐑐𐑮𐑧𐑖𐑩𐑕 𐑕𐑐𐑻𐑥 𐑦𐑯 𐑣𐑦𐑥, 𐑚𐑳𐑑, 𐑣𐑸𐑒 𐑘𐑰, 𐑣𐑰'𐑟 𐑚𐑧𐑕𐑑 𐑤𐑧𐑑 𐑩𐑤𐑴𐑯; 𐑛𐑴𐑯𐑑 𐑿 𐑔𐑦𐑙𐑒 𐑕𐑴, ·𐑒𐑨𐑐𐑑𐑦𐑯?”—𐑜𐑤𐑭𐑯𐑕𐑦𐑙 𐑨𐑑 𐑞 𐑲𐑝𐑼𐑦 𐑤𐑧𐑜.

“𐑣𐑰 𐑦𐑟. 𐑚𐑳𐑑 𐑣𐑰 𐑢𐑦𐑤 𐑕𐑑𐑦𐑤 𐑚𐑰 𐑣𐑳𐑯𐑑𐑩𐑛, 𐑓 𐑷𐑤 𐑞𐑨𐑑. 𐑢𐑪𐑑 𐑦𐑟 𐑚𐑧𐑕𐑑 𐑤𐑧𐑑 𐑩𐑤𐑴𐑯, 𐑞𐑨𐑑 𐑩𐑒𐑻𐑕𐑩𐑛 𐑔𐑦𐑙 𐑦𐑟 𐑯𐑪𐑑 𐑷𐑤𐑢𐑱𐑟 𐑢𐑪𐑑 𐑤𐑰𐑕𐑑 𐑩𐑤𐑫𐑼𐑟. 𐑣𐑰'𐑟 𐑷𐑤 𐑩 𐑥𐑨𐑜𐑯𐑩𐑑! 𐑣𐑬 𐑤𐑪𐑙 𐑕𐑦𐑯𐑕 𐑞𐑬 𐑕𐑷'𐑕𐑑 𐑣𐑦𐑥 𐑤𐑭𐑕𐑑? 𐑢𐑦𐑗 𐑢𐑱 𐑣𐑧𐑛𐑦𐑙?”

“𐑚𐑤𐑧𐑕 𐑥𐑲 𐑕𐑴𐑤, 𐑯 𐑒𐑻𐑕 𐑞 𐑓𐑬𐑤 𐑓𐑰𐑯𐑛'𐑟,” 𐑒𐑮𐑲𐑛 ·𐑚𐑳𐑙𐑼, 𐑕𐑑𐑵𐑐𐑦𐑙𐑤𐑦 𐑢𐑷𐑒𐑦𐑙 𐑮𐑬𐑯𐑛 ·𐑱𐑣𐑨𐑚, 𐑯 𐑤𐑲𐑒 𐑩 𐑛𐑪𐑜, 𐑕𐑑𐑮𐑱𐑯𐑡𐑤𐑦 𐑕𐑯𐑳𐑓𐑦𐑙; “𐑞𐑦𐑕 𐑥𐑨𐑯'𐑟 𐑚𐑤𐑳𐑛—𐑚𐑮𐑦𐑙 𐑞 𐑔𐑼𐑥𐑪𐑥𐑦𐑑𐑼!—𐑦𐑑'𐑕 𐑨𐑑 𐑞 𐑚𐑶𐑤𐑦𐑙 𐑐𐑶𐑯𐑑!—𐑣𐑦𐑟 𐑐𐑳𐑤𐑕 𐑥𐑱𐑒𐑕 𐑞𐑰𐑟 𐑐𐑤𐑨𐑙𐑒𐑕 𐑚𐑰𐑑!—𐑕𐑻!”—𐑑𐑱𐑒𐑦𐑙 𐑩 𐑤𐑭𐑯𐑕𐑦𐑑 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑐𐑪𐑒𐑩𐑑, 𐑯 𐑛𐑮𐑷𐑦𐑙 𐑯𐑽 𐑑 ·𐑱𐑣𐑨𐑚'𐑟 𐑸𐑥.

“𐑩𐑝𐑭𐑕𐑑!” 𐑮𐑹𐑛 ·𐑱𐑣𐑨𐑚, 𐑛𐑨𐑖𐑦𐑙 𐑣𐑦𐑥 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕—“𐑥𐑨𐑯 𐑞 𐑚𐑴𐑑! 𐑢𐑦𐑗 𐑢𐑱 𐑣𐑧𐑛𐑦𐑙?”

“𐑜𐑫𐑛 ·𐑜𐑪𐑛!” 𐑒𐑮𐑲𐑛 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖 ·𐑒𐑨𐑐𐑑𐑦𐑯, 𐑑 𐑣𐑵𐑥 𐑞 𐑒𐑢𐑧𐑕𐑗𐑩𐑯 𐑢𐑪𐑟 𐑐𐑫𐑑. “𐑢𐑪𐑑'𐑕 𐑞 𐑥𐑨𐑑𐑼? 𐑣𐑰 𐑢𐑪𐑟 𐑣𐑧𐑛𐑦𐑙 𐑰𐑕𐑑, 𐑲 𐑔𐑦𐑙𐑒.—𐑦𐑟 𐑘𐑹 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑒𐑮𐑱𐑟𐑦?” 𐑢𐑦𐑕𐑐𐑼𐑦𐑙 ·𐑓𐑧𐑛𐑩𐑤𐑩.

𐑚𐑳𐑑 ·𐑓𐑧𐑛𐑩𐑤𐑩, 𐑐𐑫𐑑𐑦𐑙 𐑩 𐑓𐑦𐑙𐑜𐑼 𐑪𐑯 𐑣𐑦𐑟 𐑤𐑦𐑐, 𐑕𐑤𐑦𐑛 𐑴𐑝𐑼 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕 𐑑 𐑑𐑱𐑒 𐑞 𐑚𐑴𐑑'𐑕 𐑕𐑑𐑽𐑦𐑙 𐑹, 𐑯 ·𐑱𐑣𐑨𐑚, 𐑕𐑢𐑦𐑙𐑦𐑙 𐑞 𐑒𐑳𐑑𐑦𐑙-𐑑𐑨𐑒𐑩𐑤 𐑑𐑩𐑢𐑹𐑛𐑟 𐑣𐑦𐑥, 𐑒𐑩𐑥𐑭𐑯𐑛𐑩𐑛 𐑞 𐑖𐑦𐑐'𐑕 𐑕𐑱𐑤𐑼𐑟 𐑑 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑤𐑴𐑼.

𐑦𐑯 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑣𐑰 𐑢𐑪𐑟 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑞 𐑚𐑴𐑑'𐑕 𐑕𐑑𐑻𐑯, 𐑯 𐑞 ·𐑥𐑩𐑯𐑦𐑤𐑩 𐑥𐑧𐑯 𐑢𐑻 𐑕𐑐𐑮𐑦𐑙𐑦𐑙 𐑑 𐑞𐑺 𐑹𐑟. 𐑦𐑯 𐑝𐑱𐑯 𐑞 ·𐑦𐑙𐑜𐑤𐑦𐑖 ·𐑒𐑨𐑐𐑑𐑦𐑯 𐑣𐑱𐑤𐑛 𐑣𐑦𐑥. 𐑢𐑦𐑞 𐑚𐑨𐑒 𐑑 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑖𐑦𐑐, 𐑯 𐑓𐑱𐑕 𐑕𐑧𐑑 𐑤𐑲𐑒 𐑩 𐑓𐑤𐑦𐑯𐑑 𐑑 𐑣𐑦𐑟 𐑴𐑯, ·𐑱𐑣𐑨𐑚 𐑕𐑑𐑫𐑛 𐑳𐑐𐑮𐑲𐑑 𐑑𐑦𐑤 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑝 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛.
