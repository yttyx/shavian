·𐑗𐑨𐑐𐑑𐑼 98. 𐑕𐑑𐑴𐑦𐑙 ·𐑛𐑬𐑯 𐑯 ·𐑒𐑤𐑽𐑦𐑙 𐑳𐑐.

𐑷𐑤𐑮𐑧𐑛𐑦 𐑣𐑨𐑟 𐑦𐑑 𐑚𐑰𐑯 𐑮𐑦𐑤𐑱𐑑𐑩𐑛 𐑣𐑬 𐑞 𐑜𐑮𐑱𐑑 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑦𐑟 𐑩𐑓𐑸 𐑪𐑓 𐑛𐑦𐑕𐑒𐑮𐑲𐑛 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛; 𐑣𐑬 𐑣𐑰 𐑦𐑟 𐑗𐑱𐑕𐑑 𐑴𐑝𐑼 𐑞 𐑢𐑷𐑑𐑼𐑦 𐑥𐑫𐑼𐑟, 𐑯 𐑕𐑤𐑷𐑑𐑼𐑛 𐑦𐑯 𐑞 𐑝𐑨𐑤𐑦𐑟 𐑝 𐑞 𐑛𐑰𐑐; 𐑣𐑬 𐑣𐑰 𐑦𐑟 𐑞𐑧𐑯 𐑑𐑴𐑛 𐑩𐑤𐑪𐑙𐑕𐑲𐑛 𐑯 𐑚𐑦𐑣𐑧𐑛𐑩𐑛; 𐑯 𐑣𐑬 (𐑪𐑯 𐑞 𐑐𐑮𐑦𐑯𐑕𐑩𐑐𐑩𐑤 𐑢𐑦𐑗 𐑦𐑯𐑑𐑲𐑑𐑩𐑤𐑛 𐑞 𐑣𐑧𐑛𐑟𐑥𐑩𐑯 𐑝 𐑴𐑤𐑛 𐑑 𐑞 𐑜𐑸𐑥𐑩𐑯𐑑𐑕 𐑦𐑯 𐑢𐑦𐑗 𐑞 𐑚𐑦𐑣𐑧𐑛𐑩𐑛 𐑢𐑪𐑟 𐑒𐑦𐑤𐑛) 𐑣𐑦𐑟 𐑜𐑮𐑱𐑑 𐑐𐑨𐑛𐑩𐑛 𐑕𐑻𐑑𐑬𐑑 𐑚𐑦𐑒𐑳𐑥𐑟 𐑞 𐑐𐑮𐑪𐑐𐑼𐑑𐑦 𐑝 𐑣𐑦𐑟 𐑧𐑒𐑕𐑦𐑒𐑿𐑖𐑩𐑯𐑼; 𐑣𐑬, 𐑦𐑯 𐑛𐑿 𐑑𐑲𐑥, 𐑣𐑰 𐑦𐑟 𐑒𐑩𐑯𐑛𐑧𐑥𐑛 𐑑 𐑞 𐑐𐑪𐑑𐑕, 𐑯, 𐑤𐑲𐑒 ·𐑖𐑨𐑛𐑮𐑨𐑒, ·𐑥𐑰𐑖𐑨𐑒, 𐑯 ·𐑩𐑚𐑧𐑛𐑯𐑦𐑜𐑴, 𐑣𐑦𐑟 𐑕𐑐𐑻𐑥𐑩𐑕𐑧𐑑𐑦, 𐑶𐑤, 𐑯 𐑚𐑴𐑯 𐑐𐑭𐑕 𐑳𐑯𐑕𐑒𐑱𐑞𐑛 𐑔𐑮𐑵 𐑞 𐑓𐑲𐑼;—𐑚𐑳𐑑 𐑯𐑬 𐑦𐑑 𐑮𐑦𐑥𐑱𐑯𐑟 𐑑 𐑒𐑩𐑯𐑒𐑤𐑵𐑛 𐑞 𐑤𐑭𐑕𐑑 𐑗𐑨𐑐𐑑𐑼 𐑝 𐑞𐑦𐑕 𐑐𐑸𐑑 𐑝 𐑞 𐑛𐑦𐑕𐑒𐑮𐑦𐑐𐑖𐑩𐑯 𐑚𐑲 𐑮𐑦𐑣𐑻𐑕𐑦𐑙—𐑕𐑦𐑙𐑦𐑙, 𐑦𐑓 𐑲 𐑥𐑱—𐑞 𐑮𐑴𐑥𐑨𐑯𐑑𐑦𐑒 𐑐𐑮𐑴𐑕𐑰𐑛𐑦𐑙 𐑝 𐑛𐑦𐑒𐑨𐑯𐑑𐑦𐑙 𐑪𐑓 𐑣𐑦𐑟 𐑶𐑤 𐑦𐑯𐑑𐑵 𐑞 𐑒𐑭𐑕𐑒𐑕 𐑯 𐑕𐑑𐑮𐑲𐑒𐑦𐑙 𐑞𐑧𐑥 𐑛𐑬𐑯 𐑦𐑯𐑑𐑵 𐑞 𐑣𐑴𐑤𐑛, 𐑢𐑺 𐑢𐑳𐑯𐑕 𐑩𐑜𐑱𐑯 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑮𐑦𐑑𐑻𐑯𐑟 𐑑 𐑣𐑦𐑟 𐑯𐑱𐑑𐑦𐑝 𐑐𐑮𐑩𐑓𐑳𐑯𐑛𐑦𐑑𐑦𐑟, 𐑕𐑤𐑲𐑛𐑦𐑙 𐑩𐑤𐑪𐑙 𐑚𐑦𐑯𐑰𐑔 𐑞 𐑕𐑻𐑓𐑦𐑕 𐑨𐑟 𐑚𐑦𐑓𐑹; 𐑚𐑳𐑑, 𐑩𐑤𐑨𐑕! 𐑯𐑧𐑝𐑼 𐑥𐑹 𐑑 𐑮𐑲𐑟 𐑯 𐑚𐑤𐑴.

𐑢𐑲𐑤 𐑕𐑑𐑦𐑤 𐑢𐑹𐑥, 𐑞 𐑶𐑤, 𐑤𐑲𐑒 𐑣𐑪𐑑 𐑐𐑳𐑯𐑗, 𐑦𐑟 𐑮𐑦𐑕𐑰𐑝𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑦𐑒𐑕-𐑚𐑨𐑮𐑩𐑤 𐑒𐑭𐑕𐑒𐑕; 𐑯 𐑢𐑲𐑤, 𐑐𐑼𐑣𐑨𐑐𐑕, 𐑞 𐑖𐑦𐑐 𐑦𐑟 𐑐𐑦𐑗𐑦𐑙 𐑯 𐑮𐑴𐑤𐑦𐑙 𐑞𐑦𐑕 𐑢𐑱 𐑯 𐑞𐑨𐑑 𐑦𐑯 𐑞 𐑥𐑦𐑛𐑯𐑲𐑑 𐑕𐑰, 𐑞 𐑦𐑯𐑹𐑥𐑩𐑕 𐑒𐑭𐑕𐑒𐑕 𐑸 𐑕𐑤𐑵𐑛 𐑮𐑬𐑯𐑛 𐑯 𐑣𐑧𐑛𐑩𐑛 𐑴𐑝𐑼, 𐑧𐑯𐑛 𐑓 𐑧𐑯𐑛, 𐑯 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑐𐑧𐑮𐑦𐑤𐑩𐑕𐑤𐑦 𐑕𐑒𐑵𐑑 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑕𐑤𐑦𐑐𐑼𐑦 𐑛𐑧𐑒, 𐑤𐑲𐑒 𐑕𐑴 𐑥𐑧𐑯𐑦 𐑤𐑨𐑯𐑛 𐑕𐑤𐑲𐑛𐑟, 𐑑𐑦𐑤 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑥𐑨𐑯-𐑣𐑨𐑯𐑛𐑩𐑤𐑛 𐑯 𐑕𐑑𐑱𐑛 𐑦𐑯 𐑞𐑺 𐑒𐑹𐑕; 𐑯 𐑷𐑤 𐑮𐑬𐑯𐑛 𐑞 𐑣𐑵𐑐𐑕, 𐑮𐑨𐑐, 𐑮𐑨𐑐, 𐑜𐑴 𐑨𐑟 𐑥𐑧𐑯𐑦 𐑣𐑨𐑥𐑼𐑟 𐑨𐑟 𐑒𐑨𐑯 𐑐𐑤𐑱 𐑩𐑐𐑪𐑯 𐑞𐑧𐑥, 𐑓 𐑯𐑬, 𐑧𐑒𐑕 𐑪𐑓𐑦𐑕𐑦𐑴, 𐑧𐑝𐑮𐑦 𐑕𐑱𐑤𐑼 𐑦𐑟 𐑩 𐑒𐑵𐑐𐑼.

𐑨𐑑 𐑤𐑧𐑙𐑔, 𐑢𐑧𐑯 𐑞 𐑤𐑭𐑕𐑑 𐑐𐑲𐑯𐑑 𐑦𐑟 𐑒𐑭𐑕𐑒𐑑, 𐑯 𐑷𐑤 𐑦𐑟 𐑒𐑵𐑤, 𐑞𐑧𐑯 𐑞 𐑜𐑮𐑱𐑑 𐑣𐑨𐑗𐑢𐑱𐑟 𐑸 𐑳𐑯𐑕𐑰𐑤𐑛, 𐑞 𐑚𐑬𐑩𐑤𐑟 𐑝 𐑞 𐑖𐑦𐑐 𐑸 𐑔𐑮𐑴𐑯 𐑴𐑐𐑩𐑯, 𐑯 𐑛𐑬𐑯 𐑜𐑴 𐑞 𐑒𐑭𐑕𐑒𐑕 𐑑 𐑞𐑺 𐑓𐑲𐑯𐑩𐑤 𐑮𐑧𐑕𐑑 𐑦𐑯 𐑞 𐑕𐑰. 𐑞𐑦𐑕 𐑛𐑳𐑯, 𐑞 𐑣𐑨𐑗𐑩𐑟 𐑸 𐑮𐑦𐑐𐑤𐑱𐑕𐑑, 𐑯 𐑣𐑻𐑥𐑧𐑑𐑦𐑒𐑤𐑦 𐑒𐑤𐑴𐑟𐑛, 𐑤𐑲𐑒 𐑩 𐑒𐑤𐑪𐑟𐑩𐑑 𐑢𐑷𐑤𐑛 𐑳𐑐.

𐑦𐑯 𐑞 𐑕𐑐𐑻𐑥 𐑓𐑦𐑖𐑼𐑦, 𐑞𐑦𐑕 𐑦𐑟 𐑐𐑼𐑣𐑨𐑐𐑕 𐑢𐑳𐑯 𐑝 𐑞 𐑥𐑴𐑕𐑑 𐑮𐑦𐑥𐑸𐑒𐑩𐑚𐑩𐑤 𐑦𐑯𐑕𐑦𐑛𐑩𐑯𐑑𐑕 𐑦𐑯 𐑷𐑤 𐑞 𐑚𐑦𐑟𐑯𐑩𐑕 𐑝 𐑢𐑱𐑤𐑦𐑙. 𐑢𐑳𐑯 𐑛𐑱 𐑞 𐑐𐑤𐑨𐑙𐑒𐑕 𐑕𐑑𐑮𐑰𐑥 𐑢𐑦𐑞 𐑓𐑮𐑧𐑖𐑩𐑑𐑕 𐑝 𐑚𐑤𐑳𐑛 𐑯 𐑶𐑤; 𐑪𐑯 𐑞 𐑕𐑱𐑒𐑮𐑩𐑛 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒 𐑦𐑯𐑹𐑥𐑩𐑕 𐑥𐑨𐑕𐑩𐑟 𐑝 𐑞 𐑢𐑱𐑤'𐑟 𐑣𐑧𐑛 𐑸 𐑐𐑮𐑴𐑓𐑱𐑯𐑤𐑦 𐑐𐑲𐑤𐑛; 𐑜𐑮𐑱𐑑 𐑮𐑳𐑕𐑑𐑦 𐑒𐑭𐑕𐑒𐑕 𐑤𐑲 𐑩𐑚𐑬𐑑, 𐑨𐑟 𐑦𐑯 𐑩 𐑚𐑮𐑵𐑼𐑦 𐑘𐑸𐑛; 𐑞 𐑕𐑥𐑴𐑒 𐑓𐑮𐑪𐑥 𐑞 𐑑𐑮𐑲-𐑢𐑻𐑒𐑕 𐑣𐑨𐑟 𐑚𐑦𐑕𐑫𐑑𐑩𐑛 𐑷𐑤 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕; 𐑞 𐑥𐑨𐑮𐑦𐑯𐑼𐑟 𐑜𐑴 𐑩𐑚𐑬𐑑 𐑕𐑩𐑓𐑿𐑟𐑛 𐑢𐑦𐑞 𐑳𐑙𐑒𐑗𐑵𐑩𐑕𐑯𐑩𐑕; 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑖𐑦𐑐 𐑕𐑰𐑥𐑟 𐑜𐑮𐑱𐑑 𐑤𐑩𐑝𐑲𐑩𐑔𐑩𐑯 𐑣𐑦𐑥𐑕𐑧𐑤𐑓; 𐑢𐑲𐑤 𐑪𐑯 𐑷𐑤 𐑣𐑨𐑯𐑛𐑟 𐑞 𐑛𐑦𐑯 𐑦𐑟 𐑛𐑧𐑓𐑩𐑯𐑦𐑙.

𐑚𐑳𐑑 𐑩 𐑛𐑱 𐑹 𐑑𐑵 𐑭𐑓𐑑𐑼, 𐑿 𐑤𐑫𐑒 𐑩𐑚𐑬𐑑 𐑿, 𐑯 𐑐𐑮𐑦𐑒 𐑘𐑹 𐑽𐑟 𐑦𐑯 𐑞𐑦𐑕 𐑕𐑧𐑤𐑓-𐑕𐑱𐑥 𐑖𐑦𐑐; 𐑯 𐑢𐑻 𐑦𐑑 𐑯𐑪𐑑 𐑓 𐑞 𐑑𐑧𐑤-𐑑𐑱𐑤 𐑚𐑴𐑑𐑕 𐑯 𐑑𐑮𐑲-𐑢𐑻𐑒𐑕, 𐑿 𐑢𐑫𐑛 𐑷𐑤 𐑚𐑳𐑑 𐑕𐑢𐑺 𐑿 𐑑𐑮𐑪𐑛 𐑕𐑳𐑥 𐑕𐑲𐑤𐑩𐑯𐑑 𐑥𐑻𐑗𐑩𐑯𐑑 𐑝𐑧𐑕𐑩𐑤, 𐑢𐑦𐑞 𐑩 𐑥𐑴𐑕𐑑 𐑕𐑒𐑮𐑵𐑐𐑿𐑤𐑩𐑕𐑤𐑦 𐑯𐑰𐑑 𐑒𐑩𐑥𐑭𐑯𐑛𐑼. 𐑞 𐑳𐑯𐑥𐑨𐑯𐑿𐑓𐑨𐑒𐑗𐑼𐑛 𐑕𐑐𐑻𐑥 𐑶𐑤 𐑐𐑩𐑟𐑧𐑕𐑩𐑟 𐑩 𐑕𐑦𐑙𐑜𐑿𐑤𐑼𐑤𐑦 𐑒𐑤𐑧𐑯𐑟𐑦𐑙 𐑝𐑻𐑗𐑵. 𐑞𐑦𐑕 𐑦𐑟 𐑞 𐑮𐑰𐑟𐑩𐑯 𐑢𐑲 𐑞 𐑛𐑧𐑒𐑕 𐑯𐑧𐑝𐑼 𐑤𐑫𐑒 𐑕𐑴 𐑢𐑲𐑑 𐑨𐑟 𐑡𐑳𐑕𐑑 𐑭𐑓𐑑𐑼 𐑢𐑪𐑑 𐑞𐑱 𐑒𐑷𐑤 𐑩𐑯 𐑩𐑓𐑺 𐑝 𐑶𐑤. 𐑚𐑦𐑕𐑲𐑛𐑟, 𐑓𐑮𐑪𐑥 𐑞 𐑨𐑖𐑩𐑟 𐑝 𐑞 𐑚𐑻𐑯𐑛 𐑕𐑒𐑮𐑨𐑐𐑕 𐑝 𐑞 𐑢𐑱𐑤, 𐑩 𐑐𐑴𐑑𐑩𐑯𐑑 𐑤𐑲 𐑦𐑟 𐑮𐑧𐑛𐑦𐑤𐑦 𐑥𐑱𐑛; 𐑯 𐑢𐑧𐑯𐑧𐑝𐑼 𐑧𐑯𐑦 𐑨𐑛𐑣𐑰𐑕𐑦𐑝𐑯𐑩𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑚𐑨𐑒 𐑝 𐑞 𐑢𐑱𐑤 𐑮𐑦𐑥𐑱𐑯𐑟 𐑒𐑤𐑦𐑙𐑦𐑙 𐑑 𐑞 𐑕𐑲𐑛, 𐑞𐑨𐑑 𐑤𐑲 𐑒𐑢𐑦𐑒𐑤𐑦 𐑧𐑒𐑕𐑑𐑻𐑥𐑦𐑯𐑱𐑑𐑕 𐑦𐑑. 𐑣𐑨𐑯𐑛𐑟 𐑜𐑴 𐑛𐑦𐑤𐑦𐑡𐑩𐑯𐑑𐑤𐑦 𐑩𐑤𐑪𐑙 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑯 𐑢𐑦𐑞 𐑚𐑳𐑒𐑩𐑑𐑕 𐑝 𐑢𐑷𐑑𐑼 𐑯 𐑮𐑨𐑜𐑟 𐑮𐑦𐑕𐑑𐑹 𐑞𐑧𐑥 𐑑 𐑞𐑺 𐑓𐑫𐑤 𐑑𐑲𐑛𐑦𐑯𐑩𐑕. 𐑞 𐑕𐑫𐑑 𐑦𐑟 𐑚𐑮𐑳𐑖𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑤𐑴𐑼 𐑮𐑦𐑜𐑦𐑙. 𐑷𐑤 𐑞 𐑯𐑿𐑥𐑼𐑩𐑕 𐑦𐑥𐑐𐑤𐑦𐑥𐑧𐑯𐑑𐑕 𐑢𐑦𐑗 𐑣𐑨𐑝 𐑚𐑰𐑯 𐑦𐑯 𐑿𐑕 𐑸 𐑤𐑲𐑒𐑢𐑲𐑟 𐑓𐑱𐑔𐑓𐑩𐑤𐑦 𐑒𐑤𐑧𐑯𐑟𐑛 𐑯 𐑐𐑫𐑑 𐑩𐑢𐑱. 𐑞 𐑜𐑮𐑱𐑑 𐑣𐑨𐑗 𐑦𐑟 𐑕𐑒𐑮𐑳𐑚𐑛 𐑯 𐑐𐑤𐑱𐑕𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑑𐑮𐑲-𐑢𐑻𐑒𐑕, 𐑒𐑩𐑥𐑐𐑤𐑰𐑑𐑤𐑦 𐑣𐑲𐑛𐑦𐑙 𐑞 𐑐𐑪𐑑𐑕; 𐑧𐑝𐑮𐑦 𐑒𐑭𐑕𐑒 𐑦𐑟 𐑬𐑑 𐑝 𐑕𐑲𐑑; 𐑷𐑤 𐑑𐑨𐑒𐑩𐑤𐑟 𐑸 𐑒𐑶𐑤𐑛 𐑦𐑯 𐑳𐑯𐑕𐑰𐑯 𐑯𐑫𐑒𐑕; 𐑯 𐑢𐑧𐑯 𐑚𐑲 𐑞 𐑒𐑩𐑥𐑚𐑲𐑯𐑛 𐑯 𐑕𐑦𐑥𐑩𐑤𐑑𐑱𐑯𐑾𐑕 𐑦𐑯𐑛𐑳𐑕𐑑𐑮𐑦 𐑝 𐑷𐑤𐑥𐑴𐑕𐑑 𐑞 𐑦𐑯𐑑𐑲𐑼 𐑖𐑦𐑐'𐑕 𐑒𐑳𐑥𐑐𐑩𐑯𐑦, 𐑞 𐑣𐑴𐑤 𐑝 𐑞𐑦𐑕 𐑒𐑪𐑯𐑖𐑦𐑧𐑯𐑖𐑩𐑕 𐑛𐑿𐑑𐑦 𐑦𐑟 𐑨𐑑 𐑤𐑭𐑕𐑑 𐑒𐑩𐑯𐑒𐑤𐑵𐑛𐑩𐑛, 𐑞𐑧𐑯 𐑞 𐑒𐑮𐑵 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑐𐑮𐑴𐑕𐑰𐑛 𐑑 𐑞𐑺 𐑴𐑯 𐑩𐑚𐑤𐑵𐑖𐑩𐑯𐑟; 𐑖𐑦𐑓𐑑 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑓𐑮𐑪𐑥 𐑑𐑪𐑐 𐑑 𐑑𐑴; 𐑯 𐑓𐑲𐑯𐑩𐑤𐑦 𐑦𐑖𐑿 𐑑 𐑞 𐑦𐑥𐑨𐑒𐑿𐑤𐑩𐑑 𐑛𐑧𐑒, 𐑓𐑮𐑧𐑖 𐑯 𐑷𐑤 𐑩𐑜𐑤𐑴, 𐑨𐑟 𐑚𐑮𐑲𐑛𐑜𐑮𐑵𐑥𐑟 𐑯𐑿-𐑤𐑰𐑐𐑑 𐑓𐑮𐑪𐑥 𐑬𐑑 𐑞 𐑛𐑱𐑯𐑑𐑾𐑕𐑑 ·𐑣𐑪𐑤𐑩𐑯𐑛.
