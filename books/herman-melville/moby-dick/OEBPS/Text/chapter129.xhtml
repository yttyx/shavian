·𐑗𐑨𐑐𐑑𐑼 129. 𐑞 ·𐑒𐑨𐑚𐑦𐑯.

(𐑱𐑣𐑨𐑚 𐑥𐑵𐑝𐑦𐑙 𐑑 𐑜𐑴 𐑪𐑯 𐑛𐑧𐑒; ·𐑐𐑦𐑐 𐑒𐑨𐑗𐑩𐑟 𐑣𐑦𐑥 𐑚𐑲 𐑞 𐑣𐑨𐑯𐑛 𐑑 𐑓𐑪𐑤𐑴.)

“𐑤𐑨𐑛, 𐑤𐑨𐑛, 𐑲 𐑑𐑧𐑤 𐑞𐑰 𐑞𐑬 𐑥𐑳𐑕𐑑 𐑯𐑪𐑑 𐑓𐑪𐑤𐑴 ·𐑱𐑣𐑨𐑚 𐑯𐑬. 𐑞 𐑬𐑼 𐑦𐑟 𐑒𐑳𐑥𐑦𐑙 𐑢𐑧𐑯 ·𐑱𐑣𐑨𐑚 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑕𐑒𐑺 𐑞𐑰 𐑓𐑮𐑪𐑥 𐑣𐑦𐑥, 𐑘𐑧𐑑 𐑢𐑫𐑛 𐑯𐑪𐑑 𐑣𐑨𐑝 𐑞𐑰 𐑚𐑲 𐑣𐑦𐑥. 𐑞𐑺 𐑦𐑟 𐑞𐑨𐑑 𐑦𐑯 𐑞𐑰, 𐑐𐑫𐑼 𐑤𐑨𐑛, 𐑢𐑦𐑗 𐑲 𐑓𐑰𐑤 𐑑𐑵 𐑒𐑘𐑫𐑼𐑦𐑙 𐑑 𐑥𐑲 𐑥𐑨𐑤𐑩𐑛𐑦. 𐑤𐑲𐑒 𐑒𐑘𐑫𐑼𐑟 𐑤𐑲𐑒; 𐑯 𐑓 𐑞𐑦𐑕 𐑣𐑳𐑯𐑑, 𐑥𐑲 𐑥𐑨𐑤𐑩𐑛𐑦 𐑚𐑦𐑒𐑳𐑥𐑟 𐑥𐑲 𐑥𐑴𐑕𐑑 𐑛𐑦𐑟𐑲𐑼𐑛 𐑣𐑧𐑤𐑔. 𐑛𐑵 𐑞𐑬 𐑩𐑚𐑲𐑛 𐑚𐑦𐑤𐑴 𐑣𐑽, 𐑢𐑺 𐑞𐑱 𐑖𐑨𐑤 𐑕𐑻𐑝 𐑞𐑰, 𐑨𐑟 𐑦𐑓 𐑞𐑬 ·𐑝𐑧𐑮𐑑 𐑞 𐑒𐑨𐑐𐑑𐑦𐑯. 𐑲, 𐑤𐑨𐑛, 𐑞𐑬 𐑖𐑨𐑤𐑑 𐑕𐑦𐑑 𐑣𐑽 𐑦𐑯 𐑥𐑲 𐑴𐑯 𐑕𐑒𐑮𐑵𐑛 𐑗𐑺; 𐑩𐑯𐑳𐑞𐑼 𐑕𐑒𐑮𐑵 𐑑 𐑦𐑑, 𐑞𐑬 𐑥𐑳𐑕𐑑 𐑚𐑰.”

“𐑯𐑴, 𐑯𐑴, 𐑯𐑴! 𐑘𐑰 𐑣𐑨𐑝 𐑯𐑪𐑑 𐑩 𐑣𐑴𐑤 𐑚𐑪𐑛𐑦, 𐑕𐑻; 𐑛𐑵 𐑘𐑰 𐑚𐑳𐑑 𐑿𐑟 𐑐𐑫𐑼 𐑥𐑰 𐑓 𐑘𐑹 𐑢𐑳𐑯 𐑤𐑪𐑕𐑑 𐑤𐑧𐑜; 𐑴𐑯𐑤𐑦 𐑑𐑮𐑧𐑛 𐑩𐑐𐑪𐑯 𐑥𐑰, 𐑕𐑻; 𐑲 𐑭𐑕𐑒 𐑯𐑴 𐑥𐑹, 𐑕𐑴 𐑲 𐑮𐑦𐑥𐑱𐑯 𐑩 𐑐𐑸𐑑 𐑝 𐑘𐑰.”

“𐑴! 𐑕𐑐𐑲𐑑 𐑝 𐑥𐑦𐑤𐑾𐑯 𐑝𐑦𐑤𐑩𐑯𐑟, 𐑞𐑦𐑕 𐑥𐑱𐑒𐑕 𐑥𐑰 𐑩 𐑚𐑦𐑜𐑩𐑑 𐑦𐑯 𐑞 𐑓𐑱𐑛𐑤𐑩𐑕 𐑓𐑦𐑛𐑧𐑤𐑦𐑑𐑦 𐑝 𐑥𐑨𐑯!—𐑯 𐑩 𐑚𐑤𐑨𐑒! 𐑯 𐑒𐑮𐑱𐑟𐑦!—𐑚𐑳𐑑 𐑥𐑦𐑔𐑦𐑙𐑒𐑕 𐑤𐑲𐑒-𐑒𐑘𐑫𐑼𐑟-𐑤𐑲𐑒 𐑩𐑐𐑤𐑲𐑟 𐑑 𐑣𐑦𐑥 𐑑𐑵; 𐑣𐑰 𐑜𐑮𐑴𐑟 𐑕𐑴 𐑕𐑱𐑯 𐑩𐑜𐑱𐑯.”

“𐑞𐑱 𐑑𐑧𐑤 𐑥𐑰, 𐑕𐑻, 𐑞𐑨𐑑 ·𐑕𐑑𐑳𐑚 𐑛𐑦𐑛 𐑢𐑳𐑯𐑕 𐑛𐑧𐑟𐑼𐑑 𐑐𐑫𐑼 𐑤𐑦𐑑𐑩𐑤 ·𐑐𐑦𐑐, 𐑣𐑵𐑟 𐑛𐑮𐑬𐑯𐑛 𐑚𐑴𐑯𐑟 𐑯𐑬 𐑖𐑴 𐑢𐑲𐑑, 𐑓 𐑷𐑤 𐑞 𐑚𐑤𐑨𐑒𐑯𐑩𐑕 𐑝 𐑣𐑦𐑟 𐑤𐑦𐑝𐑦𐑙 𐑕𐑒𐑦𐑯. 𐑚𐑳𐑑 𐑲 𐑢𐑦𐑤 𐑯𐑧𐑝𐑼 𐑛𐑦𐑟𐑻𐑑 𐑘𐑰, 𐑕𐑻, 𐑨𐑟 ·𐑕𐑑𐑳𐑚 𐑛𐑦𐑛 𐑣𐑦𐑥. 𐑕𐑻, 𐑲 𐑥𐑳𐑕𐑑 𐑜𐑴 𐑢𐑦𐑞 𐑘𐑰.”

“𐑦𐑓 𐑞𐑬 𐑕𐑐𐑰𐑒𐑩𐑕𐑑 𐑞𐑳𐑕 𐑑 𐑥𐑰 𐑥𐑳𐑗 𐑥𐑹, ·𐑱𐑣𐑨𐑚'𐑟 𐑐𐑻𐑐𐑩𐑕 𐑒𐑰𐑤𐑟 𐑳𐑐 𐑦𐑯 𐑣𐑦𐑥. 𐑲 𐑑𐑧𐑤 𐑞𐑰 𐑯𐑴; 𐑦𐑑 𐑒𐑨𐑯𐑪𐑑 𐑚𐑰.”

“𐑴 𐑜𐑫𐑛 𐑥𐑭𐑕𐑑𐑼, 𐑥𐑭𐑕𐑑𐑼, 𐑥𐑭𐑕𐑑𐑼!

“𐑢𐑰𐑐 𐑕𐑴, 𐑯 𐑲 𐑢𐑦𐑤 𐑥𐑻𐑛𐑼 𐑞𐑰! 𐑣𐑨𐑝 𐑩 𐑒𐑺, 𐑓 ·𐑱𐑣𐑨𐑚 𐑑𐑵 𐑦𐑟 𐑥𐑨𐑛. 𐑤𐑦𐑕𐑩𐑯, 𐑯 𐑞𐑬 𐑢𐑦𐑤𐑑 𐑪𐑓𐑩𐑯 𐑣𐑽 𐑥𐑲 𐑲𐑝𐑼𐑦 𐑓𐑫𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑛𐑧𐑒, 𐑯 𐑕𐑑𐑦𐑤 𐑯𐑴 𐑞𐑨𐑑 𐑲 𐑨𐑥 𐑞𐑺. 𐑯 𐑯𐑬 𐑲 𐑒𐑢𐑦𐑑 𐑞𐑰. 𐑞𐑲 𐑣𐑨𐑯𐑛!—𐑥𐑧𐑑! 𐑑𐑮𐑵 𐑸𐑑 𐑞𐑬, 𐑤𐑨𐑛, 𐑨𐑟 𐑞 𐑕𐑼𐑒𐑳𐑥𐑓𐑼𐑩𐑯𐑕 𐑑 𐑦𐑑𐑕 𐑕𐑧𐑯𐑑𐑼. 𐑕𐑴: 𐑜𐑪𐑛 𐑓 𐑧𐑝𐑼 𐑚𐑤𐑧𐑕 𐑞𐑰; 𐑯 𐑦𐑓 𐑦𐑑 𐑒𐑳𐑥 𐑑 𐑞𐑨𐑑,—·𐑜𐑪𐑛 𐑓 𐑧𐑝𐑼 𐑕𐑱𐑝 𐑞𐑰, 𐑤𐑧𐑑 𐑢𐑪𐑑 𐑢𐑦𐑤 𐑚𐑦𐑓𐑷𐑤.”

(𐑱𐑣𐑨𐑚 𐑜𐑴𐑟; ·𐑐𐑦𐑐 𐑕𐑑𐑧𐑐𐑕 𐑢𐑳𐑯 𐑕𐑑𐑧𐑐 𐑓𐑹𐑢𐑼𐑛.)

“𐑣𐑽 𐑣𐑰 𐑞𐑦𐑕 𐑦𐑯𐑕𐑑𐑩𐑯𐑑 𐑕𐑑𐑫𐑛; 𐑲 𐑕𐑑𐑨𐑯𐑛 𐑦𐑯 𐑣𐑦𐑟 𐑺,—𐑚𐑳𐑑 𐑲'𐑥 𐑩𐑤𐑴𐑯. 𐑯𐑬 𐑢𐑻 𐑰𐑝𐑩𐑯 𐑐𐑫𐑼 ·𐑐𐑦𐑐 𐑣𐑽 𐑲 𐑒𐑫𐑛 𐑦𐑯𐑛𐑻 𐑦𐑑, 𐑚𐑳𐑑 𐑣𐑰'𐑟 𐑥𐑦𐑕𐑦𐑙. 𐑐𐑦𐑐! 𐑐𐑦𐑐! 𐑛𐑦𐑙, 𐑛𐑪𐑙, 𐑛𐑦𐑙! 𐑣𐑵'𐑟 𐑕𐑰𐑯 ·𐑐𐑦𐑐? 𐑣𐑰 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑳𐑐 𐑣𐑽; 𐑤𐑧𐑑'𐑕 𐑑𐑮𐑲 𐑞 𐑛𐑹. 𐑢𐑪𐑑? 𐑯𐑲𐑞𐑼 𐑤𐑪𐑒, 𐑯𐑹 𐑚𐑴𐑤𐑑, 𐑯𐑹 𐑚𐑸; 𐑯 𐑘𐑧𐑑 𐑞𐑺'𐑟 𐑯𐑴 𐑴𐑐𐑩𐑯𐑦𐑙 𐑦𐑑. 𐑦𐑑 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑞 𐑕𐑐𐑧𐑤; 𐑣𐑰 𐑑𐑴𐑤𐑛 𐑥𐑰 𐑑 𐑕𐑑𐑱 𐑣𐑽: 𐑲, 𐑯 𐑑𐑴𐑤𐑛 𐑥𐑰 𐑞𐑦𐑕 𐑕𐑒𐑮𐑵𐑛 𐑗𐑺 𐑢𐑪𐑟 𐑥𐑲𐑯. 𐑣𐑽, 𐑞𐑧𐑯, 𐑲'𐑤 𐑕𐑰𐑑 𐑥𐑰, 𐑩𐑜𐑱𐑯𐑕𐑑 𐑞 𐑑𐑮𐑨𐑯𐑕𐑩𐑥, 𐑦𐑯 𐑞 𐑖𐑦𐑐'𐑕 𐑓𐑫𐑤 𐑥𐑦𐑛𐑩𐑤, 𐑷𐑤 𐑣𐑻 𐑒𐑰𐑤 𐑯 𐑣𐑻 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑𐑕 𐑚𐑦𐑓𐑹 𐑥𐑰. 𐑣𐑽, 𐑬𐑼 𐑴𐑤𐑛 𐑕𐑱𐑤𐑼𐑟 𐑕𐑱, 𐑦𐑯 𐑞𐑺 𐑚𐑤𐑨𐑒 𐑕𐑧𐑝𐑩𐑯𐑑𐑦-𐑓𐑹𐑟 𐑜𐑮𐑱𐑑 𐑨𐑛𐑥𐑼𐑩𐑤𐑟 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑕𐑦𐑑 𐑨𐑑 𐑑𐑱𐑚𐑩𐑤, 𐑯 𐑤𐑹𐑛 𐑦𐑑 𐑴𐑝𐑼 𐑮𐑴𐑟 𐑝 𐑒𐑨𐑐𐑑𐑦𐑯𐑟 𐑯 𐑤𐑧𐑓𐑑𐑧𐑯𐑩𐑯𐑑𐑕. 𐑣𐑭! 𐑢𐑪𐑑'𐑕 𐑞𐑦𐑕? 𐑧𐑐𐑩𐑤𐑧𐑑𐑕! 𐑧𐑐𐑩𐑤𐑧𐑑𐑕! 𐑞 𐑧𐑐𐑩𐑤𐑧𐑑𐑕 𐑷𐑤 𐑒𐑳𐑥 𐑒𐑮𐑬𐑛𐑦𐑙! 𐑐𐑭𐑕 𐑮𐑬𐑯𐑛 𐑞 𐑛𐑦𐑒𐑨𐑯𐑑𐑼𐑟; 𐑜𐑤𐑨𐑛 𐑑 𐑕𐑰 𐑘𐑰; 𐑓𐑦𐑤 𐑳𐑐, 𐑥𐑩𐑕𐑘𐑻𐑟! 𐑢𐑪𐑑 𐑩𐑯 𐑪𐑛 𐑓𐑰𐑤𐑦𐑙, 𐑯𐑬, 𐑢𐑧𐑯 𐑩 𐑚𐑤𐑨𐑒 𐑚𐑶'𐑟 𐑣𐑴𐑕𐑑 𐑑 𐑢𐑲𐑑 𐑥𐑧𐑯 𐑢𐑦𐑞 𐑜𐑴𐑤𐑛 𐑤𐑱𐑕 𐑩𐑐𐑪𐑯 𐑞𐑺 𐑒𐑴𐑑𐑕!—𐑥𐑩𐑕𐑘𐑻𐑟, 𐑣𐑨𐑝 𐑘𐑰 𐑕𐑰𐑯 𐑢𐑳𐑯 ·𐑐𐑦𐑐?—𐑩 𐑤𐑦𐑑𐑩𐑤 𐑯𐑰𐑜𐑮𐑴 𐑤𐑨𐑛, 𐑓𐑲𐑝 𐑓𐑰𐑑 𐑣𐑲, 𐑣𐑨𐑙-𐑛𐑪𐑜 𐑤𐑫𐑒, 𐑯 𐑒𐑬𐑼𐑛𐑤𐑦! 𐑡𐑳𐑥𐑐𐑑 𐑓𐑮𐑪𐑥 𐑩 𐑢𐑱𐑤-𐑚𐑴𐑑 𐑢𐑳𐑯𐑕;—𐑕𐑰𐑯 𐑣𐑦𐑥? 𐑯𐑴! 𐑢𐑧𐑤 𐑞𐑧𐑯, 𐑓𐑦𐑤 𐑳𐑐 𐑩𐑜𐑱𐑯, 𐑒𐑨𐑐𐑑𐑦𐑯𐑟, 𐑯 𐑤𐑧𐑑'𐑕 𐑛𐑮𐑦𐑙𐑒 𐑖𐑱𐑥 𐑩𐑐𐑪𐑯 𐑷𐑤 𐑒𐑬𐑼𐑛𐑟! 𐑲 𐑯𐑱𐑥 𐑯𐑴 𐑯𐑱𐑥𐑟. 𐑖𐑱𐑥 𐑩𐑐𐑪𐑯 𐑞𐑧𐑥! 𐑐𐑫𐑑 𐑢𐑳𐑯 𐑓𐑫𐑑 𐑩𐑐𐑪𐑯 𐑞 𐑑𐑱𐑚𐑩𐑤. 𐑖𐑱𐑥 𐑩𐑐𐑪𐑯 𐑷𐑤 𐑒𐑬𐑼𐑛𐑟.—𐑣𐑦𐑕𐑑! 𐑩𐑚𐑳𐑝 𐑞𐑺, 𐑲 𐑣𐑽 𐑲𐑝𐑼𐑦—𐑴, 𐑥𐑭𐑕𐑑𐑼! 𐑥𐑭𐑕𐑑𐑼! 𐑲 𐑨𐑥 𐑦𐑯𐑛𐑰𐑛 𐑛𐑬𐑯-𐑣𐑸𐑑𐑩𐑛 𐑢𐑧𐑯 𐑿 𐑢𐑷𐑒 𐑴𐑝𐑼 𐑥𐑰. 𐑚𐑳𐑑 𐑣𐑽 𐑲'𐑤 𐑕𐑑𐑱, 𐑞𐑴 𐑞𐑦𐑕 𐑕𐑑𐑻𐑯 𐑕𐑑𐑮𐑲𐑒𐑕 𐑮𐑪𐑒𐑕; 𐑯 𐑞𐑱 𐑚𐑳𐑤𐑡 𐑔𐑮𐑵; 𐑯 𐑶𐑕𐑑𐑼𐑟 𐑒𐑳𐑥 𐑑 𐑡𐑶𐑯 𐑥𐑰.”
