·𐑗𐑨𐑐𐑑𐑼 64. 𐑕𐑑𐑳𐑚'𐑟 ·𐑕𐑳𐑐𐑼.

𐑕𐑑𐑳𐑚'𐑟 𐑢𐑱𐑤 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑒𐑦𐑤𐑛 𐑕𐑳𐑥 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑓𐑮𐑪𐑥 𐑞 𐑖𐑦𐑐. 𐑦𐑑 𐑢𐑪𐑟 𐑩 𐑒𐑭𐑥; 𐑕𐑴, 𐑓𐑹𐑥𐑦𐑙 𐑩 𐑑𐑨𐑯𐑛𐑩𐑥 𐑝 𐑔𐑮𐑰 𐑚𐑴𐑑𐑕, 𐑢𐑰 𐑒𐑩𐑥𐑧𐑯𐑕𐑑 𐑞 𐑕𐑤𐑴 𐑚𐑦𐑟𐑯𐑩𐑕 𐑝 𐑑𐑴𐑦𐑙 𐑞 𐑑𐑮𐑴𐑓𐑦 𐑑 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛. 𐑯 𐑯𐑬, 𐑨𐑟 𐑢𐑰 𐑱𐑑𐑰𐑯 𐑥𐑧𐑯 𐑢𐑦𐑞 𐑬𐑼 𐑔𐑻𐑑𐑦-𐑕𐑦𐑒𐑕 𐑸𐑥𐑟, 𐑯 𐑢𐑳𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑯 𐑱𐑑𐑦 𐑔𐑳𐑥𐑟 𐑯 𐑓𐑦𐑙𐑜𐑼𐑟, 𐑕𐑤𐑴𐑤𐑦 𐑑𐑶𐑤𐑛 𐑬𐑼 𐑭𐑓𐑑𐑼 𐑬𐑼 𐑩𐑐𐑪𐑯 𐑞𐑨𐑑 𐑦𐑯𐑻𐑑, 𐑕𐑤𐑳𐑜𐑦𐑖 𐑒𐑹𐑐𐑕 𐑦𐑯 𐑞 𐑕𐑰; 𐑯 𐑦𐑑 𐑕𐑰𐑥𐑛 𐑣𐑸𐑛𐑤𐑦 𐑑 𐑚𐑳𐑡 𐑨𐑑 𐑷𐑤, 𐑦𐑒𐑕𐑧𐑐𐑑 𐑨𐑑 𐑤𐑪𐑙 𐑦𐑯𐑑𐑼𐑝𐑩𐑤𐑟; 𐑜𐑫𐑛 𐑧𐑝𐑦𐑛𐑩𐑯𐑕 𐑢𐑪𐑟 𐑣𐑽𐑚𐑲 𐑓𐑻𐑯𐑦𐑖𐑑 𐑝 𐑞 𐑦𐑯𐑹𐑥𐑩𐑕𐑯𐑩𐑕 𐑝 𐑞 𐑥𐑨𐑕 𐑢𐑰 𐑥𐑵𐑝𐑛. 𐑓, 𐑩𐑐𐑪𐑯 𐑞 𐑜𐑮𐑱𐑑 𐑒𐑩𐑯𐑨𐑤 𐑝 ·𐑣𐑨𐑙-·𐑣𐑴, 𐑹 𐑢𐑪𐑑𐑧𐑝𐑼 𐑞𐑱 𐑒𐑷𐑤 𐑦𐑑, 𐑦𐑯 ·𐑗𐑲𐑯𐑩, 𐑓𐑹 𐑹 𐑓𐑲𐑝 𐑤𐑱𐑚𐑼𐑼𐑟 𐑪𐑯 𐑞 𐑓𐑫𐑑-𐑐𐑭𐑔 𐑢𐑦𐑤 𐑛𐑮𐑷 𐑩 𐑚𐑳𐑤𐑒𐑦 𐑓𐑮𐑱𐑑𐑩𐑛 𐑡𐑳𐑙𐑒 𐑨𐑑 𐑞 𐑮𐑱𐑑 𐑝 𐑩 𐑥𐑲𐑤 𐑩𐑯 𐑬𐑼; 𐑚𐑳𐑑 𐑞𐑦𐑕 𐑜𐑮𐑨𐑯𐑛 𐑸𐑜𐑩𐑕𐑦 𐑢𐑰 𐑑𐑴𐑛 𐑣𐑧𐑝𐑦𐑤𐑦 𐑓𐑹𐑡𐑛 𐑩𐑤𐑪𐑙, 𐑨𐑟 𐑦𐑓 𐑤𐑱𐑛𐑩𐑯 𐑢𐑦𐑞 𐑐𐑦𐑜-𐑤𐑧𐑛 𐑦𐑯 𐑚𐑳𐑤𐑒.

𐑞 𐑴𐑤𐑛 𐑚𐑤𐑨𐑒, 𐑯𐑪𐑑 𐑦𐑯 𐑧𐑯𐑦 𐑝𐑧𐑮𐑦 𐑣𐑲 𐑜𐑤𐑰 𐑨𐑑 𐑣𐑨𐑝𐑦𐑙 𐑚𐑰𐑯 𐑐𐑮𐑰𐑝𐑾𐑕𐑤𐑦 𐑮𐑬𐑟𐑛 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑢𐑹𐑥 𐑣𐑨𐑥𐑩𐑒 𐑨𐑑 𐑩 𐑥𐑴𐑕𐑑 𐑳𐑯𐑕𐑰𐑟𐑩𐑯𐑩𐑚𐑩𐑤 𐑬𐑼, 𐑒𐑱𐑥 𐑖𐑨𐑥𐑚𐑩𐑤𐑦𐑙 𐑩𐑤𐑪𐑙 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑜𐑨𐑤𐑦, 𐑓, 𐑤𐑲𐑒 𐑥𐑧𐑯𐑦 𐑴𐑤𐑛 𐑚𐑤𐑨𐑒𐑕, 𐑞𐑺 𐑢𐑪𐑟 𐑕𐑳𐑥𐑔𐑦𐑙 𐑞 𐑥𐑨𐑑𐑼 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑯𐑰-𐑐𐑨𐑯𐑟, 𐑢𐑦𐑗 𐑣𐑰 𐑛𐑦𐑛 𐑯𐑪𐑑 𐑒𐑰𐑐 𐑢𐑧𐑤 𐑕𐑒𐑬𐑼𐑛 𐑤𐑲𐑒 𐑣𐑦𐑟 𐑳𐑞𐑼 𐑐𐑨𐑯𐑟; 𐑞𐑦𐑕 𐑴𐑤𐑛 ·𐑓𐑤𐑰𐑕, 𐑨𐑟 𐑞𐑱 𐑒𐑷𐑤𐑛 𐑣𐑦𐑥, 𐑒𐑱𐑥 𐑖𐑳𐑓𐑩𐑤𐑦𐑙 𐑯 𐑤𐑦𐑥𐑐𐑦𐑙 𐑩𐑤𐑪𐑙, 𐑩𐑕𐑦𐑕𐑑𐑦𐑙 𐑣𐑦𐑟 𐑕𐑑𐑧𐑐 𐑢𐑦𐑞 𐑣𐑦𐑟 𐑑𐑪𐑙𐑟, 𐑢𐑦𐑗, 𐑭𐑓𐑑𐑼 𐑩 𐑒𐑤𐑳𐑥𐑟𐑦 𐑓𐑨𐑖𐑩𐑯, 𐑢𐑻 𐑥𐑱𐑛 𐑝 𐑕𐑑𐑮𐑱𐑑𐑩𐑯𐑛 𐑲𐑼𐑯 𐑣𐑵𐑐𐑕; 𐑞𐑦𐑕 𐑴𐑤𐑛 ·𐑧𐑚𐑩𐑯𐑦 𐑓𐑤𐑬𐑯𐑛𐑼𐑛 𐑩𐑤𐑪𐑙, 𐑯 𐑦𐑯 𐑴𐑚𐑰𐑛𐑾𐑯𐑕 𐑑 𐑞 𐑢𐑻𐑛 𐑝 𐑒𐑩𐑥𐑭𐑯𐑛, 𐑒𐑱𐑥 𐑑 𐑩 𐑛𐑧𐑛 𐑕𐑑𐑪𐑐 𐑪𐑯 𐑞 𐑪𐑐𐑩𐑟𐑦𐑑 𐑕𐑲𐑛 𐑝 ·𐑕𐑑𐑳𐑚'𐑟 𐑕𐑲𐑛𐑚𐑹𐑛; 𐑢𐑧𐑯, 𐑢𐑦𐑞 𐑚𐑴𐑔 𐑣𐑨𐑯𐑛𐑟 𐑓𐑴𐑤𐑛𐑩𐑛 𐑚𐑦𐑓𐑹 𐑣𐑦𐑥, 𐑯 𐑮𐑧𐑕𐑑𐑦𐑙 𐑪𐑯 𐑣𐑦𐑟 𐑑𐑵-𐑤𐑧𐑜𐑩𐑛 𐑒𐑱𐑯, 𐑣𐑰 𐑚𐑴𐑛 𐑣𐑦𐑟 𐑸𐑗𐑑 𐑚𐑨𐑒 𐑕𐑑𐑦𐑤 𐑓𐑻𐑞𐑼 𐑴𐑝𐑼, 𐑨𐑑 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥 𐑕𐑲𐑛𐑢𐑱𐑟 𐑦𐑯𐑒𐑤𐑲𐑯𐑦𐑙 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑕𐑴 𐑨𐑟 𐑑 𐑚𐑮𐑦𐑙 𐑣𐑦𐑟 𐑚𐑧𐑕𐑑 𐑽 𐑦𐑯𐑑𐑵 𐑐𐑤𐑱.

“𐑒𐑫𐑒,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑤𐑦𐑓𐑑𐑦𐑙 𐑩 𐑮𐑭𐑞𐑼 𐑮𐑧𐑛𐑦𐑖 𐑥𐑹𐑕𐑩𐑤 𐑑 𐑣𐑦𐑟 𐑥𐑬𐑔, “𐑛𐑴𐑯𐑑 𐑿 𐑔𐑦𐑙𐑒 𐑞𐑦𐑕 𐑕𐑑𐑱𐑒 𐑦𐑟 𐑮𐑭𐑞𐑼 𐑴𐑝𐑼𐑛𐑳𐑯? 𐑿'𐑝 𐑚𐑰𐑯 𐑚𐑰𐑑𐑦𐑙 𐑞𐑦𐑕 𐑕𐑑𐑱𐑒 𐑑𐑵 𐑥𐑳𐑗, 𐑒𐑫𐑒; 𐑦𐑑'𐑕 𐑑𐑵 𐑑𐑧𐑯𐑛𐑼. 𐑛𐑴𐑯𐑑 𐑲 𐑷𐑤𐑢𐑱𐑟 𐑕𐑱 𐑞𐑨𐑑 𐑑 𐑚𐑰 𐑜𐑫𐑛, 𐑩 𐑢𐑱𐑤-𐑕𐑑𐑱𐑒 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑑𐑳𐑓? 𐑞𐑺 𐑸 𐑞𐑴𐑟 𐑖𐑸𐑒𐑕 𐑯𐑬 𐑴𐑝𐑼 𐑞 𐑕𐑲𐑛, 𐑛𐑴𐑯𐑑 𐑿 𐑕𐑰 𐑞𐑱 𐑐𐑮𐑦𐑓𐑻 𐑦𐑑 𐑑𐑳𐑓 𐑯 𐑮𐑺? 𐑢𐑪𐑑 𐑩 𐑖𐑦𐑯𐑛𐑦 𐑞𐑱 𐑸 𐑒𐑦𐑒𐑦𐑙 𐑳𐑐! 𐑒𐑫𐑒, 𐑜𐑴 𐑯 𐑑𐑷𐑒 𐑑 '𐑩𐑥; 𐑑𐑧𐑤 '𐑩𐑥 𐑞𐑱 𐑸 𐑢𐑧𐑤𐑒𐑩𐑥 𐑑 𐑣𐑧𐑤𐑐 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑕𐑦𐑝𐑦𐑤𐑦, 𐑯 𐑦𐑯 𐑥𐑪𐑛𐑼𐑱𐑖𐑩𐑯, 𐑚𐑳𐑑 𐑞𐑱 𐑥𐑳𐑕𐑑 𐑒𐑰𐑐 𐑒𐑢𐑲𐑩𐑑. 𐑚𐑤𐑭𐑕𐑑 𐑥𐑰, 𐑦𐑓 𐑲 𐑒𐑨𐑯 𐑣𐑽 𐑥𐑲 𐑴𐑯 𐑝𐑶𐑕. 𐑩𐑢𐑱, 𐑒𐑫𐑒, 𐑯 𐑛𐑦𐑤𐑦𐑝𐑼 𐑥𐑲 𐑥𐑧𐑕𐑦𐑡. 𐑣𐑽, 𐑑𐑱𐑒 𐑞𐑦𐑕 𐑤𐑨𐑯𐑑𐑼𐑯,” 𐑕𐑯𐑨𐑗𐑦𐑙 𐑢𐑳𐑯 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑕𐑲𐑛𐑚𐑹𐑛; “𐑯𐑬 𐑞𐑧𐑯, 𐑜𐑴 𐑯 𐑐𐑮𐑰𐑗 𐑑 '𐑩𐑥!”

𐑕𐑳𐑤𐑩𐑯𐑤𐑦 𐑑𐑱𐑒𐑦𐑙 𐑞 𐑪𐑓𐑼𐑛 𐑤𐑨𐑯𐑑𐑼𐑯, 𐑴𐑤𐑛 ·𐑓𐑤𐑰𐑕 𐑤𐑦𐑥𐑐𐑑 𐑩𐑒𐑮𐑪𐑕 𐑞 𐑛𐑧𐑒 𐑑 𐑞 𐑚𐑫𐑤𐑢𐑼𐑒𐑕; 𐑯 𐑞𐑧𐑯, 𐑢𐑦𐑞 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛 𐑛𐑮𐑪𐑐𐑦𐑙 𐑣𐑦𐑟 𐑤𐑲𐑑 𐑤𐑴 𐑴𐑝𐑼 𐑞 𐑕𐑰, 𐑕𐑴 𐑨𐑟 𐑑 𐑜𐑧𐑑 𐑩 𐑜𐑫𐑛 𐑝𐑿 𐑝 𐑣𐑦𐑟 𐑒𐑪𐑙𐑜𐑮𐑦𐑜𐑱𐑖𐑩𐑯, 𐑢𐑦𐑞 𐑞 𐑳𐑞𐑼 𐑣𐑨𐑯𐑛 𐑣𐑰 𐑕𐑪𐑤𐑩𐑥𐑤𐑦 𐑓𐑤𐑳𐑮𐑦𐑖𐑑 𐑣𐑦𐑟 𐑑𐑪𐑙𐑟, 𐑯 𐑤𐑰𐑯𐑦𐑙 𐑓𐑸 𐑴𐑝𐑼 𐑞 𐑕𐑲𐑛 𐑦𐑯 𐑩 𐑥𐑳𐑥𐑚𐑩𐑤𐑦𐑙 𐑝𐑶𐑕 𐑚𐑦𐑜𐑨𐑯 𐑩𐑛𐑮𐑧𐑕𐑦𐑙 𐑞 𐑖𐑸𐑒𐑕, 𐑢𐑲𐑤 ·𐑕𐑑𐑳𐑚, 𐑕𐑪𐑓𐑑𐑤𐑦 𐑒𐑮𐑷𐑤𐑦𐑙 𐑚𐑦𐑣𐑲𐑯𐑛, 𐑴𐑝𐑼𐑣𐑻𐑛 𐑷𐑤 𐑞𐑨𐑑 𐑢𐑪𐑟 𐑕𐑧𐑛.

“𐑓𐑧𐑤𐑴-𐑒𐑮𐑦𐑑𐑼𐑟: ·𐑲'𐑟 𐑹𐑛𐑼𐑛 𐑣𐑽 𐑑 𐑕𐑱 𐑛𐑨𐑑 𐑿 𐑥𐑳𐑕𐑑 𐑕𐑑𐑪𐑐 𐑛𐑨𐑑 𐑛𐑨𐑥 𐑯𐑶𐑟 𐑛𐑺. 𐑿 𐑣𐑽? 𐑕𐑑𐑪𐑐 𐑛𐑨𐑑 𐑛𐑨𐑥 𐑕𐑥𐑨𐑒𐑦𐑯' 𐑪𐑚 𐑛𐑦 𐑤𐑦𐑐! ·𐑥𐑭𐑕𐑩 ·𐑕𐑑𐑳𐑚 𐑕𐑱 𐑛𐑨𐑑 𐑿 𐑒𐑨𐑯 𐑓𐑦𐑤 𐑘𐑹 𐑛𐑨𐑥 𐑚𐑧𐑤𐑦𐑟 𐑳𐑐 𐑑 𐑛𐑦 𐑣𐑨𐑗𐑦𐑙𐑟, 𐑚𐑳𐑑 𐑚𐑲 ·𐑜𐑹! 𐑿 𐑥𐑳𐑕𐑑 𐑕𐑑𐑪𐑐 𐑛𐑨𐑑 𐑛𐑨𐑥 𐑮𐑨𐑒𐑩𐑑!”

“𐑒𐑫𐑒,” 𐑣𐑽 𐑦𐑯𐑑𐑼𐑐𐑴𐑟𐑛 ·𐑕𐑑𐑳𐑚, 𐑩𐑒𐑳𐑥𐑐𐑩𐑯𐑦𐑦𐑙 𐑞 𐑢𐑻𐑛 𐑢𐑦𐑞 𐑩 𐑕𐑳𐑛𐑩𐑯 𐑕𐑤𐑨𐑐 𐑪𐑯 𐑞 𐑖𐑴𐑤𐑛𐑼,—“𐑒𐑫𐑒! 𐑢𐑲, 𐑛𐑨𐑥 𐑘𐑹 𐑲𐑟, 𐑿 𐑥𐑳𐑕𐑩𐑯𐑑 𐑕𐑢𐑺 𐑞𐑨𐑑 𐑢𐑱 𐑢𐑧𐑯 𐑿'𐑼 𐑐𐑮𐑰𐑗𐑦𐑙. 𐑞𐑨𐑑'𐑕 𐑯𐑴 𐑢𐑱 𐑑 𐑒𐑩𐑯𐑝𐑻𐑑 𐑕𐑦𐑯𐑼𐑟, 𐑒𐑫𐑒!”

“𐑣𐑵 𐑛𐑨𐑑? 𐑛𐑧𐑯 𐑐𐑮𐑰𐑗 𐑑 𐑣𐑦𐑥 𐑘𐑹𐑕𐑧𐑤𐑓,” 𐑕𐑳𐑤𐑩𐑯𐑤𐑦 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑜𐑴.

“𐑯𐑴, 𐑒𐑫𐑒; 𐑜𐑴 𐑪𐑯, 𐑜𐑴 𐑪𐑯.”

“𐑢𐑧𐑤, 𐑛𐑧𐑯, ·𐑚𐑧𐑤𐑳𐑚𐑛 𐑓𐑧𐑤𐑴-𐑒𐑮𐑦𐑑𐑼𐑟:”—

“𐑮𐑲𐑑!” 𐑦𐑒𐑕𐑒𐑤𐑱𐑥𐑛 ·𐑕𐑑𐑳𐑚, 𐑩𐑐𐑮𐑵𐑝𐑦𐑙𐑤𐑦, “𐑒𐑴𐑒𐑕 '𐑩𐑥 𐑑 𐑦𐑑; 𐑑𐑮𐑲 𐑞𐑨𐑑,” 𐑯 ·𐑓𐑤𐑰𐑕 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑛.

“𐑛𐑵 𐑿 𐑦𐑟 𐑷𐑤 𐑖𐑸𐑒𐑕, 𐑯 𐑚𐑲 𐑯𐑭𐑑𐑫𐑼 𐑢𐑻𐑦 𐑢𐑹𐑱𐑖𐑩𐑕, 𐑘𐑧𐑑 𐑲 ·𐑟𐑱 𐑑 𐑿, 𐑓𐑧𐑤𐑴-𐑒𐑮𐑦𐑑𐑼𐑟, 𐑛𐑨𐑑 𐑛𐑨𐑑 𐑢𐑹𐑱𐑖𐑩𐑕𐑯𐑩𐑕—'𐑑𐑪𐑐 𐑛𐑨𐑑 𐑛𐑨𐑥 𐑕𐑤𐑨𐑐𐑦𐑯' 𐑪𐑚 𐑛𐑦 𐑑𐑱𐑤! 𐑣𐑬 𐑿 𐑑𐑦𐑙𐑒 𐑑 𐑣𐑽, 𐑕𐑐𐑴𐑟 𐑿 𐑒𐑰𐑐 𐑳𐑐 𐑕𐑳𐑗 𐑩 𐑛𐑨𐑥 𐑕𐑤𐑨𐑐𐑦𐑯' 𐑯 𐑚𐑲𐑑𐑦𐑯' 𐑛𐑺?”

“𐑒𐑫𐑒,” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, 𐑒𐑪𐑤𐑼𐑦𐑙 𐑣𐑦𐑥, “𐑲 𐑢𐑴𐑯𐑑 𐑣𐑨𐑝 𐑞𐑨𐑑 𐑕𐑢𐑺𐑦𐑙. 𐑑𐑷𐑒 𐑑 '𐑩𐑥 𐑡𐑧𐑯𐑑𐑩𐑤𐑥𐑩𐑯𐑤𐑦.”

𐑢𐑳𐑯𐑕 𐑥𐑹 𐑞 𐑕𐑻𐑥𐑩𐑯 𐑐𐑮𐑴𐑕𐑰𐑛𐑩𐑛.

“𐑘𐑹 𐑢𐑹𐑱𐑖𐑩𐑕𐑯𐑩𐑕, 𐑓𐑧𐑤𐑴-𐑒𐑮𐑦𐑑𐑼𐑟, 𐑲 𐑛𐑴𐑯𐑑 𐑚𐑤𐑱𐑥 𐑘𐑰 𐑕𐑴 𐑥𐑳𐑗 𐑓; 𐑛𐑨𐑑 𐑦𐑟 𐑯𐑭𐑑𐑫𐑼, 𐑯 𐑒𐑭𐑯𐑑 𐑚𐑰 𐑣𐑧𐑤𐑐𐑑; 𐑚𐑳𐑑 𐑑 𐑜𐑪𐑚𐑼𐑯 𐑛𐑨𐑑 𐑢𐑦𐑒𐑩𐑛 𐑯𐑭𐑑𐑫𐑼, 𐑛𐑨𐑑 𐑦𐑟 𐑛𐑦 𐑐𐑲𐑯𐑑. 𐑿 𐑦𐑟 𐑖𐑸𐑒𐑕, 𐑕𐑸𐑑𐑦𐑯; 𐑚𐑳𐑑 𐑦𐑓 𐑿 𐑜𐑪𐑚𐑼𐑯 𐑛𐑦 𐑖𐑸𐑒 𐑦𐑯 𐑿, 𐑢𐑲 𐑛𐑧𐑯 𐑿 𐑚𐑰 𐑱𐑯𐑡𐑩𐑤; 𐑓 𐑷𐑤 𐑱𐑯𐑡𐑩𐑤 𐑦𐑟 𐑯𐑪𐑑'𐑦𐑙 𐑥𐑹 𐑛𐑨𐑯 𐑛𐑦 𐑖𐑸𐑒 𐑢𐑧𐑤 𐑜𐑪𐑚𐑼𐑯𐑛. 𐑯𐑬, 𐑤𐑫𐑒 𐑣𐑽, 𐑚𐑮𐑧𐑛'𐑼𐑩𐑯, 𐑡𐑳𐑕𐑑 𐑑𐑮𐑲 𐑢𐑪𐑯𐑕𐑑 𐑑 𐑚𐑰 ·𐑕𐑦𐑚𐑦𐑤, 𐑩 𐑣𐑧𐑤𐑐𐑦𐑙 𐑘𐑹𐑕𐑧𐑤𐑚𐑕 𐑓𐑮𐑪𐑥 𐑛𐑨𐑑 𐑢𐑱𐑤. 𐑛𐑴𐑯𐑑 𐑚𐑰 𐑑𐑺𐑦𐑯' 𐑛𐑦 𐑚𐑤𐑳𐑚𐑼 𐑬𐑑 𐑘𐑹 𐑯𐑱𐑚𐑼'𐑟 𐑥𐑬𐑑, 𐑲 𐑕𐑱. 𐑦𐑟 𐑯𐑪𐑑 𐑢𐑳𐑯 𐑖𐑸𐑒 𐑛𐑵𐑛 𐑮𐑲𐑑 𐑨𐑟 𐑑𐑪𐑛𐑼 𐑑 𐑛𐑨𐑑 𐑢𐑱𐑤? 𐑯, 𐑚𐑲 ·𐑜𐑹, 𐑯𐑳𐑯 𐑪𐑯 𐑿 𐑣𐑨𐑟 𐑛𐑦 𐑮𐑲𐑑 𐑑 𐑛𐑨𐑑 𐑢𐑱𐑤; 𐑛𐑨𐑑 𐑢𐑱𐑤 𐑚𐑦𐑤𐑪𐑙 𐑑 𐑕𐑳𐑥 𐑢𐑳𐑯 𐑧𐑤𐑕. 𐑲 𐑯𐑴 𐑕𐑳𐑥 𐑴' 𐑿 𐑣𐑨𐑟 𐑚𐑧𐑮𐑦 𐑚𐑮𐑦𐑜 𐑥𐑬𐑑, 𐑚𐑮𐑦𐑜𐑼 𐑛𐑨𐑯 𐑴𐑛𐑼𐑟; 𐑚𐑳𐑑 𐑛𐑧𐑯 𐑛𐑦 𐑚𐑮𐑦𐑜 𐑥𐑬𐑑𐑕 𐑕𐑳𐑥𐑑𐑲𐑥𐑟 𐑣𐑨𐑟 𐑛𐑦 𐑕𐑥𐑷𐑤 𐑚𐑧𐑤𐑦𐑟; 𐑕𐑴 𐑛𐑨𐑑 𐑛𐑦 𐑚𐑮𐑦𐑜𐑯𐑩𐑕 𐑝 𐑛𐑦 𐑥𐑬𐑑 𐑦𐑟 𐑯𐑪𐑑 𐑑 ·𐑕𐑢𐑷𐑤𐑼 ·𐑢𐑦𐑛, 𐑚𐑳𐑑 𐑑 𐑚𐑦𐑑 𐑪𐑓 𐑛𐑦 𐑚𐑤𐑳𐑚𐑼 𐑓 𐑛𐑦 𐑕𐑥𐑷𐑤 𐑓𐑮𐑲 𐑪𐑚 𐑖𐑸𐑒𐑕, 𐑛𐑨𐑑 𐑒𐑭𐑯𐑑 𐑜𐑧𐑑 𐑦𐑯𐑑𐑵 𐑛𐑦 𐑕𐑒𐑮𐑵𐑡 𐑑 𐑣𐑧𐑤𐑐 𐑛𐑧𐑥𐑕𐑧𐑤𐑝𐑟.”

“𐑢𐑧𐑤 𐑛𐑳𐑯, 𐑴𐑤𐑛 ·𐑓𐑤𐑰𐑕!” 𐑒𐑮𐑲𐑛 ·𐑕𐑑𐑳𐑚, “𐑞𐑨𐑑'𐑕 ·𐑒𐑮𐑦𐑕𐑑𐑦𐑨𐑯𐑦𐑑𐑦; 𐑜𐑴 𐑪𐑯.”

“𐑯𐑴 𐑿𐑕 𐑜𐑴𐑦𐑯' 𐑪𐑯; 𐑛𐑦 𐑛𐑨𐑥 𐑢𐑦𐑤𐑱𐑯𐑟 𐑢𐑦𐑤 𐑒𐑰𐑐 𐑩 𐑕𐑒𐑴𐑳𐑜𐑦𐑯' 𐑯 𐑕𐑤𐑨𐑐𐑦𐑯' 𐑰𐑗 𐑴𐑛𐑼, ·𐑥𐑭𐑕𐑩 ·𐑕𐑑𐑳𐑚; 𐑛𐑱 𐑛𐑴𐑯𐑑 𐑣𐑽 𐑢𐑳𐑯 𐑢𐑻𐑛; 𐑯𐑴 𐑿𐑕 𐑩-𐑐𐑮𐑰𐑗𐑦𐑯' 𐑑 𐑕𐑳𐑗 𐑛𐑨𐑥 𐑜'𐑳𐑑𐑩𐑯𐑟 𐑨𐑟 𐑿 𐑒𐑷𐑤 '𐑩𐑥, 𐑑𐑦𐑤 𐑛𐑺 𐑚𐑧𐑤𐑦𐑟 𐑦𐑟 𐑓𐑫𐑤, 𐑯 𐑛𐑺 𐑚𐑧𐑤𐑦𐑟 𐑦𐑟 𐑚𐑪𐑑𐑩𐑥𐑤𐑩𐑕; 𐑯 𐑢𐑧𐑯 𐑛𐑱 𐑛𐑵 𐑜𐑧𐑑 '𐑩𐑥 𐑓𐑫𐑤, 𐑛𐑱 𐑢𐑴𐑯𐑑 𐑣𐑽 𐑿 𐑛𐑧𐑯; 𐑓 𐑛𐑧𐑯 𐑛𐑱 𐑕𐑦𐑙𐑒 𐑦𐑯 𐑛𐑦 𐑕𐑰, 𐑜𐑴 𐑓𐑭𐑕𐑑 𐑑 𐑕𐑤𐑰𐑐 𐑪𐑯 𐑛𐑦 𐑒𐑪𐑮𐑩𐑤, 𐑯 𐑒𐑭𐑯𐑑 𐑣𐑽 𐑯𐑪𐑑'𐑦𐑙 𐑨𐑑 𐑷𐑤, 𐑯𐑴 𐑥𐑹, 𐑓 ·𐑧𐑚𐑼 𐑯 ·𐑧𐑚𐑼.”

“𐑩𐑐𐑪𐑯 𐑥𐑲 𐑕𐑴𐑤, 𐑲 𐑨𐑥 𐑩𐑚𐑬𐑑 𐑝 𐑞 𐑕𐑱𐑥 𐑩𐑐𐑦𐑯𐑘𐑩𐑯; 𐑕𐑴 𐑜𐑦𐑝 𐑞 𐑚𐑧𐑯𐑦𐑛𐑦𐑒𐑖𐑩𐑯, ·𐑓𐑤𐑰𐑕, 𐑯 𐑲'𐑤 𐑩𐑢𐑱 𐑑 𐑥𐑲 𐑕𐑳𐑐𐑼.”

𐑩𐑐𐑪𐑯 𐑞𐑦𐑕, ·𐑓𐑤𐑰𐑕, 𐑣𐑴𐑤𐑛𐑦𐑙 𐑚𐑴𐑔 𐑣𐑨𐑯𐑛𐑟 𐑴𐑝𐑼 𐑞 𐑓𐑦𐑖𐑦 𐑥𐑪𐑚, 𐑮𐑱𐑟𐑛 𐑣𐑦𐑟 𐑖𐑮𐑦𐑤 𐑝𐑶𐑕, 𐑯 𐑒𐑮𐑲𐑛—

“𐑒𐑳𐑕𐑩𐑛 𐑓𐑧𐑤𐑴-𐑒𐑮𐑦𐑑𐑼𐑟! 𐑒𐑦𐑒 𐑳𐑐 𐑛𐑦 𐑛𐑨𐑥𐑛𐑩𐑕𐑑 𐑮𐑴 𐑨𐑟 𐑧𐑝𐑼 𐑿 𐑒𐑨𐑯; 𐑓𐑦𐑤 𐑘𐑹 𐑛𐑨𐑥' 𐑚𐑧𐑤𐑦𐑟 '𐑑𐑦𐑤 𐑛𐑱 𐑚𐑳𐑕𐑑—𐑯 𐑛𐑧𐑯 𐑛𐑲.”

“𐑯𐑬, 𐑒𐑫𐑒,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑮𐑦𐑟𐑿𐑥𐑦𐑙 𐑣𐑦𐑟 𐑕𐑳𐑐𐑼 𐑨𐑑 𐑞 𐑒𐑨𐑐𐑕𐑑𐑩𐑯; “𐑕𐑑𐑨𐑯𐑛 𐑡𐑳𐑕𐑑 𐑢𐑺 𐑿 𐑕𐑑𐑫𐑛 𐑚𐑦𐑓𐑹, 𐑞𐑺, 𐑴𐑝𐑼 𐑩𐑜𐑱𐑯𐑕𐑑 𐑥𐑰, 𐑯 𐑐𐑱 𐑐𐑼𐑑𐑦𐑒𐑿𐑤𐑼 𐑩𐑑𐑧𐑯𐑖𐑩𐑯.”

“𐑷𐑤 𐑛𐑧𐑯𐑖𐑩𐑯,” 𐑕𐑧𐑛 ·𐑓𐑤𐑰𐑕, 𐑩𐑜𐑱𐑯 𐑕𐑑𐑵𐑐𐑦𐑙 𐑴𐑝𐑼 𐑩𐑐𐑪𐑯 𐑣𐑦𐑟 𐑑𐑪𐑙𐑟 𐑦𐑯 𐑞 𐑛𐑦𐑟𐑲𐑼𐑛 𐑐𐑩𐑟𐑦𐑖𐑩𐑯.

“𐑢𐑧𐑤,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑣𐑧𐑤𐑐𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑓𐑮𐑰𐑤𐑦 𐑥𐑰𐑯𐑢𐑲𐑤; “𐑲 𐑖𐑨𐑤 𐑯𐑬 𐑜𐑴 𐑚𐑨𐑒 𐑑 𐑞 𐑕𐑳𐑚𐑡𐑧𐑒𐑑 𐑝 𐑞𐑦𐑕 𐑕𐑑𐑱𐑒. 𐑦𐑯 𐑞 𐑓𐑻𐑕𐑑 𐑐𐑤𐑱𐑕, 𐑣𐑬 𐑴𐑤𐑛 𐑸 𐑿, 𐑒𐑫𐑒?”

“𐑢𐑪𐑑 𐑛𐑨𐑑 𐑛𐑵 ·𐑢𐑦𐑛 𐑛𐑦 '𐑑𐑰𐑒,” 𐑕𐑧𐑛 𐑞 𐑴𐑤𐑛 𐑚𐑤𐑨𐑒, 𐑑𐑧𐑕𐑑𐑦𐑤𐑦.

“𐑕𐑲𐑤𐑩𐑯𐑕! 𐑣𐑬 𐑴𐑤𐑛 𐑸 𐑿, 𐑒𐑫𐑒?”

“'𐑚𐑬𐑑 𐑯𐑲𐑯𐑑𐑦, 𐑛𐑱 𐑕𐑱,” 𐑣𐑰 𐑜𐑤𐑵𐑥𐑦𐑤𐑦 𐑥𐑳𐑑𐑼𐑛.

“𐑯 𐑿 𐑣𐑨𐑝 𐑤𐑦𐑝𐑛 𐑦𐑯 𐑞𐑦𐑕 𐑢𐑻𐑤𐑛 𐑣𐑸𐑛 𐑩𐑐𐑪𐑯 𐑢𐑳𐑯 𐑣𐑳𐑯𐑛𐑮𐑩𐑛 𐑘𐑽𐑟, 𐑒𐑫𐑒, 𐑯 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑘𐑧𐑑 𐑣𐑬 𐑑 𐑒𐑫𐑒 𐑩 𐑢𐑱𐑤-𐑕𐑑𐑱𐑒?” 𐑮𐑨𐑐𐑦𐑛𐑤𐑦 𐑚𐑴𐑤𐑑𐑦𐑙 𐑩𐑯𐑳𐑞𐑼 𐑥𐑬𐑔𐑓𐑫𐑤 𐑨𐑑 𐑞 𐑤𐑭𐑕𐑑 𐑢𐑻𐑛, 𐑕𐑴 𐑞𐑨𐑑 𐑥𐑹𐑕𐑩𐑤 𐑕𐑰𐑥𐑛 𐑩 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑱𐑖𐑩𐑯 𐑝 𐑞 𐑒𐑢𐑧𐑕𐑗𐑩𐑯. “𐑢𐑺 𐑢𐑻 𐑿 𐑚𐑹𐑯, 𐑒𐑫𐑒?”

“'𐑣𐑲𐑯𐑛 𐑛𐑦 𐑣𐑨𐑗𐑢𐑱, 𐑦𐑯 𐑓𐑧𐑮𐑦-𐑚𐑴𐑑, 𐑜𐑴𐑦𐑯' 𐑴𐑚𐑼 𐑛𐑦 ·𐑮𐑴𐑯𐑴𐑒.”

“𐑚𐑹𐑯 𐑦𐑯 𐑩 𐑓𐑧𐑮𐑦-𐑚𐑴𐑑! 𐑞𐑨𐑑'𐑕 𐑒𐑢𐑽, 𐑑𐑵. 𐑚𐑳𐑑 𐑲 𐑢𐑪𐑯𐑑 𐑑 𐑯𐑴 𐑢𐑪𐑑 𐑒𐑳𐑯𐑑𐑮𐑦 𐑿 𐑢𐑻 𐑚𐑹𐑯 𐑦𐑯, 𐑒𐑫𐑒!”

“𐑛𐑦𐑛𐑩𐑯𐑑 𐑲 𐑕𐑱 𐑛𐑦 ·𐑮𐑴𐑯𐑴𐑒 𐑒𐑳𐑯𐑑𐑮𐑦?” 𐑣𐑰 𐑒𐑮𐑲𐑛 𐑖𐑸𐑐𐑤𐑦.

“𐑯𐑴, 𐑿 𐑛𐑦𐑛𐑩𐑯𐑑, 𐑒𐑫𐑒; 𐑚𐑳𐑑 𐑲'𐑤 𐑑𐑧𐑤 𐑿 𐑢𐑪𐑑 𐑲'𐑥 𐑒𐑳𐑥𐑦𐑙 𐑑, 𐑒𐑫𐑒. 𐑿 𐑥𐑳𐑕𐑑 𐑜𐑴 𐑣𐑴𐑥 𐑯 𐑚𐑰 𐑚𐑹𐑯 𐑴𐑝𐑼 𐑩𐑜𐑱𐑯; 𐑿 𐑛𐑴𐑯𐑑 𐑯𐑴 𐑣𐑬 𐑑 𐑒𐑫𐑒 𐑩 𐑢𐑱𐑤-𐑕𐑑𐑱𐑒 𐑘𐑧𐑑.”

“·𐑚𐑮𐑰𐑧𐑕 𐑥𐑲 𐑕𐑴𐑤, 𐑦𐑓 𐑲 𐑒𐑫𐑒 𐑯𐑴𐑛𐑼 𐑢𐑳𐑯,” 𐑣𐑰 𐑜𐑮𐑬𐑤𐑛, 𐑨𐑙𐑜𐑮𐑦𐑤𐑦, 𐑑𐑻𐑯𐑦𐑙 𐑮𐑬𐑯𐑛 𐑑 𐑛𐑦𐑐𐑸𐑑.

“𐑒𐑳𐑥 𐑚𐑨𐑒, 𐑒𐑫𐑒;—𐑣𐑽, 𐑣𐑨𐑯𐑛 𐑥𐑰 𐑞𐑴𐑟 𐑑𐑪𐑙𐑟;—𐑯𐑬 𐑑𐑱𐑒 𐑞𐑨𐑑 𐑚𐑦𐑑 𐑝 𐑕𐑑𐑱𐑒 𐑞𐑺, 𐑯 𐑑𐑧𐑤 𐑥𐑰 𐑦𐑓 𐑿 𐑔𐑦𐑙𐑒 𐑞𐑨𐑑 𐑕𐑑𐑱𐑒 𐑒𐑫𐑒𐑑 𐑨𐑟 𐑦𐑑 𐑖𐑵𐑛 𐑚𐑰? 𐑑𐑱𐑒 𐑦𐑑, 𐑲 𐑕𐑱”—𐑣𐑴𐑤𐑛𐑦𐑙 𐑞 𐑑𐑪𐑙𐑟 𐑑𐑩𐑢𐑹𐑛𐑟 𐑣𐑦𐑥—“𐑑𐑱𐑒 𐑦𐑑, 𐑯 𐑑𐑱𐑕𐑑 𐑦𐑑.”

𐑓𐑱𐑯𐑑𐑤𐑦 𐑕𐑥𐑨𐑒𐑦𐑙 𐑣𐑦𐑟 𐑢𐑦𐑞𐑼𐑛 𐑤𐑦𐑐𐑕 𐑴𐑝𐑼 𐑦𐑑 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑, 𐑞 𐑴𐑤𐑛 𐑯𐑰𐑜𐑮𐑴 𐑥𐑳𐑑𐑼𐑛, “𐑚𐑧𐑕𐑑 𐑒𐑫𐑒𐑑 '𐑑𐑰𐑒 𐑲 ·𐑧𐑚𐑼 𐑑𐑱𐑕𐑑; ·𐑡𐑵𐑟𐑦, 𐑚𐑧𐑮𐑦 ·𐑡𐑵𐑟𐑦.”

“𐑒𐑫𐑒,” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚, 𐑕𐑒𐑢𐑺𐑦𐑙 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑢𐑳𐑯𐑕 𐑥𐑹; “𐑛𐑵 𐑿 𐑚𐑦𐑤𐑪𐑙 𐑑 𐑞 𐑗𐑻𐑗?”

“𐑐𐑭𐑕𐑑 𐑢𐑳𐑯 𐑢𐑳𐑯𐑕 𐑦𐑯 ·𐑒𐑱𐑐-·𐑛𐑬𐑯,” 𐑕𐑧𐑛 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑕𐑳𐑤𐑩𐑯𐑤𐑦.

“𐑯 𐑿 𐑣𐑨𐑝 𐑢𐑳𐑯𐑕 𐑦𐑯 𐑘𐑹 𐑤𐑲𐑓 𐑐𐑭𐑕𐑑 𐑩 𐑣𐑴𐑤𐑦 𐑗𐑻𐑗 𐑦𐑯 ·𐑒𐑱𐑐-·𐑑𐑬𐑯, 𐑢𐑺 𐑿 𐑛𐑬𐑑𐑤𐑩𐑕 𐑴𐑝𐑼𐑣𐑻𐑛 𐑩 𐑣𐑴𐑤𐑦 𐑐𐑸𐑟𐑩𐑯 𐑩𐑛𐑮𐑧𐑕𐑦𐑙 𐑣𐑦𐑟 𐑣𐑽𐑼𐑟 𐑨𐑟 𐑣𐑦𐑟 𐑚𐑦𐑤𐑳𐑝𐑩𐑛 𐑓𐑧𐑤𐑴-𐑒𐑮𐑰𐑗𐑼𐑟, 𐑣𐑨𐑝 𐑿, 𐑒𐑫𐑒! 𐑯 𐑘𐑧𐑑 𐑿 𐑒𐑳𐑥 𐑣𐑽, 𐑯 𐑑𐑧𐑤 𐑥𐑰 𐑕𐑳𐑗 𐑩 𐑛𐑮𐑧𐑛𐑓𐑩𐑤 𐑤𐑲 𐑨𐑟 𐑿 𐑛𐑦𐑛 𐑡𐑳𐑕𐑑 𐑯𐑬, 𐑱?” 𐑕𐑧𐑛 ·𐑕𐑑𐑳𐑚. “𐑢𐑺 𐑛𐑵 𐑿 𐑦𐑒𐑕𐑐𐑧𐑒𐑑 𐑑 𐑜𐑴 𐑑, 𐑒𐑫𐑒?”

“𐑜𐑴 𐑑 𐑚𐑧𐑛 𐑚𐑧𐑮𐑦 𐑕𐑵𐑯,” 𐑣𐑰 𐑥𐑳𐑥𐑚𐑩𐑤𐑛, 𐑣𐑭𐑓-𐑑𐑻𐑯𐑦𐑙 𐑨𐑟 𐑣𐑰 𐑕𐑐𐑴𐑒.

“𐑩𐑝𐑭𐑕𐑑! 𐑣𐑰𐑝 𐑑! 𐑲 𐑥𐑰𐑯 𐑢𐑧𐑯 𐑿 𐑛𐑲, 𐑒𐑫𐑒. 𐑦𐑑'𐑕 𐑩𐑯 𐑷𐑓𐑩𐑤 𐑒𐑢𐑧𐑕𐑗𐑩𐑯. 𐑯𐑬 𐑢𐑪𐑑'𐑕 𐑘𐑹 𐑭𐑯𐑕𐑼?”

“𐑢𐑧𐑯 𐑛𐑦𐑕 𐑴𐑤𐑛 𐑚𐑮𐑨𐑒 𐑥𐑨𐑯 𐑛𐑲𐑟,” 𐑕𐑧𐑛 𐑞 𐑯𐑰𐑜𐑮𐑴 𐑕𐑤𐑴𐑤𐑦, 𐑗𐑱𐑯𐑡𐑦𐑙 𐑣𐑦𐑟 𐑣𐑴𐑤 𐑺 𐑯 𐑛𐑦𐑥𐑰𐑯𐑼, “𐑣𐑰 𐑣𐑦𐑟𐑕𐑧𐑤𐑓 𐑢𐑴𐑯𐑑 𐑜𐑴 𐑯𐑴𐑢𐑺; 𐑚𐑳𐑑 𐑕𐑳𐑥 ·𐑚𐑮𐑰𐑧𐑕𐑑 𐑱𐑯𐑡𐑩𐑤 𐑢𐑦𐑤 𐑒𐑳𐑥 𐑯 𐑓𐑧𐑗 𐑣𐑦𐑥.”

“𐑓𐑧𐑗 𐑣𐑦𐑥? 𐑣𐑬? 𐑦𐑯 𐑩 𐑒𐑴𐑗 𐑯 𐑓𐑹, 𐑨𐑟 𐑞𐑱 𐑓𐑧𐑗𐑑 ·𐑦𐑤𐑲𐑡𐑩? 𐑯 𐑓𐑧𐑗 𐑣𐑦𐑥 𐑢𐑺?”

“𐑳𐑐 𐑛𐑸,” 𐑕𐑧𐑛 ·𐑓𐑤𐑰𐑕, 𐑣𐑴𐑤𐑛𐑦𐑙 𐑣𐑦𐑟 𐑑𐑪𐑙𐑟 𐑕𐑑𐑮𐑱𐑑 𐑴𐑝𐑼 𐑣𐑦𐑟 𐑣𐑧𐑛, 𐑯 𐑒𐑰𐑐𐑦𐑙 𐑦𐑑 𐑞𐑺 𐑝𐑧𐑮𐑦 𐑕𐑪𐑤𐑩𐑥𐑤𐑦.

“𐑕𐑴, 𐑞𐑧𐑯, 𐑿 𐑦𐑒𐑕𐑐𐑧𐑒𐑑 𐑑 𐑜𐑴 𐑳𐑐 𐑦𐑯𐑑𐑵 𐑬𐑼 𐑥𐑱𐑯-𐑑𐑪𐑐, 𐑛𐑵 𐑿, 𐑒𐑫𐑒, 𐑢𐑧𐑯 𐑿 𐑸 𐑛𐑧𐑛? 𐑚𐑳𐑑 𐑛𐑴𐑯𐑑 𐑿 𐑯𐑴 𐑞 𐑣𐑲𐑼 𐑿 𐑒𐑤𐑲𐑥, 𐑞 𐑒𐑴𐑤𐑛𐑼 𐑦𐑑 𐑜𐑧𐑑𐑕? 𐑥𐑱𐑯-𐑑𐑪𐑐, 𐑱?”

“𐑛𐑦𐑛𐑩𐑯𐑑 𐑕𐑱 𐑛𐑨𐑑 𐑑'𐑷𐑤,” 𐑕𐑧𐑛 ·𐑓𐑤𐑰𐑕, 𐑩𐑜𐑱𐑯 𐑦𐑯 𐑞 𐑕𐑳𐑤𐑒𐑕.

“𐑿 𐑕𐑧𐑛 𐑳𐑐 𐑞𐑺, 𐑛𐑦𐑛𐑩𐑯𐑑 𐑿? 𐑯 𐑯𐑬 𐑤𐑫𐑒 𐑘𐑹𐑕𐑧𐑤𐑓, 𐑯 𐑕𐑰 𐑢𐑺 𐑘𐑹 𐑑𐑪𐑙𐑟 𐑸 𐑐𐑶𐑯𐑑𐑦𐑙. 𐑚𐑳𐑑, 𐑐𐑼𐑣𐑨𐑐𐑕 𐑿 𐑦𐑒𐑕𐑐𐑧𐑒𐑑 𐑑 𐑜𐑧𐑑 𐑦𐑯𐑑𐑵 𐑣𐑧𐑝𐑩𐑯 𐑚𐑲 𐑒𐑮𐑷𐑤𐑦𐑙 𐑔𐑮𐑵 𐑞 𐑤𐑳𐑚𐑼'𐑟 𐑣𐑴𐑤, 𐑒𐑫𐑒; 𐑚𐑳𐑑, 𐑯𐑴, 𐑯𐑴, 𐑒𐑫𐑒, 𐑿 𐑛𐑴𐑯𐑑 𐑜𐑧𐑑 𐑞𐑺, 𐑦𐑒𐑕𐑧𐑐𐑑 𐑿 𐑜𐑴 𐑞 𐑮𐑧𐑜𐑿𐑤𐑼 𐑢𐑱, 𐑮𐑬𐑯𐑛 𐑚𐑲 𐑞 𐑮𐑦𐑜𐑦𐑙. 𐑦𐑑'𐑕 𐑩 𐑑𐑦𐑒𐑩𐑤𐑦𐑖 𐑚𐑦𐑟𐑯𐑩𐑕, 𐑚𐑳𐑑 𐑥𐑳𐑕𐑑 𐑚𐑰 𐑛𐑳𐑯, 𐑹 𐑧𐑤𐑕 𐑦𐑑'𐑕 𐑯𐑴 𐑜𐑴. 𐑚𐑳𐑑 𐑯𐑳𐑯 𐑝 𐑳𐑕 𐑸 𐑦𐑯 𐑣𐑧𐑝𐑩𐑯 𐑘𐑧𐑑. 𐑛𐑮𐑪𐑐 𐑘𐑹 𐑑𐑪𐑙𐑟, 𐑒𐑫𐑒, 𐑯 𐑣𐑽 𐑥𐑲 𐑹𐑛𐑼𐑟. 𐑛𐑵 𐑘𐑰 𐑣𐑽? 𐑣𐑴𐑤𐑛 𐑘𐑹 𐑣𐑨𐑑 𐑦𐑯 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛, 𐑯 𐑒𐑤𐑨𐑐 𐑑𐑳𐑞𐑼 𐑩'𐑑𐑪𐑐 𐑝 𐑘𐑹 𐑣𐑸𐑑, 𐑢𐑧𐑯 𐑲'𐑥 𐑜𐑦𐑝𐑦𐑙 𐑥𐑲 𐑹𐑛𐑼𐑟, 𐑒𐑫𐑒. 𐑢𐑪𐑑! 𐑞𐑨𐑑 𐑘𐑹 𐑣𐑸𐑑, 𐑞𐑺?—𐑞𐑨𐑑'𐑕 𐑘𐑹 𐑜𐑦𐑟𐑼𐑛! 𐑩𐑤𐑪𐑓𐑑! 𐑩𐑤𐑪𐑓𐑑!—𐑞𐑨𐑑'𐑕 𐑦𐑑—𐑯𐑬 𐑿 𐑣𐑨𐑝 𐑦𐑑. 𐑣𐑴𐑤𐑛 𐑦𐑑 𐑞𐑺 𐑯𐑬, 𐑯 𐑐𐑱 𐑩𐑑𐑧𐑯𐑖𐑩𐑯.”

“𐑷𐑤 '𐑛𐑧𐑯𐑖𐑩𐑯,” 𐑕𐑧𐑛 𐑞 𐑴𐑤𐑛 𐑚𐑤𐑨𐑒, 𐑢𐑦𐑞 𐑚𐑴𐑔 𐑣𐑨𐑯𐑛𐑟 𐑐𐑤𐑱𐑕𐑑 𐑨𐑟 𐑛𐑦𐑟𐑲𐑼𐑛, 𐑝𐑱𐑯𐑤𐑦 𐑮𐑦𐑜𐑩𐑤𐑦𐑙 𐑣𐑦𐑟 𐑜𐑮𐑦𐑟𐑩𐑤𐑛 𐑣𐑧𐑛, 𐑨𐑟 𐑦𐑓 𐑑 𐑜𐑧𐑑 𐑚𐑴𐑔 𐑽𐑟 𐑦𐑯 𐑓𐑮𐑳𐑯𐑑 𐑨𐑑 𐑢𐑳𐑯 𐑯 𐑞 𐑕𐑱𐑥 𐑑𐑲𐑥.

“𐑢𐑧𐑤 𐑞𐑧𐑯, 𐑒𐑫𐑒, 𐑿 𐑕𐑰 𐑞𐑦𐑕 𐑢𐑱𐑤-𐑕𐑑𐑱𐑒 𐑝 𐑘𐑹𐑟 𐑢𐑪𐑟 𐑕𐑴 𐑝𐑧𐑮𐑦 𐑚𐑨𐑛, 𐑞𐑨𐑑 𐑲 𐑣𐑨𐑝 𐑐𐑫𐑑 𐑦𐑑 𐑬𐑑 𐑝 𐑕𐑲𐑑 𐑨𐑟 𐑕𐑵𐑯 𐑨𐑟 𐑐𐑪𐑕𐑩𐑚𐑩𐑤; 𐑿 𐑕𐑰 𐑞𐑨𐑑, 𐑛𐑴𐑯𐑑 𐑿? 𐑢𐑧𐑤, 𐑓 𐑞 𐑓𐑿𐑗𐑼, 𐑢𐑧𐑯 𐑿 𐑒𐑫𐑒 𐑩𐑯𐑳𐑞𐑼 𐑢𐑱𐑤-𐑕𐑑𐑱𐑒 𐑓 𐑥𐑲 𐑐𐑮𐑲𐑝𐑩𐑑 𐑑𐑱𐑚𐑩𐑤 𐑣𐑽, 𐑞 𐑒𐑨𐑐𐑕𐑑𐑩𐑯, 𐑲'𐑤 𐑑𐑧𐑤 𐑿 𐑢𐑪𐑑 𐑑 𐑛𐑵 𐑕𐑴 𐑨𐑟 𐑯𐑪𐑑 𐑑 𐑕𐑐𐑶𐑤 𐑦𐑑 𐑚𐑲 𐑴𐑝𐑼𐑛𐑵𐑦𐑙. 𐑣𐑴𐑤𐑛 𐑞 𐑕𐑑𐑱𐑒 𐑦𐑯 𐑢𐑳𐑯 𐑣𐑨𐑯𐑛, 𐑯 𐑖𐑴 𐑩 𐑤𐑲𐑝 𐑒𐑴𐑤 𐑑 𐑦𐑑 𐑢𐑦𐑞 𐑞 𐑳𐑞𐑼; 𐑞𐑨𐑑 𐑛𐑳𐑯, 𐑛𐑦𐑖 𐑦𐑑; 𐑛'𐑘𐑰 𐑣𐑽? 𐑯 𐑯𐑬 𐑑-𐑥𐑪𐑮𐑴, 𐑒𐑫𐑒, 𐑢𐑧𐑯 𐑢𐑰 𐑸 𐑒𐑳𐑑𐑦𐑙 𐑦𐑯 𐑞 𐑓𐑦𐑖, 𐑚𐑰 𐑖𐑫𐑼 𐑿 𐑕𐑑𐑨𐑯𐑛 𐑚𐑲 𐑑 𐑜𐑧𐑑 𐑞 𐑑𐑦𐑐𐑕 𐑝 𐑣𐑦𐑟 𐑓𐑦𐑯𐑟; 𐑣𐑨𐑝 𐑞𐑧𐑥 𐑐𐑫𐑑 𐑦𐑯 𐑐𐑦𐑒𐑩𐑤. 𐑨𐑟 𐑓 𐑞 𐑧𐑯𐑛𐑟 𐑝 𐑞 𐑓𐑤𐑵𐑒𐑕, 𐑣𐑨𐑝 𐑞𐑧𐑥 𐑕𐑬𐑕𐑑, 𐑒𐑫𐑒. 𐑞𐑺, 𐑯𐑬 𐑘𐑰 𐑥𐑱 𐑜𐑴.”

𐑚𐑳𐑑 ·𐑓𐑤𐑰𐑕 𐑣𐑨𐑛 𐑣𐑸𐑛𐑤𐑦 𐑜𐑪𐑑 𐑔𐑮𐑰 𐑐𐑱𐑕𐑩𐑟 𐑪𐑓, 𐑢𐑧𐑯 𐑣𐑰 𐑢𐑪𐑟 𐑮𐑦𐑒𐑷𐑤𐑛.

“𐑒𐑫𐑒, 𐑜𐑦𐑝 𐑥𐑰 𐑒𐑳𐑑𐑤𐑩𐑑𐑕 𐑓 𐑕𐑳𐑐𐑼 𐑑-𐑥𐑪𐑮𐑴 𐑯𐑲𐑑 𐑦𐑯 𐑞 𐑥𐑦𐑛-𐑢𐑪𐑗. ·𐑛'𐑘𐑰 𐑣𐑽? 𐑩𐑢𐑱 𐑿 𐑕𐑱𐑤, 𐑞𐑧𐑯.—𐑣𐑩𐑤𐑴! 𐑕𐑑𐑪𐑐! 𐑥𐑱𐑒 𐑩 𐑚𐑬 𐑚𐑦𐑓𐑹 𐑿 𐑜𐑴.—𐑩𐑝𐑭𐑕𐑑 𐑣𐑰𐑝𐑦𐑙 𐑩𐑜𐑱𐑯! 𐑢𐑱𐑤-𐑚𐑷𐑤𐑟 𐑓 𐑚𐑮𐑧𐑒𐑓𐑩𐑕𐑑—𐑛𐑴𐑯𐑑 𐑓𐑼𐑜𐑧𐑑.”

“𐑢𐑦𐑖, 𐑚𐑲 ·𐑜𐑹! 𐑢𐑱𐑤 𐑰𐑑 𐑣𐑦𐑥, '𐑕𐑑𐑧𐑛 𐑝 𐑣𐑦𐑥 𐑰𐑑 𐑢𐑱𐑤. 𐑲'𐑥 ·𐑚𐑮𐑰𐑧𐑕𐑑 𐑦𐑓 𐑣𐑰 𐑱𐑯𐑑 𐑥𐑹 𐑝 𐑖𐑸𐑒 𐑛𐑨𐑯 ·𐑥𐑭𐑕𐑩 ·𐑖𐑸𐑒 𐑣𐑦𐑟𐑕𐑧𐑤𐑓,” 𐑥𐑳𐑑𐑼𐑛 𐑞 𐑴𐑤𐑛 𐑥𐑨𐑯, 𐑤𐑦𐑥𐑐𐑦𐑙 𐑩𐑢𐑱; 𐑢𐑦𐑞 𐑢𐑦𐑗 𐑕𐑱𐑡 𐑦𐑡𐑨𐑒𐑿𐑤𐑱𐑖𐑩𐑯 𐑣𐑰 𐑢𐑧𐑯𐑑 𐑑 𐑣𐑦𐑟 𐑣𐑨𐑥𐑩𐑒.
