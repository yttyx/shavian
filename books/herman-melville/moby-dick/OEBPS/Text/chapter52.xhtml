·𐑗𐑨𐑐𐑑𐑼 52. 𐑞 ·𐑨𐑤𐑚𐑩𐑑𐑮𐑪𐑕.

𐑕𐑬𐑔-𐑰𐑕𐑑𐑢𐑼𐑛 𐑓𐑮𐑪𐑥 𐑞 ·𐑒𐑱𐑐, 𐑪𐑓 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑑 ·𐑒𐑮𐑴𐑟𐑧𐑑𐑕, 𐑩 𐑜𐑫𐑛 𐑒𐑮𐑵𐑟𐑦𐑙 𐑜𐑮𐑬𐑯𐑛 𐑓 ·𐑮𐑲𐑑 ·𐑢𐑱𐑤𐑥𐑧𐑯, 𐑩 𐑕𐑱𐑤 𐑤𐑵𐑥𐑛 𐑩𐑣𐑧𐑛, 𐑞 ·𐑜𐑪𐑯𐑦 (·𐑨𐑤𐑚𐑩𐑑𐑮𐑪𐑕) 𐑚𐑲 𐑯𐑱𐑥. 𐑨𐑟 𐑖𐑰 𐑕𐑤𐑴𐑤𐑦 𐑛𐑮𐑵 𐑯𐑲, 𐑓𐑮𐑪𐑥 𐑥𐑲 𐑤𐑪𐑓𐑑𐑦 𐑐𐑻𐑗 𐑨𐑑 𐑞 𐑓𐑹-𐑥𐑭𐑕𐑑-𐑣𐑧𐑛, 𐑲 𐑣𐑨𐑛 𐑩 𐑜𐑫𐑛 𐑝𐑿 𐑝 𐑞𐑨𐑑 𐑕𐑲𐑑 𐑕𐑴 𐑮𐑦𐑥𐑸𐑒𐑩𐑚𐑩𐑤 𐑑 𐑩 𐑑𐑲𐑮𐑴 𐑦𐑯 𐑞 𐑓𐑸 𐑴𐑖𐑩𐑯 𐑓𐑦𐑖𐑼𐑦𐑟—𐑩 𐑢𐑱𐑤𐑼 𐑨𐑑 𐑕𐑰, 𐑯 𐑤𐑪𐑙 𐑨𐑚𐑕𐑩𐑯𐑑 𐑓𐑮𐑪𐑥 𐑣𐑴𐑥.

𐑨𐑟 𐑦𐑓 𐑞 𐑢𐑱𐑝𐑟 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑓𐑫𐑤𐑼𐑟, 𐑞𐑦𐑕 𐑒𐑮𐑭𐑓𐑑 𐑢𐑪𐑟 𐑚𐑤𐑰𐑗𐑑 𐑤𐑲𐑒 𐑞 𐑕𐑒𐑧𐑤𐑩𐑑𐑩𐑯 𐑝 𐑩 𐑕𐑑𐑮𐑨𐑯𐑛𐑩𐑛 𐑢𐑷𐑤𐑮𐑩𐑕. 𐑷𐑤 𐑛𐑬𐑯 𐑣𐑻 𐑕𐑲𐑛𐑟, 𐑞𐑦𐑕 𐑕𐑐𐑧𐑒𐑑𐑮𐑩𐑤 𐑩𐑐𐑽𐑩𐑯𐑕 𐑢𐑪𐑟 𐑑𐑮𐑱𐑕𐑑 𐑢𐑦𐑞 𐑤𐑪𐑙 𐑗𐑨𐑯𐑩𐑤𐑟 𐑝 𐑮𐑧𐑛𐑩𐑯𐑛 𐑮𐑳𐑕𐑑, 𐑢𐑲𐑤 𐑷𐑤 𐑣𐑻 𐑕𐑐𐑸𐑟 𐑯 𐑣𐑻 𐑮𐑦𐑜𐑦𐑙 𐑢𐑻 𐑤𐑲𐑒 𐑞 𐑔𐑦𐑒 𐑚𐑮𐑭𐑯𐑗𐑩𐑟 𐑝 𐑑𐑮𐑰𐑟 𐑓𐑻𐑛 𐑴𐑝𐑼 𐑢𐑦𐑞 𐑣𐑹-𐑓𐑮𐑪𐑕𐑑. 𐑴𐑯𐑤𐑦 𐑣𐑻 𐑤𐑴𐑼 𐑕𐑱𐑤𐑟 𐑢𐑻 𐑕𐑧𐑑. 𐑩 𐑢𐑲𐑤𐑛 𐑕𐑲𐑑 𐑦𐑑 𐑢𐑪𐑟 𐑑 𐑕𐑰 𐑣𐑻 𐑤𐑪𐑙-𐑚𐑽𐑛𐑩𐑛 𐑤𐑫𐑒-𐑬𐑑𐑕 𐑨𐑑 𐑞𐑴𐑟 𐑔𐑮𐑰 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟. 𐑞𐑱 𐑕𐑰𐑥𐑛 𐑒𐑤𐑨𐑛 𐑦𐑯 𐑞 𐑕𐑒𐑦𐑯𐑟 𐑝 𐑚𐑰𐑕𐑑𐑕, 𐑕𐑴 𐑑𐑹𐑯 𐑯 𐑚𐑦𐑐𐑨𐑗𐑑 𐑞 𐑮𐑱𐑥𐑩𐑯𐑑 𐑞𐑨𐑑 𐑣𐑨𐑛 𐑕𐑼𐑝𐑲𐑝𐑛 𐑯𐑽𐑤𐑦 𐑓𐑹 𐑘𐑽𐑟 𐑝 𐑒𐑮𐑵𐑟𐑦𐑙. 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑦𐑯 𐑲𐑼𐑯 𐑣𐑵𐑐𐑕 𐑯𐑱𐑤𐑛 𐑑 𐑞 𐑥𐑭𐑕𐑑, 𐑞𐑱 𐑕𐑢𐑱𐑛 𐑯 𐑕𐑢𐑳𐑙 𐑴𐑝𐑼 𐑩 𐑓𐑨𐑞𐑩𐑥𐑤𐑩𐑕 𐑕𐑰; 𐑯 𐑞𐑴, 𐑢𐑧𐑯 𐑞 𐑖𐑦𐑐 𐑕𐑤𐑴𐑤𐑦 𐑜𐑤𐑲𐑛𐑩𐑛 𐑒𐑤𐑴𐑕 𐑳𐑯𐑛𐑼 𐑬𐑼 𐑕𐑑𐑻𐑯, 𐑢𐑰 𐑕𐑦𐑒𐑕 𐑥𐑧𐑯 𐑦𐑯 𐑞 𐑺 𐑒𐑱𐑥 𐑕𐑴 𐑯𐑲 𐑑 𐑰𐑗 𐑳𐑞𐑼 𐑞𐑨𐑑 𐑢𐑰 𐑥𐑲𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑣𐑨𐑝 𐑤𐑰𐑐𐑑 𐑓𐑮𐑪𐑥 𐑞 𐑥𐑭𐑕𐑑-𐑣𐑧𐑛𐑟 𐑝 𐑢𐑳𐑯 𐑖𐑦𐑐 𐑑 𐑞𐑴𐑟 𐑝 𐑞 𐑳𐑞𐑼; 𐑘𐑧𐑑, 𐑞𐑴𐑟 𐑓𐑼𐑤𐑹𐑯-𐑤𐑫𐑒𐑦𐑙 𐑓𐑦𐑖𐑼𐑥𐑧𐑯, 𐑥𐑲𐑤𐑛𐑤𐑦 𐑲𐑦𐑙 𐑳𐑕 𐑨𐑟 𐑞𐑱 𐑐𐑭𐑕𐑑, 𐑕𐑧𐑛 𐑯𐑪𐑑 𐑢𐑳𐑯 𐑢𐑻𐑛 𐑑 𐑬𐑼 𐑴𐑯 𐑤𐑫𐑒-𐑬𐑑𐑕, 𐑢𐑲𐑤 𐑞 𐑒𐑢𐑹𐑑𐑼-𐑛𐑧𐑒 𐑣𐑱𐑤 𐑢𐑪𐑟 𐑚𐑰𐑦𐑙 𐑣𐑻𐑛 𐑓𐑮𐑪𐑥 𐑚𐑦𐑤𐑴.

“𐑖𐑦𐑐 𐑩𐑣𐑶! 𐑣𐑨𐑝 𐑘𐑰 𐑕𐑰𐑯 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤?”

𐑚𐑳𐑑 𐑨𐑟 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡 𐑒𐑨𐑐𐑑𐑦𐑯, 𐑤𐑰𐑯𐑦𐑙 𐑴𐑝𐑼 𐑞 𐑐𐑨𐑤𐑦𐑛 𐑚𐑫𐑤𐑢𐑼𐑒𐑕, 𐑢𐑪𐑟 𐑦𐑯 𐑞 𐑨𐑒𐑑 𐑝 𐑐𐑫𐑑𐑦𐑙 𐑣𐑦𐑟 𐑑𐑮𐑳𐑥𐑐𐑩𐑑 𐑑 𐑣𐑦𐑟 𐑥𐑬𐑔, 𐑦𐑑 𐑕𐑳𐑥𐑣𐑬 𐑓𐑧𐑤 𐑓𐑮𐑪𐑥 𐑣𐑦𐑟 𐑣𐑨𐑯𐑛 𐑦𐑯𐑑𐑵 𐑞 𐑕𐑰; 𐑯 𐑞 𐑢𐑦𐑯𐑛 𐑯𐑬 𐑮𐑲𐑟𐑦𐑙 𐑩𐑥𐑱𐑯, 𐑣𐑰 𐑦𐑯 𐑝𐑱𐑯 𐑕𐑑𐑮𐑴𐑝 𐑑 𐑥𐑱𐑒 𐑣𐑦𐑥𐑕𐑧𐑤𐑓 𐑣𐑻𐑛 𐑢𐑦𐑞𐑬𐑑 𐑦𐑑. 𐑥𐑰𐑯𐑑𐑲𐑥 𐑣𐑦𐑟 𐑖𐑦𐑐 𐑢𐑪𐑟 𐑕𐑑𐑦𐑤 𐑦𐑯𐑒𐑮𐑰𐑕𐑦𐑙 𐑞 𐑛𐑦𐑕𐑑𐑩𐑯𐑕 𐑚𐑦𐑑𐑢𐑰𐑯. 𐑢𐑲𐑤 𐑦𐑯 𐑝𐑺𐑾𐑕 𐑕𐑲𐑤𐑩𐑯𐑑 𐑢𐑱𐑟 𐑞 𐑕𐑰𐑥𐑧𐑯 𐑝 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛 𐑢𐑻 𐑦𐑝𐑦𐑯𐑕𐑦𐑙 𐑞𐑺 𐑩𐑚𐑟𐑻𐑝𐑩𐑯𐑕 𐑝 𐑞𐑦𐑕 𐑪𐑥𐑦𐑯𐑩𐑕 𐑦𐑯𐑕𐑦𐑛𐑩𐑯𐑑 𐑨𐑑 𐑞 𐑓𐑻𐑕𐑑 𐑥𐑽 𐑥𐑧𐑯𐑖𐑩𐑯 𐑝 𐑞 ·𐑢𐑲𐑑 ·𐑢𐑱𐑤'𐑟 𐑯𐑱𐑥 𐑑 𐑩𐑯𐑳𐑞𐑼 𐑖𐑦𐑐, ·𐑱𐑣𐑨𐑚 𐑓 𐑩 𐑥𐑴𐑥𐑧𐑯𐑑 𐑐𐑷𐑟𐑛; 𐑦𐑑 𐑷𐑤𐑥𐑴𐑕𐑑 𐑕𐑰𐑥𐑛 𐑨𐑟 𐑞𐑴 𐑣𐑰 𐑢𐑫𐑛 𐑣𐑨𐑝 𐑤𐑴𐑼𐑛 𐑩 𐑚𐑴𐑑 𐑑 𐑚𐑹𐑛 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼, 𐑣𐑨𐑛 𐑯𐑪𐑑 𐑞 𐑔𐑮𐑧𐑑𐑩𐑯𐑦𐑙 𐑢𐑦𐑯𐑛 𐑓𐑼𐑚𐑨𐑛. 𐑚𐑳𐑑 𐑑𐑱𐑒𐑦𐑙 𐑩𐑛𐑝𐑭𐑯𐑑𐑦𐑡 𐑝 𐑣𐑦𐑟 𐑢𐑦𐑯𐑛𐑢𐑼𐑛 𐑐𐑩𐑟𐑦𐑖𐑩𐑯, 𐑣𐑰 𐑩𐑜𐑱𐑯 𐑕𐑰𐑟𐑛 𐑣𐑦𐑟 𐑑𐑮𐑳𐑥𐑐𐑩𐑑, 𐑯 𐑯𐑴𐑦𐑙 𐑚𐑲 𐑣𐑻 𐑨𐑕𐑐𐑧𐑒𐑑 𐑞𐑨𐑑 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼 𐑝𐑧𐑕𐑩𐑤 𐑢𐑪𐑟 𐑩 ·𐑯𐑨𐑯𐑑𐑳𐑒𐑩𐑑𐑼 𐑯 𐑖𐑹𐑑𐑤𐑦 𐑚𐑬𐑯𐑛 𐑣𐑴𐑥, 𐑣𐑰 𐑤𐑬𐑛𐑤𐑦 𐑣𐑱𐑤𐑛—“𐑩𐑣𐑶 𐑞𐑺! 𐑞𐑦𐑕 𐑦𐑟 𐑞 ·𐑐𐑰𐑒𐑢𐑪𐑛, 𐑚𐑬𐑯𐑛 𐑮𐑬𐑯𐑛 𐑞 𐑢𐑻𐑤𐑛! 𐑑𐑧𐑤 𐑞𐑧𐑥 𐑑 𐑩𐑛𐑮𐑧𐑕 𐑷𐑤 𐑓𐑿𐑗𐑼 𐑤𐑧𐑑𐑼𐑟 𐑑 𐑞 ·𐑐𐑩𐑕𐑦𐑓𐑦𐑒 𐑴𐑖𐑩𐑯! 𐑯 𐑞𐑦𐑕 𐑑𐑲𐑥 𐑔𐑮𐑰 𐑘𐑽𐑟, 𐑦𐑓 𐑲 𐑨𐑥 𐑯𐑪𐑑 𐑨𐑑 𐑣𐑴𐑥, 𐑑𐑧𐑤 𐑞𐑧𐑥 𐑑 𐑩𐑛𐑮𐑧𐑕 𐑞𐑧𐑥 𐑑 ——”

𐑨𐑑 𐑞𐑨𐑑 𐑥𐑴𐑥𐑧𐑯𐑑 𐑞 𐑑𐑵 𐑢𐑱𐑒𐑕 𐑢𐑻 𐑓𐑺𐑤𐑦 𐑒𐑮𐑪𐑕𐑑, 𐑯 𐑦𐑯𐑕𐑑𐑩𐑯𐑑𐑤𐑦, 𐑞𐑧𐑯, 𐑦𐑯 𐑩𐑒𐑹𐑛𐑩𐑯𐑕 𐑢𐑦𐑞 𐑞𐑺 𐑕𐑦𐑙𐑜𐑿𐑤𐑼 𐑢𐑱𐑟, 𐑖𐑴𐑤𐑟 𐑝 𐑕𐑥𐑷𐑤 𐑣𐑸𐑥𐑤𐑩𐑕 𐑓𐑦𐑖, 𐑞𐑨𐑑 𐑓 𐑕𐑳𐑥 𐑛𐑱𐑟 𐑚𐑦𐑓𐑹 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑐𐑤𐑨𐑕𐑦𐑛𐑤𐑦 𐑕𐑢𐑦𐑥𐑦𐑙 𐑚𐑲 𐑬𐑼 𐑕𐑲𐑛, 𐑛𐑸𐑑𐑩𐑛 𐑩𐑢𐑱 𐑢𐑦𐑞 𐑢𐑪𐑑 𐑕𐑰𐑥𐑛 𐑖𐑳𐑛𐑼𐑦𐑙 𐑓𐑦𐑯𐑟, 𐑯 𐑮𐑱𐑯𐑡𐑛 𐑞𐑧𐑥𐑕𐑧𐑤𐑝𐑟 𐑓𐑹 𐑯 𐑭𐑓𐑑 𐑢𐑦𐑞 𐑞 𐑕𐑑𐑮𐑱𐑯𐑡𐑼'𐑟 𐑓𐑤𐑨𐑙𐑒𐑕. 𐑞𐑴 𐑦𐑯 𐑞 𐑒𐑹𐑕 𐑝 𐑣𐑦𐑟 𐑒𐑩𐑯𐑑𐑦𐑯𐑿𐑩𐑤 𐑝𐑶𐑦𐑡𐑦𐑙𐑟 ·𐑱𐑣𐑨𐑚 𐑥𐑳𐑕𐑑 𐑪𐑓𐑩𐑯 𐑚𐑦𐑓𐑹 𐑣𐑨𐑝 𐑯𐑴𐑑𐑦𐑕𐑑 𐑩 𐑕𐑦𐑥𐑦𐑤𐑼 𐑕𐑲𐑑, 𐑘𐑧𐑑, 𐑑 𐑧𐑯𐑦 𐑥𐑪𐑯𐑩𐑥𐑱𐑯𐑦𐑨𐑒 𐑥𐑨𐑯, 𐑞 𐑝𐑧𐑮𐑾𐑕𐑑 𐑑𐑮𐑲𐑓𐑩𐑤𐑟 𐑒𐑩𐑐𐑮𐑦𐑖𐑩𐑕𐑤𐑦 𐑒𐑨𐑮𐑦 𐑥𐑰𐑯𐑦𐑙𐑟.

“𐑕𐑢𐑦𐑥 𐑩𐑢𐑱 𐑓𐑮𐑪𐑥 𐑥𐑰, 𐑛𐑵 𐑘𐑰?” 𐑥𐑻𐑥𐑼𐑛 ·𐑱𐑣𐑨𐑚, 𐑜𐑱𐑟𐑦𐑙 𐑴𐑝𐑼 𐑦𐑯𐑑𐑵 𐑞 𐑢𐑷𐑑𐑼. 𐑞𐑺 𐑕𐑰𐑥𐑛 𐑚𐑳𐑑 𐑤𐑦𐑑𐑩𐑤 𐑦𐑯 𐑞 𐑢𐑻𐑛𐑟, 𐑚𐑳𐑑 𐑞 𐑑𐑴𐑯 𐑒𐑩𐑯𐑝𐑱𐑛 𐑥𐑹 𐑝 𐑛𐑰𐑐 𐑣𐑧𐑤𐑐𐑤𐑩𐑕 𐑕𐑨𐑛𐑯𐑩𐑕 𐑞𐑨𐑯 𐑞 𐑦𐑯𐑕𐑱𐑯 𐑴𐑤𐑛 𐑥𐑨𐑯 𐑣𐑨𐑛 𐑧𐑝𐑼 𐑚𐑦𐑓𐑹 𐑦𐑝𐑦𐑯𐑕𐑑. 𐑚𐑳𐑑 𐑑𐑻𐑯𐑦𐑙 𐑑 𐑞 𐑕𐑑𐑽𐑟𐑥𐑩𐑯, 𐑣𐑵 𐑞𐑳𐑕 𐑓𐑸 𐑣𐑨𐑛 𐑚𐑰𐑯 𐑣𐑴𐑤𐑛𐑦𐑙 𐑞 𐑖𐑦𐑐 𐑦𐑯 𐑞 𐑢𐑦𐑯𐑛 𐑑 𐑛𐑦𐑥𐑦𐑯𐑦𐑖 𐑣𐑻 𐑣𐑧𐑛𐑢𐑱, 𐑣𐑰 𐑒𐑮𐑲𐑛 𐑬𐑑 𐑦𐑯 𐑣𐑦𐑟 𐑴𐑤𐑛 𐑤𐑲𐑩𐑯 𐑝𐑶𐑕,—“𐑳𐑐 𐑣𐑧𐑤𐑥! 𐑒𐑰𐑐 𐑣𐑻 𐑪𐑓 𐑮𐑬𐑯𐑛 𐑞 𐑢𐑻𐑤𐑛!”

𐑮𐑬𐑯𐑛 𐑞 𐑢𐑻𐑤𐑛! 𐑞𐑺 𐑦𐑟 𐑥𐑳𐑗 𐑦𐑯 𐑞𐑨𐑑 𐑕𐑬𐑯𐑛 𐑑 𐑦𐑯𐑕𐑐𐑲𐑼 𐑐𐑮𐑬𐑛 𐑓𐑰𐑤𐑦𐑙𐑟; 𐑚𐑳𐑑 𐑢𐑺𐑑𐑵 𐑛𐑳𐑟 𐑷𐑤 𐑞𐑨𐑑 𐑕𐑻𐑒𐑩𐑥𐑯𐑨𐑝𐑦𐑜𐑱𐑖𐑩𐑯 𐑒𐑪𐑯𐑛𐑳𐑒𐑑? 𐑴𐑯𐑤𐑦 𐑔𐑮𐑵 𐑯𐑳𐑥𐑚𐑼𐑤𐑩𐑕 𐑐𐑧𐑮𐑦𐑤𐑟 𐑑 𐑞 𐑝𐑧𐑮𐑦 𐑐𐑶𐑯𐑑 𐑢𐑧𐑯𐑕 𐑢𐑰 𐑕𐑑𐑸𐑑𐑩𐑛, 𐑢𐑺 𐑞𐑴𐑟 𐑞𐑨𐑑 𐑢𐑰 𐑤𐑧𐑓𐑑 𐑚𐑦𐑣𐑲𐑯𐑛 𐑕𐑦𐑒𐑘𐑫𐑼, 𐑢𐑻 𐑷𐑤 𐑞 𐑑𐑲𐑥 𐑚𐑦𐑓𐑹 𐑳𐑕.

𐑢𐑻 𐑞𐑦𐑕 𐑢𐑻𐑤𐑛 𐑩𐑯 𐑧𐑯𐑛𐑤𐑩𐑕 𐑐𐑤𐑱𐑯, 𐑯 𐑚𐑲 𐑕𐑱𐑤𐑦𐑙 𐑰𐑕𐑑𐑢𐑼𐑛 𐑢𐑰 𐑒𐑫𐑛 𐑓 𐑧𐑝𐑼 𐑮𐑰𐑗 𐑯𐑿 𐑛𐑦𐑕𐑑𐑩𐑯𐑕𐑩𐑟, 𐑯 𐑛𐑦𐑕𐑒𐑳𐑝𐑼 𐑕𐑲𐑑𐑕 𐑥𐑹 𐑕𐑢𐑰𐑑 𐑯 𐑕𐑑𐑮𐑱𐑯𐑡 𐑞𐑨𐑯 𐑧𐑯𐑦 ·𐑕𐑦𐑒𐑤𐑩𐑛𐑦𐑟 𐑹 ·𐑲𐑤𐑩𐑯𐑛𐑟 𐑝 ·𐑒𐑦𐑙 ·𐑕𐑪𐑤𐑩𐑥𐑩𐑯, 𐑞𐑧𐑯 𐑞𐑺 𐑢𐑻 𐑐𐑮𐑪𐑥𐑦𐑕 𐑦𐑯 𐑞 𐑝𐑶𐑦𐑡. 𐑚𐑳𐑑 𐑦𐑯 𐑐𐑼𐑕𐑿𐑑 𐑝 𐑞𐑴𐑟 𐑓𐑸 𐑥𐑦𐑕𐑑𐑼𐑦𐑟 𐑢𐑰 𐑛𐑮𐑰𐑥 𐑝, 𐑹 𐑦𐑯 𐑑𐑹𐑥𐑧𐑯𐑑𐑩𐑛 𐑗𐑱𐑕 𐑝 𐑞𐑨𐑑 𐑛𐑰𐑥𐑩𐑯 𐑓𐑨𐑯𐑑𐑩𐑥 𐑞𐑨𐑑, 𐑕𐑳𐑥 𐑑𐑲𐑥 𐑹 𐑳𐑞𐑼, 𐑕𐑢𐑦𐑥𐑟 𐑚𐑦𐑓𐑹 𐑷𐑤 𐑣𐑿𐑥𐑩𐑯 𐑣𐑸𐑑𐑕; 𐑢𐑲𐑤 𐑗𐑱𐑕𐑦𐑙 𐑕𐑳𐑗 𐑴𐑝𐑼 𐑞𐑦𐑕 𐑮𐑬𐑯𐑛 𐑜𐑤𐑴𐑚, 𐑞𐑱 𐑲𐑞𐑼 𐑤𐑰𐑛 𐑳𐑕 𐑪𐑯 𐑦𐑯 𐑚𐑨𐑮𐑩𐑯 𐑥𐑱𐑟𐑩𐑟 𐑹 𐑥𐑦𐑛𐑢𐑱 𐑤𐑰𐑝 𐑳𐑕 𐑢𐑧𐑤𐑥𐑛.
