·𐑗𐑨𐑐𐑑𐑼 120. 𐑞 ·𐑛𐑧𐑒 ·𐑑𐑩𐑢𐑹𐑛𐑟 𐑞 ·𐑧𐑯𐑛 𐑝 𐑞 𐑓𐑻𐑕𐑑 ·𐑯𐑲𐑑 ·𐑢𐑪𐑗.

𐑱𐑣𐑨𐑚 𐑕𐑑𐑨𐑯𐑛𐑦𐑙 𐑚𐑲 𐑞 𐑣𐑧𐑤𐑥. 𐑕𐑑𐑸𐑚𐑳𐑒 𐑩𐑐𐑮𐑴𐑗𐑦𐑙 𐑣𐑦𐑥.

“𐑢𐑰 𐑥𐑳𐑕𐑑 𐑕𐑧𐑯𐑛 𐑛𐑬𐑯 𐑞 𐑥𐑱𐑯-𐑑𐑪𐑐-𐑕𐑱𐑤 𐑘𐑸𐑛, 𐑕𐑻. 𐑞 𐑚𐑨𐑯𐑛 𐑦𐑟 𐑢𐑻𐑒𐑦𐑙 𐑤𐑵𐑕 𐑯 𐑞 𐑤𐑰 𐑤𐑦𐑓𐑑 𐑦𐑟 𐑣𐑭𐑓-𐑕𐑑𐑮𐑨𐑯𐑛𐑩𐑛. 𐑖𐑨𐑤 𐑲 𐑕𐑑𐑮𐑲𐑒 𐑦𐑑, 𐑕𐑻?”

“𐑕𐑑𐑮𐑲𐑒 𐑯𐑳𐑔𐑦𐑙; 𐑤𐑨𐑖 𐑦𐑑. 𐑦𐑓 𐑲 𐑣𐑨𐑛 𐑕𐑒𐑲-𐑕𐑱𐑤 𐑐𐑴𐑤𐑟, 𐑲'𐑛 𐑕𐑢𐑱 𐑞𐑧𐑥 𐑳𐑐 𐑯𐑬.”

“𐑕𐑻!—𐑦𐑯 ·𐑜𐑪𐑛'𐑟 𐑯𐑱𐑥!—𐑕𐑻?”

“𐑢𐑧𐑤.”

“𐑞 𐑨𐑙𐑒𐑼𐑟 𐑸 𐑢𐑻𐑒𐑦𐑙, 𐑕𐑻. 𐑖𐑨𐑤 𐑲 𐑜𐑧𐑑 𐑞𐑧𐑥 𐑦𐑯𐑚𐑹𐑛?”

“𐑕𐑑𐑮𐑲𐑒 𐑯𐑳𐑔𐑦𐑙, 𐑯 𐑕𐑑𐑻 𐑯𐑳𐑔𐑦𐑙, 𐑚𐑳𐑑 𐑤𐑨𐑖 𐑧𐑝𐑮𐑦𐑔𐑦𐑙. 𐑞 𐑢𐑦𐑯𐑛 𐑮𐑲𐑟𐑩𐑟, 𐑚𐑳𐑑 𐑦𐑑 𐑣𐑨𐑟 𐑯𐑪𐑑 𐑜𐑪𐑑 𐑳𐑐 𐑑 𐑥𐑲 𐑑𐑱𐑚𐑩𐑤-𐑤𐑨𐑯𐑛𐑟 𐑘𐑧𐑑. 𐑒𐑢𐑦𐑒, 𐑯 𐑕𐑰 𐑑 𐑦𐑑.—𐑚𐑲 𐑥𐑭𐑕𐑑𐑕 𐑯 𐑒𐑰𐑤𐑟! 𐑣𐑰 𐑑𐑱𐑒𐑕 𐑥𐑰 𐑓 𐑞 𐑣𐑳𐑯𐑗-𐑚𐑨𐑒𐑑 𐑕𐑒𐑦𐑐𐑼 𐑝 𐑕𐑳𐑥 𐑒𐑴𐑕𐑑𐑦𐑙 𐑕𐑥𐑨𐑒. 𐑕𐑧𐑯𐑛 𐑛𐑬𐑯 𐑥𐑲 𐑥𐑱𐑯-𐑑𐑪𐑐-𐑕𐑱𐑤 𐑘𐑸𐑛! 𐑣𐑴, 𐑜𐑤𐑵𐑐𐑪𐑑𐑕! 𐑤𐑪𐑓𐑑𐑾𐑕𐑑 𐑑𐑮𐑳𐑒𐑕 𐑢𐑻 𐑥𐑱𐑛 𐑓 𐑢𐑲𐑤𐑛𐑩𐑕𐑑 𐑢𐑦𐑯𐑛𐑟, 𐑯 𐑞𐑦𐑕 𐑚𐑮𐑱𐑯-𐑑𐑮𐑳𐑒 𐑝 𐑥𐑲𐑯 𐑯𐑬 𐑕𐑱𐑤𐑟 𐑩𐑥𐑦𐑛 𐑞 𐑒𐑤𐑬𐑛-𐑕𐑒𐑳𐑛. 𐑖𐑨𐑤 𐑲 𐑕𐑑𐑮𐑲𐑒 𐑞𐑨𐑑? 𐑴, 𐑯𐑳𐑯 𐑚𐑳𐑑 𐑒𐑬𐑼𐑛𐑟 𐑕𐑧𐑯𐑛 𐑛𐑬𐑯 𐑞𐑺 𐑚𐑮𐑱𐑯-𐑑𐑮𐑳𐑒𐑕 𐑦𐑯 𐑑𐑧𐑥𐑐𐑦𐑕𐑑 𐑑𐑲𐑥. 𐑢𐑪𐑑 𐑩 𐑙𐑩𐑮𐑵𐑖 𐑩𐑤𐑪𐑓𐑑 𐑞𐑺! 𐑲 𐑢𐑫𐑛 𐑰'𐑩𐑯 𐑑𐑱𐑒 𐑦𐑑 𐑓 𐑕𐑩𐑚𐑤𐑲𐑥, 𐑛𐑦𐑛 𐑲 𐑯𐑪𐑑 𐑯𐑴 𐑞𐑨𐑑 𐑞 𐑒𐑪𐑤𐑦𐑒 𐑦𐑟 𐑩 𐑯𐑶𐑟𐑦 𐑥𐑨𐑤𐑩𐑛𐑦. 𐑴, 𐑑𐑱𐑒 𐑥𐑧𐑛𐑦𐑕𐑦𐑯, 𐑑𐑱𐑒 𐑥𐑧𐑛𐑦𐑕𐑦𐑯!”
